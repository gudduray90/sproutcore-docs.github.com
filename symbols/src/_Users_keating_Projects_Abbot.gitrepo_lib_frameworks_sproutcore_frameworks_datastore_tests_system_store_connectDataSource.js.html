  7 module("connecting DataSource to a store");
  8 
  9 test("data source passed as string should be available as after running _getDataSource", function() {
 10   window.MyTestDataSource = SC.DataSource.extend({
 11     foo: 'bar'
 12   });
 13 
 14   var store = SC.Store.create().from("MyTestDataSource");
 15   same(store.get("dataSource"), "MyTestDataSource");
 16 
 17   var dataSource = store._getDataSource();
 18   same(dataSource.foo, 'bar');
 19 
 20   same(store.get('dataSource').foo, 'bar');
 21 });
 22 
 23 test("data source is required, if it can't be found, error should be thrown", function() {
 24   expect(1);
 25 
 26   try {
 27     SC.Store.create().from("SC.YourTestDataSource")._getDataSource();
 28   } catch (x) {
 29     same(x, 'SC.YourTestDataSource could not be found');
 30   }
 31 });
