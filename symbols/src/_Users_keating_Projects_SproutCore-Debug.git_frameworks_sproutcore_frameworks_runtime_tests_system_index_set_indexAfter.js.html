  7 
  8 var set ;
  9 
 10 module("SC.IndexSet.indexAfter", {
 11   setup: function() {
 12     set = SC.IndexSet.create(5).add(10,5).add(100);
 13   }
 14 });
 15 
 16 test("no earlier index in set", function(){ 
 17   equals(set.indexAfter(3), 5, 'set.indexAfter(3) in %@ should start of first index range'.fmt(set));
 18 });
 19 
 20 test("with index after end of set", function() {
 21   equals(set.indexAfter(1000), -1, 'set.indexAfter(1000) in %@ should return -1'.fmt(set));
 22 });
 23 
 24 test("inside of multi-index range", function() {
 25   equals(set.indexAfter(12), 13, 'set.indexAfter(12) in %@ should return next index'.fmt(set));
 26 });
 27 
 28 test("end of multi-index range", function() {
 29   equals(set.indexAfter(14), 100, 'set.indexAfter(14) in %@ should start of next range'.fmt(set));
 30 });
 31 
 32 
 33 test("single index range", function() {
 34   equals(set.indexAfter(5), 10, 'set.indexAfter(5) in %@ should start of next range multi-index range'.fmt(set));
 35 });
