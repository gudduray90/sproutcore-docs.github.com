  8 sc_require('core');
  9 /** @class
 10 
 11   dir properties
 12   @property dir
 13   @property name
 14   @property contents
 15   
 16   @extends SC.Record
 17   @version 0.1
 18 */
 19 Greenhouse.Dir = SC.Record.extend(
 20 /** @scope Greenhouse.Dir.prototype */ {
 21   type: 'Dir',
 22   nestedRecordNamespace: Greenhouse,
 23   
 24   name: SC.Record.attr(String),
 25   dir: SC.Record.attr(String),  
 26   contents: SC.Record.toMany('Greenhouse.File', {nested: YES}),
 27   
 28   primaryKey: 'id',
 29   
 30   
 31   isFile: NO,
 32 
 33   path: function(){
 34     return this.get('dir') + this.get('name');
 35   }.property('name', 'dir').cacheable(),
 36 
 37   
 38   evalBody: function(){
 39     var bodyText = this.get('body'), body, designs = [];
 40     
 41    try{
 42       body = eval(bodyText || "");
 43       body.set('needsDesigner', YES);
 44       body.set('isContainerView',YES);
 45       this.writeAttribute('currentDesign', body, YES);
 46       this.notifyPropertyChange('currentDesign');
 47       for(var v in body){
 48         if(body.hasOwnProperty(v)){
 49           if(body[v] && body[v].kindOf){
 50             if(body[v].kindOf(SC.View)){
 51               designs.push(SC.Object.create({type: 'view', view: body.get(v), name: v}));
 52             }
 53             else if(body[v].kindOf(SC.Page)){
 54               designs.push(SC.Object.create({type: 'page', view: body.get(v), name: v}));
 55             }
 56             else if(body[v].kindOf(SC.Pane)){
 57               designs.push(SC.Object.create({type: 'pane', view: body.get(v), name: v}));
 58             }
 59           }
 60         }
 61       }
 62  63       this.writeAttribute('designs', designs, YES);
 64       this.notifyPropertyChange('designs');
 65       
 66     } catch (exception) {
 67       console.log("Couldn't eval body...");
 68     }
 69     
 70   },
 71   /*
 72     if this is a dir then return if the passed
 73     file's name and type matches
 74     @returns boolean
 75   */
 76   includesFile: function(file){
 77     if(!this.get('isFile')){
 78       var contents = this.get('contents'), ret;
 79       ret = contents.find(function(item){
 80         if(item.get('type') === file.get('type') && item.get('name') === file.get('name') && item !== file) return YES;
 81       });
 82       
 83       return ret ? YES : NO;
 84     }
 85     else{
 86       return NO;
 87     }
 88   }
 89 
 90 }) ;
 91 
 92 Greenhouse.Dir.mixin({
 93 
 94 });
