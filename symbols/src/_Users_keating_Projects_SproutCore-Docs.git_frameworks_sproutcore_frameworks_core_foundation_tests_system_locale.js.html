  8 var LocaleObject;
  9 module("object.SC.Locale()", {	
 10 		setup: function() {
 11 			
 12 			LocaleObject = SC.Locale.create({
 13 				init: function(){
 14 					sc_super();
 15 					//hash of new languages
 16 					var newLocales = { deflang: 'dl', empty: '' };
 17 					
 18 					//Added the new languages to the existing list of locales
 19 					SC.Locale.addStrings(newLocales);
 20 				}
 21 			});
 22 
 23 		}
 24 	});
 25 	
 26 test("Locale.init() : Should return a flag if the language has been set during the locale initialization", function() {
 27 	// As the locale is added during initialization the value of hasString is true
 28 	equals(LocaleObject.hasStrings, true) ;
 29 	
 30 	//check the string values. 
 31 	equals(LocaleObject.strings.deflang, 'dl') ;
 32 });
 33 
 34 
 35 test("Locale.locWithDefault() : localized version of the string or the string if no match was found", function() {
 36 	//Based on the input passed it should return the default locale
 37 	equals(LocaleObject.locWithDefault("en"), "en") ;
 38 	equals(LocaleObject.locWithDefault("jp", "Japanese"), "Japanese") ;
 39 	equals(LocaleObject.locWithDefault('deflang'), "dl") ;
 40 });
 41 
 42 test("Locale.locWithDefault() : localized version of the string even if localized version is blank", function() {
 43   equals(LocaleObject.locWithDefault("empty"), "");
 44   equals(LocaleObject.locWithDefault("empty", "Empty"), "");
 45 });
 46 
 47 test("Locale.addStrings() : Should be able to add the passed hash of strings to the locale's strings table", function() {		
 48 	
 49 	//Check for the new languages. This should be false as these are not added to the list of locales
 50 	equals(false, SC.Locale.options().strings.chinese === 'zh' && SC.Locale.options().strings.dutch === 'nl') ;
 51 	
 52 	//hash of new languages
 53 	var newLocales = { chinese: 'zh', czech: 'cs', dutch: 'nl'};
 54 	
 55 	//Added the new languages to the existing list of locales
 56 	SC.Locale.addStrings(newLocales);
 57 	
 58 	//Result should be true as the new locales added to the list of default locales
 59 	equals(true, SC.Locale.options().strings.chinese === 'zh' && SC.Locale.options().strings.dutch === 'nl') ;
 60 });
 61 
 62 test("Locale.options() : Should provide the registered locales that have not been instantiated", function() {
 63 		
 64 		//hash of new languages
 65 		var newLocales = { jamaican: 'ji', korean: 'ko'};
 66 
 67 		//Added the new languages to the existing list of locales
 68 		SC.Locale.addStrings(newLocales);
 69 		
 70 		//Options should return the list of registered locales, so checking if the returned object has strings.
 71 		equals(SC.Locale.options().hasStrings, true) ;
 72 		
 73 		//Checking the strings with default locales.
 74 		equals(true, SC.Locale.options().strings.jamaican === 'ji' && SC.Locale.options().strings.korean === 'ko') ;
 75 	});
 76 	
 77 test("Locale.normalizeLanguage() : Should provide the two character language code for the passed locale", function() {
 78 	//If nothing is passed this will return the default code as 'en'
 79 	equals(SC.Locale.normalizeLanguage(), 'en') ;
 80 	
 81 	//If the language is passed as 'English' this will return the code as 'en'
 82 	equals(SC.Locale.normalizeLanguage('English'), 'en') ;
 83 	
 84 	//For any other code passed which is not in the default code it should return as it was passed
 85 	equals(SC.Locale.normalizeLanguage('ab'), 'ab') ; 
 86 });
 87 
 88 test("Locale.toString() : Should return the current language set with the guid value", function() {
 89 	// Creating the new locale by extending an existing Locale object
 90 	SC.Locale.locales['mx'] = SC.Locale.extend({ _deprecatedLanguageCodes: ['mexican'] }) ;
 91 	
 92 	//Result should return the chinese object
 93 	equals(SC.Locale.locales.mx.currentLocale.isObject, true) ;
 94 
 95 });
 96 
 97 test("Locale.createCurrentLocale() : Should create the Locale Object for the language selected", function() {
 98 	//This will match the browser language with the SC language and create the object accordingly
 99 	// This test will pass only for the default languages i.e en, fr, de, ja, es, it. 
100 	equals(true, SC.Locale.createCurrentLocale().language === SC.browser.language) ;
101 	
102 	//Resetting the default browser language
103 	SC.browser.language='kn';
104 	
105 	//This is false as currentLocale will be created as 'en'
106 	equals(false, SC.Locale.createCurrentLocale().language===SC.browser.language) ;
107 });	
108 
109 test("Locale.localeClassFor() : Should find the locale class for the names language code or creates on based on its most likely parent", function() {
110  		// Local Class for any language other than default languages will be 'en'. Therefore this condition is false
111 	equals(false, SC.Locale.localeClassFor('nl').create().language === "nl") ;
112 	
113 	// This adds the new language with the parent language to the default list
114 	SC.Locale.locales['nl'] = SC.Locale.extend({ _deprecatedLanguageCodes: ['Dutch'] }) ;
115 	
116 	//This condition is true as the local class now exists for 'nl'
117 	equals(true, SC.Locale.localeClassFor('nl').create().language==="nl") ;
118 });
119 
120 test("Locale.define() : Should be able to define a particular type of locale", function() {
121  		SC.Locale.define('xy', {
122 		longNames: 'Charles John Romonoski Gregory William'.split(' '),
123 		shortNames: ['C','A','Y','N']
124 	});
125 	
126 	//Result should return the new locale object
127 	equals(SC.Locale.locales.xy.isClass, true) ;
128 });
129 
130 test("Locale.extend() : Should make sure important properties of Locale object are copied to a new class", function() {
131 	SC.Locale.locales['mn'] = SC.Locale.extend({ _deprecatedLanguageCodes: ['newlang'] }) ;
132 	
133 	//hash of new languages
134 	var testLocales = { test: 'te', newtest: 'nt'};
135 	//Added the new languages to the existing list of locales through the new locale object
136 	SC.Locale.locales.mn.addStrings(testLocales);
137 	
138 	//Result should be true as the new locales added to the list of default locales
139 	equals(SC.Locale.locales.mn.options().strings.test,'te') ;
140 });
