 68 /** @class
 69 
 70   A toolbar view can be anchored at the top or bottom of the window to contain
 71   your main toolbar buttons.
 72 
 73   You can also override the layout property yourself or simply set the
 74   anchorLocation to `SC.ANCHOR_TOP` or `SC.ANCHOR_BOTTOM`.  This will configure
 75   the layout of your toolbar automatically when it is created.
 76 
 77   @extends SC.View
 78   @since SproutCore 1.0
 79 */
 80 SC.ToolbarView = SC.View.extend(
 81 /** @scope SC.ToolbarView.prototype */ {
 82 
 83   /**
 84     @type Array
 85     @default ['sc-toolbar-view']
 86     @see SC.View#classNames
 87   */
 88   classNames: ['sc-toolbar-view'],
 89 
 90   /**
 91     The WAI-ARIA role for toolbar view.
 92 
 93     @type String
 94     @default 'toolbar'
 95     @readOnly
 96   */
 97   ariaRole: 'toolbar',
 98 
 99   /**
100     @type String
101     @default 'toolbarRenderDelegate'
102   */
103   renderDelegateName: 'toolbarRenderDelegate',
104 
105   /**
106     Default anchor location.  This will be applied automatically to the
107     toolbar layout if you set it. Possible values:
108     
109       - SC.ANCHOR_TOP
110       - SC.ANCHOR_LEFT
111       - SC.ANCHOR_TOP_LEFT
112       - SC.ANCHOR_BOTTOM
113       - SC.ANCHOR_RIGHT
114       - SC.ANCHOR_BOTTOM_RIGHT
115     
116     @type String
117     @default null
118   */
119   anchorLocation: null,
120 
121   // ..........................................................
122   // INTERNAL SUPPORT
123   //
124 
125   /** @private */
126   layout: { left: 0, height: 32, right: 0 },
127 
128   /** @private */
129   init: function() {
130     // apply anchor location before setting up the rest of the view.
131     if (this.anchorLocation) {
132       this.layout = SC.merge(this.layout, this.anchorLocation);
133     }
134     sc_super();
135   }
136 
137 });
