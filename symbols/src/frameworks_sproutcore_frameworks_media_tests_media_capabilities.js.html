 10 /**
 11  * Is Flash supported?
 12  * 
 13  * These unit tests are largely meaningless, because we can't actually control
 14  * the navigator.plugins array or the window.ActiveXObject class, nor can we
 15  * manipulate the existence of the getUserMedia properties. Without being able
 16  * to toggle them dynamically, the best we can test is whether the return value
 17  * of the capabilities object is the same as is available via plugins... and
 18  * since that would consist of (at this time) the exact same code, it's somewhat
 19  * redundant. Still, this unit test is included in case something changes in the
 20  * future.
 21  * 
 22  * @see Department of Redundancy Department
 23  */
 24 test("Test Flash Support", function() {
 25   
 26   doesFlashExist = NO;
 27   
 28   // Non-IE detection
 29   if(navigator.plugins) {
 30     for( var i = 0; i < navigator.plugins.length; i++) {
 31       if(navigator.plugins[i].name.indexOf("Shockwave Flash") >= 0) {
 32         doesFlashExist = YES;
 33       }
 34     }
 35   } else if(window.ActiveXObject) {
 36     try {
 37       var control = new ActiveXObject('ShockwaveFlash.ShockwaveFlash');
 38       delete control;
 39       doesFlashExist = YES;
 40     } catch(e) {
 41     }
 42   }
 43   
 44   equals(SC.mediaCapabilities.isFlashSupported, doesFlashExist, "Flash plugin result must match what the browser supports.");
 45 });

 73 /**
 74  * Test version support for the HTML5 audio tag.
 75  */
 76 test("Test HTML5 Audio Support", function() {
 77   var isAudioSupported = NO;
 78   try {
 79     if(SC.browser.isMozilla && SC.browser.compare(SC.browser.mozilla, "3.6") <= 0) {
 80       throw new Error();
 81     }
 82     
 83     var doc = document.createElement('Audio');
 84     isAudioSupported = !!doc.canPlayType;
 85     delete doc;
 86     
 87   } catch(e) {
 88   }
 89   
 90   equals(SC.mediaCapabilities.isHTML5AudioSupported, isAudioSupported, "Audio Support flag must match what we've been able to determine from the browser.");
 91 });
