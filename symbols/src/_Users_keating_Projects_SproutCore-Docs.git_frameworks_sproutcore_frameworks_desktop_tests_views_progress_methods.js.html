 10 module("SC.ProgressView Methods", {
 11   setup: function() {
 12     SC.RunLoop.begin();
 13     pane = SC.MainPane.create({
 14       childViews: [
 15         SC.ProgressView.extend({
 16           value: 25,
 17           minimum: 0,
 18           maximum: 100
 19         })]
 20     });
 21     pane.append(); // make sure there is a layer...
 22     SC.RunLoop.end();
 23 
 24     view = pane.childViews[0];
 25   },
 26 
 27   teardown: function() {
 28     pane.remove();
 29     pane = view = null ;
 30   }
 31 });
 32 
 33 test("changing a progress view value to maximum", function() {
 34   equals(view.get('value'), 25, 'precon - value should be 25');
 35   equals(view.get('isIndeterminate'), NO, 'precon - value should be NO');
 36   equals(view.get('isRunning'), NO, 'precon - value should be NO');
 37   equals(view.get('isEnabled'), YES, 'precon - value should be YES');
 38 
 39   SC.RunLoop.begin();
 40   view.set('value', 100);
 41   SC.RunLoop.end();
 42 
 43   equals(view.get('value'), 100, 'should be 100');
 44 });
 45 
 46 test("changing value of a disabled progress view", function() {
 47   equals(view.get('value'), 25, 'precon - value should be 25');
 48   equals(view.get('isEnabled'), YES, 'precon - value should be YES');
 49 
 50   SC.RunLoop.begin();
 51   view.set('isEnabled', NO);
 52   view.set('value', 100);
 53   SC.RunLoop.end();
 54 
 55   // changing while disabled is allowed
 56   equals(view.get('value'), 100, 'should be 100');
 57 });
