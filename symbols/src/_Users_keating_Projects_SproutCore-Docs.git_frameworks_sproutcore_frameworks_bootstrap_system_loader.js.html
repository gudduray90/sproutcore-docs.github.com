 10 SC.setupBodyClassNames = function() {
 11   var el = document.body,
 12       browser, platform, shadows, borderRad, classNames, style, ieVersion;
 13   if (!el) return ;
 14 
 15   browser = SC.browser.current ;
 16   platform = SC.browser.isWindows ? 'windows' : SC.browser.isMac ? 'mac' : 'other-platform' ;
 17   style = document.documentElement.style;
 18   shadows = (style.MozBoxShadow !== undefined) ||
 19                 (style.webkitBoxShadow !== undefined) ||
 20                 (style.oBoxShadow !== undefined) ||
 21                 (style.boxShadow !== undefined);
 22 
 23   borderRad = (style.MozBorderRadius !== undefined) ||
 24               (style.webkitBorderRadius !== undefined) ||
 25               (style.oBorderRadius !== undefined) ||
 26               (style.borderRadius !== undefined);
 27 
 28   classNames = el.className ? el.className.split(' ') : [] ;
 29   if(shadows) classNames.push('box-shadow');
 30   if(borderRad) classNames.push('border-rad');
 31   classNames.push(browser, platform) ;
 32 
 33   // This isn't a perfectly correct way to compare versions, but should be okay
 34   // in practical usage.
 35   ieVersion = parseInt(SC.browser.version, 10);
 36   if (SC.browser.isIE) {
 37     if (ieVersion === 7) {
 38       classNames.push('ie7');
 39     }
 40     else if (ieVersion === 8) {
 41       classNames.push('ie8');
 42     }
 43     else if (ieVersion === 9) {
 44       classNames.push('ie9');
 45     }
 46   }
 47 
 48   if(browser==="safari" || browser==="chrome") classNames.push('webkit');
 49   if (SC.browser.isMobileSafari) classNames.push('mobile-safari') ;
 50   if ('createTouch' in document) classNames.push('touch');
 51   el.className = classNames.join(' ') ;
 52 } ;
