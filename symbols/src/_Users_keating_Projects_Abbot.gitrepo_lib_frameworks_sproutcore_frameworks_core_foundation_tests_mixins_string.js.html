  8 var LocaleObject;
  9 
 10 module('SC.Object', {
 11   setup: function() {
 12 
 13     LocaleObject = SC.Locale.create({
 14       init: function(){
 15         sc_super();
 16         //hash of new languages
 17         var newLocales = { deflang: 'dl', empty: '' };
 18 
 19         //Added the new languages to the existing list of locales
 20         SC.Locale.addStrings(newLocales);
 21       }
 22     });
 23     this.currentLocale = LocaleObject;
 24     
 25     SC.stringsFor('English', {
 26       'Test': '%@',
 27       'Test.Multiple': '%@ %@'
 28     });
 29 
 30     SC.metricsFor('English', {
 31       'Button.left': 10,
 32       'Button.top': 20,
 33       'Button.width': 80,
 34       'Button.height': 30
 35     });
 36   }
 37 });

 59 test("Localize a string", function() {
 60   //Based on the input passed it should return the default locale
 61   equals("en".loc(), "en", "Using String.prototype.loc") ;
 62   equals(SC.String.loc("en"), "en", "Using SC.String.loc");
 63 
 64   equals("jp".locWithDefault("Japanese"), "Japanese", "Using String.prototype.locWithDefault") ;
 65   equals(SC.String.locWithDefault("jp", "Japanese"), "Japanese", "Using SC.String.locWithDefault") ;
 66 
 67   equals('deflang'.loc(), "dl", "Using String.prototype.loc") ;
 68   equals(SC.String.loc('deflang'), "dl", "Using SC.String.loc") ;
 69 });
 70 
 71 test("Localize a string with mutliple parameters", function() {
 72   equals("Test".loc('parameter1'), 'parameter1', "Localizing with one parameter - using String.prototype.loc");
 73   equals(SC.String.loc("Test", 'parameter1'), 'parameter1', "Localizing with one parameter - using SC.String.loc");
 74 
 75   equals("Test.Multiple".loc('parameter1', 'parameter2'), 'parameter1 parameter2', "Localizing with multiple parameters - using String.prototype.loc");
 76   equals(SC.String.loc("Test.Multiple", 'parameter1', 'parameter2'), 'parameter1 parameter2', "Localizing with multiple parameters - using SC.String.loc");
 77 });
 78 
 79 test("Localize a string with null or missing parameters", function() {
 80   equals("Test".loc(null), "(null)", "Localizing with null parameter - using String.prototype.loc");
 81   equals(SC.String.loc("Test", null), "(null)", "Localizing with null parameter - using SC.String.loc");
 82 
 83   equals("Test".loc(), "", "Localizing with missing parameter - using String.prototype.loc");
 84   equals(SC.String.loc("Test"), "", "Localizing with missing parameter - using SC.String.loc");
 85 
 86   equals("Test.Multiple".loc("p1", null), "p1 (null)", "Localizing multiple with null parameter - using String.prototype.loc");
 87   equals(SC.String.loc("Test.Multiple", "p1", null), "p1 (null)", "Localizing with null parameter - using SC.String.loc");
 88 
 89   equals("Test.Multiple".loc("p1"), "p1 ", "Localizing multiple with missing parameter - using String.prototype.loc");
 90   equals(SC.String.loc("Test.Multiple", "p1"), "p1 ", "Localizing with missing parameter - using SC.String.loc");
 91 });
 92 
 93 test("Localize a string even if localized version is empty", function() {
 94   equals("empty".loc(), "", "Using String.prototype.loc");
 95   equals(SC.String.loc("empty"), "", "Using SC.String.loc");
 96 
 97   equals("empty".locWithDefault("Empty"), "", "Using String.prototype.locWithDefault");
 98   equals(SC.String.locWithDefault("empty", "Empty"), "", "Using SC.String.locWithDefault");

101 test("Access a localized metric", function() {
102   equals(10, "Button.left".locMetric());
103   equals(20, "Button.top".locMetric());
104   equals(undefined, "Button.notThere".locMetric());
105 });

107 test("Access a localized layout hash", function() {
108   // Simple case (if we ever get a full hash comparison function, we should use
109   // it here).
110   var layout = "Button".locLayout();
111   equals(10, layout.left);
112   equals(20, layout.top);
113   equals(80, layout.width);
114   equals(30, layout.height);
115   equals(undefined, layout.right);    // No localized key
116 
117 
118   // Slightly more involved case:  allow the user to specify an additional hash.
119   layout = "Button".locLayout({right:50});
120   equals(10, layout.left);
121   equals(20, layout.top);
122   equals(80, layout.width);
123   equals(30, layout.height);
124   equals(50, layout.right);    // No localized key
125 
126 
127   // Sanity-check case:  Since we have both a localized key for 'left' and we'll
128   // pass it in, an exception should be thrown.
129   should_throw(function() {
130     "Button".locLayout({left:10});
131   }, Error, "locLayout():  There is a localized value for the key 'Button.left' but a value for 'left' was also specified in the non-localized hash");
132 });
