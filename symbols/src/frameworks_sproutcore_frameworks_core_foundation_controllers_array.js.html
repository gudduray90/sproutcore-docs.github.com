 11 /**
 12   @class
 13 
 14   An ArrayController provides a way for you to publish an array of objects
 15   for CollectionView or other controllers to work with.  To work with an
 16   ArrayController, set the content property to the array you want the
 17   controller to manage.  Then work directly with the controller object as if
 18   it were the array itself.
 19 
 20   When you want to display an array of objects in a CollectionView, bind the
 21   "arrangedObjects" of the array controller to the CollectionView's "content"
 22   property.  This will automatically display the array in the collection view.
 23 
 24   @extends SC.Controller
 25   @extends SC.Array
 26   @extends SC.SelectionSupport
 27   @author Charles Jolley
 28   @since SproutCore 1.0
 29 */
 30 SC.ArrayController = SC.Controller.extend(SC.Array, SC.SelectionSupport,
 31 /** @scope SC.ArrayController.prototype */ {
 32 
 33   // ..........................................................
 34   // PROPERTIES
 35   //
 36 
 37   /**
 38     The content array managed by this controller.
 39 
 40     You can set the content of the ArrayController to any object that
 41     implements SC.Array or SC.Enumerable.  If you set the content to an object
 42     that implements SC.Enumerable only, you must also set the orderBy property
 43     so that the ArrayController can order the enumerable for you.
 44 
 45     If you set the content to a non-enumerable and non-array object, then the
 46     ArrayController will wrap the item in an array in an attempt to normalize
 47     the result.
 48 
 49     @type SC.Array
 50   */
 51   content: null,
 52 
 53   /**
 54     Makes the array editable or not.  If this is set to NO, then any attempts
 55     at changing the array content itself will throw an exception.
 56 
 57     @type Boolean
 58   */
 59   isEditable: YES,
 60 
 61   /**
 62     Used to sort the array.
 63 
 64     If you set this property to a key name, array of key names, or a function,
 65     then then ArrayController will automatically reorder your content array
 66     to match the sort order.  When using key names, you may specify the
 67     direction of the sort by appending ASC or DESC to the key name.  By default
 68     sorting is done in ascending order.
 69 
 70     For example,
 71 
 72         myController.set('orderBy', 'title DESC');
 73         myController.set('orderBy', ['lastName ASC', 'firstName DESC']);
 74 
 75     Normally, you should only use this property if you set the content of the
 76     controller to an unordered enumerable such as SC.Set or SC.SelectionSet.
 77     In this case the orderBy property is required in order for the controller
 78     to property order the content for display.
 79 
 80     If you set the content to an array, it is usually best to maintain the
 81     array in the proper order that you want to display things rather than
 82     using this method to order the array since it requires an extra processing
 83     step.  You can use this orderBy property, however, for displaying smaller
 84     arrays of content.
 85 
 86     Note that you can only use addObject() to insert new objects into an
 87     array that is ordered.  You cannot manually reorder or insert new objects
 88     into specific locations because the order is managed by this property
 89     instead.
 90 
 91     If you pass a function, it should be suitable for use in compare().
 92 
 93     @type String|Array|Function
 94   */
 95   orderBy: null,
 96 
 97   /**
 98     Set to YES if you want the controller to wrap non-enumerable content
 99     in an array and publish it.  Otherwise, it will treat single content like
100     null content.
101 
102     @type Boolean
103   */
104   allowsSingleContent: YES,
105 
106   /**
107     Set to YES if you want objects removed from the array to also be
108     deleted.  This is a convenient way to manage lists of items owned
109     by a parent record object.
110 
111     Note that even if this is set to NO, calling destroyObject() instead of
112     removeObject() will still destroy the object in question as well as
113     removing it from the parent array.
114 
115     @type Boolean
116   */
117   destroyOnRemoval: NO,
118 
119   /**
120     Returns an SC.Array object suitable for use in a CollectionView.
121     Depending on how you have your ArrayController configured, this property
122     may be one of several different values.
123 
124     @type SC.Array
125   */
126   arrangedObjects: function () {
127     return this;
128   }.property().cacheable(),
129 
130   /**
131     Computed property indicates whether or not the array controller can
132     remove content.  You can delete content only if the content is not single
133     content and isEditable is YES.
134 
135     @type Boolean
136   */
137   canRemoveContent: function () {
138     var content = this.get('content'), ret;
139     ret = !!content && this.get('isEditable') && this.get('hasContent');
140     if (ret) {
141       return !content.isEnumerable ||
142              (SC.typeOf(content.removeObject) === SC.T_FUNCTION);
143     } else return NO;
144   }.property('content', 'isEditable', 'hasContent'),
145 
146   /**
147     Computed property indicates whether you can reorder content.  You can
148     reorder content as long a the controller isEditable and the content is a
149     real SC.Array-like object.  You cannot reorder content when orderBy is
150     non-null.
151 
152     @type Boolean
153   */
154   canReorderContent: function () {
155     var content = this.get('content'), ret;
156     ret = !!content && this.get('isEditable') && !this.get('orderBy');
157     return ret && !!content.isSCArray;
158   }.property('content', 'isEditable', 'orderBy'),
159 
160   /**
161     Computed property insides whether you can add content.  You can add
162     content as long as the controller isEditable and the content is not a
163     single object.
164 
165     Note that the only way to simply add object to an ArrayController is to
166     use the addObject() or pushObject() methods.  All other methods imply
167     reordering and will fail.
168 
169     @type Boolean
170   */
171   canAddContent: function () {
172     var content = this.get('content'), ret;
173     ret = content && this.get('isEditable') && content.isEnumerable;
174     if (ret) {
175       return (SC.typeOf(content.addObject) === SC.T_FUNCTION) ||
176              (SC.typeOf(content.pushObject) === SC.T_FUNCTION);
177     } else return NO;
178   }.property('content', 'isEditable'),
179 
180   /**
181     Set to YES if the controller has valid content that can be displayed,
182     even an empty array.  Returns NO if the content is null or not enumerable
183     and allowsSingleContent is NO.
184 
185     @type Boolean
186   */
187   hasContent: function () {
188     var content = this.get('content');
189     return !!content &&
190            (!!content.isEnumerable || !!this.get('allowsSingleContent'));
191   }.property('content', 'allowSingleContent'),
192 
193   /**
194     Returns the current status property for the content.  If the content does
195     not have a status property, returns SC.Record.READY.
196 
197     @type Number
198   */
199   status: function () {
200     var content = this.get('content'),
201         ret = content ? content.get('status') : null;
202     return ret ? ret : SC.Record.READY;
203   }.property().cacheable(),
204

209   /**
210     Adds an object to the array.  If the content is ordered, this will add the
211     object to the end of the content array.  The content is not ordered, the
212     location depends on the implementation of the content.
213 
214     If the source content does not support adding an object, then this method
215     will throw an exception.
216 
217     @param {Object} object The object to add to the array.
218     @returns {SC.ArrayController} The receiver.
219   */
220   addObject: function (object) {
221     if (!this.get('canAddContent')) { throw new Error("%@ cannot add content".fmt(this)); }
222 
223     var content = this.get('content');
224     if (content.isSCArray) { content.pushObject(object); }
225     else if (content.addObject) { content.addObject(object); }
226     else { throw new Error("%@.content does not support addObject".fmt(this)); }
227 
228     return this;
229   },
230 
231   /**
232     Removes the passed object from the array.  If the underlying content
233     is a single object, then this simply sets the content to null.  Otherwise
234     it will call removeObject() on the content.
235 
236     Also, if destroyOnRemoval is YES, this will actually destroy the object.
237 
238     @param {Object} object the object to remove
239     @returns {SC.ArrayController} receiver
240   */
241   removeObject: function (object) {
242     if (!this.get('canRemoveContent')) {
243       throw new Error("%@ cannot remove content".fmt(this));
244     }
245 
246     var content = this.get('content');
247     if (content.isEnumerable) {
248       content.removeObject(object);
249     } else {
250       this.set('content', null);
251     }
252 
253     if (this.get('destroyOnRemoval') && object.destroy) { object.destroy(); }
254     return this;
255   },

261   /**
262     Compute the length of the array based on the observable content
263 
264     @type Number
265   */
266   length: function () {
267     var content = this._scac_observableContent();
268     return content ? content.get('length') : 0;
269   }.property().cacheable(),
270 
271   /** @private
272     Returns the object at the specified index based on the observable content
273   */
274   objectAt: function (idx) {
275     var content = this._scac_observableContent();
276     return content ? content.objectAt(idx) : undefined;
277   },
278 
279   /** @private
280     Forwards a replace on to the content, but only if reordering is allowed.
281   */
282   replace: function (start, amt, objects) {
283     // check for various conditions before a replace is allowed
284     if (!objects || objects.get('length') === 0) {
285       if (!this.get('canRemoveContent')) {
286         throw new Error("%@ cannot remove objects from the current content".fmt(this));
287       }
288     } else if (!this.get('canReorderContent')) {
289       throw new Error("%@ cannot add or reorder the current content".fmt(this));
290     }
291 
292     // if we can do this, then just forward the change.  This should fire
293     // updates back up the stack, updating rangeObservers, etc.
294     var content = this.get('content'); // note: use content, not observable
295     var objsToDestroy = [], i, objsLen;
296 
297     if (this.get('destroyOnRemoval')) {
298       for (i = 0; i < amt; i++) {
299         objsToDestroy.push(content.objectAt(i + start));
300       }
301     }
302 
303     if (content) { content.replace(start, amt, objects); }
304 
305     for (i = 0, objsLen = objsToDestroy.length; i < objsLen; i++) {
306 
307       objsToDestroy[i].destroy();
308     }
309     objsToDestroy = null;
310 
311     return this;
312   },
313 
314   indexOf: function (object, startAt) {
315     var content = this._scac_observableContent();
316     return content ? content.indexOf(object, startAt) : -1;
317   },

329   /** @private
330     Cached observable content property.  Set to NO to indicate cache is
331     invalid.
332   */
333   _scac_cached: NO,
334 
335   /**
336     @private
337 
338     Returns the current array this controller is actually managing.  Usually
339     this should be the same as the content property, but sometimes we need to
340     generate something different because the content is not a regular array.
341 
342     Passing YES to the force parameter will force this value to be recomputed.
343 
344     @returns {SC.Array} observable or null
345   */
346   _scac_observableContent: function () {
347     var ret = this._scac_cached;
348     if (ret) { return ret; }
349 
350     var content = this.get('content'), func, order;
351 
352     // empty content
353     if (SC.none(content)) { return (this._scac_cached = []); }
354 
355     // wrap non-enumerables
356     if (!content.isEnumerable) {
357       ret = this.get('allowsSingleContent') ? [content] : [];
358       return (this._scac_cached = ret);
359     }
360 
361     // no-wrap
362     var orderBy = this.get('orderBy');
363     if (!orderBy) {
364       if (content.isSCArray) { return (this._scac_cached = content); }
365       else { throw new Error("%@.orderBy is required for unordered content".fmt(this)); }
366     }
367 
368     // all remaining enumerables must be sorted.
369 
370     // build array - then sort it
371     var type = SC.typeOf(orderBy);
372 
373     if (type === SC.T_STRING) {
374       orderBy = [orderBy];
375     } else if (type === SC.T_FUNCTION) {
376       func = orderBy;
377     } else if (type !== SC.T_ARRAY) {
378       throw new Error("%@.orderBy must be Array, String, or Function".fmt(this));
379     }
380 
381     // generate comparison function if needed - use orderBy
382     func = func || function (a, b) {
383       var status, key, match, valueA, valueB;
384 
385       for (var i = 0, l = orderBy.get('length'); i < l && !status; i++) {
386         key = orderBy.objectAt(i);
387 
388         if (key.search(/(ASC|DESC)/) === 0) {
389           //@if(debug)
390           SC.warn("Developer Warning: SC.ArrayController's orderBy direction syntax has been changed to match that of SC.Query and MySQL.  Please change your String to 'key DESC' or 'key ASC'.  Having 'ASC' or 'DESC' precede the key has been deprecated.");
391           //@endif
392           match = key.match(/^(ASC )?(DESC )?(.*)$/);
393           key = match[3];
394         } else {
395           match = key.match(/^(\S*)\s*(DESC)?(?:ASC)?$/);
396           key = match[1];
397         }
398         order = match[2] ? -1 : 1;
399 
400         if (a) { valueA = a.isObservable ? a.get(key) : a[key]; }
401         if (b) { valueB = b.isObservable ? b.get(key) : b[key]; }
402 
403         status = SC.compare(valueA, valueB) * order;
404       }
405 
406       return status;
407     };
408 
409     return (this._scac_cached = content.toArray().sort(func));
410   },
411 
412   propertyWillChange: function (key) {
413     if (key === 'content') {
414       this.arrayContentWillChange(0, this.get('length'), 0);
415     } else {
416       return sc_super();
417     }
418   },
419 
420   _scac_arrayContentWillChange: function (start, removed, added) {
421     this.arrayContentWillChange(start, removed, added);
422     if (this._kvo_enumerable_property_chains) {
423       var removedObjects = this.slice(start, start + removed);
424       this.teardownEnumerablePropertyChains(removedObjects);
425     }
426   },
427 
428   _scac_arrayContentDidChange: function (start, removed, added) {
429     this._scac_cached = NO;
430     this.arrayContentDidChange(start, removed, added);
431 
432     // If the start & length are provided, we can also indicate if the firstObject
433     // or lastObject properties changed, thus making them independently observable.
434     if (!SC.none(start)) {
435       if (start === 0) this.notifyPropertyChange('firstObject');
436       var length = added + removed;
437       if (!SC.none(length) && start + length >= this.get('length') - 1) this.notifyPropertyChange('lastObject');
438     }
439 
440     if (this._kvo_enumerable_property_chains) {
441       var addedObjects = this.slice(start, start + added);
442       this.setupEnumerablePropertyChains(addedObjects);
443     }
444     this.updateSelectionAfterContentChange();
445   },
446 
447   /** @private
448     Whenever content changes, setup and teardown observers on the content
449     as needed.
450   */
451   _scac_contentDidChange: function () {
452     this._scac_cached = NO; // invalidate observable content
453     var content     = this.get('content'),
454         lastContent = this._scac_content,
455         didChange   = this._scac_arrayContentDidChange,
456         willChange  = this._scac_arrayContentWillChange,
457         sfunc       = this._scac_contentStatusDidChange,
458         efunc       = this._scac_enumerableDidChange,
459         newlen;
460 
461     if (content === lastContent) { return this; } // nothing to do
462 
463     // teardown old observer
464     if (lastContent) {
465       if (lastContent.isSCArray) {
466         lastContent.removeArrayObservers({
467           target: this,
468           didChange: didChange,
469           willChange: willChange
470         });
471       } else if (lastContent.isEnumerable) {
472         lastContent.removeObserver('[]', this, efunc);
473       }
474 
475       lastContent.removeObserver('status', this, sfunc);
476 
477       this.teardownEnumerablePropertyChains(lastContent);
478     }
479 
480     // save new cached values
481     this._scac_cached = NO;
482     this._scac_content = content;
483 
484     // setup new observer
485     // also, calculate new length.  do it manually instead of using
486     // get(length) because we want to avoid computed an ordered array.
487     if (content) {
488       // Content is an enumerable, so listen for changes to its
489       // content, and get its length.
490       if (content.isSCArray) {
491         content.addArrayObservers({
492           target: this,
493           didChange: didChange,
494           willChange: willChange
495         });
496 
497         newlen = content.get('length');
498       } else if (content.isEnumerable) {
499         content.addObserver('[]', this, efunc);
500         newlen = content.get('length');
501       } else {
502         // Assume that someone has set a non-enumerable as the content, and
503         // treat it as the sole member of an array.
504         newlen = 1;
505       }
506 
507       // Observer for changes to the status property, in case this is an
508       // SC.Record or SC.RecordArray.
509       content.addObserver('status', this, sfunc);
510 
511       this.setupEnumerablePropertyChains(content);
512     } else {
513       newlen = SC.none(content) ? 0 : 1;
514     }
515 
516     // finally, notify enumerable content has changed.
517     this._scac_length = newlen;
518     this._scac_contentStatusDidChange();
519 
520     this.arrayContentDidChange(0, 0, newlen);
521     this.enumerableContentDidChange(0, newlen - 1);
522     this.updateSelectionAfterContentChange();
523   }.observes('content'),
524 
525   /** @private
526     Whenever enumerable content changes, need to regenerate the
527     observableContent and notify that the range has changed.
528 
529     This is called whenever the content enumerable changes or whenever orderBy
530     changes.
531   */
532   _scac_enumerableDidChange: function () {
533     var content = this.get('content'), // use content directly
534         newlen  = content ? content.get('length') : 0,
535         oldlen  = this._scac_length;
536 
537     this._scac_length = newlen;
538     this.beginPropertyChanges();
539     this._scac_cached = NO; // invalidate
540     // If this is an unordered enumerable, we have no way
541     // of knowing which indices changed. Instead, we just
542     // invalidate the whole array.
543     this.arrayContentDidChange(0, oldlen, newlen);
544     this.enumerableContentDidChange(0, oldlen - 1);
545     this.endPropertyChanges();
546     this.updateSelectionAfterContentChange();
547   }.observes('orderBy'),
548 
549   /** @private
550     Whenever the content "status" property changes, relay out.
551   */
552   _scac_contentStatusDidChange: function () {
553     this.notifyPropertyChange('status');
554   }
