 10 var context = null;
 11 
 12 // ..........................................................
 13 // push()
 14 // 
 15 module("SC.RenderContext#push", {
 16   setup: function() {
 17     context = SC.RenderContext();
 18   },
 19   
 20   teardown: function() {
 21     context = null;
 22   }
 23 });
 24 
 25 test("it should add the line to the strings and increase the length", function() {
 26   equals(context.length, 1, "precondition - length=");
 27 
 28   context.push("sample line");
 29   equals(context.length, 2, "length should increase");
 30   equals(context.get(1), "sample line", "line should be in strings array");
 31 });
 32 
 33 test("it should accept multiple parameters, pushing each one into strings", function() {
 34 
 35   equals(context.length, 1, "precondition - length=");
 36   
 37   context.push("line1", "line2", "line3");
 38   equals(context.length, 4, "should add 3 lines to strings");
 39   equals(context.get(1), "line1", "1st item");
 40   equals(context.get(2), "line2", "2nd item");
 41   equals(context.get(3), "line3", "3rd item");
 42 });
 43 
 44 test("it should return receiver", function() {
 45   equals(context.push("line1"), context, "return value");
 46 });
 47 
 48 test("pushing a line onto a subcontext, should update the length in the parent context as well", function() {
 49   context.push("line1", "line2");
 50   var len = context.length ;
 51   
 52   var c2 = context.begin().push("line3");
 53   ok(context.length > len, "length should have increased");
 54 });
