 12 /**
 13   @namespace
 14 
 15   This private class is used to store information about observers on a
 16   particular key.  Note that this object is not observable.  You create new
 17   instances by calling SC.beget(SC.ObserverSet) ;
 18 
 19   @private
 20   @since SproutCore 1.0
 21 */
 22 SC.ObserverSet = {
 23 
 24   /**
 25     Adds the named target/method observer to the set.  The method must be
 26     a function, not a string.
 27   */
 28   add: function(target, method, context) {
 29     var targetGuid = SC.guidFor(target),
 30         methodGuid = SC.guidFor(method),
 31         targets    = this._members,
 32         members    = this.members,
 33         indexes    = targets[targetGuid],       // get the set of methods
 34         index, member;
 35 
 36     if ( !indexes ) indexes = targets[targetGuid] = {};
 37 
 38     index = indexes[methodGuid];
 39     if (index === undefined) {
 40       indexes[methodGuid] = members.length;
 41       member = [target, method, context];
 42 
 43       //@if(debug)
 44       // If deferred call logging info was specified (i.e., in debug mode when
 45       // such logging is enabled), we need to add it to the enqueued target/
 46       // method.
 47       member[3] = arguments[3];
 48       //@endif
 49 
 50       members.push(member);
 51     }
 52     else {
 53       //@if(debug)
 54       // If deferred call logging info was specified (i.e., in debug mode when
 55       // such logging is enabled), we need to add it to the enqueued target/
 56       // method.
 57       var loggingInfo = arguments[3],
 58           memberLoggingInfo;
 59 
 60       if (loggingInfo) {
 61         member            = members[index];
 62         memberLoggingInfo = member[3];
 63         if (!memberLoggingInfo) {
 64           member[3] = [loggingInfo];
 65         }
 66         else if (!(memberLoggingInfo instanceof Array)) {
 67           member[3] = [memberLoggingInfo, loggingInfo];
 68         }
 69         else {
 70           memberLoggingInfo.push(loggingInfo);
 71         }
 72       }
 73       //@endif
 74     }
 75   },
 76 
 77   /**
 78     removes the named target/method observer from the set.  If this is the
 79     last method for the named target, then the number of targets will also
 80     be reduced.
 81 
 82     returns YES if the items was removed, NO if it was not found.
 83   */
 84   remove: function(target, method) {
 85     var targetGuid = SC.guidFor(target), methodGuid = SC.guidFor(method);
 86     var indexes = this._members[targetGuid], members = this.members;
 87 
 88     if( !indexes ) return false;
 89 
 90     var index = indexes[methodGuid];
 91     if ( index === undefined) return false;
 92 
 93     if (index !== members.length - 1) {
 94       var entry = (members[index] = members[members.length - 1]);
 95       this._members[SC.guidFor(entry[0])][SC.guidFor(entry[1])] = index;
 96     }
 97 
 98     members.pop();
 99     delete this._members[targetGuid][methodGuid];
100 
101     return true;
102   },
103 
104   /**
105     Invokes the target/method pairs in the receiver.  Used by SC.RunLoop
106     Note: does not support context
107   */
108   invokeMethods: function() {
109     var members = this.members, member;
110 
111     //@if(debug)
112     var shouldLog = SC.LOG_DEFERRED_CALLS,
113         target, method, methodName, loggingInfo, loggingInfos,
114         originatingTarget, originatingMethod, originatingMethodName,
115         originatingStack, j, jLen;
116     //@endif
117 
118     for( var i=0, l=members.length; i<l; i++ ) {
119       member = members[i];
120 
121       // method.call(target);
122       member[1].call(member[0]);
123 
124       //@if(debug)
125       // If we have logging info specified for who scheduled the particular
126       // invocation, and logging is enabled, then output it.
127       if (shouldLog) {
128         target      = member[0];
129         method      = member[1];
130         methodName  = method.displayName || method;
131         loggingInfo = member[3];
132         if (loggingInfo) {
133           // If the logging info is not an array, that means only one place
134           // scheduled the invocation.
135           if (!(loggingInfo instanceof Array)) {
136             // We'll treat single-scheduler cases specially to make the output
137             // better for the user, even if it means some essentially-duplicated
138             // code.
139             originatingTarget     = loggingInfo.originatingTarget;
140             originatingMethod     = loggingInfo.originatingMethod;
141             originatingStack      = loggingInfo.originatingStack;
142             originatingMethodName = (originatingMethod ? originatingMethod.displayName : "(unknown)") || originatingMethod;
143             SC.Logger.log("Invoking runloop-scheduled method %@ on %@.  Originated by target %@,  method %@,  stack: ".fmt(methodName, target, originatingTarget, originatingMethodName), originatingStack);
144           }
145           else {
146             SC.Logger.log("Invoking runloop-scheduled method %@ on %@, which was scheduled by multiple target/method pairs:".fmt(methodName, target));
147             loggingInfos = loggingInfo;
148             for (j = 0, jLen = loggingInfos.length;  j < jLen;  ++j) {
149               loggingInfo           = loggingInfos[j];
150               originatingTarget     = loggingInfo.originatingTarget;
151               originatingMethod     = loggingInfo.originatingMethod;
152               originatingStack      = loggingInfo.originatingStack;
153               originatingMethodName = (originatingMethod ? originatingMethod.displayName : "(unknown)") || originatingMethod;
154               SC.Logger.log("  [%@]  originated by target %@,  method %@,  stack:".fmt(j, originatingTarget, originatingMethodName), originatingStack);
155             }
156           }
157         }
158         else {
159           // If we didn't capture information for this invocation, just report
160           // what we can.
161           SC.Logger.log("Invoking runloop-scheduled method %@ on %@, but we didn’t capture information about who scheduled it…".fmt(methodName, target));
162         }
163       }
164       //@endif
165     }
166   },
167 
168   /**
169     Returns a new instance of the set with the contents cloned.
170   */
171   clone: function() {
172     var newSet = SC.ObserverSet.create(), memberArray = this.members;
173 
174     newSet._members = SC.clone(this._members);
175     var newMembers = newSet.members;
176 
177     for( var i=0, l=memberArray.length; i<l; i++ ) {
178       newMembers[i] = SC.clone(memberArray[i]);
179       newMembers[i].length = 3;
180       //@if(debug)
181       newMembers[i].length = 4;
182       //@endif
183     }
184 
185     return newSet;
186   },
187 
188   /**
189     Creates a new instance of the observer set.
190   */
191   create: function() {
192     return new SC.ObserverSet.constructor();
193   },
194 
195   getMembers: function() {
196     return this.members.slice(0);
197   },
198 
199   constructor: function() {
200     this._members = {};
201     this.members = [];
202   }
203 
204 } ;
205 
206 SC.ObserverSet.constructor.prototype = SC.ObserverSet;
207 SC.ObserverSet.slice = SC.ObserverSet.clone;
208 SC.ObserverSet.copy = SC.ObserverSet.clone;
