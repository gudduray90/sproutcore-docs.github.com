 20 function evaluatePicker(pane) {
 21   ok(pane.get('isVisibleInWindow'), 'pane.isVisibleInWindow should be YES');
 22   ok(pane.$().hasClass('sc-picker'), 'pane should have sc-picker class');
 23   ok(pane.childViews[0].get('isVisibleInWindow'), 'pane.div.isVisibleInWindow should be YES');
 24   ok(pane.childViews[0].$().hasClass('sc-view'), 'pane.div should have sc-view class');
 25   
 26   equals(pane.$('.sc-view').length, 1, 'pane should have only one content view');
 27   
 28   var ret = pane.layoutStyle();
 29 
 30   equals(ret.width, '300px', 'pane should have width 300px');
 31   equals(ret.height, '200px', 'pane should have height 200px');
 32 }
