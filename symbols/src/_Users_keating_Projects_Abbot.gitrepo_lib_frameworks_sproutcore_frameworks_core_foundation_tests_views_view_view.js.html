 10 module("SC.View");
 11 
 12 test("setting themeName should trigger a theme observer", function() {
 13   var count = 0;
 14   var view = SC.View.create({
 15     themeDidChange: function() {
 16       count++;
 17     }.observes('theme')
 18   });
 19 
 20   view.set('themeName', 'hello');
 21   equals(1, count, "theme observers should get called");
 22 });
 23 
 24 test("setting themeName should trigger a theme observer when extending", function() {
 25   var count = 0;
 26   var View = SC.View.extend({
 27     themeDidChange: function() {
 28       count++;
 29     }.observes('theme')
 30   });
 31 
 32   View.create().set('themeName', 'hello');
 33   equals(1, count, "theme observers should get called");
 34 });
 35 
 36 test("it still works with the backward compatible theme property", function() {
 37   var count = 0;
 38   var view = SC.View.create({
 39     theme: 'sc-base',
 40     themeDidChange: function() {
 41       count++;
 42     }.observes('theme')
 43   });
 44 
 45   equals(SC.Theme.find('sc-base'), view.get('theme'));
 46   view.set('themeName', 'hello');
 47   equals(1, count, "theme observers should get called");
 48 });
 49 
 50 test("it still works with the backward compatible theme property when extending", function() {
 51   var count = 0;
 52   var View = SC.View.extend({
 53     theme: 'sc-base',

 56     }.observes('theme')
 57   });
 58

 63 });
