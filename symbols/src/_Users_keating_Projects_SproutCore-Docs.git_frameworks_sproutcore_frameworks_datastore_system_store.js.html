 13 
 14   The Store is where you can find all of your dataHashes. Stores can be
 15   chained for editing purposes and committed back one chain level at a time
 16   all the way back to a persistent data source.
 17

 24   be a server, local storage, or any other persistent code.

123   /**
124     Returns a new nested store instance that can be used to buffer changes
125     until you are ready to commit them.  When you are ready to commit your
126     changes, call `commitChanges()` or `destroyChanges()` and then `destroy()`
127     when you are finished with the chained store altogether.
128 
129         store = MyApp.store.chain();
130         .. edit edit edit
131         store.commitChanges().destroy();
132 
133     @param {Hash} attrs optional attributes to set on new store
134     @param {Class} newStoreClass optional the class of the newly-created nested store (defaults to SC.NestedStore)
135     @returns {SC.NestedStore} new nested store chained to receiver
136   */
137   chain: function(attrs, newStoreClass) {
138     if (!attrs) attrs = {};
139     attrs.parentStore = this;
140 
141     if (newStoreClass) {
142       // Ensure the passed-in class is a type of nested store.
143       if (SC.typeOf(newStoreClass) !== 'class') throw new Error("%@ is not a valid class".fmt(newStoreClass));
144       if (!SC.kindOf(newStoreClass, SC.NestedStore)) throw new Error("%@ is not a type of SC.NestedStore".fmt(newStoreClass));
145     }
146     else {
147       newStoreClass = SC.NestedStore;
148     }
149 
150     // Replicate parent records references
151     attrs.childRecords = this.childRecords ? SC.clone(this.childRecords) : {};
152     attrs.parentRecords = this.parentRecords ? SC.clone(this.parentRecords) : {};
153 
154     var ret    = newStoreClass.create(attrs),
155         nested = this.nestedStores;
156 
157     if (!nested) nested = this.nestedStores = [];
158     nested.push(ret);
159     return ret ;
160   },

475   /**
476     Call this method whenever you modify some editable data hash to register
477     with the Store that the attribute values have actually changed.  This will
478     do the book-keeping necessary to track the change across stores including
479     managing locks.
480 
481     @param {Number|Array} storeKeys one or more store keys that changed
482     @param {Number} rev optional new revision number. normally leave null
483     @param {Boolean} statusOnly (optional) YES if only status changed
484     @param {String} key that changed (optional)
485     @returns {SC.Store} receiver
486   */
487   dataHashDidChange: function(storeKeys, rev, statusOnly, key) {
488 
489     // update the revision for storeKey.  Use generateStoreKey() because that
490     // guarantees a universally (to this store hierarchy anyway) unique
491     // key value.
492     if (!rev) rev = SC.Store.generateStoreKey();
493     var isArray, len, idx, storeKey;
494 
495     isArray = SC.typeOf(storeKeys) === SC.T_ARRAY;
496     if (isArray) {
497       len = storeKeys.length;
498     } else {
499       len = 1;
500       storeKey = storeKeys;
501     }
502 
503     var that = this;
504     for(idx=0;idx<len;idx++) {
505       if (isArray) storeKey = storeKeys[idx];
506       this.revisions[storeKey] = rev;
507       this._notifyRecordPropertyChange(storeKey, statusOnly, key);
508 
509       this._propagateToChildren(storeKey, function(storeKey){
510         that.dataHashDidChange(storeKey, null, statusOnly, key);
511       });
512     }
513 
514     return this ;
515   },

517   /** @private
518     Will push all changes to a the recordPropertyChanges property
519     and execute `flush()` once at the end of the runloop.
520   */
521   _notifyRecordPropertyChange: function(storeKey, statusOnly, key) {
522 
523     var records      = this.records,
524         nestedStores = this.get('nestedStores'),
525         K            = SC.Store,
526         rec, editState, len, idx, store, status, keys;
527 
528     // pass along to nested stores
529     len = nestedStores ? nestedStores.length : 0 ;
530     for(idx=0;idx<len;idx++) {
531       store = nestedStores[idx];
532       status = store.peekStatus(storeKey); // important: peek avoids read-lock
533       editState = store.storeKeyEditState(storeKey);
534 
535       // when store needs to propagate out changes in the parent store
536       // to nested stores
537       if (editState === K.INHERITED) {
538         store._notifyRecordPropertyChange(storeKey, statusOnly, key);
539 
540       } else if (status & SC.Record.BUSY) {
541         // make sure nested store does not have any changes before resetting
542         if(store.get('hasChanges')) throw K.CHAIN_CONFLICT_ERROR;
543         store.reset();
544       }
545     }
546 
547     // store info in changes hash and schedule notification if needed.
548     var changes = this.recordPropertyChanges;
549     if (!changes) {
550       changes = this.recordPropertyChanges =
551         { storeKeys:      SC.CoreSet.create(),
552           records:        SC.CoreSet.create(),
553           hasDataChanges: SC.CoreSet.create(),
554           propertyForStoreKeys: {} };
555     }
556 
557     changes.storeKeys.add(storeKey);
558 
559     if (records && (rec=records[storeKey])) {
560       changes.records.push(storeKey);
561 
562       // If there are changes other than just the status we need to record
563       // that information so we do the right thing during the next flush.
564       // Note that if we're called multiple times before flush and one call
565       // has `statusOnly=true` and another has `statusOnly=false`, the flush
566       // will (correctly) operate in `statusOnly=false` mode.
567       if (!statusOnly) changes.hasDataChanges.push(storeKey);
568 
569       // If this is a key specific change, make sure that only those
570       // properties/keys are notified.  However, if a previous invocation of
571       // `_notifyRecordPropertyChange` specified that all keys have changed, we
572       // need to respect that.
573       if (key) {
574         if (!(keys = changes.propertyForStoreKeys[storeKey])) {
575           keys = changes.propertyForStoreKeys[storeKey] = SC.CoreSet.create();
576         }
577 
578         // If it's '*' instead of a set, then that means there was a previous
579         // invocation that said all keys have changed.
580         if (keys !== '*') {
581           keys.add(key);
582         }
583       }
584       else {
585         // Mark that all properties have changed.
586         changes.propertyForStoreKeys[storeKey] = '*';
587       }
588     }
589 
590     this.invokeOnce(this.flush);
591     return this;
592   },

594   /**
595     Delivers any pending changes to materialized records.  Normally this
596     happens once, automatically, at the end of the RunLoop.  If you have
597     updated some records and need to update records immediately, however,
598     you may call this manually.
599 
600     @returns {SC.Store} receiver
601   */
602   flush: function() {
603     if (!this.recordPropertyChanges) return this;
604 
605     var changes              = this.recordPropertyChanges,
606         storeKeys            = changes.storeKeys,
607         hasDataChanges       = changes.hasDataChanges,
608         records              = changes.records,
609         propertyForStoreKeys = changes.propertyForStoreKeys,
610         recordTypes = SC.CoreSet.create(),
611         rec, recordType, statusOnly, idx, len, storeKey, keys;
612 
613     storeKeys.forEach(function(storeKey) {
614       if (records.contains(storeKey)) {
615         statusOnly = hasDataChanges.contains(storeKey) ? NO : YES;
616         rec = this.records[storeKey];
617         keys = propertyForStoreKeys ? propertyForStoreKeys[storeKey] : null;
618 
619         // Are we invalidating all keys?  If so, don't pass any to
620         // storeDidChangeProperties.
621         if (keys === '*') keys = null;
622 
623         // remove it so we don't trigger this twice
624         records.remove(storeKey);
625 
626         if (rec) rec.storeDidChangeProperties(statusOnly, keys);
627       }
628 
629       recordType = SC.Store.recordTypeFor(storeKey);
630       recordTypes.add(recordType);
631 
632     }, this);
633 
634     if (storeKeys.get('length') > 0) this._notifyRecordArrays(storeKeys, recordTypes);
635 
636     storeKeys.clear();
637     hasDataChanges.clear();
638     records.clear();
639     // Provide full reference to overwrite
640     this.recordPropertyChanges.propertyForStoreKeys = {};
641 
642     return this;
643   },

679   /** @private
680     Called by a nested store on a parent store to commit any changes from the
681     store.  This will copy any changed dataHashes as well as any persistent
682     change logs.
683 
684     If the parentStore detects a conflict with the optimistic locking, it will
685     raise an exception before it makes any changes.  If you pass the
686     force flag then this detection phase will be skipped and the changes will
687     be applied even if another resource has modified the store in the mean
688     time.
689 
690     @param {SC.Store} nestedStore the child store
691     @param {SC.Set} changes the set of changed store keys
692     @param {Boolean} force
693     @returns {SC.Store} receiver
694   */
695   commitChangesFromNestedStore: function(nestedStore, changes, force) {
696     // first, check for optimistic locking problems
697     if (!force) this._verifyLockRevisions(changes, nestedStore.locks);
698 
699     // OK, no locking issues.  So let's just copy them changes.
700     // get local reference to values.
701     var len = changes.length, i, storeKey, myDataHashes, myStatuses,
702       myEditables, myRevisions, myParentRecords, myChildRecords,
703       chDataHashes, chStatuses, chRevisions, chParentRecords, chChildRecords;
704 
705     myRevisions     = this.revisions ;
706     myDataHashes    = this.dataHashes;
707     myStatuses      = this.statuses;
708     myEditables     = this.editables ;
709     myParentRecords = this.parentRecords ? this.parentRecords : this.parentRecords ={} ;
710     myChildRecords  = this.childRecords ? this.childRecords : this.childRecords = {} ;
711 
712     // setup some arrays if needed
713     if (!myEditables) myEditables = this.editables = [] ;
714     chDataHashes    = nestedStore.dataHashes;
715     chRevisions     = nestedStore.revisions ;
716     chStatuses      = nestedStore.statuses;
717     chParentRecords = nestedStore.parentRecords || {};
718     chChildRecords  = nestedStore.childRecords || {};
719 
720     for(i=0;i<len;i++) {
721       storeKey = changes[i];
722 
723       // now copy changes
724       myDataHashes[storeKey]    = chDataHashes[storeKey];
725       myStatuses[storeKey]      = chStatuses[storeKey];
726       myRevisions[storeKey]     = chRevisions[storeKey];
727       myParentRecords[storeKey] = chParentRecords[storeKey];
728       myChildRecords[storeKey]  = chChildRecords[storeKey];
729 
730       myEditables[storeKey] = 0 ; // always make dataHash no longer editable
731

1073   /**
1074     Creates a new record instance with the passed `recordType` and `dataHash`.
1075     You can also optionally specify an id or else it will be pulled from the
1076     data hash.
1077 
1078     Example:
1079 
1080       MyApp.Record = SC.Record.extend({
1081         attrA: SC.Record.attr(String, { defaultValue: 'def' }),

1087       // defaultValue will be used.
1088       MyApp.store.createRecord(MyApp.Record).get('attributes');

1092       // attribute name when creating the record, but if you specify both, only
1093       // the key name will be used.
1094       MyApp.store.createRecord(MyApp.Record, { isAttrB: YES }).get('attributes');
1095       > { attr_b: YES }
1096       MyApp.store.createRecord(MyApp.Record, { attr_b: YES }).get('attributes');
1097       > { attr_b: YES }
1098       MyApp.store.createRecord(MyApp.Record, { isAttrB: NO, attr_b: YES }).get('attributes');
1099       > { attr_b: YES }

1101     Note that the record will not yet be saved back to the server.  To save
1102     a record to the server, call `commitChanges()` on the store.
1103 
1104     @param {SC.Record} recordType the record class to use on creation
1105     @param {Hash} dataHash the JSON attributes to assign to the hash.
1106     @param {String} id (optional) id to assign to record
1107 
1108     @returns {SC.Record} Returns the created record
1109   */
1110   createRecord: function(recordType, dataHash, id) {
1111     var primaryKey, prototype, storeKey, status, K = SC.Record, changelog, defaultVal, ret;
1112 
1113     //initialize dataHash if necessary
1114     dataHash = (dataHash ? dataHash : {});
1115 
1116     // First, try to get an id.  If no id is passed, look it up in the
1117     // dataHash.
1118     if (!id && (primaryKey = recordType.prototype.primaryKey)) {
1119       id = dataHash[primaryKey];
1120       // if still no id, check if there is a defaultValue function for
1121       // the primaryKey attribute and assign that
1122       defaultVal = recordType.prototype[primaryKey] ? recordType.prototype[primaryKey].defaultValue : null;
1123       if(!id && SC.typeOf(defaultVal)===SC.T_FUNCTION) {
1124         id = dataHash[primaryKey] = defaultVal();
1125       }
1126     }
1127 
1128     // Next get the storeKey - base on id if available
1129     storeKey = id ? recordType.storeKeyFor(id) : SC.Store.generateStoreKey();
1130 
1131     // now, check the state and do the right thing.
1132     status = this.readStatus(storeKey);
1133 
1134     // check state
1135     // any busy or ready state or destroyed dirty state is not allowed
1136     if ((status & K.BUSY)  ||
1137         (status & K.READY) ||
1138         (status === K.DESTROYED_DIRTY)) {
1139       throw id ? K.RECORD_EXISTS_ERROR : K.BAD_STATE_ERROR;
1140 
1141     // allow error or destroyed state only with id
1142     } else if (!id && (status===SC.DESTROYED_CLEAN || status===SC.ERROR)) {
1143       throw K.BAD_STATE_ERROR;
1144     }
1145 
1146     // Store the dataHash and setup initial status.
1147     this.writeDataHash(storeKey, dataHash, K.READY_NEW);
1148 
1149     // Register the recordType with the store.
1150     SC.Store.replaceRecordTypeFor(storeKey, recordType);
1151     this.dataHashDidChange(storeKey);
1152 
1153     // If the attribute wasn't provided in the dataHash, attempt to insert a
1154     // default value.  We have to do this after materializing the record,
1155     // because the defaultValue property may be a function that expects
1156     // the record as an argument.
1157     ret = this.materializeRecord(storeKey);
1158     prototype = recordType.prototype;
1159     for (var prop in prototype) {
1160       var propPrototype = prototype[ prop ];
1161       if (propPrototype && propPrototype.isRecordAttribute) {
1162         // Use the record attribute key if it is defined.
1163         var attrKey = propPrototype.key || prop;
1164 
1165         if (!dataHash.hasOwnProperty(attrKey)) {
1166           if (dataHash.hasOwnProperty(prop)) {
1167             // If the attribute key doesn't exist but the name does, fix it up.
1168             // (i.e. the developer has a record attribute `endDate` with a key
1169             // `end_date` on a record and when they created the record they
1170             // provided `endDate` not `end_date`)
1171             dataHash[ attrKey ] = dataHash[ prop ];
1172             delete dataHash[ prop ];
1173           } else {
1174             // If the attribute doesn't exist in the hash at all, check for a
1175             // default value to use instead.
1176             defaultVal = propPrototype.defaultValue;
1177             if (defaultVal) {
1178               if (SC.typeOf(defaultVal)===SC.T_FUNCTION) dataHash[ attrKey ] = SC.copy(defaultVal(ret, attrKey), YES);
1179               else dataHash[ attrKey ] = SC.copy(defaultVal, YES);
1180             }
1181           }
1182         } else if (attrKey !== prop && dataHash.hasOwnProperty(prop)) {
1183           // If both attrKey and prop are provided, use attrKey only.
1184           delete dataHash[ prop ];
1185         }
1186       }
1187     }
1188 
1189     // Record is now in a committable state -- add storeKey to changelog
1190     changelog = this.changelog;
1191     if (!changelog) changelog = SC.Set.create();
1192     changelog.add(storeKey);
1193     this.changelog = changelog;
1194 
1195     // if commit records is enabled
1196     if(this.get('commitRecordsAutomatically')){
1197       this.invokeLast(this.commitRecords);
1198     }
1199 
1200     // Propagate the status to any aggregate records before returning.
1201     if (ret) ret.propagateToAggregates();
1202     return ret;
1203   },

1604   /**
1605     Retrieves a set of records from the server.  If the records has

1614

1617     @param {Array} storeKeys (optional) store keys to retrieve

1619     @param {Function|Array} callback function or array of functions
1620     @returns {Array} storeKeys to be retrieved
1621   */
1622   retrieveRecords: function(recordTypes, ids, storeKeys, isRefresh, callbacks) {
1623 
1624     var source  = this._getDataSource(),
1625         isArray = SC.typeOf(recordTypes) === SC.T_ARRAY,
1626         hasCallbackArray = SC.typeOf(callbacks) === SC.T_ARRAY,
1627         len     = (!storeKeys) ? ids.length : storeKeys.length,
1628         ret     = [],
1629         rev     = SC.Store.generateStoreKey(),
1630         K       = SC.Record,
1631         recordType, idx, storeKey, status, ok, callback;
1632 
1633     if (!isArray) recordType = recordTypes;
1634 
1635     // if no storeKeys were passed, map recordTypes + ids
1636     for(idx=0;idx<len;idx++) {
1637 
1638       // collect store key
1639       if (storeKeys) {
1640         storeKey = storeKeys[idx];
1641       } else {
1642         if (isArray) recordType = recordTypes[idx];
1643         storeKey = recordType.storeKeyFor(ids[idx]);
1644       }
1645       //collect the callback
1646       callback = hasCallbackArray ? callbacks[idx] : callbacks;
1647 
1648       // collect status and process
1649       status = this.readStatus(storeKey);
1650 
1651       // K.EMPTY, K.ERROR, K.DESTROYED_CLEAN - initial retrieval
1652       if ((status == K.EMPTY) || (status == K.ERROR) || (status == K.DESTROYED_CLEAN)) {
1653         this.writeStatus(storeKey, K.BUSY_LOADING);
1654         this.dataHashDidChange(storeKey, rev, YES);
1655         ret.push(storeKey);
1656         this._setCallbackForStoreKey(storeKey, callback, hasCallbackArray, storeKeys);
1657       // otherwise, ignore record unless isRefresh is YES.
1658       } else if (isRefresh) {
1659         // K.READY_CLEAN, K.READY_DIRTY, ignore K.READY_NEW
1660         if (status & K.READY) {
1661           this.writeStatus(storeKey, K.BUSY_REFRESH | (status & 0x03)) ;
1662           this.dataHashDidChange(storeKey, rev, YES);
1663           ret.push(storeKey);
1664           this._setCallbackForStoreKey(storeKey, callback, hasCallbackArray, storeKeys);
1665         // K.BUSY_DESTROYING, K.BUSY_COMMITTING, K.BUSY_CREATING
1666         } else if ((status == K.BUSY_DESTROYING) || (status == K.BUSY_CREATING) || (status == K.BUSY_COMMITTING)) {
1667           throw K.BUSY_ERROR ;
1668 
1669         // K.DESTROY_DIRTY, bad state...
1670         } else if (status == K.DESTROYED_DIRTY) {
1671           throw K.BAD_STATE_ERROR ;
1672 
1673         // ignore K.BUSY_LOADING, K.BUSY_REFRESH_CLEAN, K.BUSY_REFRESH_DIRTY
1674         }
1675       }
1676     }
1677 
1678     // now retrieve storekeys from dataSource.  if there is no dataSource,
1679     // then act as if we couldn't retrieve.
1680     ok = NO;
1681     if (source) ok = source.retrieveRecords.call(source, this, ret, ids);
1682 
1683     // if the data source could not retrieve or if there is no source, then
1684     // simulate the data source calling dataSourceDidError on those we are
1685     // loading for the first time or dataSourceDidComplete on refreshes.
1686     if (!ok) {
1687       len = ret.length;
1688       rev = SC.Store.generateStoreKey();
1689       for(idx=0;idx<len;idx++) {
1690         storeKey = ret[idx];
1691         status   = this.readStatus(storeKey);
1692         if (status === K.BUSY_LOADING) {
1693           this.writeStatus(storeKey, K.ERROR);
1694           this.dataHashDidChange(storeKey, rev, YES);
1695 
1696         } else if (status & K.BUSY_REFRESH) {
1697           this.writeStatus(storeKey, K.READY | (status & 0x03));
1698           this.dataHashDidChange(storeKey, rev, YES);
1699         }
1700       }
1701       ret.length = 0 ; // truncate to indicate that none could refresh
1702     }
1703     return ret ;
1704   },
