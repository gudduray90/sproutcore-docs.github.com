  8 /*global module test equals context */
  9 
 10 var context = null;
 11 
 12 module("SC.RenderContext#get", {
 13   setup: function() {
 14     context = SC.RenderContext();
 15   },
 16   
 17   teardown: function() {
 18     context = null;
 19   }
 20 });
 21 
 22 test("it should return strings array with space for top tag if no params passed and no strings pushed yet", function() {
 23   same(context.get(), [null]);
 24 });
 25 
 26 test("it should return full strings array if no params passed and no strings pushed yet", function() {
 27   context.push("line1");
 28   same(context.get(), [null, "line1"]);
 29 });
 30 
 31 test("it should return individual string if index passed that is within current length", function() {
 32   context.push("line1");
 33   equals(context.get(1), "line1");
 34 });
 35 
 36 test("it should return undefined if index passed that is outside of current range", function() {
 37   context.push("line1");
 38   equals(context.get(3), undefined);
 39 });
 40 
 41 // test this special case since the internal strings array is created lazily.
 42 test("it should return undefined if index passed and no strings set yet", function() {
 43   equals(context.get(2), undefined);
 44 });
 45 
 46 test("it should return the value based on an index from the context offset of the context is chained", function() {
 47   context.push('line1', 'line2');
 48   var childContext = context.begin();
 49   childContext.push("NEXT");
 50   equals(childContext.get(1), "NEXT", 'gets child line');
 51 }) ;
