  8 SC.AceTheme.Popover.workspaceRenderDelegate = SC.RenderDelegate.create({
  9   className: 'workspace',
 10   render: function(dataSource, context) {
 11     context.setClass({
 12       'top-toolbar': dataSource.get('hasTopToolbar'),
 13       'bottom-toolbar': dataSource.get('hasBottomToolbar')
 14     });
 15     
 16     context = context.begin('div').addClass('popover-background');
 17     this.includeSlices(dataSource, context, SC.NINE_SLICE);
 18     context.push("<div class = 'sc-pointer'></div>");
 19     context = context.end();
 20   },
 21 
 22   update: function(dataSource, jquery) {
 23     jquery.setClass({
 24       'top-toolbar': dataSource.get('hasTopToolbar'),
 25       'bottom-toolbar': dataSource.get('hasBottomToolbar')
 26     });
 27   }
 28 });
