 14     window.target = SC.Object.create({

 30 var anchor = SC.ControlTestPane.design()
 31   .add("anchor", SC.ButtonView, { 
 32      title: "Anchor Button" 
 33   });
 34 
 35 anchor.show(); // add a test to show the test pane
 36 
 37 var pane;
 38 
 39 test("check default action on pane#mouseDown", function() { 
 40   pane = SC.PickerPane.create({
 41     layout: { width: 300, height: 200 },
 42     contentView: SC.View.extend({
 43       layout: { top: 0, left: 0, bottom: 0, right: 0 }
 44     })
 45   });
 46   pane.popup(anchor.view('anchor'), SC.PICKER);
 47   
 48   ok(pane.get('isVisibleInWindow'), 'pane.isVisibleInWindow should be YES');
 49   
 50   pane.mouseDown({ pageX: 0, pageY: 0 });
 51   
 52   ok(!pane.get('isVisibleInWindow'), 'pane.isVisibleInWindow should be NO after mouseDown');
 53 }) ;
