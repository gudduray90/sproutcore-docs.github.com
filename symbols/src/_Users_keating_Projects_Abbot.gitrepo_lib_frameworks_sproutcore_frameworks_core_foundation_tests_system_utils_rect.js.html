 14 test("Get the X & Y points of a rect", function() {
 15   var frame = { x: 50, y: 40, width: 700, height: 9000 };
 16   expect(6);
 17   equals(SC.minX(frame),50,'Left edge');
 18   equals(SC.maxX(frame),750,'Right edge');
 19   equals(SC.midX(frame),400,'Horizontal midpoint');
 20   
 21   equals(SC.minY(frame),40, 'Top edge');
 22   equals(SC.maxY(frame),9040,'Bottom edge');
 23   equals(SC.midY(frame),4540,'Vertical midpoint');
 24 });

 38 test("pointInRect() to test if a given point is inside the rect", function(){
 39   var frame = { x: 50, y: 40, width: 700, height: 9000 };
 40   
 41   ok(SC.pointInRect({ x: 100, y: 100 }, frame), "Point in rect");
 42   equals(NO, SC.pointInRect({ x: 40, y: 100 }, frame), "Point out of rect horizontally");
 43   equals(NO, SC.pointInRect({ x: 600, y: 9100 }, frame), "Point out of rect vertically");
 44   equals(NO, SC.pointInRect({ x: 0, y: 0 }, frame), "Point up and left from rect");
 45   equals(NO, SC.pointInRect({ x: 800, y: 9500 }, frame), "Point down and right from rect");
 46 });

 48 test("rectsEqual() tests equality with default delta", function() {
 49   var frame = { x: 50, y: 50, width: 100, height: 100 };
 50   
 51   equals(SC.rectsEqual(frame, frame), YES, "Frames are same object");
 52   equals(SC.rectsEqual(frame, { x: 50, y: 50, width: 100, height: 100 }), YES, "Frames have same position and dimensions");
 53   equals(SC.rectsEqual(frame, { x: 50.08, y: 50, width: 100, height: 100 }), YES, "Frame.x above, within delta");
 54   equals(SC.rectsEqual(frame, { x: 49.92, y: 50, width: 100, height: 100 }), YES, "Frame.x below, within delta");
 55   equals(SC.rectsEqual(frame, { x: 50, y: 50.099, width: 100, height: 100 }), YES, "Frame.y within delta");
 56   equals(SC.rectsEqual(frame, { x: 50, y: 50, width: 100.001, height: 100 }), YES, "Frame.width within delta");
 57   equals(SC.rectsEqual(frame, { x: 50, y: 50, width: 100, height: 100.09999 }), YES, "Frame.height within delta");
 58   equals(SC.rectsEqual(frame, { x: 55, y: 50, width: 100, height: 100 }), NO, "Frame.x not equal");
 59   equals(SC.rectsEqual(frame, { x: 50, y: 55, width: 100, height: 100 }), NO, "Frame.y not equal");
 60   equals(SC.rectsEqual(frame, { x: 50, y: 50, width: 105, height: 100 }), NO, "Frame.width not equal");
 61   equals(SC.rectsEqual(frame, { x: 50, y: 50, width: 100, height: 105 }), NO, "Frame.height not equal");
 62 });
