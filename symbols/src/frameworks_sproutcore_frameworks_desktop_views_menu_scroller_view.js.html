  7 sc_require('views/scroller');
  8 
  9 /** @class
 10 
 11   Implements a complete scroller view for menus.  This class implements the
 12   arrows displayed in a menu to scroll.
 13 
 14   The main difference with SC.ScrollerView is that there is only vertical
 15   scrollers. Value Syncing between SC.MenuScrollView and SC.MenuScrollerView
 16   is done using valueBinding.
 17 
 18   @extends SC.ScrollerView
 19   @since SproutCore 1.0
 20 */
 21 SC.MenuScrollerView = SC.ScrollerView.extend(
 22 /** @scope SC.MenuScrollerView.prototype */ {
 23 
 24   // ---------------------------------------------------------------------------------------------
 25   // Properties
 26   //
 27 
 28   /**
 29     @type Array
 30     @default ['sc-menu-scroller-view']
 31     @see SC.View#classNames
 32   */
 33   classNames: ['sc-menu-scroller-view'],
 34 
 35   /**
 36     Used to set the scrolling direction of the scroller.
 37 
 38     @type Boolean
 39     @default false
 40   */
 41   scrollDown: false,
 42 
 43   /**
 44      Amount to scroll one vertical line.
 45 
 46      @type Number
 47      @default 20
 48   */
 49   verticalLineScroll: 20,
 50 
 51 
 52   // ..........................................................
 53   // INTERNAL SUPPORT
 54   //
 55 
 56   /** @private */
 57   init: function () {
 58     // Set the scrollerThickness based on controlSize
 59     switch (this.get('controlSize')) {
 60     case SC.TINY_CONTROL_SIZE:
 61       this.set('scrollerThickness', SC.MenuScrollerView.TINY_SCROLLER_THICKNESS);
 62       break;
 63     case SC.SMALL_CONTROL_SIZE:
 64       this.set('scrollerThickness', SC.MenuScrollerView.SMALL_SCROLLER_THICKNESS);
 65       break;
 66     case SC.REGULAR_CONTROL_SIZE:
 67       this.set('scrollerThickness', SC.MenuScrollerView.REGULAR_SCROLLER_THICKNESS);
 68       break;
 69     case SC.LARGE_CONTROL_SIZE:
 70       this.set('scrollerThickness', SC.MenuScrollerView.LARGE_SCROLLER_THICKNESS);
 71       break;
 72     case SC.HUGE_CONTROL_SIZE:
 73       this.set('scrollerThickness', SC.MenuScrollerView.HUGE_SCROLLER_THICKNESS);
 74       break;
 75     }
 76 
 77     return sc_super();
 78   },
 79 
 80   /** @private */
 81   render: function (context, firstTime) {
 82     context.addClass('sc-vertical');
 83     context.addClass(this.get('controlSize'));
 84     if (firstTime) {
 85       var direction = this.get('scrollDown') ? 'arrowDown' : 'arrowUp';
 86       context.push('<span class="scrollArrow ' + direction + '"> </span>');
 87     }
 88   },
 89 
 90   /** @private */
 91   willDestroyLayer: function () {
 92     var callback = this._sc_scroller_scrollDidChange;
 93     SC.Event.remove(this.$(), 'scroll', this, callback);
 94   },
 95 
 96   /** @private */
 97   mouseEntered: function (evt) {
 98     this.set('isMouseOver', YES);
 99     this._invokeScrollOnMouseOver();
100   },
101 
102   /** @private */
103   mouseExited: function (evt) {
104     this.set('isMouseOver', NO);
105   },
106 
107   /** @private */
108   _sc_scroller_armScrollTimer: function () {
109     if (!this._sc_scrollTimer) {
110       SC.run(function () {
111         var method = this._sc_scroller_scrollDidChange;
112         this._sc_scrollTimer = this.invokeLater(method, 50);
113       });
114     }
115   },
116 
117   /** @private */
118   _sc_scroller_scrollDidChange: function () {
119     var now = Date.now(),
120         last = this._sc_lastScroll,
121         layer = this.get('layer'),
122         scroll = 0;
123 
124     if (last && (now - last) < 50) return this._sc_scroller_armScrollTimer();
125     this._sc_scrollTimer = null;
126     this._sc_lastScroll = now;
127 
128     SC.run(function () {
129       if (!this.get('isEnabledInPane')) return; // nothing to do.
130 
131       this._sc_scrollValue = scroll = layer.scrollTop;
132       this.set('value', scroll); // will now enforce minimum and maximum
133     });
134   },
135 
136 
137   /** @private
138     Scroll the menu if it is is an up or down arrow. This is called by
139     the function that simulates mouseOver.
140   */
141   _scrollMenu: function () {
142     var value = this.get('value');
143 
144     if (this.get('scrollDown')) {
145       value = value + this.verticalLineScroll;
146       value = Math.min(value, this.get('maximum'));
147     } else {
148       value = value - this.verticalLineScroll;
149       value = Math.max(value, this.get('minimum'));
150     }
151 
152     this.set('value', value);
153 
154     return YES;
155   },
156 
157   /** @private
158 
159     We use this function to simulate mouseOver. It checks for the flag
160     isMouseOver which is turned on when mouseEntered is called and turned off
161     when mouseExited is called.
162   */
163   _invokeScrollOnMouseOver: function () {
164     this._scrollMenu();
165     if (this.get('isMouseOver')) {
166       this.invokeLater(this._invokeScrollOnMouseOver, 50);
167     }
168   }
169 
170 });
171 
172 
173 /**
174   @static
175   @type Number
176   @default 18
177 */
178 SC.MenuScrollerView.REGULAR_SCROLLER_THICKNESS = 18;
179 
180 /**
181   @static
182   @type Number
183   @default 10
184 */
185 SC.MenuScrollerView.TINY_SCROLLER_THICKNESS    = 10;
186 
187 /**
188   @static
189   @type Number
190   @default 14
191 */
192 SC.MenuScrollerView.SMALL_SCROLLER_THICKNESS   = 14;
193 
194 /**
195   @static
196   @type Number
197   @default 23
198 */
199 SC.MenuScrollerView.LARGE_SCROLLER_THICKNESS   = 23;
200 
201 /**
202   @static
203   @type Number
204   @default 26
205 */
206 SC.MenuScrollerView.HUGE_SCROLLER_THICKNESS    = 26;
