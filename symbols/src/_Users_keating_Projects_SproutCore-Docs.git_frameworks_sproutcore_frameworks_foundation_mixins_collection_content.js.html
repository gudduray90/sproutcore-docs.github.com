  8 /**
  9   Used for contentIndexDisclosureState().  Indicates open branch node.
 10   
 11   @type Number
 12   @constant
 13 */
 14 SC.BRANCH_OPEN = 0x0011;
 15 
 16 /**
 17   Used for contentIndexDisclosureState().  Indicates closed branch node.
 18   
 19   @type Number
 20   @constant
 21 */
 22 SC.BRANCH_CLOSED = 0x0012;
 23 
 24 /**
 25   Used for contentIndexDisclosureState().  Indicates leaf node.
 26   
 27   @type Number
 28   @constant
 29 */
 30 SC.LEAF_NODE = 0x0020;
 31 
 32 /**
 33   @namespace
 34 
 35   This mixin provides standard methods used by a CollectionView to provide
 36   additional meta-data about content in a collection view such as selection
 37   or enabled state.
 38   
 39   You can apply this mixin to a class that you set as a delegate or to the
 40   object you set as content.
 41   
 42   @since SproutCore 1.0
 43 */
 44 SC.CollectionContent = {
 45 
 46   /**
 47     Used to detect the mixin by SC.CollectionView
 48 
 49     @type Boolean
 50   */
 51   isCollectionContent: YES,
 52   
 53   /**
 54     Return YES if the content index should be selected.  Default behavior 
 55     looks at the selection property on the view.
 56     
 57     @param {SC.CollectionView} view the collection view
 58     @param {SC.Array} content the content object
 59     @param {Number} idx the content index
 60     @returns {Boolean} YES, NO, or SC.MIXED_STATE
 61   */
 62   contentIndexIsSelected: function(view, content, idx) {
 63     var sel = view.get('selection');
 64     return sel ? sel.contains(content, idx) : NO ;
 65   },
 66   
 67   /**
 68     Returns YES if the content index should be enabled.  Default looks at the
 69     isEnabled state of the collection view.
 70     looks at the selection property on the view.
 71     
 72     @param {SC.CollectionView} view the collection view
 73     @param {SC.Array} content the content object
 74     @param {Number} idx the content index
 75     @returns {Boolean} YES, NO, or SC.MIXED_STATE
 76   */
 77   contentIndexIsEnabled: function(view, content, idx) {
 78     return view.get('isEnabled');
 79   },
 80   
 81   // ..........................................................
 82   // GROUPING
 83   // 
 84   
 85   /**
 86     Optionally return an index set containing the indexes that may be group
 87     views.  For each group view, the delegate will actually be asked to 
 88     confirm the view is a group using the contentIndexIsGroup() method.
 89     
 90     If grouping is not enabled, return null.
 91     
 92     @param {SC.CollectionView} view the calling view
 93     @param {SC.Array} content the content object
 94     @return {SC.IndexSet} 
 95   */
 96   contentGroupIndexes: function(view, content) {
 97     return null;
 98   },
 99   
100   /**
101     Returns YES if the item at the specified content index should be rendered
102     using the groupExampleView instead of the regular exampleView.  Note that
103     a group view is different from a branch/leaf view.  Group views often 
104     appear with different layout and a different look and feel.
105 
106     Default always returns NO.
107     
108     @param {SC.CollectionView} view the collection view
109     @param {SC.Array} content the content object
110     @param {Number} idx the content index
111     @returns {Boolean} YES, NO, or SC.MIXED_STATE
112   */
113   contentIndexIsGroup: function(view, content, idx) {
114     return NO ;
115   },
116   
117   // ..........................................................
118   // OUTLINE VIEWS
119   // 
120   
121   /**
122     Returns the outline level for the item at the specified index.  Can be 
123     used to display hierarchical lists.
124     
125     Default always returns -1 (no outline).
126     
127     @param {SC.CollectionView} view the collection view
128     @param {SC.Array} content the content object
129     @param {Number} idx the content index
130     @returns {Boolean} YES, NO, or SC.MIXED_STATE
131   */
132   contentIndexOutlineLevel: function(view, content, idx) {
133     return -1;
134   },
135   
136   /**
137     Returns a constant indicating the disclosure state of the item.  Must be
138     one of SC.BRANCH_OPEN, SC.BRANCH_CLOSED, SC.LEAF_NODE.  If you return one
139     of the BRANCH options then the item may be rendered with a disclosure 
140     triangle open or closed.  If you return SC.LEAF_NODe then the item will 
141     be rendered as a leaf node.  
142 
143     Default returns SC.LEAF_NODE.
144     
145     @param {SC.CollectionView} view the collection view
146     @param {SC.Array} content the content object
147     @param {Number} idx the content index
148     @returns {Boolean} YES, NO, or SC.MIXED_STATE
149   */
150   contentIndexDisclosureState: function(view, content, idx) {
151     return SC.LEAF_NODE;    
152   },
153   
154   /**
155     Called to expand a content index item if it is currently in a closed 
156     disclosure state.  The default implementation does nothing.
157     
158     @param {SC.CollectionView} view the collection view
159     @param {SC.Array} content the content object
160     @param {Number} idx the content index
161     @returns {void}
162   */
163   contentIndexExpand: function(view, content, idx) {
164     SC.Logger.log('contentIndexExpand(%@, %@, %@)'.fmt(view,content,idx));
165   },
166   
167   /**
168     Called to collapse a content index item if it is currently in an open 
169     disclosure state.  The default implementation does nothing.  
170     
171     @param {SC.CollectionView} view the collection view
172     @param {SC.Array} content the content object
173     @param {Number} idx the content index
174     @returns {void}
175   */
176   contentIndexCollapse: function(view, content, idx) {
177     SC.Logger.log('contentIndexCollapse(%@, %@, %@)'.fmt(view,content,idx));
178   }
179     
180 };
