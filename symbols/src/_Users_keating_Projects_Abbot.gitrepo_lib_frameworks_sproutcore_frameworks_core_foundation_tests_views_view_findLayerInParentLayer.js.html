 12 var view, parentDom, childDom, layerId ;
 13 module("SC.View#findLayerInParentLayer", {
 14   setup: function() {
 15     
 16     layerId = 'foo-123';
 17     
 18     // manually construct a test layer.  next childDom a few layers deep
 19     childDom = document.createElement('div');
 20     SC.$(childDom).attr('id', layerId);
 21     
 22     var intermediate = document.createElement('div');
 23     intermediate.appendChild(childDom);
 24     
 25     parentDom = document.createElement('div');
 26     parentDom.appendChild(intermediate);
 27     intermediate = null;
 28     
 29     
 30     // setup view w/ layerId
 31     view = SC.View.create({ layerId: layerId });
 32   },
 33   
 34   teardown: function() {
 35     view = parentDom = childDom = layerId = null;
 36   }
 37 });
 38 
 39 test("discovers layer by finding element with matching layerId - when DOM is in document already", function() {
 40   document.body.appendChild(parentDom);
 41   equals(view.findLayerInParentLayer(parentDom), childDom, 'should find childDom');
 42   document.body.removeChild(parentDom); // cleanup or else next test may fail
 43 });
 44 
 45 test("discovers layer by finding element with matching layerId - when parent DOM is NOT in document", function() {
 46   if(parentDom.parentNode) equals(parentDom.parentNode.nodeType, 11, 'precond - NOT in parent doc');
 47   else equals(parentDom.parentNode, null, 'precond - NOT in parent doc');
 48   equals(view.findLayerInParentLayer(parentDom), childDom, 'found childDom');
 49 });
