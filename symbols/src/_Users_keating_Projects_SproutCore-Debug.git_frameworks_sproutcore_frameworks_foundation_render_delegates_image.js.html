  8 sc_require('render_delegates/render_delegate');
  9 
 10 /**
 11   @class
 12   Renders and updates DOM representations of an image.
 13 
 14   Parameters
 15  16   --------------------------
 17   Expects these properties on the data source:
 18 
 19   - image: An Image object which has completed loading
 20 
 21   If any of these are not present in the data source, the render delegate
 22   will throw an error.
 23 
 24   Optional Parameters:
 25   ---------------------------
 26   If present, these properties will be used.
 27 
 28   - imageValue: A String which represents the src or CSS class of the image
 29   - displayToolTip: A String which is rendered as a toolTip on the element
 30   - type: The type of image being rendered. One of:
 31               - SC.IMAGE_TYPE_NONE
 32               - SC.IMAGE_TYPE_URL
 33               - SC.IMAGE_TYPE_CSS_CLASS
 34           If not provided, SC.IMAGE_TYPE_URL is the default
 35 */
 36 
 37 SC.BaseTheme.imageRenderDelegate = SC.RenderDelegate.create({
 38   className: 'image',
 39 
 40   render: function(dataSource, context) {
 41     var image = dataSource.get('image'),
 42         imageValue = dataSource.get('imageValue'),
 43         type = dataSource.get('type') || SC.IMAGE_TYPE_URL,
 44         toolTip = dataSource.get('toolTip');
 45 
 46     // Place the img within a div, so that we may scale & offset the img
 47     context = context.begin('img');
 48  49     context.attr('src', image.src);
 50 
 51     if (imageValue && type === SC.IMAGE_TYPE_CSS_CLASS) {
 52       context.addClass(imageValue);
 53       this._last_class = imageValue;
 54     }
 55 
 56     if (toolTip) {
 57       context.attr('title', toolTip);
 58       context.attr('alt', toolTip);
 59     }
 60 
 61     // Adjust the layout of the img
 62     context.addStyle(this.imageStyles(dataSource));
 63 
 64     context = context.end();
 65   },
 66 
 67   update: function(dataSource, jquery) {
 68     var image = dataSource.get('image'),
 69         imageValue = dataSource.get('imageValue'),
 70         toolTip = dataSource.get('toolTip');
 71 
 72     jquery = jquery.find('img');
 73     jquery.attr('src', image.src);
 74 
 75     if (imageValue !== this._last_class) jquery.setClass(this._last_class, NO);
 76     if (imageValue) {
 77       jquery.addClass(imageValue);
 78       this._last_class = imageValue;
 79     }
 80 
 81     if (toolTip) {
 82       jquery.attr('title', toolTip);
 83       jquery.attr('alt', toolTip);
 84     }
 85 
 86     // Adjust the layout of the img
 87     jquery.css(this.imageStyles(dataSource));
 88   },
 89 
 90   imageStyles: function(dataSource) {
 91     var innerFrame = dataSource.get('innerFrame');
 92     return {
 93       'position': 'absolute',
 94       'left': Math.round(innerFrame.x),
 95       'top': Math.round(innerFrame.y),
 96       'width': Math.round(innerFrame.width),
 97       'height': Math.round(innerFrame.height)
 98     };
 99   }
100 
101 });
