 24 module("Test SC.SelectionSupport mixin with TreeController.", {
 25   setup: function() {
 26   var fruit = "Apples Bananas Cherries Dates Eggfruit".w().map(function(name) {
 27     return TestObject.create({
 28       displayName: name
 29     });
 30   });
 31     var fruitGroup = SC.Object.create(SC.TreeItemContent, {
 32       displayName: 'Fruit',
 33       treeItemChildren: fruit
 34     });
 35     var vegetables = "Arugula Beets Cucumbers Dandelions Endives".w().map(function(name) {
 36       return TestObject.create({
 37         displayName: name
 38       });
 39     });
 40       var vegetableGroup = SC.Object.create(SC.TreeItemContent, {
 41         displayName: 'Vegetables',
 42         treeItemChildren: vegetables
 43       });
 44       
 45     content = SC.Object.create(SC.TreeItemContent, {
 46       treeItemIsGrouped: YES,
 47       treeItemIsExpanded: YES,
 48       treeItemChildren: [fruitGroup, vegetableGroup]
 49     });
 50     
 51     controller = SC.TreeController.create({
 52       treeItemIsGrouped: NO,
 53       content: content
 54     });
 55   },
 56 
 57   teardown: function() {
 58   }
 59 });

 86 /**
 87   Make sure that the empty selection property is honoured.
 88 */
 89 test("SC.TreeController(SC.SelectionSupport) selection is empty only if allowed",
 90 function() {
 91   var selectionSet, source = controller.get('arrangedObjects'),
 92   indexSet;
 93 
 94   selectionSet = controller.get('selection');
 95   indexSet = selectionSet.indexSetForSource(source);
 96   ok(!controller.get('hasSelection'), 'tree controller should not have a selection');
 97   ok(indexSet === null, 'selection set should not have an indexSet');
 98 
 99   // Disable allowing empty selection
100   controller.set('allowsEmptySelection', NO);
101 
102   selectionSet = controller.get('selection');
103   indexSet = selectionSet.indexSetForSource(source);
104   ok(controller.get('hasSelection'), 'tree controller should have a selection');
105   ok(indexSet !== null, 'selection set should have an indexSet');
106   equals(selectionSet.get('length'), 1, 'selection set should have length 1');
107   equals(indexSet.length, 1, 'selection set should have an indexSet with length 1');
108   ok(selectionSet.containsObject(content.treeItemChildren[0].treeItemChildren[0]), 'selection should be the first content object');
109 });

111 /**
112   Make sure that the multiple selection property is honoured.
113 */
114 test("SC.TreeController(SC.SelectionSupport) selection is multiple only if allowed",
115 function() {
116   var selectionSet, source = controller.get('arrangedObjects'),
117   indexSet;
118 
119   // Select 3 items
120   controller.selectObjects([content.treeItemChildren[0].treeItemChildren[0], content.treeItemChildren[0].treeItemChildren[1], content.treeItemChildren[1].treeItemChildren[4]], NO);
121 
122   selectionSet = controller.get('selection');
123   indexSet = selectionSet.indexSetForSource(source);
124   ok(controller.get('hasSelection'), 'tree controller should have a selection');
125   ok(indexSet !== null, 'selection set should have an indexSet');
126   equals(selectionSet.get('length'), 3, 'selection set should have length 3');
127   equals(indexSet.length, 3, 'selection set should have an indexSet with length 3');
128   ok(selectionSet.containsObject(content.treeItemChildren[0].treeItemChildren[0]), 'selection should contain the first content object');
129   ok(selectionSet.containsObject(content.treeItemChildren[0].treeItemChildren[1]), 'selection should contain the third content object');
130   ok(selectionSet.containsObject(content.treeItemChildren[1].treeItemChildren[4]), 'selection should contain the fifth content object');
131 
132   // Disable allowing multiple selection
133   controller.set('allowsMultipleSelection', NO);
134 
135   selectionSet = controller.get('selection');
136   indexSet = selectionSet.indexSetForSource(source);
137   ok(controller.get('hasSelection'), 'tree controller should still have a selection');
138   ok(indexSet !== null, 'selection set should still have an indexSet');
139   equals(selectionSet.get('length'), 1, 'selection set should have length 1');
140   equals(indexSet.length, 1, 'selection set should have an indexSet with length 1');
141   ok(selectionSet.containsObject(content.treeItemChildren[1].treeItemChildren[4]), 'selection should be the fifth content object');
142 });
