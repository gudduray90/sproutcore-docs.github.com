 11 var pane;
 12 
 13 module("SC.Border", {
 14   setup: function() {
 15     var boxstyles = '<style>.box1-border { border-top: 10px solid red; border-left: 25px solid red; border-right: 109px solid green; border-bottom: 79px solid red; }'+
 16                     '.box2-border { border-top: 10px solid red; border-left: 25px solid red; border-right: 109px solid green; border-bottom: 79px solid red; }'+
 17                     '.box3-border { border-top: 10px solid red; border-left: 25px solid red; border-right: 109px solid green; border-bottom: 79px solid red; }</style>';
 18     htmlbody(boxstyles);
 19 
 20     SC.RunLoop.begin();
 21     pane = SC.MainPane.create();
 22     pane.append();
 23     SC.RunLoop.end();
 24   },
 25 
 26   teardown: function() {
 27     pane.remove();
 28     pane = null;
 29     clearHtmlbody();
 30   }
 31 });
 32 
 33 test('SC.View should not have border properties by default', function() {
 34   var view = SC.View.create();
 35   ok(!view.get('hasBorder'), 'view should not have hasBorder set');
 36   ok(view.get('borderStyle') === undefined, 'view should not have borderStyle set');
 37 });
 38 
 39 test('Views with SC.Border mixin should default to gray border', function() {
 40   var view = SC.View.create(SC.Border, { });
 41 
 42   equals(view.get('borderStyle'), SC.BORDER_GRAY, "borderStyle should be SC.BORDER_GRAY");
 43   equals(view.get('borderTop'), 1, 'borderTop should be 1');
 44   equals(view.get('borderBottom'), 1, 'borderBottom should be 1');
 45   equals(view.get('borderRight'), 1, 'borderRight should be 1');
 46   equals(view.get('borderLeft'), 1, 'borderLeft should be 1');
 47 });
 48 
 49 test('Views that specify a non-built-in border should have border dimensions set to 0', function() {
 50   var view = SC.View.create(SC.Border, { borderStyle: 'fat-and-chunky' });
 51   equals(view.get('borderTop'), 0, 'borderTop should be 0');
 52   equals(view.get('borderBottom'), 0, 'borderBottom should be 0');
 53   equals(view.get('borderRight'), 0, 'borderRight should be 0');
 54   equals(view.get('borderLeft'), 0, 'borderLeft should be 0');
 55 });
 56 
 57 test('Views that specify a string that ends in "-border" should have that string added as a class', function() {
 58   var view = SC.View.create(SC.Border, { borderStyle: 'fat-and-chunky-border' });
 59   pane.appendChild(view);
 60   SC.RunLoop.begin().end();
 61   ok(view.$().hasClass('fat-and-chunky-border'), 'should have fat-and-chunky-border class');
 62 });
 63 
 64 test('Views that specify a string that does not end in "-border" should not have that string added as a class', function() {
 65   var view = SC.View.create(SC.Border, { borderStyle: 'border-at-beginning' });
 66   pane.appendChild(view);
 67   SC.RunLoop.begin().end();
 68   ok(!view.$().hasClass('border-at-beginning'), 'should not have border-at-beginning class');
 69 });
 70 
 71 test('Views should rerender if borderStyle changes', function() {
 72   var view = SC.View.create(SC.Border, { borderStyle: 'first-border' });
 73   pane.appendChild(view);
 74   SC.RunLoop.begin().end();
 75   ok(view.$().hasClass('first-border'), 'precond - view should have first-border class');
 76   view.set('borderStyle', 'second-border');
 77   SC.RunLoop.begin().end();
 78   ok(view.$().hasClass('second-border'), 'view should have second-border class');
 79 });
 80 
 81 test('View frame should account for borders if border dimension properties are set', function() {
 82   var box1 = SC.View.create(SC.Border, { borderStyle: 'box1-border', borderTop: 10, borderLeft: 25, borderRight: 109, borderBottom: 79 }),
 83       box2 = SC.View.create(SC.Border, { borderStyle: 'box2-border', borderTop: 10, borderLeft: 25, borderRight: 109, borderBottom: 79 }),
 84       box3 = SC.View.create(SC.Border, { borderStyle: 'box3-border', borderTop: 10, borderLeft: 25, borderRight: 109, borderBottom: 79 });
 85 
 86   box1.appendChild(box2);
 87   box2.appendChild(box3);
 88   pane.appendChild(box1);
 89   SC.RunLoop.begin().end();
 90   var frame = box3.get('frame'),
 91       elem = box3.get('layer');
 92 
 93   equals(frame.x, elem.clientLeft, 'horizontal positions equal');
 94   equals(frame.y, elem.clientTop, 'vertical positions equal');
 95   equals(frame.width, elem.clientWidth, 'width equal');
 96   equals(frame.height, elem.clientHeight, 'height equal');
 97 });
