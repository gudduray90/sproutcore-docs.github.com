 15 Greenhouse.mixin( /** @scope Greenhouse */{
 16 
 17   inspectorStates: SC.State.design({
 18     initialSubstate: 'inspectorClosed',
 19 
 20     inspectorClosed: SC.State.design({
 21 
 22       parallelStatechart: 'inspector',
 23 
 24       // ..........................................................
 25       // Events
 26       //
 27       openInspector: function(anchor){
 28         if(anchor) Greenhouse.setPath('inspectorStates.openInspectorPicker.anchor', anchor);
 29         this.gotoState('openInspectorPicker');
 30       },
 31 
 32       toggleDockedInspector: function(){
 33         this.gotoState('dockedInspector');
 34       },
 35 
 36       floatInspector: function(){
 37         this.gotoState('inspectorPalette');
 38       }
 39     }),
 40 
 41     openInspectorPicker: SC.State.design({
 42       parallelStatechart: 'inspector',
 43 
 44       enterState: function(){
 45         var ap = Greenhouse.appPage;
 46         var picker = ap.get('inspectorPicker'),
 47             pickerContentView = ap.get('inspectorPickerContentView');
 48         var anchor = this.get('anchor') || ap.getPath('mainView.toolBar.inspector');
 49 
 50         pickerContentView.setIfChanged('nowShowing', 'Greenhouse.appPage.inspectorContentView');
 51         picker.popup(anchor, SC.PICKER_POINTER);
 52         picker.becomeFirstResponder();
 53       },
 54       exitState: function(){
 55         var ap = Greenhouse.appPage;
 56         var picker = ap.get('inspectorPicker'),
 57             pickerContentView = ap.get('inspectorPickerContentView');
 58         pickerContentView.setIfChanged('nowShowing', null);
 59         picker.remove();
 60         this.set('anchor', null);
 61       },
 62 
 63       // ..........................................................
 64       // Events
 65       //
 66       cancel: function(){
 67         this.gotoState('inspectorClosed');
 68       },
 69 
 70       floatInspector: function(){
 71         this.gotoState('inspectorPalette');
 72       },
 73 
 74       toggleDockedInspector: function(){
 75         this.gotoState('dockedInspector');
 76       }
 77     }),
 78 
 79     inspectorPalette: SC.State.design({
 80       parallelStatechart: 'inspector',
 81 
 82       enterState: function(){
 83         var ap = Greenhouse.appPage;
 84         var picker = ap.get('inspectorPicker'),
 85             pickerContentView = ap.get('inspectorPickerContentView');
 86 
 87         pickerContentView.setIfChanged('nowShowing', 'Greenhouse.appPage.inspectorContentView');
 88         picker.append();
 89         picker.set('isModal', NO);
 90         picker.set('isAnchored', NO);
 91         picker.$().toggleClass('sc-picker', NO);
 92         var content = ap.getPath('inspectorContentView.content'),
 93             toolbar = ap.getPath('inspectorContentView.toolbar');
 94 
 95         content.adjust('top', 28);
 96         toolbar.set('isVisible', YES);
 97       },
 98       exitState: function(){
 99         var ap = Greenhouse.appPage;
100         var picker = ap.get('inspectorPicker'),
101             pickerContentView = ap.get('inspectorPickerContentView');
102 
103         pickerContentView.setIfChanged('nowShowing', null);
104         picker.set('isModal', YES);
105         picker.set('isAnchored', YES);
106         picker.remove();
107 
108         var content = ap.getPath('inspectorContentView.content'),
109             toolbar = ap.getPath('inspectorContentView.toolbar');
110 
111         content.adjust('top', 0);
112         toolbar.set('isVisible', NO);
113       },
114 
115       // ..........................................................
116       // Events
117       //
118       closeInspector: function(){
119         this.gotoState('inspectorClosed');
120       },
121 
122       toggleDockedInspector: function(){
123         this.gotoState('dockedInspector');
124       }
125     }),
126 
127     dockedInspector: SC.State.design({
128       parallelStatechart: 'inspector',
129 
130       enterState: function(){
131         var iDock = Greenhouse.appPage.get('inspectorDockView');
132         iDock.setIfChanged('nowShowing', 'Greenhouse.appPage.inspectorContentView');
133       },
134       exitState: function(){
135         var iDock = Greenhouse.appPage.get('inspectorDockView');
136         iDock.setIfChanged('nowShowing', null);
137       },
138 
139       // ..........................................................
140       // Events
141       //
142       toggleDockedInspector: function(){
143         var states = Greenhouse.get('currentStates') || [];
144         if (states.indexOf(Greenhouse.getState('dockedLibrary'))) Greenhouse.sendEvent('undock');
145         this.gotoState('inspectorClosed');
146       }
147     })
148   })
149 
150 
151 
152 });
