  9 /**
 10   @class
 11 
 12   Displays a progress bar.  You can display both a defined and an
 13   indeterminate progressbar.  The progress bar itself is designed to be styled
 14   using CSS classes with the following structure:
 15 
 16       <div class="sc-progress-view"><div class="inner"></div></div>
 17 
 18   The outer can form the boundary of the bar while the inner will be adjusted
 19   to fit the percentage of the progress.
 20 
 21   Creating a ProgressView accepts a number of properties, for example:
 22 
 23       progressView: SC.ProgressView.design({
 24         value: 50,
 25         minimum: 0,
 26         maximum: 100,
 27         isIndeterminate: NO,
 28         isEnabled: YES
 29       })
 30 
 31   @extends SC.View
 32   @extends SC.Control
 33   @since SproutCore 1.0
 34 */
 35 SC.ProgressView = SC.View.extend(SC.Control,
 36 /** @scope SC.ProgressView.prototype */{
 37 
 38   /**
 39     @type Array
 40     @default ['sc-progress-view']
 41     @see SC.View#classNames
 42   */
 43   classNames: ['sc-progress-view'],
 44 
 45   /**
 46     @type Array
 47     @default ['displayValue', 'ariaValue', 'minimum', 'maximum', 'isEnabled', 'isIndeterminate', 'isRunning', 'isVisibleInWindow']
 48     @see SC.View#displayProperties
 49   */
 50   displayProperties: ['displayValue', 'ariaValue', 'minimum', 'maximum', 'isEnabled', 'isIndeterminate', 'isRunning', 'isVisibleInWindow'],
 51 
 52   /**
 53     @type String
 54     @default 'progressRenderDelegate'
 55   */
 56   renderDelegateName: 'progressRenderDelegate',
 57 
 58   // ........................................
 59   // PROPERTIES
 60   //
 61 
 62   /**
 63     Bind this to the current value of the progress bar.  Note that by default
 64     an empty value will disable the progress bar and a multiple value will make
 65     it indeterminate.
 66 
 67     @type Number
 68     @default 0.50
 69   */
 70   value: 0.50,
 71 
 72   /** @private */
 73   valueBindingDefault: SC.Binding.single().notEmpty(),
 74 
 75   /**
 76     @field
 77     @type Number
 78     @observes value
 79     @observes maximum
 80     @observes minimum
 81   */
 82   displayValue: function(){
 83     var minimum = this.get('minimum') || 0.0,
 84         maximum = this.get('maximum') || 1.0,
 85         value = this.get('value') || 0.0;
 86     value = (value - minimum) / (maximum - minimum);
 87     if (value > 1.0) value = 1.0;
 88 
 89     if(isNaN(value)) value = 0.0;
 90     // cannot be smaller then minimum
 91     if(value<minimum) value = 0.0;
 92     // cannot be larger then maximum
 93     if(value>maximum) value = 1.0;
 94 
 95     return value;
 96   }.property('value', 'maximum', 'minimum').cacheable(),
 97 
 98   /**
 99     The WAI-ARIA role for progress view.
100 
101     @type String
102     @default 'progressbar'
103     @readOnly
104   */
105   ariaRole: 'progressbar',
106 
107   /**
108     The WAI-ARIA value for the progress view. This value will be passed to any
109     rendering code as-is, not converted into percentages, etc. It is computed
110     based on the original value property.
111 
112     @property
113   */
114   ariaValue: function() {
115     return this.get('value');
116   }.property('value').cacheable(),
117 
118   /**
119     The minimum value of the progress.
120 
121     @type Number
122     @default 0
123   */
124   minimum: 0,
125 
126   /** @private */
127   minimumBindingDefault: SC.Binding.single().notEmpty(),
128 
129   /**
130     Optionally specify the key used to extract the minimum progress value
131     from the content object.  If this is set to null then the minimum value
132     will not be derived from the content object.
133 
134     @type String
135     @default null
136   */
137   contentMinimumKey: null,
138 
139   /**
140     The maximum value of the progress bar.
141 
142     @type Number
143     @default 1.0
144   */
145   maximum: 1.0,
146 
147   /** @private */
148   maximumBindingDefault: SC.Binding.single().notEmpty(),
149 
150   /**
151     Optionally specify the key used to extract the maximum progress value
152     from the content object.  If this is set to null then the maximum value
153     will not be derived from the content object.
154 
155     @type String
156     @default null
157   */
158   contentMaximumKey: null,
159 
160   /**
161     Set to true if the item in progress is indeterminate.  This may be
162     overridden by the actual value.
163 
164     @type Boolean
165     @default NO
166   */
167   isIndeterminate: NO,
168 
169   /** @private */
170   isIndeterminateBindingDefault: SC.Binding.bool(),
171 
172   /**
173     Set to YES when the process is currently running.  This will cause the
174     progress bar to animate, especially if it is indeterminate.
175 
176     @type Boolean
177     @default NO
178   */
179   isRunning: NO,
180 
181   /** @private */
182   isRunningBindingDefault: SC.Binding.bool(),
183 
184   /**
185     Optionally specify the key used to extract the isIndeterminate value
186     from the content object.  If this is set to null then the isIndeterminate
187     value will not be derived from the content object.
188 
189     @type String
190     @default null
191   */
192   contentIsIndeterminateKey: null,
193 
194   // ........................................
195   // INTERNAL SUPPORT
196   //
197 
198   /** @private */
199   contentPropertyDidChange: function(target, key) {
200     var content = this.get('content');
201     this.beginPropertyChanges()
202       .updatePropertyFromContent('value', key, 'contentValueKey', content)
203       .updatePropertyFromContent('minimum', key, 'contentMinimumKey', content)
204       .updatePropertyFromContent('maximum', key, 'contentMaximumKey', content)
205       .updatePropertyFromContent('isIndeterminate', key, 'contentIsIndeterminateKey', content)
206     .endPropertyChanges();
207   },
208 
209   /** @private */
210   didCreateLayer: function() {
211     // When using the JavaScript animation, we cannot start until the layer is
212     // created.  Then if we are indeterminate, running and visible in the
213     // window already, start animating.
214     if (this.get('isIndeterminate') && this.get('isRunning') && this.get('isVisibleInWindow')) {
215       this.displayDidChange();
216     }
217   }
218 
219 });
