  8 /** @class
  9   
 10   The default example view for a table row. Belongs to a SC.TableView.
 11 
 12   @extends SC.View
 13   @since SproutCore 1.1
 14 */
 15 
 16 SC.TableRowView = SC.View.extend({
 17 /** @scope SC.TableRowView.prototype */ 
 18 
 19   //layout: { height: 18, left: 0, right: 0, top: 0 },
 20   
 21   // ..........................................................
 22   // PROPERTIES
 23   //
 24   
 25   classNames: ['sc-table-row'],
 26   
 27   cells: [],
 28 
 29   acceptsFirstResponder: YES,
 30   
 31   /**
 32     A reference to the row's encompassing TableView.
 33     
 34     @property
 35     @type SC.TableView
 36   */
 37   tableView: null,
 38   
 39   // ..........................................................
 40   // METHODS
 41   // 
 42   
 43   init: function() {
 44     sc_super();
 45     this._sctrv_handleChildren();
 46   },
 47   
 48   /**
 49     A collection of `SC.TableColumn` objects.
 50     
 51     @property
 52     @type Array
 53   */
 54   columns: function() {
 55     return this.get('tableView').get('columns');
 56   }.property(),
 57   
 58   prepareContext: function(context, firstTime) {
 59     sc_super();
 60     context.setClass('sel', this.get('isSelected'));
 61   },
 62   
 63   render: function(context, firstTime) {
 64     var classArray = [];
 65     
 66     classArray.push((this.get('contentIndex')%2 === 0) ? 'even' : 'odd');
 67     context.addClass(classArray);
 68     
 69     sc_super();
 70   },
 71   
 72   renderChildViews: function(context, firstTime) {
 73     var cells = this.get('cells'), cell, idx;
 74     for (idx = 0; idx < cells.get('length'); idx++) {
 75       cell = cells.objectAt(idx);
 76       context = context.begin(cell.get('tagName'));
 77       cell.prepareContext(context, firstTime);
 78       context = context.end();
 79     }
 80     return context;
 81   },
 82   
 83   layoutChildViews: function() {
 84     var cells = this.get('cells'), columns = this.get('columns'),
 85         cell, column, idx;
 86     var left = 0, width, rowHeight = this.get('tableView').get('rowHeight');
 87     
 88     for (idx = 0; idx < cells.get('length'); idx++) {
 89       cell = cells.objectAt(idx);
 90       column = columns.objectAt(idx);
 91       width = column.get('width');
 92       
 93       cell.adjust({
 94         left: left,
 95         width: width,
 96         height: rowHeight
 97       });
 98       
 99       left += width;
100       cell.updateLayout();
101     }
102   },
103   
104   // ..........................................................
105   // INTERNAL SUPPORT
106   // 
107   
108   _sctrv_layoutForChildAtColumnIndex: function(index) {
109     var columns = this.get('columns'),
110         rowHeight = this.get('tableView').get('rowHeight'),
111         layout = {},
112         left = 0, idx;
113         
114     for (idx = 0; idx < index; idx++) {
115       left += columns.objectAt(idx).get('width');
116     }
117     
118     return {
119       left:   left,
120       width:  columns.objectAt(index).get('width'),
121       height: rowHeight
122     };
123   },  
124   
125   _sctrv_createTableCell: function(column, value) {
126     var cell = SC.TableCellView.create({
127       column:  column,
128       content: value
129     });
130     return cell;
131   },
132   
133   // The row needs to redraw when the selection state changes.
134   _sctrv_handleSelection: function() {
135     this.displayDidChange();
136   }.observes('isSelected'),
137   
138   _sctrv_handleChildren: function() {
139     var content = this.get('content'), columns = this.get('columns');
140     
141     this.removeAllChildren();
142     var column, key, value, cells = [], cell, idx;
143     for (idx = 0; idx < columns.get('length'); idx++) {
144       column = columns.objectAt(idx);
145       key    = column.get('key');
146       value  = content ? content.getPath(key) : "";
147       cell   = this._sctrv_createTableCell(column, value);
148       cells.push(cell);
149       this.appendChild(cell);
150     }
151 152     
153     this.set('cells', cells);
154   }
155 });
