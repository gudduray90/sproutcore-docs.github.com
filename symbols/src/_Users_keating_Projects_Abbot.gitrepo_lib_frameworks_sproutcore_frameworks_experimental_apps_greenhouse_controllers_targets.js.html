  5 /*globals Greenhouse */

  7 /** @class
  8 
  9   The full set of targets available in the application.  This is populated 
 10   automatically when you call loadTargets().
 11   
 12   
 13   This Class comes from SproutCore's test runner
 14   
 15   
 16   @extends SC.ArrayController
 17 */
 18 
 19 Greenhouse.targetsController = SC.ArrayController.create(
 20 /** @scope Greenhouse.targetsController.prototype */ {
 21 
 22   /**
 23     Call this method whenever you want to relaod the targets from the server.
 24   */
 25   reload: function() {
 26     var targets = Greenhouse.store.find(Greenhouse.TARGETS_QUERY);
 27     this.set('content', targets);
 28   },
 29   
 30   /** 
 31     Generates the Array of Apps in this project
 32   */
 33   applications: function() {
 34 
 35     var apps = [];
 36     this.forEach(function(target) { 
 37       if(target.get('sortKind') === "app") apps.pushObject(target);
 38     }, this);
 39 
 40     return apps;
 41     
 42   }.property('[]').cacheable()
 43 }) ;
