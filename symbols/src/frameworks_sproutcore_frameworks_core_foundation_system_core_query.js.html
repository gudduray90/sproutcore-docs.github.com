 10 // Alias jQuery as SC.$ and SC.CoreQuery for compatibility
 11 SC.$ = SC.CoreQuery = jQuery;

 13 // Add some plugins to SC.$. jQuery will get these also. -- test in system/core_query/additions
 14 SC.mixin(SC.$.fn,
 15   /** @scope SC.$.prototype */ {
 16 
 17   isCoreQuery: YES, // walk like a duck
 18 
 19   /** @private - better loggin */
 20   toString: function () {
 21     var values = [],
 22       len = this.length,
 23       idx = 0;
 24 
 25     for (idx = 0; idx < len; idx++) {
 26       values[idx] = '%@: %@'.fmt(idx, this[idx] ? this[idx].toString() : '(null)');
 27     }
 28     return "<$:%@>(%@)".fmt(SC.guidFor(this), values.join(' , '));
 29   },
 30 
 31   /**
 32     Returns YES if all member elements are visible.  This is provided as a
 33     common test since CoreQuery does not support filtering by
 34     psuedo-selector.
 35 
 36     @deprecated Version 1.10
 37   */
 38   isVisible: function () {
 39     return Array.prototype.every.call(this, function (elem) {
 40       return SC.$.isVisible(elem);
 41     });
 42   },
 43 
 44   /**
 45     Attempts to find the views managing the passed DOM elements and returns
 46     them.   This will start with the matched element and walk up the DOM until
 47     it finds an element managed by a view.
 48 
 49     @returns {Array} array of views or null.
 50   */
 51   view: function () {
 52     return this.map(function () {
 53       var ret = null,
 54         dom = this,
 55         value;
 56 
 57       while (!ret && dom && (dom !== document)) {
 58         if (dom.nodeType === 1 && (value = dom.getAttribute('id'))) {
 59           ret = SC.View.views[value];
 60         }
 61         dom = dom.parentNode;
 62       }
 63       dom = null;
 64 
 65       return ret;
 66     });
 67   },
 68 
 69   /**
 70     Returns YES if any of the matched elements have the passed element or CQ object as a child element.
 71   */
 72   within: function (el) {
 73     if (this.filter(el).length) { return true; }
 74     return !!this.has(el).length;
 75   }
 76 
 77 });

 79 /**
 80   Make CoreQuery enumerable.  Since some methods need to be disambiguated,
 81   we will implement some wrapper functions here.
 82 
 83   Note that SC.Enumerable is implemented on SC.Builder, which means the
 84   CoreQuery object inherits this automatically.  jQuery does not extend from
 85   SC.Builder though, so we reapply SC.Enumerable just to be safe.
 86 */
 87 (function () {
 88   var original = {},
 89       wrappers = {
 90 
 91     // if you call find with a selector, then use the jQuery way.  If you
 92     // call with a function/target, use Enumerable way
 93     find: function (callback, target) {
 94       return (target !== undefined) ? SC.Enumerable.find.call(this, callback, target) : original.find.call(this, callback);
 95     },
 96 
 97     // ditto for filter - execute SC.Enumerable style if a target is passed.
 98     filter: function (callback, target) {
 99       return (target !== undefined) ?
100         this.pushStack(SC.Enumerable.filter.call(this, callback, target)) :
101         original.filter.call(this, callback);
102     },
103 
104     // filterProperty is an SC.Enumerable thing, but it needs to be wrapped
105     // in a CoreQuery object.
106     filterProperty: function (key, value) {
107       return this.pushStack(
108         SC.Enumerable.filterProperty.call(this, key, value));
109     },
110 
111     // indexOf() is best implemented using the jQuery index()
112     indexOf: SC.$.index,
113 
114     // map() is a little tricky because jQuery is non-standard.  If you pass
115     // a context object, we will treat it like SC.Enumerable.  Otherwise use
116     // jQuery.
117     map: function (callback, target) {
118       return (target !== undefined) ?
119         SC.Enumerable.map.call(this, callback, target) :
120         original.map.call(this, callback);
121     }
122   };
123 
124   // loop through an update some enumerable methods.
125   var fn = SC.$.fn,
126     enumerable = SC.Enumerable,
127     value;
128 
129   for (var key in enumerable) {
130     if (enumerable.hasOwnProperty(key)) {
131       value = enumerable[key];
132 
133       if (key in wrappers) {
134         original[key] = fn[key];
135         value = wrappers[key];
136       }
137 
138       fn[key] = value;
139     }
140   }
141 })();

143 // Add some global helper methods.
144 SC.mixin(SC.$, {
145 
146   /** @private helper method to determine if an element is visible.  Exposed
147    for use in testing.
148 
149     @deprecated Version 1.10
150   */
151   isVisible: function (elem) {
152     //@if(debug)
153     SC.warn("Developer Warning: The isVisible() helper has been deprecated and will be removed in the future.");
154     //@endif
155     var CQ = SC.$;
156     return ("hidden" != elem.type) && (CQ.css(elem, "display") != "none") && (CQ.css(elem, "visibility") != "hidden");
157   }
158 
159 });
