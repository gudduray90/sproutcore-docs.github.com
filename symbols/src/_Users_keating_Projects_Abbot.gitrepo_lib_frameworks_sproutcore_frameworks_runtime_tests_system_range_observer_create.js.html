  9 var source, indexes, observer, obj ; // base array to work with
 10 module("SC.RangeObserver#create", {
 11   setup: function() {
 12 
 13     // create array with 5 SC.Object's in them
 14     source = [1,2,3,4,5].map(function(x) {
 15       return SC.Object.create({ item: x, foo: "bar" }) ;
 16     }, this);
 17 
 18     indexes = SC.IndexSet.create(2,2); // select 2..3
 19 
 20     observer = SC.Object.create({
 21 
 22       callCount: 0,
 23 
 24       rangeDidChange: function() {
 25         this.callCount++;
 26       }
 27 
 28     });
 29 
 30     obj = SC.RangeObserver.create(source, indexes, observer, observer.rangeDidChange, "context", YES);
 31 
 32   }
 33 });
 34 
 35 test("returns new instance", function() {
 36   ok(obj && obj.isRangeObserver, 'returns range observer');
 37 });
 38 
 39 test("sets up observing on properties for each object in range in index if isDeep", function() {
 40   var len = source.length, idx;
 41   for(idx=0;idx<len;idx++) {
 42     source[idx].set('foo', 'baz');
 43   }
 44   equals(observer.callCount, 2, 'range observer should fire twice');
 45 });
 46 
 47 test("does not observe object properties if isDeep is NO", function() {
 48   // remove unneeded observer
 49   obj.destroy();
 50 
 51   // use new observer
 52   obj = SC.RangeObserver.create(source, indexes, observer, observer.rangeDidChange, "context", NO);
 53 
 54   var len = source.length, idx;
 55   for(idx=0;idx<len;idx++) {
 56     source[idx].set('foo', 'baz');
 57   }
 58   equals(observer.callCount, 0, 'range observer should not fire');
 59 });
 60 
 61 test("SC.RangeObserver.create should accept methods specified as strings", function() {
 62   var myArray = [ SC.Object.create({ prop: 0 })],
 63       rangeObserverCount = 0,
 64       observer = SC.Object.create({
 65         rangeObserverDidFire: function(source, object, key, index) {
 66           ++rangeObserverCount;
 67         }
 68       }),
 69       rangeObserver = SC.RangeObserver.create( myArray, SC.IndexSet.create(0, 1),
 70                                                observer, 'rangeObserverDidFire',
 71                                                null, true /* isDeep */);
 72 
 73   equals(rangeObserverCount, 0, "Range observer hasn't fired yet");
 74   myArray[0].incrementProperty('prop');
 75   equals(rangeObserverCount, 1, "Range observer should fire on property change");
 76 });
