 10 // These are basic enhancements to the string class used throughout
 11 // SproutCore.
 12 /** @private */
 13 SC.STRING_TITLEIZE_REGEXP = (/([\s|\-|\_|\n])([^\s|\-|\_|\n]?)/g);
 14 SC.STRING_DECAMELIZE_REGEXP = (/([a-z])([A-Z])/g);
 15 SC.STRING_DASHERIZE_REGEXP = (/[ _]/g);
 16 SC.STRING_DASHERIZE_CACHE = {};
 17 SC.STRING_TRIM_LEFT_REGEXP = (/^\s+/g);
 18 SC.STRING_TRIM_RIGHT_REGEXP = (/\s+$/g);
 19 
 20 /**
 21   @namespace
 22 
 23   SproutCore implements a variety of enhancements to the built-in String
 24   object that make it easy to perform common substitutions and conversions.
 25 
 26   Most of the utility methods defined here mirror those found in Prototype
 27   1.6.
 28 
 29   @since SproutCore 1.0
 30   @lends String.prototype
 31 */
 32 SC.mixin(SC.String, {
 33 
 34   /**
 35     Capitalizes a string.
 36 
 37     ## Examples
 38 
 39         capitalize('my favorite items') // 'My favorite items'
 40         capitalize('css-class-name')    // 'Css-class-name'
 41         capitalize('action_name')       // 'Action_name'
 42         capitalize('innerHTML')         // 'InnerHTML'
 43 
 44     @return {String} capitalized string
 45   */
 46   capitalize: function(str) {
 47     return str.charAt(0).toUpperCase() + str.slice(1);
 48   },
 49 
 50   /**
 51     Camelizes a string.  This will take any words separated by spaces, dashes
 52     or underscores and convert them into camelCase.
 53 
 54     ## Examples
 55 
 56         camelize('my favorite items') // 'myFavoriteItems'
 57         camelize('css-class-name')    // 'cssClassName'
 58         camelize('action_name')       // 'actionName'
 59         camelize('innerHTML')         // 'innerHTML'
 60 
 61     @returns {String} camelized string
 62   */
 63   camelize: function(str) {
 64     var ret = str.replace(SC.STRING_TITLEIZE_REGEXP, function(str, separater, character) {
 65       return character ? character.toUpperCase() : '';
 66     });
 67 
 68     var first = ret.charAt(0),
 69         lower = first.toLowerCase();
 70 
 71     return first !== lower ? lower + ret.slice(1) : ret;
 72   },
 73 
 74   /**
 75     Converts a camelized string into all lower case separated by underscores.
 76 
 77     ## Examples
 78 
 79     decamelize('my favorite items') // 'my favorite items'
 80     decamelize('css-class-name')    // 'css-class-name'
 81     decamelize('action_name')       // 'action_name'
 82     decamelize('innerHTML')         // 'inner_html'
 83 
 84     @returns {String} the decamelized string.
 85   */
 86   decamelize: function(str) {
 87     return str.replace(SC.STRING_DECAMELIZE_REGEXP, '$1_$2').toLowerCase();
 88   },
 89 
 90   /**
 91     Converts a camelized string or a string with spaces or underscores into
 92     a string with components separated by dashes.
 93 
 94     ## Examples
 95 
 96     | *Input String* | *Output String* |
 97     dasherize('my favorite items') // 'my-favorite-items'
 98     dasherize('css-class-name')    // 'css-class-name'
 99     dasherize('action_name')       // 'action-name'
100     dasherize('innerHTML')         // 'inner-html'
101 102 
    @returns {String} the dasherized string.
103   */
104   dasherize: function(str) {
105     var cache = SC.STRING_DASHERIZE_CACHE,
106         ret   = cache[str];
107 
108     if (ret) {
109       return ret;
110     } else {
111       ret = SC.String.decamelize(str).replace(SC.STRING_DASHERIZE_REGEXP,'-');
112       cache[str] = ret;
113     }
114 
115     return ret;
116   },
117 
118   /**
119     Localizes the string.  This will look up the receiver string as a key
120     in the current Strings hash.  If the key matches, the loc'd value will be
121     used.  The resulting string will also be passed through fmt() to insert
122     any variables.
123 
124     @param str {String} String to localize
125     @param args {Object...} optional arguments to interpolate also
126     @returns {String} the localized and formatted string.
127   */
128   loc: function(str) {
129     // NB: This could be implemented as a wrapper to locWithDefault() but
130     // it would add some overhead to deal with the arguments and adds stack
131     // frames, so we are keeping the implementation separate.
132     if (!SC.Locale.currentLocale) { SC.Locale.createCurrentLocale(); }
133 
134     var localized = SC.Locale.currentLocale.locWithDefault(str);
135     if (SC.typeOf(localized) !== SC.T_STRING) { localized = str; }
136 
137     var args = SC.$A(arguments);
138     args.shift(); // remove str param
139     //to extend String.prototype
140     if (args.length > 0 && args[0] && args[0].isSCArray) { args = args[0]; }
141 
142     return SC.String.fmt(localized, args);
143   },
144 
145   /**
146     Returns the localized metric value for the specified key.  A metric is a
147     single value intended to be used in your interface’s layout, such as
148     "Button.Confirm.Width" = 100.
149 
150     If you would like to return a set of metrics for use in a layout hash, you
151     may prefer to use the locLayout() method instead.
152 
153     @param str {String} key
154     @returns {Number} the localized metric
155   */
156   locMetric: function(key) {
157     var K             = SC.Locale,
158         currentLocale = K.currentLocale;
159 
160     if (!currentLocale) {
161       K.createCurrentLocale();
162       currentLocale = K.currentLocale;
163     }
164     return currentLocale.locMetric(key);
165   },
166 
167   /**
168     Creates and returns a new hash suitable for use as an SC.View’s 'layout'
169     hash.  This hash will be created by looking for localized metrics following
170     a pattern based on the “base key” you specify.
171 
172     For example, if you specify "Button.Confirm", the following metrics will be
173     used if they are defined:
174 
175       Button.Confirm.left
176       Button.Confirm.top
177       Button.Confirm.right
178       Button.Confirm.bottom
179       Button.Confirm.width
180       Button.Confirm.height
181       Button.Confirm.midWidth
182       Button.Confirm.minHeight
183       Button.Confirm.centerX
184       Button.Confirm.centerY
185 
186     Additionally, you can optionally specify a hash which will be merged on top
187     of the returned hash.  For example, if you wish to allow a button’s width
188     to be configurable per-locale, but always wish for it to be centered
189     vertically and horizontally, you can call:
190 
191       locLayout("Button.Confirm", {centerX:0, centerY:0})
192 
193     …so that you can combine both localized and non-localized elements in the
194     returned hash.  (An exception will be thrown if there is a locale-specific
195     key that matches a key specific in this hash.)
196 
197 
198     For example, if your locale defines:
199 
200       Button.Confirm.left
201       Button.Confirm.top
202       Button.Confirm.right
203       Button.Confirm.bottom
204 
205 
206     …then these two code snippets will produce the same result:
207 
208       layout: {
209         left:   "Button.Confirm.left".locMetric(),
210         top:    "Button.Confirm.top".locMetric(),
211         right:  "Button.Confirm.right".locMetric(),
212         bottom: "Button.Confirm.bottom".locMetric()
213       }
214 
215       layout: "Button.Confirm".locLayout()
216 
217     The former is slightly more efficient because it doesn’t have to iterate
218     through the possible localized layout keys, but in virtually all situations
219     you will likely wish to use the latter.
220 
221     @param str {String} key
222     @param {str} (optional) additionalHash
223     @param {String} (optional) additionalHash
224     @returns {Number} the localized metric
225   */
226   locLayout: function(key, additionalHash) {
227     var K             = SC.Locale,
228         currentLocale = K.currentLocale;
229 
230     if (!currentLocale) {
231       K.createCurrentLocale();
232       currentLocale = K.currentLocale;
233     }
234     return currentLocale.locLayout(key, additionalHash);
235   },
236 
237   /**
238     Works just like loc() except that it will return the passed default
239     string if a matching key is not found.
240 
241     @param {String} str the string to localize
242     @param {String} def the default to return
243     @param {Object...} args optional formatting arguments
244     @returns {String} localized and formatted string
245   */
246   locWithDefault: function(str, def) {
247     if (!SC.Locale.currentLocale) { SC.Locale.createCurrentLocale(); }
248 
249     var localized = SC.Locale.currentLocale.locWithDefault(str, def);
250     if (SC.typeOf(localized) !== SC.T_STRING) { localized = str; }
251 
252     var args = SC.$A(arguments);
253     args.shift(); // remove str param
254     args.shift(); // remove def param
255 
256     return SC.String.fmt(localized, args);
257   },
258 
259   /**
260    Removes any extra whitespace from the edges of the string. This method is
261    also aliased as strip().
262 
263    @returns {String} the trimmed string
264   */
265   trim: jQuery.trim,
266 
267   /**
268    Removes any extra whitespace from the left edge of the string.
269 
270    @returns {String} the trimmed string
271   */
272   trimLeft: function (str) {
273     return str.replace(SC.STRING_TRIM_LEFT_REGEXP,"");
274   },
275 
276 277   /**
278    Removes any extra whitespace from the right edge of the string.
279 
280    @returns {String} the trimmed string
281   */
282   trimRight: function (str) {
283     return str.replace(SC.STRING_TRIM_RIGHT_REGEXP,"");
284   }
285 });

291   /** @scope String.prototype */ {
292 
293     trim: function() {
294       return SC.String.trim(this, arguments);
295     },
296 
297     trimLeft: function() {
298       return SC.String.trimLeft(this, arguments);
299     },
300 
301     trimRight: function() {
302       return SC.String.trimRight(this, arguments);
303     }
304   });

309 /** @scope String.prototype */ {
310 
311   loc: function() {
312     return SC.String.loc(this.toString(), SC.$A(arguments));
313   },
314 
315   locMetric: function() {
316     return SC.String.locMetric(this.toString());
317   },
318 
319   locLayout: function(additionalHash) {
320     return SC.String.locLayout(this.toString(), additionalHash);
321   }
