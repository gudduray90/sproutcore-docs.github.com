  9 /** @static
 10 
 11   This protocol defines the allowable responder methods. To implement a 
 12   specific responder method, and a method with the correct signature to your
 13   class.
 14   
 15   **DO NOT INCLUDE THIS MIXIN**.
 16   
 17   If you try and include this mixin, an error will be raised on startup.
 18   
 19   @author Erich Ocean
 20   @since SproutCore 1.0
 21 */
 22 SC.ResponderProtocol = {
 23   
 24   // .......................................................................
 25   // Mouse Event Handlers
 26   //
 27   
 28   /**
 29     Called when the mouse is pressed. You must return `YES` to receive
 30     mouseDragged and mouseUp in the future.
 31  32     
 33     @param evt {SC.Event} the mousedown event
 34     @returns {Boolean} YES to receive additional mouse events, NO otherwise
 35   */
 36   mouseDown: function(evt) {},
 37   
 38   /**
 39     Called when the mouse is released.
 40     
 41     @param evt {SC.Event} the mouseup event
 42     @returns {Boolean} YES to handle the mouseUp, NO to allow click() and doubleClick() to be called
 43   */
 44   mouseUp: function(evt) {},
 45   
 46   /**
 47     Called when the mouse is dragged, after responding `YES` to a previous `mouseDown`:
 48     call.
 49     
 50     @param evt {SC.Event} the mousemove event
 51     @returns {void}
 52   */
 53   mouseDragged: function(evt) {},
 54   
 55   /**
 56     Called when the mouse exits the view and the root responder is not in a
 57     drag session.
 58     
 59     @param evt {SC.Event} the mousemove event
 60     @returns {void}
 61   */
 62   mouseExited: function(evt) {},
 63   
 64   /**
 65     Called when the mouse enters the view and the root responder is not in a
 66     drag session.
 67     
 68     @param evt {SC.Event} the mousemove event
 69     @returns {void}
 70   */
 71   mouseEntered: function(evt) {},
 72   
 73   /**
 74     Called when the mouse moves within the view and the root responder is not in a
 75     drag session.
 76     
 77     @param evt {SC.Event} the mousemove event
 78     @returns {void}
 79   */
 80   mouseMoved: function(evt) {},
 81   
 82   
 83   /**
 84     Called when a selectstart event in IE is triggered. **ONLY IE**
 85     We use it to disable IE accelerators and text selection
 86     
 87     @param evt {SC.Event} the selectstart event
 88     @returns {void}
 89   */
 90   selectStart: function(evt) {},
 91   
 92   /**
 93      Called when a contextmenu event is triggered. Used to disable contextmenu
 94      per view.
 95      
 96      @param evt {SC.Event} the selectstart event
 97      @returns {void}
 98    */
 99   contextMenu: function(evt) {},
100   
101   // .......................................................................
102   // Event Handlers
103   //
104   // These methods are called by the input manager in response to keyboard
105   // events.  Most of these methods are defined here for you, but not actually
106   // implemented in code.
107   
108   /**
109     Insert the text or act on the key.
110     
111     @param {String} the text to insert or respond to
112     @returns {Boolean} YES if you handled the method; NO otherwise
113   */
114   insertText: function(text) {},
115   
116   /**
117     When the user presses a key-combination event, this will be called so you
118     can run the command.
119     
120     @param charCode {String} the character code
121     @param evt {SC.Event} the keydown event
122     @returns {Boolean} YES if you handled the method; NO otherwise
123   */
124   performKeyEquivalent: function(charCode, evt) { return false; },
125   
126   /**
127     This method is called if no other view in the current view hierarchy is
128     bound to the escape or command-. key equivalent.  You can use this to 
129     cancel whatever operation is running.
130     
131     @param sender {Object} the object that triggered; may be null
132     @param evt {SC.Event} the event that triggered the method
133     @returns {Boolean} YES if you handle the event; NO otherwise
134   */
135   cancel: function(sender, evt) {},
136   
137   /**
138     Delete the current selection or delete one element backward from the
139 140     current selection.
141     
142     @param sender {Object} the object that triggered the method; may be null
143     @param evt {SC.Event} the event that triggered the method; may be null
144     @returns {Boolean} YES if you handle the event; NO otherwise
145   */
146   deleteBackward: function(sender, evt) {},
147   
148   /**
149     Delete the current selection or delete one element forward from the
150     current selection.
151     
152     @param sender {Object} the object that triggered the method; may be null
153     @param evt {SC.Event} the event that triggered the method; may be null
154     @returns {Boolean} YES if you handle the event; NO otherwise
155   */
156   deleteForward: function(sender, evt) {},
157   
158   /**
159     A field editor might respond by selecting the field before it.
160     
161     @param sender {Object} the object that triggered the method; may be null
162     @param evt {SC.Event} the event that triggered the method; may be null
163     @returns {Boolean} YES if you handle the event; NO otherwise
164   */
165   insertBacktab: function(sender, evt) {},
166   
167   /**
168     Insert a newline character or end editing of the receiver.
169     
170     @param sender {Object} the object that triggered the method; may be null
171     @param evt {SC.Event} the event that triggered the method; may be null
172     @returns {Boolean} YES if you handle the event; NO otherwise
173   */
174   insertNewline: function(sender, evt) {},
175   
176   /**
177     Insert a tab or move forward to the next field.
178     
179     @param sender {Object} the object that triggered the method; may be null
180     @param evt {SC.Event} the event that triggered the method; may be null
181     @returns {Boolean} YES if you handle the event; NO otherwise
182   */
183   insertTab: function(sender, evt) {},
184   
185   /**
186     Move insertion point/selection backward one. (i.e. left arrow key)
187     
188     @param sender {Object} the object that triggered the method; may be null
189     @param evt {SC.Event} the event that triggered the method; may be null
190     @returns {Boolean} YES if you handle the event; NO otherwise
191   */
192   moveLeft: function(sender, evt) {},
193   
194   /**
195     Move the insertion point/selection forward one (i.e. right arrow key)
196     in left-to-right text, this could be the left arrow key.
197     
198     @param sender {Object} the object that triggered the method; may be null
199     @param evt {SC.Event} the event that triggered the method; may be null
200     @returns {Boolean} YES if you handle the event; NO otherwise
201   */
202   moveRight: function(sender, evt) {},
203   
204   /**
205     Move the insertion point/selection up one (i.e. up arrow key)
206     
207     @param sender {Object} the object that triggered the method; may be null
208     @param evt {SC.Event} the event that triggered the method; may be null
209     @returns {Boolean} YES if you handle the event; NO otherwise
210   */
211   moveUp: function(sender, evt) {},
212   
213   /**
214     Move the insertion point/selection down one (i.e. down arrow key)
215     
216     @param sender {Object} the object that triggered the method; may be null
217     @param evt {SC.Event} the event that triggered the method; may be null
218     @returns {Boolean} YES if you handle the event; NO otherwise
219   */
220   moveDown: function(sender, evt) {},
221   
222   /**
223     Move left, extending the selection. - shift || alt
224     
225     @param sender {Object} the object that triggered the method; may be null
226     @param evt {SC.Event} the event that triggered the method; may be null
227     @returns {Boolean} YES if you handle the event; NO otherwise
228   */
229   moveLeftAndModifySelection: function(sender, evt) {},
230   
231   /**
232     Move right, extending the seleciton - shift || alt
233     
234     @param sender {Object} the object that triggered the method; may be null
235     @param evt {SC.Event} the event that triggered the method; may be null
236     @returns {Boolean} YES if you handle the event; NO otherwise
237   */
238   moveRightAndModifySelection: function(sender, evt) {},
239   
240   /**
241     Move up, extending the selection - shift || alt
242     
243     @param sender {Object} the object that triggered the method; may be null
244     @param evt {SC.Event} the event that triggered the method; may be null
245     @returns {Boolean} YES if you handle the event; NO otherwise
246   */
247   moveUpAndModifySelection: function(sender, evt) {},
248   
249   /**
250     Move down, extending selection - shift || alt
251     
252     @param sender {Object} the object that triggered the method; may be null
253     @param evt {SC.Event} the event that triggered the method; may be null
254     @returns {Boolean} YES if you handle the event; NO otherwise
255   */
256   moveDownAndModifySelection: function(sender, evt) {},
257   
258   /**
259     Move insertion point/selection to beginning of document.
260     
261     @param sender {Object} the object that triggered the method; may be null
262     @param evt {SC.Event} the event that triggered the method; may be null
263     @returns {Boolean} YES if you handle the event; NO otherwise
264   */
265   moveToBeginningOfDocument: function(sender, evt) {},
266   
267   /**
268 269     Move insertion point/selection to end of document.
270     
271     @param sender {Object} the object that triggered the method; may be null
272     @param evt {SC.Event} the event that triggered the method; may be null
273     @returns {Boolean} YES if you handle the event; NO otherwise
274   */
275   moveToEndOfDocument: function(sender, evt) {},
276   
277   /**
278     Page down
279     
280     @param sender {Object} the object that triggered the method; may be null
281     @param evt {SC.Event} the event that triggered the method; may be null
282     @returns {Boolean} YES if you handle the event; NO otherwise
283   */
284   pageDown: function(sender, evt) {},
285   
286   /**
287     Page up
288     
289     @param sender {Object} the object that triggered the method; may be null
290     @param evt {SC.Event} the event that triggered the method; may be null
291     @returns {Boolean} YES if you handle the event; NO otherwise
292   */
293   pageUp: function(sender, evt) {},
294   
295   /**
296     Select all
297     
298     @param sender {Object} the object that triggered the method; may be null
299     @param evt {SC.Event} the event that triggered the method; may be null
300     @returns {Boolean} YES if you handle the event; NO otherwise
301   */
302   selectAll: function(sender, evt) {}
303   
304 };
