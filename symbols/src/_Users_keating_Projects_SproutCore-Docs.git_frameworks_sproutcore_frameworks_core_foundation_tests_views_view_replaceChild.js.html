  9 var parent, child;
 10 module("SC.View#replaceChild", {
 11 	setup: function() {
 12 	  child = SC.View.create();
 13 	  parent = SC.View.create({
 14 	    childViews: [SC.View, SC.View, SC.View]
 15 	  });		
 16 	}
 17  18 });
 19 
 20 
 21 test("swaps the old child with the new child", function() {
 22   var oldChild = parent.childViews[1];
 23 
 24   parent.replaceChild(child, oldChild);
 25   equals(child.get('parentView'), parent, 'child has new parent');
 26   ok(!oldChild.get('parentView'), 'old child no longer has parent');
 27   
 28   equals(parent.childViews[1], child, 'parent view has new child at loc 1');
 29   equals(parent.childViews.length, 3, 'parent has same number of children');
 30 });
