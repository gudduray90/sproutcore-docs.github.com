  8 /*global module test equals context ok */
  9 
 10 var context = null;
 11 
 12 module("SC.RenderContext#end", {
 13   setup: function() {
 14     context = SC.RenderContext();
 15   }
 16 });
 17 
 18 test("should replace opening tag with string and add closing tag, leaving middle content in place", function() {
 19   context.push("line1").end();
 20   equals(context.get(0), "<div>", "opening tag");
 21   equals(context.get(1), "line1", "opening tag");
 22   equals(context.get(2), "</div>", "closing tag");
 23 });
 24 
 25 test("should emit any CSS class names included in the tag opts.classNames array", function() {
 26   context.classNames("foo bar".w()).end();
 27   ok(context.get(0).match(/class=\"foo bar\"/), '<div> has class attr') ;
 28 });
 29 
 30 test("should emit id in tag opts.id", function() {
 31   context.id("foo").end();
 32   ok(context.get(0).match(/id=\"foo\"/), "<div> has id attr");
 33 });
 34 
 35 test("should emit style in tag if opts.styles is defined", function() {
 36   context.styles({ alpha: "beta", foo: "bar" }).end();
 37   ok(context.get(0).match(/style=\"alpha: beta; foo: bar\"/), '<div> has style="alpha: beta; foo: bar"');
 38 });
 39 
 40 test("should emit style with custom browser attributes", function() {
 41   context.styles({ mozColumnCount: '3', webkitColumnCount: '3', oColumnCount: '3', msColumnCount: '3' }).end();
 42   ok(context.get(0).match('<div style="-moz-column-count: 3; -webkit-column-count: 3; -o-column-count: 3; -ms-column-count: 3" >'),
 43                             '<div> has style="-moz-column-count: 3; -webkit-column-count: 3, -o-column-count: 3, -ms-column-count: 3"');
 44 });
 45 
 46 test("should write arbitrary attrs has in opts", function() {
 47   context.attr({ foo: "bar", bar: "baz" }).end();
 48   ok(context.get(0).match(/foo=\"bar\"/), 'has foo="bar"');
 49   ok(context.get(0).match(/bar=\"baz\"/), 'has bar="baz"');
 50 });
 51 
 52 test("classNames should override attrs.class", function() {
 53   context.classNames("foo".w()).attr({ "class": "bar" }).end();
 54   ok(context.get(0).match(/class=\"foo\"/), 'has class="foo"');
 55 });
 56 
 57 test("opts.id should override opts.attrs.id", function() {
 58   context.id("foo").attr({ id: "bar" }).end();
 59   ok(context.get(0).match(/id=\"foo\"/), 'has id="foo"');
 60 });
 61 
 62 test("opts.styles should override opts.attrs.style", function() {
 63   context.styles({ foo: "foo" }).attr({ style: "bar: bar" }).end();
 64   ok(context.get(0).match(/style=\"foo: foo\"/), 'has style="foo: foo"');
 65 });
 66 
 67 test("should return receiver if receiver has no prevObject", function() {
 68   ok(!context.prevObject, 'precondition - prevObject is null');
 69   equals(context.end(), context, 'ends as self');
 70 });
 71 
 72 test("should return prevObject if receiver has prevObject", function() {
 73   var c2 = context.begin();
 74   equals(c2.end(), context, "should return prevObject");
 75 });
 76 
 77 test("emits self closing tag if tag has no content and c._selfClosing !== NO", function() {
 78   var c2 = context.begin('input');
 79   c2.end();
 80   equals(c2.get(0), "<input />");
 81 });
 82 
 83 test("emits two tags even if tag has no content if opts.selfClosing == NO", function() {
 84   context._selfClosing = NO;
 85   
 86   context.end();
 87   equals(context.length, 2, "has two lines");
 88   equals(context.get(0), "<div>", "has opening tag");
 89   equals(context.get(1), "</div>", "has closing tag");
 90 });
 91 
 92 test("does NOT emit self closing tag if it has content, even if opts.selfClosing == YES (because that would yield invalid HTML)", function() {
 93   context._selfClosing = YES;
 94   context.push("line").end();
 95   equals(context.length, 3, "has 3 lines");
 96   equals(context.get(2), "</div>", "has closing tag");
 97 });
 98 
 99 test("it should make sure to clear reused temporary attributes object", function() {
100   
101   // generate one tag...
102   context.begin('input')
103     .id("foo")
104     .styles({ foo: "bar" })
105     .classNames("foo bar".w())
106     .push("line")
107   .end(); 
108   
109   // generate second tag...will reuse internal temporary attrs object.
110   context.begin('input').id("bar").end();
111   var str = context.get(context.length-1);
112   equals(str, "<input id=\"bar\"  />");
113 });
114 
115 test("it should work when nested more than one level deep", function() {
116   context.begin().id("foo")
117     .begin().id("bar").end()
118   .end();
119   
120   var str = context.join('');
121   ok(str.match(/id="foo"/), 'has foo');
122   ok(str.match(/id="bar"/), 'has bar');
123 });
