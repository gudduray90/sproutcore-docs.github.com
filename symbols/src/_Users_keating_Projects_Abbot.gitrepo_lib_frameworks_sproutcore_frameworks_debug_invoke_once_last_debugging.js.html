  8 // When in debug mode, it’s useful for our observer sets (which are used by
  9 // invokeOnce and invokeLast) to record which code scheduled the

 23   SC.ObserverSet.add = function(target, method, context, originatingTarget, originatingMethod, originatingStack) {
 24     var targetGuid = (target) ? SC.guidFor(target) : "__this__";
 25 
 26     // get the set of methods
 27     var methods = this[targetGuid] ;
 28     if (!methods) {
 29       methods = this[targetGuid] = SC.CoreSet.create() ;
 30       methods.target = target ;
 31       methods.isTargetSet = YES ; // used for getMembers().
 32       this.targets++ ;
 33     }
 34     methods.add(method) ;
 35 
 36     // context is really useful sometimes but not used that often so this
 37     // implementation is intentionally lazy.
 38     if (context !== undefined) {
 39       var contexts = methods.contexts || (methods.contexts = {}) ;
 40       contexts[SC.guidFor(method)] = context ;
 41     }
 42     
 43     // THIS IS THE PORTION THAT DIFFERS FROM THE STANDARD IMPLEMENTATION
 44     
 45     // Recording the calling object/function can be a useful debugging tool.
 46     // Since multiple object/functions can schedule the same target/method,
 47     // this value could either be a single value or an array.  (We won't
 48     // always use an array because that adds a certain debugging burden to the
 49     // "only one scheduled it" case.)
 50     if (originatingMethod !== undefined) {
 51       var originatingTargets = methods.originatingTargets;
 52       var originatingMethods = methods.originatingMethods;
 53       var originatingStacks  = methods.originatingStacks;
 54       if (!originatingTargets) originatingTargets = methods.originatingTargets = {};
 55       if (!originatingMethods) originatingMethods = methods.originatingMethods = {};
 56       if (!originatingStacks)  originatingStacks  = methods.originatingStacks  = {};
 57       
 58       var key = SC.guidFor(method);
 59       
 60       var existingMethod = originatingMethods[key];
 61       if (existingMethod  &&  SC.typeOf(existingMethod) !== SC.T_ARRAY) {
 62         // We previously had one entry and now we have two.  We need to
 63         // convert to an array!
 64         var existingTarget = originatingTargets[key];
 65         var existingStack  = originatingStacks[key];
 66         originatingTargets[key] = [existingTarget, originatingTarget];
 67         originatingMethods[key] = [existingMethod, originatingMethod];
 68  69         originatingStacks[key]  = [existingStack, originatingStack];
 70       }
 71       else {
 72         // We didn't previously have a value?  Then no need to use an
 73         // enclosing array.
 74         originatingTargets[key] = originatingTarget;
 75         originatingMethods[key] = originatingMethod;
 76         originatingStacks[key]  = originatingStack;
 77       }
 78     }
 79   };

 82   SC.ObserverSet.invokeMethods = function() {
 83     // iterate through the set, look for sets.
 84     for(var key in this) {
 85       if (!this.hasOwnProperty(key)) continue ;
 86       var value = this[key] ;
 87       if (value && value.isTargetSet) {
 88         var idx = value.length;
 89         var target = value.target ;
 90         
 91         // THIS IS THE PORTION THAT DIFFERS FROM THE STANDARD IMPLEMENTATION
 92         var m, log = SC.LOG_RUNLOOP_INVOCATIONS;
 93         while(--idx>=0) {
 94           m = value[idx];
 95           if (log) {
 96             var mName = m.displayName || m;
 97             
 98             var originatingKey     = SC.guidFor(m),
 99                 originatingTargets = value.originatingTargets;
100             if (!originatingTargets) {
101               // If we didn't capture information for this invocation, just
102               // report what we can.  (We assume we'll always have all three
103               // hashes or none.)
104               SC.Logger.log("Invoking runloop-scheduled method %@ on %@, but we didn’t capture information about who scheduled it…".fmt(mName, target));
105             }
106             else {
107               originatingTargets = originatingTargets[originatingKey];             // Could be one target or an array of them
108               var originatingMethods = value.originatingMethods[originatingKey];   // ditto
109               var originatingStacks  = value.originatingStacks[originatingKey];    // ditto
110 
111               // Were there multiple originating target/method pairs that
112               // scheduled this target/method?  If so, display them all nicely.
113               // Otherwise, optimize our output for only one.
114               if (originatingMethods  &&  SC.typeOf(originatingMethods) === SC.T_ARRAY) {
115                 SC.Logger.log("Invoking runloop-scheduled method %@ on %@, which was scheduled by multiple target/method pairs:".fmt(mName, target));
116               
117                 var i, len,
118                   originatingTarget,
119                   originatingMethod,
120                   originatingStack;
121                 for (i = 0, len = originatingMethods.length;  i < len;  ++i) {
122                   originatingTarget = originatingTargets[i];
123                   originatingMethod = originatingMethods[i];
124                   originatingMethod = originatingMethod.displayName || originatingMethod;
125                   originatingStack  = originatingStacks[i];
126   
127                   SC.Logger.log("[%@]  originated by target %@,  method %@,  stack:".fmt(i, originatingTarget, originatingMethod), originatingStack);
128                 }
129               }
130               else {
131                 var originatingMethodName = originatingMethods.displayName || originatingMethods;
132 
133                 SC.Logger.log("Invoking runloop-scheduled method %@ on %@.  Originated by target %@,  method %@,  stack: ".fmt(mName, target, originatingTargets, originatingMethodName), originatingStacks);
134               }
135             }
136           }
137           m.call(target);
138         }
139 140         // THIS IS THE PORTION THAT DIFFERS FROM THE STANDARD IMPLEMENTATION
141       }
142     }
143   };

146   SC.Object.prototype.invokeOnce = function(method) {
147     var originatingTarget = this ;
148     if (SC.LOG_RUNLOOP_INVOCATIONS) {
149       originatingStack  = SC.getRecentStack();
150       originatingMethod = originatingStack[0];
151     }
152     else {
153       originatingStack  = null;
154       originatingMethod = arguments.callee.caller;
155     }    SC.RunLoop.currentRunLoop.invokeOnce(this, method, originatingTarget, originatingMethod, originatingStack) ;
156     return this ;
157   };

171     SC.RunLoop.currentRunLoop.invokeLast(this, method, originatingTarget, originatingMethod, originatingStack) ;
172     return this ;

176   SC.RunLoop.prototype.invokeOnce = function(target, method, originatingTarget, originatingMethod, originatingStack) {
177     // THIS IS THE PORTION THAT DIFFERS FROM THE STANDARD IMPLEMENTATION
178     
179     // For debugging convenience, record the originating function if it was
180     // not provided for us.
181     if (!originatingTarget) originatingTarget = null;   // More obvious when debugging
182     if (!originatingMethod) {
183       if (SC.LOG_RUNLOOP_INVOCATIONS) {
184         originatingStack  = SC.getRecentStack();
185         originatingMethod = originatingStack[0];
186       }
187       else {
188         originatingStack  = null;
189         originatingMethod = arguments.callee.caller;
190       }
191     }
192     // THIS IS THE PORTION THAT DIFFERS FROM THE STANDARD IMPLEMENTATION
193     
194     
195     // normalize
196     if (method === undefined) { 
197       method = target; target = this ;
198     }
199     if (SC.typeOf(method) === SC.T_STRING) method = target[method];
200     if (!this._invokeQueue) this._invokeQueue = SC.ObserverSet.create();
201     this._invokeQueue.add(target, method, null, originatingTarget, originatingMethod, originatingStack);  // differs from standard implementation
202     return this ;
203   };

236   // Will return the recent stack as a hash with numerical keys, for nice
237   // output in some browser's debuggers.  The “recent” stack is capped at 6
238   // entries.
239   SC.getRecentStack = function() {
240     var currentFunction = arguments.callee.caller,
241         i = 0,
242         stack = {},
243         first = YES,
244         functionName;
245 
246     while (currentFunction  &&  i < 6) {
247       // Skip ourselves!
248       if (first) {
249         first = NO;
250       }
251       else {
252         functionName = currentFunction.displayName || currentFunction.toString();
253         stack[i++] = functionName;
254       }
255       currentFunction = currentFunction.caller;
256     }
257     
258     return stack;
259   };
