 20       var timeToCall = Math.max(0, 16 - (currTime - lastTime));
 21       var id = window.setTimeout(function() { callback(currTime + timeToCall); },
 22         timeToCall);
 23       lastTime = currTime + timeToCall;
 24       return id;
 25     };
