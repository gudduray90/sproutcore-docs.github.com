  7 module("SC.SelectionSet.copy");

  9 test("basic copy", function() {
 10   var content = "1 2 3 4 5 6 7 8 9".w(),
 11       set     = SC.SelectionSet.create().add(content,4,4).remove(content,6),
 12       copy    = set.copy();
 13   
 14   equals(set.get('length'), 3, 'precond - original set should have length');
 15   equals(copy.get('length'), 3, 'copy should have same length');
 16   same(copy, set, 'copy should be the same as original set');
 17 });
