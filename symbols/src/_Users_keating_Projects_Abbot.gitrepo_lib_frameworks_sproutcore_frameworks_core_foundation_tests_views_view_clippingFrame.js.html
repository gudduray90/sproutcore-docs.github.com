  9 var pane, a, aa ;
 10  11 module("SC.View#clippingFrame", {
 12   setup: function() {
 13     htmlbody('<style> .sc-view { border: 1px blue solid; position: absolute;  overflow: hidden; }</style>');
 14     SC.RunLoop.begin();
 15     pane = SC.Pane.design()
 16       .layout({ top: 0, left: 0, width: 200, height: 200 })
 17       .childView(SC.View.design()
 18         .layout({ top: 50, left: 50, width: 100, height: 100 })
 19         .childView(SC.View.design()
 20           .layout({ top: 20, left: 20, width: 40, height: 40 })))
 21       .create();
 22     pane.append();
 23     a = pane.childViews[0];
 24     aa = a.childViews[0];
 25   },
 26   
 27   teardown: function() { 
 28     pane.remove();
 29     pane = a = aa = null ;
 30     SC.RunLoop.end();
 31     clearHtmlbody();
 32   }
 33 });

 35 test("clippingFrame === frame w/ 0 offset if not partially hidden", function() {
 36   var result = pane.get('clippingFrame'), expected = pane.get('frame');
 37   expected.x = expected.y = 0 ;
 38   same(result, expected, 'pane');
 39 
 40   result = a.get('clippingFrame'); expected = a.get('frame');
 41   expected.x = expected.y = 0 ;
 42   same(result, expected, 'child');
 43 
 44   result = aa.get('clippingFrame'); expected = aa.get('frame');
 45   expected.x = expected.y = 0 ;
 46   same(result, expected, 'nested child');
 47 });

101 test("notifies receiver and each child if parent clipping frame changes", function() {
102   var callCount = 0;
103   
104   // setup observers
105   function observer() { callCount++; }
106   a.addObserver('clippingFrame', observer);
107   aa.addObserver('clippingFrame', observer);
108   
109   // now, adjust layout of child so that clipping frame will change...
110   a.adjust('top', -50);
111   
112   // IMPORTANT:  If this test fails because the callCount is > 2 it means that
113   // when you set the layout, the frame is getting invalidated more than once.
114   // This should not happen.  If this is the case, fix the view code so that
115   // it does not invalidate frame more than once before you change this 
116   // number.
117   equals(callCount, 2, 'should invoke observer on child and nested child');
118 });
