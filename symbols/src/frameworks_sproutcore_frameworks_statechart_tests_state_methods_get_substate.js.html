310 test("get non-existing substate 'abc' with using callback", function() {
311   var result, cbState, cbValue, cbKeys; 
312   
313   result = root.getSubstate('abc', function(state, value, keys) {
314     cbState = state;
315     cbValue = value;
316     cbKeys = keys;
317   });
318   
319   ok(!result, "should not return result for 'abc'");
320   equals(cbState, root, "callback state arg should be root state");
321   equals(cbValue, 'abc', "callback value arg should be 'abc'");
322   ok(!cbKeys, "callback keys arg should be none");
323 });
