 14 test("Supports backwards-compatible render method", function() {
 15   var renderCallCount = 0;
 16   var view = SC.View.create({
 17     render: function(context, firstTime) {
 18       renderCallCount++;
 19       ok(context._STYLE_REGEX, 'passes RenderContext');
 20       equals(firstTime, YES, 'passes YES for firstTime');
 21     }
 22   });
 23 
 24   view.createLayer();
 25 
 26   view.render = function(context, firstTime) {
 27     renderCallCount++;
 28     ok(context._STYLE_REGEX, 'passes RenderContext');
 29     equals(firstTime, NO, 'passes NO for firstTime');
 30   };
 31 
 32   view.updateLayer();
 33 
 34   equals(renderCallCount, 2, 'render should have been called twice');
 35 
 36   // Clean up.
 37   view.destroy();
 38 });

 40 test("Treats a view as its own render delegate", function() {
 41   var renderCallCount = 0,
 42       updateCallCount = 0;
 43 
 44   var view = SC.View.create({
 45     render: function(context) {
 46       // Check for existence of _STYLE_REGEX to determine if this is an instance
 47       // of SC.RenderContext
 48       ok(context._STYLE_REGEX, 'passes render context');
 49       renderCallCount++;
 50     },
 51 
 52     update: function(elem) {
 53      ok(elem.jquery, 'passes a jQuery object as first parameter');
 54      updateCallCount++;
 55     }
 56   });
 57 
 58   view.createLayer();
 59   view.updateLayer();
 60   equals(renderCallCount, 1, "calls render once");
 61   equals(updateCallCount, 1, "calls update once");
 62 
 63   // Clean up.
 64   view.destroy();
 65 });

 67 test("Passes data source as first parameter if render delegate is not the view", function() {
 68   var renderCallCount = 0,
 69       updateCallCount = 0;
 70 
 71   var view;
 72 
 73   var renderDelegate = SC.Object.create({
 74     render: function(dataSource, context, firstTime) {
 75       equals(dataSource, view.get('renderDelegateProxy'), "passes the view's render delegate proxy as data source");
 76       ok(context._STYLE_REGEX, "passes render context");
 77       equals(firstTime, undefined, "does not pass third parameter");
 78       renderCallCount++;
 79     },
 80 
 81     update: function(dataSource, elem) {
 82       equals(dataSource, view.get('renderDelegateProxy'), "passes view's render delegate proxy as data source");
 83       ok(elem.jquery, "passes a jQuery object as first parameter");
 84       updateCallCount++;
 85     }
 86   });
 87 
 88   view = SC.View.create({
 89     renderDelegate: renderDelegate
 90   });
 91 
 92   view.createLayer();
 93   view.updateLayer();
 94   equals(renderCallCount, 1, "calls render once");
 95   equals(updateCallCount, 1, "calls update once");
 96 
 97   // Clean up.
 98   view.destroy();
 99 });

101 test("Extending view with render delegate by implementing old render method", function() {
102   var renderCalls = 0, updateCalls = 0;
103   var parentView = SC.View.extend({
104     renderDelegate: SC.Object.create({
105       render: function(context) {
106         renderCalls++;
107       },
108 
109       update: function(cq) {
110         updateCalls++;
111       }
112     })
113   });
114 
115   var childView = parentView.create({
116     render: function(context, firstTime) {
117       sc_super();
118     }
119   });
120 
121   childView.createLayer();
122   childView.updateLayer();
123 
124   equals(renderCalls, 1, "calls render on render delegate once");
125   equals(updateCalls, 1, "calls update on render delegates once");
126 });

128 test("Views that do not override render should render their child views", function() {
129   var newStyleCount = 0, oldStyleCount = 0, renderDelegateCount = 0;
130 
131   var parentView = SC.View.design({
132     childViews: 'newStyle oldStyle renderDelegateView'.w(),
133 
134     newStyle: SC.View.design({
135       render: function(context) {
136         newStyleCount++;
137       },
138 
139       update: function() {
140         // no op
141       }
142     }),
143 
144     oldStyle: SC.View.design({
145       render: function(context, firstTime) {
146         oldStyleCount++;
147       }
148     }),
149 
150     renderDelegateView: SC.View.design({
151       renderDelegate: SC.Object.create({
152         render: function(dataSource, context) {
153           ok(dataSource.isViewRenderDelegateProxy, "Render delegate should get passed a view's proxy for its data source");
154           renderDelegateCount++;
155         },
156 
157         update: function() {
158           // no op
159         }
160       })
161     })
162   });
163 
164   parentView = parentView.create();
165 
166   parentView.createLayer();
167   parentView.updateLayer();
168 
169   equals(newStyleCount, 1, "calls render on new style view once");
170   equals(oldStyleCount, 1, "calls render on old style view once");
171   equals(renderDelegateCount, 1, "calls render on render delegate once");
172 });
