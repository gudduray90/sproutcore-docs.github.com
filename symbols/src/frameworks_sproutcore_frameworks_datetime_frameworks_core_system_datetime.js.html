 66 /** @class
 67 
 68   A Scanner reads a string and interprets the characters into numbers. You
 69   assign the scanner's string on initialization and the scanner progresses
 70   through the characters of that string from beginning to end as you request
 71   items.
 72 
 73   Scanners are used by `DateTime` to convert strings into `DateTime` objects.
 74 
 75   @extends SC.Object
 76   @since SproutCore 1.0
 77   @author Martin Ottenwaelter
 78 */
 79 SC.Scanner = SC.Object.extend(
 80 /** @scope SC.Scanner.prototype */ {
 81 
 82   /**
 83     The string to scan. You usually pass it to the create method:
 84 
 85         SC.Scanner.create({string: 'May, 8th'});
 86 
 87     @type String
 88   */
 89   string: null,
 90 
 91   /**
 92     The current scan location. It is incremented by the scanner as the
 93     characters are processed.
 94     The default is 0: the beginning of the string.
 95 
 96     @type Integer
 97   */
 98   scanLocation: 0,
 99 
100   /**
101     Reads some characters from the string, and increments the scan location
102     accordingly.
103 
104     @param {Integer} len The amount of characters to read
105     @throws {SC.SCANNER_OUT_OF_BOUNDS_ERROR} If asked to read too many characters
106     @returns {String} The characters
107   */
108   scan: function(len) {
109     if (this.scanLocation + len > this.length) throw SC.SCANNER_OUT_OF_BOUNDS_ERROR;
110     var str = this.string.substr(this.scanLocation, len);
111     this.scanLocation += len;
112     return str;
113   },
114 
115   /**
116     Reads some characters from the string and interprets it as an integer.
117 
118     @param {Integer} min_len The minimum amount of characters to read
119     @param {Integer} [max_len] The maximum amount of characters to read (defaults to the minimum)
120     @throws {SC.SCANNER_INT_ERROR} If asked to read non numeric characters
121     @returns {Integer} The scanned integer
122   */
123   scanInt: function(min_len, max_len) {
124     if (max_len === undefined) max_len = min_len;
125     var str = this.scan(max_len);
126     var re = new RegExp("^\\d{" + min_len + "," + max_len + "}");
127     var match = str.match(re);
128     if (!match) throw SC.SCANNER_INT_ERROR;
129     if (match[0].length < max_len) {
130       this.scanLocation += match[0].length - max_len;
131     }
132     return parseInt(match[0], 10);
133   },
134 
135   /**
136     Attempts to skip a given string.
137 
138     @param {String} str The string to skip
139     @throws {SC.SCANNER_SKIP_ERROR} If the given string could not be scanned
140     @returns {Boolean} YES if the given string was successfully scanned, NO otherwise
141   */
142   skipString: function(str) {
143     if (this.scan(str.length) !== str) throw SC.SCANNER_SKIP_ERROR;
144     return YES;
145   },
146 
147   /**
148     Attempts to scan any string in a given array.
149 
150     @param {Array} ary the array of strings to scan
151     @throws {SC.SCANNER_SCAN_ARRAY_ERROR} If no string of the given array is found
152     @returns {Integer} The index of the scanned string of the given array
153   */
154   scanArray: function(ary) {
155     for (var i = 0, len = ary.length; i < len; i++) {
156       if (this.scan(ary[i].length) === ary[i]) {
157         return i;
158       }
159       this.scanLocation -= ary[i].length;
160     }
161     throw SC.SCANNER_SCAN_ARRAY_ERROR;
162   }
163

167 /** @class
168 
169   A class representation of a date and time. It's basically a wrapper around
170   the Date javascript object, KVO-friendly and with common date/time
171   manipulation methods.
172 
173   This object differs from the standard JS Date object, however, in that it
174   supports time zones other than UTC and that local to the machine on which
175   it is running.  Any time zone can be specified when creating an
176   `SC.DateTime` object, e.g.
177 
178       // Creates a DateTime representing 5am in Washington, DC and 10am in London
179       var d = SC.DateTime.create({ hour: 5, timezone: 300 }); // -5 hours from UTC
180       var e = SC.DateTime.create({ hour: 10, timezone: 0 }); // same time, specified in UTC
181 
182   and it is true that `d.isEqual(e)`.
183 
184   The time zone specified upon creation is permanent, and any calls to
185   `get()` on that instance will return values expressed in that time zone. So,
186 
187       d.get('hour') returns 5.
188       e.get('hour') returns 10.
189 
190   but
191 
192       d.get('milliseconds') === e.get('milliseconds')
193 
194   is true, since they are technically the same position in time.
195 
196   You can also use SC.DateTime as a record attribute on a data model.
197 
198       SC.Record.attr(SC.DateTime); // Default format is ISO8601. See `SC.DateTime.recordFormat`
199       SC.Record.attr(SC.DateTime, { format: '%d/%m/%Y' }); // Attribute stored as a string in '%d/%m/%Y' format
200       SC.Record.attr(SC.DateTime, { useUnixTime: YES }); // Attribute stored as a number in Unix time

206   @author Jonathan Lewis

655   /**
656     @private
657     @see SC.DateTime#unknownProperty
658   */
659   _get: function(key, start, timezone) {
660     var ms, tz, doy, m, y, firstDayOfWeek, dayOfWeek, dayOfYear, prefix, suffix;
661     var currentWeekday, targetWeekday;
662     var d = this._date;
663     var originalTime, v = null;
664 
665     // Set up an absolute date/time using the given milliseconds since Jan 1, 1970.
666     // Only do it if we're given a time value, though, otherwise we want to use the
667     // last one we had because this `_get()` method is recursive.
668     //
669     // Note that because these private time calc methods are recursive, and because all DateTime instances
670     // share an internal this._date and `this._tz` state for doing calculations, methods
671     // that modify `this._date` or `this._tz` should restore the last state before exiting
672     // to avoid obscure calculation bugs.  So we save the original state here, and restore
673     // it before returning at the end.
674     originalTime = this._setCalcState(start, timezone); // save so we can restore it to how it was before we got here
675 
676     // Check this first because it is an absolute value -- no tweaks necessary when calling for milliseconds
677     if (key === 'milliseconds') {
678       v = d.getTime();
679     }
680     else if (key === 'timezone') {
681       v = this._tz;
682     }
683 
684     // 'nextWeekday' or 'lastWeekday'.
685     // We want to do this calculation in local time, before shifting UTC below.
686     if (v === null) {
687       prefix = key.slice(0, 4);
688       suffix = key.slice(4);
689       if (prefix === 'last' || prefix === 'next') {
690         currentWeekday = this._get('dayOfWeek', start, timezone);
691         targetWeekday = this._englishDayNames.indexOf(suffix);
692         if (targetWeekday >= 0) {
693           var delta = targetWeekday - currentWeekday;
694           if (prefix === 'last' && delta >= 0) delta -= 7;
695           if (prefix === 'next' && delta <  0) delta += 7;
696           this._advance({ day: delta }, start, timezone);
697           v = this._createFromCurrentState();
698         }
699       }
700     }
701 
702     if (v === null) {
703       // need to adjust for alternate display time zone.
704       // Before calculating, we need to get everything into a common time zone to
705       // negate the effects of local machine time (so we can use all the 'getUTC...() methods on Date).
706       if (timezone !== undefined) {
707         this._setCalcState(d.getTime() - (timezone * 60000), 0); // make this instance's time zone the new UTC temporarily
708       }
709 
710       // simple keys
711       switch (key) {
712         case 'year':
713           v = d.getUTCFullYear(); //TODO: investigate why some libraries do getFullYear().toString() or getFullYear()+""
714           break;
715         case 'month':
716           v = d.getUTCMonth()+1; // January is 0 in JavaScript
717           break;
718         case 'day':
719           v = d.getUTCDate();
720           break;
721         case 'dayOfWeek':
722           v = d.getUTCDay();
723           break;
724         case 'hour':
725           v = d.getUTCHours();
726           break;
727         case 'minute':
728           v = d.getUTCMinutes();
729           break;
730         case 'second':
731           v = d.getUTCSeconds();
732           break;
733         case 'millisecond':
734           v = d.getUTCMilliseconds();
735           break;
736         case 'elapsed':
737           v = +new Date() - d.getTime() - (timezone * 60000);
738           break;
739       }
740 
741       // isLeapYear
742       if ((v === null) && (key === 'isLeapYear')) {
743         y = this._get('year');
744         v = (y%4 === 0 && y%100 !== 0) || y%400 === 0;
745       }
746 
747       // daysInMonth
748       if ((v === null) && (key === 'daysInMonth')) {
749         switch (this._get('month')) {
750           case 4:
751           case 6:
752           case 9:
753           case 11:
754             v = 30;
755             break;
756           case 2:
757             v = this._get('isLeapYear') ? 29 : 28;
758             break;
759           default:
760             v = 31;
761             break;
762         }
763       }
764 
765       // dayOfYear
766       if ((v === null) && (key === 'dayOfYear')) {
767         ms = d.getTime(); // save time
768         doy = this._get('day');
769         this._setCalcStateFromHash({ day: 1 });
770         for (m = this._get('month') - 1; m > 0; m--) {
771           this._setCalcStateFromHash({ month: m });
772           doy += this._get('daysInMonth');
773         }
774         d.setTime(ms); // restore time
775         v = doy;
776       }
777 
778       // week, week0 or week1
779       if ((v === null) && (key.slice(0, 4) === 'week')) {
780         // firstDayOfWeek should be 0 (Sunday) or 1 (Monday)
781         firstDayOfWeek = key.length === 4 ? 1 : parseInt(key.slice('4'), 10);
782         dayOfWeek = this._get('dayOfWeek');
783         dayOfYear = this._get('dayOfYear') - 1;
784         if (firstDayOfWeek === 0) {
785           v = parseInt((dayOfYear - dayOfWeek + 7) / 7, 10);
786         }
787         else {
788           v = parseInt((dayOfYear - (dayOfWeek - 1 + 7) % 7 + 7) / 7, 10);
789         }
790       }
791     }
792 
793     // restore the internal calculation state in case someone else was in the
794     // middle of a calculation (we might be recursing).
795     this._setCalcState(originalTime.milliseconds, originalTime.timezone);
796 
797     return v;
798   },

831   /*
832     @private
833 
834     Converts a standard date/time options hash to an integer representing that position
835     in time relative to Jan 1, 1970
836   */
837   _toMilliseconds: function(options, start, timezone, resetCascadingly) {
838     var opts = options ? SC.clone(options) : {};
839     var d = this._date;
840     var previousMilliseconds = d.getTime(); // rather than create a new Date object, we'll reuse the instance we have for calculations, then restore it
841     var ms, tz;
842 
843     // Initialize our internal for-calculations Date object to our current date/time.
844     // Note that this object was created in the local machine time zone, so when we set
845     // its params later, it will be assuming these values to be in the same time zone as it is.
846     // It's ok for start to be null, in which case we'll just keep whatever we had in 'd' before.
847     if (!SC.none(start)) {
848       d.setTime(start); // using milliseconds here specifies an absolute location in time, regardless of time zone, so that's nice
849     }
850 
851     // We have to get all time expressions, both in 'options' (assume to be in time zone 'timezone')
852     // and in 'd', to the same time zone before we can any calculations correctly.  So because the Date object provides
853     // a suite of UTC getters and setters, we'll temporarily redefine 'timezone' as our new
854     // 'UTC', so we don't have to worry about local machine time.  We do this by subtracting
855     // milliseconds for the time zone offset.  Then we'll do all our calculations, then convert
856     // it back to real UTC.
857 
858     // (Zero time zone is considered a valid value.)
859     tz = (timezone !== undefined) ? timezone : (this.timezone !== undefined) ? this.timezone : 0;
860     d.setTime(d.getTime() - (tz * 60000)); // redefine 'UTC' to establish a new local absolute so we can use all the 'getUTC...()' Date methods
861 
862     // the time options (hour, minute, sec, millisecond)
863     // reset cascadingly (see documentation)
864     if (resetCascadingly === undefined || resetCascadingly === YES) {
865       if ( !SC.none(opts.hour) && SC.none(opts.minute)) {
866         opts.minute = 0;
867       }
868       if (!(SC.none(opts.hour) && SC.none(opts.minute))
869           && SC.none(opts.second)) {
870         opts.second = 0;
871       }
872       if (!(SC.none(opts.hour) && SC.none(opts.minute) && SC.none(opts.second))
873           && SC.none(opts.millisecond)) {
874         opts.millisecond = 0;
875       }
876     }
877 
878     // Get the current values for any not provided in the options hash.
879     // Since everything is in 'UTC' now, use the UTC accessors.  We do this because,
880     // according to javascript Date spec, you have to set year, month, and day together
881     // if you're setting any one of them.  So we'll use the provided Date.UTC() method
882     // to get milliseconds, and we need to get any missing values first...
883     if (SC.none(opts.year))        opts.year = d.getUTCFullYear();
884     if (SC.none(opts.month))       opts.month = d.getUTCMonth() + 1; // January is 0 in JavaScript
885     if (SC.none(opts.day))         opts.day = d.getUTCDate();
886     if (SC.none(opts.hour))        opts.hour = d.getUTCHours();
887     if (SC.none(opts.minute))      opts.minute = d.getUTCMinutes();
888     if (SC.none(opts.second))      opts.second = d.getUTCSeconds();
889     if (SC.none(opts.millisecond)) opts.millisecond = d.getUTCMilliseconds();
890 
891     // Ask the JS Date to calculate milliseconds for us (still in redefined UTC).  It
892     // is best to set them all together because, for example, a day value means different things
893     // to the JS Date object depending on which month or year it is.  It can now handle that stuff
894     // internally as it's made to do.
895     ms = Date.UTC(opts.year, opts.month - 1, opts.day, opts.hour, opts.minute, opts.second, opts.millisecond);
896 
897     // Now that we've done all our calculations in a common time zone, add back the offset
898     // to move back to real UTC.
899     d.setTime(ms + (tz * 60000));
900     ms = d.getTime(); // now get the corrected milliseconds value
901 
902     // Restore what was there previously before leaving in case someone called this method
903     // in the middle of another calculation.
904     d.setTime(previousMilliseconds);
905 
906     return ms;
907   },

909   /**
910     Returns a new `SC.DateTime` object advanced according the the given parameters.
911     The parameters can be:
912 
913      - none, to create a `SC.DateTime` instance initialized to the current
914        date and time in the local timezone,
915      - a integer, the number of milliseconds since
916        January, 1st 1970 00:00:00.0 UTC
917      - a options hash that can contain any of the following properties: year,
918        month, day, hour, minute, second, millisecond, timezone
919 
920     Note that if you attempt to create a `SC.DateTime` instance that has already
921     been created, then, for performance reasons, a cached value may be
922     returned.
923 
924     The timezone option is the offset, in minutes, between UTC and local time.
925     If you don't pass a timezone option, the date object is created in the
926     local timezone. If you want to create a UTC+2 (CEST) date, for example,
927     then you should pass a timezone of -120.
928 
929     @param options one of the three kind of parameters described above
930     @returns {SC.DateTime} the SC.DateTime instance that corresponds to the
931       passed parameters, possibly fetched from cache
932   */
933   create: function() {
934     var arg = arguments.length === 0 ? {} : arguments[0];
935     var timezone;
936 
937     // if simply milliseconds since Jan 1, 1970 are given, just use those
938     if (SC.typeOf(arg) === SC.T_NUMBER) {
939       arg = { milliseconds: arg };
940     }
941 
942     // Default to local machine time zone if none is given
943     timezone = (arg.timezone !== undefined) ? arg.timezone : this.timezone;
944     if (timezone === undefined) timezone = 0;
945 
946     // Desired case: create with milliseconds if we have them.
947     // If we don't, convert what we have to milliseconds and recurse.
948     if (!SC.none(arg.milliseconds)) {
949 
950       // quick implementation of a FIFO set for the cache
951       var key = 'nu' + arg.milliseconds + timezone, cache = this._dt_cache;
952       var ret = cache[key];
953       if (!ret) {
954         var previousKey, idx = this._dt_cache_index, C = this;
955         ret = cache[key] = new C([{ _ms: arg.milliseconds, timezone: timezone }]);
956         idx = this._dt_cache_index = (idx + 1) % this._DT_CACHE_MAX_LENGTH;
957         previousKey = cache[idx];
958         if (previousKey !== undefined && cache[previousKey]) delete cache[previousKey];
959         cache[idx] = key;
960       }
961       return ret;
962     }
963     // otherwise, convert what we have to milliseconds and try again
964     else {
965       var now = new Date();
966 
967       return this.create({ // recursive call with new arguments
968         milliseconds: this._toMilliseconds(arg, now.getTime(), timezone, arg.resetCascadingly),
969         timezone: timezone
970       });
971     }
972   },

988   /**
989     Returns a `SC.DateTime` object created from a given string parsed with a given
990     format. Returns `null` if the parsing fails.
991 
992     @see SC.DateTime#toFormattedString for a description of the format parameter
993     @param {String} str the string to parse
994     @param {String} fmt the format to parse the string with
995     @returns {DateTime} the DateTime corresponding to the string parameter
996   */
997   parse: function(str, fmt) {
998     // Declared as an object not a literal since in some browsers the literal
999     // retains state across function calls
1000     var re = new RegExp('(?:%([aAbBcdDhHiIjmMpsSUWwxXyYZ%])|(.))', "g");
1001     var d, parts, opts = {}, check = {}, scanner = SC.Scanner.create({string: str});
1002 
1003     if (SC.none(fmt)) fmt = SC.DATETIME_ISO8601;
1004 
1005     try {
1006       while ((parts = re.exec(fmt)) !== null) {
1007         switch(parts[1]) {
1008           case 'a': check.dayOfWeek = scanner.scanArray(this.abbreviatedDayNames); break;
1009           case 'A': check.dayOfWeek = scanner.scanArray(this.dayNames); break;
1010           case 'b': opts.month = scanner.scanArray(this.abbreviatedMonthNames) + 1; break;
1011           case 'B': opts.month = scanner.scanArray(this.monthNames) + 1; break;
1012           case 'c': throw new Error("%c is not implemented");
1013           case 'd':
1014           case 'D': opts.day = scanner.scanInt(1, 2); break;
1015           case 'e': throw new Error("%e is not implemented");
1016           case 'E': throw new Error("%E is not implemented");
1017           case 'h':
1018           case 'H': opts.hour = scanner.scanInt(1, 2); break;
1019           case 'i':
1020           case 'I': opts.hour = scanner.scanInt(1, 2); break;
1021           case 'j': throw new Error("%j is not implemented");
1022           case 'm': opts.month = scanner.scanInt(1, 2); break;
1023           case 'M': opts.minute = scanner.scanInt(1, 2); break;
1024           case 'p': opts.meridian = scanner.scanArray(this.AMPMNames); break;
1025           case 'S': opts.second = scanner.scanInt(1, 2); break;
1026           case 's': opts.millisecond = scanner.scanInt(1, 3); break;
1027           case 'U': throw new Error("%U is not implemented");
1028           case 'W': throw new Error("%W is not implemented");
1029           case 'w': throw new Error("%w is not implemented");
1030           case 'x': throw new Error("%x is not implemented");
1031           case 'X': throw new Error("%X is not implemented");
1032           case 'y': opts.year = scanner.scanInt(2); opts.year += (opts.year > 70 ? 1900 : 2000); break;
1033           case 'Y': opts.year = scanner.scanInt(4); break;
1034           case 'Z':
1035             var modifier = scanner.scan(1);
1036             if (modifier === 'Z') {
1037               opts.timezone = 0;
1038             } else if (modifier === '+' || modifier === '-' ) {
1039               var h = scanner.scanInt(2);
1040               if (scanner.scan(1) !== ':') scanner.scan(-1);
1041               var m = scanner.scanInt(2);
1042               opts.timezone = (modifier === '+' ? -1 : 1) * (h*60 + m);
1043             }
1044             break;
1045           case '%': scanner.skipString('%'); break;
1046           default:  scanner.skipString(parts[0]); break;
1047         }
1048       }
1049     } catch (e) {
1050       SC.Logger.log('SC.DateTime.createFromString ' + e.toString());
1051       return null;
1052     }
1053 
1054     if (!SC.none(opts.meridian) && !SC.none(opts.hour)) {
1055       if ((opts.meridian === 1 && opts.hour !== 12)
1056           || (opts.meridian === 0 && opts.hour === 12)) {
1057         opts.hour = (opts.hour + 12) % 24;
1058       }
1059       delete opts.meridian;
1060     }
1061 
1062     d = this.create(opts);
1063 
1064     if (!SC.none(check.dayOfWeek) && d.get('dayOfWeek') !== check.dayOfWeek) {
1065       return null;
1066     }
1067 
1068     return d;
1069   },

1092   __toFormattedString: function(part, start, timezone) {
1093     var hour, offset;
1094 
1095     // Note: all calls to _get() here should include only one
1096     // argument, since _get() is built for recursion and behaves differently
1097     // if arguments 2 and 3 are included.

1105       case 'A': return this.dayNames[this._get('dayOfWeek')];

1113       case 'H': return this._pad(this._get('hour'));

1117       case 'I':

1119         return this._pad((hour === 12 || hour === 0) ? 12 : (hour + 12) % 12);

1126       case 'u': return this._pad(this._get('utc')); //utc
1127       case 'U': return this._pad(this._get('week0'));
1128       case 'W': return this._pad(this._get('week1'));
1129       case 'w': return this._get('dayOfWeek');
1130       case 'x': return this._date.toDateString();
1131       case 'X': return this._date.toTimeString();
1132       case 'y': return this._pad(this._get('year') % 100);
1133       case 'Y': return this._get('year');
1134       case 'Z':
1135         offset = -1 * timezone;
1136         return (offset >= 0 ? '+' : '-')
1137                + this._pad(parseInt(Math.abs(offset)/60, 10))
1138                + ':'
1139                + this._pad(Math.abs(offset)%60);
