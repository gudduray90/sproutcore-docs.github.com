  3 SC.Handlebars.ViewHelper = SC.Object.create({
  4   helper: function(thisContext, path, options) {
  5     var inverse = options.inverse;
  6     var data = options.data;
  7     var view = data.view;
  8     var fn = options.fn;
  9     var hash = options.hash;
 10 
 11     var newView;
 12     if (path.isClass || path.isObject) {
 13       newView = path;
 14       if (!newView) {
 15         throw "Null or undefined object was passed to the #view helper. Did you mean to pass a property path string?";
 16       }
 17     } else {
 18       // Path is relative, look it up with this view as the root
 19       if (path.charAt(0) === '.') {
 20         newView = SC.objectForPropertyPath(path.slice(1), view);
 21       } else {
 22         // Path is absolute, look up path on global (window) object
 23         newView = SC.getPath(thisContext, path);
 24         if (!newView) {
 25           newView = SC.getPath(path);
 26         }
 27       }
 28       if (!newView) { throw "Unable to find view at path '" + path + "'"; }
 29     }
 30 
 31     if (hash.id) { hash.layerId = hash.id; }
 32 
 33     var contextOptions = {
 34       'id': hash.id,
 35       'class': hash['class'],
 36       'classBinding': hash.classBinding
 37     };
 38     delete hash.id;
 39     delete hash['class'];
 40     delete hash.classBinding;
 41 
 42     if (newView.isClass) {
 43       newView = newView.extend(hash);
 44     } else {
 45       SC.mixin(newView, hash);
 46     }
 47 
 48     var currentView = data.view;
 49 
 50     var childViews = currentView.get('childViews');
 51     var childView = currentView.createChildView(newView);
 52 
 53     // Set the template of the view to the passed block if we got one
 54     if (fn) { childView.template = fn; }
 55 
 56 
 57     childViews.pushObject(childView);
 58 
 59     var context = SC.RenderContext(childView.get('tagName'));
 60 
 61     // Add id and class names passed to view helper
 62     this.applyAttributes(contextOptions, childView, context);
 63 
 64     childView.applyAttributesToContext(context);
 65     // tomdale wants to make SproutCore slow
 66     childView.render(context, YES);
 67 
 68     return new Handlebars.SafeString(context.join());
 69   },
 70 
 71   applyAttributes: function(options, childView, context) {
 72     var id = options.id;
 73     var classNames = options['class'];
 74 
 75     if (classNames) {
 76       context.addClass(classNames.split(' '));
 77     }
 78 
 79     if (id) {
 80       childView.set('layerId', id);
 81       context.id(id);
 82     }
 83 
 84     var classBindings = options.classBinding;
 85     if (classBindings) {
 86       SC.Handlebars.bindClasses(childView, classBindings, childView).forEach(function(className) {
 87         context.setClass(className, YES);
 88       });
 89     }
 90   }
 91 });
 92 
 93 
 94 Handlebars.registerHelper('view', function(path, options) {
 95   return SC.Handlebars.ViewHelper.helper(this, path, options);
 96 });
