 11 /**
 12   @class
 13 
 14   The run loop provides a universal system for coordinating events within
 15   your application.  The run loop processes timers as well as pending
 16   observer notifications within your application.
 17 
 18   To use a RunLoop within your application, you should make sure your event
 19   handlers always begin and end with SC.RunLoop.begin() and SC.RunLoop.end()
 20 
 21   The RunLoop is important because bindings do not fire until the end of
 22   your run loop is reached.  This improves the performance of your
 23   application.
 24 
 25   Example:
 26 
 27   This is how you could write your mouseup handler in jQuery:
 28 
 29         $('#okButton').on('click', function() {
 30           SC.RunLoop.begin();
 31 
 32           // handle click event...
 33 
 34           SC.RunLoop.end(); // allows bindings to trigger...
 35         });
 36 
 37   @extends SC.Object
 38   @since SproutCore 1.0
 39 */
 40 
 41 
 42 // When in debug mode, users can log deferred calls (such as .invokeOnce()) by
 43 // setting SC.LOG_DEFERRED_CALLS.  We'll declare the variable explicitly to make
 44 // life easier for people who want to enter it inside consoles that auto-
 45 // complete.
 46 //@if(debug)
 47 if (!SC.LOG_DEFERRED_CALLS) SC.LOG_DEFERRED_CALLS = false;
 48 //@endif

 50 SC.RunLoop = SC.Object.extend(/** @scope SC.RunLoop.prototype */ {
 51 
 52   /**
 53     Call this method whenver you begin executing code.
 54 
 55     This is typically invoked automatically for you from event handlers and
 56     the timeout handler.  If you call setTimeout() or setInterval() yourself,
 57     you may need to invoke this yourself.
 58 
 59     @returns {SC.RunLoop} receiver
 60   */
 61   beginRunLoop: function() {
 62     this._start = new Date().getTime() ; // can't use Date.now() in runtime
 63     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 64       SC.Logger.log("-- SC.RunLoop.beginRunLoop at %@".fmt(this._start));
 65     }
 66     this._runLoopInProgress = YES;
 67     this._flushinvokeNextQueue();
 68     return this ;
 69   },
 70 
 71   /**
 72     YES when a run loop is in progress
 73 
 74     @property
 75     @type Boolean
 76   */
 77   isRunLoopInProgress: function() {
 78     return this._runLoopInProgress;
 79   }.property(),
 80 
 81   /**
 82     Call this method whenever you are done executing code.
 83 
 84     This is typically invoked automatically for you from event handlers and
 85     the timeout handler.  If you call setTimeout() or setInterval() yourself
 86     you may need to invoke this yourself.
 87 
 88     @returns {SC.RunLoop} receiver
 89   */
 90   endRunLoop: function() {
 91     // at the end of a runloop, flush all the delayed actions we may have
 92     // stored up.  Note that if any of these queues actually run, we will
 93     // step through all of them again.  This way any changes get flushed
 94     // out completely.
 95 
 96     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 97       SC.Logger.log("-- SC.RunLoop.endRunLoop ~ flushing application queues");
 98     }
 99 
100     this.flushAllPending();
101 
102     this._start = null ;
103 
104     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
105       SC.Logger.log("-- SC.RunLoop.endRunLoop ~ End");
106     }
107 
108     SC.RunLoop.lastRunLoopEnd = Date.now();
109     this._runLoopInProgress = NO;
110 
111     return this ;
112   },
113 
114   /**
115     Repeatedly flushes all bindings, observers, and other queued functions until all queues are empty.
116   */
117   flushAllPending: function() {
118     var didChange ;
119 
120     do {
121       didChange = this.flushApplicationQueues() ;
122       if (!didChange) didChange = this._flushinvokeLastQueue() ;
123     } while(didChange) ;
124   },
125 
126 
127   /**
128     Invokes the passed target/method pair once at the end of the runloop.
129     You can call this method as many times as you like and the method will
130     only be invoked once.
131 
132     Usually you will not call this method directly but use invokeOnce()
133     defined on SC.Object.
134 
135     Note that in development mode only, the object and method that call this
136     method will be recorded, for help in debugging scheduled code.
137 
138     @param {Object} target
139     @param {Function} method
140     @returns {SC.RunLoop} receiver
141   */
142   invokeOnce: function(target, method) {
143     // normalize
144     if (method === undefined) {
145       method = target; target = this ;
146     }
147 
148     var deferredCallLoggingInfo;      // Used only in debug mode
149 
150     //@if(debug)
151     // When in debug mode, SC.Object#invokeOnce() will pass in the originating
152     // method, target, and stack.  That way, we'll record the interesting parts
153     // rather than having most of these calls seemingly coming from
154     // SC.Object#invokeOnce().
155     //
156     // If it was not specified, we'll record the originating function ourselves.
157     var shouldLog = SC.LOG_DEFERRED_CALLS;
158     if (shouldLog) {
159       var originatingTarget = arguments[2],
160           originatingMethod = arguments[3],
161           originatingStack  = arguments[4];
162 
163       if (!originatingTarget) originatingTarget = null;   // More obvious when debugging
164       if (!originatingMethod) {
165         originatingStack  = SC._getRecentStack();
166         originatingMethod = originatingStack[0];
167       }
168 
169       deferredCallLoggingInfo = {
170         originatingTarget: originatingTarget,
171         originatingMethod: originatingMethod,
172         originatingStack:  originatingStack
173       };
174     }
175 176     //@endif
177 
178     if (typeof method === "string") method = target[method];
179     if (!this._invokeQueue) this._invokeQueue = SC.ObserverSet.create();
180     if ( method ) this._invokeQueue.add(target, method, undefined, deferredCallLoggingInfo);
181     return this ;
182   },
183 
184   /**
185     Invokes the passed target/method pair at the very end of the run loop,
186     once all other delayed invoke queues have been flushed.  Use this to
187     schedule cleanup methods at the end of the run loop once all other work
188     (including rendering) has finished.
189 
190     If you call this with the same target/method pair multiple times it will
191     only invoke the pair only once at the end of the runloop.
192 
193     Usually you will not call this method directly but use invokeLast()
194     defined on SC.Object.
195 
196     Note that in development mode only, the object and method that call this
197     method will be recorded, for help in debugging scheduled code.
198 
199     @param {Object} target
200     @param {Function} method
201     @returns {SC.RunLoop} receiver
202   */
203   invokeLast: function(target, method) {
204     // normalize
205     if (method === undefined) {
206       method = target; target = this ;
207     }
208 
209     var deferredCallLoggingInfo;      // Used only in debug mode
210 
211     //@if(debug)
212     // When in debug mode, SC.Object#invokeOnce() will pass in the originating
213     // method, target, and stack.  That way, we'll record the interesting parts
214     // rather than having most of these calls seemingly coming from
215     // SC.Object#invokeOnce().
216     //
217     // If it was not specified, we'll record the originating function ourselves.
218     var shouldLog = SC.LOG_DEFERRED_CALLS;
219     if (shouldLog) {
220       var originatingTarget = arguments[2],
221           originatingMethod = arguments[3],
222           originatingStack  = arguments[4];
223 
224       if (!originatingTarget) originatingTarget = null;   // More obvious when debugging
225       if (!originatingMethod) {
226         originatingStack  = SC._getRecentStack();
227         originatingMethod = originatingStack[0];
228       }
229 
230       deferredCallLoggingInfo = {
231         originatingTarget: originatingTarget,
232         originatingMethod: originatingMethod,
233         originatingStack:  originatingStack
234       };
235     }
236     //@endif
237 
238 
239     if (typeof method === "string") method = target[method];
240     if (!this._invokeLastQueue) this._invokeLastQueue = SC.ObserverSet.create();
241     this._invokeLastQueue.add(target, method, undefined, deferredCallLoggingInfo);
242     return this ;
243   },
244 
245   /**
246     Invokes the passed target/method pair once at the beginning of the next
247     runloop, before any other methods (including events) are processed.
248     Use this to defer painting to make views more responsive.
249 
250     If you call this with the same target/method pair multiple times it will
251     only invoke the pair only once at the beginning of the next runloop.
252 
253     Usually you will not call this method directly but use invokeNext()
254     defined on SC.Object.
255 
256     Note that in development mode only, the object and method that call this
257     method will be recorded, for help in debugging scheduled code.
258 
259     @param {Object} target
260     @param {Function} method
261     @returns {SC.RunLoop} receiver
262   */
263   invokeNext: function(target, method) {
264     // normalize
265     if (method === undefined) {
266       method = target; target = this ;
267     }
268 
269     if (typeof method === "string") method = target[method];
270     if (!this._invokeNextQueue) this._invokeNextQueue = SC.ObserverSet.create();
271     this._invokeNextQueue.add(target, method);
272     return this ;
273   },
274 
275   /**
276     Executes any pending events at the end of the run loop.  This method is
277     called automatically at the end of a run loop to flush any pending
278     queue changes.
279 
280     The default method will invoke any one time methods and then sync any
281     bindings that might have changed.  You can override this method in a
282     subclass if you like to handle additional cleanup.
283 
284     This method must return YES if it found any items pending in its queues
285     to take action on.  endRunLoop will invoke this method repeatedly until
286     the method returns NO.  This way if any if your final executing code
287     causes additional queues to trigger, then can be flushed again.
288 
289     @returns {Boolean} YES if items were found in any queue, NO otherwise
290   */
291   flushApplicationQueues: function() {
292     var hadContent = NO,
293         // execute any methods in the invokeQueue.
294         queue = this._invokeQueue;
295     if ( queue && queue.getMembers().length ) {
296       this._invokeQueue = null; // reset so that a new queue will be created
297       hadContent = YES ; // needs to execute again
298       queue.invokeMethods();
299     }
300 
301     // flush any pending changed bindings.  This could actually trigger a
302     // lot of code to execute.
303     return SC.Binding.flushPendingChanges() || hadContent ;
304   },
305 
306   _flushinvokeLastQueue: function() {
307     var queue = this._invokeLastQueue, hadContent = NO ;
308     if (queue && queue.getMembers().length ) {
309       this._invokeLastQueue = null; // reset queue.
310       hadContent = YES; // has targets!
311       if (hadContent) queue.invokeMethods();
312     }
313     return hadContent ;
314   },
315 
316   _flushinvokeNextQueue: function() {
317     var queue = this._invokeNextQueue, hadContent = NO ;
318     if (queue && queue.getMembers().length ) {
319       this._invokeNextQueue = null; // reset queue.
320       hadContent = YES; // has targets!
321       if (hadContent) queue.invokeMethods();
322     }
323     return hadContent ;
324   }
325 
326 });

329 //@if(debug)
330  /**
331   Will return the recent stack as a hash with numerical keys, for nice output
332   in some browsers’ debuggers.  The “recent” stack is capped at 6 entries.
333 
334   This is used by, amongst other places, SC.LOG_DEFERRED_CALLS.
335 
336   @returns {Hash}
337 */
338 SC._getRecentStack = function() {
339   var currentFunction = arguments.callee.caller,
340       i               = 0,
341       stack           = {},
342       first           = YES,
343       functionName;
344 
345   while (currentFunction  &&  i < 10) {
346     // Skip ourselves!
347     if (first) {
348       first = NO;
349     }
350     else {
351       functionName = currentFunction.displayName || currentFunction.toString().substring(0, 40);
352       stack[i++]   = functionName;
353     }
354     currentFunction = currentFunction.caller;
355   }
356 
357   return stack;
358 };

363 /**
364   The current run loop.  This is created automatically the first time you
365   call begin().
366 
367   @type SC.RunLoop
368 */
369 SC.RunLoop.currentRunLoop = null;
370 
371 /**
372   The default RunLoop class.  If you choose to extend the RunLoop, you can
373   set this property to make sure your class is used instead.
374 
375   @type Class
376 */
377 SC.RunLoop.runLoopClass = SC.RunLoop;
378 
379 /**
380   Begins a new run loop on the currentRunLoop.  If you are already in a
381   runloop, this method has no effect.
382 
383   @returns {SC.RunLoop} receiver
384 */
385 SC.RunLoop.begin = function() {
386   var runLoop = this.currentRunLoop;
387   if (!runLoop) runLoop = this.currentRunLoop = this.runLoopClass.create();
388   runLoop.beginRunLoop();
389   return this ;
390 };
391 
392 /**
393   Ends the run loop on the currentRunLoop.  This will deliver any final
394   pending notifications and schedule any additional necessary cleanup.
395 
396   @returns {SC.RunLoop} receiver
397 */
398 SC.RunLoop.end = function() {
399   var runLoop = this.currentRunLoop;
400   if (!runLoop) {
401     throw "SC.RunLoop.end() called outside of a runloop!";
402   }
403   runLoop.endRunLoop();
404   return this ;
405 } ;

408 /**
409   Call this to kill the current run loop--stopping all propagation of bindings
410   and observers and clearing all timers.
411 
412   This is useful if you are popping up an error catcher: you need a run loop
413   for the error catcher, but you don't want the app itself to continue
414   running.
415 */
416 SC.RunLoop.kill = function() {
417   this.currentRunLoop = this.runLoopClass.create();
418   return this;
419 };

422   Returns YES when a run loop is in progress
423 
424   @return {Boolean}

426 SC.RunLoop.isRunLoopInProgress = function() {
427   if(this.currentRunLoop) return this.currentRunLoop.get('isRunLoopInProgress');
428   return NO;

431 /**
432   Executes a passed function in the context of a run loop. If called outside a
433   runloop, starts and ends one. If called inside an existing runloop, is
434   simply executes the function unless you force it to create a nested runloop.
435 
436   If an exception is thrown during execution, we give an error catcher the
437   opportunity to handle it before allowing the exception to bubble again.
438 
439   @param {Function} callback callback to execute
440   @param {Object} target context for callback
441   @param {Boolean} if YES, starts/ends a new runloop even if one is already running
442 */
443 SC.run = function(callback, target, forceNested) {
444   var alreadyRunning = SC.RunLoop.isRunLoopInProgress();
445 
446   // Only use a try/catch block if we have an ExceptionHandler
447   // since in some browsers try/catch causes a loss of the backtrace
448   if (SC.ExceptionHandler && SC.ExceptionHandler.enabled) {
449     try {
450       if(forceNested || !alreadyRunning) SC.RunLoop.begin();
451       if (callback) callback.call(target);
452       if(forceNested || !alreadyRunning) SC.RunLoop.end();
453     } catch (e) {
454       var handled = SC.ExceptionHandler.handleException(e);
455 
456       // If the exception was not handled, throw it again so the browser
457       // can deal with it (and potentially use it for debugging).
458       // (We don't throw it in IE because the user will see two errors)
459       if (!handled && !SC.browser.isIE) {
460         throw e;
461       }
462     }
463   } else {
464     if(forceNested || !alreadyRunning) SC.RunLoop.begin();
465     if (callback) callback.call(target);
466     if(forceNested || !alreadyRunning) SC.RunLoop.end();
467   }
468 };

470 /**
471   Wraps the passed function in code that ensures a run loop will
472   surround it when run.
473 */
474 SC.RunLoop.wrapFunction = function(func) {
475   var ret = function() {
476     var alreadyRunning = SC.RunLoop.isRunLoopInProgress();
477     if(!alreadyRunning) SC.RunLoop.begin();
478     var ret = arguments.callee.wrapped.apply(this,arguments);
479     if(!alreadyRunning) SC.RunLoop.end();
480     return ret;
481   };
482   ret.wrapped = func;
483   return ret;
484 };
