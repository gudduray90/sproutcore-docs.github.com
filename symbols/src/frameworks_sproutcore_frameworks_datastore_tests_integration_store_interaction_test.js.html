  6 /*global module, test, equals, same */
  7 
  8 
  9 var store, Employee, Engineer, colin;
 10 
 11 /** Test the interaction of a polymorphic record with the store. */
 12 module("SC.Record:Interaction with SC.Store", {
 13 
 14   setup: function() {
 15     SC.run(function () {
 16       store = SC.Store.create();
 17 
 18       Employee = SC.Record.extend({
 19         name: SC.Record.attr(String)
 20       });
 21 
 22       Employee.isPolymorphic = true;
 23 
 24       Engineer = Employee.extend({
 25         isEngineer: YES
 26       });
 27 
 28       colin = store.createRecord(Engineer, { guid: 1, name: 'Colin' });
 29     });
 30   },
 31 
 32   teardown: function() {
 33     SC.run(function () {
 34       store.destroy();
 35 
 36       store = Employee = Engineer = colin = null;
 37     });
 38   }
 39 
 40 });
 41 
 42 test("store.replaceIdFor() should update the storeKeysById() object on the record class", function() {
 43   var storeKey = colin.get('storeKey');
 44 
 45   equals(Engineer.storeKeysById()['1'], storeKey, "The storeKey should be the same.");
 46   equals(Engineer.storeKeysById()['1'], Employee.storeKeysById()['1'], "The storeKey should match at any requested polymorphic level.");
 47   same(Engineer.storeKeysById(), Employee.storeKeysById(), "The storeKey to id mapping is actually the same instance between each polymorphic level.");
 48 
 49   SC.Store.replaceIdFor(storeKey, 2);
 50 
 51   equals(Engineer.storeKeysById()['2'], storeKey, "The storeKey should still be the same after replacing the id.");
 52   equals(Engineer.storeKeysById()['2'], Employee.storeKeysById()['2'], "The storeKey should still match at any requested polymorphic level after replacing the id.");
 53   same(Engineer.storeKeysById(), Employee.storeKeysById(), "The storeKey to id mapping is actually the same instance between each polymorphic level.");
 54 });
