 10 /**
 11   @class
 12 
 13   Represents a state within a statechart. 
 14   
 15   The statechart actively manages all states belonging to it. When a state is created, 
 16   it immediately registers itself with it parent states. 
 17   
 18   You do not create an instance of a state itself. The statechart manager will go through its 
 19   state heirarchy and create the states itself.
 20 
 21   For more information on using statecharts, see SC.StatechartManager.
 22

 24   @extends SC.Object

201   /**
202     Used to initialize this state. To only be called by the owning statechart.
203   */
204   initState: function() {
205     if (this.get('stateIsInitialized')) return;
206     
207     this._registerWithParentStates();
208     
209     var key = null, 
210         value = null,
211         state = null,
212         substates = [],
213         matchedInitialSubstate = NO,
214         initialSubstate = this.get('initialSubstate'),
215         substatesAreConcurrent = this.get('substatesAreConcurrent'),
216         statechart = this.get('statechart'),
217         i = 0,
218         len = 0,
219         valueIsFunc = NO,
220         historyState = null;
221             
222     if (SC.kindOf(initialSubstate, SC.HistoryState) && initialSubstate.isClass) {
223       historyState = this.createHistoryState(initialSubstate, { parentState: this, statechart: statechart });
224       this.set('initialSubstate', historyState);
225       
226       if (SC.none(historyState.get('defaultState'))) {
227         this.stateLogError("Initial substate is invalid. History state requires the name of a default state to be set");
228         this.set('initialSubstate', null);
229         historyState = null;
230       }
231     }
232     
233     // Iterate through all this state's substates, if any, create them, and then initialize
234     // them. This causes a recursive process.
235     for (key in this) {
236       value = this[key];
237       valueIsFunc = SC.typeOf(value) === SC.T_FUNCTION;
238       
239       if (valueIsFunc && value.isEventHandler) {
240         this._registerEventHandler(key, value);
241         continue;
242       }
243       
244       if (valueIsFunc && value.isStateObserveHandler) {
245         this._registerStateObserveHandler(key, value);
246         continue;
247       }
248       
249       if (valueIsFunc && value.statePlugin) {
250         value = value.apply(this);
251       }
252       
253       if (SC.kindOf(value, SC.State) && value.isClass && this[key] !== this.constructor) {
254         state = this.createSubstate(value, { name: key, parentState: this, statechart: statechart });
255         substates.push(state);
256         this[key] = state;
257         state.initState();
258         if (key === initialSubstate) {
259           this.set('initialSubstate', state);
260           matchedInitialSubstate = YES;
261         } else if (historyState && historyState.get('defaultState') === key) {
262           historyState.set('defaultState', state);
263           matchedInitialSubstate = YES;
264         }
265       }
266     }
267     
268     if (!SC.none(initialSubstate) && !matchedInitialSubstate) {
269       this.stateLogError("Unable to set initial substate %@ since it did not match any of state's %@ substates".fmt(initialSubstate, this));
270     }
271     
272     if (substates.length === 0) {
273       if (!SC.none(initialSubstate)) {
274         this.stateLogWarning("Unable to make %@ an initial substate since state %@ has no substates".fmt(initialSubstate, this));
275       }
276     } 
277     else if (substates.length > 0) {
278       if (SC.none(initialSubstate) && !substatesAreConcurrent) {
279         state = this.createEmptyState({ parentState: this, statechart: statechart });
280         this.set('initialSubstate', state);
281         substates.push(state);
282         this[state.get('name')] = state;
283         state.initState();
284         this.stateLogWarning("state %@ has no initial substate defined. Will default to using an empty state as initial substate".fmt(this));
285       } 
286       else if (!SC.none(initialSubstate) && substatesAreConcurrent) {
287         this.set('initialSubstate', null);
288         this.stateLogWarning("Can not use %@ as initial substate since substates are all concurrent for state %@".fmt(initialSubstate, this));
289       }
290     }
291     
292     this.set('substates', substates);
293     this.set('currentSubstates', []);
294     this.set('enteredSubstates', []);
295     this.set('stateIsInitialized', YES);
296   },

319   /** @private 
320   
321     Registers event handlers with this state. Event handlers are special
322     functions on the state that are intended to handle more than one event. This
323     compared to basic functions that only respond to a single event that reflects
324     the name of the method.
325   */
326   _registerEventHandler: function(name, handler) {
327     var events = handler.events,
328         event = null,
329         len = events.length,
330         i = 0;
331         
332     this._registeredEventHandlers[name] = handler;
333     
334     for (; i < len; i += 1) {
335       event = events[i];
336       
337       if (SC.typeOf(event) === SC.T_STRING) {
338         this._registeredStringEventHandlers[event] = {
339           name: name,
340           handler: handler
341         };
342         continue;
343       }
344       
345       if (event instanceof RegExp) {
346         this._registeredRegExpEventHandlers.push({
347           name: name,
348           handler: handler,
349           regexp: event
350         });
351         continue;
352       }
353       
354       this.stateLogError("Invalid event %@ for event handler %@ in state %@".fmt(event, name, this));
355     }
356   },

358 359   /** @private 
360   
361     Registers state observe handlers with this state. State observe handlers behave just like
362     when you apply observes() on a method but will only be active when the state is currently 
363     entered, otherwise the handlers are inactive until the next time the state is entered
364   */
365   _registerStateObserveHandler: function(name, handler) {
366     var i = 0, 
367         args = handler.args, 
368         len = args.length, 
369         arg, validHandlers = YES;
370     
371     for (; i < len; i += 1) {
372       arg = args[i];
373       if (SC.typeOf(arg) !== SC.T_STRING || SC.empty(arg)) { 
374         this.stateLogError("Invalid argument %@ for state observe handler %@ in state %@".fmt(arg, name, this));
375         validHandlers = NO;
376       }
377     }
378     
379     if (!validHandlers) return;
380     
381     this._registeredStateObserveHandlers[name] = handler.args;
382   },

384   /** @private
385     Will traverse up through this state's parent states to register
386     this state with them.
387   */
388   _registerWithParentStates: function() {
389     this._registerSubstate(this);
390     var parent = this.get('parentState');
391     while (!SC.none(parent)) {
392       parent._registerSubstate(this);
393       parent = parent.get('parentState');
394     }

397   /** @private
398     Will register a given state as a substate of this state
399   */
400   _registerSubstate: function(state) {
401     var path = state.pathRelativeTo(this);
402     if (SC.none(path)) return; 
403     
404     // Create special private member variables to help
405     // keep track of substates and access them.
406     if (SC.none(this._registeredSubstatePaths)) {
407       this._registeredSubstatePaths = {};
408       this._registeredSubstates = [];
409     }
410     
411     this._registeredSubstates.push(state);
412     
413     // Keep track of states based on their relative path
414     // to this state. 
415     var regPaths = this._registeredSubstatePaths;
416     if (regPaths[state.get('name')] === undefined) {
417       regPaths[state.get('name')] = { __ki_paths__: [] };
418     }
419     
420     var paths = regPaths[state.get('name')];
421     paths[path] = state;
422     paths.__ki_paths__.push(path);
423   },
424   
425   /**
426     Will generate path for a given state that is relative to this state. It is
427     required that the given state is a substate of this state.
428     
429     If the heirarchy of the given state to this state is the following:
430     A > B > C, where A is this state and C is the given state, then the 
431     relative path generated will be "B.C"
432   */
433   pathRelativeTo: function(state) {
434     var path = this.get('name'),
435         parent = this.get('parentState');
436     
437     while (!SC.none(parent) && parent !== state) {
438       path = "%@.%@".fmt(parent.get('name'), path);
439       parent = parent.get('parentState');
440     }
441     
442     if (parent !== state && state !== this) {
443       this.stateLogError('Can not generate relative path from %@ since it not a parent state of %@'.fmt(state, this));
444       return null;
445     }
446     
447     return path;
448   },

450   /**
451     Used to get a substate of this state that matches a given value. 
452     
453     If the value is a state object, then the value will be returned if it is indeed 
454     a substate of this state, otherwise null is returned. 
455     
456     If the given value is a string, then the string is assumed to be a path to a substate. 
457     The value is then parsed to find the closes match. If there is no match then null 
458     is returned. If there is more than one match then null is return and an error 
459     is generated indicating ambiguity of the given value. 
460     
461     Note that when the value is a string, it is assumed to be a path relative to this 
462     state; not the root state of the statechart.
463   */
464   getSubstate: function(value) {
465     var valueType = SC.typeOf(value);
466     
467     // If the value is an object then just check if the value is 
468     // a registered substate of this state, and if so return it. 
469     if (valueType === SC.T_OBJECT) {
470       return this._registeredSubstates.indexOf(value) > -1 ? value : null;
471     }
472     
473     if (valueType !== SC.T_STRING) {
474       this.stateLogError("Can not find matching subtype. value must be an object or string: %@".fmt(value));
475       return null;
476     }
477     
478     // The value is a string. Therefore treat the value as a relative path to 
479     // a substate of this state.
480     
481     // Extract that last part of the string. Ex. 'foo' => 'foo', 'foo.bar' => 'bar'
482     var matches = value.match(/(^|\.)(\w+)$/);
483     if (!matches) return null;
484 
485     // Get all the paths related to the matched value. If no paths then return null.
486     var paths = this._registeredSubstatePaths[matches[2]];
487     if (SC.none(paths)) return null;
488     
489     // Do a quick check to see if there is a path that exactly matches the given
490     // value, and if so return the corresponding state
491     var state = paths[value];
492     if (!SC.none(state)) return state;
493     
494     // No exact match found. If the value given is a basic string with no ".", then check
495     // if there is only one path containing that string. If so, return it. If there is
496     // more than one path then it is ambiguous as to what state is trying to be reached.
497     if (matches[1] === "") {
498       if (paths.__ki_paths__.length === 1) return paths[paths.__ki_paths__[0]];
499       if (paths.__ki_paths__.length > 1) {
500         var msg = 'Can not find substate matching %@ in state %@. Ambiguous with the following: %@';
501         this.stateLogError(msg.fmt(value, this, paths.__ki_paths__));
502       }
503     } 
504     
505     return null;
506   },

508   /**
509     Used to go to a state in the statechart either directly from this state if it is a current state,
510     or from one of this state's current substates.
511     
512     Note that if the value given is a string, it will be assumed to be a path to a state. The path
513     will be relative to the statechart's root state; not relative to this state.
514     
515     Method can be called in the following ways: 
516     
517         // With one argument
518         gotoState(<state>)
519       
520         // With two arguments
521         gotoState(<state>, <hash>)
522     
523     Where <state> is either a string or a SC.State object and <hash> is a regular JS hash object.
524     
525     @param state {SC.State|String} the state to go to
526     @param context {Hash} Optional. context object that will be supplied to all states that are
527            exited and entered during the state transition process
528   */
529   gotoState: function(state, context) {
530     var fromState = null;
531     
532     if (this.get('isCurrentState')) {
533       fromState = this;
534     } else if (this.get('hasCurrentSubstates')) {
535       fromState = this.get('currentSubstates')[0];
536     }
537     
538     this.get('statechart').gotoState(state, fromState, context);
539   },

687   
688   /**
689     Called by the statechart to allow a state to try and handle the given event. If the
690     event is handled by the state then YES is returned, otherwise NO.
691     
692     There is a particular order in how an event is handled by a state:
693     
694      1. Basic function whose name matches the event
695      2. Registered event handler that is associated with an event represented as a string
696      3. Registered event handler that is associated with events matching a regular expression
697      4. The unknownEvent function
698       
699     Use of event handlers that are associated with events matching a regular expression may
700     incur a performance hit, so they should be used sparingly.
701     
702     The unknownEvent function is only invoked if the state has it, otherwise it is skipped. Note that
703     you should be careful when using unknownEvent since it can be either abused or cause unexpected
704     behavior.
705     
706     Example of a state using all four event handling techniques:
707     
708         SC.State.extend({
709       
710           // Basic function handling event 'foo'
711           foo: function(arg1, arg2) { ... },
712         
713           // event handler that handles 'frozen' and 'canuck'
714           eventHandlerA: function(event, arg1, arg2) {
715             ...
716           }.handleEvent('frozen', 'canuck'),
717         
718           // event handler that handles events matching the regular expression /num\d/
719           //   ex. num1, num2
720           eventHandlerB: function(event, arg1, arg2) {
721             ...
722           }.handleEvent(/num\d/),
723         
724           // Handle any event that was not handled by some other
725           // method on the state
726           unknownEvent: function(event, arg1, arg2) {
727         
728           }
729       
730         });
731   */
732   tryToHandleEvent: function(event, arg1, arg2) {
733 
734     var trace = this.get('trace');
735 
736     // First check if the name of the event is the same as a registered event handler. If so,
737     // then do not handle the event.
738     if (this._registeredEventHandlers[event]) {
739       this.stateLogWarning("state %@ can not handle event %@ since it is a registered event handler".fmt(this, event));
740       return NO;
741     }    
742     
743     if (this._registeredStateObserveHandlers[event]) {
744       this.stateLogWarning("state %@ can not handle event %@ since it is a registered state observe handler".fmt(this, event));
745       return NO;
746     }
747     
748     // Now begin by trying a basic method on the state to respond to the event
749     if (SC.typeOf(this[event]) === SC.T_FUNCTION) {
750       if (trace) this.stateLogTrace("will handle event %@".fmt(event));
751       return (this[event](arg1, arg2) !== NO);
752     }
753     
754     // Try an event handler that is associated with an event represented as a string
755     var handler = this._registeredStringEventHandlers[event];
756     if (handler) {
757       if (trace) this.stateLogTrace("%@ will handle event %@".fmt(handler.name, event));
758       return (handler.handler.call(this, event, arg1, arg2) !== NO);
759     }
760     
761     // Try an event handler that is associated with events matching a regular expression
762     
763     var len = this._registeredRegExpEventHandlers.length,
764         i = 0;
765         
766     for (; i < len; i += 1) {
767       handler = this._registeredRegExpEventHandlers[i];
768       if (event.match(handler.regexp)) {
769         if (trace) this.stateLogTrace("%@ will handle event %@".fmt(handler.name, event));
770         return (handler.handler.call(this, event, arg1, arg2) !== NO);
771       }
772     }
773     
774     // Final attempt. If the state has an unknownEvent function then invoke it to 
775     // handle the event
776     if (SC.typeOf(this['unknownEvent']) === SC.T_FUNCTION) {
777       if (trace) this.stateLogTrace("unknownEvent will handle event %@".fmt(event));
778       return (this.unknownEvent(event, arg1, arg2) !== NO);
779     }
780     
781     // Nothing was able to handle the given event for this state
782     return NO;
783   },

898   /** @private 
899   
900     Primary method used to either add or remove this state as an observer
901     based on all the state observe handlers that have been registered with
902     this state.
903     
904     Note: The code to add and remove the state as an observer has been
905     taken from the observerable mixin and made slightly more generic. However,
906     having this code in two different places is not ideal, but for now this
907     will have to do. In the future the code should be refactored so that
908     there is one common function that both the observerable mixin and the 
909     statechart framework use.  
910   */
911   _configureAllStateObserveHandlers: function(action) {
912     var key, values, value, dotIndex, path, observer, i, root;
913 
914     for (key in this._registeredStateObserveHandlers) {
915       values = this._registeredStateObserveHandlers[key];
916       for (i = 0; i < values.length; i += 1) {
917         path = values[i]; observer = key;
918   
919         // Use the dot index in the path to determine how the state
920         // should add itself as an observer.
921   
922         dotIndex = path.indexOf('.');
923 
924         if (dotIndex < 0) {
925           this[action](path, this, observer);
926         } else if (path.indexOf('*') === 0) {
927           this[action](path.slice(1), this, observer);
928         } else {
929           root = null;
930 
931           if (dotIndex === 0) {
932             root = this; path = path.slice(1);
933           } else if (dotIndex === 4 && path.slice(0, 5) === 'this.') {
934             root = this; path = path.slice(5);
935           } else if (dotIndex < 0 && path.length === 4 && path === 'this') {
936             root = this; path = '';
937           }
938 
939           SC.Observers[action](path, this, observer, root);
940         }
941       }
942     }
943   },

1043 /**
1044   Use this when you want to plug-in a state into a statechart. This is beneficial
1045   in cases where you split your statechart's states up into multiple files and
1046   don't want to fuss with the sc_require construct.
1047   
1048   Example:
1049   
1050       MyApp.statechart = SC.Statechart.create({
1051         rootState: SC.State.design({
1052           initialSubstate: 'a',
1053           a: SC.State.plugin('path.to.a.state.class'),
1054           b: SC.State.plugin('path.to.another.state.class')
1055         })
1056       });
1057     
1058   You can also supply hashes the plugin feature in order to enhance a state or
1059   implement required functionality:
1060 
1061       SomeMixin = { ... };
1062 
1063       stateA: SC.State.plugin('path.to.state', SomeMixin, { ... })
1064 
1065   @param value {String} property path to a state class
1066   @param args {Hash,...} Optional. Hash objects to be added to the created state
1067 */
1068 SC.State.plugin = function(value) {
1069   var args = SC.A(arguments); args.shift();
1070   var func = function() {
1071     var klass = SC.objectForPropertyPath(value);
1072     if (!klass) {
1073       console.error('SC.State.plugin: Unable to determine path %@'.fmt(value));
1074       return undefined;
1075     }
1076     if (!klass.isClass || !klass.kindOf(SC.State)) {
1077       console.error('SC.State.plugin: Unable to extend. %@ must be a class extending from SC.State'.fmt(value));
1078       return undefined;
1079     }
1080     return klass.extend.apply(klass, args);
1081   };
1082   func.statePlugin = YES;
1083   return func;
1084 };
