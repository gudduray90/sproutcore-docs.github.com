  8 // Extensions to the core SC.Object class
  9 SC.mixin(SC.Object.prototype, /** @scope SC.Object.prototype */ {
 10 
 11   /**
 12     Invokes the named method after the specified period of time.
 13 
 14     This is a convenience method that will create a single run timer to
 15     invoke a method after a period of time.  The method should have the
 16     signature:
 17 
 18         methodName: function(timer)
 19 
 20     If you would prefer to pass your own parameters instead, you can instead
 21     call invokeLater() directly on the function object itself.
 22 
 23     @param methodName {String} method name to perform.
 24     @param interval {Number} period from current time to schedule.
 25     @returns {SC.Timer} scheduled timer.
 26   */
 27   invokeLater: function(methodName, interval) {
 28     if (interval === undefined) { interval = 1 ; }
 29     var f = methodName, args, func;
 30 
 31     // if extra arguments were passed - build a function binding.
 32     if (arguments.length > 2) {
 33       args = SC.$A(arguments).slice(2);
 34       if (SC.typeOf(f) === SC.T_STRING) { f = this[methodName] ; }
 35       func = f ;
 36       f = function() { return func.apply(this, args); } ;
 37     }
 38 
 39     // schedule the timer
 40     return SC.Timer.schedule({ target: this, action: f, interval: interval });
 41   },
 42   
 43   /**
 44     A convenience method which makes it easy to coalesce invocations to ensure 
 45     that the method is only called once. This is useful if you need to schedule 
 46     a call but only want it to trigger once after some defined interval has 
 47     passed.
 48     
 49     @param {Function|String} method reference or method name
 50     @param {Number} interval
 51   */
 52   invokeOnceLater: function(method, interval) {
 53     if (interval === undefined) { interval = 1 ; }
 54 
 55     var timers = this._sc_invokeOnceLaterTimers,
 56         methodGuid, existingTimer, f, newTimer;
 57 
 58     // ensure we always deal with real functions
 59     if (SC.typeOf(method) === SC.T_STRING) {
 60       method = this[method];
 61     }
 62     
 63     methodGuid = SC.guidFor(method);
 64     
 65     if(!timers) {
 66       this._sc_invokeOnceLaterTimers = timers = {};
 67     }
 68     
 69     existingTimer = timers[methodGuid];
 70     if(existingTimer) existingTimer.invalidate();
 71     
 72     f = function() {
 73       // GC assistance for IE
 74       delete timers[methodGuid];
 75       return method.apply(this, arguments);
 76     };
 77     
 78     // schedule the timer
 79     newTimer = SC.Timer.schedule({ target: this, action: f, interval: interval });
 80     timers[methodGuid] = newTimer;
 81     
 82     return newTimer;
 83   },
 84 
 85   /**
 86     Lookup the named property path and then invoke the passed function,
 87     passing the resulting value to the function.
 88 
 89     This method is a useful way to handle deferred loading of properties.
 90     If you want to defer loading a property, you can override this method.
 91     When the method is called, passing a deferred property, you can load the
 92     property before invoking the callback method.
 93 
 94     You can even swap out the receiver object.
 95 
 96     The callback method should have the signature:
 97 
 98     function callback(objectAtPath, sourceObject) { ... }
 99 
100     You may pass either a function itself or a target/method pair.
101 
102     @param {String} pathName
103     @param {Object} target target or method
104     @param {Function|String} method
105     @returns {SC.Object} receiver
106   */
107   invokeWith: function(pathName, target, method) {
108     // normalize target/method
109     if (method === undefined) {
110       method = target; target = this;
111     }
112     if (!target) { target = this ; }
113     if (SC.typeOf(method) === SC.T_STRING) { method = target[method]; }
114 
115     // get value
116     var v = this.getPath(pathName);
117 
118     // invoke method
119     method.call(target, v, this);
120     return this ;
121   }
122 
123 });
