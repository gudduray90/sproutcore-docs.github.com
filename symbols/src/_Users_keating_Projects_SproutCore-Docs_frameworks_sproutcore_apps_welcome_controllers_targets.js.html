  5 /*globals CoreTools Welcome */
  6 
  7 /**
  8 
  9   Manages the list of targets
 10 
 11   @extends SC.ArrayController
 12 */
 13 Welcome.targetsController = SC.ArrayController.create(
 14 /** @scope Welcome.targetsController.prototype */ {
 15 
 16   /**
 17     Call this method whenever you want to reload the targets from the server.
 18   */
 19   reload: function() {
 20     var targets = Welcome.store.find(CoreTools.TARGETS_QUERY);
 21     this.set('content', targets);
 22   },
 23 
 24   appsOnly: function() {
 25     return this.filter(function(t) {
 26       return t.get('kind') === 'app' && !t.get('name').match(/sproutcore\/(welcome|experimental)/);
 27     });
 28   }.property('[]').cacheable(),
 29 
 30   loadApplication: function() {
 31     var app = this.get('selection').firstObject(),
 32         url = app ? app.get('appUrl') : null;
 33 
 34     if (url) {
 35       this.set('canLoadApp', NO);
 36       this.invokeLater(function() {
 37         window.location.href = url; // load new app
 38       });
 39     }
 40   },
 41 
 42   launchEnabled: function() {
 43     var canLoadApp = this.get('canLoadApp'),
 44         selection = this.get('selection'),
 45         selectedObject = selection.firstObject();
 46     return canLoadApp && selectedObject && selectedObject.get('name') !== '/sproutcore';
 47   }.property('canLoadApp', 'selection').cacheable(),
 48 
 49   // used to disable all controls
 50   canLoadApp: YES,
 51 
 52   allowsEmptySelection: NO,
 53   allowsMultipleSelection: NO
 54 
 55 }) ;
