  2 /**
  3   If true, then all SC.Controls can be focused when the
  4   user presses the tab key. Otherwise, only TextFieldViews

  8   @constant

 18     // @if (debug)
 19     SC.Logger.warn("SC.SAFARI_FOCUS_BEHAVIOR is deprecated. Please use SC.FOCUS_ALL_CONTROLS instead");
 20     // @endif
 21     SC.FOCUS_ALL_CONTROLS = SC.SAFARI_FOCUS_BEHAVIOR;
 22   }
 23 });
