  4 /*globals SC */
  5 
  6 var statechart = null;
  7 var monitor, root, stateA, stateB, stateC, stateD, stateE, stateF;
  8 
  9 module("SC.Statechart: With Concurrent States - Goto State Basic Tests", {
 10   setup: function() {
 11     
 12     statechart = SC.Statechart.create({
 13       
 14       monitorIsActive: YES,
 15       
 16       rootState: SC.State.design({
 17         
 18         substatesAreConcurrent: YES,
 19 
 20         a: SC.State.design({
 21           initialSubstate: 'c',
 22           c: SC.State.design(),
 23           d: SC.State.design()
 24         }),
 25 
 26         b: SC.State.design({
 27           initialSubstate: 'e',
 28           e: SC.State.design(),
 29           f: SC.State.design()
 30         })
 31       })
 32       
 33     });
 34     
 35     statechart.initStatechart();
 36     
 37     monitor = statechart.get('monitor');
 38     root = statechart.get('rootState');
 39     stateA = statechart.getState('a');
 40     stateB = statechart.getState('b');
 41     stateC = statechart.getState('c');
 42     stateD = statechart.getState('d');
 43     stateE = statechart.getState('e');
 44     stateF = statechart.getState('f');
 45   },
 46   
 47   teardown: function() {
 48     statechart.destroy();
 49     statechart = monitor = root = stateA = stateB = stateC = stateD = stateE = stateF = null;
 50   }
 51 });
 52 
 53 test("check statechart initialization", function() {
 54   equals(monitor.get('length'), 5, 'initial state sequence should be of length 5');
 55   equals(monitor.matchSequence().begin()
 56                                   .entered(root)
 57                                   .beginConcurrent()
 58                                     .beginSequence()
 59                                       .entered('a', 'c')
 60                                     .endSequence()
 61                                     .beginSequence()
 62                                       .entered('b', 'e')
 63                                     .endSequence()
 64                                   .endConcurrent()
 65                                 .end(), 
 66     true, 'initial sequence should be entered[ROOT, a, c, b, e]');
 67   equals(monitor.matchSequence().begin()
 68                                   .entered(root)
 69                                   .beginConcurrent()
 70                                     .entered('a', 'b')
 71                                   .endConcurrent()
 72                                   .beginConcurrent()
 73                                     .entered('c', 'e')
 74                                   .endConcurrent()
 75                                 .end(), 
 76     false, 'initial sequence should not be entered[ROOT, a, b, c, e]');
 77   
 78   equals(statechart.get('currentStateCount'), 2, 'current state count should be 2');
 79   
 80   equals(statechart.stateIsCurrentState('c'), true, 'current state should be c');
 81   equals(statechart.stateIsCurrentState('e'), true, 'current state should be e');
 82   equals(statechart.stateIsCurrentState('d'), false, 'current state should not be d');
 83   equals(statechart.stateIsCurrentState('f'), false, 'current state should not be f');
 84   
 85   equals(stateA.stateIsCurrentSubstate('c'), true, 'state a\'s current substate should be state c');
 86   equals(stateA.stateIsCurrentSubstate('d'), false, 'state a\'s current substate should not be state d');
 87   equals(stateB.stateIsCurrentSubstate('e'), true, 'state a\'s current substate should be state e');
 88   equals(stateB.stateIsCurrentSubstate('f'), false, 'state a\'s current substate should not be state f');
 89   
 90   equals(stateA.get('isCurrentState'), false, 'state a should not be current state');
 91   equals(stateB.get('isCurrentState'), false, 'state b should not be current state');
 92   equals(stateC.get('isCurrentState'), true, 'state c should be current state');
 93   equals(stateD.get('isCurrentState'), false, 'state d should not be current state');
 94   equals(stateE.get('isCurrentState'), true, 'state e should be current state');
 95   equals(stateF.get('isCurrentState'), false, 'state f should not be current state');
 96   
 97   ok(monitor.matchEnteredStates(root, 'a', 'c', 'b', 'e'), 'states root, A, C, B and E should all be entered');
 98 });
 99 
100 test("from state c, go to state d, and from state e, go to state f", function() {
101   monitor.reset();
102   
103   stateC.gotoState('d');
104   equals(monitor.get('length'), 2, 'state sequence should be of length 2');
105   equals(monitor.matchSequence().begin().exited('c').entered('d').end(), true, 'sequence should be exited[c], enterd[d]');
106   
107   monitor.reset();
108   
109   stateE.gotoState('f');
110   equals(monitor.get('length'), 2, 'state sequence should be of length 2');
111   equals(monitor.matchSequence().begin().exited('e').entered('f').end(), true, 'sequence should be exited[e], enterd[f]');
112   
113   equals(statechart.get('currentStateCount'), 2, 'current state count should be 2');
114   
115   equals(statechart.stateIsCurrentState('d'), true, 'current state should be d');
116   equals(statechart.stateIsCurrentState('f'), true, 'current state should be f');
117   
118   equals(stateA.stateIsCurrentSubstate('c'), false, 'state a\'s current substate should not be state c');
119   equals(stateA.stateIsCurrentSubstate('d'), true, 'state a\'s current substate should be state d');
120   equals(stateB.stateIsCurrentSubstate('e'), false, 'state b\'s current substate should not be state e');
121   equals(stateB.stateIsCurrentSubstate('f'), true, 'state b\'s current substate should be state f');
122   
123   equals(stateA.get('isCurrentState'), false, 'state a should not be current state');
124   equals(stateB.get('isCurrentState'), false, 'state b should not be current state');
125   equals(stateC.get('isCurrentState'), false, 'state c should not be current state');
126   equals(stateD.get('isCurrentState'), true, 'state d should be current state');
127   equals(stateE.get('isCurrentState'), false, 'state e should not be current state');
128   equals(stateF.get('isCurrentState'), true, 'state f should be current state');
129   
130   ok(monitor.matchEnteredStates(root, 'a', 'd', 'b', 'f'), 'states root, A, D, B and F should all be entered');
131 });
132 
133 test("from state a, go to sibling concurrent state b", function() {
134   monitor.reset();
135   
136 137   // Expect to get an error to be outputted in the JS console, which is what we want since
138   // the pivot state is the root state and it's substates are concurrent
139   console.log('expecting to get an error...');
140   stateA.gotoState('b');
141   
142   equals(monitor.get('length'), 0, 'state sequence should be of length 0');
143   equals(statechart.get('currentStateCount'), 2, 'current state count should be 2');
144   equals(statechart.stateIsCurrentState('c'), true, 'current state should be c');
145   equals(statechart.stateIsCurrentState('e'), true, 'current state should be e');
146   equals(stateA.stateIsCurrentSubstate('c'), true, 'state a\'s current substate should be state c');
147   equals(stateA.stateIsCurrentSubstate('d'), false, 'state a\'s current substate should not be state d');
148   equals(stateB.stateIsCurrentSubstate('e'), true, 'state a\'s current substate should be state e');
149   equals(stateB.stateIsCurrentSubstate('f'), false, 'state a\'s current substate should not be state f');
150   
151   ok(monitor.matchEnteredStates(root, 'a', 'c', 'b', 'e'), 'states root, A, C, B and E should all be entered');
152 });
