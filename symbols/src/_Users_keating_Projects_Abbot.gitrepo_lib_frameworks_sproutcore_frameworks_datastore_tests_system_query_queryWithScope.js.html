  8 // test parsing of query string
  9 var q, scope1, scope2;
 10 module("SC.Query#queryWithScope", {
 11   setup: function() {
 12     q = SC.Query.create({
 13       conditions: "foo = bar",
 14       parameters: { foo: "bar" },
 15       orderBy: "foo",
 16       recordType: SC.Record,
 17       recordTypes: [SC.Record],
 18       location: SC.Query.REMOTE
 19     }).freeze();
 20     
 21     scope1 = SC.CoreSet.create();
 22     scope2 = SC.CoreSet.create();
 23   },
 24   
 25   teardown: function() {
 26     q = scope1 = scope2 = null;
 27   }
 28 });
 29 
 30 function verifyCopy(copy, original) {
 31   var keys = 'conditions orderBy recordType recordTypes parameters location'.w();
 32   keys.forEach(function(key) {
 33     equals(copy.get(key), original.get(key), 'copy.%@ should equal original.%@'.fmt(key, key));
 34   });
 35 }
 36 
 37 test("getting into scope first time", function() {
 38   
 39   var q2 = q.queryWithScope(scope1);
 40   verifyCopy(q2, q);
 41   equals(q2.get('scope'), scope1, 'new query should have scope1');
 42   
 43   var q3 = q.queryWithScope(scope1);
 44   equals(q3, q2, 'calling again with same scope should return same instance');
 45 });
 46 
 47 test("chained scope", function() {
 48   var q2 = q.queryWithScope(scope1) ;
 49   var q3 = q2.queryWithScope(scope2);
 50   
 51   verifyCopy(q3, q2);
 52   equals(q3.get('scope'), scope2, 'new query should have scope2');
 53   
 54   var q4 = q2.queryWithScope(scope2);
 55   equals(q4, q3, 'calling again with same scope should return same instance');
 56 });
