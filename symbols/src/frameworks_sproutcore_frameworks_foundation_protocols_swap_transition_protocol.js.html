  8 /** @namespace
  9   This protocol defines the allowable transition plugin methods.
 10 
 11   SC.ContainerView uses transition plugins to setup, execute and cleanup the
 12   swapping between views and expects the given transition plugin object
 13   to implement the methods in this protocol.
 14 */
 15 SC.SwapTransitionProtocol = {
 16 
 17   /**
 18     This optional method is called to set up the entrance transition.
 19 
 20     Use this method to adjust the layout of the container and new content so
 21     that it may be properly animated.  For example, you may need to adjust the
 22     content from a flexible layout (i.e. { left: 0, top: 0, right: 0, bottom: 0 })
 23     to a fixed layout (i.e. { left: 0, top: 0, width: 100, height: 200 })
 24     so that it can be moved.
 25 
 26     @param {SC.ContainerView} container The SC.ContainerView using this plugin.
 27     @param {SC.View} content The new view added to the container.
 28     @param {SC.View} previousStatechart The current content statechart in the container.
 29     @param {Object} options Options to modify the transition.
 30   */
 31   willBuildInToView: function (container, content, previousStatechart, options) {},
 32 
 33   /**
 34     This optional method is called to transition the new content in.
 35 
 36     When the transition completes, the function must call the entered()
 37     method on the statechart.
 38 
 39     @param {SC.ContainerContentStatechart} statechart The statechart for the content view.
 40     @param {SC.ContainerView} container The SC.ContainerView using this plugin.
 41     @param {SC.View} content The new view added to the container.
 42     @param {SC.View} previousStatechart The current content statechart in the container.
 43     @param {Object} options Options to modify the transition.
 44   */
 45   buildInToView: function (statechart, container, content, previousStatechart, options) {},
 46 
 47   /**
 48     This optional method is called to cancel an active entrance transition.
 49 
 50     Use this method to stop the animation and immediately clean up the views.
 51 
 52     @param {SC.ContainerView} container The SC.ContainerView using this plugin.
 53     @param {SC.View} content The new view in the container, which is still transitioning in.
 54     @param {Object} options Options to modify the transition.
 55   */
 56   buildInDidCancel:  function (container, content, options) {},
 57 
 58   /**
 59     This optional method is called to clean up the entrance the transition.
 60 
 61     Use this method to adjust the layout of the container and new content after
 62     the transition completes.  For example, you may need to adjust the layout
 63     from a fixed layout (i.e. { left: 0, top: 0, width: 100, height: 200 })
 64     to a flexible layout (i.e. { left: 0, top: 0, right: 0, bottom: 0 }).
 65 
 66     @param {SC.ContainerView} container The SC.ContainerView using this plugin.
 67     @param {SC.View} content The new view added to the container.
 68     @param {Object} options Options that were used to modify the transition.
 69   */
 70   didBuildInToView: function (container, content, options) {},
 71 
 72   /**
 73     This optional method is called to set up the exit transition.
 74 
 75     Use this method to adjust the layout of the container and new content so
 76     that it may be properly animated.  For example, you may need to adjust the
 77     content from a flexible layout (i.e. { left: 0, top: 0, right: 0, bottom: 0 })
 78     to a fixed layout (i.e. { left: 0, top: 0, width: 100, height: 200 })
 79     so that it can be moved.
 80 
 81     @param {SC.ContainerView} container The SC.ContainerView using this plugin.
 82     @param {SC.View} content The old view being removed from the container.
 83     @param {Object} options Options to modify the transition.
 84   */
 85   willBuildOutFromView: function (container, content, options) {},
 86 
 87   /**
 88     This optional method is called to transition the old content out.
 89 
 90     When the transition completes, the function must call the exited()
 91     method on the statechart.
 92 
 93     Note that a view may be repeatedly built out before it has completed.  In
 94     order to accelerate the build out transition, the exitCount parameter will
 95     be incremented each time buildOutFromView is called.  The initial value
 96     will always be 1.
 97 
 98     @param {SC.ContainerContentStatechart} statechart The statechart for the content view.
 99     @param {SC.ContainerView} container The SC.ContainerView using this plugin.
100     @param {SC.View} content The old view being removed from the container.
101     @param {Object} options Options to modify the transition.
102     @param {Number} exitCount The number of times the content is being built out.
103   */
104   buildOutFromView: function (statechart, container, content, options, exitCount) {},
105 
106   /**
107     This optional method is called to cancel an active exit transition.
108 
109     Use this method to stop the animation and immediately clean up the views.
110 
111     @param {SC.ContainerView} container The SC.ContainerView using this plugin.
112     @param {SC.View} content The old view being removed from the container, which is still transitioning out.
113     @param {Object} options Options to modify the transition.
114   */
115   buildOutDidCancel: function (container, content, options) {},
116 
117   /**
118     This optional method is called to clean up the entrance the transition.
119 
120     Use this method to adjust the layout of the container and new content after
121     the transition completes.  For example, you may need to adjust the layout
122     from a fixed layout (i.e. { left: 0, top: 0, width: 100, height: 200 })
123     to a flexible layout (i.e. { left: 0, top: 0, right: 0, bottom: 0 }).
124 
125     @param {SC.ContainerView} container The SC.ContainerView using this plugin.
126     @param {SC.View} content The new view added to the container.
127     @param {Object} options Options that were used to modify the transition.
128   */
129   didBuildOutFromView: function (container, content, options) {},
130 
131   /**
132     This optional method is called to adjust the clippingFrame during the
133     transition.
134 
135     Because some childViews are altered by the clippingFrame of their parent
136     views (notably collection views), we may need to provide a modified
137     clipping frame while the transition is in process.
138 
139     For example, a push transition should double the regular clippingFrame
140     of the container to fit both the new and current content while the
141     transition is in progress.
142 
143     @param {SC.ContainerView} container The SC.ContainerView using this plugin.
144     @param {Object} clippingFrame The current clippingFrame of the container.
145     @param {Object} options Options used to modify the transition.
146     @returns clippingFrame
147   */
148   transitionClippingFrame: function (container, clippingFrame, options) {}
149 
150 };
