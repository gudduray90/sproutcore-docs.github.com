  8 /** @class
  9 
 10   Represents a Checkbox Button.
 11   
 12   The view is an `SC.ButtonView` put into toggle mode and with the 'theme' property
 13   set to "checkbox".
 14   
 15   Rendering
 16   ----------------------------
 17   SC.ButtonView delegates its rendering to its theme. As the theme is set
 18   to "checkbox", the way the checkbox renders (including DOM) will actually
 19   be different than SC.ButtonView's.
 20   
 21   @extends SC.ButtonView
 22   @since SproutCore 1.0
 23 */
 24 SC.CheckboxView = SC.ButtonView.extend(SC.StaticLayout,
 25 /** @scope SC.CheckboxView.prototype */ {
 26 
 27   /**
 28     @type Array
 29     @default ['sc-checkbox-view', 'sc-checkbox-control']
 30     @see SC.View#classNames
 31   */
 32   classNames: ['sc-checkbox-view', 'sc-checkbox-control'],
 33 
 34   /**
 35     The WAI-ARIA role of checkbox.
 36 
 37     @type String
 38     @readOnly
 39   */
 40   ariaRole: 'checkbox',
 41 
 42   // no special theme for Checkbox; button defaults to 'square', so we have to stop that.
 43   themeName: null,
 44   
 45   /**
 46     @type String
 47     @default 'checkboxRenderDelegate'
 48   */
 49   renderDelegateName: 'checkboxRenderDelegate',
 50 
 51   /**
 52     Ellipsis is disabled by default to allow multiline text
 53     
 54     @type Boolean
 55     @default NO
 56   */
 57   needsEllipsis: NO,
 58   
 59   /**
 60     `YES` if `isEnabled` is `YES`, `NO` otherwise
 61     
 62     @type Boolean
 63     @default NO
 64     @observes isEnabled
 65   */
 66   acceptsFirstResponder: function() {
 67     if (SC.FOCUS_ALL_CONTROLS) { return this.get('isEnabled'); }
 68     return NO;
 69   }.property('isEnabled'),
 70   
 71   /** @private */
 72   _toggleValue: function(){
 73     var isOn = this.get('value') === this.get('toggleOnValue');
 74     this.set('value', isOn ? this.get('toggleOffValue') : this.get('toggleOnValue'));
 75   },
 76   
 77   /** @private */
 78   mouseDown: function(evt) {
 79     if(!this.get('isEnabled')) return YES;
 80     this.set('isActive', YES);
 81     this._isMouseDown = YES;
 82     if (evt && this.get('acceptsFirstResponder')) evt.allowDefault();
 83     return YES;
 84   },
 85   
 86   /** @private */
 87   mouseUp: function(evt) {
 88     if(!this.get('isEnabled')) return YES;
 89 
 90     this.set('isActive', NO);
 91     this._isMouseDown = NO;
 92 
 93     // fire action
 94     if (this.get('buttonBehavior') !== SC.HOLD_BEHAVIOR) {
 95       if (this.$().within(evt.target)) {
 96         this._toggleValue();
 97         this._action(evt);
 98       }
 99     }
100 
101     return YES;
102 
103   },
104   
105   /** @private */
106   keyDown: function(evt) {
107     // handle tab key
108     if(!this.get('isEnabled')) return YES;
109 
110     if (evt.which === 9 || evt.keyCode === 9) {
111       var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
112       if(view) view.becomeFirstResponder();
113       else evt.allowDefault();
114       return YES ; // handled
115     }
116 
117     if (evt.which === 13 || evt.which === 32) {
118       this._toggleValue();
119 
120       // fire action
121       if (this.get('buttonBehavior') !== SC.HOLD_BEHAVIOR) {
122         if (this.$().within(evt.target)) { this._action(evt); }
123       }
124 
125       return YES ; // handled
126     }
127 
128     // let other keys through to browser
129     evt.allowDefault();
130 
131     return NO;
132   },
133   
134   
135   
136   /** @private */
137   touchStart: function(evt) {
138     return this.mouseDown(evt);
139   },
140   
141   /** @private */
142   touchEnd: function(evt) {
143     return this.mouseUp(evt);
144   }
145     
146 });
