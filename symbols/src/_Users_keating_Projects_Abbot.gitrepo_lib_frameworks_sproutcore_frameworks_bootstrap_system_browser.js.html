  8 window.SC = window.SC || { MODULE_INFO: {}, LAZY_INSTANTIATION: {} };
  9 
 10 SC._detectBrowser = function(userAgent, language) {
 11   var version, webkitVersion, browser = {};
 12 
 13   userAgent = (userAgent || navigator.userAgent).toLowerCase();
 14   language = language || navigator.language || navigator.browserLanguage;
 15 
 16   // Gibberish at the end is to determine when the browser version is done
 17   version = browser.version = (userAgent.match( /.*(?:rv|chrome|webkit|opera|ie)[\/: ](.+?)([ \);]|$)/ ) || [])[1];
 18   webkitVersion = (userAgent.match( /webkit\/(.+?) / ) || [])[1];
 19 
 20   /**
 21     @name SC.browser.isWindows
 22     @type Boolean
 23   */
 24   browser.windows = browser.isWindows = !!/windows/.test(userAgent);
 25 
 26   /**
 27     @name SC.browser.isMac
 28     @type Boolean
 29   */
 30   browser.mac = browser.isMac = !!/macintosh/.test(userAgent) || (/mac os x/.test(userAgent) && !/like mac os x/.test(userAgent));
 31 
 32   /**
 33     @name SC.browser.isiPhone
 34     @type Boolean
 35   */
 36   browser.iPhone = browser.isiPhone = !!/iphone/.test(userAgent);
 37 
 38   /**
 39     @name SC.browser.isiPod
 40     @type Boolean
 41   */
 42   browser.iPod = browser.isiPod = !!/ipod/.test(userAgent);
 43 
 44   /**
 45     @name SC.browser.isiPad
 46     @type Boolean
 47   */
 48   browser.iPad = browser.isiPad = !!/ipad/.test(userAgent);
 49 
 50   /**
 51     @name SC.browser.isiOS
 52     @type Boolean
 53   */
 54   browser.iOS = browser.isiOS = browser.iPhone || browser.iPod || browser.iPad;
 55 
 56   /**
 57     @name SC.browser.isAndroid
 58     @type Boolean
 59   */
 60   browser.android = browser.isAndroid = !!/android/.test(userAgent);
 61 
 62   /**
 63     @name SC.browser.opera
 64     @type String
 65   */
 66   browser.opera = /opera/.test(userAgent) ? version : 0;
 67 
 68   /**
 69     @name SC.browser.isOpera
 70     @type Boolean
 71   */
 72   browser.isOpera = !!browser.opera;
 73 
 74   /**
 75     @name SC.browser.msie
 76     @type String
 77   */
 78   browser.msie = /msie/.test(userAgent) && !browser.opera ? version : 0;
 79 
 80   /**
 81     @name SC.browser.isIE
 82     @type Boolean
 83   */
 84   browser.isIE = !!browser.msie;
 85 
 86   /**
 87     @name SC.browser.isIE8OrLower
 88     @type Boolean
 89   */
 90   browser.isIE8OrLower = !!(browser.msie && parseInt(browser.msie, 10) <= 8);
 91 
 92   /**
 93     @name SC.browser.mozilla
 94     @type String
 95   */
 96   browser.mozilla = /mozilla/.test(userAgent) && !/(compatible|webkit|msie)/.test(userAgent) ? version : 0;
 97 
 98   /**
 99     @name SC.browser.isMozilla
100     @type Boolean
101   */
102   browser.isMozilla = !!browser.mozilla;
103 
104   /**
105     @name SC.browser.webkit
106     @type String
107   */
108   browser.webkit = /webkit/.test(userAgent) ? webkitVersion : 0;
109 
110   /**
111     @name SC.browser.isWebkit
112     @type Boolean
113   */
114   browser.isWebkit = !!browser.webkit;
115 
116   /**
117     @name SC.browser.chrome
118     @type String
119   */
120   browser.chrome = /chrome/.test(userAgent) ? version: 0;
121 
122   /**
123     @name SC.browser.isChrome
124     @type Boolean
125   */
126   browser.isChrome = !!browser.chrome;
127 
128   /**
129     @name SC.browser.mobileSafari
130     @type String
131   */
132   browser.mobileSafari = /apple.*mobile/.test(userAgent) && browser.iOS ? webkitVersion : 0;
133 
134   /**
135     @name SC.browser.isMobileSafari
136     @type Boolean
137   */
138   browser.isMobileSafari = !!browser.mobileSafari;
139 
140   /**
141     @name SC.browser.iPadSafari
142     @type String
143   */
144   browser.iPadSafari = browser.iPad && browser.isMobileSafari ? webkitVersion : 0;
145 
146   /**
147     @name SC.browser.isiPadSafari
148     @type Boolean
149   */
150   browser.isiPadSafari = !!browser.iPadSafari;
151 
152   /**
153     @name SC.browser.iPhoneSafari
154     @type String
155   */
156   browser.iPhoneSafari = browser.iPhone && browser.isMobileSafari ? webkitVersion : 0;
157 
158   /**
159     @name SC.browser.isiPhoneSafari
160     @type Boolean
161   */
162   browser.isiPhoneSafari = !!browser.iphoneSafari;
163 
164   /**
165     @name SC.browser.iPodSafari
166     @type String
167   */
168   browser.iPodSafari = browser.iPod && browser.isMobileSafari ? webkitVersion : 0;
169 
170   /**
171     @name SC.browser.isiPodSafari
172     @type Boolean
173   */
174   browser.isiPodSafari = !!browser.iPodSafari;
175 
176   /**
177     @name SC.browser.isiOSHomeScreen
178     @type Boolean
179   */
180   browser.isiOSHomeScreen = browser.isMobileSafari && !/apple.*mobile.*safari/.test(userAgent);
181 
182   /**
183     @name SC.browser.safari
184     @type String
185   */
186   browser.safari = browser.webkit && !browser.chrome && !browser.iOS && !browser.android ? webkitVersion : 0;
187 
188   /**
189     @name SC.browser.isSafari
190     @type Boolean
191   */
192   browser.isSafari = !!browser.safari;
193 
194   /**
195     @name SC.browser.language
196     @type String
197   */
198   browser.language = language.split('-', 1)[0];
199 
200   /**
201     Possible values:
202 
203       - 'msie'
204       - 'mozilla'
205       - 'chrome'
206       - 'safari'
207       - 'opera'
208       - 'mobile-safari'
209       - 'unknown'
210 
211     @name SC.browser.current
212     @type String
213     @default 'unknown'
214   */
215   browser.current = browser.msie ? 'msie' : browser.mozilla ? 'mozilla' : browser.chrome ? 'chrome' : browser.safari ? 'safari' : browser.opera ? 'opera' : browser.mobileSafari ? 'mobile-safari' : browser.android ? 'android' : 'unknown';
216   return browser;
217 };
218 
219 
220 /** @class
221 
222   Contains information about the browser environment that SproutCore
223   is running in. String properties, such as `SC.browser.webkit` or
224   `SC.browser.msie`, will have a value that represents the browser build
225   number if that browser is being used. Otherwise, they will have a
226   falsey value. For convenience, Boolean counterparts for all of the
227   versioned properties are provided.
228 
229   @since SproutCore 1.0
230 */
231 SC.browser = SC._detectBrowser();
