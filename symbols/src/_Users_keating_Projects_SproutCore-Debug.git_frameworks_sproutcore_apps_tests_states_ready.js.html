  7 /*globals TestRunner */
  8 
  9 /**
 10   Initial state of application before it has loaded targets.
 11 */
 12 TestRunner.READY = SC.Responder.create({
 13   
 14   /**
 15     Invoked when you select a target.  Set the target controller then show 
 16     list state if needed.
 17   */
 18   selectTarget: function(sender, target) {
 19     if (target && target.isEnumerable) target = target.firstObject();
 20 
 21     TestRunner.sourceController.selectObject(target);
 22     
 23     if (target) {
 24       var tests = target.get('tests');
 25       if (tests && (tests.get('status') & SC.Record.BUSY)) {
 26         TestRunner.makeFirstResponder(TestRunner.READY_LOADING);
 27       } else if (!tests || (tests.get('length')===0)) {
 28         TestRunner.makeFirstResponder(TestRunner.READY_NO_TESTS);
 29       } else TestRunner.makeFirstResponder(TestRunner.READY_LIST);
 30       
 31     } else TestRunner.makeFirstResponder(TestRunner.READY_EMPTY);
 32   },
 33 
 34   /**
 35     Invoked when you select the test.
 36   */
 37   selectTest: function(sender, test) {
 38     if (!TestRunner.targetController.get('hasContent')) return NO ;
 39 
 40     if (test && test.isEnumerable) test = test.firstObject();
 41     TestRunner.detailController.set('content', test);
 42     TestRunner.set('routeName', test ? test.get('filename') : null);
 43 
 44     if (test) TestRunner.makeFirstResponder(TestRunner.READY_DETAIL);
 45     else TestRunner.makeFirstResponder(TestRunner.READY_LIST);
 46   },
 47   
 48   route: function(sender, params) {
 49     var target = TestRunner.computeRouteTarget(),
 50         test   = TestRunner.computeRouteTest();
 51 
 52     if (test) TestRunner.sendAction('selectTest', this, test);
 53     else TestRunner.sendAction('selectTarget', this, target);
 54   }
 55       
 56 });
