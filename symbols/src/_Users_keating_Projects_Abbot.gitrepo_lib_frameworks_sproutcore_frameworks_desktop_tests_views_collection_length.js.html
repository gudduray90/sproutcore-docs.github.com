  8 var view, content1, content2 ;
  9 
 10 module("SC.CollectionView.length", {
 11   setup: function() {
 12     
 13     // stub in collection view to verify that proper method are called
 14     view = SC.CollectionView.create({
 15       
 16       observer: CoreTest.stub('observer(length)').observes('length'),
 17       computeLayout: CoreTest.stub('computeLayout'),
 18       
 19       reset: function(){
 20         this.observer.reset();
 21         this.computeLayout.reset();
 22       }    
 23     });
 24     
 25     content1 = "a b c".w();
 26     content2 = "d e f g h".w();
 27   }
 28 });
 29 
 30 test("no content should have length of 0", function() {
 31   equals(view.get('content'), null, 'precond - content is null');
 32   equals(view.get('length'), 0, 'length should be 0');
 33 });
 34 
 35 test("length should be set property on newly inited object with content already set", function() {
 36   view = SC.CollectionView.create({ content: content1 });
 37   equals(view.get('length'), content1.get('length'), 'view.length should be content.length');
 38 });
 39 
 40 test("setting content should update length & notify", function() {
 41   view.set('content', content1);
 42   equals(view.get('length'), content1.get('length'), 'view.length should equal new length');
 43   view.observer.expect(1);
 44 });
 45 
 46 test("changing the content should update length & notify", function() {
 47   view.set('content', content1);
 48   view.reset(); // don't care.
 49   ok(content1.get('length') !== content2.get('length'), 'precond - content1.length should not equal content2.length');
 50   
 51   view.set('content', content2);
 52   
 53   equals(view.get('length'), content2.get('length'), 'view.length should equal new length');
 54   view.observer.expect(1);
 55 });
 56 
 57 test("modifying content to make it shorter should update view length and notify",function() {
 58   view.set('content', content1);
 59   view.reset(); // don't care.
 60   
 61   var len = content1.get('length');
 62   content1.removeAt(1);
 63 
 64   equals(view.get('length'), len-1, 'view.length should equal new length');
 65   view.observer.expect(1);
 66 });
 67 
 68 test("modifying content to add length should update view length and notify",function() {
 69   view.set('content', content1);
 70   view.reset(); // don't care.
 71 
 72   var len = content1.get('length');
 73   content1.insertAt(1, 'foo');
 74 
 75   equals(view.get('length'), len+1, 'view.length should equal new length');
 76   view.observer.expect(1);
 77 });
 78 
 79 test("modifying content so that it does not change the length should NOT change view length OR notify", function() {
 80   view.set('content', content1);
 81   view.reset(); // don't care.
 82 
 83   var len = content1.get('length');
 84   content1.replace(1, 1, 'foo');
 85 
 86   equals(view.get('length'), len, 'view.length should equal new length');
 87   view.observer.expect(0);
 88 });
