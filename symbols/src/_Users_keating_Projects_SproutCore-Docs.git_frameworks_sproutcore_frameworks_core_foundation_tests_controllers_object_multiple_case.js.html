  8 var src, src2, content, controller;
  9 
 10 // ..........................................................
 11 // NO MULTIPLE CONTENT
 12 // 
 13 
 14 module("SC.ObjectController - multiple_case - ALLOWSMULTIPLE = NO", {
 15   setup: function() {
 16     src        = SC.Object.create({ foo: "foo1", bar: "bar1" });
 17     src2       = SC.Object.create({ foo: "foo2", bar: "bar1" });
 18     content    = [src, src2];
 19     controller = SC.ObjectController.create({ 
 20       content: content,
 21       allowsMultipleContent: NO 
 22     });
 23   },
 24   
 25   teardown: function() {
 26     controller.destroy();
 27   }
 28 });
 29 
 30 test("hasContent", function() {
 31   equals(controller.get("hasContent"), NO, 'hasContent should be NO');
 32 });
 33 
 34 test("getting any value should return undefined", function() {
 35   equals(controller.get("foo"), undefined, 'controller.get(foo)');
 36   equals(controller.get("bar"), undefined, 'controller.get(bar)');
 37 });
 38 
 39 test("setting any unknown value should have no effect", function() {
 40   equals(controller.set("foo", "FOO"), controller, 'controller.set(foo, FOO) should return self');  
 41   equals(controller.set("bar", "BAR"), controller, 'controller.set(bar, BAR) should return self');
 42   equals(src.get("foo"), "foo1", 'src.get(foo)');
 43   equals(src.get("bar"), "bar1", 'src.get(bar)');
 44 });

 46 // ..........................................................
 47 // MULTIPLE CONTENT
 48 // 
 49 
 50 module("SC.ObjectController - multiple_case - ALLOWSMULTIPLE = YES", {
 51   setup: function() {
 52     src        = SC.Object.create({ foo: "foo1", bar: "bar1" });
 53     src2       = SC.Object.create({ foo: "foo2", bar: "bar1" });
 54     content    = [src, src2];
 55     controller = SC.ObjectController.create({ 
 56       content: content,
 57       allowsMultipleContent: YES 
 58     });
 59   },
 60   
 61   teardown: function() {
 62     controller.destroy();
 63   }
 64 });
 65 
 66 test("getting any unknown value", function() {
 67   same(controller.get("foo"), ["foo1", "foo2"], 'controller.get(foo) should return array with all items');
 68   same(controller.get("bar"), "bar1", 'controller.get(bar) should return single property since all items match');
 69 });
 70 
 71 test("setting any unknown value should pass through", function() {
 72   equals(controller.set("foo", "EDIT"), controller, 'controller.set(foo, EDIT) should return self');  
 73   equals(controller.set("bar", "EDIT"), controller, 'controller.set(bar, EDIT) should return self');
 74   equals(controller.set("baz", "EDIT"), controller, 'controller.set(baz, EDIT) should return self');
 75   
 76   equals(src.get("foo"), "EDIT", 'src.get(foo)');
 77   equals(src.get("bar"), "EDIT", 'src.get(bar)');
 78   equals(src.get("baz"), "EDIT", 'src.get(bar)');
 79 
 80   equals(src2.get("foo"), "EDIT", 'src2.get(foo)');
 81   equals(src2.get("bar"), "EDIT", 'src2.get(bar)');
 82   equals(src2.get("baz"), "EDIT", 'src2.get(bar)');
 83 });
 84 
 85 test("changing a property on a content object", function() {
 86   var callCount = 0;
 87   controller.addObserver("bar", function() { callCount++; });
 88 
 89   equals(controller.get("bar"), "bar1", "controller.get(bar) before edit should have original value");
 90 
 91   src.set("bar", "EDIT");
 92   same(controller.get("bar"), ["EDIT", "bar1"], "controller.get(bar) after edit should have updated value");
 93   equals(callCount, 1, 'observer on controller should have fired');
 94 });
 95 
 96 test("hasContent", function() {
 97   equals(controller.get("hasContent"), YES, 'should have content');
 98   
 99   var callCount = 0;
100   controller.addObserver("hasContent", function() { callCount++; });
101   
102   controller.set("content", null);
103   equals(controller.get("hasContent"), NO, "hasContent should == NO after setting to null");
104   ok(callCount > 0, 'hasContent observer should fire when setting to null');
105   
106   callCount = 0;
107   controller.set("content", content);
108   equals(controller.get("hasContent"), YES, "hasContent should == YES after setting back to content");
109   ok(callCount > 0, "hasContent observer should fire");
110 });
