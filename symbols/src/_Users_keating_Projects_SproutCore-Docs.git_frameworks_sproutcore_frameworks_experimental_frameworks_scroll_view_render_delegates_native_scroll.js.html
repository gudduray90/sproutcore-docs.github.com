  7 SC.BaseTheme.nativeScrollRenderDelegate = SC.RenderDelegate.create({
  8   className: 'native-scroll',
  9 
 10   render: function (dataSource, context) {},
 11 
 12   update: function (dataSource, context) {
 13     var K = 'scrollRenderDelegate',
 14         scroll = context.find('> .sc-scroll-container-view');
 15 
 16     if (dataSource.didChangeFor(K, 'canScrollHorizontal')) {
 17       scroll.css('overflow-x', dataSource.get('canScrollHorizontal') ? 'scroll' : 'hidden');
 18     }
 19 
 20     if (dataSource.didChangeFor(K, 'canScrollVertical')) {
 21       scroll.css('overflow-y', dataSource.get('canScrollVertical') ? 'scroll' : 'hidden');
 22     }
 23   }
 24 
 25 });
