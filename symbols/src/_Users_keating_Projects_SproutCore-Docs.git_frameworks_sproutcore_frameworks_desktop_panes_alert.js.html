 38 /** @class
 39   Displays a preformatted modal alert pane.
 40   
 41   Alert panes are a simple way to provide modal messaging that otherwise 
 42   blocks the user's interaction with your application.  Alert panes are 
 43   useful for showing important error messages and confirmation dialogs. They
 44   provide a better user experience than using the OS-level alert dialogs.
 45   
 46   ## Displaying an Alert Pane
 47   
 48   The easiest way to display an alert pane is to use one of the various
 49   class methods defined on `SC.AlertPane`, passing the message and an optional
 50   detailed description and caption.
 51   
 52   There are four variations of this method can you can invoke:
 53   
 54    - `warn()` -- displays an alert pane with a warning icon to the left.
 55    - `error()` -- displays an alert with an error icon to the left
 56    - `info()` -- displays an alert with an info icon to the left
 57    - `plain()` -- displays an alert w/o any icon
 58    - `show()` -- displays an alert with a customizable icon to the left
 59   
 60   In addition to passing a message, description and caption, you can also customize
 61   the title of the button 1 (OK) and add an optional button 2 and 3 (Cancel or Extra).
 62   
 63    - button1 -- 1st button from the right. default:OK
 64    - button2 -- 2nd button from the right. Optional. Could be Cancel or 2nd action.
 65    - button3 -- 1st button from the left. Optional. Could be Cancel or alternative option.
 66   
 67   Additionally, you can define a delegate object.  This delegate's 
 68   `alertPaneDidDismiss()` method will be called when the pane
 69   is dismissed, passing the pane instance and a key indicating which 
 70   button was pressed.
 71   
 72   ## Examples
 73   
 74   Show a simple AlertPane with an OK button:
 75   
 76       SC.AlertPane.warn({
 77         message: "Could not load calendar",
 78         description: "Your internet connection may be unavailable or our servers may be down.",
 79         caption: "Try again in a few minutes."
 80       });
 81   
 82   Show an AlertPane with a customized OK title (title will be 'Try Again') and
 83   custom action:
 84   
 85       SC.AlertPane.warn({
 86         message: "Could not load calendar",
 87         description: "Your internet connection may be unavailable or our servers may be down.",
 88         caption: "Try again in a few minutes.",
 89         buttons: [
 90           { title: "Try Again" }
 91         ]
 92       });
 93   
 94   Show an AlertPane with a custom OK, a Cancel button and an Extra button, 
 95   each with custom titles.  Also, pass a delegate that will be invoked when
 96   the user's dismisses the dialog.
 97   
 98       MyApp.calendarController = SC.Object.create({
 99         alertPaneDidDismiss: function(pane, status) {
100           switch(status) {
101             case SC.BUTTON1_STATUS:
102               this.tryAgain();
103               break;
104             
105             case SC.BUTTON2_STATUS:
106               // do nothing
107               break;
108             
109             case SC.BUTTON3_STATUS:
110               this.showMoreInfo();
111               break;
112           }
113         },
114         ...
115       });
116     
117     
118       SC.AlertPane.warn({
119         message: "Could not load calendar",
120         description: "Your internet connection may be unavailable or our servers may be down.",
121         caption: "Try again in a few minutes.",
122         delegate: MyApp.calendarController,
123         buttons: [
124           { title: "Try Again" },
125           { title: "Cancel" },
126           { title: "More Info…" }
127         ]
128       });
129   
130 131   Instead of using the delegate pattern above, you can also specify a target
132   and an action, similar to SC.ButtonView.
133   
134       SC.AlertPane.warn({
135         message: "Could not load calendar",
136         description: "Your internet connection may be unavailable or our servers may be down.",
137         caption: "Try again in a few minutes.",
138         buttons: [
139           { 
140             title: "OK", 
141             action: "didClickOK",
142             target: MyApp.calendarController
143           }
144         ]
145       });
146   
147   Also note that in addition to passing the action as a string of the method
148   name that will be invoked, you can also give a function reference as the
149   action.
150   
151   
152   @extends SC.PanelPane
153   @since SproutCore 1.0
154 */

178     The method to be called on your delegate will be:
179

367   /**
368     Action triggered whenever any button is pressed. Also the hides the 
369     alertpane itself.
370     
371     This will trigger the following chain of events:
372 
373      1. If a delegate was given, and it has alertPaneDidDismiss it will be called
374      2. Otherwise it will look for the action of the button and call:
375       a) The action function reference if one was given
376       b) The action method on the target if one was given
377       c) If both a and b are missing, call the action on the rootResponder
378     
379     @param {SC.View} sender - the button view that was clicked
380   */
381   dismiss: function(sender) {
382     var del = this.delegate,
383         rootResponder, action, target;
384     
385     if (del && del.alertPaneDidDismiss) {
386       del.alertPaneDidDismiss(this, sender.get('actionKey'));
387     }
388 
389     if (action = (sender && sender.get('customAction'))) {
390       if (SC.typeOf(action) === SC.T_FUNCTION) {
391         action.call(action);
392       } else {
393         rootResponder = this.getPath('pane.rootResponder');
394         if(rootResponder) {
395           target = sender.get('customTarget');
396           rootResponder.sendAction(action, target || del, this, this, null, this);
397         }
398       }
399     }
400     
401     this.remove(); // hide alert
402   },

410     if (v) v.displayDidChange(); // re-render message
411   }.observes('icon', 'message', 'displayDescription', 'displayCaption')

415 SC.AlertPane.mixin(
416 /** @scope SC.AlertPane */{

418   /**
419     Show a dialog with a given set of hash attributes:
420     
421         SC.AlertPane.show({
422           message: "Could not load calendar",
423           description: "Your internet connection may be unavailable or our servers may be down.",
424           caption: "Try again in a few minutes.",
425           delegate: MyApp.calendarController
426         });
427     
428     See more examples for how to configure buttons and individual actions in the 
429     documentation for the `SC.AlertPane` class.
430     
431     @param {Hash} args
432     @return {SC.AlertPane} the pane shown
433   */
434   show: function(args) {
435     // normalize the arguments if this is a deprecated call
436     args = SC.AlertPane._argumentsCall.apply(this, arguments);
437     
438     var pane = this.create(args), 
439         idx = 0, 
440         buttons = args.buttons,
441         buttonView, title, action, target, themeName;
442     
443     if(buttons) {
444       buttons.forEach(function(button) {
445         idx++;
446         if(!button) return;
447         buttonView = pane.get('button%@'.fmt(idx));
448         
449         title = button.title;
450         action = button.action;
451         target = button.target;
452         themeName = args.themeName || 'capsule';
453         
454         buttonView.set('title'.fmt(idx), title);
455         if(action) buttonView.set('customAction'.fmt(idx), action);
456         if(target) buttonView.set('customTarget'.fmt(idx), target);
457         buttonView.set('isVisible', !!title);
458         buttonView.set('themeName', themeName);
459       });
460     } else {
461       // if there are no buttons defined, just add the standard OK button
462       buttonView = pane.get('button1');
463       buttonView.set('title', "OK");
464       buttonView.set('isVisible', YES);
465     }
466     
467     var show = pane.append(); // make visible.
468     pane.adjust('height', pane.childViews[0].$().height());
469     pane.updateLayout();
470     return show;
471   },
472   
473   /**
474     Same as `show()` just that it uses sc-icon-alert-48 CSS classname
475     as the dialog icon
476     
477     @param {Hash} args
478     @return {SC.AlertPane} the pane shown
479   */
480   warn: function(args) {
481     // normalize the arguments if this is a deprecated call
482     args = SC.AlertPane._argumentsCall.apply(this, arguments);
483     
484     args.icon = 'sc-icon-alert-48';
485     return this.show(args);
486   },
487

533   /** @private
534     Set properties to new structure for call that use the old arguments 
535     structure.
536     
537     Deprecated API but is preserved for now for backwards compatibility.
538     
539     @deprecated
540   */
541   _argumentsCall: function(args) {
542     var ret = args;
543     if(SC.typeOf(args)!==SC.T_HASH) {
544       //@if(debug)
545       SC.debug('SC.AlertPane has changed the signatures for show(), info(), warn(), error() and plain(). Please update accordingly.');
546       //@endif
547       var normalizedArgs = this._normalizeArguments(arguments);
548       
549       // now convert it to the new format for show()
550       ret = {
551         message: normalizedArgs[0],
552         description: normalizedArgs[1],
553         caption: normalizedArgs[2],
554         delegate: normalizedArgs[7],
555         icon: (normalizedArgs[6] || 'sc-icon-alert-48'),
556         themeName: 'capsule'
557       };
558       
559       // set buttons if there are any (and check if it's a string, since last
560       // argument could be the delegate object)
561       if(SC.typeOf(normalizedArgs[3])===SC.T_STRING || SC.typeOf(normalizedArgs[4])===SC.T_STRING || SC.typeOf(normalizedArgs[5])===SC.T_STRING) {
562         ret.buttons = [
563           { title: normalizedArgs[3] },
564           { title: normalizedArgs[4] },
565           { title: normalizedArgs[5] }
566         ];
567       }
568       
569     }
570     return ret;
571   },
572   
573   /** @private
574     internal method normalizes arguments for processing by helper methods.
575   */
576   _normalizeArguments: function(args) {
577     args = SC.A(args); // convert to real array
578     var len = args.length, delegate = args[len-1];
579     if (SC.typeOf(delegate) !== SC.T_STRING) {
580       args[len-1] = null;
581     } else delegate = null ;
582     args[7] = delegate ;
583     return args ;
584   }
