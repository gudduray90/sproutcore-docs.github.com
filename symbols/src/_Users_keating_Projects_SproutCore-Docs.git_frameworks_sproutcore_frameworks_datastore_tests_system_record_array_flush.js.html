  8 // test core array-mapping methods for RecordArray
  9 var store, storeKey, json, rec, storeKeys, recs, query, recsController, fooQuery, fooRecs, fooRecsController;
 10 module("SC.RecordArray core methods", {
 11   setup: function() {
 12     // setup dummy store
 13     store = SC.Store.create();
 14 
 15     storeKey = SC.Record.storeKeyFor('foo');
 16     json = {  guid: "foo", foo: "foo" };
 17 
 18     store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
 19 
 20 
 21     // get record
 22     rec = store.materializeRecord(storeKey);
 23     equals(rec.get('foo'), 'foo', 'record should have json');
 24 
 25     // get record array.
 26     query = SC.Query.create({ recordType: SC.Record });
 27     recs = SC.RecordArray.create({ store: store, query: query });
 28 
 29     recsController = SC.Object.create({
 30       content: recs,
 31       bigCost: NO,
 32       veryExpensiveObserver: function() {
 33         this.set('bigCost', YES);
 34       }.observes('.content.[]')
 35     });
 36 
 37     fooQuery = SC.Query.create({ recordType: SC.Record, conditions: "foo='foo'" });
 38     fooRecs = SC.RecordArray.create({ store: store, query: fooQuery });
 39 
 40     fooRecsController = SC.Object.create({
 41       content: fooRecs,
 42       bigCost: NO,
 43       veryExpensiveObserver: function() {
 44         this.set('bigCost', YES);
 45       }.observes('.content.[]')
 46     });
 47   }
 48 });

 83 test("calling storeDidChangeStoreKeys() with a matching recordType", function() {
 84   recs.flush(); // do initial setup
 85   var orig = recs.get('storeKeys').copy();
 86 
 87   // do it this way instead of using store.createRecord() to isolate the
 88   // method call.
 89   storeKey = SC.Record.storeKeyFor("bar");
 90   json     = {  guid: "bar", foo: "bar" };
 91   store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
 92 
 93   equals(recs.get('needsFlush'), NO, 'PRECOND - should not need flush');
 94   same(recs.get('storeKeys'), orig, 'PRECOND - storeKeys should not have changed yet');
 95 
 96   recs.storeDidChangeStoreKeys([storeKey], SC.Set.create().add(SC.Record));
 97 
 98   orig.push(storeKey); // update - must be last b/c id.bar.storeKey < id.foo.storeKey
 99   equals(recs.get('needsFlush'), NO, 'should not need flush anymore');
100   same(recs.get('storeKeys'), orig, 'storeKeys should now be updated - rec1[%@]{%@} = %@, rec2[%@]{%@} = %@'.fmt(
101     rec.get('id'), rec.get('storeKey'), rec,
102 
103     store.materializeRecord(storeKey).get('id'),
104     storeKey,
105     store.materializeRecord(storeKey)));
106 
107 });

109 test("calling storeDidChangeStoreKeys() with a non-matching recordType", function() {
110 
111   var Foo = SC.Record.extend(),
112       Bar = SC.Record.extend();
113 
114   storeKey = Foo.storeKeyFor('foo2');
115   json = { guid: "foo2" };
116 
117   store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
118 
119   query = SC.Query.create({ recordType: Foo });
120   recs = SC.RecordArray.create({ store: store, query: query });
121 
122   equals(recs.get('length'), 1, 'should have a Foo record');
123 
124   // now simulate adding a Bar record
125   storeKey = Bar.storeKeyFor('bar');
126   json = { guid: "bar" };
127   store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
128 
129   recs.storeDidChangeStoreKeys([storeKey], SC.Set.create().add(Bar));
130   equals(recs.get('needsFlush'), NO, 'should not have indicated it needed a flush');
131 
132 });

134 test("calling storeDidChangeStoreKeys() to remove a record", function() {
135 
136   equals(recs.get('length'), 1, 'PRECOND - should have storeKey');
137 
138   store.writeStatus(storeKey, SC.Record.DESTROYED_CLEAN);
139   equals(recs.get('storeKeys').length, 1, 'should still have storeKey');
140   recs.storeDidChangeStoreKeys([storeKey], SC.Set.create().add(SC.Record));

145 test("calling storeDidChangeStoreKeys() with a matching recordType should not unnecessarily call enumerableContentDidChange", function() {
146   // do initial setup
147   recs.flush();
148   fooRecs.flush();
149 
150   recsController.set('bigCost', NO);
151   fooRecsController.set('bigCost', NO);
152 
153   // do it this way instead of using store.createRecord() to isolate the
154   // method call.
155   storeKey = SC.Record.storeKeyFor("bar");
156   json     = {  guid: "bar", foo: "bar" };
157   store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
158 
159   equals(recsController.get('bigCost'), NO, 'PRECOND - recsController should not have spent big cost');
160   equals(fooRecsController.get('bigCost'), NO, 'PRECOND - fooRecsController should not have spent big cost');
161 
162   recs.storeDidChangeStoreKeys([storeKey], SC.Set.create().add(SC.Record));
163   fooRecs.storeDidChangeStoreKeys([storeKey], SC.Set.create().add(SC.Record));
164 
165   equals(recsController.get('bigCost'), YES, 'recsController should have spent big cost');
166   equals(fooRecsController.get('bigCost'), NO, 'fooRecsController should not have spent big cost');

192 module("SC.RecordArray core methods", {
193   setup: function() {
194     // setup dummy store
195     store = SC.Store.create();
196 
197     storeKey = SC.Record.storeKeyFor('foo');
198     json = {  guid: "foo", name: "foo" };
199     store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
200     foo = store.materializeRecord(storeKey);
201     equals(foo.get('name'), 'foo', 'record should have json');
202 
203     storeKey = SC.Record.storeKeyFor('bar');
204     json2 = { guid: "bar", name: "bar" };
205     store.writeDataHash(storeKey, json2, SC.Record.READY_CLEAN);
206     bar = store.materializeRecord(storeKey);
207     equals(bar.get('name'), 'bar', 'record should have json');
208 
209     // get record array.
210     query = SC.Query.create({ recordType: SC.Record, orderBy: 'name' });
211     recs = store.find(query);
212   }
213 });
214 
215 test("local query should notify changes", function() {
216   // note: important to retrieve records from RecordArray first to prime
217   // any cache
218   same(recs.mapProperty('id'), ['bar', 'foo'], 'PRECOND - bar should appear before foo');
219 
220   SC.stopIt = YES;
221 
222   SC.RunLoop.begin();
223   bar.set('name', 'zzbar');
224   SC.RunLoop.end(); // should resort record array
225 
226   same(recs.mapProperty('id'), ['foo', 'bar'], 'order of records should change');
227 });
