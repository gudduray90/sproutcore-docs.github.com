  8 module("SC.View - Static Layout functionality");
  9 
 10 test("Static layout", function() {
 11   var view = SC.View.create({
 12     useStaticLayout: YES
 13   });
 14 
 15   view.createLayer();
 16 
 17   ok(view.$().is('.sc-static-layout'), "views with useStaticLayout get the sc-static-layout class");
 18 });
 19 
 20 test("Background color", function() {
 21   var view = SC.View.create({
 22     backgroundColor: "red"
 23   });
 24 
 25   view.createLayer();
 26 
 27   equals(view.get('layer').style.backgroundColor, "red", "backgroundColor sets the CSS background-color class");
 28 });
