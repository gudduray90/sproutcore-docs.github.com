  8 module('SC.Math');

 10 test("0.12 rounded default decimal place", function() {
 11   equals(SC.Math.round(0.12), 0);
 12 });
 13 
 14 test("1.23456789 rounded into decimals", function() {
 15   equals(SC.Math.round(1.23456789), 1, "0 decimal places");
 16   equals(SC.Math.round(1.23456789, 1), 1.2, "1 decimal place");
 17   equals(SC.Math.round(1.23456789, 2), 1.23, "2 decimal places");
 18   equals(SC.Math.round(1.23456789, 3), 1.235, "3 decimal places");
 19   equals(SC.Math.round(1.23456789, 4), 1.2346, "4 decimal places");
 20   equals(SC.Math.round(1.23456789, 7), 1.2345679, "7 decimal places");
 21 });
 22 
 23 test("123456.7890 rounded into wholes", function() {
 24   equals(SC.Math.round(123456.7890), 123457, "0 places");
 25   equals(SC.Math.round(123456.7890, -1), 123460, "1 place");
 26   equals(SC.Math.round(123456.7890, -2), 123500, "2 places");
 27   equals(SC.Math.round(123456.7890, -3), 123000, "3 places");
 28   equals(SC.Math.round(123456.7890, -4), 120000, "4 places");
 29 });
 30 
 31 test("near", function() {
 32   ok(SC.Math.near(1, 1), "Equal numbers should be considered near -- obviously");
 33   ok(!SC.Math.near(2, 1), "Unequal whole numbers should not be near");
 34   ok(SC.Math.near(1.00000001, 1), "Numbers inside lambda range should be considered near");
 35 });
 36 
 37 test("near - custom lambda", function() {
 38   ok(SC.Math.near(2, 1, 1), "Lambda should define an inclusive range -- ie. 2 with lambda 1 is considered near 1");
 39 });
