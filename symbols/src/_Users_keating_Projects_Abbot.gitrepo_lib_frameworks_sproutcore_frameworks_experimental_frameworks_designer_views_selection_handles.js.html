  7 /**
  8   This View is used by Greenhouse when application is in design mode
  9   This view draws selection handles for a given designer.  It will also 
 10   forward any mouse events to the underlying designer.
 11 */
 12 SC.SelectionHandlesView = SC.View.extend(
 13 /** @scope SC.SelectionHandlesView.prototype */ {
 14 
 15   /**
 16     The designer that owns this selection.  mouse and keyboard events are 
 17     forwarded to this.
 18   */
 19   designer: null,
 20   
 21   classNames: 'handles',
 22   
 23   render: function(context, firstTime) {
 24     var designer = this.get('designer'),
 25         vertical = designer ? designer.get('canResizeVertical') : NO,
 26         horizontal = designer ? designer.get('canResizeHorizontal') : NO,
 27         handles ;
 28         
 29     // render handles
 30     if (firstTime || (vertical !== this._vertical) || (horizontal === this._horizontal)) {
 31       this._vertical = vertical ;
 32       this._horizontal = horizontal;
 33 
 34       if (vertical && horizontal) {
 35         handles = ['top left', 'top right', 'bottom left','bottom right'];
 36       } else if (vertical) {
 37         handles = ['top', 'bottom'];
 38       } else if (horizontal) {
 39         handles = ['left', 'right'];
 40       } else handles = [];
 41 
 42 
 43       handles.forEach(function(classNames) {
 44         context.begin('span')
 45           .classNames(SC.String.w(classNames))
 46           .addClass('handle')
 47           .end();
 48       }, this);
 49     }
 50     
 51   },
 52 
 53   // ..........................................................
 54   // EVENT HANDLING
 55   // 
 56   // By default just forward to designer
 57   
 58   mouseDown: function(evt) {
 59     var d = this.designer;
 60     return (d && d.mouseDown) ? d.mouseDown(evt) : null;
 61   },
 62   
 63   mouseUp: function(evt) {
 64     var d = this.designer;
 65     return (d && d.mouseUp) ? d.mouseUp(evt) : null;
 66   },
 67   
 68   mouseMoved: function(evt) {
 69     var d = this.designer;
 70     return (d && d.mouseMoved) ? d.mouseMoved(evt) : null;
 71   },
 72   
 73   mouseDragged: function(evt) {
 74     var d = this.designer;
 75     return (d && d.mouseDragged) ? d.mouseDragged(evt) : null;
 76   }
 77   
 78 });
