  8 var pane;
  9 
 10 module("SC.View#touch", {
 11   setup: function() {
 12     SC.run(function() {
 13      pane = SC.Pane.create({
 14        layout: { width: 200, height: 200, left: 0, top: 0 },
 15        childViews: ['outerView'],
 16 
 17        outerView: SC.View.extend({
 18          childViews: ['innerView'],
 19 
 20          innerView: SC.View.extend({
 21            layout: { width: 50, height: 50, left: 100, top: 100 }
 22          })
 23        })
 24      }).append();
 25     });
 26   },
 27 
 28   teardown: function() {
 29     pane.remove();
 30     pane = null;
 31   }
 32 });
 33 
 34 function testTouches (view, left, top, boundary) {
 35   var frame = view.get('frame');
 36 
 37   // Just outside the touchBoundary
 38   ok(!view.touchIsInBoundary({ pageX: left - boundary - 1, pageY: top }), '{ pageX: %@, pageY: %@ } is not inside %@'.fmt(left - boundary - 1, top, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 39 
 40   // Just inside the touchBoundary
 41   ok(view.touchIsInBoundary({ pageX: left - boundary, pageY: top }), '{ pageX: %@, pageY: %@ } is inside %@'.fmt(left - boundary, top, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 42 
 43   // Just inside the edge of the view
 44   ok(view.touchIsInBoundary({ pageX: left + frame.width, pageY: top }), '{ pageX: %@, pageY: %@ } is inside %@'.fmt(left + frame.width, top, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 45 
 46   // Just inside the touchBoundary
 47   ok(view.touchIsInBoundary({ pageX: left + frame.width + boundary, pageY: top }), '{ pageX: %@, pageY: %@ } is inside %@'.fmt(left + frame.width + boundary, top, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 48 
 49   // Just outside the touchBoundary
 50   ok(!view.touchIsInBoundary({ pageX: left + frame.width + boundary + 1, pageY: top }), '{ pageX: %@, pageY: %@ } is not inside %@'.fmt(left + frame.width + boundary + 1, top, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 51 
 52   // Just outside the touchBoundary
 53   ok(!view.touchIsInBoundary({ pageX: left, pageY: top - boundary - 1 }), '{ pageX: %@, pageY: %@ } is not inside %@'.fmt(left, top - boundary - 1, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 54 
 55   // Just inside the touchBoundary
 56   ok(view.touchIsInBoundary({ pageX: left, pageY: top - boundary }), '{ pageX: %@, pageY: %@ } is inside %@'.fmt(left, top - boundary, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 57 
 58   // Just inside the edge of the view
 59   ok(view.touchIsInBoundary({ pageX: left, pageY: top + frame.height }), '{ pageX: %@, pageY: %@ } is inside %@'.fmt(left, top + frame.height, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 60 
 61   // Just inside the touchBoundary
 62   ok(view.touchIsInBoundary({ pageX: left, pageY: top + frame.height + boundary }), '{ pageX: %@, pageY: %@ } is inside %@'.fmt(left, top + frame.height + boundary, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 63 
 64   // Just outside the touchBoundary
 65   ok(!view.touchIsInBoundary({ pageX: left, pageY: top + frame.height + boundary + 1 }), '{ pageX: %@, pageY: %@ } is not inside %@'.fmt(left, top + frame.height + boundary + 1, SC.stringFromRect(view.get('_touchBoundaryFrame'))));
 66 
 67 }
 68 
 69 test("touchIsInBoundary() should return appropriate values", function() {
 70   var outerView = pane.get('outerView'),
 71     innerView = outerView.get('innerView');
 72 
 73   testTouches(innerView, 100, 100, 25);
 74 
 75   // Move the inner view
 76   SC.run(function() {
 77     innerView.adjust('top', 150);
 78   });
 79   testTouches(innerView, 100, 150, 25);
 80 
 81   // Move the outer view
 82   SC.run(function() {
 83     outerView.adjust('left', 100);
 84   });
 85   testTouches(innerView, 200, 150, 25);
 86 
 87   // Expand the touch boundary
 88   SC.run(function() {
 89     innerView.set('touchBoundary', { left: 50, bottom: 50, top: 50, right: 50 });
 90   });
 91   testTouches(innerView, 200, 150, 50);
 92 
 93   // Contract the touch boundary
 94   SC.run(function() {
 95     innerView.set('touchBoundary', { left: 5, bottom: 5, top: 5, right: 5 });
 96   });
 97   testTouches(innerView, 200, 150, 5);
 98 });
 99 
100 test("touchIsInBoundary() should return appropriate values for a newly appended view", function() {
101   var outerView = pane.get('outerView'),
102     innerView = outerView.get('innerView');
103 
104   // Append a view
105   var newView = SC.View.create({
106     layout: { width: 10, height: 10, left: 50, top: 50 }
107   });
108 
109   SC.run(function() {
110     outerView.appendChild(newView);
111   });
112   testTouches(newView, 50, 50, 25);
113 });
