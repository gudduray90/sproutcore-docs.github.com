  8 /*
  9   Tests SplitView methods; specifically, the API that users of SplitView use
 10   (not the methods that they may override).
 11   
 12   The methods tested are:
 13   
 14   - getPositionForChild
 15   - adjustPositionForChild
 16   
 17   They are tested in several different ways, with child views configured in many
 18   different ways.
 19   
 20   We also run the entire test suite twice: once in each orientation. The numbers should
 21   be the same either way. In the ui tests, we test that they get applied properly.
 22 */
 23 
 24 function setupTestSuite(layoutDirection) {
 25   
 26 var splitView, left, right, middleLeft, middleRight;
 27 module(
 28   "SplitView " +
 29   (layoutDirection === SC.HORIZONTAL_LAYOUT ? "Horizontal" : "Vertical") + 
 30   " - Methods", 
 31   {
 32   setup: function() {
 33     SC.RunLoop.begin();
 34     // we test with no split divider because we want to easily calculate layout.
 35     splitView = SC.SplitView.create({
 36       childViews: [
 37         SC.View.create(SC.SplitChild, { canCollapse: YES, collapseAtSize: 50 }),
 38         SC.View.create(SC.SplitChild, { canCollapse: YES, collapseAtSize: 50 })
 39       ],
 40       
 41       layout: { left: 0, top: 0, width: 500, height: 500 },
 42       
 43       layoutDirection: layoutDirection,
 44       
 45       splitDividerView: null
 46     });
 47     
 48     SC.RunLoop.end();
 49     
 50     left = splitView.childViews[0];
 51     right = splitView.childViews[1];
 52   },
 53   
 54   teardown: function() {
 55     
 56   }
 57 });
 58 
 59 test("Initial positions are correct", function() {
 60   equals(splitView.childViews.length, 2, "SplitView has 2 children");
 61   equals(splitView.getPositionForChild(left), 0, "Left should always be at 0");
 62   equals(splitView.getPositionForChild(right), 250, "Right should be at 1/2: 50");
 63 });
 64 
 65 test("Can adjust position for a child", function() {
 66   var result = splitView.adjustPositionForChild(right, 100);
 67   equals(result, 100, "Result of adjustPositionForChild(right, 100)");
 68   equals(left.get('position'), 0, "Left position (should always be 0)");
 69   equals(left.get('size'), 100, "Left size");
 70   equals(right.get('position'), 100, "Right position");
 71   equals(right.get('size'), 400, "Right size");
 72 });
 73 
 74 
 75 test("SplitView - Methods: Minimum and Maximum widths and heights", function() {
 76   // first, test with the built-in minimum size
 77   var result = splitView.adjustPositionForChild(right, 80);
 78   equals(result, 100, "Result of adjustPositionForChild(right, 80)");
 79   equals(left.get('position'), 0, "Left position (should always be 0)");
 80   equals(left.get('size'), 100, "Left size");
 81   equals(right.get('position'), 100, "Right position");
 82   equals(right.get('size'), 400, "Right size");
 83   
 84   // now, add a max and test that
 85   left.set('maximumSize', 150);
 86   
 87   result = splitView.adjustPositionForChild(right, 180);
 88   equals(result, 150, "Result of adjustPositionForChild(right, 180) when left's max size = 150");
 89   equals(left.get('position'), 0, "Left position (should always be 0)");
 90   equals(left.get('size'), 150, "Left size");
 91   equals(right.get('position'), 150, "Right position");
 92   equals(right.get('size'), 350, "Right size");
 93   
 94   // and now go back to middle
 95   result = splitView.adjustPositionForChild(right, 130);
 96   equals(result, 130, "Result of adjustPositionForChild(right, 130) when left's max size = 150");
 97   equals(left.get('position'), 0, "Left position (should always be 0)");
 98   equals(left.get('size'), 130, "Left size");
 99   equals(right.get('position'), 130, "Right position");
100   equals(right.get('size'), 370, "Right size");
101 });
102 
103 test("SplitView - Methods: Collapsing a child", function() {
104   var result = splitView.adjustPositionForChild(right, 40);
105   
106   // first was collapsed, so position of seconds is now at 0.
107   equals(result, 0, "Result of adjustPositionForChild(right, 40)");
108   equals(left.get('position'), 0, "Left position (should always be 0)");
109   equals(left.get('size'), 0, "Left size after collapsing");
110   equals(right.get('position'), 0, "Right position");
111   equals(right.get('size'), 500, "Right size");
112 });

115 module(
116   "SplitView 4-pane " +
117   (layoutDirection === SC.HORIZONTAL_LAYOUT ? "Horizontal" : "Vertical") + 
118   " - Methods",
119   {
120   setup: function() {
121     SC.RunLoop.begin();
122     // we test with no split divider because we want to easily calculate layout.
123     splitView = SC.SplitView.create({
124       childViews: [
125         SC.View.create(SC.SplitChild, { canCollapse: YES, collapseAtSize: 50, autoResizeStyle: SC.RESIZE_MANUAL }),
126         SC.View.create(SC.SplitChild, { autoResizeStyle: SC.RESIZE_AUTOMATIC }),
127         SC.View.create(SC.SplitChild, { autoResizeStyle: SC.RESIZE_AUTOMATIC }),
128         SC.View.create(SC.SplitChild, { autoResizeStyle: SC.FIXED_SIZE, canCollapse: YES, collapseAtSize: 50 })
129       ],
130       
131       layout: { left: 0, top: 0, width: 800, height: 800 },
132       
133       layoutDirection: layoutDirection,
134       
135       splitDividerView: null
136     });
137     
138     SC.RunLoop.end();
139     
140     left = splitView.childViews[0];
141     middleLeft = splitView.childViews[1];
142     middleRight = splitView.childViews[2];
143     right = splitView.childViews[3];
144   }
145 });
146 
147 function confirmPositions() {
148   for (var i = 0; i < arguments.length; i++) {
149     equals(splitView.childViews[i].get('position'), arguments[i]);
150   }
151 }
152 
153 test("Initial positions are correct", function() {
154   // the left and right sides do not resize unless forced to; as such, they
155   // stay at 100.
156   confirmPositions(0, 100, 400, 700);
157 });
158 
159 test("Repositioning handles indirectness", function() {
160   var result;
161   
162   // moving within available space works
163   result = splitView.adjustPositionForChild(middleLeft, 200);
164   equals(result, 200, "Moved to target position successfully.");
165   confirmPositions(0, 200, 400, 700);
166   
167   // indirect being turned off for the next view, moving past min size
168   // for middleLeft should still work because it is an immediate sibling
169   result = splitView.adjustPositionForChild(middleLeft, 400);
170   equals(result, 400, "Moved to 400");
171   confirmPositions(0, 400, 500, 700);
172   
173   // moving so far as to move the fourth view, however, should not work (it is fixed size)
174   result = splitView.adjustPositionForChild(middleLeft, 700);
175   equals(result, 500, "Limited to 500 due to minimum sizes and because last cannot move.");
176   confirmPositions(0, 500, 600, 700);
177   
178   // changing last from fixed should not allow it through if indirect is not allowed
179   right.set('autoResizeStyle', SC.RESIZE_AUTOMATIC);
180   right.set('allowsIndirectAdjustments', NO);
181   
182   result = splitView.adjustPositionForChild(middleLeft, 700);
183   equals(result, 500, "Limited to 500 due to minimum sizes and because last cannot move.");
184   confirmPositions(0, 500, 600, 700);
185   
186   // but it should work if indirect _is_ allowed (rightmost should collapse)
187   right.set('allowsIndirectAdjustments', YES);
188   
189   result = splitView.adjustPositionForChild(middleLeft, 700);
190   equals(result, 600, "Limited to 600 due to minimum sizes and because last collapses.");
191   confirmPositions(0, 600, 700, 800);
192 });
193 
194 test("Repositioning without compensation", function() {
195   // first, disable all compensation
196   left.set('compensatesForMovement', NO);
197   middleLeft.set('compensatesForMovement', NO);
198   middleRight.set('compensatesForMovement', NO);
199   right.set('compensatesForMovement', NO);
200   
201   var result;
202   
203   // repositioning middle-left should fail because rightmost can't move.
204   result = splitView.adjustPositionForChild(middleLeft, 800);
205   equals(result, 100, "Failed to move to target because compensation is off.");
206   confirmPositions(0, 100, 400, 700);
207   
208   // turning compensation on for second-to-right makes _it_ move to max
209   middleRight.set('compensatesForMovement', YES);
210   result = splitView.adjustPositionForChild(middleLeft, 800);
211   equals(result, 300, "Moved to min width of middleRight");
212   confirmPositions(0, 300, 600, 700);
213 });
214 
215 test("Resizing split view", function() {
216   SC.RunLoop.begin();
217   // note: we test both horizontal and vertical
218   splitView.adjust('width', 1600);
219   splitView.adjust('height', 1600);
220   SC.RunLoop.end();
221   
222   // should have resized the auto resizable parts
223   confirmPositions(0, 100, 800, 1500);
224   
225   // manually resize left part; we're going to test that, when it gets small enough,
226   // even the RESIZE_MANUAL will change sizes
227   splitView.adjustPositionForChild(middleLeft, 300);
228   
229   // so now, just to be clear, the positions should be:
230   confirmPositions(0, 300, 800, 1500);
231   
232   // so, if we now resize to _really really tiny_
233   SC.RunLoop.begin();
234   splitView.adjust('width', 400);
235   splitView.adjust('height', 400);
236   SC.RunLoop.end();
237   
238   // we should see that even the RESIZE_MANUAL has resized.
239   confirmPositions(0, 100, 200, 300);
240   
241   // and if we shrink further still...
242   SC.RunLoop.begin();
243   splitView.adjust('width', 300);
244   splitView.adjust('height', 300);
245   SC.RunLoop.end();
246   
247   // there should be change, because the left side collapses.
248   confirmPositions(0, 0, 100, 200);
249   
250   // and if we shrink further still...
251   SC.RunLoop.begin();
252   splitView.adjust('width', 200);
253   splitView.adjust('height', 200);
254   SC.RunLoop.end();
255   
256   // there should be no change
257   confirmPositions(0, 0, 100, 200);
258   
259   // and if we shrink further still, way beyond the smallest it can get...
260   SC.RunLoop.begin();
261   splitView.adjust('width', 100);
262   splitView.adjust('height', 100);
263   SC.RunLoop.end();
264   
265   // there should be no change
266   confirmPositions(0, 0, 100, 200);
267 });
268 
269 test("Fitting to fill disabled", function() {
270   splitView.set('shouldResizeChildrenToFit', NO);
271   
272   // so, first thing: Resize, and ensure things are where they started.
273   SC.RunLoop.begin();
274   // note: we test both horizontal and vertical
275   splitView.adjust('width', 1600);
276   splitView.adjust('height', 1600);
277   SC.RunLoop.end();
278   
279   confirmPositions(0, 100, 400, 700);
280   
281   // now, move one, and see that compensation, etc. still works properly
282   splitView.adjustPositionForChild(middleLeft, 500);
283   confirmPositions(0, 500, 600, 700);
284   
285   // move back, and make sure compensation still took effect
286   splitView.adjustPositionForChild(middleLeft, 100);
287   confirmPositions(0, 100, 600, 700);
288   
289   // disable compensation and move again
290   middleLeft.set('compensatesForMovement', NO);
291   middleRight.set('compensatesForMovement', NO);
292   right.set('compensatesForMovement', NO);
293   
294   splitView.adjustPositionForChild(middleLeft, 500);
295   confirmPositions(0, 500, 1000, 1100);
296   
297   // disable indirect for last item and check that movement cannot occur
298   right.set('allowsIndirectAdjustments', NO);
299   
300   var result = splitView.adjustPositionForChild(middleLeft, 700);
301   equals(result, 500, "Should not be able to move (last does not allow indirect adjustments)");
302   confirmPositions(0, 500, 1000, 1100);
303   
304   result = splitView.adjustPositionForChild(middleLeft, 200);
305   equals(result, 500, "Should not be able to move (last does not allow indirect adjustments)");
306   confirmPositions(0, 500, 1000, 1100);
307 });
308 
309 }
310 
311 setupTestSuite(SC.LAYOUT_HORIZONTAL);
312 setupTestSuite(SC.LAYOUT_VERTICAL);
