 24 function timeShouldBeEqualToHash(t, h, message) {
 25   if (h === undefined) h = options;
 26   if (h.timezone === undefined) h.timezone = SC.DateTime.timezone;
 27   if (message === undefined) message = "%@ of time should be equal to hash";
 28   
 29   if (t === null) {
 30     ok(false, 'Time should not be null');
 31     return;
 32   }
 33     
 34   equals(t.get('year'), h.year , message.fmt('year'));
 35   equals(t.get('month'), h.month, message.fmt('month'));
 36   equals(t.get('day'), h.day, message.fmt('day'));
 37   equals(t.get('hour'), h.hour, message.fmt('hour'));
 38   equals(t.get('minute'), h.minute, message.fmt('minute'));
 39   equals(t.get('second'), h.second, message.fmt('second'));
 40   equals(t.get('millisecond'), h.millisecond, message.fmt('millisecond'));
 41   equals(t.get('timezone'), h.timezone, message.fmt('timezone'));
 42 }

 44 function formatTimezone(offset) {
 45   var modifier = offset < 0 ? '+' : '-';
 46   offset = Math.abs(offset);
 47   var minutes = offset % 60;
 48   var hours = (offset - minutes) / 60;
 49   return modifier + SC.DateTime._pad(hours) + ':' + SC.DateTime._pad(minutes);
 50 }

 52 test('_toMilliseconds()', function() {
 53   var originalTimezone = options.timezone;
 54   var originalHour = options.hour;
 55 
 56   dt = SC.DateTime;
 57   timezone = 300;
 58   startTime = 1264958583000; // Sun, 31 Jan 2010 17:23:03 GMT (a randomly chosen time with which to re-init the internal date object for more robustness)
 59 
 60   // Check the default behavior
 61   equals(dt._toMilliseconds(null, ms, timezone), ms, "Should equal start milliseconds when no options hash provided");
 62   equals(dt._toMilliseconds({}, ms, timezone), ms, "Should equal start milliseconds when empty options hash provided");
 63   
 64   // Test a completely defined date/time hash with no specified start milliseconds.
 65   equals(dt._toMilliseconds(options, null, options.timezone), ms, "Milliseconds should express the parsed options hash");
 66 
 67   // Now specify the same time in timezone (+60), one hour west of the prime meridian.
 68   // Pass in 'startTime' to force a reset of the internal date object so we can be sure we're not
 69   // succeeding because of old values.
 70   options.hour = originalHour - 1;
 71   equals(dt._toMilliseconds(options, startTime, options.timezone + 60), ms, "Should get same milliseconds when expressing time in westerly time zone");
 72 
 73   // Now specify the same time in timezone (-60), one hour east of the prime meridian
 74   options.hour = originalHour + 1;
 75   equals(dt._toMilliseconds(options, startTime, options.timezone - 60), ms, "Should get same milliseconds when expressing time in easterly time zone");
 76 
 77   // Now start at the original 1985 time, but modify only the hour as specified in time zone 60.
 78   options = { hour: originalHour - 1 };
 79   equals(dt._toMilliseconds(options, ms, originalTimezone + 60, NO), ms, "Should get same result modifying only the hour as expressed in westerly time zone");
 80 
 81   // Now do the same thing the other way
 82   options = { hour: originalHour + 1 };
 83   equals(dt._toMilliseconds(options, ms, originalTimezone - 60, NO), ms, "Should get same result modifying only the hour as expressed in westerly time zone");

148 test('Adjust with hashes expressed in various time zones', function() {
149   timezones.forEach(function(timezone) {
150     var newHour;
151 
152     options.timezone = timezone;
153     dt = SC.DateTime.create(options);
154 
155     // According to Date specs, must specify all three of year, month, and date if we specify one of them,
156     // for the calculation to be correct.  Calling adjust to change only one can have
157     // unpredictable results, depending on what the other two values already were.
158     timeShouldBeEqualToHash(dt.adjust({ year: 2005, month: 9, day: 30 }), {year: 2005, month: 9, day: 30, hour: options.hour, minute: options.minute, second: options.second, millisecond: options.millisecond, timezone: timezone});
159 
160     // Setting only the hour should cascade minute, second, and millisecond to 0, etc
161     timeShouldBeEqualToHash(dt.adjust({ hour:         3 }), { year: options.year, month: options.month, day: options.day, hour: 3, minute: 0, second: 0, millisecond: 0, timezone: timezone});
162     timeShouldBeEqualToHash(dt.adjust({ minute:       1 }), { year: options.year, month: options.month, day: options.day, hour: options.hour, minute: 1, second: 0, millisecond: 0, timezone: timezone});
163     timeShouldBeEqualToHash(dt.adjust({ second:      12 }), { year: options.year, month: options.month, day: options.day, hour: options.hour, minute: options.minute, second: 12, millisecond: 0, timezone: timezone});
164     timeShouldBeEqualToHash(dt.adjust({ millisecond: 18 }), { year: options.year, month: options.month, day: options.day, hour: options.hour, minute: options.minute, second: options.second, millisecond: 18, timezone: timezone});
165     
166     // Test taking each to time zone 0.  Manually calculate what the hour should be
167     // then test that a call to get() returns that value.
168     newHour = Math.floor((options.hour + 48 + (timezone / 60)) % 24); // small hack -- add 48 hours to ensure positive results when adding negative time zone offsets (doesn't affect the calculation since we mod by 24)
