  8 /**
  9   @namespace
 10   Child views of SplitViews should use this mixin to get their positioning
 11   logic and default settings.
 12   
 13   By default, SplitView merely sets the position and size properties on its
 14   child views. This mixin observes the position and size properties and 
 15   calls adjust().
 16 */
 17 SC.SplitChild = 
 18   /** @scope SC.SplitChild.prototype */{
 19   /**
 20    * Set to YES if your Split Child is a divider view.
 21    *
 22    * @type Boolean
 23   */
 24   isSplitDivider: NO,
 25   
 26   /**
 27    * The minimum size for the SC.SplitView child. This view will
 28    * be unable to be resized smaller than this size.
 29    *
 30    * @default 100
 31    *
 32    * @type {Number}
 33   */
 34   minimumSize: 100,
 35 
 36   /**
 37    * The maximum size for the SC.SplitView child. This view will
 38    * be unable to be resized to a size larger than this value.
 39    *
 40    * If undefined, there is no limit.
 41    * 
 42    * @default 100
 43    *
 44    * @type {Number}
 45   */
 46   maximumSize: undefined,
 47   
 48   /**
 49    * The current size of the SC.SplitView child. Use this to set the default
 50    * size.
 51    *
 52    * If you specify a size, autoResizeStyle will default to SC.RESIZE_MANUAL;
 53    * if you don't, SplitView will assume you want it to fill the available space
 54    * and use SC.RESIZE_AUTOMATIC.
 55    *
 56    * @default undefined
 57    *
 58    * @type Number
 59   */
 60   size: undefined,
 61   
 62   /**
 63    * [RO] The current position of the SC.SplitView. This is read-only, and is set
 64    * by the SplitView as it positions the views.
 65    *
 66    * @type Number
 67   */
 68   position: 0,
 69   
 70   /**
 71    * An amount to add to the position when adjusting layout.
 72    * 
 73    * For example, if SC.SplitView positions this view at 100, and
 74    * positionOffset is -5, the view's layout will will have a position of 95.
 75    *
 76    * This can be used in conjunction with sizeOffset to make dividers
 77    * overlap the other views in the SplitView to have a larger grab area.
 78    *
 79    * TODO: Get this value from theme.
 80    *
 81    * @type Number
 82   */
 83   positionOffset: SC.propertyFromRenderDelegate('splitPositionOffset', 0),
 84   
 85   /**
 86    * An amount to add to the assigned size when adjusting layout.
 87    * 
 88    * For example, if SC.SplitView gives this view a size of 1, but the
 89    * sizeOffset is 10, the view's layout will be given a width or height of 11.
 90    *
 91    * This can be used in conjunction with positionOffset to make dividers 
 92    * overlap the other views in the SplitView to have a larger grab area.
 93    *
 94    * TODO: Get this value from theme.
 95    *
 96    * @type Number
 97   */
 98   sizeOffset: SC.propertyFromRenderDelegate('splitSizeOffset', 0),
 99 
100   /**
101    * If YES, the SC.SplitView can collapse this view when the user
102    * double-clicks an adjacent divider, or when the view is resized
103    * to a size smaller than the collapseAtSize.
104    *
105    * @type {Boolean}
106   */
107   canCollapse: NO,
108   
109   /**
110    * If the user attempts to resize the view to a size below this number,
111    * the view will collapse.
112    *
113    * If undefined, the view cannot be collapsed due to a resize.
114    *
115    * @type {Number}
116   */
117   collapseAtSize: undefined,
118 
119   /**
120    * When (and if) the view should automatically resize due to the SplitView changing size.
121    *
122    * - SC.RESIZE_AUTOMATIC: always resize when the SplitView changes size.
123    * - SC.RESIZE_MANUAL: resize only when the user moves a divider, or all
124    *   SC.RESIZE_AUTOMATIC views have already been resized as much as possible.
125    * - SC.FIXED_SIZE: Never resize.
126    *
127    * If you specify an autoResizeStyle, it will be used. If you leave it at `undefined`, 
128    * it will look at `size`: if a `size` is supplied, it will use `SC.RESIZE_MANUAL`; otherwise
129    * it will use `SC.RESIZE_AUTOMATIC`.
130    *
131    * @default based on size
132    * @type {AutoResizeStyle}
133   */
134   autoResizeStyle: undefined,
135 
136   /**
137    * If NO, moving the divider before this view will not resize the view, but
138    * instead, move it—causing further views to move as well. Most SplitView
139    * children will want this set to YES.
140    *
141    * @type Boolean
142   */
143   compensatesForMovement: YES,
144   
145   /**
146    * When NO, the view can only be adjusted when the user drags a divider
147    * immediately adjacent to it. If YES, the view can be adjusted indirectly
148    * from earlier dividers as well.
149    *
150    * For example, assume you have a 3-pane setup. Increasing the size of left pane will
151    * resize the middle one until its minimum size is reached. SplitView will then
152    * try to adjust the right pane. If the right pane has allowsIndirectAdjustments
153    * set to YES, SplitView will shrink the right pane to make way. If NO, the
154    * left pane will be unable to resize further.
155    *
156    * Default: YES.
157    *
158    * @type {Boolean}
159   */
160   allowsIndirectAdjustments: YES,
161   
162   
163   //
164   // Positioning logic
165   //
166   _scsvc_positionOrSizeDidChange: function() {
167     this.invokeOnce('splitChildLayoutDidChange');
168   }.observes('position', 'size'),
169   
170   /**
171    * Called when either the position or size of the child has changed, and layout
172    * needs to be updated accordingly. You may override this method to take into
173    * account any custom layout. The default handles the position- and sizeOffset
174    * properties by adding them to the position and size, respectively.
175    *
176    * For instance, while the default implementation changes left/right/width/height
177    * to fill in one direction and fit in the allocated position in the other, you could
178    * make it only set left/width or top/height.
179    *
180   */
181   splitChildLayoutDidChange: function() {
182     var split = this.get('splitView');
183     if (!split) return;
184     
185     var position = this.get('position') + this.get('positionOffset'),
186         size = this.get('size') + this.get('sizeOffset');
187     
188     
189     if (split.get('layoutDirection') === SC.LAYOUT_HORIZONTAL) {
190       this.set('layout', {
191         left: position,
192         width: size,
193         top: 0, bottom: 0
194       });
195     } else {
196       this.set('layout', {
197         top: position,
198         height: size,
199         right: 0, left: 0
200       });
201     }
202   },
203   
204   
205   /**
206    * The SplitView that contains the child views to be adjusted.
207    *
208    * This is computed to be the first SplitView found in a search
209    * up the view hierarchy. You can substitute your own SplitView.
210    *
211    * Note: this property is duplicated between this mixin and the SplitThumb
212    * mixin, as both have qualities that need the controlling SplitView.
213    *
214    * @property SC.SplitView
215   */
216   splitView: function() {
217     var view = this ;
218     while (view && !view.isSplitView) view = view.get('parentView') ;
219     return view ;
220   }.property('parentView').cacheable(),
221   
222   /**
223    * The layoutDirection of the SplitView. This is observed so that we
224    * can update positioning if the layoutDirection changes but the position
225    * and size properties do not.
226    *
227    * @type {LayoutDirection}
228   */
229   // NOTE: While an edge case, this is implemented because it makes it _much_
230   // easier to write the sample in the Test Controls app.
231   splitViewLayoutDirection: null,
232   splitViewLayoutDirectionBinding: '*splitView.layoutDirection',
233   
234   splitViewLayoutDirectionDidChange: function() {
235     this.invokeOnce('splitChildLayoutDidChange');
236   }.observes('splitViewLayoutDirection')
237 };
