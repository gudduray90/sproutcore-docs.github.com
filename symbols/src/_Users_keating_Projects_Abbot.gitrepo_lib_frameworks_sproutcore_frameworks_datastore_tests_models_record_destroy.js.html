  8 var MyFoo = null, callInfo ;
  9 module("SC.Record#destroy", {
 10   setup: function() {
 11     SC.RunLoop.begin();
 12     MyApp = SC.Object.create({
 13       store: SC.Store.create()
 14     })  ;
 15   
 16     MyApp.Foo = SC.Record.extend();
 17     MyApp.json = { 
 18       foo: "bar", 
 19       number: 123,
 20       bool: YES,
 21       array: [1,2,3] 
 22     };
 23     
 24     MyApp.foo = MyApp.store.createRecord(MyApp.Foo, MyApp.json);
 25     
 26     // modify store so that everytime refreshRecords() is called it updates 
 27     // callInfo
 28     callInfo = null ;
 29     MyApp.store.__orig = MyApp.store.destroyRecord;
 30     MyApp.store.destroyRecord = function(records) {
 31       callInfo = SC.A(arguments) ; // save method call
 32       MyApp.store.__orig.apply(MyApp.store, arguments); 
 33     };
 34     SC.RunLoop.end();
 35   }
 36 });
 37 
 38 test("calling destroy on a newRecord will mark the record as destroyed and calls destroyRecords on the store", function() {
 39   equals(MyApp.foo.get('status'), SC.Record.READY_NEW, 'precond - status is READY_NEW');
 40   SC.RunLoop.begin();
 41   MyApp.foo.destroy();
 42   SC.RunLoop.end();
 43   same(callInfo, [null, null, MyApp.foo.storeKey], 'destroyRecords() should not be called');
 44   
 45   equals(MyApp.foo.get('status'), SC.Record.DESTROYED_CLEAN, 'status should be SC.Record.DESTROYED_CLEAN');
 46 });
 47 
 48 test("calling destroy on existing record should call destroyRecord() on store", function() {
 49 
 50   // Fake it till you make it...
 51   MyApp.store.writeStatus(MyApp.foo.storeKey, SC.Record.READY_CLEAN)
 52     .dataHashDidChange(MyApp.foo.storeKey, null, YES);
 53     
 54   equals(MyApp.foo.get('status'), SC.Record.READY_CLEAN, 'precond - status is READY CLEAN');
 55   
 56   SC.RunLoop.begin();
 57   MyApp.foo.destroy();
 58   SC.RunLoop.end();
 59   
 60   same(callInfo, [null, null, MyApp.foo.storeKey], 'destroyRecord() should not be called');
 61   equals(MyApp.foo.get('status'), SC.Record.DESTROYED_DIRTY, 'status should be SC.Record.DESTROYED_DIRTY');
 62 });
 63 
 64 test("calling destroy on a record that is already destroyed should do nothing", function() {
 65 
 66   // destroy once
 67   SC.RunLoop.begin();
 68   MyApp.foo.destroy();
 69   SC.RunLoop.end();
 70   equals(MyApp.foo.get('status'), SC.Record.DESTROYED_CLEAN, 'status should be DESTROYED_CLEAN');
 71   
 72   SC.RunLoop.begin();
 73   MyApp.foo.destroy();
 74   SC.RunLoop.end();
 75   equals(MyApp.foo.get('status'), SC.Record.DESTROYED_CLEAN, 'status should be DESTROYED_CLEAN');
 76 });
 77 
 78 test("should return receiver", function() {
 79   equals(MyApp.foo.destroy(), MyApp.foo, 'should return receiver');
 80 });
 81 
 82 test("destroy should update status cache", function() {
 83   var st = MyApp.foo.get('status');
 84   ok(st !== SC.Record.DESTROYED_CLEAN, 'precond - foo should not be destroyed');
 85 
 86   SC.RunLoop.begin();
 87   MyApp.foo.destroy();
 88   equals(MyApp.foo.get('status'), SC.Record.DESTROYED_CLEAN, 'status should be DESTROYED_CLEAN immediately when destroyed directly by record');
 89   SC.RunLoop.end();
 90 });
