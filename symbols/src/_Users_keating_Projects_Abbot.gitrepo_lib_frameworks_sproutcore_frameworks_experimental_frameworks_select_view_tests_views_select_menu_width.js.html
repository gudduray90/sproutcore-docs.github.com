1 // Tests that the minimum menu width changes with the size of the SelectView
  2 //
  3 var View, pane, obj;
  4 
  5 module("SelectView - Minimum Menu Width", {
  6   setup: function() {
  7     pane = SC.Pane.create();
  8 
  9     obj = SC.Object.create({title: 'Lost', value: 'Found'});
 10 
 11     View = SC.SelectView.extend({
 12       items: [
 13         {title: 'Hello', value: 'hi'},
 14         {title: 'Yo', value: 'Jimbo'},
 15       ],
 16 
 17  18       itemTitleKey: 'title',
 19       itemValueKey: 'value'
 20     });
 21   },
 22 
 23   teardown: function() {
 24     
 25   }
 26 });
 27 
 28 function createView(settings) {
 29   var view = View.create(settings || {});
 30   pane.appendChild(view);
 31   return view;
 32 }
 33 
 34 test("Menu width changes with minimumMenuWidth", function() {
 35   SC.RunLoop.begin();
 36   var view = createView({ minimumMenuWidth: 500 });
 37   view.showMenu();
 38   SC.RunLoop.end();
 39 
 40   equals(view.get('menu').get('frame').width, 500, "Width is 500");
 41 
 42   SC.RunLoop.begin();
 43   view.set('minimumMenuWidth', 1000);
 44   SC.RunLoop.end();
 45 
 46   equals(view.get('menu').get('frame').width, 1000, "Width has changed to 1000");
 47 
 48   view.hideMenu();
 49 
 50 });
