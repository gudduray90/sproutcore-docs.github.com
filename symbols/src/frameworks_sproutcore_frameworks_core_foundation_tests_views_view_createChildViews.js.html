 14 test("calls createChildView() for each class or string in childViews array", function() {
 15   var called = [];
 16   var v = SC.View.create({
 17     childViews: [
 18       SC.View.extend({ key: 0 }), // class - should be called
 19       SC.View.create({ key: 1 }), // instance - will be called
 20       'customClassName'           // string - should be called
 21     ],
 22 
 23     // this should be used for the 'customClassName' item above
 24     customClassName: SC.View.extend({ key: 2 }),
 25 
 26     // patch to record results...
 27     createChildView: function(childView) {
 28       if(childView.isClass) {
 29         called.push(childView.prototype.key);
 30       } else {
 31         called.push(childView.key);
 32       }
 33       return sc_super();
 34     }
 35   });
 36 
 37   // createChildViews() is called automatically during create.
 38   same(called, [0,1,2], 'called createChildView for correct children');
 39 
 40   // make sure childViews array is correct now.
 41   var cv = v.childViews, len = cv.length, idx;
 42   for(idx=0;idx<len;idx++) {
 43     equals(cv[idx].key, idx, 'has correct index key');
 44     ok(cv[idx].isObject, 'isObject - %@'.fmt(cv[idx]));
 45   }
 46 });

 48 test("should not error when there is a dud view name in childViews list.", function() {
 49   var called = [];
 50   var v = SC.View.create({
 51     childViews: [
 52       'nonExistantClassName',       // string - should NOT be called
 53       null,                       // null - should NOT be called
 54       '',                         // empty string - should NOT be called
 55       'customClassName'          // string - should be called
 56     ],
 57     // this should be used for the 'customClassName' item above
 58     customClassName: SC.View.extend({ key: 2 }),
 59 
 60     // patch to record results...
 61     createChildView: function(childView) {
 62       called.push(childView.prototype.key);
 63       ok(childView.isClass, "childView: %@ isClass".fmt(childView));
 64       return sc_super();
 65     }
 66   });
 67 
 68   // createChildViews() is called automatically during create.
 69   same(called, [2], 'called createChildView for correct children');
 70   equals(v.getPath('childViews.length'), 1, "The childViews array should not contain any invalid childViews after creation.");
 71 });

 98 var view, myViewClass ;
 99 module("SC.View#createChildView", {
100   setup: function() {
101     view = SC.View.create({ page: SC.Object.create() });
102     myViewClass = SC.View.extend({ isMyView: YES, foo: 'bar' });
103   }
104 });
105 
106 test("should create view from class with any passed attributes", function() {
107   var v = view.createChildView(myViewClass, { foo: "baz" });
108   ok(v.isMyView, 'v is instance of myView');
109   equals(v.foo, 'baz', 'view did get custom attributes');
110 });
111 
112 test("should set newView.parentView to receiver", function() {
113   var v = view.createChildView(myViewClass) ;
114   equals(v.get('parentView'), view, 'v.parentView == view');
115 });
116 
117 test("should set newView.page to receiver.page unless custom attr is passed", function() {
118   var v = view.createChildView(myViewClass) ;
119   equals(v.get('page'), view.get('page'), 'v.page == view.page');
120 
121   var myPage = SC.Object.create();
122   v = view.createChildView(myViewClass, { page: myPage }) ;
123   equals(v.get('page'), myPage, 'v.page == custom page');
124 });
