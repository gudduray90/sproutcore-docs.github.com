  9 var source, indexes, observer, obj ; // base array to work with
 10 module("SC.RangeObserver#update", {
 11   setup: function() {
 12     
 13     // create array with 5 SC.Object's in them
 14     source = [1,2,3,4,5].map(function(x) {
 15       return SC.Object.create({ item: x, foo: "bar" }) ;
 16     }, this); 
 17 
 18     indexes = SC.IndexSet.create(2,2); // select 2..3
 19     
 20     observer = SC.Object.create({
 21       
 22       callCount: 0, 
 23       
 24       rangeDidChange: function() { 
 25         this.callCount++;
 26       }
 27       
 28     });
 29 
 30     obj = SC.RangeObserver.create(source, indexes, observer, observer.rangeDidChange, "context", YES);
 31     
 32   }
 33 });
 34 
 35 test("returns receiver", function() {
 36   ok(obj === obj.update(source, SC.IndexSet.create()), 'should return receiver');
 37 });
 38 
 39 test("switches to observing new range - no previous updated", function() {
 40   obj.update(source, SC.IndexSet.create(0));
 41   observer.callCount = 0 ;
 42   
 43   var len = source.length, idx;
 44   for(idx=0;idx<len;idx++) source[idx].set('foo', 'baz');
 45   
 46   // since new index set length is different use this as a proxy to verify
 47   // that range changed
 48   equals(observer.callCount, 1, 'range observer should fire on new range');
 49 });
 50 
 51 test("switches to observing new range - previously updated", function() {
 52   var len = source.length, idx;
 53   for(idx=0;idx<len;idx++) source[idx].set('foo', 'baz');
 54   observer.callCount = 0 ;
 55 
 56   obj.update(source, SC.IndexSet.create(0));
 57   observer.callCount = 0 ;
 58   
 59   for(idx=0;idx<len;idx++) source[idx].set('foo', 'bar');
 60   
 61   // since new index set length is different use this as a proxy to verify
 62   // that range changed
 63   equals(observer.callCount, 1, 'range observer should fire on new range');
 64 });
