 12 // Create a fake content array.  Generates a list with whatever length you
 13 // want of objects with a title based on the index.  Cannot mutate.
 14 var ContentArray = SC.Object.extend(SC.Array, {
 15 
 16   length: 0,
 17 
 18   objectAt: function(idx) {
 19     if (idx >= this.get('length')) { return undefined; }
 20 
 21     var content = this._content, ret ;
 22     if (!content) { content = this._content = []; }
 23 
 24     ret = content[idx];
 25     if (!ret) {
 26       ret = content[idx] = SC.Object.create({
 27         title: "ContentItem %@".fmt(idx),
 28         isDone: (idx % 3)===0,
 29         unread: (Math.random() > 0.5) ? Math.floor(Math.random() * 100) : 0
 30       });
 31     }
 32 
 33     return ret ;
 34   }
 35 });

 37 var pane = SC.ControlTestPane.design()
 38   .add("basic", SC.ScrollView.design({
 39     borderStyle: SC.BORDER_NONE,
 40     layout: { left: 0, right: 0, top: 0, height: 300 },
 41     hasHorizontalScroller: NO,
 42     contentView: SC.ListView.design({
 43       content: ContentArray.create({ length: 5 }),
 44       contentValueKey: "title",
 45       contentCheckboxKey: "isDone",
 46       contentUnreadCountKey: "unread",
 47       rowHeight: 20
 48     })
 49   }));

 54 test("drag on default list view", function() {
 55   var ev,
 56     itemView,
 57     layer,
 58     listView = pane.view("basic").get('contentView');
 59 
 60   itemView = listView.itemViewForContentIndex(0);
 61   layer = itemView.get('layer');
 62   ev = SC.Event.simulateEvent(layer, 'mousedown');
 63   SC.Event.trigger(layer, 'mousedown', [ev]);
 64 
 65   ev = SC.Event.simulateEvent(layer, 'mousemove');
 66   SC.Event.trigger(layer, 'mousemove', [ev]);
 67 
 68   equals(listView.get('dragContent'), null, 'dragContent should not be set, because the default implementation should prevent dragging');
 69 
 70   // Clean up
 71   ev = SC.Event.simulateEvent(layer, 'mouseup');
 72   SC.Event.trigger(layer, 'mouseup', [ev]);
 73 });

 76 test("drag on list view with SC.DROP_ON support", function() {
 77   var ev,
 78     itemView,
 79     layer,
 80     listView = pane.view("basic").get('contentView');
 81 
 82   // Configure the view to accept drop on.
 83   listView.set('canReorderContent', YES);
 84   listView.set('isDropTarget', YES);
 85   listView.set('delegate', SC.Object.create(SC.CollectionViewDelegate, {
 86      collectionViewValidateDragOperation: function(view, drag, op, proposedInsertionIndex, proposedDropOperation) {
 87       return SC.DRAG_ANY;
 88     }
 89   }));
 90 
 91   itemView = listView.itemViewForContentIndex(0);
 92   layer = itemView.get('layer');
 93   ev = SC.Event.simulateEvent(layer, 'mousedown');
 94   SC.Event.trigger(layer, 'mousedown', [ev]);
 95 
 96   var f = function() {
 97     var itemView2,
 98       point;
 99 
100     SC.RunLoop.begin();
101     ev = SC.Event.simulateEvent(layer, 'mousemove');
102     SC.Event.trigger(layer, 'mousemove', [ev]);
103 
104     equals(listView.get('dragContent').content, listView.get('content'), "dragContent.content should be equal to the ListView's content");
105     ok(listView.get('dragContent').indexes.isEqual(SC.IndexSet.create(0)), "dragContent.indexes should be equal to indexes equal to [{0}]");
106     SC.RunLoop.end();
107 
108     // Drag over 2nd item
109     itemView2 = listView.itemViewForContentIndex(1);
110     layer = itemView2.get('layer');
111     point = SC.offset(layer);
112 
113     ev = SC.Event.simulateEvent(layer, 'mousemove', { pageX: point.x + 1, pageY: point.y + 1 });
114     SC.Event.trigger(layer, 'mousemove', [ev]);
115 
116 
117     ok(itemView2.get('isDropTarget'), "second list item should have isDropTarget set to true");
118 
119     // This test only works because SC.ListItemView supports adding the class to match the property.
120     ok(itemView2.$().hasClass('drop-target'), "second list item should add drop-target class");
121 
122     // Drag over 3rd item
123     itemView = listView.itemViewForContentIndex(2);
124     layer = itemView.get('layer');
125     point = SC.offset(layer);
126 
127     ev = SC.Event.simulateEvent(layer, 'mousemove', { pageX: point.x + 1, pageY: point.y + 1 });
128     SC.Event.trigger(layer, 'mousemove', [ev]);
129 
130     ok(itemView.get('isDropTarget'), "third list item should have isDropTarget set to true");
131     ok(!itemView2.get('isDropTarget'), "second list item should not have isDropTarget set to true");
132 
133     // This test only works because SC.ListItemView supports adding the class to match the property.
134     ok(itemView.$().hasClass('drop-target'), "third list item should add drop-target class");
135     ok(!itemView2.$().hasClass('drop-target'), "second list item should not add drop-target class");
136 
137     // Clean up
138     ev = SC.Event.simulateEvent(layer, 'mouseup');
139     SC.Event.trigger(layer, 'mouseup', [ev]);
140 
141     window.start();
142   };
143 
144   stop(); // stops the test runner
145   setTimeout(f, 200);
146 });
