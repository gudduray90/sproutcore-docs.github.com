 12 module("SC.Store#commitRecord", {

 15     ds = SC.DataSource.create({
 16       
 17       callCount: 0,
 18       
 19       commitRecords: function(store, toCreate, toUpdate, toDestroy, params) {
 20         this.toCreate = toCreate;
 21         this.toUpdate = toUpdate;
 22         this.toDestroy = toDestroy;
 23         this.params = params;
 24         this.callCount++;
 25       },
 26       
 27       reset: function() {
 28         this.toCreate = this.toUpdate = this.toDestroy = this.params = null;

 30       },

 36         
 37         if (toCreate !== undefined) {
 38           same(this.toCreate, toCreate, 'expect toCreate to have items');
 39         }

 43         }
 44         
 45         if (toDestroy !== undefined) {
 46           same(this.toDestroy, toDestroy, 'expect toDestroy to have items');
 47         }
 48
