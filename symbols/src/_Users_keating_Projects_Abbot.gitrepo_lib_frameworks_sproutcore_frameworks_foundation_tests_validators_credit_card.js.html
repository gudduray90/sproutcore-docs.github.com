 11 module("credit_card validation - fieldValueForObject()");
 12 test("Expects a string of 16 digits.  Will split into groups of 4 for display.",function(){
 13 	var obj=SC.Validator.CreditCard.create();
 14 	var string = "1234567890123456";
 15 	var grouped_string = obj.fieldValueForObject("1234567891234567");
 16 	equals(grouped_string,"1234 5678 9123 4567");	
 17 });
 18 test("Removes all whitespace or dashes to make a single string",function(){
 19 	var obj=SC.Validator.CreditCard.create();
 20 	var string = "1234 5678-9012-3456";
 21 	var clean_string = obj.objectForFieldValue(string);
 22 	equals(clean_string,"1234567890123456");	
 23 });
 24 
 25 test("Checks all valid card numbers or dashes to make a single string",function(){
 26 	var obj=SC.Validator.CreditCard.create();
 27 	var invalid_cc = "1234 5678-9012-3456";
 28 	equals(obj.checkNumber(invalid_cc),false);
 29 	var valid_cc = "4111 1111 1111 1111";
 30 	equals(obj.checkNumber(valid_cc),true);	
 31 });
