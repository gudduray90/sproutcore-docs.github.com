 83 
 84   // setup special binding
 85   fromObject = SC.Object.create({
 86     value1: 'value1',
 87     value2: 'value2'
 88   });
 89 
 90   toObject = SC.Object.create({
 91     value1: 'value1',
 92     value2: 'value2',
 93 
 94     callCount: 0,
 95 
 96     observer: function () {
 97       equals(this.get('value1'), 'CHANGED', 'value1 when observer fires');
 98       equals(this.get('value2'), 'CHANGED', 'value2 when observer fires');
 99       this.callCount++;
100     }.observes('value1', 'value2')
101   });
102

104   toObject.bind('value2', fromObject, 'value2');

107   // fire after bindings are done flushing.
108   fromObject.set('value1', 'CHANGED').set('value2', 'CHANGED');
109   SC.Binding.flushPendingChanges();
110 
111   equals(toObject.callCount, 2, 'should call observer twice');
