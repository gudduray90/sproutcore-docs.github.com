 14 module("basic object binding", {
 15 
 16   setup: function () {
 17     fromObject = SC.Object.create({ value: 'start' }) ;
 18     midObject = SC.Object.create({ value: 'middle' });
 19     toObject = SC.Object.create({ value: 'end' }) ;
 20     binding1 = SC.Binding.from("value", fromObject).to("value", midObject).connect() ;
 21     binding2 = SC.Binding.from("value", midObject).to("value", toObject).connect();
 22     SC.Binding.flushPendingChanges() ; // actually sets up up the connection
 23   },
 24 
 25   teardown: function () {
 26     fromObject.destroy();
 27     midObject.destroy();
 28     toObject.destroy();
 29     fromObject = midObject = toObject = binding1 = binding2 = null;
 30   }
 31 });

 82 test("suspended observing during bindings", function() {
 83 
 84   // setup special binding
 85   fromObject = SC.Object.create({
 86     value1: 'value1',
 87     value2: 'value2'
 88   });
 89 
 90   toObject = SC.Object.create({
 91     value1: 'value1',
 92     value2: 'value2',
 93 
 94     callCount: 0,
 95 
 96     observer: function() {
 97       equals(this.get('value1'), 'CHANGED', 'value1 when observer fires');
 98       equals(this.get('value2'), 'CHANGED', 'value2 when observer fires');
 99       this.callCount++;
100     }.observes('value1', 'value2')
101   });
102 
103   toObject.bind('value1', fromObject, 'value1');
104   toObject.bind('value2', fromObject, 'value2');
105 
106   // change both value1 + value2, then  flush bindings.  observer should only
107   // fire after bindings are done flushing.
108   fromObject.set('value1', 'CHANGED').set('value2', 'CHANGED');
109   SC.Binding.flushPendingChanges();
110 
111   equals(toObject.callCount, 2, 'should call observer twice');
112 });

206 module("chained binding", {
207 
208   setup: function() {
209     first = SC.Object.create({ output: 'first' }) ;
210 
211     second = SC.Object.create({
212       input: 'second',
213       output: 'second',
214 
215       inputDidChange: function() {
216         this.set("output", this.get("input")) ;
217       }.observes("input")
218     }) ;
219 
220     third = SC.Object.create({ input: "third" }) ;
221 
222     binding1 = SC.Binding.from("output", first).to("input", second).connect() ;
223     binding2 = SC.Binding.from("output", second).to("input", third).connect() ;
224     SC.Binding.flushPendingChanges() ; // actually sets up up the connection
225   }
226 
227 });
228 
229 test("changing first output should propagate to third after flush", function() {
230   first.set("output", "change") ;
231   equals("change", first.get("output"), "first.output") ;
232   ok("change" !== third.get("input"), "third.input") ;
233 
234   var didChange = YES;
235   while(didChange) didChange = SC.Binding.flushPendingChanges() ;
236 
237   // bindings should not have bending changes
238   ok(!SC.Binding._changeQueue.contains(binding1), "the binding should not be in the _changeQueue");
239   ok(!SC.Binding._changeQueue.contains(binding2), "the binding should not be in the _changeQueue");
240 
241   equals("change", first.get("output"), "first.output") ;
242   equals("change", second.get("input"), "second.input") ;
243   equals("change", second.get("output"), "second.output") ;
244   equals("change", third.get("input"), "third.input") ;
245 });
