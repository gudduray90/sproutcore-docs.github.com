 10 module("PanelPane - Methods");
 11 
 12 function getViewCount() {
 13   var i = 0; for (key in SC.View.views) ++i;
 14   return i;
 15 }
 16 
 17 test("PanelPane destroy", function() {
 18   var start = getViewCount();
 19 
 20   var pane = SC.PanelPane.create({
 21     isModal: YES
 22   });
 23   pane.append();
 24   pane.remove();
 25   pane.destroy();
 26 
 27   var end = getViewCount();
 28   equals(start, end, "No extra views lying about after calling .destroy");
 29 });
