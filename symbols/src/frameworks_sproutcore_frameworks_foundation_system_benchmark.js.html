 11 /** @namespace
 12 
 13   This bit of meta-programming magic can install a benchmark handler on any
 14   object method.  When a benchmark is installed, the time required to execute
 15   the method will be printed to the console log every time the method is
 16   called.
 17 
 18   This class can be used to implement benchmarking.  To use this object, just
 19   call start() with a key name and end() with a keyname.  The benchmark will
 20   be logged.  If you set verbose = true, then benchmark will log every time it
 21   saves a bench.  Otherwise, it just keeps stats.  You can get the stats by
 22   calling report().
 23 
 24   Benchmark does not require anything other than the date.js class.  It also
 25   does not rely on SC.Object so that you can benchmark code in that area as
 26   well.
 27 
 28   The benchmark has three types of reports.
 29 
 30     - report(): Returns an abbreviated list with just the durations of the bench.
 31               Also, it averages multiple runs. Everything is reported on the top
 32               level only.
 33     - timelineReport(): Returns an list of benchmarks and sub-benchmarks. If the
 34                       the globalStartTime is set, then it will show relative
 35                       time from that time.
 36     - timelineChart(): Displays a chart of all the benchmarks (not sub-benchmarks)
 37                      relative to the first time capture or to the globalStartTime.
 38                      Hide this by calling hideChart()
 39 
 40   @since SproutCore 1.0
 41 */

 62   /**
 63     Events are a way of assigning specific, individual times to names, rather than
 64     durations of time. A benchmark event can only occur once—if it occurs again, it
 65     will overwrite the old event.
 66 
 67     The purpose of events is different than the purpose for normal benchmarks. Normal
 68     benchmarks may be used to benchmark a specific process, and may do so repeatedly;
 69     events, conversely, are usually used to benchmark things like startup time, and
 70     occur only once. For instance, an 'event' is registered when the document is ready.
 71 
 72     Events are kept as a hash of names to timestamps. To add an event, just set it:
 73 
 74         SC.Benchmark.events['myEventName'] = new Date().getTime();
 75 
 76         // Or, more conveniently:
 77         SC.Benchmark.addEvent('myEventName', [optional time]);
 78 
 79     On a timeline chart, events are usually represented as lines rather than bars. However,
 80     if you add eventNameStart and eventNameEnd, they will be automatically added as standard
 81     benchmarks.
 82 
 83     This is useful when adding preload events to SC.benchmarkPreloadEvents; as SC.Benchmark
 84     does not yet exist, you cannot call .start() and .end(), but adding the items to
 85     SC.benchmarkPreloadEvents will ensure they are included.
 86 
 87     @type Hash
 88     @default {}
 89   */
 90   events: {},

 98       - *name*: an optional longer name you assigned to the stat key.  Set this  using name().

116   /**
117     Adds an 'event' to the events hash.
118 
119     Unlike normal benchmarks, recorded with start/end and that represent a block of time,

123 
124     @param {String} name
125       A name that identifies the event. If addEvent is called again with the same name,
126       the previous call's timestamp will be overwritten.

135   /**
136     Call this method at the start of whatever you want to collect.
137     If a parentKey is passed, then you will attach the stat to the parent,
138     otherwise it will be on the top level. If topLevelOnly is passed, then
139     recursive calls to the start will be ignored and only the top level call
140     will be benchmarked.
141 
142     @param {String} key
143       A unique key that identifies this benchmark.  All calls to start/end
144       with the same key will be groups together.
145     @param {String} parentKey
146       A unique key that identifies the parent benchmark.  All calls to
147       start/end with the same key will be groups together.
148     @param {Boolean} topLevelOnly
149       If true then recursive calls to this method with the same key will be
150       ignored.
151     @param {Number} time
152       Only pass if you want to explicitly set the start time.  Otherwise the
153       start time is now.
154     @returns {String} the passed key
155   */
156   start: function(key, parentKey, time, topLevelOnly) {
157     if (!this.enabled) return ;
158 
159     var start = (time || Date.now()), stat;
160 
161     if (parentKey) stat = this._subStatFor(key, parentKey) ;
162     else stat = this._statFor(key) ;
163 
164     if (topLevelOnly && stat._starts.length > 0) stat._starts.push('ignore');
165     else stat._starts.push(start) ;
166 
167     stat._times.push({start: start, _subStats: {}});
168     return key;
169   },
170 
171   /**
172     Call this method at the end of whatever you want to collect.  This will
173     save the collected benchmark.
174 
175     @param {String} key
176       The benchmark key you used when you called start()
177     @param {String} parentKey
178       The benchmark parent key you used when you called start()
179     @param {Number} time
180       Only pass if you want to explicitly set the end time.  Otherwise start
181       time is now.
182   */
183   end: function(key, parentKey, time) {
184     var stat;
185     if (!this.enabled) return ;
186     if(parentKey)
187     {
188       stat = this._subStatFor(key, parentKey) ;
189     }
190     else
191     {
192       stat = this._statFor(key) ;
193     }
194     var start = stat._starts.pop() ;
195     if (!start) {
196       SC.Logger.log('SC.Benchmark "%@" ended without a matching start.  No information was saved.'.fmt(key));
197       return ;
198     }
199 
200     // top level only.
201     if (start == 'ignore') return ;
202 
203     var end = (time || Date.now()) ;
204     var dur = end - start;
205 
206     stat._times[stat._times.length-1].end = end;
207     stat._times[stat._times.length-1].dur = dur;
208 
209     stat.amt += dur ;
210     stat.runs++ ;
211 
212     if (this.verbose) this.log(key) ;
213   },

223   /**
224     This is a simple way to benchmark a function.  The function will be
225     run with the name you provide the number of times you indicate.  Only the
226     function is a required param.
227   */
228   bench: function(func, key, reps) {
229     if (!key) key = "bench%@".fmt(this._benchCount++) ;
230     if (!reps) reps = 1 ;
231     var ret ;
232 
233     while(--reps >= 0) {
234       var timeKey = SC.Benchmark.start(key) ;
235       ret = func();
236       SC.Benchmark.end(timeKey) ;
237     }
238 
239     return ret ;
240   },

242   /**
243     This bit of metaprogramming magic install a wrapper around a method and
244     benchmark it whenever it is run.
245   */
246   install: function(object,method, topLevelOnly) {
247     // vae the original method.
248     object['b__' + method] = object[method] ;
249     var __func = object['b__' + method];
250 
251     // replace with this helper.
252     object[method] = function() {
253       var key = '%@(%@)'.fmt(method, $A(arguments).join(', ')) ;
254       SC.Benchmark.start(key, topLevelOnly) ;
255       var ret = __func.apply(this, arguments) ;
256       SC.Benchmark.end(key) ;
257       return ret ;
258     } ;
259   },

272     This method will return a string containing a report of the stats
273     collected so far.  If you pass a key, only the stats for that key will

527   /** @private
528     Loads data from both the browser's own event hash and SC's pre-load event hash.
529   */
530   loadPreloadEvents: function() {
531     var preloadEvents = SC.benchmarkPreloadEvents, events = [], idx, len, evt;
532 
533     // the browsers may have their own event hash. Ours uses the same format, so
534     // all that we need to do is mixin the browser's to our own.
535     if (typeof webkitPerformnce !== 'undefined') SC.mixin(preloadEvents, webkitPerformane.timing);
536 
537     // we will attempt to find when the loading started and use that as our
538     // global start time, but only do so if the global start time is not already set.
539     if (!this.globalStartTime) {
540       // the potential events representing start time can be either from the browser
541       // or our own recordings. We prefer the browser.
542       var globalStartEvents = ['navigation', 'navigationStart', 'headStart'];
543       len = globalStartEvents.length;
544 
545       for (idx = 0; idx < len; idx++) {
546         if (preloadEvents[globalStartEvents[idx]]) {
547           this.globalStartTime = preloadEvents[globalStartEvents[idx]];
548           break;
549         }
550       }
551     }
552 
553     // the JavaScript start time will be one recorded by us
554     // we record headStart in bootstrap.
555     this.javascriptStartTime = preloadEvents['headStart'];
556 
557     // finally, mix in the events to our own events hash
558     SC.mixin(this.events, preloadEvents);
559 
560     this._hasLoadedPreloadEvents = true;
561   },

563   /** @private
564     Some events represent a beginning and end. While this is not common for events
565     that take place after the app loads (as they can just use SC.Benchmark.start/end),
566     SC.Benchmark.start/end is not available before load—as such, code will add
567     *Start and *End events to the event hash.
568 
569     This method iterates over the event hash and removes those items that represent
570     starts and ends, calling .start/.end for them.
571   */
572   _loadBenchmarksFromEvents: function() {
573     if (!this._hasLoadedPreloadEvents) this.loadPreloadEvents();
574 
575     var events = this.events;
576     for (var i in events) {
577       if (i.substr(-3) !== 'End') continue;
578 
579       var stem = i.substr(0, i.length - 3);
580       if (!events[stem + 'Start']) continue;
581 
582       SC.Benchmark.start(stem, undefined, events[stem + 'Start']);
583       SC.Benchmark.end(stem, undefined, events[stem + 'End']);
584 
585       delete events[stem + 'Start'];
586       delete events[stem + 'End'];
587     }
588   },

590   /** @private
591     Generates, sorts, and returns the array of all the data that has been captured.
592   */
593   _compileChartData: function(showSub) {
594     this._loadBenchmarksFromEvents();
595 
596     var chart = [], dispKey;
597     for(var key in this.stats)
598     {
599       var stat = this.stats[key];
600       for(var i=0; i<stat._times.length; i++)
601       {
602         var st = stat._times[i];
603         dispKey = (stat._times.length > 1) ? (i+1)+' - '+key : key;
604         chart.push([dispKey, st.start, st.end, st.dur, false]);
605         if(showSub)
606         {
607           var subStats = st._subStats;
608           for(var k in subStats)
609           {
610 
611             var subStat = subStats[k];
612             for(var j=0; j<subStat._times.length; j++)
613             {
614               var s = subStat._times[j];
615               dispKey = (subStat._times.length > 1) ? (j+1)+' - '+k : k;
616               chart.push([dispKey, s.start, s.end, s.dur, true]);
617 
618             }
619           }
620         }
621       }
622     }
623 
624     chart.sort(function(a,b)
625     {
626       if(a[1] < b[1])
627       {
628         return -1;
629       }
630       else if(a[1] == b[1])
631       {
632         if(a[3] && !b[3]) return -1;
633         if(!a[3] && b[3]) return 1;
634         return 0;
635       }
636       return 1;
637     });
638 
639     return chart;
640   },

680   // returns a stats hash for the named key and parent key.  If the hash does not exist yet,
681   // creates it.
682   /** @private */
683   _subStatFor: function(key, parentKey) {
684     var parentTimeLen = this.stats[parentKey]._times.length;
685     if(parentTimeLen === 0) return;
686     var parentSubStats = this.stats[parentKey]._times[this.stats[parentKey]._times.length-1]._subStats;
687     var ret = parentSubStats[key] ;
688     if (!ret) {
689       parentSubStats[key] = {
690         runs: 0, amt: 0, name: key, _starts: [], _times: []
691       };
692       ret = parentSubStats[key];
693     }
694     return ret ;
695   },
