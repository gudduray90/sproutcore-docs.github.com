  8 /**
  9   @type String
 10   @constant
 11 */
 12 SC.SCALE_NONE = "none";
 13 
 14 /**
 15  16   Stretch/shrink the shape to fill the frame
 17 
 18   @type String
 19   @constant
 20 */
 21 SC.FILL = "fill";
 22 
 23 /**
 24   Stretch/shrink the shape to fill the frame while maintaining aspect ratio, such
 25   that the shortest dimension will just fit within the frame and the longest dimension will
 26   overflow and be cropped.
 27 
 28   @type String
 29   @constant
 30 */
 31 SC.FILL_PROPORTIONALLY = "fillProportionally";
 32 
 33 /**
 34   Stretch/shrink the shape to fit the frame while maintaining aspect ratio, such that the
 35   longest dimension will just fit within the frame
 36 
 37   @type String
 38   @constant
 39 */
 40 SC.BEST_FIT = "fitBest";
 41 
 42 /**
 43   Shrink the shape to fit the frame while maintaining aspect ratio, such that
 44   the longest dimension will just fit within the frame.  Do not stretch the shape if the shape's
 45   width is less than the frame's width.
 46 
 47   @type String
 48   @constant
 49 */
 50 SC.BEST_FIT_DOWN_ONLY = "fitBestDown";
 51 
 52 /**
 53   @namespace
 54 
 55   InnerFrame provides the innerFrameForSize function, which will return a frame for the given size adjusted
 56   to fit within the given outer size, according to the align and scale properties.
 57 
 58   View's that render images will find this mixin particularly useful for fitting their images.
 59  */
 60 SC.InnerFrame = {
 61 
 62   /**
 63     Align the shape within its frame. Possible values:
 64 
 65       - SC.ALIGN_TOP_LEFT
 66       - SC.ALIGN_TOP
 67       - SC.ALIGN_TOP_RIGHT
 68       - SC.ALIGN_LEFT
 69       - SC.ALIGN_CENTER
 70       - SC.ALIGN_RIGHT
 71       - SC.ALIGN_BOTTOM_LEFT
 72       - SC.ALIGN_BOTTOM
 73       - SC.ALIGN_BOTTOM_RIGHT
 74 
 75     @type String
 76     @default SC.ALIGN_CENTER
 77   */
 78   align: SC.ALIGN_CENTER,
 79 
 80   /**
 81     Returns a frame (x, y, width, height) fitting the source size (sourceWidth & sourceHeight) within the
 82     destination size (destWidth & destHeight) according to the align and scale properties.  This is essential to
 83     positioning child views or elements within parent views or elements in elegant ways.
 84 
 85     @param {Number} sourceWidth
 86     @param {Number} sourceHeight
 87     @param {Number} destWidth
 88     @param {Number} destHeight
 89     @returns {Object} the inner frame with properties: {x: value, y: value, width: value, height: value }
 90   */
 91   innerFrameForSize: function(sourceWidth, sourceHeight, destWidth, destHeight) {
 92     var align = this.get('align'),
 93         scale = this.get('scale'),
 94         scaleX,
 95         scaleY,
 96         result;
 97 
 98     // Fast path
 99     result = { x: 0, y: 0, width: destWidth, height: destHeight };
100     if (scale === SC.FILL) return result;
101 
102     // Determine the appropriate scale
103     scaleX = destWidth / sourceWidth;
104     scaleY = destHeight / sourceHeight;
105 
106     switch (scale) {
107       case SC.FILL_PROPORTIONALLY:
108         scale = scaleX > scaleY ? scaleX : scaleY;
109         break;
110       case SC.BEST_FIT:
111         scale = scaleX < scaleY ? scaleX : scaleY;
112         break;
113       case SC.BEST_FIT_DOWN_ONLY:
114         if ((sourceWidth > destWidth) || (sourceHeight > destHeight)) {
115           scale = scaleX < scaleY ? scaleX : scaleY;
116         } else {
117           scale = 1.0;
118         }
119         break;
120       case SC.SCALE_NONE:
121         scale = 1.0;
122         break;
123       default: // Number
124         if (isNaN(window.parseFloat(scale)) || (window.parseFloat(scale) <= 0)) {
125           SC.Logger.warn("SC.InnerFrame: The scale '%@' was not understood.  Scale must be one of SC.FILL, SC.FILL_PROPORTIONALLY, SC.BEST_FIT, SC.BEST_FIT_DOWN_ONLY or a positive number greater than 0.00.".fmt(scale));
126 
127           // Don't attempt to scale or offset the image
128           return result;
129         }
130     }
131 
132     sourceWidth *= scale;
133     sourceHeight *= scale;
134     result.width = Math.round(sourceWidth);
135     result.height = Math.round(sourceHeight);
136 
137     // Align the image within its frame
138     switch (align) {
139       case SC.ALIGN_LEFT:
140         result.x = 0;
141         result.y = (destHeight / 2) - (sourceHeight / 2);
142         break;
143       case SC.ALIGN_RIGHT:
144         result.x = destWidth - sourceWidth;
145         result.y = (destHeight / 2) - (sourceHeight / 2);
146         break;
147       case SC.ALIGN_TOP:
148         result.x = (destWidth / 2) - (sourceWidth / 2);
149         result.y = 0;
150         break;
151       case SC.ALIGN_BOTTOM:
152         result.x = (destWidth / 2) - (sourceWidth / 2);
153         result.y = destHeight - sourceHeight;
154         break;
155       case SC.ALIGN_TOP_LEFT:
156         result.x = 0;
157         result.y = 0;
158         break;
159       case SC.ALIGN_TOP_RIGHT:
160         result.x = destWidth - sourceWidth;
161         result.y = 0;
162         break;
163       case SC.ALIGN_BOTTOM_LEFT:
164         result.x = 0;
165         result.y = destHeight - sourceHeight;
166         break;
167       case SC.ALIGN_BOTTOM_RIGHT:
168         result.x = destWidth - sourceWidth;
169         result.y = destHeight - sourceHeight;
170         break;
171       default: // SC.ALIGN_CENTER || SC.ALIGN_MIDDLE
172         if (align !== SC.ALIGN_CENTER && align !== SC.ALIGN_MIDDLE) {
173           SC.Logger.warn("SC.InnerFrame: The align '%@' was not understood.  Align must be one of SC.ALIGN_CENTER/SC.ALIGN_MIDDLE, SC.ALIGN_LEFT, SC.ALIGN_RIGHT, SC.ALIGN_TOP, SC.ALIGN_BOTTOM, SC.ALIGN_TOP_LEFT, SC.ALIGN_TOP_RIGHT, SC.ALIGN_BOTTOM_LEFT or SC.ALIGN_BOTTOM_RIGHT.".fmt(align));
174         }
175         result.x = (destWidth / 2) - (sourceWidth / 2);
176         result.y = (destHeight / 2) - (sourceHeight / 2);
177     }
178 
179     return result;
180   },
181 
182   /**
183     Determines how the shape will scale to fit within its containing space. Possible values:
184     
185       - SC.SCALE_NONE
186       - SC.FILL
187       - SC.FILL_PROPORTIONALLY
188       - SC.BEST_FIT
189       - SC.BEST_FIT_DOWN_ONLY
190 
191     @type String
192     @default SC.FILL
193   */
194   scale: SC.FILL
195 };
