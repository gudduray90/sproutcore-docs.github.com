  7 /** @class
  8 
  9   Layout controller properties used by the layout palette.  The content should
 10   be bound to the current page design controller selection.
 11 
 12   @extends SC.ObjectController
 13 */
 14 Greenhouse.layoutController = SC.ObjectController.create(
 15 /** @scope Greenhouse.layoutController.prototype */ {
 16 
 17   contentBinding: "Greenhouse.pageController.designController.selection",
 18   allowsMultipleContent: YES, // palette like behavior
 19 
 20   /**
 21     Determines which set of dimensions should be visible in the layout 
 22     palette in the horizontal direction.
 23   */
 24   hDimNowShowing: function() {
 25     var loc = this.get('anchorLocation'),
 26         K   = SC.ViewDesigner, 
 27         ret = 'leftDimensions';
 28         
 29     if (loc & K.ANCHOR_LEFT) ret = 'leftDimensions';
 30     else if (loc & K.ANCHOR_RIGHT) ret = 'rightDimensions';
 31     else if (loc & K.ANCHOR_CENTERX) ret = 'centerXDimensions';
 32     else if (loc & K.ANCHOR_WIDTH) ret = 'widthDimensions';
 33     return ret ;
 34   }.property('anchorLocation').cacheable(),
 35   
 36   /**
 37     Determines which set of dimensions should be visible in the layout 
 38     palette in the vertical direction.
 39   */
 40   vDimNowShowing: function() {
 41     var loc = this.get('anchorLocation'),
 42         K   = SC.ViewDesigner, 
 43         ret = 'topDimensions';
 44         
 45     if (loc & K.ANCHOR_TOP) ret = 'topDimensions';
 46     else if (loc & K.ANCHOR_BOTTOM) ret = 'bottomDimensions';
 47     else if (loc & K.ANCHOR_CENTERY) ret = 'centerYDimensions';
 48     else if (loc & K.ANCHOR_HEIGHT) ret = 'heightDimensions';
 49     return ret ;
 50   }.property('anchorLocation').cacheable()
 51   
 52 }) ;
