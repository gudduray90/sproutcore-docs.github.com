  9 /**
 10   @class
 11 
 12   Implements support for WebSocket.
 13 
 14   Example Usage:
 15 
 16     var ws = SC.WebSocket.create({
 17       server: 'ws://server',
 18     }).connect();
 19 
 20     ws.notify(this, 'wsReceivedMessage');
 21 
 22     ws.send('message');
 23 
 24   @since SproutCore 1.11
 25   @extends SC.Object
 26   @extends SC.DelegateSupport
 27   @author Nicolas BADIA
 28 */
 29 SC.WebSocket = SC.Object.extend(SC.DelegateSupport, SC.WebSocketDelegate, {
 30 
 31   /**
 32     URL of the WebSocket server.
 33 
 34     @type String
 35     @default null
 36   */
 37   server: null,
 38 
 39   /**
 40     Determines if the browser support the WebSocket protocol.
 41 
 42     @type Boolean
 43     @default null
 44     @readOnly
 45   */
 46   isSupported: null,
 47 
 48   /**
 49     Determines if the connection is open or not.
 50 
 51     @type Boolean
 52     @readOnly
 53   */
 54   isConnected: false,
 55 
 56   /**
 57     In order to handle authentification, set `isAuth` to NO in the
 58     `webSocketDidOpen` delegate method just after sending a request to
 59     authentificate the connection. This way, any futher message will be put in
 60     queue until the server tels you the connection is authentified. Once it
 61     did, you should set `isAuth` to YES to resume the queue.
 62 
 63     If you don't need authentification, leave `isAuth` to null.
 64 
 65     @type Boolean
 66     @default null
 67   */
 68   isAuth: null,
 69 
 70   /**
 71     Processes the messages as JSON if possible.
 72 
 73     @type Boolean
 74     @default true
 75   */
 76   isJSON: true,
 77 
 78   /**
 79     A WebSocket delegate.
 80 
 81     @see SC.WebSocketDelegate
 82     @type {SC.WebSocketDelegate}
 83     @default null
 84   */
 85   delegate: null,
 86 
 87   /**
 88     Determines if we should attempt to try an automatic reconnection
 89     if the connection is close.
 90 
 91     @type {SC.WebSocketDelegate}
 92     @default null
 93   */
 94   autoReconnect: true,
 95 
 96   /**
 97     The interval in milliseconds to waits before trying a reconnection.
 98 
 99     @type {SC.WebSocketDelegate}
100     @default null
101   */
102   reconnectInterval: 10000, // 10 secondes
103

106   // PUBLIC METHODS
107   //
108 
109   /**
110     Call this method to open a connection.
111 
112     @returns {SC.WebSocket}
113   */
114   connect: function() {
115     var that = this;
116 
117     if (this.isSupported === null) this.set('isSupported', !!window.WebSocket);
118 
119     if (!this.isSupported || this.socket) return this;
120 
121     try {
122       var socket = this.socket = new WebSocket(this.get('server'));
123 
124       socket.onopen = function() {
125         SC.RunLoop.begin();
126         that.onOpen.apply(that, arguments);
127         SC.RunLoop.end();
128       };
129 
130       socket.onmessage = function() {
131         SC.RunLoop.begin();
132         that.onMessage.apply(that, arguments);
133         SC.RunLoop.end();
134       };
135 
136       socket.onclose = function() {
137         SC.RunLoop.begin();
138         that.onClose.apply(that, arguments);
139         SC.RunLoop.end();
140       };
141 
142       socket.onerror = function() {
143         SC.RunLoop.begin();
144         that.onError.apply(that, arguments);
145         SC.RunLoop.end();
146       };
147     } catch(e) {
148       SC.error('An error has occurred while connnecting to the websocket server: '+e);
149     }
150 
151     return this;
152   },
153 
154   /**
155     Call this method to close a connection.
156 
157     @param code {Number} A numeric value indicating the status code explaining why the connection is being closed. If this parameter is not specified, a default value of 1000 (indicating a normal "transaction complete" closure) is assumed.
158     @param reason {String} A human-readable string explaining why the connection is closing. This string must be no longer than 123 bytes of UTF-8 text (not characters).
159     @returns {SC.WebSocket}
160   */
161   close: function(code, reason) {
162     var socket = this.socket;
163 
164     if (socket && socket.readyState === SC.WebSocket.OPEN) {
165       this.socket.close(code, reason);
166     }
167 
168     return this;
169   },
170 
171   /**
172     Configures a callback to execute when an event happen. You must pass
173     at least a target and action/method to this and optionally an event name.
174 
175     You may also pass additional arguments which will then be passed along to
176     your callback.
177 
178     Example:
179 
180         var websocket = SC.WebSocket.create({ server: 'ws://server' }).connect();
181 
182         webSocket.notify('onopen', this, 'wsWasOpen');
183         webSocket.notify('onmessage', this, 'wsReceivedMessage'); // You can ommit onmessage here
184         webSocket.notify('onclose', this, 'wsWasClose');
185         webSocket.notify('onerror', this, 'wsDidError');
186 
187     ## Callback Format
188 
189     Your notification callback should expect to receive the WebSocket object as
190     the first parameter and the event or message; plus any additional parameters that you pass. If your callback handles the notification and to prevent further handling, it
191     should return YES.
192 
193     @param target {String} String Event name.
194     @param target {Object} The target object for the callback action.
195     @param action {String|Function} The method name or function to call on the target.
196     @returns {SC.WebSocket} The SC.WebSocket object.
197   */
198   notify: function(event, target, action) {
199     var args,
200       i, len;
201 
202     if (SC.typeOf(event) !== SC.T_STRING) {
203       // Fast arguments access.
204       // Accessing `arguments.length` is just a Number and doesn't materialize the `arguments` object, which is costly.
205       args = new Array(arguments.length - 2); //  SC.A(arguments).slice(2)
206       for (i = 0, len = args.length; i < len; i++) { args[i] = arguments[i + 2]; }
207 
208       // Shift the arguments
209       action = target;
210       target = event;
211       event = 'onmessage';
212     } else {
213       if (arguments.length > 3) {
214         // Fast arguments access.
215         // Accessing `arguments.length` is just a Number and doesn't materialize the `arguments` object, which is costly.
216         args = new Array(arguments.length - 3); //  SC.A(arguments).slice(3)
217         for (i = 0, len = args.length; i < len; i++) { args[i] = arguments[i + 3]; }
218       } else {
219         args = [];
220       }
221     }
222 
223     var listeners = this.get('listeners');
224     if (!listeners) { this.set('listeners', listeners = {}); }
225     if(!listeners[event]) { listeners[event] = []; }
226 
227     //@if(debug)
228     for (i = listeners[event].length - 1; i >= 0; i--) {
229       var listener = listeners[event][i];
230       if (listener.event === event && listener.target === target && listener.action === action) {
231         SC.warn("Developer Warning: This listener is already defined.");
232       }
233     }
234     //@endif
235 
236     // Add another listener for the given event name.
237     listeners[event].push({target: target, action: action, args: args});
238 
239     return this;
240   },
241 
242   /**
243     Send the passed message. If the connection is not yet open or anthentified,
244     the message will be put in the queue.
245 
246     @param message {String|Object} The message to send.
247     @returns {SC.WebSocket}
248   */
249   send: function(message) {
250     if (this.isConnected === true && this.isAuth !== false) {
251       if (this.isJSON) {
252         message = JSON.stringify(message);
253       }
254 
255       this.socket.send(message);
256     }
257     else {
258       this.addToQueue(message);
259     }
260     return this;
261   },
262 
263   // ..........................................................
264   // PRIVATE METHODS
265   //
266 
267   /**
268      @private
269   */
270   onOpen: function(event) {
271     var del = this.get('objectDelegate');
272 
273     this.set('isConnected', true);
274 
275     var ret = del.webSocketDidOpen(this, event);
276     if (ret !== true) this._notifyListeners('onopen', event);
277 
278     this.fireQueue();
279   },
280 
281   /**
282      @private
283   */
284   onMessage: function(messageEvent) {
285     if (messageEvent) {
286       var del = this.get('objectDelegate'),
287         message,
288         data,
289         ret;
290 
291       message = data = messageEvent.data;
292       ret = del.webSocketDidReceiveMessage(this, data);
293 
294       if (ret !== true) {
295         if (this.isJSON) {
296           message = JSON.parse(data);
297         }
298         this._notifyListeners('onmessage', message);
299       }
300     }
301 
302     // If there is message in the queue, we fire them
303     this.fireQueue();
304   },
305 
306   /**
307      @private
308   */
309   onClose: function(closeEvent) {
310     var del = this.get('objectDelegate');
311 
312     this.set('isConnected', false);
313     this.set('isAuth', null);
314     this.socket = null;
315 
316     var ret = del.webSocketDidClose(this, closeEvent);
317 
318     if (ret !== true) {
319       this._notifyListeners('onclose', closeEvent);
320       this.tryReconnect();
321     }
322   },
323 
324   /**
325      @private
326   */
327   onError: function(event) {
328     var del = this.get('objectDelegate'),
329       ret = del.webSocketDidError(this, event);
330 
331     if (ret !== true) this._notifyListeners('onerror', event);
332   },
333 
334   /**
335      @private
336 
337      Add the message to the queue
338   */
339   addToQueue: function(message) {
340     var queue = this.queue;
341     if (!queue) { this.queue = queue = []; }
342 
343     queue.push(message);
344   },
345 
346   /**
347      @private
348 
349      Send the messages from the queue.
350   */
351   fireQueue: function() {
352     var queue = this.queue;
353     if (!queue || queue.length === 0) return;
354 
355     queue = SC.A(queue);
356     this.queue = null;
357 
358     for (var i = 0, len = queue.length; i < len; i++) {
359       var message = queue[i];
360       this.send(message);
361     }
362   },
363 
364   /**
365     @private
366   */
367   tryReconnect: function() {
368     if (!this.get('autoReconnect')) return;
369 
370     var that = this;
371     setTimeout(function() { that.connect(); }, this.get('reconnectInterval'));
372   },
373 
374   /**
375     @private
376 
377     Will notify each listener. Returns true if any of the listeners handle.
378   */
379   _notifyListeners: function(event, message) {
380     var listeners = (this.listeners || {})[event], notifier, target, action, args;
381     if (!listeners) { return NO; }
382 
383     var handled = NO,
384       len = listeners.length;
385 
386     for (var i = 0; i < len; i++) {
387       notifier = listeners[i];
388       args = (notifier.args || []).copy();
389       args.unshift(message);
390       args.unshift(this);
391 
392       target = notifier.target;
393       action = notifier.action;
394       if (SC.typeOf(action) === SC.T_STRING) { action = target[action]; }
395 
396       handled = action.apply(target, args);
397       if (handled === true) return handled;
398     }
399 
400     return handled;
401   },
402 
403   /**
404     @private
405   */
406   objectDelegate: function () {
407     var del = this.get('delegate');
408     return this.delegateFor('isWebSocketDelegate', del, this);
409   }.property('delegate').cacheable(),
410
