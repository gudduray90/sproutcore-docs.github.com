  8 // test parsing of query string
  9 var store, storeKey, rec1, rec2, rec3, rec4, rec5, rec6, rec7, MyApp, q;
 10 module("SC.Query evaluation of records", {
 11   setup: function() {
 12     
 13     SC.RunLoop.begin();
 14     
 15     // setup dummy app and store
 16     MyApp = SC.Object.create({
 17       store: SC.Store.create()
 18     });
 19     
 20     // setup dummy models
 21     MyApp.Animal = SC.Record.extend({});
 22     MyApp.Person = SC.Record.extend({
 23       pet: SC.Record.toOne(MyApp.Animal, { nested: true })
 24     });
 25 
 26     // load some data
 27     MyApp.store.loadRecords(MyApp.Person, [
 28       { guid: 1, firstName: "John", lastName: "Doe", married: true },
 29       { guid: 2, firstName: "Jane", lastName: "Doe", married: false },
 30       { guid: 3, firstName: "Emily", lastName: "Parker", bornIn: 1975, married: true },
 31       { guid: 4, firstName: "Johnny", lastName: "Cash", married: true },
 32       { guid: 5, firstName: "Bert", lastName: "Berthold", married: true },
 33       { guid: 6, firstName: "Ronald", lastName: "Fitzgerald", parents: { father: "Frank", mother: "Nancy" }},
 34       { guid: 7, firstName: "Jack", lastName: "Dimpleton", pet: { guid: 1, name: "Fido" } }
 35     ]);
 36     
 37     rec1 = MyApp.store.find(MyApp.Person,1);
 38     rec2 = MyApp.store.find(MyApp.Person,2);
 39     rec3 = MyApp.store.find(MyApp.Person,3);
 40     rec4 = MyApp.store.find(MyApp.Person,4);
 41     rec5 = MyApp.store.find(MyApp.Person,5);
 42     rec6 = MyApp.store.find(MyApp.Person,6);
 43     rec7 = MyApp.store.find(MyApp.Person,7);
 44     
 45     SC.RunLoop.end();
 46     
 47     q = SC.Query.create();
 48   }
 49 });
 50

 52 // ..........................................................
 53 // RECORD PROPERTIES
 54 // 
 55 
 56 test("should get record properties correctly", function() {
 57 
 58   q.conditions = "fakeProp = 'Foo'";
 59   q.parse();
 60   equals(q.contains(rec1), false, 'John should not match: fakeProp = "Foo"');
 61   
 62   q.conditions = "firstName = 'John'";
 63   q.parse();
 64   equals(q.contains(rec1), true, 'John should match: firstName = "John"');
 65   equals(q.contains(rec2), false, 'Jane should not match: firstName = "John"');
 66 
 67   q.conditions = "parents.father = 'Frank'";
 68   q.parse();
 69   equals(q.contains(rec6), true, "Ronald should match: parents.father = 'Frank'");
 70   equals(q.contains(rec1), false, "John should not match: parents.father = 'Frank'");
 71 
 72   q.conditions = "pet.name = 'Fido'";
 73   q.parse();
 74   equals(q.contains(rec7), true, "Jack should match: pet.name = 'Fido'");
 75   equals(q.contains(rec6), false, "Ronald should not match: pet.name = 'Fido'");
 76 
 77   q.conditions = "pet.parentRecord.firstName = 'Jack'";
 78   q.parse();
 79   equals(q.contains(rec7), true, "Jack should match: pet.parentRecord.firstName = 'Jack'");
 80   equals(q.contains(rec6), false, "Ronald should not match: pet.parentRecord.firstName = 'Jack'");
 81 
 82   q.conditions = "lastName BEGINS_WITH firstName";
 83   q.parse();
 84   equals(q.contains(rec5), true, 'Bert Berthold should match: lastName BEGINS_WITH firstName');
 85   equals(q.contains(rec2), false, 'Jane Doe should not match: lastName BEGINS_WITH firstName');
 86   
 87   q.conditions = "lastName CONTAINS firstName";
 88   q.parse();
 89   equals(q.contains(rec5), true, 'Bert Berthold should match: lastName CONTAINS firstName');
 90   equals(q.contains(rec2), false, 'Jane Doe should not match: lastName CONTAINS firstName');
 91 
 92 });

 95 test("should handle undefined record properties correctly", function() {
 96   
 97   q.conditions = "bornIn = 1975";
 98   q.parse();
 99   equals(q.contains(rec3), true, 'record with bornIn set should match');
100   equals(q.contains(rec2), false, 'record without bornIn set should not match');
101   
102   q.conditions = "bornIn = undefined";
103   q.parse();
104   equals(q.contains(rec3), false, 'record with bornIn set different to null should not match');
105   equals(q.contains(rec2), true, 'record without bornIn set should match');
106   
107 }); 
108 
109 test("should handle boolean correctly", function() {
110

112   q.parse();

114   equals(q.contains(rec2), false, 'record without married set should not match');
115
