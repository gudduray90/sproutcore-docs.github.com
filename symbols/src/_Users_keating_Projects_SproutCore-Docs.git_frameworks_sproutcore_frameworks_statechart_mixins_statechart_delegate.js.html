 10 /**
 11   @class
 12   
 13   Apply to objects that are to represent a delegate for a SC.Statechart object.
 14   When assigned to a statechart, the statechart and its associated states will
 15   use the delegate in order to make various decisions.
 16   
 17   @see SC.Statechart#delegate 
 18 
 19   @author Michael Cohen
 20 */
 21 
 22 SC.StatechartDelegate = /** @scope SC.StatechartDelegate.prototype */ {
 23   
 24   // Walk like a duck
 25   isStatechartDelegate: YES,
 26   
 27   // Route Handling Management
 28   
 29   /**
 30     Called to update the application's current location. 
 31     
 32     The location provided is dependent upon the application's underlying
 33     routing mechanism.
 34  35     
 36     @param {SC.StatechartManager} statechart the statechart
 37     @param {String|Hash} location the new location 
 38     @param {SC.State} state the state requesting the location update
 39   */
 40   statechartUpdateLocationForState: function(statechart, location, state) {
 41     SC.routes.set('location', location);
 42   },
 43   
 44   /**
 45     Called to acquire the application's current location.
 46     
 47     @param {SC.StatechartManager} statechart the statechart
 48     @param {SC.State} state the state requesting the location
 49     @returns {String} the location 
 50   */
 51   statechartAcquireLocationForState: function(statechart, state) {
 52     return SC.routes.get('location');
 53   },
 54   
 55   /**
 56     Used to bind a state's handler to a route. When the application's location
 57     matches the given route, the state's handler is to be invoked. 
 58     
 59     The statechart and states remain completely independent of how the underlying 
 60     routing mechanism works thereby providing a looser coupling and more flexibility 
 61     in how routing is to work. Given this flexiblity, it is important that a route
 62     assigned (using the {@link SC.State#representRoute} property) to a state strictly 
 63     conforms to the underlying routing mechanism's criteria in order for the given 
 64     handler to be properly invoked.
 65     
 66     By default the {@link SC.routes} mechanism is used to bind the state's handler with
 67     the given route.
 68     
 69     @param {SC.StatechartManager} statechart the statechart
 70     @param {SC.State} state the state to bind the route to
 71     @param {String|Hash} route the route that is to be bound to the state
 72     @param {Function|String} handler the method on the state to be invoked when the route
 73       gets triggered.
 74       
 75     @see SC.State#representRoute
 76   */
 77   statechartBindStateToRoute: function(statechart, state, route, handler) {
 78     SC.routes.add(route, state, handler);
 79   },
 80   
 81   /**
 82     Invoked by a state that has been notified to handle a triggered route. The state
 83     asks if it should go ahead an actually handle the triggered route. If no then
 84     the state's handler will no longer continue and finish by calling this delegate's
 85     `statechartStateCancelledHandlingTriggeredRoute` method. If yes then the state will 
 86     continue with handling the triggered route.
 87     
 88     By default `YES` is returned.
 89     
 90     @param {SC.StatechartManager} statechart the statechart
 91     @param {SC.State} state the state making the request
 92     @param {SC.StateRouteHandlerContext} routeContext contextual information about the handling 
 93       of a route
 94     
 95     @see #statechartStateCancelledHandlingTriggeredRoute
 96   */
 97   statechartShouldStateHandleTriggeredRoute: function(statechart, state, context) {
 98     return YES;
 99   },
100   
101   /**
102     Invoked by a state that has been informed by the delegate to not handle a triggered route.
103     Used this for any additional clean up or processing that you may wish to perform.
104     
105     @param {SC.StatechartManager} statechart the statechart
106     @param {SC.State} state the state making the request
107     @param {SC.StateRouteHandlerContext} routeContext contextual information about the handling 
108       of a route
109     
110     @see #statechartShouldStateHandleTriggeredRoute
111   */
112   statechartStateCancelledHandlingTriggeredRoute: function(statechart, state, context) { }
113   
114 };
