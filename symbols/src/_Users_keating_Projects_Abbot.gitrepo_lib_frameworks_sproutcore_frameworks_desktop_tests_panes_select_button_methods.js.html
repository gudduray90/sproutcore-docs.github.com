 10 var pane, view , view1, view2, view3, view4 ;
 11 
 12 module("SC.SelectButtonView",{
 13 
 14   //setup
 15   setup: function() {
 16     SC.RunLoop.begin();
 17     var isDue = NO ;
 18 
 19     //pane
 20     pane = SC.MainPane.create({
 21       objs : ["Around","The","World"],
 22       objs2 : [{ title: "Around", pos: 3},
 23         { title: "The", pos: 1},
 24         { title: "World", pos: 2 },
 25         { title: "Again", pos: 4}],
 26       selectedValue: "World",
 27       isDue: YES,
 28       childViews: [
 29 
 30         //view1
 31         SC.SelectButtonView.extend({
 32           objects: ["To","Back", "You"],
 33           disableSort: NO
 34         }),
 35 
 36         //view2
 37         SC.SelectButtonView.extend({
 38           objects: ["Drop","Down", "Menu"]
 39         }),
 40 
 41         //view3
 42         SC.SelectButtonView.extend({
 43           objectsBinding: '*owner.objs',
 44           valueBinding: '*owner.selectedValue',
 45           isVisibleBinding: '*owner.isDue'
 46         }),
 47 
 48         //view4
 49         SC.SelectButtonView.extend({
 50           objectsBinding: '*owner.objs2',
 51           valueBinding: '*owner.selectedValue',
 52           valueKey: 'title',
 53           nameKey: 'title',
 54           sortKey: 'pos'
 55         }),
 56         
 57         //view5
 58         SC.SelectButtonView.extend({
 59           objects: ["My","New", "List"]
 60         }),
 61         
 62         //view6
 63         SC.SelectButtonView.extend({
 64           objects: ["My","New", "List"],
 65           customViewClassName: 'custom-menu-item',
 66           customViewMenuOffsetWidth: 46
 67         })
 68       ]
 69     });
 70 
 71     view1 = pane.childViews[0] ;
 72     view2 = pane.childViews[1] ;
 73     view3 = pane.childViews[2] ;
 74     view4 = pane.childViews[3] ;
 75     view5 = pane.childViews[4] ;
 76     view6 = pane.childViews[5] ;
 77     
 78     pane.append(); // make sure there is a layer...
 79     SC.RunLoop.end();
 80   },
 81 
 82   //teardown
 83   teardown: function() {
 84     pane.remove() ;
 85     pane = view = null ;
 86   }
 87 });
