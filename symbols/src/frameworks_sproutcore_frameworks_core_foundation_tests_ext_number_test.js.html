  9 var defaultLocale;
 10 module("Number#ordinal", {
 11   setup: function () {
 12     // Cache the current locale.
 13     defaultLocale = SC.Locale.currentLocale;
 14   },
 15 
 16   teardown: function () {
 17     // Return the current locale.
 18     SC.Locale.currentLocale = defaultLocale;
 19   }
 20 });
 21 
 22 /**
 23  * Admitedly not exhaustive, but tests the numbers from 1-100
 24  */
 25 test("Properly Computes the Ordinal in english", function () {
 26 
 27   // Force it to English
 28   String.preferredLanguage = 'en';
 29   SC.Locale.currentLocale = SC.Locale.createCurrentLocale();
 30 
 31   equals(SC.Locale.currentLocale.language, 'en');
 32 
 33   var sts = [1, 21, 31, 41, 51, 61, 71, 81, 91, 101],
 34     nds = [2, 22, 32, 42, 52, 62, 72, 82, 92, 102],
 35     rds = [3, 23, 33, 43, 53, 63, 73, 83, 93, 103];
 36 
 37   sts.forEach(function (number) {
 38     equals(number.ordinal(), 'st');
 39   });
 40 
 41   nds.forEach(function (number) {
 42     equals(number.ordinal(), 'nd');
 43   });
 44 
 45   rds.forEach(function (number) {
 46     equals(number.ordinal(), 'rd');
 47   });
 48 
 49   var ths = [];
 50   for (var i = 0; i < 100; i++) {
 51     ths.push(i);
 52   }
 53 
 54   ths.removeObjects(sts);
 55   ths.removeObjects(nds);
 56   ths.removeObjects(rds);
 57 
 58   ths.forEach(function (number) {
 59     equals(number.ordinal(), 'th');
 60   });
 61 
 62 });
 63 
 64 test("Do not compute the ordinal in a language if no method is defined", function () {
 65 
 66   // Force it to Japanese
 67   String.preferredLanguage = 'jp';

 70   equals(SC.Locale.currentLocale.language, 'ja');
 71 
 72   var st = 1,
 73     nd = 2,
 74     rd = 3,
 75     th = 4;
 76 
 77   equals(st.ordinal(), '');
 78   equals(nd.ordinal(), '');
 79   equals(rd.ordinal(), '');
 80   equals(th.ordinal(), '');
 81 });
