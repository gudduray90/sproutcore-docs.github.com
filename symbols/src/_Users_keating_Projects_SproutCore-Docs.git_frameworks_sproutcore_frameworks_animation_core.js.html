 10 /** @deprecated
 11   @namespace
 12   A simple mixin called Animatable is provided. What does it do?
 13   It makes CSS transitions for you, and if they aren't available,
 14   implements them in JavaScript.
 15 
 16   ## Animatable things:
 17 
 18     - layout. You can animate any layout property, even centerX and centerY
 19     - opacity.
 20     - display, in a way. All animating display does is delay setting display:none
 21       until <em>after</em> the transition duration has passed. This allows you
 22       to set display:none after fading out. If mixing with CSS transitions, you will
 23       need to set the delay a tad longer to accommodate any delays in beginning the
 24       transition.
 25 
 26   ## Example Usage:
 27 
 28       aView: SC.LabelView.design(SC.Animatable, {
 29         transitions: {
 30           left: {duration: .25},
 31           top: .25, // only possible during design; otherwise you must use long form.
 32           width: {duration: .25, timing: SC.Animatable.TRANSITION_EASE_IN_OUT }
 33         }
 34       })
 35 */
 36 SC.warn("SC.Animatable is deprecated and will be removed in the next release.  Animation can be achieved by using the .animate method on SC.View.");

190   /**
191   Adds support for some style properties to adjust.
192 
193   These added properties are currently:
194 
195    - opacity.
196    - display.
197 
198   This is a complete rewrite of adjust. Its performance can probably be boosted. Do it!

200   adjust: function(dictionary, value)
201   {
202     if (!SC.none(value)) {
203       var key = dictionary;
204       dictionary = { };
205       dictionary[key] = value;
206     }
207     else {
208       dictionary = SC.clone(dictionary);
209     }
210 
211     var style = SC.clone(this.get('style')), didChangeStyle = NO, layout = SC.clone(this.get('layout')), didChangeLayout = NO;
212     var sprops = this._styleProperties;
213     for (var i in dictionary)
214     {
215       var didChange = NO;
216 
217       var current = (sprops.indexOf(i) >= 0) ? style : layout;
218       var cval = current[i], nval = dictionary[i];
219 
220       if (nval !== undefined && cval !== nval)
221       {
222         if (nval === null)
223         {
224           if (cval !== undefined) didChange = YES;
225           delete current[i];
226         }
227         else
228         {
229           current[i] = nval;
230           didChange = YES;
231         }
232       }
233 
234       if (didChange) {
235         if (current === style) didChangeStyle = YES; else didChangeLayout = YES;
236       }
237     }
238 
239     if (didChangeStyle) this.set('style', style);
240     if (didChangeLayout) this.set('layout', layout);
241 
242     // call base with whatever is leftover
243     return this;
244   },

308   _getStartStyleHash: function(start, target)
309   {
310     // temporarily set layout to "start", in the fastest way possible;
311     // note that start is an entire style structure—get("frame") doesn't care! HAH!
312     var original_layout = this.layout;
313     this.layout = start;
314 
315     // get our frame and parent's frame
316     var p = this.computeParentDimensions();
317     var f = this.computeFrameWithParentFrame(p);
318 
319     // set back to target
320     this.layout = original_layout;
321 
322     // prepare a new style set, empty.
323     var l = {};
324 
325     // loop through properties in target
326     for (var i in target)
327     {
328       if (f)
329       {
330         if (i == "left") { l[i] = f.x; continue; }
331         else if (i == "top") { l[i] = f.y; continue; }
332         else if (i == "right") { l[i] = p.width - f.x - f.width; continue; }
333         else if (i == "bottom") { l[i] = p.height - f.y - f.height; continue; }
334         else if (i == "width") { l[i] = f.width; continue; }
335         else if (i == "height") { l[i] = f.height; continue; }
336         else if (i == "centerX") { l[i] = f.x + (f.width / 2) - (p.width / 2); continue; }
337         else if (i == "centerY") { l[i] = f.y + (f.height / 2) - (p.height / 2); continue; }
338       }
339 
340       if (SC.none(l[i])) {
341         if (!SC.none(start[i])) l[i] = start[i];
342         else l[i] = target[i];
343       }
344     }
345     return l;

394   /**
395   Immediately applies styles to elements, and starts any needed transitions.
396 
397   Called automatically when style changes, but if you need styles to be adjusted
398   immediately (for instance, if you have temporarily disabled animation to set a
399   start state), you may want to call manually too.
400   */
401   updateStyle: function()
402   {
403     // get the layer. We need it for a great many things.
404     var layer = this.get("layer");
405 
406     // cont. with other stuff
407     var newStyle = this.get("style");
408 
409     /* SPECIAL CASES (done now because they need to happen whether or not animation will take place) */
410     ////**SPECIAL TRANSFORM CASE**////
411     var specialTransform = NO, specialTransformValue = "";
412     if (this.get('hasAcceleratedLayer')) {
413 
414       var nT = newStyle['top'],
415           nB = newStyle['bottom'],
416           nH = newStyle['height'],
417           nL = newStyle['left'],
418           nR = newStyle['right'],
419           nW = newStyle['width'];
420 
421       // NOTE: This needs to match exactly the conditions in layoutStyles
422       if (
423         (SC.empty(nT) || (!SC.isPercentage(nT) && !SC.empty(nH))) &&
424         (SC.empty(nL) || (!SC.isPercentage(nL) && !SC.empty(nW))) &&
425         (this.transitions && (this.transitions['top'] || this.transitions['left']))
426       ) {
427         specialTransform = YES;
428         this._useSpecialCaseTransform = YES;
429       } else {
430         this._useSpecialCaseTransform = NO;
431       }
432     }
433     ////**/SPECIAL TRANSFORM CASE**////
434 
435     // make sure there _is_ a previous style to animate from. Otherwise,
436     // we don't animate—and this is sometimes used to temporarily disable animation.
437     var i;
438     if (!this._animatableCurrentStyle || this._disableAnimation > 0 || !layer)
439     {
440       // clone it to be a nice starting point next time.
441       this._animatableSetCSS = "";
442       this._animatableCurrentStyle = {};
443       for (i in newStyle)
444       {
445         if (i[0] != "_") this._animatableCurrentStyle[i] = newStyle[i];
446       }
447 
448       if (layer) this._animatableApplyStyles(layer, newStyle);
449       return this;
450     }
451 
452     // no use doing anything else if no layer.
453     if (!layer) return;
454 
455     // get a normalized starting point based off of our style
456     var startingPoint = this._getStartStyleHash(this._animatableCurrentStyle, newStyle);
457     var endingPoint = {};
458 
459     // prepare stuff for timing function calc
460     var timing;
461 
462     // also prepare an array of CSS transitions to set up. Do this always so we get (and keep) all transitions.
463     var cssTransitions = this._TMP_CSS_TRANSITIONS;
464     if (SC.platform.supportsCSSTransitions) {
465       // first, handle special cases
466       var timing_function;
467 
468       ////**SPECIAL TRANSFORM CASE**////
469       // this is a VERY special case. If right or bottom are supplied, can't do it. If left+top need
470       // animation at different speeds: can't do it.
471       if (specialTransform) {
472         var transitionForTiming = this.transitions['left'] || this.transitions['top'];
473         timing_function = this.cssTimingStringFor(transitionForTiming);
474         cssTransitions.push("-"+SC.platform.cssPrefix+"-transform " + transitionForTiming.duration + "s " + timing_function);
475       }
476       ////**END SPECIAL TRANSFORM CASE**////
477 
478       // loop
479       for (i in this.transitions) {
480         if (!this._cssTransitionFor[i]) continue;
481 
482         ////**SPECIAL TRANSFORM CASE**////
483         if (specialTransform && (i == "left" || i == "top")) {
484           if (this.transitions["left"].action){
485             this._transitionCallbacks["-"+SC.platform.cssPrefix+"-transform"] = {
486               source: this,
487               target: (this.transitions["left"].target || this),
488               action: this.transitions["left"].action
489             };
490           }
491 
492           if (this.transitions["top"].action){
493             this._transitionCallbacks["-"+SC.platform.cssPrefix+"-transform"] = {
494               source: this,
495               target: (this.transitions["top"].target || this),
496               action: this.transitions["top"].action
497             };
498           }
499           continue;
500         }
501         ////**END SPECIAL TRANSFORM CASE**////
502 
503         // get timing function
504         timing_function = this.cssTimingStringFor(this.transitions[i]);
505 
506         // sanitize name
507         cssTransitions.push(this._cssTransitionFor[i] + " " + this.transitions[i].duration + "s " + timing_function);
508       }
509     }
510 
511     for (i in newStyle)
512     {
513       if (i[0] == "_") continue; // guid (or something else we can't deal with anyway)
514 
515 
516       // if it needs to be set right away since it is not animatable, _getStartStyleHash
517       // will have done that. But if we aren't supposed to animate it, we need to know, now.
518       var shouldSetImmediately = !this.transitions[i] || newStyle[i] == startingPoint[i];
519       if (i == "display" && newStyle[i] != "none") shouldSetImmediately = true;
520 
521       if (shouldSetImmediately)
522       {
523         // set
524         startingPoint[i] = newStyle[i];
525 
526         // you can't easily stop the animator. So just set its endpoint and make it end soon.
527         var animator = this._animators[i];
528         if (animator)
529         {
530           animator.endValue = newStyle[i];
531           animator.end = 0;
532         }
533         continue;
534       }
535 
536       // If there is an available CSS transition, use that.
537       if (SC.platform.supportsCSSTransitions && this._cssTransitionFor[i])
538       {
539         // the transition is already set up.
540         // we can just set it as part of the starting point
541         endingPoint[i] = newStyle[i];
542 
543         if (this.transitions[i].action){
544           this._transitionCallbacks[this._cssTransitionFor[i]] = {
545             source: this,
546             target: (this.transitions[i].target || this),
547             action: this.transitions[i].action
548           };
549         }
550 
551         continue;
552       }
553 
554       // well well well... looks like we need to animate. Prepare an animation structure.
555       // (WHY ARE WE ALWAYS PREPARING?)
556       var applier = this._animateTickPixel,
557       property = i,
558       startValue = startingPoint[i],
559       endValue = newStyle[i];
560 
561       // special property stuff
562       if (property == "centerX" || property == "centerY")
563       {
564         // uh... need a special applier; it needs to update currentlayout differently than actual
565         // layout, since one gets "layout," and the other gets styles.
566         applier = this._animateTickCenter;
567       }
568       else if (property == "opacity")
569       {
570         applier = this._animateTickNumber;
571       }
572       else if (property == "display")
573       {
574         applier = this._animateTickDisplay;
575       }
576 
577       // cache animator objects, not for memory, but so we can modify them.
578       if (!this._animators[i]) this._animators[i] = {};
579 
580       // used to mixin a struct. But I think that would create a new struct.
581       // also, why waste cycles on a SC.mixin()? So I go the direct approach.
582       var a = this._animators[i];
583 
584       // set settings...
585       // start: Date.now(), // you could put this here. But it is better to wait. The animation is smoother
586       // if its beginning time is whenever the first frame fires.
587       // otherwise, if there is a big delay before the first frame (perhaps we are animating other elements)
588       // the items will "jump" unattractively
589       a.start = null;
590       a.duration = Math.floor(this.transitions[i].duration * 1000);
591       a.startValue = startValue;
592       a.endValue = endValue;
593       a.layer = layer;
594       a.property = property;
595       a.action = applier;
596       a.style = layer.style;
597       a.holder = this;
598 
599       if (this.transitions[i].action){
600         a.callback = {
601           source: this,
602           target: (this.transitions[i].target || this),
603           action: this.transitions[i].action
604         };
605       }
606 
607       timing = this.transitions[i].timing || SC.Animatable.defaultTimingFunction;
608       if (timing && SC.typeOf(timing) != SC.T_STRING) a.timingFunction = timing;
609 
610       // add timer
611       if (!a.going) this._animationsToStart[i] = a;
612     }
613 
614     // start animations, in order
615     var ao = this._animationOrder, l = this._animationOrder.length;
616     for (i = 0; i < l; i++)
617     {
618       var nextAnimation = ao[i];
619       if (this._animationsToStart[nextAnimation])
620       {
621         SC.Animatable.addTimer(this._animationsToStart[nextAnimation]);
622         delete this._animationsToStart[nextAnimation];
623       }
624     }
625 
626     // and update layout to the normalized start.
627     var css = cssTransitions.join(",");
628     cssTransitions.length = "";
629     this._animatableSetCSS = css;
630 
631     // apply starting styles directly to layer
632     this._animatableApplyStyles(layer, startingPoint, endingPoint);
633 
634     // all our timers are scheduled, we should be good to go. YAY.
635     return this;
636 
637   },

639   /**
640   @private
641   Adjusts display and queues a change for the other properties.
642 
643   layer: the layer to modify
644   styles: the styles to set
645   delayed: styles to set after a brief delay (if any)
646   */
647   _animatableApplyStyles: function(layer, styles, delayed)
648   {
649     if (!layer) return;
650 
651     // handle a specific style first: display. There is a special case because it disrupts transitions.
652     var needsRender = NO;
653     if (styles["display"] && layer.style["display"] !== styles["display"]) {
654       layer.style["display"] = styles["display"];
655       needsRender = YES;
656     }
657 
658     // set CSS transitions very first thing
659     if (this._animatableSetCSS != this._last_transition_css) {
660       layer.style[SC.platform.domCSSPrefix+"Transition"] = this._animatableSetCSS;
661       this._last_transition_css = this._animatableSetCSS;
662       needsRender = YES;
663     }
664 
665     if (!this._animators["display-styles"]) this._animators["display-styles"] = {};
666 
667     // get timer
668     var timer = this._animators["display-styles"];
669 
670     // set settings
671     timer.holder = this;
672     timer.action = this._animatableApplyNonDisplayStylesFromTimer;
673     timer.inLoopAction = this._animatableApplyNonDisplayStyles;
674     timer.layer = layer;
675     timer.styles = styles;
676     timer.delayed = delayed;
677     this._animatableCurrentStyle = styles;
678 
679     // schedule.
680     if (this._disableAnimation > 0 || !needsRender) {
681       timer.inLoopAction();
682     } else {
683       // after setting transition or display, we must wait a moment;
684       // otherwise, no animation will happen.
685       SC.Animatable.addTimer(timer);
686     }
687   },

790   /**
791   @private
792   Solves cubic bezier curves. Basically, returns the Y for the supplied X.
793 
794   I have only a vague idea of how this works. But I do have a vague idea. It is originally
795   from WebKit's source code:
796   http://trac.webkit.org/browser/trunk/WebCore/platform/graphics/UnitBezier.h?rev=31808
797   */
798   _solveBezierForT: function(ax, ay, bx, by, cx, cy, x, duration) {
799     // determines accuracy. Which means animation is slower for longer duration animations.
800     // that seems ironic, for some reason, but I don't know why.
801 
802     // SOME OPTIMIZATIONS COULD BE DONE, LIKE MOVING THIS INTO ITS OWN BIT AT BEGINNING OF ANIMATION.
803     var epsilon = 1.0 / (200.0 * duration);
804 
805     // a method I have NO idea about... Newton's method
806     var t0, t1, t2, x2, d2, i;
807     for (t2 = x, i = 0; i < 8; i++) {
808       x2 = ((ax * t2 + bx) * t2 + cx) * t2 - x; // sample curve x for t2, - x
809       if (Math.abs(x2) < epsilon) return t2; // obviously, this is determining the accuracy
810       d2 = (3.0 * ax * t2 + 2.0 * bx) * t2 + cx;
811       if (Math.abs(d2) < Math.pow(10, -6)) break;
812       t2 = t2 - x2 / d2;
813     }
814 
815     // fall back to bisection
816     t0 = 0.0;
817     t1 = 1.0;
818     t2 = x;
819     if (t2 < t0) return t0;
820     if (t2 > t1) return t1;
821     while (t0 < t1) {
822       x2 = ((ax * t2 + bx) * t2 + cx) * t2;
823       if (Math.abs(x2 - x) < epsilon) return t2;
824 
825       if (x > x2) t0 = t2;
826       else t1 = t2;
827 
828       t2 = (t1 - t0) * 0.5 + t0;
829     }
830 
831     return t2; // on failure
832   },
833 
834   _solveBezier: function(p1x, p1y, p2x, p2y, x, duration) {
835     // calculate coefficients
836     var cx = 3.0 * p1x;
837     var bx = 3.0 * (p2x - p1x) - cx;
838     var ax = 1.0 - cx - bx;
839 
840     var cy = 3.0 * p1y;
841     var by = 3.0 * (p2y - p1y) - cy;
842     var ay = 1.0 - cy - by;
843 
844     var t = this._solveBezierForT(ax, ay, bx, by, cx, cy, x, duration);
845 
846     // now calculate Y
847     return ((ay * t + by) * t + cy) * t;
848   },
