  2 
  3 sc_require('controllers/statechart_controller');
  4 
  5 Test.statechart = SC.Statechart.create({
  6   
  7   trace: YES,
  8   
  9   initialState: 'loggedOutState',
 10   
 11   delegate: Test.statechartController,
 12   
 13   loggedOutState: SC.State.design({
 14     
 15     enterState: function() {

 28         ctx.retry();
 29       } else {
 30         this.gotoState('loggedInState');
 31       }
 32     }
 33     
 34   }),
 35   
 36   loggedInState: SC.State.design({
 37     
 38     enterState: function() {
 39       Test.mainPage.get('mainPane').append();
 40     },
 41

 43       this.gotoState('fooState');
 44     },
 45     
 46     switchToBarMode: function() {

 48     },
 49     
 50     initialSubstate: 'fooState',
 51     
 52     fooState: SC.State.design({
 53       
 54       representRoute: 'foo',
 55

 59       }
 60       
 61     }),
 62     
 63     barState: SC.State.design({
 64 
 65       representRoute: 'bar/:id',
 66       
 67       enterState: function() {
