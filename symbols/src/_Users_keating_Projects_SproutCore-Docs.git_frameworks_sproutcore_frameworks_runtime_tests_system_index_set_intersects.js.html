  9 var set ;
 10 module("SC.IndexSet#intersects", {
 11   setup: function() {
 12     set = SC.IndexSet.create().add(1000, 10).add(2000,1);
 13   }
 14 });
 15 
 16 // ..........................................................
 17 // SINGLE INDEX
 18 // 
 19 
 20 test("handle index in set", function() {
 21   equals(set.intersects(1001), YES, 'index 1001 should be in set %@'.fmt(set));
 22   equals(set.intersects(1009), YES, 'index 1009 should be in set %@'.fmt(set));
 23   equals(set.intersects(2000), YES, 'index 2000 should be in set %@'.fmt(set));
 24 });
 25 
 26 test("handle index not in set", function() {
 27   equals(set.intersects(0), NO, 'index 0 should not be in set');
 28   equals(set.intersects(10), NO, 'index 10 should not be in set');
 29   equals(set.intersects(1100), NO, 'index 1100 should not be in set');
 30 });
 31 
 32 test("handle index past end of set", function() {
 33   equals(set.intersects(3000), NO, 'index 3000 should not be in set');
 34 });
 35 
 36 // ..........................................................
 37 // RANGE
 38 // 
 39 
 40 test("handle range inside set", function() {
 41   equals(set.intersects(1001,4), YES, '1001..1003 should be in set');
 42 });
 43 
 44 test("handle range outside of set", function() {
 45   equals(set.intersects(100,4), NO, '100..1003 should NOT be in set');
 46 });
 47 
 48 test("handle range partially inside set", function() {
 49   equals(set.intersects(998,4), YES,'998..1001 should be in set');
 50 });
 51 
 52 // ..........................................................
 53 // INDEX SET
 54 // 
 55 
 56 test("handle set inside IndexSet", function() {
 57   var test = SC.IndexSet.create().add(1001,4).add(1005,2);
 58   equals(set.intersects(test), YES, '%@ should be in %@'.fmt(test, set));
 59 });
 60 
 61 test("handle range outside of IndexSet", function() {
 62   var test = SC.IndexSet.create().add(100,4).add(105,2);
 63   equals(set.intersects(test), NO, '%@ should be in %@'.fmt(test, set));
 64 });
 65 
 66 test("handle range partially inside IndexSet", function() {
 67   var test = SC.IndexSet.create().add(1001,4).add(100,2);
 68   equals(set.intersects(test), YES, '%@ should be in %@'.fmt(test, set));
 69 });
 70 
 71 test("handle self", function() {
 72   equals(set.contains(set), YES, 'should return YES when passed itself');  
 73 });
