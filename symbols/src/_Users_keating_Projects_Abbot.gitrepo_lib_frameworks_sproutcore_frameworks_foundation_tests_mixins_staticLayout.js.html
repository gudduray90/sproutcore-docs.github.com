 19 /*
 20   helper method to test the layout of a view.  Applies the passed layout to a
 21   view, then compares both its frame and layoutStyle properties both before
 22   and after adding the view to a parent view.  
 23 
 24   You can pass frame rects with some properties missing and they will be
 25   filled in for you just so you don't have to write so much code.
 26   
 27   @param {Hash} layout layout hash to test
 28   @param {Hash} no_f expected frame for view with no parent
 29   @param {Hash} no_s expected layoutStyle for view with no parent
 30   @param {Hash} with_f expected frame for view with parent
 31   @param {Hash} with_s expected layoutStyle for view with parent
 32   @returns {void}
 33 */
 34 function performLayoutTest(layout, no_f, no_s, with_f, with_s) {
 35   
 36   // make sure we add null properties and convert numbers to 'XXpx' to style layout.
 37   var keys = 'width height top bottom marginLeft marginTop left right zIndex minWidth maxWidth minHeight maxHeight'.w();
 38   keys.forEach(function(key) {
 39     if (no_s[key]===undefined) no_s[key] = null;
 40     if (with_s[key]===undefined) with_s[key] = null;  
 41 
 42     if (typeof no_s[key] === 'number') no_s[key] = no_s[key].toString() + 'px';
 43     if (typeof with_s[key] === 'number') with_s[key] = no_s[key].toString() + 'px';
 44   });
 45   
 46   // set layout
 47   child.set('layout', layout) ;
 48 
 49   // test
 50   keys.forEach(function(key) {
 51     equals(child.get('layoutStyle')[key], no_s[key], "STYLE NO PARENT %@".fmt(key)) ;  
 52   });
 53   
 54   // add parent
 55   SC.RunLoop.begin();
 56   parent.appendChild(child);
 57   SC.RunLoop.end();
 58   
 59   // test again
 60   keys.forEach(function(key) {
 61     equals(child.get('layoutStyle')[key], with_s[key], "STYLE NO PARENT %@".fmt(key)) ;  
 62   });
 63 }

 89 test("Test that auto as a value for width height is set correctly when"
 90   +" setting the element style", function() {
 91   child = SC.View.create({
 92     useStaticLayout:YES,
 93     render: function(context) {
 94       // needed for auto
 95       context.push('<div style="padding: 10px"></div>');
 96     }
 97   });
 98 
 99   // parent MUST have a layer.
100   parent.createLayer();
101   var layer = parent.get('layer');
102   document.body.appendChild(layer);
103   
104   var layout = { top: 0, left: 0, width: 'auto', height: 'auto' };
105   var no_f = { x: 0, y: 0, width: 0, height: 0 };
106   var with_f = { x: 0, y: 0, width: 20, height: 20 };
107   var s = { top: 0, left: 0, width: 'auto', height: 'auto' };
108   
109   performLayoutTest(layout, no_f, s, with_f, s);
110   
111   layer.parentNode.removeChild(layer);
112 });
