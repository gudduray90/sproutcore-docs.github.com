 10     loudly: function(string) {
 11       return string + this.get('exclaim');

 13     exclaim: "!"

 15 
 16   Klass.reopen({

163 // When creating a new instance of a class or extending a class, SproutCore
164 // keeps a record of the object that calls to sc_super should be looked up
165 // on.
166 //
167 // Calls to sc_super are dynamic, which means that you can modify a class or
168 // superclass of an object at runtime, and sc_super will correctly pick up
169 // those changes. This is especially important for calls to reopen and
170 // enhance.
171 test("__sc_super__ semantics", function() {
172   var rootObject = SC.Object.create({});
173   ok(rootObject.__sc_super__ === SC.Object.prototype, "SproutCore remembers that new SC.Objects should super to SC.Object.prototype");
174 
175   var basicObject = new SC.Object();
176   ok(basicObject.__sc_super__ === SC.Object.prototype, "SproutCore remembers that SC.Objects created by new SC.Object should super to SC.Object.prototype");
177 
178   var Klass = SC.Object.extend({});
179   ok(Klass.__sc_super__ === SC.Object.prototype, "SproutCore remembers the original begetted prototype for subclasses");
180 
181   var object = Klass.create({});
182   ok(object.__sc_super__ === Klass.prototype, "SproutCore remembers the original prototype for new instances");
183 
184   var basicSubclassObject = new Klass();
185   ok(basicSubclassObject.__sc_super__ === Klass.prototype, "SproutCore remembers the original prototype for new instances created with new");
186 
187   var SubKlass = Klass.extend({});
188   ok(SubKlass.__sc_super__ === Klass.prototype, "SproutCore remembers the original begetted prototype for custom subclasses");
189 
190   SubKlass.reopen({});
191   ok(SubKlass.__sc_super__ === Klass.prototype, "Reopen doesn't break prototype recordkeeping");
192 });

208 /**
209   There was a bug that defining a subclass prior to reopening the parent class
210   and adding a computed property, binding or observer to the parent, caused the
211   subclass not to register the property, binding or observer correctly.
212 */
213 test("reopening a class that has been subclassed, updates the subclasses properties, bindings and observers", function() {
214   var propertyChanged = 0, observerCalled = 0;
215   MyBindable = SC.Object.create({ a: 1, b: 2 });
216   var MyClass = SC.Object.extend({ property: function() { }.property() });
217 
218   // Define a subclass prior to the reopen().
219   var MySubclass = MyClass.extend({ anotherProperty: function() { return "My Own Property"; }.property(), c: 3 });
220 
221   // Reopen and add a computed property, binding and observer to the parent class
222   MyClass.reopen({
223     anotherProperty: function() {
224       return "Another Property";
225     }.property(),
226     yetAnotherProperty: function() {
227       return "Yet Another Property";
228     }.property('property'),
229     a: 1,
230     aBinding: SC.Binding.oneWay("MyBindable.a"),
231     bObserver: function() {
232       observerCalled++;
233     }.observes('MyBindable.b'),
234     c: 4
235   });
236 
237   // Define another subclass after the reopen().
238   var MyOtherSubclass = MyClass.extend({ stillOneOtherProperty: function() { }.property() });
239 
240   // Create instances of each for comparison.
241   var myClass = MyClass.create();
242   myClass.addObserver('yetAnotherProperty', function() {
243     propertyChanged++;
244   });
245 
246   var mySubclass = MySubclass.create();
247   mySubclass.addObserver('yetAnotherProperty', function() {
248     propertyChanged++;
249   });
250 
251   var myOtherSubclass = MyOtherSubclass.create();
252   myOtherSubclass.addObserver('yetAnotherProperty', function() {
253     propertyChanged++;
254   });
255 
256   SC.run(function() {
257     myClass.set('property', "foo");
258     mySubclass.set('property', "foo");
259     myOtherSubclass.set('property', "foo");
260     MyBindable.set('a', 2);
261     MyBindable.set('b', 3);
262   });
263 
264   equals(propertyChanged, 3, "property invalidated thrice");
265   equals(observerCalled, 3, "fires observer thrice");
266   equals(myClass.get('a'), 2, "myClass bound value should be 2");
267   equals(mySubclass.get('a'), 2, "mySubclass bound value should be 2");
268   equals(myOtherSubclass.get('a'), 2, "myOtherSubclass bound value should be 2");
269   equals(mySubclass.get('c'), 3, "mySubclass overridden property should still be set");
270   equals(mySubclass.get('anotherProperty'), "My Own Property", "mySubclass overridden computed property should still exist");
271 });
