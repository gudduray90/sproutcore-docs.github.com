  8 /**
  9  * Returns the ordinal associated for the current number.
 10  */
 11 SC.methodForLocale('en', 'ordinalForNumber', function(number) {
 12   var d = number % 10;
 13   return (~~ (number % 100 / 10) === 1) ? 'th' :
 14          (d === 1) ? 'st' :
 15          (d === 2) ? 'nd' :
 16          (d === 3) ? 'rd' : 'th';
 17 });
