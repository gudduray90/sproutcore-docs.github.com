  8 var view, del, content;
  9 module("SC.ListView.rowDelegate", {
 10   setup: function() {
 11     del = SC.Object.create(SC.CollectionRowDelegate);
 12 
 13     // fake empty array that implements delegate
 14     content = SC.Object.create(SC.CollectionRowDelegate, SC.Array, SC.Freezable, {
 15 
 16       isFrozen: YES,
 17 
 18       length: 0,
 19 
 20       objectAt: function(idx) { return undefined; }
 21     });

 23   }
 24 });

 30 test("no delegate and no content", function() {
 31   view = SC.ListView.create();
 32   equals(view.get('delegate'), null, 'precond - delegate should be null');
 33   equals(view.get('content'), null, 'precond - content should be null');

 37 test("with no delegate and content not delegate", function() {
 38   var array = [];
 39   view = SC.ListView.create({ content: array });
 40 
 41   equals(view.get('delegate'), null, 'precond - delegate should be null');
 42   ok(!view.get('content').isCollectionRowDelegate, 'precond - content should not be CollectionRowDelegate');
 43   equals(view.get('rowDelegate'), view, 'default row delegate should view');

 51   equals(view.get('rowDelegate'), content, 'row delegate should be content');
 52 });

 66 test("modifying delegate and content", function() {
 67   var callCount = 0 ;
 68 
 69   view = SC.ListView.create();
 70   view.addObserver('rowDelegate', function() { callCount++; });
 71 
 72   equals(view.get('delegate'), null, 'precond - delegate should be null');
 73   equals(view.get('content'), null, 'precond - content should be null');
 74   equals(view.get('rowDelegate'), view, 'default row delegate should view');
 75 
 76   // test setting content
 77   callCount=0;
 78   SC.run(function() { view.set('content', content); });
 79   ok(callCount>0, 'rowDelegate should change when setting content');
 80   equals(view.get('rowDelegate'), content, 'rowDelegate should change after setting content');
 81 
 82   // test setting delegate
 83   callCount=0;
 84   SC.run(function() { view.set('delegate', del); });
 85   ok(callCount>0, 'rowDelegate should change when setting delegate');
 86   equals(view.get('rowDelegate'), del, 'rowDelegate should change to delegate after setting delegate');
 87 
 88   // test changing content
 89   callCount=0;
 90   SC.run(function() { view.set('content', []); });
 91   ok(callCount>0, 'rowDelegate should change when setting content');
 92   equals(view.get('rowDelegate'), del, 'rowDelegate should stay delegate as long as delegate remains');
 93 
 94   // test changing delegate
 95   callCount=0;
 96   SC.run(function() { view.set('delegate', null); });
 97   ok(callCount>0, 'rowDelegate should change when setting delegate');
 98   equals(view.get('rowDelegate'), view, 'rowDelegate should go back to view when delegate and content cleared or do not implement');
 99 
100 });

106 test("changing the rowHeight should invalidate all row heights", function() {
107   var indexes = null;
108   view = SC.ListView.create({
109     content:  '1 2 3 4 5'.w(), // provide some content
110     delegate: del,
111 
112     // override for testing
113     rowSizeDidChangeForIndexes: function(passed) {
114       indexes = passed;
115     }
116   });
117 
118   indexes = null;
119   del.set('rowHeight', 30);
120   same(indexes, SC.IndexSet.create(0,5), 'changing row height should call rowSizeDidChangeForIndexes(0..5)');
121 
122   // remove del and try again to verify that it stops tracking changes
123   view.set('delegate', null);
124   indexes = null;
125   del.set('rowHeight', 23);
126   equals(indexes, null, 'once delegate is removed changed rowHeight should not impact anything');
127 
128   // change row height on the view without a delegate to verify new observers
129   // are setup.
130   indexes = null;
131   view.set('rowHeight', 14);
132   same(indexes, SC.IndexSet.create(0,5), 'changing row height should call rowSizeDidChangeForIndexes(0..5)');
133 
134 });

137 test("changing the customRowHeightIndexes should invalidate impacted row heights", function() {
138   var indexes = null;
139   view = SC.ListView.create({
140     content:  '1 2 3 4 5'.w(), // provide some content
141     delegate: del,
142 
143     // override for testing
144     rowSizeDidChangeForIndexes: function(passed) {
145       indexes = passed;
146     }
147   });
148 
149   // try changing the index set
150   indexes = null;
151   var set = SC.IndexSet.create(2,2);
152   del.set('customRowHeightIndexes', set);
153   same(indexes, set, 'setting customRowHeightIndexes for first time should invalidate indexes only');
154 
155   // modify the set
156   indexes = null ;
157   set.add(1,3);
158   same(indexes, set, 'modifying index set should invalidate all old and new indexes');
159 
160   // changing to a new set
161   indexes =null ;
162   var set2 = SC.IndexSet.create(0,1);
163   del.set('customRowHeightIndexes', set2);
164   same(indexes, set.copy().add(set2), 'setting new indexes should invalidate both old and new indexes');
165 
166 
167   // remove an set
168   indexes =null ;
169   del.set('customRowHeightIndexes', null);
170   same(indexes, set2, 'setting indexes to null should invalidate old index set');
171 
172   // try removing delegate
173   del.set('customRowHeightIndexes', set2);
174   indexes = null;
175   view.set('delegate', null);
176   same(indexes, set2, 'removing delegate should invalidate old index set');
177 
178   // change delegate once removed.
179   indexes = null ;
180   del.set('customRowHeightIndexes', set);
181   equals(indexes, null, 'modifying delegate once removed should not change view');
182 
183 });
