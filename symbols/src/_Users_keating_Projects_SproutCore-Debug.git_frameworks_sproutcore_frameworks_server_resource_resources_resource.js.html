 27   createRoutes: function() {
 28     var routeNames = this.get('routeNames'),
 29         routeName,
 30         route;
 31 
 32     this.beginPropertyChanges();
 33     
 34     // Swap the array
 35     for (var i = 0, length = routeNames.length; i < length; i++) {
 36 
 37       routeName = routeNames[i];
 38       route = this[routeName];
 39       
 40       if (route) {
 41         route = this.createRoute(routeName, route);
 42       } else {
 43         SC.Logger.error("%@: No route with name %@ was found in the class.".fmt(this, routeName));
 44       }
 45     }
 46     
 47     this.endPropertyChanges();
 48   },
 49 
 50   createRoute: function(routeName, route, attrs) {
 51     console.log('createRoute(%@, %@)'.fmt(routeName, route));
 52     var silentRoutes = this.get('silentRoutes'),
 53         routes = this.get('routes'),
 54         name = this.get('name'),
 55         path;
 56 
 57     if (!route.isClass) {
 58       attrs = route;
 59     } else {
 60       // attrs should always exist...
 61       if (!attrs) { attrs = {}; }
 62       // clone the hash that was given so we dont pollute it if it's being reused
 63       else { attrs = SC.clone(attrs); }
 64     }
 65     
 66     attrs.name = routeName;
 67     attrs.resource = this;
 68 
 69     // Now add this to the attributes and create.
 70     if (route.isClass) { route = route.create(attrs); }
 71 
 72     // Store the route in routes and replace it with a function of the same name
 73     routes[routeName] = route;
 74     this[routeName] = this.routeMethod(route);
 75 
 76     return route;
 77   },
 78 
 79   routeMethod: function(route) {
 80     return function(id, params) {
 81       this.callRoute(route.get('name'), id, params);
 82     };
 83   },

 85   /**

 89    */
 90   callRoute: function(routeName, id, params) {
 91     var routes = this.get('routes'),
 92         route,
 93         address,
 94         body,
 95         request;
 96     
 97     console.log('callRoute');
 98     
 99     if (params && SC.typeOf(params) !== SC.T_HASH) {
100       SC.Logger.error("%@ params must be an object (ie. hash)".fmt(this));
101     }
102     
103     if (routeName && routes[routeName]) {
104       route = routes[routeName];
105     } else {
106       SC.Logger.error("%@.callRoute() must be passed a routeName argument.".fmt(this));
107     }
108     
109     if (route) {
110       route.willSendRequest(params);
111       address = this.path(routeName);
112       
113       if (params) {
114         if (route.httpMethod === SC.HTTP_METHOD.GET || route.httpMethod === SC.HTTP_METHOD.DELETE) address += "?" + this.stringFromParams(params);
115         else body = route.bodyForRequest(params);
116       }
117       
118       request = SC.Request.create({
119         type: route.httpMethod,
120         isJSON: route.format === SC.RESOURCE_ROUTE_FORMAT.JSON,
121         isXML: route.format === SC.RESOURCE_ROUTE_FORMAT.XML,
122         address: address,
123         body: body
124       });
125       
126       request.notify(route, route.didComplete).send();
127     }
128   },
129 
130   path: function(routeName) {
131     var silentRoutes = this.get('silentRoutes'),
132         name = this.get('name'),
133         path;
134     
135     path = name.pluralize();
136     if (routeName && silentRoutes.indexOf(routeName) < 0) path += "/" + routeName;
137     
138     return path;
139   },
