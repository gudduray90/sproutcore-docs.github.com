 12 module("SC.ERROR");

 14 test("SC.Error.desc creates an error instance with description,label and code", function() {
 15   var c = SC.Error.desc('This is an error instance','Error Instance', "FOO", 99999);
 16   equals(SC.T_ERROR,SC.typeOf(c),'Error instance');
 17   equals('This is an error instance',c.message,'Description');
 18   equals('Error Instance',c.label,'Label');
 19   equals(c.get('errorValue'), "FOO", 'error value should be set');
 20   equals(99999,c.code,'Code');
 21 });
 22 
 23 test("SC.$error creates an error instance with description,label and code",function(){
 24   var d = SC.$error('This is a new error instance','New Error Instance', "FOO", 99999);
 25   equals(SC.T_ERROR,SC.typeOf(d),'New Error instance');
 26   equals('This is a new error instance',d.message,'Description');
 27   equals('New Error Instance',d.label,'Label');
 28   equals(d.get('errorValue'), "FOO", 'error value should be set');
 29   equals(99999,d.code,'Code');
 30 });

 32 test("SC.$ok should return YES if the passed value is an error object or false", function() {
 33   ok(SC.$ok(true), '$ok(true) should be YES');
 34   ok(!SC.$ok(false), '$ok(false) should be NO');
 35   ok(SC.$ok(null), '$ok(null) should be YES');
 36   ok(SC.$ok(undefined), '$ok(undefined) should be YES');
 37   ok(SC.$ok("foo"), '$ok(foo) should be YES');
 38   ok(!SC.$ok(SC.$error("foo")), '$ok(SC.Error) should be NO');
 39 
 40   ok(!SC.$ok(new SC.Error()), '$ok(Error) should be NO');
 41   ok(!SC.$ok(SC.Object.create({ isError: YES })), '$ok({ isError: YES }) should be NO');
 42 });
 43 
 44 test("SC.$val should return the error value if it has one", function() {
 45   equals(SC.val(true), true, 'val(true) should be true');
 46   equals(SC.val(false), false, 'val(false) should be false');
 47   equals(SC.val(null), null, 'val(null) should be YES');
 48   equals(SC.val(undefined), undefined, '$ok(undefined) should be YES');
 49   equals(SC.val("foo"), "foo", 'val(foo) should be YES');
 50   equals(SC.val(SC.$error("foo", "FOO", "BAZ")), "BAZ", 'val(SC.Error) should be BAZ');
 51   equals(SC.val(SC.$error("foo", "FOO")), undefined, 'val(SC.Error) should be undefined');
 52   equals(SC.val(new SC.Error()), null, 'val(Error) should be null');
 53   equals(SC.val(SC.Object.create({ isError: YES, errorValue: "BAR" })), "BAR", 'val({ isError: YES, errorValue: BAR }) should be BAR');
 54 });
 55 
 56 test("errorObject property should return the error itself", function() {
 57   var er = SC.$error("foo");
 58   equals(er.get('errorObject'), er, 'errorObject should return receiver');
 59 });
