 10 /**
 11   @class
 12   Use this mixin to make your view automatically resize based upon its value,
 13   title, or other string property. Only works for views that support automatic
 14   resizing.
 15 
 16   Supporting Automatic Resizing
 17   -------------------------------------
 18   To support automatic resizing, your view must provide these properties:
 19 
 20   - *`supportsAutoResize`.* Must be set to YES.
 21 
 22   - *`autoResizeLayer`* A DOM element to use as a template for resizing the
 23     view. Font sizes and other styles will be copied to the measuring element
 24     SproutCore uses to measure the text.
 25 
 26   - *`autoResizeText`.* The text to measure. A button view might make a proxy
 27     to its `displayTitle`, for instance.
 28 
 29   Your view may also supply:
 30 
 31   - *`autoResizePadding`.* An amount to add to the measured size. This may be either
 32     a single number to be added to both width and height, or a hash containing
 33     separate `width` and `height` properties.
 34 
 35 
 36   NOTE: these properties are not defined in the mixin itself because the supporting view,
 37   rather than the user of SC.AutoResize, will be providing the properties, and mixing
 38   SC.AutoResize into the view should not override these properties.
 39 */
 40 SC.AutoResize = {
 41   /*@scope SC.AutoResize.prototype */
 42 
 43   /**
 44     If YES, automatically resizes the view (default). If NO, only measures,
 45     setting 'measuredSize' to the measured value (you can bind to measuredSize
 46     and update size manually).
 47 
 48     @type Boolean
 49     @default YES
 50   */
 51   shouldAutoResize: YES,
 52 
 53   /**
 54     If NO, prevents SC.AutoResize from doing anything at all.
 55 
 56  57     @type Boolean
 58     @default YES
 59   */
 60   shouldMeasureSize: YES,
 61 
 62   /**
 63     Determines if the view's width should be resized
 64     on calculation.
 65 
 66     @type Boolean
 67     @default YES
 68   */
 69   shouldResizeWidth: YES,
 70 
 71   /**
 72     Determines if the view's height should be resized
 73     on calculation. Default is NO to retain backwards
 74     compatibility.
 75     
 76     @type Boolean
 77     @default NO
 78   */
 79   shouldResizeHeight: NO,
 80   
 81   /**
 82     The measured size of the view's content (the value of the autoResizeField). 
 83     This property is observable, and, if used in conjunction with setting
 84     shouldAutoResize to NO, allows you to customize the 'sizing' part, using
 85     SC.AutoResize purely for its measuring code.
 86     
 87     @type Rect
 88   */
 89   measuredSize: { width: 0, height: 0 },
 90 
 91   /**
 92     @private
 93     Begins observing the auto resize field.
 94   */
 95   initMixin: function() {
 96     // @if (debug)
 97     if (!this.get('supportsAutoResize')) {
 98       throw "View `%@` does not support automatic resize. See documentation for SC.AutoResize".fmt(this);
 99     }
100     // @endif
101   },
102 
103   /**
104     Observe the autoResizePadding so we can update our measurements if it changes.
105 
106     @private
107   */
108   _scar_autoResizePaddingDidChange: function() {
109     this.invokeOnce('measureSize');
110   }.observes('autoResizePadding'),
111 
112 
113   /**
114     If this property is provided, all views that share the same value for this property will be resized as a batch for increased performance.
115 
116     @type String
117   */
118   batchResizeId: null,
119 
120   /** @private */
121   _scar_measurementPending: NO,
122   
123   /** @private */
124   _scar_requestedBatchResizeId: null,
125 
126   /** @private
127     If the batch id changed while a request is out, we have to fix it
128   */
129   _scar_batchResizeIdDidChange: function() {
130     var batchResizeId = this.get('batchResizeId'),
131     requestedBatchResizeId = this._scar_requestedBatchResizeId;
132 
133     // check if a request is out and the id changed
134     if(this._scar_measurementPending && this._scar_requestedBatchResizeId !== batchResizeId) {
135       // if so, cancel the old request and make a new one
136       SC.AutoResizeManager.cancelMeasurementForView(this, requestedBatchResizeId);
137       SC.AutoResizeManager.scheduleMeasurementForView(this, batchResizeId);
138 
139       // update the requested batchResizeId to the new id
140       this._scar_requestedBatchResizeId = batchResizeId;
141     }
142   }.observes('batchResizeId'),
143 
144   /**
145     Schedules a measurement to happen later, in batch mode. Only valid when the view
146     has a `batchResizeId`.
147   */
148   scheduleMeasurement: function() {
149     if (!this.get('shouldMeasureSize')) {
150       return;
151     }
152 
153     var batchResizeId = this.get('batchResizeId');
154     SC.AutoResizeManager.scheduleMeasurementForView(this, batchResizeId);
155 
156     this._scar_measurementPending = YES;
157     this._scar_requestedBatchResizeId = batchResizeId;
158   }.observes('isVisible'),
159 
160   /**
161     Measures the size of the view.
162 
163     @param batch For internal use during batch resizing.
164   */
165   measureSize: function(batch) {
166     var metrics, layer, value = this.get('autoResizeText'),
167         autoSizePadding, paddingHeight, paddingWidth,
168         ignoreEscape = !this.get('escapeHTML');
169 
170     // There are two special cases.
171     //   - empty: we should do nothing. The metrics are 0.
172     //   - batch mode: just call measureString.
173     //
174     // If we are in neither of those special cases, we should go ahead and
175     // resize normally.
176     //
177     if (SC.none(value) || value === "") {
178       metrics = { width: 0, height: 0 };
179     } else if (batch) {
180       metrics = SC.measureString(value, ignoreEscape);
181     } else {
182       // Normal resize pattern: get our own layer, pass it as a template to SC.metricsForString.
183       layer = this.get('autoResizeLayer');
184 
185       if(!layer) {
186         return;
187       }
188 
189       metrics = SC.metricsForString(value, layer, this.get('classNames'), ignoreEscape);
190     }
191 
192     // metrics should include padding
193     autoSizePadding = this.get('autoResizePadding') || 0;
194     if(SC.typeOf(autoSizePadding) === SC.T_NUMBER) {
195       paddingHeight = paddingWidth = autoSizePadding;
196     } else {
197       paddingHeight = autoSizePadding.height;
198       paddingWidth = autoSizePadding.width;
199     }
200 
201     metrics.width += paddingWidth;
202     metrics.height += paddingHeight;
203 
204     // In any case, we set measuredSize.
205     this.set('measuredSize', metrics);
206 
207     if (this.get('shouldAutoResize')) {
208       // if we are allowed to autoresize, adjust the layout
209       if (this.get('shouldResizeWidth')) {
210         this.adjust('width', metrics.width);
211       }
212 
213       if (this.get('shouldResizeHeight')) {
214 215         this.adjust('height', metrics.height);
216       }
217 
218     }
219 
220     this._scar_measurementPending = NO;
221 
222     return metrics;
223   },
224 
225   /**
226     @private
227   */
228   _scar_valueDidChange: function() {
229     this.scheduleMeasurement();
230   }.observes('autoResizeText'),
231 
232   /**
233     @private
234     When the layer is first created, measurement will need to take place.
235   */
236   didCreateLayer: function(orig) {
237     orig();
238 
239     this.scheduleMeasurement();
240   }.enhance()
241 };
242 
243 /**
244  * @private
245  * @class
246  * Manages batch auto resizing.
247  *
248  * This used to be part of SC.AutoResize, but we shouldn't mix these
249  * methods/properties into each view.
250  */
251 SC.AutoResizeManager = {
252 
253   /**
254     A hash of views needing resizing, mapped from batch resize ID to SC.CoreSets
255     of views.
256   */
257   viewsNeedingResize: null,
258 
259   /**
260     Views queued for batch resizing, but with no batch resize id.
261 
262     @property {SC.CoreSet}
263   */
264   untaggedViews: null,
265 
266   /**
267     Schedules a re-measurement for the specified view in the batch with the
268     given id.
269 
270     If a batch does not exist by that id, it will be created. If there is no id,
271     the view will be measured individually.
272 
273     @param view The view to measure.
274     @param id The id of the batch to measure the view in.
275   */
276   scheduleMeasurementForView: function(view, id) {
277     // views with no tag just get put in their own list
278     if(SC.none(id)) {
279       var untaggedViews = this.untaggedViews || (this.untaggedViews = SC.CoreSet.create());
280 
281       untaggedViews.add(view);
282 
283     // views with a batch resize id set for each tag
284     } else {
285       var needResize = this.viewsNeedingResize || (this.viewsNeedingResize = {}),
286       views = needResize[id] || (needResize[id] = SC.CoreSet.create());
287 
288       views.add(view);
289     }
290 
291     SC.RunLoop.currentRunLoop.invokeLast(this.doBatchResize);
292   },
293 
294   /**
295     Cancels a scheduled measurement for a view in the named batch id.
296 
297     @param view The view that was scheduled for measurement.
298     @param id The batch id the view was scheduled in.
299   */
300   cancelMeasurementForView: function(view, id) {
301     var set = SC.none(id) ? this.untaggedViews : this.viewsNeedingResize[id];
302 
303     if(set) {
304       set.remove(view);
305     }
306   },
307 
308   /**
309     Processes all autoResize batches. This will automatically be invoked at the
310     end of any run loop in which measurements were scheduled.
311   */
312   doBatchResize: function() {
313     // make sure we are called from the correct scope.
314     // this will make our property references below clearer.
315     if (this !== SC.AutoResizeManager) {
316       return SC.AutoResizeManager.doBatchResize();
317     }
318 
319     var tag, views, view, layer, batches, prepared;
320 
321     // first measure all the batched views
322     batches = this.viewsNeedingResize;
323     for(tag in batches) {
324       if (batches.hasOwnProperty(tag)) {
325         views = batches[tag];
326 
327         // now measure the rest using the same settings
328         while ((view = views.pop())) {
329           if(view.get('isVisible') && (layer = view.get('autoResizeLayer'))) {
330             if(!prepared) SC.prepareStringMeasurement(layer);
331 
332             view.measureSize(YES);
333           }
334         }
335 
336         // if they were all isVisible:NO, then prepare was never called
337         // so dont call teardown
338         if(prepared) SC.teardownStringMeasurement();
339       }
340     }
341 
342     // measure views with no batch id
343     views = this.untaggedViews;
344 
345     if(!views) {
346       return;
347     }
348 
349     while((view = views.pop())) {
350       view.measureSize();
351     }
352   }
353
