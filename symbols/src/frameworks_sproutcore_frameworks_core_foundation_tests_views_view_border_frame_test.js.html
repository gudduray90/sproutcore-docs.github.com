  6 /*global module, test, same*/

  8 var view;
  9 module("SC.View#borderFrame", {
 10   setup: function () {
 11     SC.run(function () {
 12       view = SC.View.create({
 13         layout: { width: 100, height: 100 }
 14       });
 15     });
 16   },
 17 
 18   teardown: function () {
 19     view.destroy();
 20     view = null;
 21   }
 22 });
 23 
 24 test("The borderFrame property of the view should include the borders from the layout.", function () {
 25   var borderFrame,
 26     frame;
 27 
 28   // No borders.
 29   frame = view.get('frame');
 30   borderFrame = view.get('borderFrame');
 31 
 32   same(frame, { x: 0, y: 0, width: 100, height: 100 }, "The frame without borders is");
 33   same(borderFrame, { x: 0, y: 0, width: 100, height: 100 }, "The borderFrame without borders is");
 34 
 35   // Right 5px border.
 36   SC.run(function () {
 37     view.adjust('borderRight', 5);
 38   });
 39 
 40   frame = view.get('frame');
 41   borderFrame = view.get('borderFrame');
 42 
 43   same(frame, { x: 0, y: 0, width: 95, height: 100 }, "The frame with 5px right border is");
 44   same(borderFrame, { x: 0, y: 0, width: 100, height: 100 }, "The borderFrame with 5px right border is");
 45 
 46   // Top 10px border.
 47   SC.run(function () {
 48     view.adjust('borderTop', 10);
 49   });
 50 
 51   frame = view.get('frame');
 52   borderFrame = view.get('borderFrame');
 53 
 54   same(frame, { x: 0, y: 10, width: 95, height: 90 }, "The frame with 5px right border & 10px top border is");
 55   same(borderFrame, { x: 0, y: 0, width: 100, height: 100 }, "The borderFrame with 5px right border & 10px top border is");
 56 
 57   // Left 15px border.
 58   SC.run(function () {
 59     view.adjust('borderLeft', 15);
 60   });
 61 
 62   frame = view.get('frame');
 63   borderFrame = view.get('borderFrame');
 64 
 65   same(frame, { x: 15, y: 10, width: 80, height: 90 }, "The frame with 5px right border & 10px top border & 15px left border is");
 66   same(borderFrame, { x: 0, y: 0, width: 100, height: 100 }, "The borderFrame with 5px right border & 10px top border & 15px left border is");
 67 
 68   // Bottom 20px border.
 69   SC.run(function () {
 70     view.adjust('borderBottom', 20);
 71   });
 72 
 73   frame = view.get('frame');
 74   borderFrame = view.get('borderFrame');
 75 
 76   same(frame, { x: 15, y: 10, width: 80, height: 70 }, "The frame with 5px right border & 10px top border & 15px left border & 20px bottom border is");
 77   same(borderFrame, { x: 0, y: 0, width: 100, height: 100 }, "The borderFrame with 5px right border & 10px top border & 15px left border & 20px bottom border is");
 78 
 79   // 25px border.
 80   SC.run(function () {
 81     view.set('layout', { width: 100, height: 100, border: 25 });
 82   });
 83 
 84   frame = view.get('frame');
 85   borderFrame = view.get('borderFrame');
 86 
 87   same(frame, { x: 25, y: 25, width: 50, height: 50 }, "The frame with 25px border is");
 88   same(borderFrame, { x: 0, y: 0, width: 100, height: 100 }, "The borderFrame with 25px border is");
 89 
 90 });
