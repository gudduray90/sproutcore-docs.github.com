  8 // NOTE: The test below are based on the Data Hashes state chart.  This models
  9 // the "write" event in the NestedStore portion of the diagram.
 10 
 11 var parent, store, child, storeKey, json;
 12 module("SC.NestedStore#writeDataHash", {
 13   setup: function() {
 14     parent = SC.Store.create();
 15 
 16     json = {
 17       string: "string",
 18       number: 23,
 19       bool:   YES
 20     };
 21 
 22     storeKey = SC.Store.generateStoreKey();
 23 
 24     parent.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
 25     parent.editables = null; // manually patch to setup test state
 26 
 27     store = parent.chain(); // create nested store
 28     child = store.chain();  // test multiple levels deep
 29   }
 30 });
 31 
 32 // ..........................................................
 33 // BASIC STATE TRANSITIONS
 34 //
 35 
 36 // The transition from each base state performs the same operation, so just
 37 // run the same test on each state.
 38 function testWriteDataHash() {
 39   var oldrev = store.revisions[storeKey];
 40 
 41   // perform test
 42   var json2 = { foo: "bar" };
 43   equals(store.writeDataHash(storeKey, json2, SC.Record.READY_NEW), store, 'should return receiver');
 44 
 45   // verify
 46   equals(store.storeKeyEditState(storeKey), SC.Store.EDITABLE, 'new edit state should be editable');
 47 
 48   equals(store.readDataHash(storeKey), json2, 'should have new json data hash');
 49   equals(store.readStatus(storeKey), SC.Record.READY_NEW, 'should have new status');
 50 
 51   equals(store.revisions[storeKey], oldrev, 'should not change revision');
 52   if (!SC.none(oldrev)) {
 53     ok(store.revisions.hasOwnProperty(storeKey), 'should clone reference to revision');
 54   }
 55 }

 90 test("writing a new hash", function() {
 91   storeKey = SC.Store.generateStoreKey(); // new store key!
 92   equals(parent.readDataHash(storeKey), null, 'precond - parent should not have a data hash for store key yet');
 93   equals(store.storeKeyEditState(storeKey), SC.Store.INHERITED, 'precond - edit status should be inherited');
 94 
 95   // perform write
 96   equals(store.writeDataHash(storeKey, json, SC.Record.READY_NEW), store, 'should return receiver');
 97 
 98   // verify change
 99   equals(store.storeKeyEditState(storeKey), SC.Store.EDITABLE, 'new status should be editable');
100   equals(store.readDataHash(storeKey), json, 'should match new json');
101   equals(store.readStatus(storeKey), SC.Record.READY_NEW, 'should have new record status');
102 });

108 test("change should propogate to child if child edit state = INHERITED", function() {
109 
110   // verify preconditions
111   equals(child.storeKeyEditState(storeKey), SC.Store.INHERITED, 'precond - child edit state should be INHERITED');
112 
113   // perform change
114   var json2 = { version: 2 };
115   store.writeDataHash(storeKey, json2, SC.Record.READY_NEW);
116 
117   // verify
118   same(child.readDataHash(storeKey), json2, 'child should pick up change');
119   equals(parent.readDataHash(storeKey), json, 'parent should still have old json');
120 
121   equals(child.readStatus(storeKey), SC.Record.READY_NEW, 'child should pick up new status');
122   equals(parent.readStatus(storeKey), SC.Record.READY_CLEAN, 'parent should still have old status');
123 
124 });

159 test("writeDataHash on parent records should be propagated to nested children and eventually to the parent store", function() {
160   var model = SC.Record.extend( {
161                 v: SC.Record.attr( String ),
162                 c: SC.Record.toMany( SC.Record.extend( {
163                                        v: SC.Record.attr( String ),
164                                      } ),
165                                      {
166                                        nested: YES
167                                      } )
168               } );
169   var json1 = { guid: "p1", v: "p1-1", c: [ { guid: "c1", v: "c1-1" }, { id: "c2", v: "c2-1" } ] };
170   var json12 = { guid: "p1", v: "p1-2", c: [ { guid: "c1", v: "c1-2" }, { id: "c2", v: "c2-1" } ] };
171 
172   SC.RunLoop.begin();
173 
174   // load the data into the main store
175   var storeKey1 = parent.loadRecord( model, json1 );
176   var rec1 = parent.find( model, "p1" );
177 
178   equals( rec1.get( "v" ), "p1-1", "the data loaded into the record should be correct" );
179   equals( rec1.get( "c" ).objectAt( 0 ).get( "v" ), "c1-1", "the data loaded into the 1st child record should be correct" );
180   equals( rec1.get( "c" ).objectAt( 1 ).get( "v" ), "c2-1", "the data loaded into the 2nd child record should be correct" );
181 
182   // start an "editing" session into the nested store store
183   var rec12 = store.find( rec1 );
184 
185   equals( rec12.get( "c" ).objectAt( 0 ).get( "v" ), "c1-1", "the data loaded into the 1st child record should be correct" );
186   equals( rec12.get( "c" ).objectAt( 1 ).get( "v" ), "c2-1", "the data loaded into the 2nd child record should be correct" );
187 
188   // replace the content and notify the changes
189   store.writeDataHash( storeKey1, json12, SC.Record.READY_CLEAN );
190   store.dataHashDidChange( storeKey1 );
191 
192   SC.RunLoop.end();
193 
194   SC.RunLoop.begin();
195   // check if the values from the hash are properly propagated to the record + children
196   equals( rec12.get( "v" ), "p1-2", "after writeDataHash the data loaded into the record should be updated" );
197   equals( rec12.get( "c" ).objectAt( 0 ).get( "v" ), "c1-2", "after writeDataHash the data loaded into the 1st child record should be updated" );
198   equals( rec12.get( "c" ).objectAt( 1 ).get( "v" ), "c2-1", "after writeDataHash the data loaded into the 2nd child record should not be updated" );
199 
200   // push the changes to the parent store
201   store.commitChanges(YES);
202   SC.RunLoop.end();
203 
204   SC.RunLoop.begin();
205   // check if the values updated into the nested store are properly propagated to the record + children
206   equals( rec1.get( "v" ), "p1-2", "after commitChanges the data loaded into the record should be updated" );
207   equals( rec1.get( "c" ).objectAt( 0 ).get( "v" ), "c1-2", "after commitChanges the data loaded into the 1st child record should be updated" );
208   equals( rec1.get( "c" ).objectAt( 1 ).get( "v" ), "c2-1", "the data loaded into the 2nd child record should be correct" );
209   SC.RunLoop.end();
210 });
