 61 /**
 62   Simulates clicking on the specified view.
 63 
 64   @param {SC.View} view the view
 65   @param {Boolean} [shiftKey] simulate shift key pressed
 66   @param {Boolean} [ctrlKey] simulate ctrlKey pressed
 67 */
 68 function clickOn(view, shiftKey, ctrlKey) {
 69   var layer    = view.get('layer'),
 70       opts     = { shiftKey: !!shiftKey, ctrlKey: !!ctrlKey },
 71       ev;
 72 
 73   ok(layer, 'precond - view %@ should have layer'.fmt(view.toString()));
 74 
 75   ev = SC.Event.simulateEvent(layer, 'mousedown', opts);
 76   SC.Event.trigger(layer, 'mousedown', [ev]);
 77 
 78   ev = SC.Event.simulateEvent(layer, 'mouseup', opts);
 79   SC.Event.trigger(layer, 'mouseup', [ev]);
 80   SC.RunLoop.begin();
 81   SC.RunLoop.end();
 82   layer = null ;
 83 }
 84 
 85 /**
 86   Simulates a key press on the specified view.
 87 
 88   @param {SC.View} view the view
 89   @param {Number} keyCode key to simulate
 90   @param {Boolean} [isKeyPress] simulate key press event
 91   @param {Boolean} [shiftKey] simulate shift key pressed
 92   @param {Boolean} [ctrlKey] simulate ctrlKey pressed
 93 */
 94 function keyPressOn(view, keyCode, isKeyPress, shiftKey, ctrlKey) {
 95   var layer    = view.get('layer'),
 96       opts     = {
 97         shiftKey: !!shiftKey,
 98         ctrlKey: !!ctrlKey,
 99         keyCode: keyCode,
100         charCode: isKeyPress ? keyCode : 0,
101         which: keyCode
102       },
103       ev;
104 
105   ok(layer, 'precond - view %@ should have layer'.fmt(view.toString()));
106 
107   ev = SC.Event.simulateEvent(layer, 'keydown', opts);
108   SC.Event.trigger(layer, 'keydown', [ev]);
109 
110   if (isKeyPress) {
111     ev = SC.Event.simulateEvent(layer, 'keypress', opts);
112     SC.Event.trigger(layer, 'keypress', [ev]);
113   }
114 
115   ev = SC.Event.simulateEvent(layer, 'keyup', opts);
116   SC.Event.trigger(layer, 'keyup', [ev]);
117   SC.RunLoop.begin();
118   SC.RunLoop.end();
119   layer = null ;
120 }
