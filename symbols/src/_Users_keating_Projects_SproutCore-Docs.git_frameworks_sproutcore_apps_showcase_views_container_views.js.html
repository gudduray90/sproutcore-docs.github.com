 10 /**
 11   This pre-configured view demonstrates SC.ContainerView.
 12 */
 13 Showcase.containerViews = SC.ScrollView.design({
 14   contentView: Showcase.ViewsListView.design({
 15     content: [
 16       Showcase.ViewsItemContent.create({
 17         title: 'Regular',
 18         example: "SC.ContainerView.extend({  layout: { left: 20, right: 20, top: 20, bottom: 20 },  nowShowing: 'blueSampleView',  blueSampleView: SC.LabelView.extend({  classNames:['blue-sample-view'],  value: 'Blue View'  }),  redSampleView: SC.LabelView.extend({  classNames:['red-sample-view'],  value: 'Red View'  })  });",
 19         supportAction: function() {
 20           var containerView = this.getPath('parentView.exampleBox.contentView');
 21           if (containerView.get('nowShowing') === 'blueSampleView') containerView.set('nowShowing', 'redSampleView');
 22           else containerView.set('nowShowing', 'blueSampleView');
 23         },
 24         supportTitle: 'Toggle nowShowing'
 25       })
 26     ],
 27     exampleHeight: 300
 28   })
 29 });
