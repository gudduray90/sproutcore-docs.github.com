  8 var view, exampleView;
  9 
 10 module("SC.CollectionFastPath", {
 11   setup: function() {
 12 
 13     view = SC.CollectionView.create({
 14       content: "a b c d e f".w().map(function(x) {
 15         return SC.Object.create({ title: x });
 16       }),
 17       useFastPath: YES
 18     });
 19 
 20     exampleView = SC.View.extend({
 21       isPoolable: YES,
 22       layerIsCacheable: YES
 23     });
 24 
 25   },
 26 
 27   teardown: function() {
 28     view = exampleView = null;
 29   }
 30 });
 31 
 32 
 33 /**
 34   There was a bug that if you called itemViewForContentIndex() on a Collection
 35   with SC.CollectionFastPath mixed in BEFORE it was visible, it would throw an
 36   exception (because this._mapView wasn't initialized properly in
 37   CollectionFastPath).
 38 */
 39 test("Calling itemViewForContentIndex() before the Collection is visible.", function() {
 40   try {
 41     var itemView = view.itemViewForContentIndex(0);
 42     ok(true, 'Requesting itemViewForContentIndex() should not throw an exception prior to reloadIfNeeded being called.');
 43   } catch (ex) {
 44     ok(false, 'Requesting itemViewForContentIndex() should not throw an exception prior to reloadIfNeeded being called.');
 45   }
 46 
 47   // The next test just shows how that when isVisibleInWindow changes, causing
 48   // reloadIfNeeded to be called, then the request would succeed.
 49   try {
 50     SC.RunLoop.begin();
 51     view.set('isVisibleInWindow', YES);
 52     SC.RunLoop.end();
 53     itemView = view.itemViewForContentIndex(0);
 54     ok(true, 'Requesting itemViewForContentIndex() should not throw an exception after reloadIfNeeded being called.');
 55   } catch (ex) {
 56     ok(false, 'Requesting itemViewForContentIndex() should not throw an exception after reloadIfNeeded being called.');
 57   }
 58 });
