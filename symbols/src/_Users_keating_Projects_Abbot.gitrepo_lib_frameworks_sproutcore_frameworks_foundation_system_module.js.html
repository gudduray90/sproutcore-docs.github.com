 12 /**
 13   SC.Module is responsible for dynamically loading in JavaScript and other
 14   resources. These packages of code and resources, called bundles, can be
 15   loaded by your application once it has finished loading, allowing you to
 16   reduce the time taken for it to launch.
 17 
 18   You can explicitly load a module by calling SC.Module.loadModule(), or you
 19   can mark a module as prefetched in your Buildfile. In those cases,
 20   SproutCore will automatically start to load the bundle once the application
 21   has loaded and the user has remained idle for more than one second.
 22 */
 23 
 24 SC.Module = SC.Object.create(/** @scope SC.Module */ {
 25 
 26   /**
 27     Returns YES if the module is ready; NO if it is not loaded or its
 28     dependencies have not yet loaded.
 29 
 30     @param {String} moduleName the name of the module to check
 31     @returns {Boolean}
 32   */
 33   isModuleReady: function(moduleName) {
 34     var moduleInfo = SC.MODULE_INFO[moduleName] ;
 35     return moduleInfo ? !!moduleInfo.isReady : NO ;
 36   },
 37 
 38   /**
 39     Asynchronously loads a module if it is not already loaded. If you pass
 40     a function, or a target and action, it will be called once the module
 41     has finished loading.
 42 
 43     If the module you request has dependencies (as specified in the Buildfile)
 44     that are not yet loaded, it will load them first before executing the
 45     requested module.
 46 
 47     @param moduleName {String}
 48     @param target {Function}
 49     @param method {Function}
 50     @returns {Boolean} YES if already loaded, NO otherwise
 51   */
 52   loadModule: function(moduleName, target, method) {
 53     var module = SC.MODULE_INFO[moduleName], callbacks, targets;
 54     var args   = SC.A(arguments).slice(3);
 55     var log    = SC.LOG_MODULE_LOADING;
 56     var idx, len;
 57 
 58     // Treat the first parameter as the callback if the target is a function and there is
 59     // no method supplied.
 60     if (method === undefined && SC.typeOf(target) === SC.T_FUNCTION) {
 61       method = target;
 62       target = null;
 63     }
 64 
 65     if (log) {
 66       SC.Logger.log("SC.Module: Attempting to load '%@'".fmt(moduleName));
 67     }
 68 
 69     // If we couldn't find anything in the SC.MODULE_INFO hash, we don't have any record of the
 70     // requested module.
 71     if (!module) {
 72       throw "SC.Module: could not find module '%@'".fmt(moduleName) ;
 73     }
 74 
 75     // If this module was in the middle of being prefetched, we now need to
 76     // execute it immediately when it loads.
 77     module.isPrefetching = NO;
 78 
 79     // If the module is already loaded, execute the callback immediately if SproutCore is loaded,
 80     // or else as soon as SC has finished loading.
 81     if (module.isLoaded) {
 82       if (log) SC.Logger.log("SC.Module: Module '%@' already loaded.".fmt(moduleName));
 83 
 84       // If the module has finished loading and we have the string
 85       // representation, try to evaluate it now.
 86       if (module.source) {
 87         if (log) SC.Logger.log("SC.Module: Evaluating JavaScript for module '%@'.".fmt(moduleName));
 88         this._evaluateStringLoadedModule(module);
 89       }
 90 
 91       if (method) {
 92         if (SC.isReady) {
 93           SC.Module._invokeCallback(moduleName, target, method, args);
 94         } else {
 95           // Queue callback for when SC has finished loading.
 96           SC.ready(SC.Module, function() {
 97             SC.Module._invokeCallback(moduleName, target, method, args);
 98           });
 99         }
100       }
101 
102       return YES;
103     }
104     // The module is not yet loaded, so register the callback and, if necessary, begin loading
105     // the code.
106     else {
107       if (log) SC.Logger.log("SC.Module: Module '%@' is not loaded, loading now.".fmt(moduleName));
108 
109       // If this method is called more than once for the same module before it is finished
110       // loading, we might have multiple callbacks that need to be executed once it loads.
111 
112       // Retrieve array of callbacks from MODULE_INFO hash.
113       callbacks = module.callbacks || [] ;
114 
115       if (method) {
116         callbacks.push(function() {
117           SC.Module._invokeCallback(moduleName, target, method, args);
118         });
119       }
120 
121       module.callbacks = callbacks;
122 
123       // If this is the first time the module has been requested, determine its dependencies
124       // and begin loading them as well as the JavaScript for this module.
125       if (!module.isLoading) {
126         this._loadDependenciesForModule(moduleName);
127         this._loadCSSForModule(moduleName);
128         this._loadJavaScriptForModule(moduleName);
129         module.isLoading = YES;
130       }
131 
132       return NO;
133     }
134   },
135 
136   /**
137     @private
138 
139     Loads a module in string form. If you prefetch a module, its source will
140     be held as a string in memory until SC.Module.loadModule() is called, at
141     which time its JavaScript will be evaluated.
142 
143     You shouldn't call this method directly; instead, mark modules as
144     prefetched in your Buildfile. SproutCore will automatically prefetch those
145     modules once your application has loaded and the user is idle.
146 
147     @param {String} moduleName the name of the module to prefetch
148   */
149   prefetchModule: function(moduleName) {
150     var module = SC.MODULE_INFO[moduleName];
151 
152     if (module.isLoading || module.isLoaded) return;
153 
154     if (SC.LOG_MODULE_LOADING) SC.Logger.log("SC.Module: Prefetching module '%@'.".fmt(moduleName));
155     this._loadDependenciesForModule(moduleName);
156     this._loadCSSForModule(moduleName);
157     this._loadJavaScriptForModule(moduleName);
158     module.isLoading = YES;
159     module.isPrefetching = YES;
160   },

166   /** @private
167     If a module is marked for lazy instantiation, this method will execute the closure and call
168     any registered callbacks.
169   */
170   _executeLazilyInstantiatedModule: function(moduleName, targetName, methodName){
171     var lazyInfo =  SC.LAZY_INSTANTIATION[moduleName];
172     var target;
173     var method;
174     var idx, len;
175 
176     if (SC.LOG_MODULE_LOADING) {
177       SC.Logger.log("SC.Module: Module '%@' is marked for lazy instantiation, instantiating it now…".fmt(moduleName));
178     }
179 
180     len = lazyInfo.length;
181     for (idx = 0; idx < len; idx++) {
182       // Iterate through each function associated with this module, and attempt to execute it.
183       try {
184         lazyInfo[idx]();
185       } catch(e) {
186         SC.Logger.error("SC.Module: Failed to lazily instatiate entry for  '%@'".fmt(moduleName));
187       }
188     }
189 
190     // Free up memory containing the functions once they have been executed.
191     delete SC.LAZY_INSTANTIATION[moduleName];
192 
193     // Now that we have executed the functions, try to find the target and action for the callback.
194     target = this._targetForTargetName(targetName);
195     method = this._methodForMethodNameInTarget(methodName, target);
196 
197     if (!method) {
198       throw "SC.Module: could not find callback for lazily instantiated module '%@'".fmt(moduleName);
199     }
200   },

202   /**
203     Evaluates a module's JavaScript if it is stored in string format, then
204     deletes that code from memory.
205 
206     @param {Hash} module the module to evaluate
207   */
208   _evaluateStringLoadedModule: function(module) {
209     var moduleSource = module.source;
210 
211     jQuery.globalEval(moduleSource);
212     delete module.source;
213 
214     if (module.cssSource) {
215       var el = document.createElement('style');
216       el.setAttribute('type', 'text/css');
217       if (el.styleSheet) {
218         el.styleSheet.cssText = module.cssSource;
219       } else {
220         var content = document.createTextNode(module.cssSource);
221         el.appendChild(content);
222       }
223 
224       document.getElementsByTagName('head')[0].appendChild(el);
225     }
226 
227     module.isReady = YES;
228   },
229 
230   /**
231     @private
232 
233     Creates <link> tags for every CSS resource in a module.
234 
235     @param {String} moduleName the name of the module whose CSS should be loaded
236   */
237   _loadCSSForModule: function(moduleName) {
238     var head = document.getElementsByTagName('head')[0] ;
239     var module = SC.MODULE_INFO[moduleName];
240     var styles = module.styles || [];
241     var len = styles.length;
242     var url;
243     var el;
244     var idx;
245 
246     if (!head) head = document.documentElement ; // fix for Opera
247     len = styles.length;
248 
249     for (idx = 0; idx < len; idx++) {
250       url = styles[idx] ;
251 
252       if (url.length > 0) {
253         if (SC.LOG_MODULE_LOADING) SC.Logger.log("SC.Module: Loading CSS file in '%@' -> '%@'".fmt(moduleName, url));
254         el = document.createElement('link') ;
255         el.setAttribute('href', url) ;
256         el.setAttribute('rel', "stylesheet") ;
257         el.setAttribute('type', "text/css") ;
258         head.appendChild(el) ;
259       }
260     }
261 
262     el = null;
263   },
264 
265   _loadJavaScriptForModule: function(moduleName) {
266     var module = SC.MODULE_INFO[moduleName];
267     var el;
268     var url;
269     var dependencies = module.dependencies;
270     var dependenciesAreLoaded = YES;
271 
272     // If this module has dependencies, determine if they are loaded.
273     if (dependencies && dependencies.length > 0) {
274       dependenciesAreLoaded = this._dependenciesMetForModule(moduleName);
275     }
276 
277     // If the module is prefetched, always load the string representation.
278     if (module.isPrefetched) {
279       url = module.stringURL;
280     } else {
281       if (dependenciesAreLoaded) {
282         // Either we have no dependencies or they've all loaded already,
283         // so just execute the code immediately once it loads.
284         url = module.scriptURL;
285       } else {
286         // Because the dependencies might load after this module, load the
287         // string representation so we can execute it once all dependencies
288         // are in place.
289         url = module.stringURL;
290       }
291     }
292 
293     if (url.length > 0) {
294       if (SC.LOG_MODULE_LOADING) SC.Logger.log("SC.Module: Loading JavaScript file in '%@' -> '%@'".fmt(moduleName, url));
295 
296       el = document.createElement('script') ;
297       el.setAttribute('type', "text/javascript") ;
298       el.setAttribute('src', url) ;
299 
300       if (SC.browser.isIE) {
301         el.onreadystatechange = function() {
302           if (this.readyState == 'complete' || this.readyState == 'loaded') {
303             SC.Module._moduleDidLoad(moduleName);
304           }
305         };
306       } else {
307         el.onload = function() {
308           SC.Module._moduleDidLoad(moduleName);
309         };
310       }
311 
312       document.body.appendChild(el) ;
313     }
314   },

316   /**
317     @private
318 
319     Returns YES if all of the dependencies for a module are ready.
320 
321     @param {String} moduleName the name of the module being checked
322     @returns {Boolean} whether the dependencies are loaded
323   */
324   _dependenciesMetForModule: function(moduleName) {
325     var dependencies = SC.MODULE_INFO[moduleName].dependencies || [];
326     var idx, len = dependencies.length;
327     var dependencyName;
328     var module;
329 
330     for (idx = 0; idx < len; idx++) {
331       dependencyName = dependencies[idx];
332       module = SC.MODULE_INFO[dependencyName];
333 
334       if (!module) throw "SC.loadModule: Unable to find dependency %@ for module %@.".fmt(dependencyName, moduleName);
335 
336       if (!module.isReady) {
337         return NO;
338       }
339     }
340 
341     return YES;
342   },
343 
344   /**
345     Loads all unloaded dependencies for a module, then creates the <script> and <link> tags to
346     load the JavaScript and CSS for the module.
347   */
348   _loadDependenciesForModule: function(moduleName) {
349       // Load module's dependencies first.
350       var moduleInfo      = SC.MODULE_INFO[moduleName];
351       var log             = SC.LOG_MODULE_LOADING;
352       var dependencies    = moduleInfo.dependencies || [];
353       var dependenciesMet = YES;
354       var len             = dependencies.length;
355       var idx;
356       var requiredModuleName;
357       var requiredModule;
358       var dependents;
359 
360       for (idx = 0; idx < len; idx++) {
361         requiredModuleName = dependencies[idx];
362         requiredModule = SC.MODULE_INFO[requiredModuleName];
363 
364         // Try to find dependent module in MODULE_INFO
365         if (!requiredModule) {
366           throw "SC.Module: could not find required module '%@' for module '%@'".fmt(requiredModuleName, moduleName) ;
367         } else {
368 
369           // Required module has been requested but hasn't loaded yet.
370           if (requiredModule.isLoading) {
371             dependenciesMet = NO ;
372 
373             dependents = requiredModule.dependents;
374             if (!dependents) requiredModule.dependents = dependents = [];
375             dependents.push(moduleName);
376           }
377 
378           // Required module has already been loaded and evaluated, no need to worry about it.
379           else if (requiredModule.isReady) {
380             continue ;
381           }
382           // Required module has not been loaded nor requested yet.
383           else {
384             dependenciesMet = NO ;
385 
386             // Register this as a dependent module (used by SC._moduleDidLoad()...)
387             dependents = requiredModule.dependents;
388             if (!dependents) requiredModule.dependents = dependents = [];
389 
390             dependents.push(moduleName) ;
391 
392             if (log) SC.Logger.log("SC.Module: '%@' depends on '%@', loading dependency…".fmt(moduleName, requiredModuleName));
393 
394             // Load dependencies
395             SC.Module.loadModule(requiredModuleName) ;
396           }
397         }
398       }
399     },
400 
401     /**
402       @private
403 
404       Calls an action on a target to notify the target that a module has loaded.
405     */
406     _invokeCallback: function(moduleName, targetName, methodName, args) {
407       var method;
408       var target;
409 
410       target = this._targetForTargetName(targetName);
411       method = this._methodForMethodNameInTarget(methodName, target);
412 
413       // If we weren't able to find the callback, this module may be lazily instantiated and
414       // the callback won't exist until we execute the closure that it is wrapped in.
415       if (!method) {
416         if (SC.LAZY_INSTANTIATION[moduleName]) {
417           this._executeLazilyInstantiatedModule(moduleName, targetName, methodName);
418 
419           target = this._targetForTargetName(targetName);
420           method = this._methodForMethodNameInTarget(methodName, target);
421         } else {
422           throw "SC.Module: could not find callback for '%@'".fmt(moduleName);
423         }
424       }
425 
426       if (!args) {
427         args = [];
428       }
429 
430       // The first parameter passed to the callback is the name of the module.
431       args.unshift(moduleName);
432 
433       // Invoke the callback. Wrap it in a run loop if we are not in a runloop already.
434       var needsRunLoop = !!SC.RunLoop.currentRunLoop;
435       if (needsRunLoop) {
436         SC.run(function() {
437           method.apply(target, args);
438         });
439       } else {
440         method.apply(target, args);
441       }
442     },
443 
444     /** @private
445       Given a module name, iterates through all registered callbacks and calls them.
446     */
447     _invokeCallbacksForModule: function(moduleName) {
448       var moduleInfo = SC.MODULE_INFO[moduleName], callbacks ;
449       if (!moduleInfo) return ; // shouldn't happen, but recover anyway
450 
451       if (SC.LOG_MODULE_LOADING) SC.Logger.log("SC.Module: Module '%@' has completed loading, invoking callbacks.".fmt(moduleName));
452 
453       callbacks = moduleInfo.callbacks || [] ;
454 
455 456       for (var idx=0, len=callbacks.length; idx<len; ++idx) {
457         callbacks[idx]() ;
458       }
459     },
460 
461     _evaluateAndInvokeCallbacks: function(moduleName) {
462       var moduleInfo = SC.MODULE_INFO;
463       var module = moduleInfo[moduleName];
464       var log = SC.LOG_MODULE_LOADING;
465 
466       if (log) SC.Logger.log("SC.Module: Evaluating and invoking callbacks for '%@'.".fmt(moduleName));
467 
468       if (module.source) {
469         this._evaluateStringLoadedModule(module);
470       }
471       module.isReady = YES;
472 
473       if (SC.isReady) {
474         SC.Module._invokeCallbacksForModule(moduleName) ;
475         delete module.callbacks;
476       } else {
477         SC.ready(SC, function() {
478           SC.Module._invokeCallbacksForModule(moduleName) ;
479           delete module.callbacks;
480         });
481       }
482 
483       // for each dependent module, try and load them again...
484       var dependents = module.dependents || [] ;
485       var dependentName, dependent;
486 
487       for (var idx = 0, len = dependents.length; idx < len; idx++) {
488         dependentName = dependents[idx];
489         dependent = moduleInfo[dependentName];
490         if (dependent.isLoaded && this._dependenciesMetForModule(dependentName)) {
491           if (log) SC.Logger.log("SC.Module: Now that %@ has loaded, all dependencies for a dependent %@ are met.".fmt(moduleName, dependentName));
492           this._evaluateAndInvokeCallbacks(dependentName);
493         }
494       }
495     },
496 
497   /** @private
498     Called when the JavaScript for a module finishes loading.
499 
500     Any pending callbacks are called (if SC.isReady), and any dependent
501     modules which were waiting for this module to load are notified so they
502     can continue loading.
503 
504     @param moduleName {String} the name of the module that just loaded
505 */
506   _moduleDidLoad: function(moduleName) {
507     var module = SC.MODULE_INFO[moduleName];
508     var log    = SC.LOG_MODULE_LOADING;
509     var dependenciesMet;
510     var callbacks, targets;
511 
512     if (log) SC.Logger.log("SC.Module: Module '%@' finished loading.".fmt(moduleName));
513 
514     if (!module) {
515       if (log) SC.Logger.log("SC._moduleDidLoad() called for unknown module '@'.".fmt(moduleName));
516       module = SC.MODULE_INFO[moduleName] = { isLoaded: YES, isReady: YES } ;
517       return;
518     }
519 
520     if (module.isLoaded) {
521       if (log) SC.Logger.log("SC._moduleDidLoad() called more than once for module '%@'. Skipping.".fmt(moduleName));
522       return ;
523     }
524 
525     // Remember that we're loaded.
526     delete module.isLoading ;
527     module.isLoaded = YES ;
528 
529     if (!module.isPrefetching) {
530       dependenciesMet = this._dependenciesMetForModule(moduleName);
531       if (dependenciesMet) {
532         this._evaluateAndInvokeCallbacks(moduleName);
533       } else {
534         if (log) SC.Logger.log("SC.Module: Dependencies for '%@' not met yet, waiting to evaluate.".fmt(moduleName));
535       }
536     } else {
537       delete module.isPrefetching;
538       if (log) SC.Logger.log("SC.Module: Module '%@' was prefetched, not evaluating until needed.".fmt(moduleName));
539     }
540   },

542   /**
543     @private
544 
545     If necessary, converts a property path into a target object.
546 
547     @param {String|Object} targetName the string or object representing the target
548     @returns Object
549   */
550   _targetForTargetName: function(targetName){
551     if (SC.typeOf(targetName) === SC.T_STRING) {
552       return SC.objectForPropertyPath(targetName);
553     }
554 
555     return targetName;
556   },
557 
558   /**
559     @private
560 
561     If necessary, converts a property path into a method object.
562 
563     @param {String|Object} methodName the string or object representing the method
564     @param {Object} target the target from which to retrieve the method
565     @returns Object
566   */
567   _methodForMethodNameInTarget: function(methodName, target){
568     if (SC.typeOf(methodName) === SC.T_STRING) {
569       return SC.objectForPropertyPath(methodName, target);
570     }
571 
572     return methodName;
573   }

576 /**
577 Inspect the list of modules and, for every prefetched module, create a
578 background task to load the module when the user remains idle.
579 */
580 SC.ready(function() {
581   var moduleInfo = SC.MODULE_INFO;
582   var moduleName;
583   var module;
584   var task;
585 
586   // Iterate through all known modules and look for those that are marked
587   // as prefetched.
588   for (moduleName in moduleInfo) {
589     module = moduleInfo[moduleName];
590 
591     if (module.isPrefetched) {
592       var prefetchedModuleName = moduleName;
593 
594       // Create a task that will load the module, and then register it with
595       // the global background task queue.
596       task = SC.Task.create({
597         run: function() {
598           SC.Module.prefetchModule(prefetchedModuleName);
599         }
600       });
601 
602       SC.backgroundTaskQueue.push(task);
603     }
604   }
