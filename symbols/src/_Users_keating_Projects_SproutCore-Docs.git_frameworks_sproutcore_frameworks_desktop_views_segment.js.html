  7 /**
  8   @class
  9 
 10   SegmentViews are the views used and arranged by SC.SegmentedView and are very similar to a SC.ButtonView
 11   without any event handling.  The event handling is done by the parent view.
 12 
 13   @extends SC.View
 14   @since SproutCore 1.5
 15 */
 16 SC.SegmentView = SC.View.extend(SC.Control,
 17 /** @scope SC.SegmentView.prototype */{
 18 
 19   /**
 20     @type String
 21     @default 'tab'
 22     @readOnly
 23   */
 24   //ariaRole: 'tab',
 25   ariaRole: 'button', // workaround for <rdar://problem/10444670>; switch back to 'tab' later with <rdar://problem/10463928> (also see segmented.js)
 26 
 27   /**
 28     @type Boolean
 29     @default YES
 30     @readOnly
 31   */
 32   isSegment: YES,
 33 
 34   /**
 35     @type Array
 36     @default ['sc-segment-view']
 37     @see SC.View#classNames
 38   */
 39   classNames: ['sc-segment-view'],
 40 
 41   /**
 42     @type String
 43     @default null
 44     @see SC.View#toolTip
 45   */
 46   toolTip: null,
 47 
 48   /**
 49     @type Boolean
 50     @default YES
 51     @see SC.Control#isEnabled
 52   */
 53   isEnabled: YES,
 54 
 55   /**
 56     @type Boolean
 57     @default NO
 58     @see SC.Control#isActive
 59   */
 60   isActive: NO,
 61 
 62   /**
 63     @type Boolean
 64     @default NO
 65     @see SC.Control#isSelected
 66   */
 67   isSelected: NO,
 68 
 69   /**
 70     Change the layout direction to make this a vertical segment instead of horizontal ones. 
 71     Possible values:
 72     
 73       - SC.LAYOUT_HORIZONTAL
 74       - SC.LAYOUT_VERTICAL
 75     
 76     @type String
 77     @default SC.LAYOUT_HORIZONTAL
 78   */
 79   layoutDirection: SC.LAYOUT_HORIZONTAL,   
 80 
 81   /**
 82     @type String
 83     @default null
 84     @see SC.Control#controlSize
 85   */
 86   controlSize: null,
 87 
 88   /**
 89     @type Boolean
 90     @default NO
 91     @see SC.ButtonView#supportFocusRing
 92   */
 93   supportFocusRing: NO,
 94 
 95   // TODO: isDefault, isCancel, value not really used by render delegate
 96   displayProperties: ['icon', 'displayTitle', 'value', 'displayToolTip', 'isDefault', 'isCancel', 'width', 'isSegment','isFirstSegment', 'isMiddleSegment', 'isLastSegment', 'isOverflowSegment', 'index', 'layoutDirection'],
 97 
 98   /**
 99     @type String
100     @default 'segmentRenderDelegate'
101   */
102   renderDelegateName: 'segmentRenderDelegate',
103 
104   /**
105     @type Boolean
106     @default YES
107   */
108   useStaticLayout: YES,
109 
110 
111   // ..........................................................
112   // Properties
113   // 
114 
115   /**
116     @type String
117     @default ""
118   */
119   title: "",
120 
121   /**
122     @type Object
123     @default null
124   */
125   value: null,
126 
127   /**
128     @type String
129     @default null
130   */
131   icon: null,
132 
133   /**
134     @type Boolean
135     @default null
136   */
137   localize: NO,
138 
139   /**
140     @type String
141     @default null
142   */
143   keyEquivalent: null,
144 
145   // TODO: Modification currently unsupported in SegmentedView
146   /**
147     @type Boolean
148     @default YES
149   */
150   escapeHTML: YES,
151 
152   // TODO: Modification currently unsupported in SegmentedView
153   /**
154     @type Boolean
155     @default YES
156   */
157   needsEllipsis: YES,
158 
159   /**
160     Localized title.
161     
162     @field
163     @type String
164     @default ""
165   */
166   displayTitle: function() {
167     var ret = this.get('title');
168     if (this.get('localize')) ret = SC.String.loc(ret);
169     return ret;
170   }.property('title', 'localize').cacheable(),
171 
172   /**
173     @type Number
174     @default null
175   */
176   width: null,
177 
178   /**
179     The item represented by this view.
180 
181     @type Object
182     @default null
183   */
184   localItem: null,
185 
186   /** @private
187     Whenever the width property changes, adjust our layout accordingly.
188   */
189   widthDidChange: function() {
190     this.adjust('width', this.get('width'));
191   }.observes('width'),
192 
193   /** @private
194     Update our properties according to our matching item.
195   */
196   updateItem: function(parentView, item) {
197     var itemKeys = parentView.get('itemKeys'),
198         itemKey,
199         viewKeys = parentView.get('viewKeys'),
200         viewKey,
201         i;
202 
203     for (i = itemKeys.get('length') - 1; i >= 0; i--) {
204       itemKey = parentView.get(itemKeys.objectAt(i));
205       viewKey = viewKeys.objectAt(i);
206 
207       // Don't overwrite the default value if none exists in the item
208       if (!SC.none(item.get(itemKey))) this.set(viewKey, item.get(itemKey));
209     }
210 
211     this.set('localItem', item);
212   }
213 });
