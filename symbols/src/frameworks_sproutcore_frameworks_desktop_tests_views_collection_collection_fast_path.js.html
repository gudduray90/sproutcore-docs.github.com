  8 var view, exampleView;
  9 
 10 module("SC.CollectionFastPath", {
 11   setup: function() {
 12 
 13     SC.run(function () {
 14       view = SC.CollectionView.create({
 15         content: "a b c d e f".w().map(function(x) {
 16           return SC.Object.create({ title: x });
 17         }),
 18         // STUB: reloadIfNeeded
 19         reloadIfNeeded: CoreTest.stub('reloadIfNeeded', SC.CollectionView.prototype.reloadIfNeeded)
 20       });
 21     });
 22 
 23     exampleView = SC.View.extend({});
 24 
 25   },
 26 
 27   teardown: function() {
 28     view.destroy();
 29     view = exampleView = null;
 30   }
 31 });
 32 
 33 
 34 /**
 35   There was a bug that if you called itemViewForContentIndex() on a Collection
 36   with SC.CollectionFastPath mixed in BEFORE it was visible, it would throw an
 37   exception (because this._mapView wasn't initialized properly in
 38   CollectionFastPath).
 39 */
 40 test("Calling itemViewForContentIndex() before the Collection is visible.", function() {
 41   try {
 42     var itemView = view.itemViewForContentIndex(0);
 43     ok(true, 'Requesting itemViewForContentIndex() should not throw an exception prior to reloadIfNeeded being called.');
 44 
 45     view.reloadIfNeeded.expect(0);
 46   } catch (ex) {
 47     ok(false, 'Requesting itemViewForContentIndex() should not throw an exception prior to reloadIfNeeded being called.');
 48   }
 49 
 50   // The next test just shows how that when isVisibleInWindow changes, causing
 51   // reloadIfNeeded to be called, then the request would succeed.
 52   try {
 53     SC.run(function () {
 54       view.createLayer();
 55       view._doAttach(document.body);
 56     });
 57 
 58     view.reloadIfNeeded.expect(1);
 59     itemView = view.itemViewForContentIndex(0);
 60     ok(true, 'Requesting itemViewForContentIndex() should not throw an exception after reloadIfNeeded being called.');
 61   } catch (ex) {
 62     ok(false, 'Requesting itemViewForContentIndex() should not throw an exception after reloadIfNeeded being called.');
 63   }
 64 });
