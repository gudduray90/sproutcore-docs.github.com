 10 /** @class
 11   The views that make up the column header cells in a typical `SC.TableView`.
 12   
 13   In addition, this view is in charge of rendering the _entire_ table column
 14   (both header and body) when the table is in the "drag-reorder" state. This
 15   is the state that occurs when the user clicks and holds on a table header,
 16   then drags that header horizontally.
 17 
 18   @extends SC.View
 19   @since SproutCore 1.1
 20 */
 21 SC.TableHeaderView = SC.View.extend({
 22   
 23   classNames: ['sc-table-header'],
 24   
 25   displayProperties: ['sortState', 'isInDragMode'],
 26   
 27   acceptsFirstResponder: YES,
 28   
 29   isInDragMode: NO,
 30   
 31   hasHorizontalScroller: NO,
 32   hasVerticalScroller: NO,
 33   
 34   childViews: ['dragModeView'],
 35   
 36   
 37   /**
 38     The view that is visible when the column is in drag mode.
 39   */
 40   dragModeView: SC.ListView.extend({
 41     isVisible: NO,
 42     
 43     layout: { left: 0, right: 0, bottom: 0 },
 44     
 45     init: function() {
 46       sc_super();
 47 
 48       var tableHeaderView = this.get('parentView');
 49       
 50       if (tableHeaderView) {
 51         tableHeaderView.addObserver('isInDragMode', this,
 52             '_scthv_dragModeDidChange');
 53       }
 54       
 55     },
 56     
 57     _scthv_dragModeDidChange: function() {
 58       // var isInDragMode = this.get('tableHeaderView').get('isInDragMode');
 59       // this.set('isVisible', isInDragMode);
 60     }
 61   }),
 62 
 63   /**
 64     The SC.TableColumn object this header cell is bound to.
 65   */
 66   column:  null,
 67   
 68   render: function(context, firstTime) {
 69     var column = this.get('column'), icon = column.get('icon'), html;
 70     var span = context.begin('span');
 71     if (icon) {
 72       html = '<img src="%@" class="icon" />'.fmt(icon);
 73       span.push(html);
 74     } else {
 75       span.push(this.get('label'));
 76     }
 77     span.end();
 78   },
 79     
 80   // ========================================================
 81   // = For the column we look after, set up some observers. =
 82   // ========================================================
 83   init: function() {
 84     sc_super();
 85 
 86     var column = this.get('column');
 87     column.addObserver('width',     this, '_scthv_layoutDidChange');
 88     column.addObserver('maxWidth',  this, '_scthv_layoutDidChange');
 89     column.addObserver('minWidth',  this, '_scthv_layoutDidChange');
 90     column.addObserver('sortState', this, '_scthv_sortStateDidChange');
 91     column.addObserver('tableContent', this, '_scthv_tableContentDidChange');
 92     
 93     // var tableContent = column.get('tableContent');
 94     // var columnContent = this._scthv_columnContentFromTableContent(tableContent);
 95     // this.set('content', columnContent);
 96   },
 97   
 98   /**
 99     The sortState of the header view's column.
100   */
101   sortState: function() {
102     return this.get('column').get('sortState');
103   }.property(),
104   
105   mouseDown: function(evt) {
106     var tableView = this.get('tableView');    
107     return tableView ? tableView.mouseDownInTableHeaderView(evt, this) :
108      sc_super();
109   },
110   
111   mouseUp: function(evt) {
112     var tableView = this.get('tableView');
113 114     return tableView ? tableView.mouseUpInTableHeaderView(evt, this) :
115      sc_super();
116   },
117   
118   mouseDragged: function(evt) {
119     var tableView = this.get('tableView');
120     return tableView ? tableView.mouseDraggedInTableHeaderView(evt, this) :
121      sc_super();
122   },
123   
124   _scthv_dragViewForHeader: function() {
125     var dragLayer = this.get('layer').cloneNode(true);
126     var view = SC.View.create({ layer: dragLayer, parentView: this });
127     
128     // cleanup weird stuff that might make the drag look out of place
129     SC.$(dragLayer).css('backgroundColor', 'transparent')
130       .css('border', 'none')
131       .css('top', 0).css('left', 0);      
132     
133     return view;
134   },
135   
136   _scthv_enterDragMode: function() {
137     this.set('isInDragMode', YES);
138   },
139   
140   _scthv_exitDragMode: function() {
141     this.set('isInDragMode', NO);
142   },
143   
144   // _scthv_hideViewInDragMode: function() {    
145   //   var shouldBeVisible = !this.get('isInDragMode'), layer = this.get('layer');
146   //   console.log('should be visible: %@'.fmt(!this.get('isInDragMode')));
147   //   SC.RunLoop.begin();
148   //   SC.$(layer).css('display', shouldBeVisible ? 'block' : 'none');
149   //   SC.RunLoop.end();
150   // }.observes('isInDragMode'),
151   
152   // _scthv_setupDragMode: function() {
153   //   var isInDragMode = this.get('isInDragMode');
154   //   if (isInDragMode) {
155   //     });      
156   //   } else {
157   //     //
158   //   }
159   //   
160   //   
161   // }.observes('isInDragMode'),
162   
163   _scthv_dragModeViewDidChange: function() {
164     var dragModeView = this.get('dragModeView');
165     if (dragModeView && dragModeView.set) {
166       dragModeView.set('tableHeadView', this);
167       dragModeView.set('tableView', this.get('tableView'));
168     }
169   }.observes('dragModeView'),
170   
171   _scthv_layoutDidChange: function(sender, key, value, rev) {
172     var pv = this.get('parentView');
173     pv.invokeOnce(pv.layoutChildViews);
174     
175     // Tell the container view how tall the header is so that it can adjust
176     // itself accordingly.
177     var layout = this.get('layout');    
178     //this.get('dragModeView').adjust('top', layout.height);
179   },
180   
181   // When our column's tableContent property changes, we need to go back and get our column content
182   _scthv_tableContentDidChange: function() {
183     var tableContent = this.get('column').get('tableContent');    
184     var columnContent = this.get('parentView')._scthv_columnContentFromTableContent(tableContent, this.get('columnIndex'));
185     this.set('content', columnContent);
186   },
187   
188   _scthv_sortStateDidChange: function() {
189     SC.RunLoop.begin();
190     var sortState  = this.get('column').get('sortState');
191     var classNames = this.get('classNames');
192     
193     classNames.removeObject('sc-table-header-sort-asc');
194     classNames.removeObject('sc-table-header-sort-desc');
195     classNames.removeObject('sc-table-header-sort-active');
196     
197     if (sortState !== null) {
198       classNames.push('sc-table-header-sort-active');
199     }
200     
201     if (sortState === SC.SORT_ASCENDING) {
202       classNames.push('sc-table-header-sort-asc');
203     }
204     
205     if (sortState === SC.SORT_DESCENDING) {
206       classNames.push('sc-table-header-sort-desc');
207     }
208     
209     // TODO: Figure out why it's not enough to simply call
210     // `displayDidChange` here.
211     this.displayDidChange();
212     this.invokeOnce('updateLayer');
213     SC.RunLoop.end();
214   }
215 });
