  8 /*global module test equals context */
  9 
 10 var context = null;
 11 
 12 module("SC.RenderContext#join", {
 13   setup: function() {
 14     context = SC.RenderContext().push("line1", "line2") ;
 15   },
 16   
 17   teardown: function() {
 18     context = null;
 19   }
 20 });
 21 
 22 test("it should return joined lines with no separator string by default", function() {
 23   equals(context.join(), '<div>line1line2</div>');
 24 });
 25 
 26 test("it should return joined lines with separator string if passed", function() {
 27   equals(context.join(","), "<div>,line1,line2,</div>") ;
 28 });
