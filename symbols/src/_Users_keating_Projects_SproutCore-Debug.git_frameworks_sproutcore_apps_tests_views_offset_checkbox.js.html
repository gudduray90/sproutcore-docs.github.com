  7 /** @class
  8 
  9   This special view class will automatically adjusts its left offset based 
 10   on an "offset" value, which is will be bound to the width of the split view.
 11   
 12   This way when you resize the split view, the checkbox view will move also.
 13 
 14   @extends SC.CheckboxView
 15 */
 16 TestRunner.OffsetCheckboxView = SC.CheckboxView.extend(
 17 /** @scope TestRunner.OffsetCheckboxView.prototype */ {
 18 
 19   /** bind to thickness of splitview (though a controller) */
 20   offset: 0,
 21   
 22   offsetDidChange: function() {
 23     this.adjust('left', this.get('offset')+6);
 24   }.observes('offset')
 25 
 26 });
