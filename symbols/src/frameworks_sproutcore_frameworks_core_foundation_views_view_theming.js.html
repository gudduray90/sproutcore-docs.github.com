 44   /**
 45     The SC.Theme instance which this view should use to render.
 46 
 47     Note: the actual code for this function is in _themeProperty for backwards-compatibility:
 48     some older views specify a string value for 'theme', which would override this property,
 49     breaking it.
 50 
 51     @property {SC.Theme}
 52   */
 53   theme: function() {
 54     var base = this.get('baseTheme'), themeName = this.get('themeName');
 55 
 56     // find theme, if possible
 57     if (themeName) {
 58       // Note: theme instance "find" function will search every parent
 59       // _except_ global (which is not a parent)
 60       var theme;
 61       if (base) {
 62         theme = base.find(themeName);
 63         if (theme) { return theme; }
 64       }
 65 
 66       theme = SC.Theme.find(themeName);
 67       if (theme) { return theme; }
 68 
 69       // Create a new invisible subtheme. This will cause the themeName to
 70       // be applied as a class name.
 71       return base.invisibleSubtheme(themeName);
 72     }
 73 
 74     // can't find anything, return base.
 75     return base;
 76   }.property('baseTheme', 'themeName').cacheable(),

119   /**
120     The object to which rendering and updating the HTML representation of this
121     view should be delegated.
122 
123     By default, views are responsible for creating their own HTML
124     representation. In some cases, however, you may want to create an object
125     that is responsible for rendering all views of a certain type. For example,
126     you may want rendering of SC.ButtonView to be controlled by an object that
127     is specific to the current theme.
128 
129     By setting a render delegate, the render and update methods will be called
130     on that object instead of the view itself.
131 
132     For your convenience, the view will provide its displayProperties to the
133     RenderDelegate. In some cases, you may have a conflict between the RenderDelegate's
134     API and your view's. For instance, you may have a 'value' property that is
135     any number, but the render delegate expects a percentage. Make a 'displayValue'
136     property, add _it_ to displayProperties instead of 'value', and the Render Delegate
137     will automatically use that when it wants to find 'value.'
138 
139     You can also set the render delegate by using the 'renderDelegateName' property.
140 
141     @type Object
142   */
143   renderDelegate: function(key, value) {
144     if (value) { this._setRenderDelegate = value; }
145     if (this._setRenderDelegate) { return this._setRenderDelegate; }
146 
147     // If this view does not have a render delegate but has
148     // renderDelegateName set, try to retrieve the render delegate from the
149     // theme.
150     var renderDelegateName = this.get('renderDelegateName'), renderDelegate;
151 
152     if (renderDelegateName) {
153       renderDelegate = this.get('theme')[renderDelegateName];
154       if (!renderDelegate) {
155         throw new Error("%@: Unable to locate render delegate \"%@\" in theme.".fmt(this, renderDelegateName));
156       }
157 
158       return renderDelegate;
159     }
160 
161     return null;
162   }.property('renderDelegateName', 'theme'),

196   /**
197     Invoked whenever your view needs to create its HTML representation.
198 
199     You will normally override this method in your subclassed views to
200     provide whatever drawing functionality you will need in order to
201     render your content.
202 
203     This method is usually only called once per view. After that, the update
204     method will be called to allow you to update the existing HTML
205     representation.
206 
207 
208     The default implementation of this method calls renderChildViews().
209 
210     For backwards compatibility, this method will also call the appropriate
211     method on a render delegate object, if your view has one.
212 
213     @param {SC.RenderContext} context the render context
214     @returns {void}
215   */

217     var renderDelegate = this.get('renderDelegate');
218 
219     if (renderDelegate) {
220       if (firstTime) {
221         renderDelegate.render(this.get('renderDelegateProxy'), context);
222       } else {
223         renderDelegate.update(this.get('renderDelegateProxy'), context.$());
224       }
225     }

252   /**
253     Invokes a method on the render delegate, if one is present and it implements
254     that method.
255 
256     @param {String} method The name of the method to call.
257     @param arg One or more arguments.
258   */
259   invokeRenderDelegateMethod: function(method, args) {
260     var renderDelegate = this.get('renderDelegate');
261     if (!renderDelegate) return undefined;
262 
263     if (SC.typeOf(renderDelegate[method]) !== SC.T_FUNCTION) return undefined;
264 
265     args = SC.$A(arguments);
266     args.shift();
267     args.unshift(this.get('renderDelegateProxy'));
268     return renderDelegate[method].apply(renderDelegate, args);

272 /**
273   @class
274   @private
275   View Render Delegate Proxies are tool SC.Views use to:
276 
277   - look up 'display*' ('displayTitle' instead of 'title') to help deal with
278     differences between the render delegate's API and the view's.
279 
280   RenderDelegateProxies are fully valid data sources for render delegates. They
281   act as proxies to the view, interpreting the .get and .didChangeFor commands
282   based on the view's displayProperties.
283 
284   This tool is not useful outside of SC.View itself, and as such, is private.
285 */
286 SC.View._RenderDelegateProxy = {
287 
288   //@if(debug)
289   // for testing:
290   isViewRenderDelegateProxy: YES,
291   //@endif
292 
293   /**
294     Creates a View Render Delegate Proxy for the specified view.
295 
296     Implementation note: this creates a hash of the view's displayProperties
297     array so that the proxy may quickly determine whether a property is a
298     displayProperty or not. This could cause issues if the view's displayProperties
299     array is modified after instantiation.
300 
301     @param {SC.View} view The view this proxy should proxy to.
302     @returns SC.View._RenderDelegateProxy
303   */
304   createForView: function(view) {
305     var ret = SC.beget(this);
306 
307     // set up displayProperty lookup for performance
308     var dp = view.get('displayProperties'), lookup = {};
309     for (var idx = 0, len = dp.length; idx < len; idx++) {
310       lookup[dp[idx]] = YES;
311     }
312 
313     // also allow the few special properties through
314     lookup.theme = YES;
315 
316     ret._displayPropertiesLookup = lookup;
317     ret.renderState = {};
318 
319     ret._view = view;
320     return ret;
321   },
322 
323 
324   /**
325     Provides the render delegate with any property it needs.
326 
327     This first looks up whether the property exists in the view's
328     displayProperties, and whether it exists prefixed with 'display';
329     for instance, if the render delegate asks for 'title', this will
330     look for 'displayTitle' in the view's displayProperties array.
331 
332    @param {String} property The name of the property the render delegate needs.
333    @returns The value.
334   */
335   get: function(property) {
336     if (this[property] !== undefined) { return this[property]; }
337 
338     var displayProperty = 'display' + property.capitalize();
339 
340     if (this._displayPropertiesLookup[displayProperty]) {
341       return this._view.get(displayProperty);
342     } else {
343       return this._view.get(property);
344     }
345   },
346 
347   /**
348    Checks if any of the specified properties have changed.
349 
350    For each property passed, this first determines whether to use the
351    'display' prefix. Then, it calls view.didChangeFor with context and that
352    property name.
353   */
354   didChangeFor: function(context) {
355     var len = arguments.length, idx;
356     for (idx = 1; idx < len; idx++) {
357       var property = arguments[idx],
358           displayProperty = 'display' + property.capitalize();
359 
360       if (this._displayPropertiesLookup[displayProperty]) {
361         if (this._view.didChangeFor(context, displayProperty)) { return YES; }
362       } else {
363         if (this._view.didChangeFor(context, property)) { return YES; }
364       }
365     }
366 
367     return NO;
368   }
369 };

371 /**
372   Generates a computed property that will look up the specified property from
373   the view's render delegate, if present. You may specify a default value to
374   return if there is no such property or is no render delegate.
375 
376   The generated property is read+write, so it may be overridden.
377 
378   @param {String} propertyName The name of the property to get from the render delegate..
379   @param {Value} def The default value to use if the property is not present.
380 */
381 SC.propertyFromRenderDelegate = function(propertyName, def) {
382   return function(key, value) {
383     // first, handle set() case
384     if (value !== undefined) {
385       this['_set_rd_' + key] = value;
386     }
387 
388     // use any value set manually via set()  -- two lines ago.
389     var ret = this['_set_rd_' + key];
390     if (ret !== undefined) return ret;
391 
392     // finally, try to get it from the render delegate
393     var renderDelegate = this.get('renderDelegate');
394     if (renderDelegate && renderDelegate.get) {
395       var proxy = this.get('renderDelegateProxy');
396       ret = renderDelegate.getPropertyFor(proxy, propertyName);
397     }
398 
399     if (ret !== undefined) return ret;
400 
401     return def;
402   }.property('renderDelegate').cacheable();
403 };
