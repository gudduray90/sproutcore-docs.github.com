  8 var content, controller;
  9 
 10 // ..........................................................
 11 // NULL ARRAY
 12 // 
 13 
 14 module("SC.ObjectController - empty_case - NULL", {
 15   setup: function() {
 16     content = null;
 17     controller = SC.ObjectController.create({ content: content });
 18   },
 19  20   
 21   teardown: function() {
 22     controller.destroy();
 23   }
 24 });
 25 
 26 test("getting any value should return undefined", function() {
 27   equals(controller.get("foo"), undefined, 'controller.get(foo)');
 28   equals(controller.get("bar"), undefined, 'controller.get(bar)');
 29 });
 30 
 31 test("setting any unknown value should have no effect", function() {
 32   equals(controller.set("foo", "FOO"), controller, 'controller.set(foo, FOO) should return self');  
 33   equals(controller.set("bar", "BAR"), controller, 'controller.set(bar, BAR) should return self');
 34   equals(controller.get("foo"), undefined, 'controller.get(foo)');
 35   equals(controller.get("bar"), undefined, 'controller.get(bar)');
 36 });
 37 
 38 test("hasContent", function() {
 39   equals(controller.get("hasContent"), NO, 'hasContent should be NO');
 40 });
