 11 /*global module, test, ok, equals */
 12 
 13 var responder;
 14 
 15 module("SC.RootResponder", {
 16 	setup: function() {
 17 		responder = SC.RootResponder.create();
 18 	},
 19 	
 20 	teardown: function() {
 21     responder.destroy();
 22   }
 23 });
 24 
 25 test("Basic requirements", function() {
 26   ok(SC.RootResponder, "SC.RootResponder");
 27   ok(SC.RootResponder.responder && SC.RootResponder.responder, "SC.RootResponder.responder");
 28   equals(
 29     SC.RootResponder.responder ? SC.RootResponder.responder.constructor : "no responder!",
 30     SC.RootResponder,
 31     "SC.RootResponder.responder is an instance of"
 32   );
 33 });
 34 
 35 test("root_responder.ignoreTouchHandle() : Should ignore TEXTAREA, INPUT, A, and SELECT elements", function () {
 36  var wasMobileSafari = SC.browser.isMobileSafari;
 37  SC.browser.isMobileSafari = YES;
 38 
 39  ["A", "INPUT", "TEXTAREA", "SELECT"].forEach(function (tag) {
 40    ok(responder.ignoreTouchHandle({
 41      target: { tagName: tag },
 42      allowDefault: SC.K
 43    }), "should pass touch events through to <" + tag + "> elements");
 44  });
 45 
 46  ["AUDIO", "B", "Q", "BR", "BODY", "BUTTON", "CANVAS", "FORM",
 47   "IFRAME", "IMG", "OPTION", "P", "PROGRESS", "STRONG",
 48   "TABLE", "TBODY", "TR", "TH", "TD", "VIDEO"].forEach(function (tag) {
 49    ok(!responder.ignoreTouchHandle({
 50      target: { tagName: tag },
 51      allowDefault: SC.K
 52    }), "should NOT pass touch events through to <" + tag + "> elements");
 53  });
 54 
 55  SC.browser.isMobileSafari = wasMobileSafari;
 56 });
