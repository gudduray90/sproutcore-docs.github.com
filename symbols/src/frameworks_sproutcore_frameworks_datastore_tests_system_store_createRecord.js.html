 83 /**
 84   There is new functionality in the store that allows the developer to reference
 85   attribute names or attribute keys when creating a new record.
 86 
 87   For example, if the Record is defined as:
 88 
 89     MyApp.Record = SC.Record.extend({
 90       attrA: SC.RecordAttribute(String, { key: 'attr_a' });
 91     })
 92 
 93   Previously, passing the hash { attrA: 'test' } would see `attrA` added to the
 94   hash when it should instead be `attr_a`.  The new functionality, recognizes
 95   that the attribute key should be used.
 96 
 97   Therefore, either of these will result in the same data hash in the store:
 98 
 99     MyApp.store.createRecord(MyApp.Record, { attrA: 'test' })
100     MyApp.store.createRecord(MyApp.Record, { attr_a: 'test' })
101 
102 */
103 test("Creating a keyed record", function() {
104   var hash1, hash2, hash3, hash4, expectedHash, sk;
105 
106   // The actual hash that should be created.
107   expectedHash = {
108     string_key: "abcd",
109     number_key: 1,
110     bool_key:   NO,
111     array_key:  [],
112     funcDef_key: [1, 2]
113   };
114 
115   // Uses only the attribute names
116   hash1 = {
117     string: "abcd",
118     number: 1,
119     bool:   NO,
120     array:  [],
121     funcDef: [1, 2]
122   };
123 
124   // Uses only the attribute keys
125   hash2 = {
126     string_key: "abcd",
127     number_key: 1,
128     bool_key:   NO,
129     array_key:  [],
130     funcDef_key: [1, 2]
131   };
132 
133   // Uses a mix of attribute names and keys
134   hash3 = {
135     string: "abcd",
136     number_key: 1,
137     bool_key:   NO,
138     array:  [],
139     funcDef: [1, 2]
140   };
141 
142   // Uses duplicate attribute names and keys in different orders
143   hash4 = {
144     string: "efgh",
145     string_key: "abcd",
146     number_key: 1,
147     bool_key:   NO,
148     bool:   YES,
149     array_key:  [],
150     array:  ['a'],
151     funcDef: [1, 2]
152   };
153 
154   rec = store.createRecord(MyKeyedRecordType, hash1, 'test1');
155   sk = store.storeKeyFor(MyKeyedRecordType, rec.get('id'));
156   same(store.readDataHash(sk), expectedHash, "data hashes are equivalent when given only names");
157 
158   rec = store.createRecord(MyKeyedRecordType, hash2, 'test2');
159   sk = store.storeKeyFor(MyKeyedRecordType, rec.get('id'));
160   same(store.readDataHash(sk), expectedHash, "data hashes are equivalent when given only keys");
161 
162   rec = store.createRecord(MyKeyedRecordType, hash3, 'test3');
163   sk = store.storeKeyFor(MyKeyedRecordType, rec.get('id'));
164   same(store.readDataHash(sk), expectedHash, "data hashes are equivalent when given names and keys");
165 
166   rec = store.createRecord(MyKeyedRecordType, hash4, 'test4');
167   sk = store.storeKeyFor(MyKeyedRecordType, rec.get('id'));
168   same(store.readDataHash(sk), expectedHash, "data hashes are equivalent when given names and keys with conflicts");
169 });

180 test("Initializing default values", function() {
181   var rec1, rec2, sk1, sk2;
182 
183   //create 2 records
184   rec1 = store.createRecord(MyRecordType, null, 'test1');
185   rec2 = store.createRecord(MyRecordType, null, 'test2');
186 
187   //get storKeys
188   sk1 = store.storeKeyFor(MyRecordType, rec1.get('id'));
189   sk2 = store.storeKeyFor(MyRecordType, rec2.get('id'));
190 
191   ok(sk1, "a first record with default values was created");
192 
193   equals(store.readDataHash(sk1)['string'], "Untitled", "the default value for 'string' was initialized");
194   equals(store.readDataHash(sk1)['number'], 5, "the default value for 'number' was initialized");
195   equals(store.readDataHash(sk1)['bool'], YES, "the default value for 'bool' was initialized");
196   same(store.readDataHash(sk1)['array'], [1, 2], "the default value for 'array' was initialized");
197   same(store.readDataHash(sk1)['funcDef'], [1, 3], "the default value for 'funcDef' was initialized");
198   equals(store.readDataHash(sk1)['noDefault'], null, "no value for 'noDefault' was initialized");
199 
200   ok(sk2, "a second record with default values was created");
201 
202   rec2.get('array').push(3);
203   rec2.get('funcDef').push(2);
204 
205   same(store.readDataHash(sk2)['array'], [1, 2, 3], "the array for 'array' was updated");
206   same(store.readDataHash(sk2)['funcDef'], [1, 3, 2], "the array for 'funcDef' was updated");
207 
208   ok(store.readDataHash(sk2)['array'] !== store.readDataHash(sk1)['array'], "the default value for 'array' is a copy not a reference");
209   ok(store.readDataHash(sk2)['funcDef'] !== store.readDataHash(sk1)['funcDef'], "the default value for 'funcDef' is a copy not a reference");
210 });
