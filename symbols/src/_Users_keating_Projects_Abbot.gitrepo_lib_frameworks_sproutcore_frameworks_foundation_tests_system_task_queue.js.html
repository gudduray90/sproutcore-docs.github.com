 10 /*globals module test ok isObj equals expects */
 11 var taskQueue;
 12 module("Task Queue",{
 13   setup: function(){
 14     taskQueue = SC.TaskQueue.create();
 15   }
 16 });
 17 
 18 
 19 
 20 test("Adding a task should not cause it to run.",function(){
 21   var task = SC.Task.create({ run: function(){ this.ran = YES; } });
 22   taskQueue.push(task);
 23   
 24   ok(!task.ran, "Task should not have run");
 25 });
 26 
 27 
 28 test("Adding a task and calling run() should cause the task to be run.",function(){
 29   var task = SC.Task.create({ run: function(){ this.ran = YES; } });
 30   taskQueue.push(task);
 31   taskQueue.run();
 32   
 33   ok(task.ran, "Task should have run");
 34 });
 35 
 36 test("Adding multiple tasks and calling run should run the tasks in order.",function(){
 37   var ri = 0;
 38   var task = SC.Task.extend({ run: function(){ this.ran = ri++; } }), t1, t2, t3, t4;
 39   taskQueue.push(t1 = task.create());
 40   taskQueue.push(t2 = task.create());
 41   taskQueue.push(t3 = task.create());
 42   taskQueue.push(t4 = task.create());
 43   taskQueue.run();
 44   
 45   equals(t1.ran, 0, "Task 1 should be first");
 46   equals(t2.ran, 1, "Task 1 should be second");
 47   equals(t3.ran, 2, "Task 1 should be third");
 48   equals(t4.ran, 3, "Task 1 should be fourth");
 49 });
