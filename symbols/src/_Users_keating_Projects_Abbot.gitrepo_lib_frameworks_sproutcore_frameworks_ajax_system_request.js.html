 10 /**
 11   @class
 12 
 13   Implements support for Ajax requests using XHR and other prototcols.
 14 
 15   SC.Request is much like an inverted version of the request/response objects
 16   you receive when implementing HTTP servers.
 17 
 18   To send a request, you just need to create your request object, configure
 19   your options, and call send() to initiate the request.
 20 
 21   @extends SC.Object
 22   @extends SC.Copyable
 23   @extends SC.Freezable
 24   @since SproutCore 1.0

222   concatenatedProperties: 'COPY_KEYS',

224   /** @private */
225   COPY_KEYS: ['attachIdentifyingHeaders', 'isAsynchronous', 'isJSON', 'isXML', 'address', 'type', 'timeout', 'body', 'responseClass', 'willSend', 'didSend', 'willReceive', 'didReceive'],

227   /**
228     Returns a copy of the current request. This will only copy certain
229     properties so if you want to add additional properties to the copy you
230     will need to override copy() in a subclass.
231 
232     @returns {SC.Request} new request
233   */
234   copy: function() {
235     var ret = {},
236         keys = this.COPY_KEYS,
237         loc = keys.length,
238         key, listeners, headers;
239 
240     while(--loc >= 0) {
241       key = keys[loc];
242       if (this.hasOwnProperty(key)) {
243         ret[key] = this.get(key);
244       }
245     }
246 
247     if (this.hasOwnProperty('listeners')) {
248       ret.listeners = SC.copy(this.get('listeners'));
249     }
250 
251     if (this.hasOwnProperty('_headers')) {
252       ret._headers = SC.copy(this._headers);
253     }
254 
255     ret.source = this.get('source') || this;
256 
257     return this.constructor.create(ret);
258   },
259 
260   /**
261     To set headers on the request object. Pass either a single key/value
262     pair or a hash of key/value pairs. If you pass only a header name, this
263     will return the current value of the header.
264 
265     @param {String|Hash} key
266     @param {String} value
267     @returns {SC.Request|Object} receiver
268   */
269   header: function(key, value) {
270     var headers;
271 
272     if (SC.typeOf(key) === SC.T_STRING) {
273       headers = this._headers;
274       if (arguments.length === 1) {
275         return headers ? headers[key] : null;
276       } else {
277         this.propertyWillChange('headers');
278         if (!headers) { headers = this._headers = {}; }
279         headers[key] = value;
280         this.propertyDidChange('headers');
281         return this;
282       }
283 
284     // handle parsing hash of parameters
285     } else if (value === undefined) {
286       headers = key;
287       this.beginPropertyChanges();
288       for(key in headers) {
289         if (!headers.hasOwnProperty(key)) { continue; }
290         this.header(key, headers[key]);
291       }
292       this.endPropertyChanges();
293       return this;
294     }
295 
296     return this;
297   },

355   /** 
356     Called just before a request is enqueued.  This will encode the body 
357     into JSON if it is not already encoded, and set identifying headers
358   */
359   _prep: function() {
360     var hasContentType = !!this.header('Content-Type');
361 
362     if(this.get('attachIdentifyingHeaders')) {
363       this.header('X-Requested-With', 'XMLHttpRequest');
364       this.header('X-SproutCore-Version', SC.VERSION);
365     }
366     
367     if (this.get('isJSON') && !hasContentType) {
368       this.header('Content-Type', 'application/json');
369     } else if (this.get('isXML') && !hasContentType) {
370       this.header('Content-Type', 'text/xml');
371     }
372     return this;
373   },
374   
375   /**
376     Will fire the actual request. If you have set the request to use JSON
377     mode then you can pass any object that can be converted to JSON as the
378     body. Otherwise you should pass a string body.
379     
380     @param {String|Object} [body]
381     @returns {SC.Response} New response object
382   */  
383   send: function(body) {
384     // Sanity-check: Be sure a timeout value was not specified if the request
385     // is synchronous (because it wouldn't work).
386     var timeout = this.get('timeout');
387     if (timeout && !this.get('isAsynchronous')) {
388       throw "Timeout values cannot be used with synchronous requests";
389     } else if (timeout === 0) {
390       throw "The timeout value must either not be specified or must be greater than 0";
391     }
392 
393     if (body) { this.set('body', body); }
394     return SC.Request.manager.sendRequest(this.copy()._prep());
395   },
396 
397   /**
398     Resends the current request. This is more efficient than calling send()
399     for requests that have already been used in a send. Otherwise acts just
400     like send(). Does not take a body argument.
401 
402     @returns {SC.Response} new response object
403   */
404   resend: function() {
405     var req = this.get('source') ? this : this.copy()._prep();
406     return SC.Request.manager.sendRequest(req);
407   },

409   /**
410     Configures a callback to execute when a request completes. You must pass
411     at least a target and action/method to this and optionally a status code.
412     You may also pass additional parameters which will be passed along to your
413     callback. If your callback handled the notification, it should return YES.
414 
415     ## Scoping With Status Codes
416 
417     If you pass a status code as the first option to this method, then your
418     notification callback will only be called if the response status matches
419     the code. For example, if you pass 201 (or SC.Request.CREATED) then
420     your method will only be called if the response status from the server
421     is 201.
422 
423     You can also pass "generic" status codes such as 200, 300, or 400, which
424     will be invoked anytime the status code is the range if a more specific
425     notifier was not registered first and returned YES.
426 
427     Finally, passing a status code of 0 or no status at all will cause your
428     method to be executed no matter what the resulting status is unless a
429     more specific notifier was registered and returned YES.
430 
431     ## Callback Format
432 
433     Your notification callback should expect to receive the Response object
434     as the first parameter plus any additional parameters that you pass.
435 
436     @param {Number} status
437     @param {Object} target
438     @param {String|Function} action
439     @param {Hash} params
440     @returns {SC.Request} receiver
441   */
442   notify: function(status, target, action, params) {
443     // normalize status
444     var hasStatus = YES;
445     if (SC.typeOf(status) !== SC.T_NUMBER) {
446       params = SC.A(arguments).slice(2);
447       action = target;
448       target = status;
449       status = 0;
450       hasStatus = NO;
451     } else {
452       params = SC.A(arguments).slice(3);
453     }
454 
455     var listeners = this.get('listeners');
456     if (!listeners) { this.set('listeners', listeners = {}); }
457     if(!listeners[status]) { listeners[status] = []; }
458 
459     listeners[status].push({target: target, action: action, params: params});
460 
461     return this;
462   }

517 /**
518   @class
519 
520   The request manager coordinates all of the active XHR requests. It will
521   only allow a certain number of requests to be active at a time; queuing
522   any others. This allows you more precise control over which requests load
523 524   in which order.
525 
526   @since SproutCore 1.0
527 */
528 SC.Request.manager = SC.Object.create(
529 /** @scope SC.Request.manager */{
530 
531   /**
532     Maximum number of concurrent requests allowed. 6 for all browsers.
533     
534     @type Number
535     @default 6
536   */
537   maxRequests: 6,
538 
539   /**
540     Current requests that are inflight.
541 
542     @type Array
543     @default []
544   */
545   inflight: [],
546 
547   /**
548     Requests that are pending and have not been started yet.
549 
550     @type Array
551     @default []
552   */
553   pending: [],
554 
555 
556   // ..........................................................
557   // METHODS
558   // 
559   
560   /**
561     Invoked by the send() method on a request. This will create a new low-
562     level transport object and queue it if needed.
563     
564     @param {SC.Request} request the request to send
565     @returns {SC.Object} response object
566   */
567   sendRequest: function(request) {
568     if (!request) { return null; }
569 
570     // create low-level transport.  copy all critical data for request over
571     // so that if the request has been reconfigured the transport will still
572     // work.
573     var response = request.get('responseClass').create({ request: request });
574 
575     // add to pending queue
576     this.get('pending').pushObject(response);
577     this.fireRequestIfNeeded();
578 
579     return response;
580   },
581 
582   /** 
583     Cancels a specific request. If the request is pending it will simply
584     be removed. Otherwise it will actually be cancelled.
585 
586     @param {Object} response a response object
587     @returns {Boolean} YES if cancelled
588   */
589   cancel: function(response) {
590     var pending = this.get('pending'),
591         inflight = this.get('inflight'),
592         idx;
593 
594     if (pending.indexOf(response) >= 0) {
595       this.propertyWillChange('pending');
596       pending.removeObject(response);
597       this.propertyDidChange('pending');
598       return YES;
599     } else if (inflight.indexOf(response) >= 0) {
600       response.cancel();
601 
602       inflight.removeObject(response);
603       this.fireRequestIfNeeded();
604       return YES;
605     }
606 
607     return NO;
608   },
609 
610   /**
611     Cancels all inflight and pending requests.
612 
613     @returns {Boolean} YES if any items were cancelled.
614   */
615   cancelAll: function() {
616     if (this.get('pending').length || this.get('inflight').length) {
617       this.set('pending', []);
618       this.get('inflight').forEach(function(r) { r.cancel(); });
619       this.set('inflight', []);
620       return YES;
621     }
622 
623     return NO;
624   },
625   
626   /**
627     Checks the inflight queue. If there is an open slot, this will move a
628     request from pending to inflight.
629 
630     @returns {Object} receiver
631   */
632   fireRequestIfNeeded: function() {
633     var pending = this.get('pending'),
634         inflight = this.get('inflight'),
635         max = this.get('maxRequests'),
636         next;
637 
638     if ((pending.length>0) && (inflight.length<max)) {
639       next = pending.shiftObject();
640       inflight.pushObject(next);
641       next.fire();
642     }
643   },
644 
645   /**
646     Called by a response/transport object when finishes running. Removes
647     the transport from the queue and kicks off the next one.
648   */
649   transportDidClose: function(response) {
650     this.get('inflight').removeObject(response);
651     this.fireRequestIfNeeded();
652   }
653
