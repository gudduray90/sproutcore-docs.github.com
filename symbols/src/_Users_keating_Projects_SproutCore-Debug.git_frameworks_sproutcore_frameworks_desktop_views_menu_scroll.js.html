198   /** @private */
199   _sc_scroller_armScrollTimer: function() {
200     if (!this._sc_scrollTimer) {
201       SC.RunLoop.begin() ;
202       var method = this._sc_scroller_scrollDidChange ;
203       this._sc_scrollTimer = this.invokeLater(method, 50) ;
204       SC.RunLoop.end() ;

208   /** @private */
209   _sc_scroller_scrollDidChange: function() {
210     var now = Date.now(), 
211         last = this._sc_lastScroll, 
212         layer = this.get('layer'), 
213         scroll = 0 ;
214     
215     if (last && (now-last)<50) return this._sc_scroller_armScrollTimer() ;
216     this._sc_scrollTimer = null ;
217     this._sc_lastScroll = now ;
218     
219     SC.RunLoop.begin();
220     
221     if (!this.get('isEnabled')) return ; // nothing to do.
222     
223     this._sc_scrollValue = scroll = layer.scrollTop ;
224     this.set('value', scroll) ; // will now enforce minimum and maximum
225     
226     SC.RunLoop.end();
227   },

230   /** @private
231     Scroll the menu if it is is an up or down arrow. This is called by
232     the function that simulates mouseOver.
233   */
234   _scrollMenu: function(){
235     var val = this.get('value'), newval;
236     if(this.get('scrollDown')) {
237       newval = val+this.verticalLineScroll;
238       if(newval<=this.get('maximum')){
239         this.set('value', newval);
240       }
241     }
242     else {
243       newval = val-this.verticalLineScroll;
244       if(newval>=0){
245 246         this.set('value', newval);
247       }else if(val<=this.verticalLineScroll && val>0){
248         this.set('value', 0);
249       }
250     }
251     return YES;
252   },
253   
254   /** @private
255     
256     We use this function to simulate mouseOver. It checks for the flag 
257     isMouseOver which is turned on when mouseEntered is called and turned off
258     when mouseExited is called. 
259   */
260   _invokeScrollOnMouseOver: function(){
261     this._scrollMenu();
262     if(this.get('isMouseOver')){
263       this.invokeLater(this._invokeScrollOnMouseOver, 50);
264     }
265   }

478   /**
479     @param {SC.View} view
480   */
481   scrollToVisible: function(view) {
482     // if no view is passed, do default
483     if (arguments.length === 0) return sc_super();
484 
485     var contentView = this.get('contentView') ;
486     if (!contentView) return NO; // nothing to do if no contentView.
487 
488     // get the frame for the view - should work even for views with static
489     // layout, assuming it has been added to the screen.
490     var vf = view.get('frame');
491     if (!vf) return NO; // nothing to do
492 
493     // convert view's frame to an offset from the contentView origin.  This
494     // will become the new scroll offset after some adjustment.
495     vf = contentView.convertFrameFromView(vf, view.get('parentView')) ;
496 
497     var vscroll2 = this.get('verticalScrollerView2');
498     if (vscroll2 && vscroll2.get('isVisible')) {
499       vf.height += vscroll2.get('frame').height;
500     }
501 
502     var vscroll = this.get('verticalScrollerView');
503     if (vscroll && vscroll.get('isVisible')) {
504       vf.top -= vscroll.get('frame').height;
505     }
506 
507     return this.scrollToRect(vf);
508   },

510   /** @private
511     Adjusts the layout for the various internal views.  This method is called
512     once when the scroll view is first configured and then anytime a scroller
513     is shown or hidden.  You can call this method yourself as well to retile.
514     
515     You may also want to override this method to handle layout for any
516     additional controls you have added to the view.
517   */
518   tile: function() {
519     // get vertical scroller/determine if we should have a scroller
520     var hasScroller, vscroll, vscroll2, hasVertical, clip, clipLayout, viewportHeight;
521     hasScroller = this.get('hasVerticalScroller');
522     vscroll = hasScroller ? this.get('verticalScrollerView') : null ;
523     vscroll2 = hasScroller ? this.get('verticalScrollerView2') : null ;
524     hasVertical = vscroll && this.get('isVerticalScrollerVisible') ;
525     
526     // get the containerView
527     clip = this.get('containerView') ;
528     clipLayout = { left: 0, top: 0 } ;
529     
530     if (hasVertical) {
531       viewportHeight =0;
532       var scrollerThickness = vscroll.get('scrollerThickness') || vscroll2.get('scrollerThickness');
533       var view   = this.get('contentView'), view2, 
534             f      = (view) ? view.get('frame') : null, 
535             height = (f) ? f.height : 0,
536             elem = this.containerView.$()[0],
537             verticalOffset = this.get('verticalScrollOffset'),
538             topArrowInvisible = { height: 0, top: 0, right: 0, left: 0 },
539             topArrowVisible = { height: scrollerThickness, top: 0, right: 0, left: 0 },
540             bottomArrowVisible = { height: scrollerThickness, bottom: 0, right: 0, left: 0 },
541             bottomArrowInvisible = { height: 0, bottom: 0, right: 0, left: 0 };
542       
543       if(elem) viewportHeight = elem.offsetHeight;
544       
545       if(verticalOffset===0){
546         clipLayout.top = 0 ;
547         clipLayout.bottom = scrollerThickness;
548         vscroll.set('layout', topArrowInvisible) ;
549         vscroll2.set('layout', bottomArrowVisible) ;
550       }else if(verticalOffset>=(height-viewportHeight-scrollerThickness)){
551         clipLayout.top = scrollerThickness ;
552         clipLayout.bottom = 0 ;
553         vscroll.set('layout', topArrowVisible) ;
554         vscroll2.set('layout', bottomArrowInvisible) ;
555       }else{
556         clipLayout.top = scrollerThickness ;
557         clipLayout.bottom = scrollerThickness ;
558         vscroll.set('layout', topArrowVisible) ;
559         vscroll2.set('layout', bottomArrowVisible) ;
560       }
561     } 
562     if (vscroll){
563      vscroll.set('isVisible', hasVertical) ;
564      vscroll2.set('isVisible', hasVertical) ;
565     }
566     clip.set('layout', clipLayout) ;
567   },

581   /** @private
582     Instantiate scrollers & container views as needed.  Replace their classes
583     in the regular properties.
584   */
585   createChildViews: function() {
586     var childViews = [], view, view2, controlSize = this.get('controlSize') ;
587     
588     // create the containerView.  We must always have a container view. 
589     // also, setup the contentView as the child of the containerView...
590     if (SC.none(view = this.containerView)) view = SC.ContainerView;
591     
592     childViews.push(this.containerView = this.createChildView(view, {
593       contentView: this.contentView
594     }));
595     
596     // and replace our own contentView...
597     this.contentView = this.containerView.get('contentView');
598     
599     // create a vertical scroller 
600     if ((view=this.verticalScrollerView) && (view2=this.verticalScrollerView2)) {
601       if (this.get('hasVerticalScroller')) {
602         view = this.verticalScrollerView = this.createChildView(view, {
603           layout: {top: 0, left: 0, right: 0},
604           controlSize: controlSize,
605           valueBinding: '*owner.verticalScrollOffset'
606         }) ;
607         childViews.push(view);
608         view2 = this.verticalScrollerView2 = this.createChildView(view2, {
609           scrollDown: YES,
610           layout: {bottom: 0, left: 0, right: 0 },
611           controlSize: controlSize,
612           valueBinding: '*owner.verticalScrollOffset'
613         }) ;
614         childViews.push(view2);
615       } else {
616         this.verticalScrollerView = null ;
617         this.verticalScrollerView2 = null ;
618       }
619     }
620     
621     // set childViews array.
622     this.childViews = childViews ;
623     
624     this.contentViewFrameDidChange() ; // setup initial display...
625     this.tile() ; // set up initial tiling
626   },

651   /** @private
652     Invoked whenever the contentView's frame changes.  This will update the 
653     scroller maximum and optionally update the scroller visibility if the
654     size of the contentView changes.  We don't care about the origin since
655     that is tracked separately from the offset values.
656   */
657   contentViewFrameDidChange: function() {
658     var view   = this.get('contentView'), view2, 
659         f      = (view) ? view.get('frame') : null,
660         width  = (f) ? f.width : 0,
661         height = (f) ? f.height : 0,
662         dim    = this.get('frame'),
663         viewportHeight, elem ;
664         
665     // cache out scroll settings...
666     //if ((width === this._scroll_contentWidth) && (height === this._scroll_contentHeight)) return ;
667     this._scroll_contentWidth = width;
668     this._scroll_contentHeight = height ;
669     
670     if (this.get('hasVerticalScroller') && (view = this.get('verticalScrollerView')) && (view2 = this.get('verticalScrollerView2'))) {
671       height -= 1 ; // accurately account for our layout
672       // decide if it should be visible or not
673       if (this.get('autohidesVerticalScroller')) {
674         this.set('isVerticalScrollerVisible', height > dim.height);
675       }
676       height -= this.get('verticalScrollerBottom') ;
677       viewportHeight = 0;
678       elem = this.containerView.$()[0];
679       if(elem) viewportHeight = elem.offsetHeight;
680       height = height - viewportHeight;
681       view.setIfChanged('maximum', height) ;
682       view2.setIfChanged('maximum', height) ;
683     }
684   },
