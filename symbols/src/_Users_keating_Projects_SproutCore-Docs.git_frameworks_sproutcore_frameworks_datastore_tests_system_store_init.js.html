  8 // This file tests the initial state of the store when it is first created
  9 // either independently or as a chained store.
 10 
 11 module("SC.Store#init");
 12 
 13 test("initial setup for root store", function() {
 14   var store = SC.Store.create();
 15   
 16   equals(SC.typeOf(store.dataHashes), SC.T_HASH, 'should have dataHashes');
 17   equals(SC.typeOf(store.revisions), SC.T_HASH, 'should have revisions');
 18   equals(SC.typeOf(store.statuses), SC.T_HASH, 'should have statuses');
 19   ok(!store.editables, 'should not have editables');
 20   ok(!store.recordErrors, 'should not have recordErrors');
 21   ok(!store.queryErrors, 'should not have queryErrors');
 22 });
