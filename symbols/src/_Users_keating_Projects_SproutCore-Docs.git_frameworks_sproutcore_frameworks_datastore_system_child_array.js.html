  8 /**
  9   @class
 10 
 11   A `ChildArray` is used to map an array of `ChildRecord` objects.
 12 
 13   @extends SC.Enumerable
 14   @extends SC.Array
 15   @since SproutCore 1.0
 16 */
 17 
 18 SC.ChildArray = SC.Object.extend(SC.Enumerable, SC.Array,
 19   /** @scope SC.ChildArray.prototype */ {
 20 
 21   /**
 22     If set, it is the default record `recordType`
 23 
 24  25     @default null
 26     @type String
 27   */
 28   defaultRecordType: null,
 29 
 30   /**
 31     If set, the parent record will be notified whenever the array changes so that
 32     it can change its own state
 33 
 34     @default null
 35     @type {SC.Record}
 36   */
 37   record: null,
 38 
 39   /**
 40     If set will be used by the many array to get an editable version of the
 41     `storeId`s from the owner.
 42 
 43     @default null
 44     @type String
 45   */
 46   propertyName: null,
 47 
 48   /**
 49     Actual references to the hashes
 50 
 51     @default null
 52     @type {SC.Array}
 53   */
 54   children: null,
 55 
 56   /**
 57     The store that owns this record array.  All record arrays must have a
 58     store to function properly.
 59 
 60     @type SC.Store
 61     @property
 62   */
 63   store: function() {
 64     return this.getPath('record.store');
 65   }.property('record').cacheable(),
 66 
 67   /**
 68     The storeKey for the parent record of this many array.  Editing this
 69     array will place the parent record into a `READY_DIRTY state.
 70 
 71     @type Number
 72     @property
 73   */
 74   storeKey: function() {
 75     return this.getPath('record.storeKey');
 76   }.property('record').cacheable(),
 77 
 78   /**
 79     Returns the storeIds in read only mode.  Avoids modifying the record
 80     unnecessarily.
 81 
 82     @type SC.Array
 83     @property
 84   */
 85  86   readOnlyChildren: function() {
 87     return this.get('record').readAttribute(this.get('propertyName'));
 88   }.property(),
 89 
 90   /**
 91     Returns an editable array of child hashes.  Marks the owner records as
 92     modified.
 93 
 94     @type {SC.Array}
 95     @property
 96   */
 97   editableChildren: function() {
 98     var store    = this.get('store'),
 99         storeKey = this.get('storeKey'),
100         pname    = this.get('propertyName'),
101         ret, hash;
102 
103     ret = store.readEditableProperty(storeKey, pname);
104     if (!ret) {
105       hash = store.readEditableDataHash(storeKey);
106       ret = hash[pname] = [];
107     }
108 
109     return ret ;
110   }.property(),
111 
112   // ..........................................................
113   // ARRAY PRIMITIVES
114   //
115 
116   /** @private
117     Returned length is a pass-through to the storeIds array.
118 
119     @type Number
120     @property
121   */
122   length: function() {
123     var children = this.get('readOnlyChildren');
124     return children ? children.length : 0;
125   }.property('readOnlyChildren'),
126 
127   /**
128     Looks up the store id in the store ids array and materializes a
129     records.
130 
131     @param {Number} idx index of the object to retrieve.
132     @returns {SC.Record} The record if found or undefined.
133   */
134   objectAt: function(idx) {
135     var recs      = this._records,
136         children = this.get('readOnlyChildren'),
137         hash, ret, pname = this.get('propertyName'),
138         parent = this.get('record');
139     var len = children ? children.length : 0;
140 
141     if (!children) return undefined; // nothing to do
142     if (recs && (ret=recs[idx])) return ret ; // cached
143     if (!recs) this._records = recs = [] ; // create cache
144 
145     // If not a good index return undefined
146     if (idx >= len) return undefined;
147     hash = children.objectAt(idx);
148     if (!hash) return undefined;
149 
150     // not in cache, materialize
151     recs[idx] = ret = parent.registerNestedRecord(hash, pname, pname+'.'+idx);
152 
153     return ret;
154   },
155 
156   /**
157     Pass through to the underlying array.  The passed in objects must be
158     records, which can be converted to `storeId`s.
159 
160     @param {Number} idx index of the object to replace.
161     @param {Number} amt number of records to replace starting at idx.
162     @param {Number} recs array with records to replace.
163     @returns {SC.Record} The record if found or undefined.
164 
165   */
166   replace: function(idx, amt, recs) {
167     var children = this.get('editableChildren'),
168         len      = recs ? (recs.get ? recs.get('length') : recs.length) : 0,
169         record   = this.get('record'), newRecs,
170 
171         pname    = this.get('propertyName'),
172         cr, recordType;
173     
174     newRecs = this._processRecordsToHashes(recs);
175     children.replace(idx, amt, newRecs);
176     // notify that the record did change...
177     if (newRecs !== this._prevChildren){
178       this._performRecordPropertyChange(null, false);
179       this.arrayContentWillChange(idx, amt, len);
180       this._childrenContentDidChange(idx, amt, len);
181     } 
182     record.recordDidChange(pname);
183 
184     return this;
185   },
186 
187   /** @private
188 
189     Converts a records array into an array of hashes.
190 
191     @param {SC.Array} recs records to be converted to hashes.
192     @returns {SC.Array} array of hashes.
193   */
194   _processRecordsToHashes: function(recs){
195     var store, sk;
196     recs = recs || [];
197     recs.forEach( function(me, idx){
198       if (me.isNestedRecord){
199         store = me.get('store');
200         sk = me.storeKey;
201         recs[idx] = store.readDataHash(sk);
202       }
203     });
204 
205     return recs;
206   },
207 
208   /**
209     Calls normalize on each object in the array
210   */
211   normalize: function(){
212     this.forEach(function(child,id){
213       if(child.normalize) child.normalize();
214     });
215   },
216

218   // INTERNAL SUPPORT
219   //
220 
221   /**
222     Invoked whenever the children array changes.  Observes changes.
223 
224     @param {SC.Array} keys optional
225     @returns {SC.ChildArray} itself.
226   */
227   recordPropertyDidChange: function(keys) {
228     this._performRecordPropertyChange(keys, true);
229     return this;
230   },
231   
232   /** @private
233     Invoked when the object is changed from the parent or an outside source
234     
235     will cause the entire array to reset
236     
237     @param {SC.Array} keys optional
238     @param {Boolean} doReset optional
239     @returns {SC.ChildArray} itself.
240   */
241   _performRecordPropertyChange: function(keys, doReset){
242     if (keys && !keys.contains(this.get('propertyName'))) return this;
243 
244     var children = this.get('readOnlyChildren'), oldLen = 0, newLen = 0;
245     var prev = this._prevChildren, f = this._childrenContentDidChange;
246     doReset = SC.none(doReset) ? true : doReset;
247 
248     if (children === prev) return this; // nothing to do
249 
250     if (prev) {
251       prev.removeArrayObservers({
252         target: this,
253         willChange: this.arrayContentWillChange,
254         didChange: f
255       });
256 
257       oldLen = prev.get('length');
258     }
259 
260     if (children) {
261       children.addArrayObservers({
262 263         target: this,
264         willChange: this.arrayContentWillChange,
265         didChange: f
266       });
267 
268       newLen = children.get('length');
269     }
270     this._prevChildren = children;
271     
272     if (doReset){
273       this.arrayContentWillChange(0, oldLen, newLen);
274       this._childrenContentDidChange(0, oldLen, newLen);      
275     }
276 
277     return this;
278   },
279 
280   /** @private
281     Invoked whenever the content of the children array changes.  This will
282     dump any cached record lookup and then notify that the enumerable content
283     has changed.
284 
285     @param {Number} target
286     @param {Number} key
287     @param {Number} value
288     @param {Number} rev
289   */
290   _childrenContentDidChange: function(start, removedCount, addedCount) {
291     this._records = null ; // clear cache
292     this.arrayContentDidChange(start, removedCount, addedCount);
293   },
294 
295   /** @private */
296   init: function() {
297     sc_super();
298     this.recordPropertyDidChange();
299   }
