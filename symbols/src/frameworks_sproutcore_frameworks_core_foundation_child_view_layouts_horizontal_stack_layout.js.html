 12   /**
 13     This child layout plugin automatically positions the view's child views in a
 14     horizontal stack and optionally adjusts the parent view's height to fit.  It
 15     does this by checking the height of each child view and positioning the
 16     following child view accordingly.  By default any time that a child view's
 17     height changes, the view will use this plugin to re-adjust all other child
 18     views and its own height appropriately.
 19 
 20     A typical usage scenario is a long "form" made of multiple subsection
 21     views.  If we want to adjust the height of a subsection, to make space for
 22     an error label for example, it would be a lot of work to manually
 23     reposition all the following sections below it.  A much easier to code and
 24     cleaner solution is to just set the childViewLayout plugin on the wrapper
 25     view.
 26 
 27     For example,
 28 
 29         MyApp.MyView = SC.View.extend({
 30 
 31           // Child views will be stacked in order horizontally.
 32           childViewLayout: SC.View.HORIZONTAL_STACK,
 33 
 34           // The order of child views is important!
 35           childViews: ['sectionA', 'sectionB', 'sectionC'],
 36 
 37           // Actual layout will become { left: 10, bottom: 20, top: 20, width: 270 }
 38           layout: { left: 10, bottom: 20, top: 20 }, // Don't need to specify layout.width, this is automatic.
 39 
 40           sectionA: SC.View.design({
 41             // Actual layout will become { left: 0, bottom: 0, top: 0, width: 100 }
 42             layout: { width: 100 } // Don't need to specify layout.left, this is automatic.
 43           }),
 44 
 45           sectionB: SC.View.design({
 46             // Actual layout will become { border: 1, left: 100, bottom: 0, top: 0, width: 50 }
 47             layout: { border: 1, width: 50 } // Don't need to specify layout.left, this is automatic.
 48           }),
 49 
 50           sectionC: SC.View.design({
 51             // Actual layout will become { left: 150, bottom: 10, top: 10, width: 120 }
 52             layout: { right: 10, top: 10, width: 120 } // Don't need to specify layout.left, this is automatic.
 53           })
 54 
 55         });
 56 
 57     ## Modify all child view layouts with `childViewLayoutOptions`
 58 
 59     To modify the plugin behavior for all child view layouts, you can set the
 60     following child view layout options in `childViewLayoutOptions` on the view:
 61 
 62       - paddingBefore - Adds padding before the first child view.  Default: 0
 63       - paddingAfter - Adds padding after the last child view.  Default: 0
 64       - spacing - Adds spacing between each child view.  Default: 0
 65       - resizeToFit - Whether to resize the view to fit the child views (requires that each child view has a layout width).  Default: true
 66 
 67     For example,
 68 
 69         MyApp.MyView = SC.View.extend({
 70 
 71           // Child views will be stacked in order horizontally.
 72           childViewLayout: SC.View.HORIZONTAL_STACK,
 73 
 74           // Change the behavior of the HORIZONTAL_STACK plugin
 75           childViewLayoutOptions: {
 76             paddingBefore: 10,
 77             paddingAfter: 20,
 78             spacing: 5
 79           },
 80 
 81           // The order of child views is important!
 82           childViews: ['sectionA', 'sectionB', 'sectionC'],
 83 
 84           // Actual layout will become { left: 10, bottom: 20, top: 20, width: 310 }
 85           layout: { left: 10, bottom: 20, top: 20 }, // Don't need to specify layout.width, this is automatic.
 86 
 87           sectionA: SC.View.design({
 88             // Actual layout will become { left: 10, bottom: 0, top: 0, width: 100 }
 89             layout: { width: 100 } // Don't need to specify layout.left, this is automatic.
 90           }),
 91 
 92           sectionB: SC.View.design({
 93             // Actual layout will become { border: 1, left: 115, bottom: 0, top: 0, width: 50 }
 94             layout: { border: 1, width: 50 } // Don't need to specify layout.left, this is automatic.
 95           }),
 96 
 97           sectionC: SC.View.design({
 98             // Actual layout will become { left: 170, top: 10, bottom: 10, width: 120 }
 99             layout: { top: 10, bottom: 10, width: 120 } // Don't need to specify layout.left, this is automatic.
100           })
101 
102         });
103 
104     If `resizeToFit` is set to `false`, the view will not adjust itself to fit
105     its child views.  This means that when `resizeToFit` is false, the view should
106     specify its width component in its layout.  This also means that you can
107     ignore the last child view's layout width if you want the last child view
108     to stretch to fill the parent view.
109 
110     For example,
111 
112         MyApp.MyView = SC.View.extend({
113 
114           // Child views will be stacked in order horizontally.
115           childViewLayout: SC.View.HORIZONTAL_STACK,
116 
117           // Change the behavior of the HORIZONTAL_STACK plugin
118           childViewLayoutOptions: {
119             paddingBefore: 10,
120             paddingAfter: 20,
121             spacing: 5
122           },
123 
124           // The order of child views is important!
125           childViews: ['sectionA', 'sectionB', 'sectionC'],
126 
127           // Actual layout will become { left: 10, bottom: 20, top: 20, width: 500 }
128           layout: { left: 10, bottom: 20, top: 20, width: 500 }, // Need to specify layout.width.
129 
130           sectionA: SC.View.design({
131             // Actual layout will become { left: 10, bottom: 0, top: 0, width: 100 }
132             layout: { width: 100 } // Don't need to specify layout.left, this is automatic.
133           }),
134 
135           sectionB: SC.View.design({
136             // Actual layout will become { border: 1, left: 115, bottom: 0, top: 0, width: 50 }
137             layout: { border: 1, width: 50 } // Don't need to specify layout.left, this is automatic.
138           }),
139 
140           sectionC: SC.View.design({
141             // Actual layout will become { left: 170, top: 10, bottom: 10, right: 20 }
142             layout: { top: 10, bottom: 10 } // Don't need to specify layout.left, layout.right or layout.width, this is automatic.
143           })
144 
145         });
146 
147     ## Modify specific child view layouts
148 
149     To adjust the child layout on a granular level per child view, you can
150     also set the following properties on each child view:
151 
152       - marginBefore - Specify the minimum spacing above the child view.
153       - marginAfter - Specify the minimum spacing below the child view.
154       - useAbsoluteLayout - Don't include this child view in automatic layout, use absolute positioning based on the child view's `layout` property.
155       - useStaticLayout - Don't include this child view in automatic layout.  This child view uses relative positioning and is not eligible for automatic layout.
156       - isVisible - Non-visible child views are not included in the stack.
157 
158       For example,
159 
160         MyApp.MyView = SC.View.extend({
161 
162           // Child views will be stacked in order horizontally.
163           childViewLayout: SC.View.HORIZONTAL_STACK,
164 
165           // Actual layout will become { left: 10, right: 10, top: 20, width: 570 }
166           layout: { left: 10, right: 10, top: 20 },
167 
168           // Keep the child views ordered!
169           childViews: ['sectionA', 'float', 'sectionB', 'sectionC'],
170 
171           sectionA: SC.View.design({
172             // Actual layout will become { left: 0, right: 50, top: 0, width: 100 }
173             layout: { right: 50, width: 100 },
174             // The following child view will be at least 50px further right.
175             marginAfter: 50
176           }),
177 
178           float: SC.View.design({
179             // This view will not be included in automatic layout and will not effect the stack.
180             layout: { top: 5, right: 5, height: 50, width: 50 },
181             useAbsoluteLayout: true
182           }),
183 
184           sectionB: SC.View.design({
185             // Actual layout will become { left: 1500, right: 0, top: 0, width: 120 }
186             layout: { width: 120 }
187           }),
188 
189           sectionC: SC.View.design({
190             // Actual layout will become { left: 470, bottom: 0, top: 0, width: 100 }
191             layout: { width: 100 },
192             // This child view will be at least 200px to the right of the previous.
193             marginBefore: 200
194           })
195 
196         });
197 
198     ### A Note About Spacing
199 
200     Note that the spacing attribute in `childViewLayoutOptions` becomes the
201     _minimum margin between child views, without explicitly overriding it from
202     both sides using `marginAfter` and `marginBefore`_.  For example, if `spacing`
203     is 25, setting `marginAfter` to 10 on a child view will not result in the
204     next child view being 10px to the right of it, unless the next child view also
205     specified `marginBefore` as 10.
206 
207     What this means is that it takes less configuration if you set `spacing` to
208     be the _smallest margin you wish to exist between child views_ and then use
209     the overrides to grow the margin if necessary.  For example, if `spacing`
210     is 5, setting `marginAfter` to 10 on a child view will result in the next
211     child view being 10px to the right of it, without having to also specify
212     `marginBefore` on that next child view.
213 
214     @extends SC.ChildViewLayoutProtocol
215     @since Version 1.10
216   */
217   HORIZONTAL_STACK: {
218 
219     /** @private Properties to observe on child views that affect the overall child view layout. */
220     childLayoutProperties: ['marginBefore', 'marginAfter', 'isVisible'],
221 
222     /** @private */
223     layoutChildViews: function (view) {
224       var childViews = view.get('childViews'),
225         options = view.get('childViewLayoutOptions') || {},
226         resizeToFit = SC.none(options.resizeToFit) ? true : options.resizeToFit,
227         lastMargin = 0, // Used to avoid adding spacing to the final margin.
228         marginAfter = options.paddingBefore || 0,
229         paddingAfter = options.paddingAfter || 0,
230         position = 0,
231         spacing = options.spacing || 0,
232         i, len;
233 
234       for (i = 0, len = childViews.get('length'); i < len; i++) {
235         var childView = childViews.objectAt(i),
236           layout,
237           marginBefore;
238 
239         // Ignore child views with useAbsoluteLayout true, useStaticLayout true or that are not visible.
240         if (!childView.get('isVisible') ||
241           childView.get('useAbsoluteLayout') ||
242           childView.get('useStaticLayout')) {
243           continue;
244         }
245 
246         layout = childView.get('layout');
247         //@if(debug)
248         // Add some developer support.
249         if (SC.none(layout.width) && (i < len - 1 || resizeToFit)) {
250           SC.warn('Developer Warning: The SC.View.HORIZONTAL_STACK plugin requires that each childView layout contains at least a width and optionally also top and bottom, top and height, bottom and height or centerY and height.  The childView %@ has an invalid layout: %@'.fmt(childView, SC.stringFromLayout(layout)));
251           return;
252         }
253         //@endif
254 
255         // Determine the left margin.
256         marginBefore = childView.get('marginBefore') || 0;
257         position += Math.max(marginAfter, marginBefore);
258 
259         if (layout.left !== position) {
260           childView.adjust('left', position);
261 
262           // Allow the last child view to stretch.
263           if (!resizeToFit && !layout.width && !layout.right) {
264             childView.adjust('right', paddingAfter);
265           }
266         }
267         position += childView.getPath('borderFrame.width');
268 
269         // Determine the right margin.
270         lastMargin = childView.get('marginAfter') || 0;
271         marginAfter = lastMargin || spacing;
272       }
273 
274       // Adjust our frame to fit as well, this ensures that scrolling works.
275       position += Math.max(lastMargin, paddingAfter);
276       if (resizeToFit && view.getPath('layout.width') !== position) {
277         view.adjust('width', position);
278       }
279     }
280 
281   }
282 
283 });
