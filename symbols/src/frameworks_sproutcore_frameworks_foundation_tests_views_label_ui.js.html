 10 var iconURL= "http://www.freeiconsweb.com/Icons/16x16_people_icons/People_046.gif";
 11 (function() {
 12 var pane = SC.ControlTestPane.design()
 13   .add("basic", SC.LabelView, {
 14     value:'hello'
 15   })
 16 
 17   .add("disabled", SC.LabelView, {
 18     value:'hello',
 19     isEnabled: NO
 20   })
 21 
 22   .add("hint", SC.LabelView, {
 23     hint: 'Get on with it!',
 24     isEditable: true
 25   })
 26 
 27   .add("selectable", SC.LabelView, {
 28     value:'hello',
 29     isTextSelectable: YES
 30   })
 31 
 32   .add("iconclass", SC.LabelView, {
 33     icon: 'icon-class',
 34     value: "hello"
 35   })
 36 
 37   .add("icon", SC.LabelView, {
 38      value: "hello",
 39      icon: iconURL
 40   })
 41 
 42   .add("regular size", SC.LabelView, {
 43      value: "hello",
 44      controlSize: SC.REGULAR_CONTROL_SIZE
 45   })
 46 
 47   .add("small size", SC.LabelView, {
 48      value: "hello",
 49      controlSize: SC.SMALL_CONTROL_SIZE
 50   })
 51 
 52   .add("tiny size", SC.LabelView, {
 53      value: "hello",
 54      controlSize: SC.TINY_CONTROL_SIZE
 55   })
 56 
 57   .add("editable", SC.LabelView, {
 58      value: "double click me",
 59      isEditable: YES
 60   })
 61 
 62   .add("null value", SC.LabelView, {
 63      value: null
 64   })
 65 
 66   .add("undefined value", SC.LabelView, {
 67      value: undefined
 68   });
 69 
 70 
 71 module('SC.LabelView ui', {
 72   setup: function() {
 73     htmlbody('<style> .sc-static-layout { border: 1px red dotted; } </style>');
 74     pane.standardSetup().setup();
 75   },
 76   teardown: function(){
 77     pane.standardSetup().teardown();
 78     clearHtmlbody();
 79   }
 80 });
 81 
 82 test("Check that all Label are visible", function() {
 83   ok(pane.view('basic').get('isVisibleInWindow'), 'basic.isVisibleInWindow should be YES');
 84   ok(pane.view('disabled').get('isVisibleInWindow'), 'title.isVisibleInWindow should be YES');
 85   ok(pane.view('selectable').get('isVisibleInWindow'), 'icon.isVisibleInWindow should be YES');
 86   ok(pane.view('icon').get('isVisibleInWindow'), 'title,icon,disabled.isVisibleInWindow should be YES');
 87   ok(pane.view('regular size').get('isVisibleInWindow'), 'title,icon,default.isVisibleInWindow should be YES');
 88   ok(pane.view('small size').get('isVisibleInWindow'), 'title.icon,selected.isVisibleInWindow should be YES');
 89   ok(pane.view('tiny size').get('isVisibleInWindow'), 'title,toolTip.isVisibleInWindow should be YES');
 90   ok(pane.view('editable').get('isVisibleInWindow'), 'title,toolTip.isVisibleInWindow should be YES');
 91   ok(pane.view('null value').get('isVisibleInWindow'), 'null value.isVisibleInWindow should be YES');
 92 });
 93 
 94 
 95 test("Check that all labels have the right classes and styles set", function() {
 96   var viewElem=pane.view('basic').$();
 97   ok(viewElem.hasClass('sc-view'), 'basic.hasClass(sc-view) should be YES');
 98   ok(viewElem.hasClass('sc-label-view'), 'basic.hasClass(sc-label-view) should be YES');
 99   ok(!viewElem.hasClass('icon'), 'basic.hasClass(icon) should be NO');
100   ok(!viewElem.hasClass('disabled'), 'basic.hasClass(disabled) should be YES');
101 
102   viewElem=pane.view('disabled').$();
103   ok(viewElem.hasClass('sc-view'), 'title.hasClass(sc-view) should be YES');
104   ok(viewElem.hasClass('sc-label-view'), 'title.hasClass(sc-label-view) should be YES');
105   ok(!viewElem.hasClass('icon'), 'title.hasClass(icon) should be NO');
106   ok(viewElem.hasClass('disabled'), 'title.hasClass(disabled) should be NO');
107 
108   viewElem=pane.view('selectable').$();
109   ok(viewElem.hasClass('sc-view'), 'icon.hasClass(sc-view) should be YES');
110   ok(viewElem.hasClass('sc-label-view'), 'icon.hasClass(sc-label-view) should be YES');
111   ok(viewElem.hasClass('sc-regular-size'), 'icon.hasClass(sc-regular-size) should be YES');
112   ok(!viewElem.hasClass('icon'), 'icon.hasClass(icon) should be YES');
113   ok(!viewElem.hasClass('sel'), 'icon.hasClass(sel) should be NO');
114   ok(!viewElem.hasClass('disabled'), 'icon.hasClass(disabled) should be NO');
115 
116   viewElem = pane.view('iconclass').$();
117   ok(viewElem.hasClass('icon'), 'view element should have "icon" class');
118   ok(viewElem.find('img').hasClass('icon'), 'img element inside view should have "icon" class');
119 
120   viewElem=pane.view('icon').$();
121   ok(viewElem.hasClass('sc-view'), 'title,icon,disabled.hasClass(sc-view) should be YES');
122   ok(viewElem.hasClass('sc-label-view'), 'title,icon,disabled.hasClass(sc-label-view) should be YES');
123   ok(viewElem.hasClass('icon'), 'title,icon,disabled.hasClass(icon) should be YES');
124   ok(!viewElem.hasClass('disabled'), 'title,icon,disabled.hasClass(disabled) should be YES');
125 
126   viewElem=pane.view('regular size').$();
127   ok(viewElem.hasClass('sc-view'), 'title,icon,default.hasClass(sc-view) should be YES');
128   ok(viewElem.hasClass('sc-label-view'), 'title,icon,default.hasClass(sc-label-view) should be YES');
129   ok(viewElem.hasClass('sc-regular-size'), 'title,icon,default.hasClass(sc-regular-size) should be YES');
130   ok(!viewElem.hasClass('disabled'), 'title,icon,default.hasClass(disabled) should be NO');
131 });

134 test("Check that the title is set or not and if it is in the appropriate element", function() {
135   var viewElem=pane.view('basic').$();
136   equals(viewElem.text(), 'hello', 'has a value set');
137 
138   viewElem=pane.view('icon').$('img');
139   ok((viewElem!==null), 'should have an image corresponding to an icon');
140 
141   viewElem=pane.view('null value').$();
142   equals(viewElem.text(), '', 'has correct empty value set');
143 
144   viewElem=pane.view('undefined value').$();
145   equals(viewElem.text(), '', 'has correct empty value set');
146 });
147 
148 test("The hint property should appear if the label is editable and has no value.", function () {
149   var viewElem = pane.view('hint').$();
150 
151   viewElem = viewElem.find('.sc-hint');
152   equals(viewElem.length, 1, "has an .sc-hint span inside");
153   equals(viewElem.text(), 'Get on with it!', 'has correct hint value set');
154 });
