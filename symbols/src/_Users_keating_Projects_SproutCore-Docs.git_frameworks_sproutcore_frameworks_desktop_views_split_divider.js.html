  9 /**
 10   @class
 11 
 12   A SplitDividerView displays a divider between two views within a SplitView.
 13   Clicking and dragging the divider will change the thickness of each view
 14   either to the top/left or bottom/right of the divider.
 15 
 16   Double-clicking on the SplitDividerView will try to collapse the first
 17   view within the SplitView that has property canCollapse set to true,
 18   so it is not visible, unless you have canCollapse disabled on the SplitView.
 19 
 20   This view must be a direct child of the split view it works with. It must
 21   be surrounded by two other views.
 22 
 23   @extends SC.View
 24   @author Charles Jolley
 25   @author Lawrence Pit
 26   @author Erich Ocean
 27   @test in split
 28 */
 29 SC.SplitDividerView = SC.View.extend(
 30 /** @scope SC.SplitDividerView.prototype */ {
 31 
 32   /**
 33     @type Array
 34     @default ['sc-split-divider-view']
 35     @see SC.View#classNames
 36   */
 37   classNames: ['sc-split-divider-view'],
 38   
 39   /** @private */
 40 
 41   /** @private */
 42   mouseDown: function(evt) {
 43     var splitView = this.get('splitView');
 44     return (splitView) ? splitView.mouseDownInThumbView(evt, this) : sc_super();
 45   },
 46   
 47   /** @private */
 48   doubleClick: function(evt) {
 49     var splitView = this.get('splitView');
 50     return (splitView) ? splitView.doubleClickInThumbView(evt, this) : sc_super();
 51   },
 52   
 53   /** @private */
 54   touchStart: function(evt){
 55     return this.mouseDown(evt);
 56   }
 57   
 58 });
