/** @class
 11   This is a basic designer used for all `SC.Object`s that are created in
 12   design mode.
 13   
 14   FIXME: have `SC.ViewDesigner` subclass this designer.....
 15 
 16   @extends SC.Object
 17   @since SproutCore 1.0
 18 */
 19 SC.ObjectDesigner = SC.Object.extend(
 20 /** @scope SC.ViewDesigner.prototype */ {
 21 
 22   /** The object managed by this designer. */
 23   object: null,
 24   
 25   /** The class for the design.  Set when the object is created. */
 26   objectClass: null,
 27   
 28   /** Set to `YES` if the object is currently selected for editing. */
 29   designIsSelected: NO,
 30 
 31   /** Set to `YES` if this particular designer should not be enabled. */
 32   designIsEnabled: YES,
 33   
 34   /**
 35     The current page.  Comes from the object.
 36     
 37     @property {SC.Page}
 38   */
 39   page: function() {
 40     var v = this.get('object');
 41     return (v) ? v.get('page') : null;
 42   }.property('object').cacheable(),
 43   
 44   /**
 45     The design controller from the page.  Comes from page
 46     
 47     @property {SC.PageDesignController}
 48   */
 49   designController: function() {
 50     var p = this.get('page');
 51     return (p) ? p.get('designController') : null ;  
 52   }.property('page').cacheable(),
 53   
 54   
 55   concatenatedProperties: ['designProperties', 'localizedProperties', 'excludeProperties'],
 56 
 57   // ..........................................................
 58   // GENERIC PROPERTIES
 59   // 
 60   // Adds support for adding generic properties to a object.  These will
 61   // overwrite whatever you write out using specifically supported props.
 62     
 63   // ..........................................................
 64   // HANDLE ENCODING OF VIEW DESIGN
 65   // 
 66 
 67   /**
 68     Encodes any simple properties that can just be copied from the object onto
 69     the coder.  This is used by encodeDesignProperties() and 
 70     encodeLocalizedProperties().
 71   */
 72   encodeSimpleProperties: function(props, coder) {
 73     var object = this.get('object'), proto = this.get('objectClass').prototype ;
 74     props.forEach(function(prop) {
 75       var val = object[prop] ; // avoid get() since we don't want to exec props
 76       if (val !== undefined && (val !== proto[prop])) {
 77         coder.encode(prop, val) ;
 78       }
 79     }, this);
 80   },
 81   
 82 
 83   /** 
 84     Array of properties that can be encoded directly.  This is an easy way to
 85     add support for simple properties that need to be written to the design
 86     without added code.  These properties will be encoded by 
 87     `encodeDesignProperties()`.
 88     
 89     You can add to this array in your subclasses.
 90   */
 91   designProperties: [],
 92   
 93   /*
 94     Array of properties specifically not displayed in the editable properties
 95     list
 96   */
 97   
 98   excludeProperties: [],
 99   
100   
101   /*
102     Array of properties avaliaible to edit in greenhouse
103     
104   */
105   editableProperties: function(){
106 
107     var con = this.get('designAttrs'), 
108         obj = this.get('object'),
109         ret = [],
110         designProperties = this.get('designProperties'),
111         excludeProperties = this.get('excludeProperties');
112     if(con) con = con[0];
113     for(var i in con){
114       if(con.hasOwnProperty(i) && excludeProperties.indexOf(i) < 0){
115         if(!SC.none(obj[i])) ret.pushObject(SC.Object.create({value: obj[i], key: i, view: obj}));
116       }
117     }
118     designProperties.forEach(function(k){
119       if(excludeProperties.indexOf(k) < 0){
120         ret.pushObject(SC.Object.create({value: obj[k], key: k, view: obj}));
121       }
122 123     });
124     
125     return ret; 
126   }.property('designProperties').cacheable(),
127   
128   /** 
129     Invoked by a design coder to encode design properties.  The default 
130     implementation invoked `encodeDesignProperties()` and
131     `encodeChildViewsDesign()`.  You can override this method with your own
132     additional encoding if you like.
133   */
134   encodeDesign: function(coder) {
135     coder.set('className', SC._object_className(this.get('objectClass')));
136     this.encodeDesignProperties(coder);
137     return YES ;
138   },
139 
140   /**
141     Encodes the design properties for the object.  These properties are simply
142     copied from the object onto the coder.  As an optimization, the value of 
143     each property will be checked against the default value in the class. If
144     they match, the property will not be emitted.
145   */
146   encodeDesignProperties: function(coder) {
147     return this.encodeSimpleProperties(this.get('designProperties'), coder);
148   },
149   
150   /** 
151     Array of localized that can be encoded directly.  This is an easy way to
152     add support for simple properties that need to be written to the 
153     localization without added code.  These properties will be encoded by 
154     `encodeLocalizedProperties()`.
155     
156     You can add to this array in your subclasses.
157   */
158   localizedProperties: [],
159   
160   /** 
161     Invoked by a localization coder to encode design properties.  The default 
162     implementation invoked `encodeLocalizedProperties()` and
163     `encodeChildViewsLoc()`.  You can override this method with your own
164     additional encoding if you like.
165   */
166   encodeLoc: function(coder) {
167     coder.set('className', SC._object_className(this.get('objectClass')));
168     this.encodeLocalizedProperties(coder);
169     return YES ;
170   },
171 
172   /**
173     Encodes the localized properties for the object.  These properties are 
174     simply copied from the object onto the coder.  As an optimization, the value 
175     of  each property will be checked against the default value in the class. 
176     If they match, the property will not be emitted.
177   */
178   encodeLocalizedProperties: function(coder) {
179     return this.encodeSimpleProperties(this.get('localizedProperties'),coder);
180   },
181 
182   /**
183     This method is invoked when the designer is instantiated.  You can use 
184     this method to reload any state saved in the object.  This method is called
185     before any observers or bindings are setup to give you a chance to 
186     configure the initial state of the designer.
187   */
188   awakeDesign: function() {},
189   
190   /**
191     The `unknownProperty` handler will pass through to the object by default.
192     This will often provide you the support you need without needing to 
193     customize the Designer.  Just make sure you don't define a conflicting
194     property name on the designer itself!
195   */
196   unknownProperty: function(key, value) {
197     if (value !== undefined) {
198       this.object.set(key, value);
199       return value ;
200     } else return this.object.get(key);
201   },
202   
203   // ......................................
204   // PRIVATE METHODS
205   //
206   
207   init: function() {
208     
209     // setup design from object state...
210     this.awakeDesign();
211     
212     // setup bindings, etc
213     sc_super();
214         
215     // and register with designController, if defined...
216     var c= this.get('designController');
217     if (c) c.registerDesigner(this) ;
218     
219   },
220 
221   destroy: function() {
222     sc_super();
223     this.set('object', null); // clears the object observer...  
224   },
225     
226   tryToPerform: function(methodName, arg1, arg2) {
227     // only handle event if we are in design mode
228     var page = this.object ? this.object.get('page') : null ;
229     var isDesignMode = page ? page.get('needsDesigner') || page.get('isDesignMode') : NO ;
230 
231     // if we are in design mode, route event handling to the designer
232     // otherwise, invoke default method.
233     if (isDesignMode) {
234       return sc_super();
235     } else {
236       return SC.Object.prototype.tryToPerform.apply(this.object, arguments);
237     }
238   }
239 }) ;
240 
241 // Set default Designer for object
242 if (!SC.Object.Designer) SC.Object.Designer = SC.ObjectDesigner ;
243 
244 // ..........................................................
245 // DESIGN NOTIFICATION METHODS
246 //
247 // These methods are invoked automatically on the designer class whenever it 
248 // is loaded.
249 
250 SC.ObjectDesigner.mixin({
251   /**
252     Invoked whenever a designed object is loaded.  This will save the design
253     attributes for later use by a designer.
254   */
255   didLoadDesign: function(designedObject, sourceObject, attrs) {
256     designedObject.isDesign = YES ; // indicates that we need a designer.
257     designedObject.designAttrs = attrs;
258     //designedObject.sourceObject = sourceObject; TODO: don't need this..
259   },
260 
261   /**
262     Invoked whenever a location is applied to a designed object.  Saves the 
263     attributes separately for use by the design object.
264   */
265   didLoadLocalization: function(designedObject, attrs) {
266     // nothing to do for now.
267   },
268   
269   /**
270     Invoked whenver a object is created.  This will create a peer designer if 
271     needed.
272   */
273   didCreateObject: function(object, attrs) {
274     // add designer if page is in design mode
275     var page = object.get('page'), design = object.constructor;
276     
277     if (design.isDesign && page && page.get('needsDesigner')) {
278       
279       // find the designer class
280       var cur = design, origDesign = design;
281       while(cur && !cur.Designer) cur = cur.superclass;
282       var DesignerClass = (cur) ? cur.Designer : SC.Object.Designer;
283       
284       // next find the first superclass object that is not a design (and a real
285       // class).  This is important to make sure that we can determine the 
286       // real name of a object's class.
287       while (design && design.isDesign) design = design.superclass;
288       if (!design) design = SC.Object;
289       
290       object.designer = DesignerClass.create({
291         object: object,
292         objectClass: design,
293         designAttrs: origDesign.designAttrs
294         //sourceObject: origDesign.sourceObject TODO: don't need this
295       });
296     }
297   }
298   
299 });
