 10 var CommentView = SC.View.extend(SC.StaticLayout, SC.Control, {
 11   
 12   useStaticLayout: YES,
 13   
 14   content: null,
 15   
 16   classNames: 'comment-view',
 17 
 18   contentKeys: {
 19     'contentFromKey': 'from',
 20     'contentDateKey': 'date',
 21     'contentCommentKey': 'comment'
 22   },
 23 
 24   contentFromKey: 'from',
 25 
 26   contentDateKey: 'date',
 27 
 28   contentCommentKey: 'comment',
 29 
 30   contentPropertyDidChange: function(target, key) {
 31 
 32     // update everything!
 33     if (key === '*') {      
 34       this.updateFromLabel().updateCommentLabel();
 35     } else if (key === 'from' || key === 'date') {
 36       this.updateFromLabel();
 37     } else if (key === 'comment') {
 38       this.updateCommentLabel();
 39     }
 40     
 41     if (this.owner && this.owner.updateHeight) this.owner.updateHeight();
 42   },
 43   
 44   updateFromLabel: function() {
 45     var content = this.get('content'),
 46         from    = content ? content.get('from') : 'Anonymous',
 47         date    = content ? content.get('date') : 'some date',
 48         str     = "%@ wrote %@: ".fmt(from, date);
 49     this.fromLabel.set('value', str);
 50     return this ;
 51   },
 52   
 53   updateCommentLabel: function() {
 54     var content = this.get('content'),
 55         comment = content ? content.get('comment') : '(No Comment)';
 56     this.commentLabel.set('value', comment);
 57   },
 58 
 59   // ..........................................................
 60   // BASE VIEWS
 61   // 
 62 
 63   childViews: 'fromLabel commentLabel editButton replyButton'.w(),
 64   
 65   fromLabel: SC.LabelView.extend({
 66     tagName: "label",
 67     layout: { left: 0, top: 0, right: 0, height: 18 }
 68   }),
 69   
 70   commentLabel: SC.LabelView.extend(SC.StaticLayout, {
 71     useStaticLayout: YES,
 72     tagName: "p"
 73   }),
 74 
 75   editButton: SC.ButtonView.extend({
 76     title: "Edit",
 77     titleMinWidth: 0,
 78     layout: { bottom: 4, right: 58, height: 21, width: 50 }
 79   }),
 80 
 81   replyButton: SC.ButtonView.extend({
 82     title: "Reply",
 83     titleMinWidth: 0,
 84     layout: { bottom: 4, right: 4, height: 21, width: 50 }
 85   })
 86   
 87 });

152 test("removing an item should delete childView and adjust height", function() {
153   var view = pane.view('basic'),
154       item = content[0];
155       
156   equals(view.getPath('childViews.length'), content.length, 'precond - should have child views equal to current content');
157   var oldHeight = view.get('frame').height; // save height.
158   
159   SC.run(function() { content.removeAt(0); }); // remove first item
160 
161 
162   equals(view.getPath('childViews.length'), content.length, 'view should remove childView for removed content items');
163   var newHeight = view.get('frame').height; 
164   ok(newHeight < oldHeight, 'view height should adjust to reflect new content. (old height: %@ current height: %@)'.fmt(oldHeight, newHeight));
165   
166   // restore old content
167   SC.run(function() { content.insertAt(0, item); });
168   
169 });

173 test("inserting an item should add childView and adjust height", function() {
174   var view = pane.view('basic'),
175       item = extra; // we will insert another one
176       
177   equals(view.getPath('childViews.length'), content.length, 'precond - should have child views equal to current content');
178   var oldHeight = view.get('frame').height; // save height.
179   
180   SC.run(function() { content.pushObject(item); }); // add another item
181 
182 
183   equals(view.getPath('childViews.length'), content.length, 'view should add childView for added content item');
184   var newHeight = view.get('frame').height; 
185   ok(newHeight > oldHeight, 'view height should adjust to reflect new content. (old height: %@ current height: %@)'.fmt(oldHeight, newHeight));
186 
187   // restore
188   SC.run(function() { content.popObject(); });
189 });

192 test("editing an item should automatically adjust the height", function() {
193   var view = pane.view('basic'),
194       item = content[0],
195       childView = view.childViews[0], 
196       old = item.get('comment');
197   
198   ok(childView, 'stacked view should have child view');
199   equals(childView.get('content'), item, 'first childView should own first content item');
200 
201   var height = view.get('frame').height;  // save old height
202 
203   // change comment
204   SC.run(function() { item.set('comment', 'This is a new comment'); });
205   
206   // should have updated UI and adjusted height of collection
207   equals(childView.$().find('p').text(), 'This is a new comment', 'Item view should now contain comment');
208   
209   var newHeight = view.get('frame').height;
210   ok(newHeight < height, 'view height should adjust to reflect new content. (old height: %@ current height: %@)'.fmt(height, newHeight));
211 
212   // restore
213   SC.run(function() { item.set('comment', old); });
214   
215   newHeight = view.get('frame').height;
216   equals(newHeight, height, 'view height should restore to old height when content is edited again. (old height: %@ current height: %@)'.fmt(height, newHeight));
217   
218 });

232       
233   SC.run(function() { content.removeAt(0); });
234   SC.run(function() { content.insertAt(0, item); });
235   SC.run(function() { content.pushObject(extra); });
236   SC.run(function() { content.popObject(); });
237   
238   SC.run(function() { item.set('comment', 'Short comment'); });
239   SC.run(function() { item.set('comment', old); });
240   
241   var newHeight = view.get('frame').height;
242   equals(newHeight, height, 'view height should restore to old height when content is edited again. (old height: %@ current height: %@)'.fmt(height, newHeight));
243   
244 });
