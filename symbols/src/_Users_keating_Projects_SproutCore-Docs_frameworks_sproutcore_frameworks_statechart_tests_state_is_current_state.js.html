1 // ==========================================================================
  2 // SC.Statechart Unit Test
  3 // ==========================================================================
  4 /*globals SC */
  5 
  6 var statechart = null;
  7 
  8 module("SC.Statechart: State - isCurrentState Property Tests", {
  9   setup: function() {
 10 
 11     statechart = SC.Statechart.create({
 12       
 13       monitorIsActive: YES,
 14       
 15       rootState: SC.State.design({
 16         
 17         initialSubstate: 'a',
 18         
 19         a: SC.State.design(),
 20         
 21         b: SC.State.design()
 22         
 23       })
 24       
 25     });
 26     
 27     statechart.initStatechart();
 28   },
 29   
 30   teardown: function() {
 31     statechart.destroy();
 32     statechart = null;
 33   }
 34 });
 35 
 36 test("check binding to isCurrentState", function() {
 37   var a = statechart.getState('a');
 38 
 39   var o = SC.Object.create({
 40     value: null,
 41     valueBinding: SC.Binding.oneWay().from('isCurrentState', a)
 42   });
 43   
 44   SC.run();
 45   equals(a.get('isCurrentState'), true);
 46   equals(o.get('value'), true);
 47   
 48   SC.run(function() { statechart.gotoState('b'); });
 49   equals(a.get('isCurrentState'), false);
 50   equals(o.get('value'), false);
 51   
 52   SC.run(function() { statechart.gotoState('a'); });
 53   equals(a.get('isCurrentState'), true);
 54   equals(o.get('value'), true);
 55   
 56   SC.run(function() { statechart.gotoState('b'); });
 57   equals(a.get('isCurrentState'), false);
 58   equals(o.get('value'), false);
 59 
 60 });
