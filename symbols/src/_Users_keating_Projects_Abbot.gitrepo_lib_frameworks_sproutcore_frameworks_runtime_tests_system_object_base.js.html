 14 module("A new SC.Object instance", {
 15   
 16   setup: function() {
 17     obj = SC.Object.create({
 18       foo: "bar",
 19       total: 12345,
 20       aMethodThatExists: function() {},
 21       aMethodThatReturnsTrue: function() { return true; },
 22       aMethodThatReturnsFoobar: function() { return "Foobar"; },
 23       aMethodThatReturnsFalse: function() { return NO; }
 24     });
 25   },
 26   
 27   teardown: function() {
 28     obj = undefined ;
 29   }
 30   
 31 });
 32 
 33 test("Should identify it's methods using the 'respondsTo' method", function() {
 34   equals(obj.respondsTo('aMethodThatExists'), true) ;
 35   equals(obj.respondsTo('aMethodThatDoesNotExist'), false) ;
 36 });
 37 
 38 test("Should return false when asked to perform a method it does not have", function() {
 39   equals(obj.tryToPerform('aMethodThatDoesNotExist'), false) ;
 40 });
 41 
 42 test("Should pass back the return YES if method returned YES, NO if method not implemented or returned NO", function() {
 43   equals(obj.tryToPerform('aMethodThatReturnsTrue'), YES, 'method that returns YES') ;
 44   equals(obj.tryToPerform('aMethodThatReturnsFoobar'), YES, 'method that returns non-NO') ;
 45   equals(obj.tryToPerform('aMethodThatReturnsFalse'), NO, 'method that returns NO') ;
 46   equals(obj.tryToPerform('imaginaryMethod'), NO, 'method that is not implemented') ;
 47 });
 48 
 49 test("Should return it's properties when requested using SC.Object#get", function() {
 50   equals(obj.get('foo'), 'bar') ;
 51   equals(obj.get('total'), 12345) ;
 52 });
 53 
 54 test("Should allow changing of those properties by calling SC.Object#set", function() {
 55   equals(obj.get('foo'), 'bar') ;
 56   equals(obj.get('total'), 12345) ;
 57   
 58   obj.set( 'foo', 'Chunky Bacon' ) ;
 59   obj.set( 'total', 12 ) ;
 60   
 61   equals(obj.get('foo'), 'Chunky Bacon') ;
 62   equals(obj.get('total'), 12) ;
 63 });
 64 
 65 test("Should only advertise changes once per request to SC.Object#didChangeFor", function() {
 66   obj.set( 'foo', 'Chunky Bacon' );
 67   equals(obj.didChangeFor( this, 'foo' ), true) ;
 68   equals(obj.didChangeFor( this, 'foo' ), false) ;
 69 });
 70 
 71 test("Should advertise changes once per request to SC.Object#didChangeFor when setting property to NULL", function() {
 72   obj.set( 'foo', null );
 73   equals(obj.didChangeFor( this, 'foo' ), true) ;
 74   equals(obj.didChangeFor( this, 'foo' ), false) ;
 75 });

 77 test("When the object is destroyed the 'isDestroyed' status should change accordingly", function() {
 78 	equals(obj.get('isDestroyed'), NO);
 79 	obj.destroy();
 80 	equals(obj.get('isDestroyed'), YES);

 83 module("SC.Object observers", {
 84   setup: function() {
 85     // create a namespace
 86     TestNamespace = {
 87       obj: SC.Object.create({
 88         value: "test"
 89       })
 90     };
 91     
 92     // create an object
 93     obj = SC.Object.create({
 94       prop1: null,
 95       
 96       // normal observer
 97       observer: function(){
 98         this._normal = YES;
 99       }.observes("prop1"),
100       
101       globalObserver: function() {
102         this._global = YES;
103       }.observes("TestNamespace.obj.value"),
104       
105       bothObserver: function() {
106         this._both = YES;
107       }.observes("prop1", "TestNamespace.obj.value")
108     });
109     
110   }
111 });
112 
113 test("Local observers work", function() {
114   obj._normal = NO;
115   obj.set("prop1", NO);
116   equals(obj._normal, YES, "Normal observer did change.");
117 });
118 
119 test("Global observers work", function() {
120   obj._global = NO;
121   TestNamespace.obj.set("value", "test2");
122   equals(obj._global, YES, "Global observer did change.");
123 });
124 
125 test("Global+Local observer works", function() {
126   obj._both = NO;
127   obj.set("prop1", NO);
128   equals(obj._both, YES, "Both observer did change.");
129 });
