  7 /*global module, test, equals */
  8 
  9 var family, grandma, momma, child1, child2, child3, child4, child5, observerFiredCount, observer2FiredCount;
 10 module("SC.Observable - Observing with @each", {
 11   setup: function () {
 12     momma = SC.Object.create({ children: [] });
 13     grandma = SC.Object.create({ momma: momma });
 14     family = SC.Object.create({
 15       grandma: grandma,
 16       eachCallback: function () {
 17         observerFiredCount++;
 18       },
 19       childrenCallback: function () {
 20         observer2FiredCount++;
 21       }
 22     });
 23 
 24     child1 = SC.Object.create({ name: "Bartholomew" });
 25     child2 = SC.Object.create({ name: "Agnes" });
 26     child3 = SC.Object.create({ name: "Dan" });
 27     child4 = SC.Object.create({ name: "Nancy" });
 28     child5 = SC.Object.create({ name: "Constance" });
 29 
 30     momma.set('children', [child1, child2, child3]);
 31 
 32     observerFiredCount = 0;
 33     observer2FiredCount = 0;
 34   },
 35 
 36   teardown: function () {
 37     family.destroy();
 38     grandma.destroy();
 39     momma.destroy();
 40     child1.destroy();
 41     child2.destroy();
 42     child3.destroy();
 43     child4.destroy();
 44     child5.destroy();
 45 
 46     family = grandma = momma = child1 = child2 = child3 = child4 = child5 = null;
 47   }
 48 });
 49 
 50 test("chained observers on enumerable properties are triggered when the observed property of any item changes", function () {
 51   family.addObserver('grandma.momma.children.@each.name', family, family.eachCallback);
 52   family.addObserver('grandma.momma.children', family, family.childrenCallback);
 53 
 54   observerFiredCount = 0;
 55   SC.run(function () { momma.get('children').setEach('name', 'Juan'); });
 56   equals(observerFiredCount, 3, "observer fired after changing child names");
 57 
 58   observerFiredCount = 0;
 59   SC.run(function () { momma.children.pushObject(child4); });
 60   equals(observerFiredCount, 1, "observer fired after adding a new item");
 61 
 62   observerFiredCount = 0;
 63   SC.run(function () { child4.set('name', "Herbert"); });
 64   equals(observerFiredCount, 1, "observer fired after changing property on new object");
 65 
 66   var oldChildren = momma.get('children');
 67   momma.set('children', []);
 68   observerFiredCount = 0;
 69   SC.run(function () {
 70     oldChildren.pushObject(child5);
 71     oldChildren.objectAt(0).set('name', "Hanna");
 72   });
 73   equals(observerFiredCount, 0, "observer did not fire after changing removed array and property on an object in removed array");
 74   equals(observer2FiredCount, 1, "children observer did fire only once by replacing children with empty array");
 75 
 76   observerFiredCount = 0;
 77   observer2FiredCount = 0;
 78   SC.run(function () { child1.set('name', "Hanna"); });
 79   equals(observerFiredCount, 0, "observer did not fire after changing property on a removed object");
 80 
 81   observerFiredCount = 0;
 82   SC.run(function () { momma.set('children', [child1, child2, child3, child4]); });
 83   equals(observerFiredCount, 0, "@each observer did not fire after replacing children with 4 objects");
 84   equals(observer2FiredCount, 1, "children observer did fire only once after replacing children with 4 objects");
 85 
 86   observerFiredCount = 0;
 87   observer2FiredCount = 0;
 88   SC.run(function () { momma.set('children', []); });
 89   equals(observerFiredCount, 0, "@each observer did not fire after replacing children with empty array");
 90   equals(observer2FiredCount, 1, "children observer did fire only once after replacing children with empty array");
 91 
 92   observerFiredCount = 0;
 93   SC.run(function () { momma.get('children').pushObjects([child1, child2, child3, child4]); });
 94   equals(observerFiredCount, 1, "observer did fire only once after adding 4 objects");
 95 
 96   observerFiredCount = 0;
 97   SC.run(function () { momma.get('children').removeObjects([child1, child2]); });
 98   equals(observerFiredCount, 1, "observer did fire once after removing 2 of 4 objects");
 99 
100   observerFiredCount = 0;
101   SC.run(function () { momma.get('children').removeObjects([child3, child4]); });
102   equals(observerFiredCount, 1, "observer did fire once after removing all objects");
103 
104   // New grandma.
105   observerFiredCount = 0;
106   grandma.destroy();
107   grandma = SC.Object.create({ momma: momma });
108   family.set('grandma', grandma);
109   equals(observerFiredCount, 0, "observer did not fire after replacing dear old grandma");
110 
111   oldChildren = momma.get('children');
112   momma.destroy();
113   momma = SC.Object.create({ children: oldChildren });
114   grandma.set('momma', momma);
115   equals(observerFiredCount, 0, "observer did not fire after replacing dear old momma");
116 
117   // Clean up.
118   family.removeObserver('grandma.momma.children.@each.name', family, family.eachCallback);
119 });

122 test("observer cleanup", function () {
123   family.addObserver('grandma.momma.children.@each.name', family, family.eachCallback);
124   family.removeObserver('grandma.momma.children.@each.name', family, family.eachCallback);
125 
126   // Clean up.
127   equals(family._kvo_observed_keys.length, 0, "family has no observed keys");
128   equals(grandma._kvo_observed_keys.length, 0, "grandma has no observed keys");
129   equals(momma._kvo_observed_keys.length, 0, "momma has no observed keys: %@".fmt(momma._kvo_observed_keys.toArray()));
130   equals(momma.children._kvo_observed_keys.length, 0, "momma.children has no observed keys: %@".fmt(momma.children._kvo_observed_keys.toArray()));
131   equals(child1._kvo_observed_keys.length, 0, "child1 has no observed keys");
132 });

134 test("content observers are removed correctly", function () {
135   var child1 = SC.Object.create({ name: "Bartholomew", age: 15 });
136   var child2 = SC.Object.create({ name: "Agnes", age: 12 });
137   var children = [child1, child2];
138 
139   var observerFiredCount = 0;
140   var observerFunc = function () { observerFiredCount++; };
141 
142   children.addObserver('@each.name', this, observerFunc);
143   children.removeObserver('@each.name', this, observerFunc);
144   observerFiredCount = 0;
145   SC.run(function () { children.setEach('name', "Hanna"); });
146   equals(observerFiredCount, 0, "name observer did not fire after it was removed");
147 
148   children.addObserver('@each.age', this, observerFunc);
149   children.removeObserver('@each.age', this, observerFunc);
150   observerFiredCount = 0;
151   SC.run(function () { children.setEach('age', 14); });
152   equals(observerFiredCount, 0, "age observer did not fire after it was removed");
153 });
