  7 (function() {
  8   var observerObject, array;
  9 
 10   var willChangeStart, willChangeAdded, willChangeRemoved, willChangeSource, willChangeCallCount;
 11   var didChangeStart, didChangeAdded, didChangeRemoved, didChangeSource, didChangeCallCount;
 12 
 13   module("Enumerable Observers", {
 14     setup: function() {
 15       array = [1, 2, 3];
 16 
 17       observerObject = SC.Object.create({
 18         arrayContentWillChange: function(start, removedCount, addedCount, source) {
 19           willChangeStart = start;
 20           willChangeAdded = addedCount;
 21           willChangeRemoved = removedCount;
 22           willChangeSource = source;
 23           willChangeCallCount = willChangeCallCount ? willChangeCallCount++ : 1;
 24         },
 25 
 26         arrayContentDidChange: function(start, removedCount, addedCount, source) {
 27           didChangeStart = start;
 28           didChangeAdded = addedCount;
 29           didChangeRemoved = removedCount;
 30           didChangeSource = source;
 31           didChangeCallCount = didChangeCallCount ? didChangeCallCount++ : 1;
 32         }
 33       });
 34 
 35       array.addArrayObservers({
 36         target: observerObject,
 37         willChange: 'arrayContentWillChange',
 38         didChange: 'arrayContentDidChange'
 39       });
 40     }
 41   });
 42 
 43   test("should be called when an object is added to an enumerable", function() {
 44     array.pushObject(4);
 45 
 46     equals(willChangeCallCount, 1, "calls arrayContentWillChange once");
 47     equals(didChangeCallCount, 1, "calls arrayContentDidChange once");
 48 
 49     equals(didChangeSource.objectAt(willChangeStart), 4);
 50     equals(didChangeAdded, 1, "specifies one object added");
 51     equals(didChangeRemoved, 0, "specifies no objects removed");
 52   });
 53 
 54   test("should not be called when the observer is removed", function() {
 55     array.pushObject(4);
 56     equals(didChangeCallCount, 1, "precond - observer fires when added");
 57     didChangeCallCount = 0;
 58 
 59     array.removeArrayObservers({
 60       target: observerObject,
 61       willChange: 'arrayContentWillChange',
 62       didChange: 'arrayContentDidChange'
 63     });
 64     array.pushObject(5);
 65     equals(didChangeCallCount, 0, "observer does not fire after being removed");
 66   });
 67 
 68   test("should include both added and removed objects", function() {
 69     array.replace(1, 1, [6, 7, 8]);
 70 
 71     equals(willChangeStart, 1);
 72     equals(willChangeRemoved, 1);
 73     equals(willChangeAdded, 3);
 74     equals(willChangeSource, array);
 75 
 76     equals(didChangeStart, 1);
 77     equals(didChangeRemoved, 1);
 78     equals(didChangeAdded, 3);
 79     equals(didChangeSource, array);
 80   });
 81 })();
