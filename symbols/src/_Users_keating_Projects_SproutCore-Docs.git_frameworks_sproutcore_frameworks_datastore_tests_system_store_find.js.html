  8 // test querying through find() on the store
  9 module("SC.Query querying find() on a store", {
 10   setup: function() {
 11     SC.RunLoop.begin();
 12     // setup dummy app and store
 13     MyApp = SC.Object.create({});
 14 
 15     // setup a dummy model
 16     MyApp.Foo = SC.Record.extend();
 17     MyApp.Bar = SC.Record.extend();
 18 
 19     // setup data source that just returns cached storeKeys
 20     MyApp.DataSource = SC.DataSource.create({
 21 
 22       fetch: function(store, query) {
 23         this.query = query;
 24         this.store = store;
 25         this.fetchCount++ ;
 26 
 27         // used by tests to verify remote queries
 28         if (query.get('location') === SC.Query.REMOTE) {
 29           if (query.get('recordType') === MyApp.Foo) {
 30             store.loadQueryResults(query, this.get('storeKeys'));
 31           }
 32         }
 33 
 34         return YES ;
 35       },
 36 
 37       reset: function() {
 38         this.query = this.store = null ;
 39         this.fetchCount = this.prepareCount = 0 ;
 40       },
 41 
 42       fetchEquals: function(store, query, count, desc) {
 43         if (desc===undefined && typeof count === 'string') {
 44           desc = count;  count = undefined;
 45         }
 46         if (count===undefined) count = 1;
 47 
 48         equals(this.store, store, desc + ': should get store');
 49         equals(this.query, query, desc + ': should get query');
 50         equals(this.fetchCount, count, desc + ': should get count');
 51       },
 52 
 53       destroyRecord: function(store, storeKey){
 54         store.dataSourceDidDestroy(storeKey);
 55         return YES;
 56       }
 57 
 58     });
 59 
 60     MyApp.store = SC.Store.create().from(MyApp.DataSource);
 61 
 62     var records = [
 63       { guid: 1, firstName: "John", lastName: "Doe", married: true },
 64       { guid: 2, firstName: "Jane", lastName: "Doe", married: false },
 65       { guid: 3, firstName: "Emily", lastName: "Parker", bornIn: 1975, married: true },
 66       { guid: 4, firstName: "Johnny", lastName: "Cash", married: true },
 67       { guid: 5, firstName: "Bert", lastName: "Berthold", married: true }
 68     ];
 69  70 
    // load some data
 71     MyApp.DataSource.storeKeys = MyApp.store.loadRecords(MyApp.Foo, records);
 72     SC.RunLoop.end();
 73 
 74     SC.RunLoop.begin();
 75     // for sanity check, load two record types
 76     MyApp.store.loadRecords(MyApp.Bar, records);
 77     SC.RunLoop.end();
 78 
 79   },
 80 
 81   teardown: function() {
 82     MyApp = null ;
 83     SC.Record.subclasses.clear(); //reset
 84   }
 85 
 86 });

132   equals(r2.get('store'), child, 'return object should be owned by child store');
133 
134   ok(r2 !== r1, 'return value for child store should not be same as parent');
135   equals(r3, r2, 'return value from child store should be the same after multiple calls');
136 
137   // check underlying queries
