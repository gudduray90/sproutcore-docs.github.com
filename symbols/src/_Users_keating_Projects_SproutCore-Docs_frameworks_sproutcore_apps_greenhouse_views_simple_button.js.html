 21 Greenhouse.SimpleButton = {
 22 /* SimpleButton Mixin */
 23   target: null,
 24   action: null,
 25   hasState: NO,
 26   hasHover: NO,
 27   inState: NO,
 28   _hover: NO,
 29   stateClass: 'state',
 30   hoverClass: 'hover',
 31   activeClass: 'active', // Used to show the button as being active (pressed)
 32 
 33   _isMouseDown: NO,
 34 
 35   displayProperties: ['inState'],
 36 
 37   /** @private
 38     On mouse down, set active only if enabled.
 39   */
 40   mouseDown: function(evt) {
 41     //console.log('SimpleButton#mouseDown()...');
 42     if (!this.get('isEnabledInPane')) return YES ; // handled event, but do nothing
 43     //this.set('isActive', YES);
 44     this._isMouseDown = YES;
 45     this.displayDidChange();
 46     return YES ;
 47   },
 48 
 49   /** @private
 50     Remove the active class on mouseExited if mouse is down.
 51   */
 52   mouseExited: function(evt) {
 53     //console.log('SimpleButton#mouseExited()...');
 54     if ( this.get('hasHover') ){
 55       this._hover = NO;
 56       this.displayDidChange();
 57     }
 58     //if (this._isMouseDown) this.set('isActive', NO);
 59     return YES;
 60   },
 61 
 62   /** @private
 63     If mouse was down and we renter the button area, set the active state again.
 64   */
 65   mouseEntered: function(evt) {
 66     //console.log('SimpleButton#mouseEntered()...');
 67     if ( this.get('hasHover') ){
 68       this._hover = YES;
 69       this.displayDidChange();
 70     }
 71     //this.set('isActive', this._isMouseDown);
 72     return YES;
 73   },
 74 
 75   /** @private
 76     ON mouse up, trigger the action only if we are enabled and the mouse was released inside of the view.
 77   */
 78   mouseUp: function(evt) {
 79     if (!this.get('isEnabledInPane')) return YES;
 80     //console.log('SimpleButton#mouseUp()...');
 81     //if (this._isMouseDown) this.set('isActive', NO); // track independently in case isEnabled has changed
 82     this._isMouseDown = false;
 83     // Trigger the action
 84     var target = this.get('target') || null;
 85     var action = this.get('action');
 86     // Support inline functions
 87     if (this._hasLegacyActionHandler()) {
 88       // old school...
 89       this._triggerLegacyActionHandler(evt);
 90     } else {
 91       // newer action method + optional target syntax...
 92       this.getPath('pane.rootResponder').sendAction(action, target, this, this.get('pane'));
 93     }
 94     if (this.get('hasState')) {
 95       this.set('inState', !this.get('inState'));
 96     }
 97     this.displayDidChange();
 98     return YES;
 99   },
100 
101   renderMixin: function(context, firstTime) {
102     if (this.get('hasHover')) {
103       var hoverClass = this.get('hoverClass');
104       context.setClass(hoverClass, this._hover && !this._isMouseDown); // addClass if YES, removeClass if NO
105     }
106 
107     if (this.get('hasState')) {
108       var stateClass = this.get('stateClass');
109       context.setClass(stateClass, this.inState); // addClass if YES, removeClass if NO
110     }
111 
112     var activeClass = this.get('activeClass');
113     context.setClass(activeClass, this._isMouseDown);
114 
115     // If there is a toolTip set, grab it and localize if necessary.
116     var toolTip = this.get('toolTip') ;
117     if (SC.typeOf(toolTip) === SC.T_STRING) {
118       if (this.get('localize')) toolTip = toolTip.loc();
119       context.attr('title', toolTip);
120       context.attr('alt', toolTip);
121     }
122   },
123 
124   /**
125     @private
126     From ButtonView
127     Support inline function definitions
128    */
129   _hasLegacyActionHandler: function(){
130     var action = this.get('action');
131     if (action && (SC.typeOf(action) === SC.T_FUNCTION)) return true;
132     if (action && (SC.typeOf(action) === SC.T_STRING) && (action.indexOf('.') !== -1)) return true;
133     return false;
134   },
135 
136   /** @private */
137   _triggerLegacyActionHandler: function(evt){
138     var target = this.get('target');
139     var action = this.get('action');
140 
141     // TODO: [MB/EG] Review: MH added the else if so that the action executes
142     // in the scope of the target, if it is specified.
143     if (target === undefined && SC.typeOf(action) === SC.T_FUNCTION) {
144       this.action(evt);
145     }
146     else if (target !== undefined && SC.typeOf(action) === SC.T_FUNCTION) {
147       action.apply(target, [evt]);
148     }
149 
150     if (SC.typeOf(action) === SC.T_STRING) {
151       eval("this.action = function(e) { return "+ action +"(this, e); };");
152       this.action(evt);
153     }
154   }
155 
156 };
