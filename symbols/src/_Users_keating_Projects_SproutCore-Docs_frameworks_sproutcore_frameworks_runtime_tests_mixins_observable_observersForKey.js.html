 20 test("should get observers", function() {
 21   var o1 = SC.Object.create({ foo: 100 }),
 22       o2 = SC.Object.create({ func: function() {} }),
 23       o3 = SC.Object.create({ func: function() {} }),
 24       observers = null;
 25       
 26   equals(o1.observersForKey('foo').get('length'), 0, "o1.observersForKey should return empty array");
 27   
 28   o1.addObserver('foo', o2, o2.func);
 29   o1.addObserver('foo', o3, o3.func);
 30   
 31   observers = o1.observersForKey('foo');
 32     
 33   equals(observers.get('length'), 2, "o2.observersForKey should return an array with length 2");
 34   equals(observers[0][0], o2, "first item in observers array should be o2");
 35   equals(observers[1][0], o3, "second item in observers array should be o3");
 36 });
