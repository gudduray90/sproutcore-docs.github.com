  7 /** @class
  8 
  9   Describes a target in the build system.
 10 
 11   @extends SC.Record
 12 */
 13 CoreTools.Target = SC.Record.extend(
 14 /** @scope CoreTools.Target.prototype */ {
 15 
 16   primaryKey: "name",
 17   
 18   /**
 19     Name of target.  This is also the primary key.
 20   */
 21   name: SC.Record.attr(String),
 22   
 23   /**
 24     Parent of target.  Only non-null for nested targets.
 25   */
 26   parent: SC.Record.toOne("CoreTools.Target"),
 27 
 28   /**
 29     URL to use to load tests.
 30   */
 31   testsUrl: SC.Record.attr(String, { key: "link_tests" }),
 32   
 33  34   /**  
 35     URL to use to load the app.  If no an app, returns null
 36   */
 37   appUrl: function() {
 38     return (this.get('kind') === 'app') ? CoreTools.attachUrlPrefix(this.get('name')) : null;
 39   }.property('kind', 'name').cacheable(),
 40   
 41   /**
 42     The isExpanded state.  Defaults to NO on load.
 43   */
 44   isExpanded: SC.Record.attr(Boolean, { defaultValue: NO }),
 45   
 46   /**
 47     Children of this target.  Computed by getting the loaded targets
 48   */
 49   children: function() {
 50     var store = this.get('store'),
 51         query = CoreTools.TARGETS_QUERY,
 52         ret   = store.find(query).filterProperty('parent', this);
 53         
 54     if (ret) ret = ret.sortProperty('kind', 'displayName');
 55     return (ret && ret.get('length')>0) ? ret : null ;
 56   }.property().cacheable(),
 57   
 58   /**
 59     Display name for this target
 60   */
 61   displayName: function() {
 62     var name = (this.get('name') || '(unknown)').split('/');
 63     return name[name.length-1];
 64   }.property('name').cacheable(),
 65   
 66   /**
 67     The icon to display.  Based on the type.
 68   */
 69   targetIcon: function() {
 70     var ret = 'sc-icon-document-16';
 71     switch(this.get('kind')) {
 72       case "framework":
 73         ret = 'sc-icon-folder-16';
 74         break;
 75         
 76       case "app":
 77         ret = 'sc-icon-options-16';
 78         break;
 79     }
 80     return ret ;
 81   }.property('kind').cacheable(),
 82   
 83   /**
 84     This is the group key used to display.  Will be the kind unless the item
 85     belongs to the sproutcore target.
 86   */
 87   sortKind: function() {
 88     if (this.get('name') === '/sproutcore') return null;
 89     var parent = this.get('parent');
 90     if (parent && (parent.get('name') === '/sproutcore')) return 'sproutcore';
 91     else return (this.get('kind') || 'unknown').toLowerCase();
 92   }.property('kind', 'parent').cacheable(),
 93   
 94   
 95   testsQuery: function() {
 96     return SC.Query.remote(CoreTools.Test, { url: this.get('testsUrl') });
 97   }.property('testsUrl').cacheable(),
 98   
 99   /**
100     Returns all of the tests associated with this target by fetching the
101     testsUrl.
102   */
103   tests: function() {
104     return this.get('store').find(this.get('testsQuery'));
105   }.property('testsQuery').cacheable()
106   
107 }) ;
108 
109 CoreTools.TARGETS_QUERY = SC.Query.remote(CoreTools.Target);
