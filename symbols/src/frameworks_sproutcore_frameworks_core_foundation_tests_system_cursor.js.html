  5 /*global module, test, htmlbody*/
  6 
  7 module("SC.Cursor", {
  8   setup: function () {
  9     htmlbody('<style title="wrong-style"></style>');
 10   },
 11 
 12   teardown: function () {
 13     clearHtmlbody();
 14   }
 15 });
 16 
 17 /**
 18   There was a bug that if any additional style elements exist in the body
 19   the cursor would create a stylesheet in the head, but then retrieve the
 20   last stylesheet object which would be wrong.
 21 */
 22 test("The cursor's stylesheet object should be the right object.", function () {
 23   var actual = SC.Cursor.sharedStyleSheet(),
 24     wrong;
 25 
 26   wrong = document.styleSheets[document.styleSheets.length - 1];
 27   ok(actual !== wrong, "The last stylesheet is not correct.");
 28 });
