 18 /** @class
 19   A Pane is like a regular view except that it does not need to live within a
 20   parent view.  You usually use a Pane to form the root of a view hierarchy in
 21   your application, such as your main application view or for floating
 22   palettes, popups, menus, etc.
 23 
 24   Usually you will not work directly with the SC.Pane class, but with one of
 25   its subclasses such as SC.MainPane, SC.Panel, or SC.PopupPane.
 26 
 27   ## Showing a Pane
 28 
 29   To make a pane visible, you need to add it to your HTML document.  The
 30   simplest way to do this is to call the append() method:
 31 
 32       myPane = SC.Pane.create();
 33       myPane.append(); // adds the pane to the document
 34 
 35   This will insert your pane into the end of your HTML document body, causing
 36   it to display on screen.  It will also register your pane with the
 37   SC.RootResponder for the document so you can start to receive keyboard,
 38   mouse, and touch events.
 39 
 40   If you need more specific control for where you pane appears in the
 41   document, you can use several other insertion methods such as appendTo(),
 42   prependTo(), before() and after().  These methods all take a an element to
 43   indicate where in your HTML document you would like you pane to be inserted.
 44 
 45   Once a pane is inserted into the document, it will be sized and positioned
 46   according to the layout you have specified.  It will then automatically
 47   resize with the window if needed, relaying resize notifications to children
 48   as well.
 49 
 50   ## Hiding a Pane
 51 
 52   When you are finished with a pane, you can hide the pane by calling the
 53   remove() method.  This method will actually remove the Pane from the
 54   document body, as well as deregistering it from the RootResponder so that it
 55   no longer receives events.
 56 
 57   The isVisibleInWindow method will also change to NO for the Pane and all of
 58   its childViews and the views will no longer have their updateDisplay methods
 59   called.
 60 
 61   You can readd a pane to the document again any time in the future by using
 62   any of the insertion methods defined in the previous section.
 63 
 64   ## Receiving Events
 65 
 66   Your pane and its child views will automatically receive any mouse or touch
 67   events as long as it is on the screen.  To receive keyboard events, however,
 68   you must focus the keyboard on your pane by calling makeKeyPane() on the
 69   pane itself.  This will cause the RootResponder to route keyboard events to
 70   your pane.  The pane, in turn, will route those events to its current
 71   keyView, if there is any.
 72 
 73   Note that all SC.Views (anything that implements SC.ClassicResponder,
 74   really) will be notified when it is about or gain or lose keyboard focus.
 75   These notifications are sent both when the view is made keyView of a
 76   particular pane and when the pane is made keyPane for the entire
 77   application.
 78 
 79   You can prevent your Pane from becoming key by setting the acceptsKeyPane
 80   to NO on the pane.  This is useful when creating palettes and other popups
 81   that should not steal keyboard control from another view.
 82 
 83   @extends SC.View
 84   @extends SC.ResponderContext
 85   @since SproutCore 1.0
 86 */

107     The rootResponder for this pane.  Whenever you add a pane to a document,

115   /**
116     Attempts to send the event down the responder chain for this pane.  If you
117     pass a target, this method will begin with the target and work up the
118     responder chain.  Otherwise, it will begin with the current rr
119     and walk up the chain looking for any responder that implements a handler
120     for the passed method and returns YES when executed.
121 
122     @param {String} action
123     @param {SC.Event} evt
124     @param {Object} target
125     @returns {Object} object that handled the event
126   */
127   sendEvent: function(action, evt, target) {
128     var handler;
129 
130     // walk up the responder chain looking for a method to handle the event
131     if (!target) target = this.get('firstResponder') ;
132     while(target) {
133       if (action === 'touchStart') {
134         // first, we must check that the target is not already touch responder
135         // if it is, we don't want to have "found" it; that kind of recursion is sure to
136         // cause really severe, and even worse, really odd bugs.
137         if (evt.touchResponder === target) {
138           target = null;
139           break;
140         }
141 
142         // now, only pass along if the target does not already have any touches, or is
143         // capable of accepting multitouch.
144         if (!target.get("hasTouch") || target.get("acceptsMultitouch")) {
145           if (target.tryToPerform("touchStart", evt)) break;
146         }
147       } else if (action === 'touchEnd' && !target.get("acceptsMultitouch")) {
148         if (!target.get("hasTouch")) {
149           if (target.tryToPerform("touchEnd", evt)) break;
150         }
151       } else {
152         if (target.tryToPerform(action, evt)) break;
153       }
154 
155       // even if someone tries to fill in the nextResponder on the pane, stop
156       // searching when we hit the pane.
157       target = (target === this) ? null : target.get('nextResponder') ;
158     }
159 
160     // if no handler was found in the responder chain, try the default
161     if (!target && (target = this.get('defaultResponder'))) {
162       if (typeof target === SC.T_STRING) {
163         target = SC.objectForPropertyPath(target);
164       }
165 
166       if (!target) target = null;
167       else target = target.tryToPerform(action, evt) ? target : null ;
168     }
169 
170     // if we don't have a default responder or no responders in the responder
171     // chain handled the event, see if the pane itself implements the event
172     else if (!target && !(target = this.get('defaultResponder'))) {
173       target = this.tryToPerform(action, evt) ? this : null ;
174     }
175 
176     return evt.mouseHandler || target ;
177   },

243   /**
244     Makes the passed view (or any object that implements SC.Responder) into
245     the new firstResponder for this pane.  This will cause the current first
246     responder to lose its responder status and possibly keyResponder status as
247     well.
248 
249     @param {SC.View} view
250     @param {Event} evt that cause this to become first responder
251     @returns {SC.Pane} receiver
252   */
253   makeFirstResponder: function(original, view, evt) {
254 255     // firstResponder should never be null
256     if(!view) view = this;
257 258 
    var current=this.get('firstResponder'), isKeyPane=this.get('isKeyPane');
259     if (current === view) return this ; // nothing to do
260     if (SC.platform.touch && view && view.kindOf(SC.TextFieldView) && !view.get('focused')) return this;
261 
262     // if we are currently key pane, then notify key views of change also
263     if (isKeyPane) {
264       if (current) { current.tryToPerform('willLoseKeyResponderTo', view); }
265       if (view) {
266         view.tryToPerform('willBecomeKeyResponderFrom', current);
267       }
268     }
269 
270     if (current) {
271       current.beginPropertyChanges();
272       current.set('isKeyResponder', NO);
273     }
274 
275     if (view) {
276       view.beginPropertyChanges();
277       view.set('isKeyResponder', isKeyPane);
278     }
279 
280     original(view, evt);
281 
282     if(current) current.endPropertyChanges();
283     if(view) view.endPropertyChanges();
284 
285     // and notify again if needed.
286     if (isKeyPane) {
287       if (view) {
288         view.tryToPerform('didBecomeKeyResponderFrom', current); }
289       if (current) {
290         current.tryToPerform('didLoseKeyResponderTo', view);
291       }
292     }
293 
294     return this ;

430     Removes the pane from the document.  This will remove the
431     DOM node and deregister you from the document window.
432 
433     @returns {SC.Pane} receiver
434   */
435   remove: function() {
436     if (!this.get('isVisibleInWindow')) return this ; // nothing to do
437     if (!this.get('isPaneAttached')) return this ; // nothing to do
438 
439     // remove layer...
440     var dom = this.get('layer') ;
441     if (dom && dom.parentNode) dom.parentNode.removeChild(dom) ;
442     dom = null ;
443 
444     // layer is cached by SC.View in this._view_layer, remove it so that the DOM node can be freed
445     this.destroyLayer();
446 
447     // remove intercept
448     this._removeIntercept();
449 
450     // resign keyPane status, if we had it
451     this.resignKeyPane();
452 
453     // remove the pane
454     var rootResponder = this.rootResponder ;
455     if (this.get('isMainPane')) rootResponder.makeMainPane(null) ;
456     rootResponder.panes.remove(this) ;
457     this.rootResponder = null ;
458 
459     // clean up some of my own properties
460     this.set('isPaneAttached', NO) ;
461     this.parentViewDidChange();
462     return this ;
463   },

597   /**
598     Updates the isVisibleInWindow state on the pane and its childViews if
599     necessary.  This works much like SC.View's default implementation, but it
600     does not need a parentView to function.
601 
602     @returns {SC.Pane} receiver
603   */
604   recomputeIsVisibleInWindow: function() {
605     if (this.get('designer') && SC.suppressMain) return sc_super();
606     var previous = this.get('isVisibleInWindow'),
607         current  = this.get('isVisible') && this.get('isPaneAttached');
608 
609     // If our visibility has changed, then set the new value and notify our
610     // child views to update their value.
611     if (previous !== current) {
612       this.set('isVisibleInWindow', current);
613 
614       var childViews = this.get('childViews'), len = childViews.length, idx, view;
615       for(idx=0;idx<len;idx++) {
616         view = childViews[idx];
617         if (view.recomputeIsVisibleInWindow) {
618           view.recomputeIsVisibleInWindow(current);
619         }
620       }
621 
622 
623       // For historical reasons, we'll also layout the child views if
624       // necessary.
625       if (current) {
626         if (this.get('childViewsNeedLayout')) {
627           this.invokeOnce(this.layoutChildViewsIfNeeded);
628         }
629       }
630       else {
631         // Also, if we were previously visible and were the key pane, resign
632         // it.  This more appropriately belongs in a 'isVisibleInWindow'
633         // observer or some such helper method because this work is not
634         // strictly related to computing the visibility, but view performance
635         // is critical, so avoiding the extra observer is worthwhile.
636         if (this.get('isKeyPane')) { this.resignKeyPane(); }
637       }
638     }
639 
640     // If we're in this function, then that means one of our ancestor views
641     // changed, or changed its 'isVisibleInWindow' value.  That means that if
642     // we are out of sync with the layer, then we need to update our state
643     // now.
644     //
645     // For example, say we're isVisible=NO, but we have not yet added the
646     // 'sc-hidden' class to the layer because of the "don't update the layer if
647     // we're not visible in the window" check.  If any of our parent views
648     // became visible, our layer would incorrectly be shown!
649     this.updateLayerIfNeeded(YES);
650 
651     return this;
652   },
