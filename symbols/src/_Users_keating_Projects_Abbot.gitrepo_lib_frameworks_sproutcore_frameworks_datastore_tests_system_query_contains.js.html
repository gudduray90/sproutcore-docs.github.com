  8 // test parsing of query string
  9 var store, storeKey, foo1, foo2, bar1, bar2, baz, barChild, MyApp, q;
 10 module("SC.Query comparison of records", {
 11   setup: function() {
 12     
 13     SC.RunLoop.begin();
 14     
 15     // setup dummy app and store
 16     MyApp = SC.Object.create({
 17       store: SC.Store.create()
 18     });
 19     
 20     // setup a dummy model
 21     MyApp.Foo = SC.Record.extend();
 22     MyApp.Bar = SC.Record.extend();
 23     MyApp.BarChild = MyApp.Bar.extend();
 24     MyApp.Baz = SC.Record.extend();
 25     
 26     // load some data
 27     MyApp.store.loadRecords(MyApp.Foo, [
 28       { guid: 1, firstName: "John", lastName: "Doe", year: 1974 },
 29       { guid: 2, firstName: "Jane", lastName: "Doe", year: 1975 }
 30     ]);
 31     
 32     MyApp.store.loadRecords(MyApp.Bar, [
 33       { guid: 3, firstName: "Emily", lastName: "Parker", year: 1975, active: null },
 34       { guid: 4, firstName: "Johnny", lastName: "Cash", active: false }
 35     ]);
 36     
 37     MyApp.store.loadRecords(MyApp.Baz, [
 38       { guid: 5, firstName: "Bert", lastName: "Berthold", active: true }
 39     ]);
 40 
 41     MyApp.store.loadRecords(MyApp.BarChild, [
 42       { guid: 6, firstName: "Bert", lastName: "Ernie", active: true }
 43     ]);
 44     
 45     SC.RunLoop.end();
 46     
 47     foo1 = MyApp.store.find(MyApp.Foo,1);
 48     foo2 = MyApp.store.find(MyApp.Foo,2);
 49     bar1 = MyApp.store.find(MyApp.Bar,3);
 50     bar2 = MyApp.store.find(MyApp.Bar,4);
 51     barChild = MyApp.store.find(MyApp.BarChild, 6);
 52     baz  = MyApp.store.find(MyApp.Baz,5);
 53     
 54   },
 55   
 56   teardown: function() {
 57     MyApp = foo1 = foo2 = bar1 = bar2 = baz = barChild = q = null;
 58   }
 59 });
 60

 65 test("should only contain records matching recordType or recordTypes", function() {
 66   
 67   q = SC.Query.create({ recordType: MyApp.Foo });
 68   equals(q.contains(foo1), YES, 'q with recordType=Foo should contain record of type Foo');
 69   equals(q.contains(bar1), NO, 'q with recordType=Foo should NOT contain record of type Bar');
 70   equals(q.contains(barChild), NO, 'q with recordType=Foo should NOT contain record of type BarChild');
 71   
 72   equals(q.contains(baz),  NO, 'q with recordType=Foo should NOT contain record of type Baz');
 73   
 74   q = SC.Query.create({ recordTypes: [MyApp.Foo, MyApp.Bar] });
 75   equals(q.contains(foo1), YES, 'q with recordTypes=Foo,Bar should contain record of type Foo');
 76   equals(q.contains(bar1), YES, 'q with recordTypes=Foo,Bar should contain record of type Bar');
 77   equals(SC.kindOf(barChild, MyApp.Bar), YES, 'SC.kindOf(barChild, MyApp.Bar)');
 78   
 79   equals(q.contains(barChild), YES, 'q with recordTypes=Foo,Bar should contain record of type BarChild');
 80 
 81   equals(q.contains(baz),  NO, 'q with recordTypes=Foo,Bar should NOT contain record of type Baz');
 82 
 83   q = SC.Query.create();
 84   equals(q.contains(foo1), YES, 'no recordType should contain record of type Foo');
 85   equals(q.contains(bar1), YES, 'no recordType should contain record of type Foo');
 86   equals(q.contains(barChild), YES, 'no recordType should contain record of type BarChild');
 87   equals(q.contains(baz), YES, 'no recordType should contain record of type Foo');
 88   
 89 });

 91 test("should only contain records within parent scope, if one is defined", function() {
 92   
 93   q = SC.Query.create({ scope: SC.Set.create().add(foo1).add(bar1) });
 94   equals(q.contains(foo1), YES, 'scope=[foo1,bar1] should return YES for foo1');
 95   equals(q.contains(foo2), NO, 'scope=[foo1,bar1] should return NO for foo2');
 96   equals(q.contains(bar1), YES, 'scope=[bar1] should return YES for bar1');
 97   equals(q.contains(bar2), NO, 'scope=[foo1,bar1] should return NO for bar2');
 98 });
 99 
100 test("should evaluate query against record", function() {
101   q = SC.Query.create({ 
102     conditions: "firstName = {firstName}", 
103     parameters: { firstName: 'Bert' }
104   });
105   
106   equals(q.contains(bar2), NO, 'q(firstName=Bert) should return NO for bar[firstName=Johnny]');
107   equals(q.contains(baz), YES, 'q(firstName=Bert) should return YES for baz[firstName=Bert]');
108   equals(q.contains(barChild), YES, 'q(firstName=Bert) should return YES for barChild[firstName=Bert]');
109 
110   var p  = { firstName: "Johnny" };
111   equals(q.contains(bar2, p), YES, 'q(firstName=Johnny) should return YES for bar[firstName=Johnny]');
112   equals(q.contains(baz, p), NO, 'q(firstName=Johnny) should return NO for baz[firstName=Bert]');
113   equals(q.contains(barChild, p), NO, 'q(firstName=Johnny) should return NO for barChild[firstName=Bert]');
114   
115 });
116 
117 test("should consider recordType + query conditions", function() {
118   q = SC.Query.create({
119     conditions: "firstName = {firstName}",
120     recordType: MyApp.Bar,
121     parameters: { firstName: "Bert" }
122   });
123   
124   equals(q.contains(bar1), NO, 'should not contain bar1 (wrong firstName)');
125   equals(q.contains(bar2), NO, 'should not contain bar2 (wrong firstName)');
126   equals(q.contains(barChild), YES, 'should contain barChild');
127   equals(q.contains(baz), NO, 'should contain baz (wrong type)');
128   
129 });
