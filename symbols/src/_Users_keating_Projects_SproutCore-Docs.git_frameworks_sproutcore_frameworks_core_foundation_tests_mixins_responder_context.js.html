  7 /*global module test equals context ok same */
  8 var S, A, a, B, Manager;
  9 module("ResponderContext", {
 10   setup: function() {
 11     Manager = SC.Responder.create(SC.ResponderContext, {  });
 12     var TestResponder = SC.Responder.extend({
 13       didBecomeFirstResponder: function() {
 14         this.didBecome = YES;
 15         this.hasFirst = YES;
 16       },
 17       willLoseFirstResponder: function() {
 18         this.didLose = YES;
 19         this.hasFirst = NO;
 20       }
 21     });
 22     
 23     A = TestResponder.create();
 24     a = TestResponder.create({nextResponder: A});
 25     B = TestResponder.create();
 26   }
 27 });
 28 
 29 test("Can enter and exit states.", function() {
 30   Manager.makeFirstResponder(A);
 31   ok(A.didBecome, "A did become first responder.");
 32   ok(A.hasFirst, "A has first responder.");
 33   
 34   Manager.makeFirstResponder(B);
 35   ok(A.didLose, "A did lose first responder.");
 36   ok(!A.hasFirst, "A does not have first responder.");
 37 });
 38 
 39 test("Can enter and exit chained states.", function() {
 40   Manager.makeFirstResponder(A);
 41   ok(A.didBecome, "A did become first responder.");
 42   ok(A.hasFirst, "A has first responder.");
 43   
 44   Manager.makeFirstResponder(a);
 45   ok(!A.didLose, "A did not lose first responder.");
 46   ok(A.hasFirst, "A has first responder.");
 47   ok(A.didBecome, "a did become first responder.");
 48   ok(A.hasFirst, "a has first responder.");
 49   
 50   Manager.makeFirstResponder(B);
 51   ok(a.didLose, "a did lose first responder.");
 52   ok(!a.hasFirst, "a does not have first responder.");
 53   ok(A.didLose, "A did lose first responder.");
 54   ok(!A.hasFirst, "A does not have first responder.");
 55 });
 56 
 57 test("Setting responder to the current responder does not reenter.", function() {
 58   Manager.makeFirstResponder(A);
 59   ok(A.didBecome, "A did become first responder.");
 60   ok(A.hasFirst, "A has first responder.");
 61 
 62   A.didBecome = NO;
 63   Manager.makeFirstResponder(A);
 64   ok(!A.didBecome, "A did become first responder.");
 65 });
 66 
 67 test("Calling 'resetFirstResponder' reenters the first responder.", function() {
 68   Manager.makeFirstResponder(A);
 69   ok(A.didBecome, "A did become first responder.");
 70   ok(A.hasFirst, "A has first responder.");
 71 
 72   A.didBecome = NO;
 73   Manager.resetFirstResponder();
 74   ok(A.didLose, "A did leave.");
 75   ok(A.didBecome, "A did reenter.");
 76 });
