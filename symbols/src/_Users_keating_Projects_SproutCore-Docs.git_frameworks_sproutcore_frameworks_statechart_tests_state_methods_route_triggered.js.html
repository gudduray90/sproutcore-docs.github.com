  6 var sc, del, foo;

  8 module("SC.State: routeTriggered method Tests", {
  9   
 10   setup: function() {
 11     
 12     del = SC.Object.create(SC.StatechartDelegate, {
 13       
 14       info: {},
 15       
 16       returnValue: YES,
 17       
 18       statechartShouldStateHandleTriggeredRoute: function(statechart, state, context) {
 19         this.info.statechartShouldStateHandleTriggeredRoute = {
 20           statechart: statechart, 
 21           state: state, 
 22           context: context
 23         };
 24         
 25         return this.get('returnValue');
 26       },
 27       
 28       statechartStateCancelledHandlingTriggeredRoute: function(statechart, state, context) {
 29         this.info.statechartStateCancelledHandlingTriggeredRoute = {
 30           statechart: statechart, 
 31           state: state, 
 32           context: context
 33         };
 34       }
 35       
 36     });
 37     
 38     sc = SC.Statechart.create({
 39       
 40       initialState: 'foo',
 41       
 42       delegate: del,
 43     
 44       foo: SC.State.design({
 45         
 46         info: {},
 47         
 48         location: 'foo/bar',
 49         
 50         createStateRouteHandlerContext: function(attr) {
 51           this.info.createStateRouteHandlerContext = {
 52             attr: attr
 53           };
 54           return sc_super();
 55         },
 56         
 57         handleTriggeredRoute: function(context) {
 58           this.info.handleTriggeredRoute = {
 59             context: context
 60           };
 61         }
 62         
 63       })
 64       
 65     });
 66     
 67     sc.initStatechart();
 68     foo = sc.getState('foo');
 69   },
 70   
 71   teardown: function() {
 72     sc = del = foo = null;
 73   }
 74   
 75 });

 77 test("invoke routeTriggered where delegate does allow state to handle route", function() {
 78   var info, context, params = { value: 'test' };
 79   
 80   foo.routeTriggered(params);
 81 
 82   info = foo.info.createStateRouteHandlerContext;
 83 
 84   ok(info, "state.createStateRouteHandlerContext should have been invoked");
 85   ok(info.attr, "state.createStateRouteHandlerContext should be provided attr param");
 86   
 87   info = foo.info.handleTriggeredRoute;
 88   
 89   ok(info, "state.handleTriggeredRoute should have been invoked");
 90   
 91   context = info.context;
 92   
 93   ok(SC.kindOf(context, SC.StateRouteHandlerContext), "state.handleTriggeredRoute should be provided a state route handler context object");
 94   equals(context.get('state'), foo, "context.state should be state foo");
 95   equals(context.get('location'), 'foo/bar', "context.location should be 'foo/bar'");
 96   equals(context.get('params'), params, "context.params should be value passed to state.routeTriggered method");
 97   equals(context.get('handler'), foo.routeTriggered, "context.handler should be reference to state.routeTriggered");
 98   
 99   info = del.info.statechartShouldStateHandleTriggeredRoute;
100   
101   ok(info, "del.statechartShouldStateHandleTriggeredRoute should have been invoked");
102   equals(info.statechart, sc, "del.statechartShouldStateHandleTriggeredRoute should be provided a statechart");
103   equals(info.state, foo, "del.statechartShouldStateHandleTriggeredRoute should be provided a state");
104   
105   context = info.context;
106   
107   ok(SC.kindOf(context, SC.StateRouteHandlerContext), "state.statechartShouldStateHandleTriggeredRoute should be provided a state route handler context object");
108   equals(context.get('state'), foo, "context.state should be state foo");
109   equals(context.get('location'), 'foo/bar', "context.location should be 'foo/bar'");
110   equals(context.get('params'), params, "context.params should be value passed to state.routeTriggered method");
111   equals(context.get('handler'), foo.routeTriggered, "context.handler should be reference to state.routeTriggered");
112   
113   info = del.info.statechartStateCancelledHandlingTriggeredRoute;
114   
115   ok(!info, "del.statechartStateCancelledHandlingTriggeredRoute should have been invoked");
116 });
