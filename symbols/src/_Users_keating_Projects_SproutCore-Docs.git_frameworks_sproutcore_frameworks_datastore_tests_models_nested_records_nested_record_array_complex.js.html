346 test("Advanced Array Functionality: pushObject", function() {
347   var ppl, that = this;
348   that.willChange = 0;
349   that.didChange = 0;
350   that.removedCount = 0;
351   that.addedCount = 0;
352 
353   this.arrayContentWillChange = function(start, removedCount, addedCount) {
354     that.willChange += 1;
355   };
356   this.arrayContentDidChange = function(start, removedCount, addedCount) {
357     that.didChange += 1;
358     that.removedCount += removedCount;
359     that.addedCount += addedCount;
360   };
361 
362   // Create thea rray and add observers
363   ppl = testParent.get('people');
364   ppl.addArrayObservers({
365     target: this,
366     willChange: this.arrayContentWillChange,
367     didChange: this.arrayContentDidChange
368   });
369 
370   // This should fire willChange and didChange once each
371   ppl.pushObject(peopleData2[0]);
372   equals(that.willChange, 1, "arrayContentWillChange should have been executed once");
373   equals(that.didChange, 1, "arrayContentDidChange should have been executed once");
374   equals(that.removedCount, 0, "Zero records were removed");
375   equals(that.addedCount, 1, "1 Record was added");
376 
377   // Reset
378   that.willChange = 0;
379   that.didChange = 0;
380   that.removedCount = 0;
381   that.addedCount = 0;
382 
383   // This should fire willChange and didChange once each
384   ppl.pushObject(peopleData2[1]);
385   equals(that.willChange, 1, "arrayContentWillChange should have been executed once");
386   equals(that.didChange, 1, "arrayContentDidChange should have been executed once");
387   equals(that.removedCount, 0, "Zero records were removed");
388   equals(that.addedCount, 1, "A second record was added");
389 });
