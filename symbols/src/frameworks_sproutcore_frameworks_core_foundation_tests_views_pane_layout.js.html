  9 var pane;
 10 
 11 module("SC.Pane#layout", {
 12   setup: function() {
 13     pane = SC.Pane.create({
 14       layout: { top: 0, left: 0, width: 1, height: 1}
 15     });
 16     pane.append();
 17   },
 18 
 19   teardown: function() {
 20     pane.remove();
 21     pane.destroy();
 22   }
 23 });
 24 
 25 test("make sure that a call to adjust actually adjusts the view's size", function() {
 26   SC.RunLoop.begin();
 27   pane.adjust({ width: 100, height: 50 });
 28   SC.RunLoop.end();
 29 
 30   equals(pane.$()[0].style.width, '100px', 'width should have been adjusted');
 31   equals(pane.$()[0].style.height, '50px', 'height should have been adjusted');
 32 });
