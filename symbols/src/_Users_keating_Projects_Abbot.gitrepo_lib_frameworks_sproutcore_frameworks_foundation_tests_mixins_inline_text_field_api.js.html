 10 var field;
 11 
 12 /**
 13   Track the public functions and properties of the class.  This will serve as an early warning
 14   when functions that people may depend on disappear between versions to ensure that we don't
 15   break promised support without proper deprecations.
 16   
 17   tylerkeating: This is probably redundant since each of these functions and properties should
 18   be individually tested elsewhere.
 19 */
 20 module("Test the public functions and properties of SC.InlineTextFieldView", {
 21   setup: function() {
 22     field = SC.InlineTextFieldView.create({});
 23   },
 24 
 25   teardown: function() {
 26     field = null;
 27   }
 28 });
 29 
 30 
 31 test("contains all public class functions",
 32 function() {
 33   ok(SC.typeOf(SC.InlineTextFieldView['beginEditing']) === SC.T_FUNCTION, "should respond to beginEditing()");
 34   ok(SC.typeOf(SC.InlineTextFieldView['commitEditing']) === SC.T_FUNCTION, "should respond to commitEditing()");
 35   ok(SC.typeOf(SC.InlineTextFieldView['discardEditing']) === SC.T_FUNCTION, "should respond to discardEditing()");
 36 });
 37 
 38 
 39 test("contains all public functions",
 40 function() {
 41   ok(field.respondsTo('beginEditing'), "should respond to beginEditing()");
 42   ok(field.respondsTo('commitEditing'), "should respond to commitEditing()");
 43   ok(field.respondsTo('discardEditing'), "should respond to discardEditing()");
 44   ok(field.respondsTo('blurEditor'), "should respond to blurEditor()");
 45   ok(field.respondsTo('cancel'), "should respond to cancel()");
 46 });
 47 
 48 
 49 test("contains all public properties",
 50 function() {
 51   ok(field.get('isEditing') !== undefined, "should have isEditing property");
 52 });
