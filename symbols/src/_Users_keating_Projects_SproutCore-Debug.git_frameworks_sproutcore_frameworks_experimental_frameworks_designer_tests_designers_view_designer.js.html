  9 (function() {
 10   var basic, testView, page;
 11   // ..........................................................
 12   // TEST VIEWS
 13   // 
 14   module('SC.ViewDesigner', {
 15     setup: function () {
 16       testView = SC.View.extend({
 17         mouseDown: function(){
 18           var page = this.get('page');
 19           page.set('mouseDownInView', YES);
 20         }
 21       });
 22       testView.Designer = SC.ViewDesigner.extend({
 23         mouseDown: function(){
 24           var page = this.get('page');
 25           page.set('mouseDownInDesigner', YES);
 26         }
 27       });
 28       basic = SC.Page.design({
 29         needsDesigner: true, 
 30         mouseDownInView: false, 
 31         mouseDownInDesigner: false,
 32         view: testView.design({})
 33       });
 34       
 35       
 36     },
 37     teardown: function () {}
 38   });
 39 
 40   test("tryToPerform redirects to designer",function() {
 41     var view = basic.get('view');
 42     view.tryToPerform('mouseDown', {});
 43     ok(basic.get('mouseDownInDesigner'), "designer got mouseDown");
 44  45     ok(!basic.get('mouseDownInView'), "view did not get mouseDown");
 46   });
 47 
 48 })();
