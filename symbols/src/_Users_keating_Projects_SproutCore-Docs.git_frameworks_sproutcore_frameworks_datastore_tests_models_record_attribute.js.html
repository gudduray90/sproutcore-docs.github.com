  8 // test core array-mapping methods for RecordArray with RecordAttribute
  9 var storeKeys, rec, rec2, rec3, bar, MyApp;
 10 
 11 module("SC.RecordAttribute core methods", {
 12   setup: function() {
 13 
 14     MyApp = SC.Object.create({
 15       store: SC.Store.create()
 16     });
 17 
 18     // stick it to the window object so that objectForPropertyPath works
 19     window.MyApp = MyApp;
 20 
 21     MyApp.Foo = SC.Record.extend({
 22 
 23       // test simple reading of a pass-through prop
 24       firstName: SC.Record.attr(String),
 25 
 26       // test mapping to another internal key
 27       otherName: SC.Record.attr(String, { key: "firstName" }),
 28 
 29       // test mapping Date
 30       date: SC.Record.attr(Date),
 31       nonIsoDate: SC.Record.attr(Date, { useIsoDate: false }),
 32 
 33       // test SC.DateTimes
 34       dateTime: SC.Record.attr(SC.DateTime),
 35 
 36       // test Array
 37       anArray: SC.Record.attr(Array),
 38 
 39       // test Object
 40       anObject: SC.Record.attr(Object),
 41 
 42       // test Number
 43       aNumber: SC.Record.attr(Number),
 44 
 45       // used to test default value
 46       defaultValue: SC.Record.attr(String, {
 47         defaultValue: "default"
 48       }),
 49 
 50       // used to test default value
 51       defaultComputedValue: SC.Record.attr(Number, {
 52         defaultValue: function() {
 53           return Math.floor(Math.random()*3+1);
 54         }
 55       }),
 56 
 57       // test toOne relationships
 58       relatedTo: SC.Record.toOne('MyApp.Foo'),
 59 
 60       // test toOne relationship with computed type
 61       relatedToComputed: SC.Record.toOne(function() {
 62         // not using .get() to avoid another transform which will
 63         // trigger an infinite loop
 64         return (this.readAttribute('relatedToComputed').indexOf("foo")===0) ? MyApp.Foo : MyApp.Bar;
 65       }),
 66 
 67       // test readONly
 68       readOnly: SC.Record.attr(String, { isEditable: NO })
 69 
 70     });
 71 
 72     MyApp.Bar = SC.Record.extend({
 73       parent: SC.Record.toOne('MyApp.Foo', { aggregate: YES }),
 74       relatedMany: SC.Record.toMany('MyApp.Foo', { aggregate: YES })
 75     });
 76 
 77     SC.RunLoop.begin();
 78     storeKeys = MyApp.store.loadRecords(MyApp.Foo, [
 79       {
 80         guid: 'foo1',
 81         firstName: "John",
 82         lastName: "Doe",
 83         date: "2009-03-01T20:30-08:00",
 84         dateTime: new Date(1235939425000),
 85         anArray: ['one', 'two', 'three'],
 86         anObject: { 'key1': 'value1', 'key2': 'value2' },
 87         aNumber: '123',
 88         readOnly: 'foo1'
 89       },
 90 
 91       {
 92         guid: 'foo2',
 93         firstName: "Jane",
 94         lastName: "Doe",
 95         relatedTo: 'foo1',
 96         relatedToAggregate: 'bar1',
 97         dateTime: "2009-03-01T20:30:25Z",
 98         anArray: 'notAnArray',
 99         anObject: 'notAnObject',
100         aNumber: '123',
101         nonIsoDate: "2009/06/10 8:55:50 +0000"
102       },
103 
104       {
105         guid: 'foo3',
106         firstName: "Alex",
107         lastName: "Doe",
108         relatedToComputed: 'bar1',
109         dateTime: SC.DateTime.create(1235939425000),
110         anArray: ['one', 'two', 'three'],
111         anObject: { 'key1': 'value1', 'key2': 'value2' },
112         aNumber: '123'
113       }
114 
115     ]);
116 
117     MyApp.store.loadRecords(MyApp.Bar, [
118       { guid: 'bar1', city: "Chicago", parent: 'foo2', relatedMany: ['foo1', 'foo2'] }
119     ]);
120 
121     SC.RunLoop.end();
122 
123     rec = MyApp.store.find(MyApp.Foo, 'foo1');
124     rec2 = MyApp.store.find(MyApp.Foo, 'foo2');
125     rec3 = MyApp.store.find(MyApp.Foo, 'foo3');
126 
127     bar = MyApp.store.find(MyApp.Bar, 'bar1');
128     equals(rec.storeKey, storeKeys[0], 'should find record');
129 
130   }
131 });

147 });

151 });

173 });

195 test("writing pass-through should simply set value", function() {
196   rec.set("firstName", "Foo");
197   equals(rec.readAttribute("firstName"), "Foo", "should write string");
198 
199   rec.set("firstName", 23);
200   equals(rec.readAttribute("firstName"), 23, "should write number");
201 
202   rec.set("firstName", YES);
203   equals(rec.readAttribute("firstName"), YES, "should write bool");
204 
205 });

207 test("writing when isEditable is NO should ignore", function() {
208   var v = rec.get('readOnly');
209   rec.set('readOnly', 'NEW VALUE');
210   equals(rec.get('readOnly'), v, 'read only value should not change');
211 });
212 
213 test("writing a value should override default value", function() {
214   equals(rec.get('defaultValue'), 'default', 'precond - returns default');
215   rec.set('defaultValue', 'not-default');
216   equals(rec.get('defaultValue'), 'not-default', 'newly written value should replace default value');
217 });

219 test("writing a string to a number attribute should store a number" ,function() {
220      equals(rec.set('aNumber', "456"), rec, 'returns receiver');
221      equals(rec.get('aNumber'), 456, 'should have new value');
222      equals(typeof rec.get('aNumber'), 'number', 'new value should be a number');
223 });
224 
225 test("writing a date should generate an ISO date" ,function() {
226   var date = new Date(1238650083966);
227 
228   // Work with timezones
229   var utcDate = new Date(Number(date) + (date.getTimezoneOffset() * 60000)); // Adjust for timezone offset
230   utcDate.getTimezoneOffset = function(){ return 0; }; // Hack the offset to respond 0
231 
232   equals(rec.set('date', utcDate), rec, 'returns receiver');
233   equals(rec.readAttribute('date'), '2009-04-02T05:28:03Z', 'should have time in ISO format');
234 });

244 });
245

248   equals(rec2.get('status'), SC.Record.READY_CLEAN, "precond - rec2 should be READY_CLEAN");
249 
250   bar.set('city', 'Oslo');
251   bar.get('store').flush();
252

254   equals(rec2.get('status'), SC.Record.READY_DIRTY, "foo2 should be READY_DIRTY");
255 });
256 
257 test("writing an attribute should make many relationship aggregate dirty and add the aggregate to the store" ,function() {
258   equals(bar.get('status'), SC.Record.READY_CLEAN, "precond - bar should be READY_CLEAN");
259   equals(rec2.get('status'), SC.Record.READY_CLEAN, "precond - rec2 should be READY_CLEAN");
