  6 /*global module, test, equals, ok, start, stop */
  7 
  8 var linkView,
  9   pane;
 10 
 11 
 12 module("SC.LinkView", {
 13 
 14   setup: function () {
 15     linkView = SC.LinkView.create();
 16     pane = SC.Pane.create({
 17       childViews: [linkView],
 18       layout: { width: 400, height: 400 }
 19     });
 20   },
 21 
 22   teardown: function () {
 23     pane.destroy();
 24     linkView = pane = null;
 25   }
 26 
 27 });
 28 
 29 test("The view renders as expected: Default values", function () {
 30   pane.append();
 31 
 32   var layer = linkView.get('layer');
 33 
 34   // body: ""
 35   // fileName: null
 36   // escapeHTML: true
 37   // href: '#'
 38   // language: null
 39   // ping: null
 40   // rel: null
 41   // target: '_blank'
 42   // type: null
 43   equals(layer.innerText, "", "The innerText of the layer is");
 44   if (SC.platform.a.download) { equals(layer.download, "", "The download attribute of the layer is"); }
 45   equals(layer.href, document.URL + "#", "The href attribute of the layer is");
 46   equals(layer.hreflang, SC.Locale.currentLocale.language, "The hreflang attribute of the layer is");
 47   equals(layer.ping, "", "The ping attribute of the layer is");
 48   equals(layer.rel, "", "The rel attribute of the layer is");
 49   equals(layer.target, "_blank", "The target attribute of the layer is");
 50   equals(layer.type, "", "The type attribute of the layer is");
 51 });
 52 
 53 test("The view renders as expected: Custom values", function () {
 54   // Customize the link view.
 55   linkView.set('body', '<div>ABC</div>');
 56   linkView.set('fileName', 'ABC.txt');
 57   linkView.set('href', 'http://bogus-site.com/files/abc.txt');
 58   linkView.set('language', 'fr');
 59   linkView.set('ping', ['http://logback.com/?clicked=file']);
 60   linkView.set('rel', ['author']);
 61   linkView.set('target', '_self');
 62   linkView.set('type', 'text/plain');
 63 
 64   pane.append();
 65 
 66   var layer = linkView.get('layer');
 67 
 68   equals(layer.innerText, "<div>ABC</div>", "The innerText of the layer is");
 69   if (SC.platform.a.download) { equals(layer.download, "ABC.txt", "The download attribute of the layer is"); }
 70   equals(layer.href, "http://bogus-site.com/files/abc.txt", "The href attribute of the layer is");
 71   equals(layer.hreflang, 'fr', "The hreflang attribute of the layer is");
 72   if (SC.platform.a.ping) { equals(layer.ping, "http://logback.com/?clicked=file", "The ping attribute of the layer is"); }
 73   equals(layer.rel, "author", "The rel attribute of the layer is");
 74   equals(layer.target, "_self", "The target attribute of the layer is");
 75   equals(layer.type, "text/plain", "The type attribute of the layer is");
 76 });
 77 
 78 test("The view renders as expected: Unescaped body", function () {
 79   // Customize the link view.
 80   linkView.set('escapeHTML', false);
 81   linkView.set('body', '<div>ABC</div>');
 82 
 83   pane.append();
 84 
 85   var layer = linkView.get('layer');
 86 
 87   equals(layer.innerHTML, "<div>ABC</div>", "The innerHTML of the layer is");
 88   equals(layer.innerText.trimRight(), "ABC", "The innerText of the layer is");
 89 });
 90 
 91 test("The view updates as expected: Custom values", function () {
 92   // Customize the link view.
 93   linkView.set('body', '<div>ABC</div>');
 94   linkView.set('fileName', 'ABC.txt');
 95   linkView.set('href', 'http://bogus-site.com/files/abc.txt');
 96   linkView.set('language', 'fr');
 97   linkView.set('ping', ['http://logback.com/?clicked=file']);
 98   linkView.set('rel', ['author']);
 99   linkView.set('target', '_self');
100   linkView.set('type', 'text/plain');
101 
102   pane.append();
103 
104   var layer = linkView.get('layer');
105 
106   equals(layer.innerText, "<div>ABC</div>", "The innerText of the layer is");
107   if (SC.platform.a.download) { equals(layer.download, "ABC.txt", "The download attribute of the layer is"); }
108   equals(layer.href, "http://bogus-site.com/files/abc.txt", "The href attribute of the layer is");
109   equals(layer.hreflang, 'fr', "The hreflang attribute of the layer is");
110   if (SC.platform.a.ping) { equals(layer.ping, "http://logback.com/?clicked=file", "The ping attribute of the layer is"); }
111   equals(layer.rel, "author", "The rel attribute of the layer is");
112   equals(layer.target, "_self", "The target attribute of the layer is");
113   equals(layer.type, "text/plain", "The type attribute of the layer is");
114 
115   SC.run(function () {
116     linkView.set('body', '<div>DEF</div>');
117     linkView.set('fileName', 'DEF.js');
118     linkView.set('href', 'http://bogus-site.com/files/def.js');
119     linkView.set('language', 'es');
120     linkView.set('ping', ['https://logback.com/?clicked=file']);
121     linkView.set('rel', ['sidebar']);
122     linkView.set('target', '_blank');
123     linkView.set('type', 'text/javascript');
124   });
125 
126   equals(layer.innerText, "<div>DEF</div>", "The innerText of the layer is now");
127   if (SC.platform.a.download) { equals(layer.download, "DEF.js", "The download attribute of the layer is now"); }
128   equals(layer.href, "http://bogus-site.com/files/def.js", "The href attribute of the layer is now");
129   equals(layer.hreflang, 'es', "The hreflang attribute of the layer is now");
130 
131   // Some properties don't result in re-rendering and aren't updated for performance reasons.
132   if (SC.platform.a.ping) { equals(layer.ping, "http://logback.com/?clicked=file", "The ping attribute of the layer is still"); }
133   equals(layer.rel, "author", "The rel attribute of the layer is still");
134   equals(layer.target, "_self", "The target attribute of the layer is still");
135   equals(layer.type, "text/plain", "The type attribute of the layer is still");
136 });
