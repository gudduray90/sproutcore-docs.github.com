  9 module("SC.View#destroyLayer");

 11 test("it if has no layer, does nothing", function() {
 12   var callCount = 0;
 13   var view = SC.View.create({ 
 14     willDestroyLayer: function() { callCount++; }
 15   });
 16   ok(!view.get('layer'), 'precond - does NOT have layer');
 17   
 18   view.destroyLayer();
 19   equals(callCount, 0, 'did not invoke callback');
 20 });
 21 
 22 test("if it has a layer, calls willDestroyLayer on receiver and child views then deletes the layer", function() {
 23   var callCount = 0;
 24   
 25   var view = SC.View.create({
 26     willDestroyLayer: function() { callCount++; },
 27     childViews: [SC.View.extend({
 28       // no willDestroyLayer here... make sure no errors are thrown
 29       childViews: [SC.View.extend({
 30         willDestroyLayer: function() { callCount++; }
 31       })]
 32     })]
 33   });
 34   view.createLayer();
 35   ok(view.get('layer'), 'precond - view has layer');
 36   
 37   view.destroyLayer();
 38   equals(callCount, 2, 'invoked destroy layer');  
 39   ok(!view.get('layer'), 'view no longer has layer');
 40 });

test("if it has a layer, calls willDestroyLayerMixin on receiver and child views if defined (comes from mixins)", function() {
 43   var callCount = 0;
 44 
 45   // make sure this will call both mixins...
 46   var mixinA = {
 47     willDestroyLayerMixin: function() { callCount++; }
 48   };
 49   
 50   var mixinB = {
 51     willDestroyLayerMixin: function() { callCount++; }
 52   };
 53   
 54   var view = SC.View.create(mixinA, mixinB, {
 55     childViews: [SC.View.extend(mixinA, mixinB, {
 56       childViews: [SC.View.extend(mixinA)]
 57     })]
 58   });
 59   view.createLayer();
 60   view.destroyLayer();
 61   equals(callCount, 5, 'invoked willDestroyLayerMixin on all mixins');  
 62 });

 64 test("returns receiver", function() {

 66   equals(view.destroyLayer(), view, 'returns receiver');

 69 test("removes layer from parentNode if in DOM", function() {
 70   var view = SC.View.create();
 71   var layer = view.createLayer().get('layer');
 72   
 73   ok(layer, 'precond - has layer');
 74   document.body.appendChild(layer); // add to document body
 75   
 76   view.destroyLayer();
 77 
 78   if(layer.parentNode) equals(layer.parentNode.nodeType, 11, 'layer no longer in parent node');
 79   else equals(layer.parentNode, null, 'layer no longer in parent node');
 80   layer = null; // cleanup
 81 });
