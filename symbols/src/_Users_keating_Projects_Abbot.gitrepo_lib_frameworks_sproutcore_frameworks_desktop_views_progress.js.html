  8 /**
  9   @class
 10 
 11   Displays a progress bar.  You can display both a defined and an 
 12   indeterminate progressbar.  The progress bar itself is designed to be styled
 13   using CSS classes with the following structure:
 14   
 15       <div class="sc-progress-view"><div class="inner"></div></div>
 16   
 17   The outer can form the boundary of the bar while the inner will be adjusted 
 18   to fit the percentage of the progress.
 19   
 20   Creating a ProgressView accepts a number of properties, for example:
 21   
 22       progressView: SC.ProgressView.design({
 23         value: 50, 
 24         minimum: 0, 
 25         maximum: 100,
 26         isIndeterminate: NO,
 27         isEnabled: YES
 28       })
 29   
 30   @extends SC.View
 31   @extends SC.Control
 32   @since SproutCore 1.0
 33 */
 34 SC.ProgressView = SC.View.extend(SC.Control,
 35 /** @scope SC.ProgressView.prototype */{
 36   
 37   /**
 38     @type Array
 39     @default ['sc-progress-view']
 40     @see SC.View#classNames
 41   */
 42   classNames: ['sc-progress-view'],
 43   
 44   /**
 45     @type Array
 46     @default 'displayValue ariaValue minimum maximum isRunning isEnabled isIndeterminate animatedBackgroundMatrix offsetRange'.w()
 47     @see SC.View#displayProperties
 48   */
 49   displayProperties: ['displayValue', 'ariaValue', 'minimum', 'maximum', 'isRunning', 'isEnabled', 'isIndeterminate', 'animatedBackgroundMatrix', 'offsetRange'],
 50 
 51   /**
 52     @type String
 53     @default 'progressRenderDelegate'
 54   */
 55   renderDelegateName: 'progressRenderDelegate',
 56 
 57   // ........................................
 58   // PROPERTIES
 59   //
 60 
 61   /**
 62     Bind this to the current value of the progress bar.  Note that by default 
 63     an empty value will disable the progress bar and a multiple value will make 
 64     it indeterminate.
 65     
 66     @type Number
 67     @default 0.50
 68   */
 69   value: 0.50,
 70 
 71   /** @private */
 72   valueBindingDefault: SC.Binding.single().notEmpty(),
 73   
 74   /**
 75     @field
 76     @type Number
 77     @observes value
 78     @observes maximum
 79     @observes minimum
 80   */
 81   displayValue: function(){
 82     var minimum = this.get('minimum') || 0.0,
 83         maximum = this.get('maximum') || 1.0,
 84         value = this.get('value') || 0.0;
 85     value = (value - minimum) / (maximum - minimum);
 86     if (value > 1.0) value = 1.0;
 87 
 88     if(isNaN(value)) value = 0.0;
 89     // cannot be smaller then minimum
 90     if(value<minimum) value = 0.0;
 91     // cannot be larger then maximum
 92     if(value>maximum) value = 1.0;
 93     
 94     return value;
 95   }.property('value', 'maximum', 'minimum').cacheable(),
 96 
 97   /**
 98     The WAI-ARIA role for progress view.
 99 
100     @type String
101     @default 'progressbar'
102     @readOnly
103   */
104   ariaRole: 'progressbar',
105 
106   /**
107     The WAI-ARIA value for the progress view. This value will be passed to any
108     rendering code as-is, not converted into percentages, etc. It is computed
109     based on the original value property.
110 
111     @property
112   */
113   ariaValue: function() {
114     return this.get('value');
115   }.property('value').cacheable(),
116   
117   /**
118     The minimum value of the progress.
119     
120     @type Number
121 122     @default 0
123   */ 
124   minimum: 0,
125   
126   /** @private */
127   minimumBindingDefault: SC.Binding.single().notEmpty(),
128 
129   /**
130     Optionally specify the key used to extract the minimum progress value 
131     from the content object.  If this is set to null then the minimum value
132     will not be derived from the content object.
133     
134     @type String
135     @default null
136   */
137   contentMinimumKey: null,
138   
139   /**
140     The maximum value of the progress bar.
141     
142     @type Number
143     @default 1.0
144   */
145   maximum: 1.0,
146   
147   /** @private */
148   maximumBindingDefault: SC.Binding.single().notEmpty(),
149 
150   /**
151     Deprecated. This is a render setting, and as such, should be adjusted in
152     the theme. Investigate your theme's progressRenderDelegate.
153     
154     @deprecated This should now be changed in themes.
155     @type Integer
156   */
157   offsetRange: undefined,
158 
159   /**
160     Optionally specify the key used to extract the maximum progress value 
161     from the content object.  If this is set to null then the maximum value
162     will not be derived from the content object.
163     
164     @type String
165     @default null
166   */
167   contentMaximumKey: null,
168 
169   /** 
170     Set to true if the item in progress is indeterminate.  This may be 
171     overridden by the actual value.
172     
173     @type Boolean
174     @default NO
175   */
176   isIndeterminate: NO,
177   
178   /** @private */
179   isIndeterminateBindingDefault: SC.Binding.bool(),
180 
181   /**
182     Set to YES when the process is currently running.  This will cause the 
183     progress bar to animate, especially if it is indeterminate.
184     
185     @type Boolean
186     @default NO
187   */
188   isRunning: NO,
189   
190   /** @private */
191   isRunningBindingDefault: SC.Binding.bool(),
192 
193   /** 
194     Set to the matrix used for background image position for animation.
195     [1st image y-location, offset, total number of images]
196     
197     @type Array
198     @default null
199   */
200   animatedBackgroundMatrix: null,
201   
202   /**
203     Optionally specify the key used to extract the isIndeterminate value 
204     from the content object.  If this is set to null then the isIndeterminate 
205     value will not be derived from the content object.
206     
207     @type String
208     @default null
209 210   */
211   contentIsIndeterminateKey: null,
212 
213   
214   // ........................................
215   // INTERNAL SUPPORT
216   //
217 
218   /** @private */
219   _backgroundOffset: 0,
220   
221   /** @private */
222   _currentBackground: 1,
223   
224   /** @private */
225   _nextBackground: 1,
226 
227   /** @private */
228   init: function() {
229     sc_super();
230     this.animateProgressBar();
231   },
232   
233   /** @private */
234   animateProgressBar: function() {
235     if (this.get('isRunning') && this.get('isVisibleInWindow')) {
236       this._animateProgressBar(500); // wait to start to avoid probs
237     }
238   }.observes('isRunning', 'isVisibleInWindow'),
239 
240   /** @private */
241   _animateProgressBar: function(delay) {
242     if (delay===0) delay = 1000/30;
243     if (this.get('isRunning') && this.get('isVisibleInWindow')) {
244       this.displayDidChange();
245       this.invokeLater(this._animateProgressBar, delay, 600);
246     }
247   },
248 
249   /** @private */
250   contentPropertyDidChange: function(target, key) {
251     var content = this.get('content');
252     this.beginPropertyChanges()
253       .updatePropertyFromContent('value', key, 'contentValueKey', content)
254       .updatePropertyFromContent('minimum', key, 'contentMinimumKey', content)
255       .updatePropertyFromContent('maximum', key, 'contentMaximumKey', content)
256       .updatePropertyFromContent('isIndeterminate', key, 'contentIsIndeterminateKey', content)
257     .endPropertyChanges();
258   }
259 
260 });
