 31 function createCookie(name,value,days) {
 32   var expires;
 33 	if (days) {
 34 		var date = new Date();
 35 		date.setTime(date.getTime()+(days*24*60*60*1000));
 36 		expires = "; expires="+date.toGMTString();
 37 	}
 38 	else expires = "";
 39 	document.cookie = name+"="+value+expires+"; path=/";
 40 }
 41 
 42 function readCookie(name) {
 43 	var nameEQ = name + "=";
 44 	var ca = document.cookie.split(';');
 45 	for(var i=0;i < ca.length;i++) {
 46 		var c = ca[i];
 47 		while (c.charAt(0)==' ') c = c.substring(1,c.length);
 48 		if (c.indexOf(nameEQ) === 0) return decodeURIComponent(c.substring(nameEQ.length,c.length));
 49 	}
 50 	return null;
 51 }
 52 
 53 function eraseCookies() {
 54   setCookies.forEach(function(cookie) {
 55     createCookie(cookie, "", -1);
 56   });
 57 }
