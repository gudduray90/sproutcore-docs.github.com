  7 sc_require('views/views_item_view.js');

 10 /**
 11   This list view is for Views & Controls examples.
 12 */
 13 Showcase.ViewsListView = SC.ListView.extend({
 14 
 15     classNames: ['views-list-view'],
 16 
 17     customRowHeightIndexes: function() {
 18       var content = this.get('content'),
 19           customRowHeightIndexes = SC.IndexSet.create();
 20 
 21       for (var i = content.get('length') - 1; i >= 0; i--) {
 22         if (content.objectAt(i).get('isShowingSnippet')) {
 23           customRowHeightIndexes.add(i, 1);
 24         }
 25       }
 26 
 27       return customRowHeightIndexes;
 28     }.property().idempotent(),
 29 
 30     exampleHeight: 120,
 31 
 32     exampleView: Showcase.ViewsItemView,
 33 
 34     itemHeight: function() {
 35       var exampleHeight = this.get('exampleHeight');
 36 
 37       return exampleHeight + 120;
 38     }.property('exampleHeight').cacheable(),
 39 
 40     rowSpacing: 2,
 41 
 42     snippetHeight: 180,
 43 
 44     contentIndexesInRect: function(rect) {
 45       return null; // select all
 46     },
 47 
 48     contentIndexRowHeight: function(view, content, contentIndex) {
 49       return this.get('rowHeight') + 180;
 50     },
 51 
 52     createItemView: function(exampleClass, idx, attrs) {
 53       var length = this.get('length');
 54       attrs.isLastItem = idx + 1 == length;
 55 
 56       return exampleClass.create(attrs);
 57     },
 58 
 59     init: function() {
 60       sc_super();
 61 
 62       var content = this.content;
 63       for (var i = content.get('length') - 1; i >= 0; i--) {
 64         content.objectAt(i).addObserver('isShowingSnippet', this, this._sclv_customRowHeightIndexesDidChange);
 65       }
 66     }
 67 
 68 });
