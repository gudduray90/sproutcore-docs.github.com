 10 var root, content, controller, extra, flattened;
 11 
 12 var TestObject = SC.Object.extend({
 13   toString: function() { return "TestObject(%@)".fmt(this.get('title')); }
 14 });
 15 
 16 
 17 module("SC.TreeController - tree_case", {
 18   setup: function() {
 19     
 20     // setup a basic tree 
 21     content = [
 22       TestObject.create({
 23         title: "A",
 24         isExpanded: YES,
 25         outline: 0,
 26         
 27         children: [
 28           TestObject.create({ title: "A.i", outline: 1 }),
 29 
 30           TestObject.create({ title: "A.ii",
 31             outline: 1,
 32             isExpanded: NO,
 33             children: [
 34               TestObject.create({ title: "A.ii.1", outline: 2 }),
 35               TestObject.create({ title: "A.ii.2", outline: 2 }),
 36               TestObject.create({ title: "A.ii.3", outline: 2 })]
 37           }),
 38 
 39           TestObject.create({ title: "A.iii", outline: 1 })]
 40       }),
 41 
 42       TestObject.create({
 43         title: "B",
 44         isExpanded: YES,
 45         outline: 0,
 46         children: [
 47           TestObject.create({ title: "B.i",
 48             isExpanded: YES,
 49             outline: 1,
 50             children: [
 51               TestObject.create({ title: "B.i.1", outline: 2 }),
 52               TestObject.create({ title: "B.i.2", outline: 2 }),
 53               TestObject.create({ title: "B.i.3", outline: 2 })]
 54           }),
 55 
 56           TestObject.create({ title: "B.ii", outline: 1 }),
 57           TestObject.create({ title: "B.iii", outline: 1 })]
 58       }),
 59 
 60       TestObject.create({
 61         outline: 0,
 62         title: "C"
 63       })];
 64 
 65     root = TestObject.create({
 66       title: "ROOT",
 67       children: content,
 68       isExpanded: YES
 69     });
 70     
 71     flattened = [
 72       content[0],
 73       content[0].children[0],
 74       content[0].children[1],
 75       content[0].children[2],
 76       content[1],
 77       content[1].children[0],
 78       content[1].children[0].children[0],
 79       content[1].children[0].children[1],
 80       content[1].children[0].children[2],
 81       content[1].children[1],
 82       content[1].children[2],
 83       content[2]];
 84 
 85     controller = SC.TreeController.create({ 
 86       content: root,
 87       treeItemChildrenKey: "children",
 88       treeItemIsExpandedKey: "isExpanded",
 89       treeItemIsGrouped: YES 
 90     });
 91   },
 92   
 93   teardown: function() {
 94     controller.destroy();
 95   }

 97 
 98 test("arrangedObjects", function() {
 99   ok(controller.get('arangedObjects') !== controller, 'should have its own arrangedObjects');
100   
101   var ao = controller.get('arrangedObjects');
102   equals(ao.get('length'), flattened.get('length'), 'arrangedObjects should have expected length');
103   
104   var idx, len = flattened.get('length');
105   for(idx=0;idx<len;idx++) {
106     equals(ao.objectAt(idx), flattened[idx], 'arrangedObjects.objectAt(%@) should be expected object in flattened[%@]'.fmt(idx,idx));
107   }
108 });
