 86 function testStates(canLoad) {
 87   var msg, status;
 88   
 89   SC.RunLoop.begin();
 90   
 91   store.retrieveRecord(undefined, undefined, storeKey1, YES);
 92   status = store.readStatus( storeKey1);
 93   if (canLoad) {
 94     equals(status, SC.Record.BUSY_LOADING, "the status should have changed to BUSY_LOADING");
 95   } else {
 96     equals(status, SC.Record.ERROR, "the status should remain empty");
 97   }
 98   
 99   
100   store.retrieveRecord(undefined, undefined, storeKey2, YES);
101   status = store.readStatus( storeKey2);
102   if (canLoad) {
103     equals(status, SC.Record.BUSY_LOADING, "the status should have changed to BUSY_LOADING");
104   } else {
105     equals(status, SC.Record.ERROR, "the status should become empty");
106   }
107   
108   store.retrieveRecord(undefined, undefined, storeKey3, YES);
109   status = store.readStatus( storeKey3);
110   if (canLoad) {
111     equals(status, SC.Record.BUSY_LOADING, "the status should have changed to BUSY_LOADING");
112   } else {
113     equals(status, SC.Record.ERROR, "the status should become empty");
114   }
115   
116   try{
117     store.retrieveRecord(undefined, undefined, storeKey4, YES);
118     msg='';
119   }catch(error1){
120     msg=error1.message;
121   }
122   equals(msg, SC.Record.BUSY_ERROR.message, "should throw error");
123 
124   try{
125     store.retrieveRecord(undefined, undefined, storeKey5, YES);
126     msg='';
127   }catch(error2){
128     msg=error2.message;
129   }
130   equals(msg, SC.Record.BUSY_ERROR.message, "should throw error");
131   
132   try{
133     store.retrieveRecord(undefined, undefined, storeKey6, YES);
134     msg='';
135   }catch(error3){
136     msg=error3.message;
137   }
138   equals(msg, SC.Record.BUSY_ERROR.message, "should throw error");
139 
140   try{
141     store.retrieveRecord(undefined, undefined, storeKey7, YES);
142     msg='';
143   }catch(error4){
144     msg=error4.message;
145   }
146   equals(msg, SC.Record.BAD_STATE_ERROR.message, "should throw error");
147 
148 
149   store.retrieveRecord(undefined, undefined, storeKey8, YES);
150   status = store.readStatus( storeKey8);
151   if (canLoad) {
152     ok(SC.Record.BUSY_REFRESH | (status & 0x03), "the status changed to BUSY_REFRESH.");
153   } else {
154     equals(status, SC.Record.READY_CLEAN, "the status should remain ready clean");
155   }
156   
157   SC.RunLoop.end();
158 }
