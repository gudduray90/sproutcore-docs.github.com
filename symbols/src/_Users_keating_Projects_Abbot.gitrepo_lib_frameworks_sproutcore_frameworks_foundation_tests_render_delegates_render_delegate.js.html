 20 test("sc_super works.", function() {
 21 
 22   var tick = 0, base_called_on = -1, derived_called_on = -1;
 23 
 24   var base = SC.RenderDelegate.create({
 25     aMethod: function() {
 26       base_called_on = tick;
 27       tick++;
 28     }
 29   });
 30 
 31   var derived = base.create({
 32     aMethod: function() {
 33       derived_called_on = tick;
 34       tick++;
 35       sc_super();
 36     }
 37   });
 38 
 39   derived.aMethod();
 40 
 41   equals(derived_called_on, 0, "Derived method called on tick 0");
 42   equals(base_called_on, 1, "Base called on tick 1");
 43 });
 44 
 45 test("Function.prototype.enhance works.", function() {
 46 
 47   var tick = 0, base_called_on = -1, derived_called_on = -1;
 48 
 49   var base = SC.RenderDelegate.create({
 50     aMethod: function(arg1, arg2) {
 51       equals(arg1, "ARG2", "First argument is swapped");
 52       equals(arg2, "ARG1", "Second argument is swapped");
 53 
 54 
 55       base_called_on = tick;
 56       tick++;
 57     }
 58   });
 59 
 60   var derived = base.create({
 61     aMethod: function(orig, arg1, arg2) {
 62       equals(arg1, "ARG1", "First argument is correct");
 63       equals(arg2, "ARG2", "Second argument is correct");
 64 
 65       derived_called_on = tick;
 66       tick++;
 67 
 68       // swap arguments
 69       orig(arg2, arg1);
 70     }.enhance()
 71   });
 72 
 73   derived.aMethod("ARG1", "ARG2");
 74 
 75  76   equals(derived_called_on, 0, "Derived method called on tick 0");
 77   equals(base_called_on, 1, "Base called on tick 1");
 78 });
