 19 /**
 20   Add functionality to the SC.Request.manager object.
 21   
 22   */
 23 SC.ready(function() {
 24   
 25   // SC.Request.manager._simulatedResources = {};
 26   
 27   SC.Request.manager.registerSimulatedResource = function(resource) {
 28     var route;
 29     console.log("registerSimulatedResource(%@)".fmt(resource.get('name')));
 30     
 31     // Since SC.routes provides functionality to match routes, we will use it.
 32     // However, to prevent collisions with any possible app location routes, prepend a private path
 33     route = "/sc_simulated_resource_route%@".fmt(route);
 34     console.log("route = %@".fmt(route));
 35     SC.routes.add(route);
 36   };
 37   
 38   /**
 39     Invoked by the send() method on a request.  This will create a new low-
 40     level transport object and queue it if needed.
 41     
 42     @param {SC.Request} request the request to send
 43     @returns {SC.Object} response object
 44   */
 45   SC.Request.manager.sendRequest = function(request) {
 46     //console.log("sendRequest(%@)".fmt(request.get('address')));
 47     var location,
 48         params,
 49         firstRoute,
 50         route,
 51         response;
 52     
 53     if (!request) return null;
 54     
 55     location = request.get('address');
 56     location = "sc_simulated_resource_route%@".fmt(location);
 57     //console.log("location = %@".fmt(location));
 58     
 59     firstRoute = SC.routes._firstRoute;
 60     
 61     if (firstRoute) {
 62       // TODO: this may not be necessary
 63       params = SC.routes._extractParametersAndRoute({ route: location });
 64       //console.dirxml(params);
 65       location = params.route;
 66       //console.log("location = %@".fmt(location));
 67       delete params.route;
 68       delete params.params;
 69       route = firstRoute.routeForParts(location.split('/'), params);
 70     }
 71     
 72     //console.log("route = %@".fmt(route));
 73     // create low-level transport.  copy all critical data for request over
 74     // so that if the request has been reconfigured the transport will still
 75     // work.
 76     if (route) {
 77       response = SC.SimulatedResponse.create({ request: request });
 78     } else {
 79       response = request.get('responseClass').create({ request: request });
 80     }
 81     
 82     // add to pending queue
 83     this.get('pending').pushObject(response);
 84     this.fireRequestIfNeeded();
 85     
 86     return response ;
 87   };
 88 });
