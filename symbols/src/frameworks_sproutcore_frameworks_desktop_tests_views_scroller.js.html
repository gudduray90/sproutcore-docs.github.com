  8 /*global module test htmlbody ok equals same stop start */
  9 var pane, view, view1, view2 ;
 10 module("SC.ScrollerView",{
 11   setup: function() {
 12     SC.RunLoop.begin();
 13 	pane = SC.MainPane.create({
 14 	  childViews: [
 15 	    SC.ScrollerView.extend({
 16 		}),
 17 		SC.ScrollerView.extend({
 18 		  minimum:10,
 19 		  maximum:100,
 20 		  isEnabled:NO,
 21 		  layoutDirection: SC.LAYOUT_HORIZONTAL
 22 		}),
 23 		SC.ScrollerView.extend({
 24       layout:{ top: 0, bottom: 0, right: 0, width: 20 },
 25 		  minimum:0,
 26 		  maximum:100,
 27 		  hasButtons: NO
 28 		})
 29 
 30 	  ]
 31 	});
 32 	pane.append(); // make sure there is a layer...
 33 	SC.RunLoop.end();
 34 	view = pane.childViews[0];
 35 	view1= pane.childViews[1];
 36   view2= pane.childViews[2];
 37   },
 38 
 39   teardown: function() {
 40    	pane.remove();
 41    	pane = view = null ;
 42   }
 43 });
 44 
 45 test('value & displayValue', function() {
 46   equals(view1.get('maximum'), 100, 'precond - view has maximum of 100');
 47   equals(view1.get('minimum'), 10, 'precond - view has a minimum of 10');
 48   view1.set('value', 300);
 49   equals(view1.get('displayValue'), view1.get('maximum'), 'displayValue is set to maximum if attempting to set value higher than maximum');
 50   view1.set('maximum', 50);
 51   equals(view1.get('displayValue'), view1.get('maximum'), 'displayValue should change if maximum changes');
 52   view1.set('value', 0);
 53   equals(view1.get('displayValue'), view1.get('minimum'), 'displayValue is set to minimum if attempt to set value lower than minimum');
 54   view1.set('minimum', 15);
 55   equals(view1.get('displayValue'), view1.get('minimum'), 'displayValue should change if minimum changes');
 56 });
 57 
 58 test('isEnabled', function() {
 59   ok(!view.$().hasClass('disabled'), 'scrollers should be enabled by default');
 60   view.set('isEnabled', NO);
 61   SC.RunLoop.begin().end();
 62   ok(view.$().hasClass('disabled'), 'scrollers should have disabled class set if isEnabled is set to NO');
 63 
 64   var layer = view.$('.button-down'),
 65       evt = SC.Event.simulateEvent(layer, 'mousedown'),
 66       scrollerHeight = view.get('frame').height;
 67 
 68   view.set('maximum', scrollerHeight+100);
 69   view.set('value', 500);
 70   SC.Event.trigger(layer, 'mousedown', [evt]);
 71   equals(view.get('value'), 500, 'scrollers should not respond to mouse events if they are disabled');
 72   view.set('isEnabled', YES);
 73   SC.Event.trigger(layer, 'mousedown', [evt]);
 74   ok(view.get('value') < (scrollerHeight+100), 'scrollers should respond to mouse events if they are not disabled');
 75 });
 76 
 77 test('layoutDirection', function() {
 78   equals(view.get('layoutDirection'), SC.LAYOUT_VERTICAL, 'scrollers should default to vertical direction');
 79   ok(view.$().hasClass('sc-vertical'), 'scroller with vertical layoutDirection has sc-vertical class name');
 80   equals(view1.get('layoutDirection'), SC.LAYOUT_HORIZONTAL, 'precond - view1 has horizontal direction');
 81   ok(view1.$().hasClass('sc-horizontal'), 'scroller with horizontal layoutDirection has sc-horizontal class name');
 82 });
 83 
 84 test('hasButtons', function() {
 85   equals(view.get('hasButtons'), YES, 'scrollers should default to having buttons');
 86   equals(view.$('.endcap').length, 0, 'scrollers with buttons should not have an endcap');
 87   equals(view.$('.button-bottom, .button-top').length, 2, 'scrollers with buttons should have an up and a down button');
 88   equals(view2.$('.endcap').length, 1, 'scrollers with buttons should have an endcap');
 89   equals(view2.$('.button-bottom, .button-top').length, 0, 'scrollers with buttons should not have an up or a down button');  
 90 });
 91 
 92 test('hasThumb', function() {
 93   equals(view.$('.thumb').height(), view.get('thumbLength'),  'vertical scroller should have the right height');
 94   equals(view1.$('.thumb').width(), view1.get('thumbLength'), 'horizontal scroller should have the right width');
 95 });
