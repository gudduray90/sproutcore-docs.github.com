  7 /*globals TestControls */
  8 
  9 TestControls.alertController = SC.Controller.create({
 10   alertType: 'error',
 11   buttonCount: 1,
 12 
 13   message: "Hello, World!",
 14   description: "This is a description...",
 15   caption: "And this is a caption.",
 16 
 17   button1Title: "OK",
 18   button2Title: "Cancel",
 19   button3Title: "Nokay",
 20 
 21   showAlert: function() {
 22     var args = [];
 23     args.push(this.get('message'));
 24     args.push(this.get('description'));
 25     args.push(this.get('caption'));
 26     args.push(this.get('button1Title'));
 27 
 28     if (this.get('buttonCount') > 1) {
 29       args.push(this.get('button2Title'));
 30     }
 31 
 32     if (this.get('buttonCount') > 2) {
 33       args.push(this.get('button3Title'));
 34     }
 35 
 36     SC.AlertPane[this.get('alertType')].apply(SC.AlertPane, args);
 37   }
 38 });
