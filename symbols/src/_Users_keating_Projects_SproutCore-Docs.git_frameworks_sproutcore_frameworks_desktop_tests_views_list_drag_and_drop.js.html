 12 // Create a fake content array.  Generates a list with whatever length you
 13 // want of objects with a title based on the index.  Cannot mutate.
 14 var ContentArray = SC.Object.extend(SC.Array, {
 15 
 16   length: 0,
 17 
 18   objectAt: function(idx) {
 19     if (idx >= this.get('length')) { return undefined; }
 20 
 21     var content = this._content, ret ;
 22     if (!content) { content = this._content = []; }
 23 
 24     ret = content[idx];
 25     if (!ret) {
 26       ret = content[idx] = SC.Object.create({
 27         title: "ContentItem %@".fmt(idx),
 28         isDone: (idx % 3)===0,
 29         unread: (Math.random() > 0.5) ? Math.floor(Math.random() * 100) : 0
 30       });
 31     }
 32 
 33     return ret ;
 34   }
 35 });

 37 var pane = SC.ControlTestPane.design()
 38   .add("basic", SC.ScrollView.design({
 39     borderStyle: SC.BORDER_NONE,
 40     layout: { left: 0, right: 0, top: 0, height: 300 },
 41     hasHorizontalScroller: NO,
 42     contentView: SC.ListView.design({
 43       content: ContentArray.create({ length: 5 }),
 44       contentValueKey: "title",
 45       contentCheckboxKey: "isDone",
 46       contentUnreadCountKey: "unread",
 47       rowHeight: 20
 48     })
 49   }));
 50 
 51 pane.show(); // add a test to show the test pane
 52 window.pane = pane;
 53 
 54 
 55 module("SC.ListView - drag and drop", pane.standardSetup());
 56 
 57 test("drag on default list view", function() {
 58   var ev,
 59     itemView,
 60     layer,
 61     listView = pane.view("basic").get('contentView');
 62 
 63   itemView = listView.itemViewForContentIndex(0);
 64   layer = itemView.get('layer');
 65   ev = SC.Event.simulateEvent(layer, 'mousedown');
 66   SC.Event.trigger(layer, 'mousedown', [ev]);
 67 
 68   ev = SC.Event.simulateEvent(layer, 'mousemove');
 69   SC.Event.trigger(layer, 'mousemove', [ev]);
 70 
 71   equals(listView.get('dragContent'), null, 'dragContent should not be set, because the default implementation should prevent dragging');
 72 
 73   // Clean up
 74   ev = SC.Event.simulateEvent(layer, 'mouseup');
 75   SC.Event.trigger(layer, 'mouseup', [ev]);
 76 });

 79 test("drag on list view with SC.DROP_ON support", function() {
 80   var ev,
 81     itemView,
 82     layer,
 83     listView = pane.view("basic").get('contentView');
 84 
 85   // Configure the view to accept drop on.
 86   listView.set('canReorderContent', YES);
 87   listView.set('isDropTarget', YES);
 88   listView.set('delegate', SC.Object.create(SC.CollectionViewDelegate, {
 89      collectionViewValidateDragOperation: function(view, drag, op, proposedInsertionIndex, proposedDropOperation) {
 90       return SC.DRAG_ANY;
 91     }
 92   }));
 93 
 94   itemView = listView.itemViewForContentIndex(0);
 95   layer = itemView.get('layer');
 96   ev = SC.Event.simulateEvent(layer, 'mousedown');
 97   SC.Event.trigger(layer, 'mousedown', [ev]);
 98 
 99   var f = function() {
100     var itemView2,
101       point;
102 
103     SC.RunLoop.begin();
104     ev = SC.Event.simulateEvent(layer, 'mousemove');
105     SC.Event.trigger(layer, 'mousemove', [ev]);
106 
107     equals(listView.get('dragContent').content, listView.get('content'), "dragContent.content should be equal to the ListView's content");
108     ok(listView.get('dragContent').indexes.isEqual(SC.IndexSet.create(0)), "dragContent.indexes should be equal to indexes equal to [{0}]");
109     SC.RunLoop.end();
110 
111     // Drag over 2nd item
112     itemView2 = listView.itemViewForContentIndex(1);
113     layer = itemView2.get('layer');
114     point = SC.offset(layer);
115 
116     ev = SC.Event.simulateEvent(layer, 'mousemove', { pageX: point.x + 1, pageY: point.y + 1 });
117     SC.Event.trigger(layer, 'mousemove', [ev]);
118 
119 
120     ok(itemView2.get('isDropTarget'), "second list item should have isDropTarget set to true");
121 
122     // This test only works because SC.ListItemView supports adding the class to match the property.
123     ok(itemView2.$().hasClass('drop-target'), "second list item should add drop-target class");
124 
125     // Drag over 3rd item
126     itemView = listView.itemViewForContentIndex(2);
127     layer = itemView.get('layer');
128     point = SC.offset(layer);
129 
130     ev = SC.Event.simulateEvent(layer, 'mousemove', { pageX: point.x + 1, pageY: point.y + 1 });
131     SC.Event.trigger(layer, 'mousemove', [ev]);
132 
133     ok(itemView.get('isDropTarget'), "third list item should have isDropTarget set to true");
134     ok(!itemView2.get('isDropTarget'), "second list item should not have isDropTarget set to true");
135 
136     // This test only works because SC.ListItemView supports adding the class to match the property.
137     ok(itemView.$().hasClass('drop-target'), "third list item should add drop-target class");
138     ok(!itemView2.$().hasClass('drop-target'), "second list item should not add drop-target class");
139 
140     // Clean up
141     ev = SC.Event.simulateEvent(layer, 'mouseup');
142     SC.Event.trigger(layer, 'mouseup', [ev]);
143 
144     window.start();
145   };
146 
147   stop(); // stops the test runner
148   setTimeout(f, 200);
149 });
