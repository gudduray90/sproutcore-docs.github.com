 17 module("SC.RenderContext#attr", {
 18   setup: function() {
 19     context = SC.RenderContext().setAttr({ foo: 'foo' }) ;
 20   }
 21 });
 22 
 23 test("should add passed name to value", function() {
 24   context.setAttr('bar', 'bar');
 25   equals(context._attrs.bar, 'bar', 'verify attr name');
 26 });
 27 
 28 test("should replace passed name  value in attrs", function() {
 29   context.setAttr('foo', 'bar');
 30   equals(context._attrs.foo, 'bar', 'verify attr name');
 31 });
 32 
 33 test("should return receiver", function() {
 34   equals(context, context.setAttr('foo', 'bar'));
 35 });
 36 
 37 test("should create attrs hash if needed", function() {
 38   context = SC.RenderContext().begin();
 39   equals(context._attrs, null, 'precondition - has no attrs');
 40 
 41   context.setAttr('foo', 'bar');
 42   equals(context._attrs.foo, 'bar', 'has styles');
 43 });
 44 
 45 test("should assign all attrs if a hash is passed", function() {
 46   context.setAttr({ foo: 'bar', bar: 'bar' });

 48 });
