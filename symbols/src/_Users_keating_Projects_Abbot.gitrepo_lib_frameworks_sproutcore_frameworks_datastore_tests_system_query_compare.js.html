  8 // test parsing of query string
  9 var store, storeKey, rec1, rec2, rec3, rec4, rec5, MyApp, q;
 10 module("SC.Query comparison of records", {
 11   setup: function() {
 12     
 13     SC.RunLoop.begin();
 14     
 15     // setup dummy app and store
 16     MyApp = SC.Object.create({
 17       store: SC.Store.create()
 18     });
 19     
 20     // setup a dummy model
 21     MyApp.Foo = SC.Record.extend({});
 22     
 23     // load some data
 24     MyApp.store.loadRecords(MyApp.Foo, [
 25       { guid: 1, firstName: "John", lastName: "Doe", year: 1974 },
 26       { guid: 2, firstName: "Jane", lastName: "Doe", year: 1975 },
 27       { guid: 3, firstName: "Emily", lastName: "Parker", year: 1975, active: null },
 28       { guid: 4, firstName: "Johnny", lastName: "Cash", active: false },
 29       { guid: 5, firstName: "Bert", lastName: "Berthold", active: true }
 30     ]);
 31     
 32     SC.RunLoop.end();
 33     
 34     rec1 = MyApp.store.find(MyApp.Foo,1);
 35     rec2 = MyApp.store.find(MyApp.Foo,2);
 36     rec3 = MyApp.store.find(MyApp.Foo,3);
 37     rec4 = MyApp.store.find(MyApp.Foo,4);
 38     rec5 = MyApp.store.find(MyApp.Foo,5);
 39     
 40     
 41     q = SC.Query.create();
 42   }
 43 });
 44  
 45

 49 
 50 test("parse() should work with conditions = null", function(){
 51   q.parse();
 52 });
 53  
 54 test("building the order", function() {
 55   // undefined orderBy
 56   q.orderBy = null;
 57   q.parse();
 58   equals(q._order.length, 0, 'order should be empty');
 59   
 60   // empty orderBy
 61   q.orderBy = "";
 62   q.parse();
 63   equals(q._order.length, 0, 'order should be empty');
 64   
 65   // single property
 66   q.orderBy = "firstName";
 67   q.parse();
 68   equals(q._order[0].propertyName,'firstName', 'propertyName should be firstName');
 69   
 70   // more properties
 71   q.orderBy = "lastName, firstName";
 72   q.parse();
 73   equals(q._order[0].propertyName,'lastName', 'propertyName should be lastName');
 74   equals(q._order[1].propertyName,'firstName', 'propertyName should be firstName');
 75   
 76   // more properties with direction
 77   q.orderBy = "lastName, firstName, year DESC";
 78   q.parse();
 79   equals(q._order[0].propertyName,'lastName', 'propertyName should be lastName');
 80   ok(!q._order[0].descending, 'descending should be false');
 81   equals(q._order[1].propertyName,'firstName', 'propertyName should be firstName');
 82   ok(!q._order[1].descending, 'descending should be false');
 83   equals(q._order[2].propertyName,'year', 'propertyName should be year');
 84   ok(q._order[2].descending, 'descending should be true');
 85 });
 86 
 87 test("no order should result in comparison by guid", function() {
 88   q.orderBy = null;
 89   q.parse();
 90   equals(q.compare(rec1,rec2), -1, 'guid 1 should be before guid 2');
 91 });
 92 
 93 test("comparing non existent properties", function() {
 94   q.orderBy = "year";
 95   q.parse();
 96   equals(q.compare(rec5,rec1), -1, 'null should be before 1974');
 97 });
 98  
 99 test("comparing null and boolean properties", function() {
100   q.orderBy = "active";
101   q.parse();
102   equals(q.compare(rec3,rec4), -1, 'null should be before false');
103   equals(q.compare(rec4,rec5), -1, 'false should be before true');
104 });
105  
106 test("comparing number properties", function() {
107   q.orderBy = "year";
108   q.parse();
109   equals(q.compare(rec1,rec2), -1, '1974 should be before 1975');
110   
111   q.orderBy = "year DESC";
112   q.parse();
113 114   equals(q.compare(rec1,rec2), 1, '1974 should be after 1975 with DESC');
115 }); 
116  
117  
118 test("comparing string properties", function() {
119   q.orderBy = "firstName";
120   q.parse();
121   equals(q.compare(rec1,rec2), 1, 'John should be after Jane');
122   
123   q.orderBy = "firstName DESC";
124   q.parse();
125   equals(q.compare(rec1,rec2), -1, 'John should be before Jane with DESC');
126 }); 
127 
128 test("comparing by equal properties should use guid for order", function() {
129   q.orderBy = "lastName";
130   q.parse();
131   equals(q.compare(rec1,rec2), -1, 'guid 1 should be before guid 2');
132 });
133 
134 test("specifying a custom orderBy comparison function", function() {
135   var usedCustomFunction = NO;
136   q.orderBy = function(rec1, rec2) {
137     // We'll be explicit about our use of a custom comparison function, in
138     // addition to returning later years first.
139     usedCustomFunction = YES;
140     return SC.compare(rec2.get('year'), rec1.get('year'));
141   };
142   q.parse();
143   equals(q.compare(rec1,rec2), 1, 'guid 2 should be before guid 1');
144   equals(usedCustomFunction, YES, 'we should have used our custom comparison function');
145 });
