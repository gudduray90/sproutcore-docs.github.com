  8 sc_require('ext/menu_item');
  9 
 10 /**
 11   @class
 12   Extends SC.MenuPane to add support for automatic resizing.
 13 */
 14 
 15 SC.AutoResizingMenuPane = SC.MenuPane.extend(
 16 /** @scope SC.AutoResizingMenuPane.prototype */ {
 17 
 18   /**
 19     If YES, the menu should automatically resize its width to fit its items.
 20 
 21     This will swap out the default SC.MenuItemView. If you are using a custom
 22     exampleView, you will need to mix SC.AutoResize into your exampleView
 23     and set shouldAutoResize to NO (the actual resizing will be handled
 24     by SC.MenuPane).
 25 
 26     This property must be set before instantiation; any changes after instantiation
 27     will not function properly.
 28 
 29     @property
 30     @type {Boolean}
 31     @default YES
 32   */
 33   shouldAutoResize: YES,
 34 
 35   /**
 36     The minimum width for this menu if it is to be automatically resized.
 37 
 38     If no value is specified, it will be determined from the controlSize.
 39 
 40     @type Number
 41     @default minimumMenuWidth from render delegate, or 0.
 42   */
 43   minimumMenuWidth: SC.propertyFromRenderDelegate('minimumMenuWidth', 0),
 44 
 45   /**
 46     The amount to add to any calculated width.
 47 
 48     If no value is specified, it will be determined from the controlSize.
 49 
 50     @type Number
 51     @default menuWidthPadding from render delegate, or 0
 52   */
 53   menuWidthPadding: SC.propertyFromRenderDelegate('menuWidthPadding', 0),
 54 
 55   /**
 56     @private
 57     In addition to the normal init, we need to schedule an automatic resize.
 58   */
 59   init: function() {
 60     sc_super();
 61 
 62     this.set('exampleView', SC.AutoResizingMenuItemView);
 63 
 64     if (this.get('shouldAutoResize')) {
 65       this.invokeOnce('_updateMenuWidth');
 66     }
 67   },
 68 
 69   /**
 70     The array of child menu item views that compose the menu.
 71 
 72     This computed property parses @displayItems@ and constructs an SC.MenuItemView (or whatever class you have set as the @exampleView@) for every item.
 73 
 74     @property
 75     @type Array
 76     @readOnly
 77     @private
 78   */
 79   createMenuItemViews: function() {
 80     // EXTENDED to set shouldMeasureSize to its initial value and to
 81     // observe the measured size.
 82     var views = sc_super();
 83 
 84     var idx, len = views.length, view;
 85     if (this.get('shouldAutoResize')) {
 86       for (idx = 0; idx < len; idx++) {
 87         view = views[idx];
 88 
 89         // set up resizing if we want
 90         view.set('shouldMeasureSize', YES);
 91         view.addObserver('measuredSize', this, this._menuItemMeasuredSizeDidChange);
 92       }
 93     }
 94 
 95     return views;
 96   },
 97 
 98   _menuItemViewsDidChange: function() {
 99     if (this.get('shouldAutoResize')) this.invokeOnce('_updateMenuWidth');
100   }.observes('menuItemViews'),
101 
102   _menuItemMeasuredSizeDidChange: function(menuItem) {
103     this.invokeOnce('_updateMenuWidth');
104   },
105 
106   _menuMinimumMenuWidthDidChange: function() {
107     this.invokeOnce('_updateMenuWidth');
108   }.observes('minimumMenuWidth'),
109 
110   _updateMenuWidth: function() {
111     var menuItemViews = this.get('menuItemViews');
112     if (!menuItemViews) return;
113 
114     var len = menuItemViews.length, idx, view,
115         width = this.get('minimumMenuWidth');
116 
117     for (idx = 0; idx < len; idx++) {
118       view = menuItemViews[idx];
119       width = Math.max(width, view.get('measuredSize').width + this.get('menuWidthPadding'));
120     }
121 
122     this.adjust('width', width);
123     this.positionPane();
124   }
125 });
