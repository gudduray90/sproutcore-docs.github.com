  8 
  9 (function() {
 10   var pane = SC.ControlTestPane.design()
 11 
 12   .add("basic", SC.WebView, {
 13     value: "/clock",
 14     layout: {
 15       width: 250,
 16       height: 150
 17     }
 18   }).add("change src/value", SC.WebView, {
 19     value: "/clock",
 20     layout: {
 21       width: 250,
 22       height: 150
 23     }
 24   }).add("auto resize", SC.WebView, {
 25     value: sc_static('iframe'),
 26     includeImage: sc_static('a_sample_image.jpg'), // this simply ensures that the image is added to the build
 27     shouldAutoResize: YES,
 28     layout: {
 29       width: 250,
 30       height: 150
 31     }
 32   }).add("don\'t auto resize", SC.WebView, {
 33     shouldAutoResize: NO,
 34     //set the value later when you need it
 35     layout: {
 36       width: 250,
 37       height: 150
 38     }
 39   });
 40 
 41   pane.show(); // add a test to show the test pane
 42   // ..........................................................
 43   // TEST VIEWS
 44   // 
 45   module('SC.WebView UI');
 46 
 47   test("attributes tests",
 48   function() {
 49     var view = pane.view('basic');
 50     var iframe = view.$('iframe');
 51     var rootElement = view.$();
 52     ok(view, 'view should exist');
 53     ok(view.get('value'), 'should have value property');
 54     equals(view.get('shouldAutoResize'), NO, 'should have autoresize off by default, shouldAutoResize');
 55     ok(rootElement.hasClass('sc-view'), 'should have sc-view css class');
 56     ok(rootElement.hasClass('sc-web-view'), 'should have sc-web-view css class');
 57     equals(rootElement.height(), 150, 'should have height');
 58     equals(rootElement.width(), 250, 'should have width');
 59     ok(iframe, 'should have iframe element');
 60     equals(iframe.attr('src'), "/clock", "should have source as /clock");
 61     ok(iframe[0].contentWindow, 'should have content window if src is from the same domain');
 62 
 63   });
 64 
 65   test('change src/value',
 66   function() {
 67     var view = pane.view('change src/value');
 68     var iframe = view.$('iframe');
 69     equals(view.get('value'), "/clock", "should have value property in view as \'/clock\' before changing source");
 70     SC.RunLoop.begin();
 71     view.set('value', "/store_configurator");
 72     SC.RunLoop.end();
 73     equals(view.get('value'), "/store_configurator", "should have value property in view as \'/store_configurator\' after changing source");
 74     equals(iframe.attr('src'), "/store_configurator", "should have source in the iframe as \'/store_configurator\'");
 75   });
 76 
 77   test('auto resize tests, shouldAutoResize:YES',
 78   function() {
 79     var view = pane.view('auto resize');
 80     //set the test wrapper element's overflow to auto so that you can see the resize magic'
 81     view.$()[0].parentNode.style.overflow = "auto";
 82     equals(view.get('shouldAutoResize'), YES, 'should have auto resize flag as true');
 83     stop();
 84     SC.Event.add(view.$('iframe')[0], 'load', this,
 85     function() {
 86       ok(view.$().height() > 150, "height of view should change based on content");
 87       ok(view.$().width() > 200, "width of view should change based on content");
 88       start();
 89     });
 90   });
 91 
 92   test('auto resize tests, shouldAutoResize:NO',
 93   function() {
 94     var view = pane.view('don\'t auto resize');
 95     equals(view.get('shouldAutoResize'), NO, 'should have auto resize flag as false');
 96     
 97     /** set the src a bit late so that the following onload event get's hooked 
 98     up before it actually loads
 99     */
100     SC.RunLoop.begin();
101     view.set('value', sc_static('iframe'));
102     SC.RunLoop.end();
103     stop();
104     SC.Event.add(view.$('iframe')[0], 'load', this,
105     function() {
106       ok(view.$().height() === 150, "height of view should not change based on content");
107       ok(view.$().width() === 250, "width of view should not change based on content");
108       start();
109     });
110   });
111 
112 })();
