 10 module("SC.SliderView Methods", {
 11   setup: function() {
 12     // SC.RunLoop.begin();
 13     pane = SC.MainPane.create({
 14       layout: { width: 500 },
 15       childViews: [
 16         SC.SliderView.extend({
 17           value: 50,
 18           minimum: 0,
 19           maximum: 100,
 20           step: 25
 21         })]
 22     });
 23     pane.append(); // make sure there is a layer...
 24     // SC.RunLoop.end();
 25 
 26     view = pane.childViews[0];
 27   },
 28 
 29   teardown: function() {
 30     pane.remove();
 31     pane = view = null ;
 32   }
 33 });
 34 
 35 test("changing value of the slider will change its left position", function() {
 36   equals(view.get('value'), 50, 'precond - value should be 50');
 37   equals(parseFloat(view.$('.sc-handle').css('left')), 250, 'left of sc-handle should be 50%');
 38 
 39   var elem = view.get('layer');
 40 
 41   SC.RunLoop.begin();
 42   view.set('value', 100);
 43   SC.RunLoop.end();
 44 
 45   equals(view.get('value'), 100, 'value should now be 100');
 46   equals(parseFloat(view.$('.sc-handle').css('left')), 500, 'left of sc-handle should be 100%');
 47 
 48 });
 49 
 50 test("going over maximum slider limit", function() {
 51   equals(view.get('value'), 50, 'precond - value should be 50');
 52 
 53   var elem = view.get('layer');
 54 
 55   SC.RunLoop.begin();
 56   view.set('value', 150);
 57   SC.RunLoop.end();
 58 
 59   // TODO: should we allow setting value higher then maximum?
 60   // Yes I think so: the value (e.g. from a record) should not be constrained by the view layer just because its value was
 61   // bound to an unused slider. - DCP
 62   equals(view.get('value'), 150, 'value should now be 150');
 63   equals(parseFloat(view.$('.sc-handle').css('left')), 500, 'left of sc-handle should be 100%');
 64 });
 65 
 66 test("going below minimum slider limit", function() {
 67   equals(view.get('value'), 50, 'precond - value should be 50');
 68 
 69   var elem = view.get('layer');
 70 
 71   SC.RunLoop.begin();
 72   view.set('value', -10);
 73   SC.RunLoop.end();
 74 
 75   // TODO: should we allow setting value lower then minimum?
 76   // Yes I think so: the value (e.g. from a record) should not be constrained by the view layer just because its value was
 77   // bound to an unused slider. - DCP
 78   equals(view.get('value'), -10, 'value should now be -10');
 79   equals(parseFloat(view.$('.sc-handle').css('left')), 0, 'left of sc-handle should be 0%');
 80 });
 81 
 82 test("steps and stepPositions give the correct values.", function() {
 83   // This test is of course sensitive to the view's min, max and step.
 84   var steps = view.get('steps'),
 85     positions = view.get('stepPositions');
 86   ok(steps.length === 5 && steps[0] === 0 && steps[1] === 25 && steps[2] === 50 && steps[3] === 75 && steps[4] === 100,
 87     "The view's steps property returns [0, 25, 50, 75, 100].");
 88   ok(positions.length === 5 && positions[0] === 0 && positions[1] === 0.25 && positions[2] === 0.5 && positions[3] === 0.75 && positions[4] === 1,
 89     "The view's stepPositions property returns [0, 0.25, 0.5, 0.75, 1].");
 90 })
