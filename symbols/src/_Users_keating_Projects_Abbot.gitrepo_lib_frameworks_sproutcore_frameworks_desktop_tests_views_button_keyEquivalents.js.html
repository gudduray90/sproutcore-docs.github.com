 10 // performKeyEquivalent() - verify that return value is correct.
 11 //
 12 var pane = SC.ControlTestPane.design()
 13   .add("basic", SC.ButtonView, { 
 14     triggerAction: function() { return YES; },
 15     title:'hello world',
 16     keyEquivalent: 'return'
 17   });
 18   
 19 pane.show(); // add a test to show the test pane
 20 
 21   
 22  
 23 module('SC.ButtonView#performKeyEquivalent', pane.standardSetup());
 24  
 25 test("handles matching key equivalent 'return'", function() {
 26   var view = pane.view('basic');
 27   view.triggerAction = function(evt) { return YES; }; // act like we handled it if we get here
 28   var u = view.pane();
 29   ok(u.performKeyEquivalent('return'), "should return truthy value indicating it handled the key equivalent 'return'");
 30 });
 31  
 32 test("ignores non-matching key equivalent 'wrong_key'", function() {
 33   var view = pane.view('basic');
 34   view.triggerAction = function(evt) { return YES; }; // act like we handled it if we get here (we shouldn't in this case)
 35   ok(!view.performKeyEquivalent('wrong_key'), "should return falsy value indicating it ignored the non-matching key equivalent 'wrong_key'");
 36 });
 37 
 38 test("triggers on return if isDefault is set and no keyEquivalent is set", function() {
 39   var view = pane.view('basic');
 40   view.set('isDefault', YES);
 41   view.set('keyEquivalent', null);
 42   ok(view.performKeyEquivalent('return'), 'should be handled');
 43   
 44   view.set('keyEquivalent', 'a');
 45   ok(!view.performKeyEquivalent('return'), 'should NOT be handled');
 46 });
 47 
 48 
 49 test("triggers on escape if isCancel is set and no keyEquivalent is set", function() {
 50   var view = pane.view('basic');
 51   view.set('isCancel', YES);
 52   view.set('keyEquivalent', null);
 53   ok(view.performKeyEquivalent('escape'), 'should be handled');
 54   
 55   view.set('keyEquivalent', 'a');
 56   ok(!view.performKeyEquivalent('escape'), 'should NOT be handled');
 57 });
