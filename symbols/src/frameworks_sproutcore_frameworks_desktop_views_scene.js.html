  8 /** @class
  9 
 10   This view was significantly more limited than what can be achieved
 11   using the new transition plugins in SproutCore 1.10.  Please use the
 12   `transition` property in SC.ContainerView for animating between
 13   views.
 14 
 15   @extends SC.ContainerView
 16   @deprecated Version 1.10
 17   @see SC.ContainerView#transition
 18   @since Version 1.0
 19 */
 20 SC.SceneView = SC.ContainerView.extend(
 21 /** @scope SC.SceneView.prototype */ {
 22 
 23   //@if(debug)
 24   init: function () {
 25     sc_super();
 26 
 27     SC.warn("Developer Warning: SC.SceneView has been deprecated.  Please use the transition property of SC.ContainerView for animated view swapping.");
 28   },
 29   //@endif
 30 
 31   /**
 32     Array of scene names.  Scenes will slide on and off screen in the order
 33     that you specify them here.  That is, if you shift from a scene at index
 34     2 to a scene at index 1, the scenes will animation backwards.  If you
 35     shift to a scene at index 3, the scenes will animate forwards.
 36 
 37     @type Array
 38     @default ['master', 'detail']
 39   */
 40   scenes: ['master', 'detail'],
 41 
 42   /**
 43     The currently showing scene.  Changing this property will cause the
 44     scene view to transition to the new scene.  If you set this property to
 45     null, an empty string, or a non-existent scene, then the scene will appear
 46     empty.
 47 
 48     @type String
 49     @default null
 50   */
 51   nowShowing: null,
 52 
 53   /**
 54     Speed of transition.  Should be expressed in msec.
 55 
 56     @type Number
 57     @default 200
 58   */
 59   transitionDuration: 200,
 60 
 61   /** @private */
 62   _state: 'NO_VIEW', // no view
 63 
 64   /** @private
 65     Whenever called to change the content, save the nowShowing state and
 66     then animate in by adjusting the layout.
 67   */
 68   replaceContent: function(content) {
 69     if (content && this._state===this.READY) this.animateScene(content);
 70     else this.replaceScene(content);
 71     return this ;
 72   },
 73 
 74   /** @private
 75     Invoked whenever we just need to swap the scenes without playing an
 76     animation.
 77   */
 78   replaceScene: function(newContent) {
 79     var oldContent = this._targetView,
 80         layout     = this.STANDARD_LAYOUT,
 81         scenes     = this.get('scenes'),
 82         idx        = scenes ? scenes.indexOf(this.get('nowShowing')) : -1;
 83 
 84     // cleanup animation here too..
 85     this._targetView = newContent ;
 86     this._targetIndex  = idx;
 87 
 88     if (this._timer) this._timer.invalidate();
 89     this._leftView = this._rightView = this._start = this._end = null;
 90     this._timer = null;
 91 
 92 
 93     this.removeAllChildren();
 94 
 95     if (oldContent) oldContent.set('layout', layout);
 96     if (newContent) newContent.set('layout', layout);
 97 
 98     if (newContent) this.appendChild(newContent);
 99     this._state = newContent ? this.READY : this.NO_VIEW ;
100   },
101 
102   /** @private
103     Invoked whenever we need to animate in the new scene.
104   */
105   animateScene: function(newContent) {
106     var oldContent = this._targetView,
107         outIdx     = this._targetIndex,
108         scenes     = this.get('scenes'),
109         inIdx      = scenes ? scenes.indexOf(this.get('nowShowing')) : -1,
110         layout;
111 
112     if (outIdx<0 || inIdx<0 || outIdx===inIdx) {
113       return this.replaceScene(newContent);
114     }
115 
116     this._targetView = newContent ;
117     this._targetIndex = inIdx;
118 
119     // save some info needed for animation
120     if (inIdx > outIdx) {
121       this._leftView  = oldContent;
122       this._rightView = newContent;
123       this._target    = -1;
124     } else {
125       this._leftView  = newContent ;
126       this._rightView = oldContent ;
127       this._target    = 1 ;
128     }
129 
130     // setup views
131     this.removeAllChildren();
132 
133     if (oldContent) this.appendChild(oldContent)
134     if (newContent) this.appendChild(newContent);
135 
136     // setup other general state
137     this._start   = Date.now();
138     this._end     = this._start + this.get('transitionDuration');
139     this._state   = this.ANIMATING;
140     this.tick();
141   },
142 
143   /** @private
144     Called while the animation runs.  Compute the new layout for
145     the left and right views based on the portion completed.  When we finish
146     call replaceScene().
147   */
148   tick: function() {
149     this._timer = null ; // clear out
150 
151     var now    = Date.now(),
152         pct    = (now-this._start)/(this._end-this._start),
153         target = this._target,
154         left   = this._leftView,
155         right  = this._rightView,
156         layout, adjust;
157 
158     if (pct<0) pct = 0;
159 
160     // if we're done or the view is no longer visible, just replace the
161     // scene.
162     if (!this.get('isVisibleInWindow') || (pct>=1)) {
163       return this.replaceScene(this._targetView);
164     }
165 
166     // ok, now let's compute the new layouts for the two views and set them
167     layout = SC.clone(this.get('frame'));
168     adjust = Math.floor(layout.width * pct);
169 
170     // set the layout for the views, depending on the direction
171     if (target>0) {
172       layout.left = 0-(layout.width-adjust);
173       left.set('layout', layout);
174 
175       layout = SC.clone(layout);
176       layout.left = adjust ;
177       right.set('layout', layout);
178 
179     } else {
180       layout.left = 0-adjust ;
181       left.set('layout', layout);
182 
183       layout = SC.clone(layout);
184       layout.left = layout.width-adjust;
185       right.set('layout', layout);
186     }
187 
188     this._timer = this.invokeLater(this.tick, 20);
189     return this;
190   },
191 
192 
193   /** @private */
194   NO_VIEW: 'NO_VIEW',
195 
196   /** @private */
197   ANIMATING: 'ANIMATING',
198 
199   /** @private */
200   READY: 'READY',
201 
202   /** @private
203     Standard layout assigned to views at rest
204   */
205   STANDARD_LAYOUT: { top: 0, left: 0, bottom: 0, right: 0 }
206 
207 });
