 24 /*
 25   Verifies that the item views for the passed collection view match exactly the
 26   content array passed.  If shouldShowAllContent is also YES then verifies 
 27   that the nowShowing range is showing the entire content range.
 28   
 29   @param {SC.CollectionView} view the view to test
 30   @param {SC.Array} content the content array
 31   @param {Boolean} shouldShowAllContent
 32   @param {String} testName optional test name
 33   @returns {void}
 34 */
 35 function verifyItemViews(view, content, shouldShowAllContent, testName) {
 36   var nowShowing = view.get('nowShowing'),
 37       childViews = view.get('childViews');
 38 
 39   if (testName === undefined) testName='';
 40   
 41   if (shouldShowAllContent) {
 42     ok(nowShowing.isEqual(SC.IndexSet.create(0, content.get('length'))), '%@ nowShowing (%@) should equal (0..%@)'.fmt(testName, nowShowing, content.get('length')-1));
 43   }
 44   
 45   equals(childViews.get('length'), nowShowing.get('length'), '%@ view.childViews.length should match nowShowing.length'.fmt(testName));
 46   
 47   // childViews should be in same order as nowShowing indexes at all times.
 48   var iter= 0;
 49   nowShowing.forEach(function(idx) {
 50     var itemView = childViews.objectAt(iter),
 51         item     = content.objectAt(idx);
 52     ok(itemView, 'childViews[%@] should have itemView'.fmt(iter));
 53     if (itemView) {
 54       equals(itemView.get('content'), item, '%@ childViews[%@].content should equal content[%@]'.fmt(testName, iter,idx));
 55     }
 56     iter++;    
 57   });
 58 }

 64 test("should only reload when isVisibleInWindow", function() {
 65 
 66   view.set('isVisibleInWindow', NO);
 67   //view.isVisibleInWindow = NO ;
 68   
 69   var len = view.getPath('childViews.length');
 70   
 71   SC.run(function() {
 72     view.reload();
 73   });
 74 
 75   equals(view.getPath('childViews.length'), len, 'view.childViews.length should not change while offscreen');
 76   
 77   SC.RunLoop.begin();
 78   view.set('isVisibleInWindow', YES);
 79  80   SC.RunLoop.end();
 81 
 82   equals(view.getPath('childViews.length'), content.get('length'), 'view.childViews.length should change when moved onscreen if reload is pending');
 83 });

102 test("reload(index set) should update item view for items in index only", function() {
103 
104   // make sure views are loaded first time
105   SC.run(function() { view.reload(); });
106   
107   // now get a couple of child views.
108   var cv1 = view.childViews[1], cv2 = view.childViews[3];
109   
110   // and then reload them
111   SC.run(function() { view.reload(SC.IndexSet.create(1).add(3)); });
112   
113   ok(cv1 !== view.childViews[1], 'view.childViews[1] should be new instance after view.reload(<1,3>) actual: %@ expected: %@'.fmt(view.childViews[1], cv1));
114   ok(cv2 !== view.childViews[3], 'view.childViews[3] should be new instance after view.reload(<1,3>) actual: %@ expected: %@'.fmt(view.childViews[3], cv2));
115   
116   // verify integrity
117   verifyItemViews(view, content, YES);
118 });

140 test("remove and readd item", function() {
141   // first remove an item.
142   var item = content.objectAt(0);
143   SC.run(function() { content.removeAt(0); });
144   verifyItemViews(view, content, YES, 'after content.removeAt(0)');
145 
146   // then readd the item
147   SC.run(function() { content.insertAt(0, item); });
148   verifyItemViews(view, content, YES, 'after content.insertAt(0,item)');
149 
150   // then add another item
151   item = SC.Object.create();
152   SC.run(function() { content.pushObject(item); });
153   verifyItemViews(view, content, YES, 'after content.pushObject(item)');
154 
155   // and remove the item
156   SC.run(function() { content.popObject(); });
157   verifyItemViews(view, content, YES, 'after content.popObject(item)');
158 
159 });

161 test("reloading should only render nowShowing component", function() {
162   var expected = SC.IndexSet.create(0,2).add(6);
163 
164   view = SC.CollectionView.create({
165     content: content, 
166     computeNowShowing: function() {
167       return expected;
168     },
169     isVisibleInWindow: YES
170   });
171   
172   SC.RunLoop.begin();
173   view.reload();
174   SC.RunLoop.end();
175   
176   same(view.get('nowShowing'), expected, 'precond - should have limited now showing');
177   equals(view.get('childViews').get('length'), expected.get('length'), 'should only render number of child views in IndexSet');
178 });
