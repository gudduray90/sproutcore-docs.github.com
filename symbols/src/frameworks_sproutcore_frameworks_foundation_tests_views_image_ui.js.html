166   function testScale(imageView, isImg) {
167     stop();
168 
169     // Default is SC.FILL
170     imageView.addObserver('status', this, function () {
171 
172       // Status has changed, but the observer fires immediately, so pause in order to have the DOM updated
173       setTimeout(function () {
174         var imgEl,
175             innerFrame,
176             testImg = !imageView.get('useCanvas');
177 
178         // Default is SC.FILL
179         innerFrame  = imageView.get('innerFrame');
180         equals(innerFrame.width, 588, "SC.FILL width");
181         equals(innerFrame.height, 90, "SC.FILL height");
182         if (testImg) {
183           imgEl = imageView.$('img');
184 
185           equals(imgEl.css('width'), "588px", "SC.FILL width");
186           equals(imgEl.css('height'), "90px", "SC.FILL height");
187         }
188 
189         SC.run(function () {
190           imageView.set('scale', SC.SCALE_NONE);
191         });
192 
193         innerFrame = imageView.get('innerFrame');
194         equals(innerFrame.width, 271, "SC.SCALE_NONE width");
195         equals(innerFrame.height, 60, "SC.SCALE_NONE height");
196         if (testImg) {
197           equals(imgEl.css('width'), "271px", "SC.SCALE_NONE width");
198           equals(imgEl.css('height'), "60px", "SC.SCALE_NONE height");
199         }
200 
201         SC.run(function () {
202           imageView.set('scale', SC.BEST_FILL);
203         });
204 
205         innerFrame = imageView.get('innerFrame');
206         equals(innerFrame.width, 588, "SC.BEST_FILL width");
207         equals(innerFrame.height, 130, "SC.BEST_FILL height");
208         if (testImg) {
209           equals(imgEl.css('width'), "588px", "SC.BEST_FILL width");
210           equals(imgEl.css('height'), "130px", "SC.BEST_FILL height");
211         }
212 
213         SC.run(function () {
214           imageView.set('scale', SC.BEST_FIT);
215         });
216 
217         innerFrame = imageView.get('innerFrame');
218         equals(innerFrame.width, 407, "SC.BEST_FIT width");
219         equals(innerFrame.height, 90, "SC.BEST_FIT height");
220         if (testImg) {
221           equals(imgEl.css('width'), "407px", "SC.BEST_FIT width");
222           equals(imgEl.css('height'), "90px", "SC.BEST_FIT height");
223         }
224 
225         SC.run(function () {
226           imageView.set('scale', SC.BEST_FIT_DOWN_ONLY);
227         });
228 
229         innerFrame = imageView.get('innerFrame');
230         equals(innerFrame.width, 271, "SC.BEST_FIT_DOWN_ONLY width (larger frame)");
231         equals(innerFrame.height, 60, "SC.BEST_FIT_DOWN_ONLY height (larger frame)");
232         if (testImg) {
233           equals(imgEl.css('width'), "271px", "SC.BEST_FIT_DOWN_ONLY width (larger frame)");
234           equals(imgEl.css('height'), "60px", "SC.BEST_FIT_DOWN_ONLY height (larger frame)");
235         }
236 
237         SC.run(function () {
238           if (!imageView.get('useStaticLayout')) {
239             imageView.set('layout', { top: 0, left: 0, width: 147, height: 90 });
240 
241             innerFrame = imageView.get('innerFrame');
242             equals(innerFrame.width, 147, "SC.BEST_FIT_DOWN_ONLY width (smaller size frame)");
243             equals(innerFrame.height, 33, "SC.BEST_FIT_DOWN_ONLY height (smaller size frame)");
244             if (testImg) {
245               equals(imgEl.css('width'), "147px", "SC.BEST_FIT_DOWN_ONLY width (smaller size frame)");
246               equals(imgEl.css('height'), "33px", "SC.BEST_FIT_DOWN_ONLY height (smaller size frame)");
247             }
248           }
249         });
250 
251 
252         window.start(); // starts the test runner
253       }, 150);
