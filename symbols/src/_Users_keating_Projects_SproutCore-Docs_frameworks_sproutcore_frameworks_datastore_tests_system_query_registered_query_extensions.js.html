 48 test("SC.Query.queryExtensions", function(){
 49   SC.Query.registerQueryExtension('STARTS_WITH_J', {
 50     reservedWord: true,
 51     leftType: 'PRIMITIVE',
 52     evalType: 'BOOLEAN',
 53     evaluate: function (r,w) {
 54                 var word = this.leftSide.evaluate(r,w);
 55                 return ( word.substr(0,1) == 'J' );
 56               }
 57   });
 58 
 59   ok(q.queryLanguage['STARTS_WITH_J'], 'extension STARTS_WITH_J should be set');
 60   
 61   q.conditions = "firstName STARTS_WITH_J";
 62   q.parse();
 63   ok(q.contains(rec2), "Jane should match");
 64   ok(!q.contains(rec3), "Emily should not match");
 65   
 66   q.conditions = "firstName STARTS_WITH_J OR lastName STARTS_WITH_J";
 67   q.parse();
 68   ok(q.contains(rec2), "Jane Doe should match");
 69   ok(q.contains(rec5), "Bert Jules should match");
 70 
 71 });
