 10 /** @class
 11   Displays a non-modal, default positioned, drag&drop-able palette pane.
 12 
 13   The default way to use the palette pane is to simply add it to your page like this:
 14   
 15       SC.PalettePane.create({
 16         layout: { width: 400, height: 200, right: 0, top: 0 },
 17         contentView: SC.View.extend({
 18         })
 19       }).append();
 20   
 21   This will cause your palette pane to display.
 22   
 23   Palette pane is a simple way to provide non-modal messaging that won't 
 24   blocks the user's interaction with your application.  Palette panes are 
 25   useful for showing important detail informations with flexsible position.
 26   They provide a better user experience than modal panel.
 27   
 28   @extends SC.PanelPane
 29   @since SproutCore 1.0
 30 */
 31 SC.PalettePane = SC.PanelPane.extend(
 32 /** @scope SC.PalettePane.prototype */ {
 33   
 34   /**
 35     @type Array
 36     @default ['sc-palette']
 37     @see SC.View#classNames
 38   */
 39   classNames: ['sc-palette'],
 40   
 41   /**
 42     Palettes are not modal by default
 43     
 44  45     @type Boolean
 46     @default NO
 47   */
 48   isModal: NO,
 49   
 50   /**
 51     @type SC.View
 52     @default SC.ModalPane
 53   */
 54   modalPane: SC.ModalPane,
 55   
 56   /**
 57     @type Boolean
 58     @default NO
 59   */
 60   isAnchored: NO,
 61   
 62   /** @private */
 63   _mouseOffsetX: null,
 64 
 65   /** @private */
 66   _mouseOffsetY: null,
 67 
 68   /** @private
 69     Drag & drop palette to new position.
 70   */
 71   mouseDown: function(evt) {
 72     var f=this.get('frame');
 73     this._mouseOffsetX = f ? (f.x - evt.pageX) : 0;
 74     this._mouseOffsetY = f ? (f.y - evt.pageY) : 0;
 75     return YES;
 76   },
 77 
 78   /** @private */
 79   mouseDragged: function(evt) {
 80     if(!this.isAnchored) {
 81       this.set('layout', { width: this.layout.width, height: this.layout.height, left: this._mouseOffsetX + evt.pageX, top: this._mouseOffsetY + evt.pageY });
 82       this.updateLayout();
 83     }
 84     return YES;
 85   },
 86   
 87   /** @private */
 88   touchStart: function(evt){
 89     return this.mouseDown(evt);
 90   },
 91   
 92   /** @private */
 93   touchesDragged: function(evt){
 94     return this.mouseDragged(evt);
 95   }
 96 
 97 });
