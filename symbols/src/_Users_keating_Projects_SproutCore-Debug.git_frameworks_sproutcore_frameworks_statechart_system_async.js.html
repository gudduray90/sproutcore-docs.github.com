 10 /**
 11   @class
 12 
 13   Represents a call that is intended to be asynchronous. This is
 14   used during a state transition process when either entering or
 15   exiting a state.
 16 
 17   @extends SC.Object
 18   @author Michael Cohen
 19 */
 20 SC.Async = SC.Object.extend(
 21   /** @scope SC.Async.prototype */{
 22 
 23   func: null,
 24 
 25   arg1: null,
 26 
 27   arg2: null,
 28 
 29   /** @private
 30     Called by the statechart
 31   */
 32   tryToPerform: function(state) {
 33     var func = this.get('func'),
 34         arg1 = this.get('arg1'),
 35         arg2 = this.get('arg2'),
 36         funcType = SC.typeOf(func);
 37 
 38     if (funcType === SC.T_STRING) {
 39       state.tryToPerform(func, arg1, arg2);
 40     }
 41     else if (funcType === SC.T_FUNCTION) {
 42       func.apply(state, [arg1, arg2]);
 43     }
 44   }
 45 
 46  47 });
 48 
 49 /**
 50   Singleton
 51 */
 52 SC.Async.mixin(/** @scope SC.Async */{
 53 
 54   /**
 55     Call in either a state's enterState or exitState method when you
 56     want a state to perform an asynchronous action, such as an animation.
 57 
 58     Examples:
 59 
 60       SC.State.extend({
 61 
 62         enterState: function() {
 63           return SC.Async.perform('foo');
 64         },
 65 
 66         exitState: function() {
 67           return SC.Async.perform('bar', 100);
 68         }
 69  70 
        foo: function() { ... },
 71 
 72         bar: function(arg) { ... }
 73 
 74       });
 75 
 76     @param func {String|Function} the function to be invoked on a state
 77     @param arg1 Optional. An argument to pass to the given function
 78     @param arg2 Optional. An argument to pass to the given function
 79     @return {SC.Async} a new instance of a SC.Async
 80   */
 81   perform: function(func, arg1, arg2) {
 82     return SC.Async.create({ func: func, arg1: arg1, arg2: arg2 });
 83   }
 84 
 85 });
