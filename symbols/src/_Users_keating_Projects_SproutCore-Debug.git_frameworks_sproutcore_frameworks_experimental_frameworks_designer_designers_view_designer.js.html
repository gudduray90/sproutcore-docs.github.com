 11 /** @class
 12 
 13   A Designer class provides the core editing functionality you need to edit
 14   a view in the UI.  When your app loads in `design.mode`, a peer Designer
 15   instance is created for every view using the class method Designer or
 16   `SC.ViewDesigner` if the view class does not define a Designer class.
 17 
 18   Whenever you put your app into design mode, all events will be routed first
 19   to the peer designer for an object, which will have an opportunity to
 20   prosent a design UI.
 21 
 22   Likewise, the designer palettes provided by the view builder will focus on
 23   the designer instead of the view itself.
 24 
 25   ## Designer UI
 26 
 27   The basic ViewDesigner class automatically handles the UI interaction for
 28   layout.  You can also double click on the view to perform a default action.
 29 
 30   For views with `isContainerView` set to `YES`, double clicking on the view will
 31   automatically "focus" the view.  This allows you to select the view's
 32   children instead of the view itself.
 33 
 34   @extends SC.Object
 35   @since SproutCore 1.0
 36 */

141   /**
142     The current anchor location.  This determines which of the other dimension
143     metrics are actually used to compute the layout.  The value may be one of:
144 
145        TOP_LEFT, TOP_CENTER, TOP_RIGHT, TOP_HEIGHT,
146        CENTER_LEFT, CENTER_CENTER, CENTER_RIGHT, CENTER_HEIGHT
147        BOTTOM_LEFT, BOTTOM_CENTER, BOTTOM_RIGHT, BOTTOM_HEIGHT,
148        WIDTH_LEFT, WIDTH_CENTER, WIDTH_RIGHT, WIDTH_HEIGHT,
149        null
150 
151     @property
152     @type {Number}
153   */
154   anchorLocation: function(key, value) {
155     var layout = this.get('layout'),
156         K      = SC.ViewDesigner,
157         h, v, frame, view, pview, pframe, ret;
158 
159     if (!layout) return null;
160 
161     // update to refelct new anchor locations...
162     if (value !== undefined) {
163 
164       ret    = {};
165       view   = this.get('view');
166       frame  = view.get('frame');
167       pview  = view.get('parentView');
168       pframe = pview ? pview.get('frame') : null;
169       if (!pframe) pframe = SC.RootResponder.responder.computeWindowSize();
170 
171       // compute new layout in each direction
172       if (value & K.ANCHOR_LEFT) {
173         ret.left = frame.x;
174         ret.width = frame.width;
175 
176       } else if (value & K.ANCHOR_RIGHT) {
177         ret.right = (pframe.width - SC.maxX(frame));
178         ret.width = frame.width;
179 
180       } else if (value & K.ANCHOR_CENTERX) {
181         ret.centerX = Math.round(SC.midX(frame) - (pframe.width/2)) ;
182         ret.width = frame.width;
183 
184       } else if (value & K.ANCHOR_WIDTH) {
185         ret.left = frame.x;
186         ret.right = (pframe.width - SC.maxX(frame));
187       }
188 
189       // vertical
190       if (value & K.ANCHOR_TOP) {
191         ret.top = frame.y;
192         ret.height = frame.height;
193 
194       } else if (value & K.ANCHOR_BOTTOM) {
195         ret.bottom = (pframe.height - SC.maxY(frame));
196         ret.height = frame.height;
197 
198       } else if (value & K.ANCHOR_CENTERY) {
199         ret.centerY = Math.round(SC.midY(frame) - (pframe.height/2)) ;
200         ret.height = frame.height;
201 
202       } else if (value & K.ANCHOR_HEIGHT) {
203         ret.top = frame.y;
204         ret.bottom = (pframe.height - SC.maxY(frame));
205       }
206 
207       this.set('layout', ret);
208       layout = ret ;
209     }
210 
211 212     if (!SC.none(layout.left)) {
213       h = SC.none(layout.width) ? K.ANCHOR_WIDTH : K.ANCHOR_LEFT;
214     } else if (!SC.none(layout.right)) h = K.ANCHOR_RIGHT;
215     else if (!SC.none(layout.centerX)) h = K.ANCHOR_CENTERX;
216     else h = 0;
217 
218     if (!SC.none(layout.top)) {
219       v = SC.none(layout.height) ? K.ANCHOR_HEIGHT : K.ANCHOR_TOP;
220     } else if (!SC.none(layout.bottom)) v = K.ANCHOR_BOTTOM ;
221     else if (!SC.none(layout.centerY)) v = K.ANCHOR_CENTERY ;
222     else v = 0;
223 
224     return v | h;
225   }.property('layout').cacheable(),

313   /**
314     Encodes any simple properties that can just be copied from the view onto
315     the coder.  This is used by `encodeDesignProperties()` and
316     `encodeLocalizedProperties()`.
317   */
318   encodeSimpleProperties: function(props, coder) {
319     var view = this.get('view'), proto = this.get('viewClass').prototype ;
320     props.forEach(function(prop) {
321       var val = view[prop] ; // avoid get() since we don't want to exec props
322 
323       //handle bindings
324       if (prop.length > 7 && prop.slice(-7) === "Binding" && val !== undefined){
325         coder.js(prop,val.encodeDesign());
326       }
327       else{
328         if (val !== undefined && (val !== proto[prop])) {
329           coder.encode(prop, val) ;
330         }
331       }
332     }, this);
333   },

503   /**
504     Invoked whenever the view changes.  This will observe all property
505     changes on the new view.
506   */
507   viewDidChange: function() {
508     var view = this.get('view'), old = this._designer_view ;
509     if (view === old) return; // nothing to do
510 
511     var func = this.viewPropertyDidChange ;
512     if (old) old.removeObserver('*', this, func);
513     this._designer_view = view ;
514     if (view) view.addObserver('*', this, func);
515     this.viewPropertyDidChange(view, '*', null, null);
516   }.observes('view'),
517 
518   /**
519     Invoked whenever a property on the view has changed.  The passed key will
520     be '*' when the entire view has changed.  The default implementation here
521     will notify the property as changed on the receiver if the
522     property value is undefined on the receiver.
523 
524     It will notify all properties changed for '*'.  You may override this
525     method with your own behavior if you like.
526   */
527   viewPropertyDidChange: function(view, key) {
528     if (key === '*') this.allPropertiesDidChange();
529     else if (this[key] === undefined) this.notifyPropertyChange(key) ;
530 
531     if ((key === '*') || (key === 'layout')) {
532       if (this.get('designIsSelected') && this._handles) {
533         this._handles.set('layout', SC.clone(view.get('layout')));
534       }
535     }
536   },
537 
538   /**
539     The `unknownProperty` handler will pass through to the view by default.
540     This will often provide you the support you need without needing to
541     customize the Designer.  Just make sure you don't define a conflicting
542     property name on the designer itself!
543   */
544   unknownProperty: function(key, value) {
545     if (value !== undefined) {
546       this.view.set(key, value);
547       return value ;
548     } else return this.view.get(key);
549   },

695   /**
696     Select on `mouseDown`.  If `metaKey` or `shiftKey` is pressed, add to
697     selection.  Otherwise just save starting info for dragging
698   */
699   mouseDown: function(evt) {
700     this.shouldReleaseRootDesigner(evt);
701     if (!this.get('designIsEnabled') || !this.get('parentDesignerIsRoot')) return NO ;
702 
703     // save mouse down info
704     var view = this.get('view'),
705         info, vert, horiz, repos, frame, pview, margin, canH, canV;
706 
707     if (!view) return NO; // nothing to do
708 
709     // save mouse down state for later use
710     this._mouseDownInfo = info = {
711       layout:   SC.copy(view.get('layout')),
712       selected: this.get('designIsSelected'),
713       dragged:  NO,
714       metaKey:  evt.metaKey || evt.shiftKey,
715       source:   this,
716       x: evt.pageX, y: evt.pageY
717     };
718     info.hanchor = info.vanchor = info.reposition = NO;
719 
720     // detect what operations are available.
721     repos = this.get('canReposition');
722     horiz = vert = NO ;
723     if (info.selected) {
724       frame = view.get('frame');
725       pview = view.get('parentView');
726       if (frame && pview) frame = pview.convertFrameToView(frame, null);
727 
728       margin = this.HANDLE_MARGIN;
729 
730       // detect if we are in any hotzones
731       if (frame) {
732         if (Math.abs(info.x - SC.minX(frame)) <= margin) {
733           horiz = "left";
734         } else if (Math.abs(info.x - SC.maxX(frame)) <= margin) {
735           horiz = "right";
736         }
737 
738         if (Math.abs(info.y - SC.minY(frame)) <= margin) {
739           vert = "top";
740         } else if (Math.abs(info.y - SC.maxY(frame)) <= margin) {
741           vert = "bottom";
742         }
743       }
744 
745       canH = this.get('canResizeHorizontal');
746       canV = this.get('canResizeVertical');
747 
748       // look for corners if can resize in both directions...

939   /**
940     Generic resizer.  Must pass one set of keys: VKEYS, HKEYS
941   */
942   _mouseResize: function(evt, info, keys, ret) {
943 
944     var delta  = evt[keys.evtPoint] - info[keys.point],
945         layout = info.layout,
946         view   = this.get('view'),
947         min    = this.get(keys.min),
948         max    = this.get(keys.max),
949 
950         headKey   = keys.head,
951         tailKey   = keys.tail,
952         centerKey = keys.center,
953         sizeKey   = keys.size,
954 
955         hasHead   = !SC.none(layout[keys.head]),
956         hasTail   = !SC.none(layout[keys.tail]),
957         hasCenter = !SC.none(layout[keys.center]),
958         hasSize   = !SC.none(layout[keys.size]),
959         w;
960 
961     if (info[keys.anchor] === headKey) {
962 
963       // if left aligned, adjust left size and width if set.
964       if (hasHead) {
965         if (hasSize) {
966           w = layout[sizeKey];
967           ret[sizeKey] = Math.min(max, Math.max(min, Math.floor(layout[sizeKey] - delta)));
968           min = (layout[headKey]+w) - min;
969           max = (layout[headKey]+w) - max;
970 
971           ret[headKey] = Math.max(max, Math.min(min, Math.floor(layout[headKey]+delta)));
972 
973         } else {
974           ret[headKey] = Math.floor(layout[headKey]+delta);
975         }
976 
977       // if right aligned or centered, adjust the width...
978       } else if (hasTail || hasCenter) {
979         if (hasCenter) delta *= 2;
980         ret[sizeKey] = Math.max(min, Math.min(max, Math.floor((layout[sizeKey]||0)-delta)));
981 
982       // otherwise, adjust left
983       } else ret[headKey] = Math.floor((layout[headKey]||0)+delta);
984 
985     } else if (info[keys.anchor] === tailKey) {
986 
987       // reverse above
988       if (hasTail) {
989 
990         if (hasSize) {
991           w = layout[sizeKey];
992           ret[sizeKey] = Math.min(max, Math.max(min, Math.floor(layout[sizeKey] + delta)));
993           min = (layout[tailKey]+w)-min;
994           max = (layout[tailKey]+w)-max;
995 
996           ret[tailKey] = Math.max(max, Math.min(min, Math.floor(layout[tailKey]-delta)));
997 
998         } else {
999           ret[tailKey] = Math.floor(layout[tailKey]-delta);
1000         }
1001 
1002       } else {
1003         if (hasCenter) delta *= 2;
1004         ret[sizeKey] = Math.max(min, Math.min(max, Math.floor((layout[sizeKey]||0)+delta)));
1005       }
1006     }
1007 
1008     return this;
1009   },
