 12 module("SC.Statechart: - Respond to Events Tests", {
 13   setup: function() {
 14     
 15     TestState = SC.State.extend({
 16       returnValue: null,
 17       handledEvent: NO,
 18       
 19       handleEvent: function() {
 20         this.set('handledEvent', YES);
 21         return this.get('returnValue');
 22       },
 23       
 24       reset: function() {
 25         this.set('returnValue', null);
 26         this.set('handledEvent', NO);
 27       }
 28     });
 29     
 30     statechart = SC.Statechart.create({
 31       
 32       someFunctionInvoked: NO,
 33       someFunctionReturnValue: null,
 34       
 35       someFunction: function() {
 36         this.set('someFunctionInvoked', YES);
 37         return this.get('someFunctionReturnValue');
 38       },
 39       
 40       rootState: TestState.design({
 41         
 42         eventA: function(sender, context) {
 43           return this.handleEvent();
 44         },
 45         
 46         eventHandler: function(event, sender, context) {
 47           return this.handleEvent();
 48         }.handleEvents('eventB'),
 49         
 50         initialSubstate: 'a',
 51         
 52         a: TestState.design({
 53           foo: function(sender, context) {
 54             return this.handleEvent();
 55           }
 56         }),
 57         
 58         b: TestState.design({
 59           bar: function(sender, context) { 
 60             return this.handleEvent();
 61           },
 62           
 63           eventHandler: function(event, sender, context) {
 64             return this.handleEvent();
 65           }.handleEvents('frozen', 'canuck')
 66         }),
 67         
 68         c: TestState.design({
 69           eventHandlerA: function(event, sender, context) {
 70             return this.handleEvent();
 71           }.handleEvents('yes'),
 72           
 73           eventHandlerB: function(event, sender, context) {
 74             return this.handleEvent();
 75           }.handleEvents(/^num/i)
 76         }),
 77         
 78         d: TestState.design({
 79           unknownEvent: function(event, sender, context) {
 80             return this.handleEvent();
 81           }
 82         }),
 83         
 84         e: TestState.design({
 85           eventHandler: function(event,sender, context) {
 86             return this.handleEvent();
 87           }.handleEvents('plus', 'minus'),
 88           
 89           initialSubstate: 'f',
 90           
 91           f: TestState.design({
 92             foo: function(sender, context) {
 93               return this.handleEvent();
 94             }
 95           })
 96         }),
 97         
 98         z: TestState.design({
 99           blue: function(sender, context) {
100             return this.handleEvent();
101           },
102           
103           substatesAreConcurrent: YES,
104           
105           x: TestState.design({
106             yellow: function(sender, context) {
107               return this.handleEvent();
108             }
109           }),
110           
111           y: TestState.design({
112             orange: function(sender,context) {
113               return this.handleEvent();
114             }
115           })
116         })
117         
118       })
119       
120     });
121     
122     statechart.initStatechart();
123     root = statechart.get('rootState');
124     stateA = statechart.getState('a');
125     stateB = statechart.getState('b');
126     stateC = statechart.getState('c');
127     stateD = statechart.getState('d');
128     stateE = statechart.getState('e');
129     stateF = statechart.getState('f');
130     stateX = statechart.getState('x');
131     stateY = statechart.getState('y');
132     stateZ = statechart.getState('z');
133   },
134   
135   teardown: function() {
136     statechart = TestState = root = null;
137     stateA = stateB = stateC = stateD = stateE = stateF = stateX = stateY = stateZ = null;
138   }
