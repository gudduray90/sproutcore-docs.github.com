289   /**
290     This is the core method invoked to update a view layer whenever it has
291     changed.  This method simply creates a render context focused on the
292     layer element and then calls your render() method.
293 
294     You will not usually call or override this method directly.  Instead you
295     should set the layerNeedsUpdate property to YES to cause this method to
296     run at the end of the run loop, or you can call updateLayerIfNeeded()
297     to force the layer to update immediately.
298 
299     Instead of overriding this method, consider overidding the render() method
300     instead, which is called both when creating and updating a layer.  If you
301     do not want your render() method called when updating a layer, then you
302     should override this method instead.
303 
304     @param optionalContext provided only for backwards-compatibility.
305 
306     @returns {SC.View} receiver
307   */
308   updateLayer: function(optionalContext) {
309     var mixins, idx, len, hasLegacyRenderMethod;
310 
311     var context = optionalContext || this.renderContext(this.get('layer')) ;
312     this._renderLayerSettings(context, NO);
313 
314     // If the render method takes two parameters, we assume that it is a
315     // legacy implementation that takes context and firstTime. If it has only
316     // one parameter, we assume it is the render delegates style that requires
317     // only context. Note that, for backwards compatibility, the default
318     // SC.View implementation of render uses the old style.
319     hasLegacyRenderMethod = !this.update;
320     // Call render with firstTime set to NO to indicate an update, rather than
321     // full re-render, should be performed.
322     if (hasLegacyRenderMethod) {
323       this.render(context, NO);
324 325     }
326     else {
327       this.update(context.$());
328     }
329     if (mixins = this.renderMixin) {
330       len = mixins.length;
331       for(idx=0; idx<len; ++idx) { mixins[idx].call(this, context, NO) ; }
332     }
333 
334     context.update() ;
335     if (context._innerHTMLReplaced) {
336       var pane = this.get('pane');
337       if(pane && pane.get('isPaneAttached')) {
338         this._notifyDidAppendToDocument();
339       }
340     }
341 
342     // If this view uses static layout, then notify that the frame (likely)
343     // changed.
344     if (this.useStaticLayout) { this.viewDidResize(); }
345 
346     if (this.didUpdateLayer) { this.didUpdateLayer(); } // call to update DOM
347     if(this.designer && this.designer.viewDidUpdateLayer) {
348       this.designer.viewDidUpdateLayer(); //let the designer know
349     }
350     return this ;
351   },

375   /**

393     this.renderToContext(context) ;

402   /** @private -
403     Invokes the receivers didCreateLayer() method if it exists and then
404     invokes the same on all child views.
405   */
406   _notifyDidCreateLayer: function() {
407     this.notifyPropertyChange('layer');
408 
409     if (this.didCreateLayer) { this.didCreateLayer() ; }
410 
411     // and notify others
412     var mixins = this.didCreateLayerMixin, len, idx,
413         childViews = this.get('childViews'),
414         childView;
415     if (mixins) {
416       len = mixins.length ;
417       for (idx=0; idx<len; ++idx) { mixins[idx].call(this) ; }
418     }
419 
420     len = childViews.length ;
421     for (idx=0; idx<len; ++idx) {
422       childView = childViews[idx];
423       if (!childView) { continue; }
424 
425       // A parent view creating a layer might result in the creation of a
426       // child view's DOM node being created via a render context without
427       // createLayer() being invoked on the child.  In such cases, if anyone
428       // had requested 'layer' and it was cached as null, we need to
429       // invalidate it.
430       childView.notifyPropertyChange('layer');
431 
432       // A strange case, that a childView's frame won't be correct before
433       // we have a layer, if the childView doesn't have a fixed layout
434       // and we are using static layout
435       if (this.get('useStaticLayout')) {
436         if (!childView.get('isFixedLayout')) { childView.viewDidResize(); }
437       }
438 
439       childView._notifyDidCreateLayer() ;
440     }

522   /**
523     This method is called when a view changes its location in the view
524     hierarchy.  This method will update the underlying DOM-location of the
525     layer so that it reflects the new location.
526 
527     @returns {SC.View} receiver
528   */
529   updateLayerLocation: function() {
530     // collect some useful value
531     // if there is no node for some reason, just exit
532     var node = this.get('layer'),
533         parentView = this.get('parentView'),
534         parentNode = parentView ? parentView.get('containerLayer') : null ;
535 
536     // remove node from current parentNode if the node does not match the new
537     // parent node.
538     if (node && node.parentNode && node.parentNode !== parentNode) {
539       node.parentNode.removeChild(node);
540     }
541 
542     // CASE 1: no new parentView.  just remove from parent (above).
543     if (!parentView) {
544       if (node && node.parentNode) { node.parentNode.removeChild(node); }
545 
546     // CASE 2: parentView has no layer, view has layer.  destroy layer
547     // CASE 3: parentView has no layer, view has no layer, nothing to do
548     } else if (!parentNode) {
549       if (node) {
550         if (node.parentNode) { node.parentNode.removeChild(node); }
551         this.destroyLayer();
552       }
553 
554     // CASE 4: parentView has layer, view has no layer.  create layer & add
555     // CASE 5: parentView has layer, view has layer.  move layer
556     } else {
557       if (!node) {
558         this.createLayer() ;
559         node = this.get('layer') ;
560         if (!node) { return; } // can't do anything without a node.
561       }
562 
563       var siblings = parentView.get('childViews'),
564           nextView = siblings.objectAt(siblings.indexOf(this)+1),
565           nextNode = (nextView) ? nextView.get('layer') : null ;
566 
567       // before we add to parent node, make sure that the nextNode exists...
568       if (nextView && (!nextNode || nextNode.parentNode!==parentNode)) {
569         nextView.updateLayerLocationIfNeeded() ;
570         nextNode = nextView.get('layer') ;
571       }
572 
573       // add to parentNode if needed.
574       if ((node.parentNode!==parentNode) || (node.nextSibling!==nextNode)) {
575         parentNode.insertBefore(node, nextNode) ;
576       }
577     }
578 
579     parentNode = parentView = node = nextNode = null ; // avoid memory leaks
580 
581     this.set('layerLocationNeedsUpdate', NO) ;
582 
583     return this ;
584   },

607   /**
608     @private
609 
610     Renders to a context.
611     Rendering only happens for the initial rendering. Further updates happen in updateLayer,
612     and are not done to contexts, but to layers.
613     Note: You should not generally override nor directly call this method. This method is only
614     called by createLayer to set up the layer initially, and by renderChildViews, to write to
615     a context.
616 
617     @param {SC.RenderContext} context the render context.
618     @param {Boolean} firstTime Provided for compatibility when rendering legacy views only.
619   */
620   renderToContext: function(context, firstTime) {
621     var hasLegacyRenderMethod, mixins, idx, len;
622 
623     this.beginPropertyChanges() ;
624     this.set('layerNeedsUpdate', NO) ;
625 
626     if (SC.none(firstTime)) { firstTime = YES; }
627 
628     this._renderLayerSettings(context, firstTime);
629 
630     // If the render method takes two parameters, we assume that it is a
631     // legacy implementation that takes context and firstTime. If it has only
632     // one parameter, we assume it is the render delegates style that requires
633     // only context. Note that, for backwards compatibility, the default
634     // SC.View implementation of render uses the old style.
635     hasLegacyRenderMethod = !this.update;
636 
637     // Let the render method handle rendering. If we have a render delegate
638     // object set, it will be used there.
639     if (hasLegacyRenderMethod) {
640       this.render(context, firstTime);
641     }
642     // This view implements the render delegate protocol.
643     else {
644       if (firstTime) {
645         this.render(context);
646       } else {
647         this.update(context.$());
648       }
649     }
650 
651     // If we've made it this far and renderChildViews() was never called,
652     // render any child views now.
653     if (firstTime && !this._didRenderChildViews) { this.renderChildViews(context, firstTime); }
654     // Reset the flag so that if the layer is recreated we re-render the child views
655     this._didRenderChildViews = NO;
656 
657 
658     if (mixins = this.renderMixin) {
659       len = mixins.length;
660       for(idx=0; idx<len; ++idx) { mixins[idx].call(this, context, firstTime) ; }
661     }
662 
663     this.endPropertyChanges() ;
664   },

694   /**
695     @private
696 
697     Iterates over the view's `classNameBindings` array, inserts the value
698     of the specified property into the `classNames` array, then creates an
699     observer to update the view's element if the bound property ever changes
700     in the future.
701   */
702   _applyClassNameBindings: function() {
703     var classBindings = this.get('classNameBindings'),
704         classNames = this.get('classNames'),
705         elem, newClass, dasherizedClass;
706 
707     if (!classBindings) { return; }
708 
709     // Loop through all of the configured bindings. These will be either
710     // property names ('isUrgent') or property paths relative to the view
711     // ('content.isUrgent')
712     classBindings.forEach(function(property) {
713 
714       // Variable in which the old class value is saved. The observer function
715       // closes over this variable, so it knows which string to remove when
716       // the property changes.
717       var oldClass;
718 
719       // Set up an observer on the context. If the property changes, toggle the
720       // class name.
721       observer = function() {
722         // Get the current value of the property
723         newClass = this._classStringForProperty(property);
724         elem = this.$();
725 
726         // If we had previously added a class to the element, remove it.

1634 /**
1635   @class
1636 
1637   Base class for managing a view.  Views provide two functions:
1638 
1639    1. They translate state and events into drawing instructions for the
1640      web browser and
1641    2. They act as first responders for incoming keyboard, mouse, and
1642      touch events.
1643 
1644   View Initialization
1645   ====
1646 
1647   When a view is setup, there are several methods you can override that
1648   will be called at different times depending on how your view is created.
1649   Here is a guide to which method you want to override and when:
1650 
1651    - `init` -- override this method for any general object setup (such as
1652      observers, starting timers and animations, etc) that you need to happen
1653      everytime the view is created, regardless of whether or not its layer
1654      exists yet.
1655    - `render` -- override this method to generate or update your HTML to reflect
1656      the current state of your view.  This method is called both when your view
1657      is first created and later anytime it needs to be updated.
1658    - `didCreateLayer` -- the render() method is used to generate new HTML.
1659      Override this method to perform any additional setup on the DOM you might
1660      need to do after creating the view.  For example, if you need to listen
1661      for events.
1662    - `willDestroyLayer` -- if you implement didCreateLayer() to setup event
1663      listeners, you should implement this method as well to remove the same
1664      just before the DOM for your view is destroyed.
1665    - `updateLayer` -- Normally, when a view needs to update its content, it will
1666      re-render the view using the render() method.  If you would like to
1667      override this behavior with your own custom updating code, you can
1668      replace updateLayer() with your own implementation instead.
1669    - `didAppendToDocument` -- in theory all DOM setup could be done
1670      in didCreateLayer() as you already have a DOM element instantiated.
1671      However there is cases where the element has to be first appended to the
1672      Document because there is either a bug on the browser or you are using
1673      plugins which objects are not instantiated until you actually append the
1674      element to the DOM. This will allow you to do things like registering
1675      DOM events on flash or quicktime objects.
1676 
1677   @extends SC.Responder
1678   @extends SC.DelegateSupport
1679   @since SproutCore 1.0
1680 
1681 */
1682 SC.View = SC.CoreView.extend(/** @scope SC.View.prototype */{
1683   classNames: ['sc-view'],
1684   
1685   displayProperties: ['isFirstResponder']
1686 });
