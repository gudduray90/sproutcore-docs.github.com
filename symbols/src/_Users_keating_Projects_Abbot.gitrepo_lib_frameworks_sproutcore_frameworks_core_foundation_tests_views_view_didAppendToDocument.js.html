 11 var counter, pane, view, additionalView;
 12 
 13 module("SC.View#didAppendToDocument", {
 14   setup: function() {
 15     counter = 0;
 16 
 17     pane = SC.MainPane.create({
 18       childViews: [
 19         SC.View.extend({
 20           render: function (context, firstTime) {
 21             context.push('new string');
 22           },
 23           didAppendToDocument: function(){
 24             ok(document.getElementById(this.get('layerId')), "view layer should exist");
 25             counter++;
 26           }
 27         })
 28       ]
 29     });
 30     view = pane.childViews[0];
 31 
 32     additionalView = SC.View.create({
 33       didAppendToDocument: function(){
 34         ok(document.getElementById(this.get('layerId')), "additionalView layer should exist");
 35         counter++;
 36       }
 37     });
 38   },
 39 
 40   teardown: function() {
 41     pane.remove().destroy();
 42     pane = null;
 43   }
 44 });
 45 
 46 test("Check that didAppendToDocument gets called at the right moment", function() {
 47 
 48   equals(counter, 0, "precond - has not been called yet");
 49   pane.append(); // make sure there is a layer...
 50   equals(counter, 1, "didAppendToDocument was called once");
 51 
 52   SC.run(function() {
 53     view.updateLayer();
 54   });
 55 
 56   equals(counter, 2, "didAppendToDocument is called every time a new DOM element is created");
 57 
 58   pane.appendChild(additionalView);
 59 
 60   SC.RunLoop.begin().end();
 61   equals(counter, 3, "");
 62 });

 65 // Test for bug: when a childView has a non-fixed layout and we request its frame before the parentView has
 66 // a layer and the parentView uses static layout, then the frame returned will be {x: 0, y:0, width: 0, height: 0}
 67 // and any further requests for the childView's frame will not return a new value unless the parentViewDidChange
 68 // or parentViewDidResize.  A weird case, but we prevent it from failing anyhow.
 69 test("Check that childView is updated if the pane has a static layout and view doesn't have a fixed layout", function() {
 70   var childFrame,
 71       wrongFrame = {x:0, y:0, width: 0, height: 0},
 72       correctFrame;
 73 
 74   pane.set('useStaticLayout', YES);
 75 
 76   childFrame = view.get('frame');
 77   same(childFrame, wrongFrame, 'getting frame before layer exists on non-fixed layout childView should return an empty frame');
 78 
 79   SC.run(function() {
 80     pane.append(); // make sure there is a layer...
 81   });
 82   childFrame = view.get('frame');
 83   correctFrame = pane.get('frame');
 84 
 85   same(childFrame, correctFrame, 'getting frame after layer exists on non-fixed layout childView should return actual frame');
 86 });
 87 
 88 
 89 test("Check that childView is updated if it has a static layout", function() {
 90   var childFrame,
 91       wrongFrame = {x:0, y:0, width: 0, height: 0},
 92       correctFrame;
 93 
 94   view.set('useStaticLayout', YES);
 95 
 96   equals(counter, 0, "precond - has not been called yet");
 97   pane.append(); // make sure there is a layer...
 98   equals(counter, 1, "didAppendToDocument was called once");
 99 });
