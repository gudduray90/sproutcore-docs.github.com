 17 module("SC.offset", {
 18 
 19   setup: function() {
 20 
 21     htmlbody('<style> .sc-main { height: 2500px; width: 2500px; } </style>');
 22 
 23     SC.RunLoop.begin();
 24 
 25     // Even though a full SC app doesn't really allow the viewport to be scaled or scrolled by default (thus
 26     // the offset by viewport will always equal offset by document), we simulate an app that uses a
 27     // scrollable viewport to test the validity of the functions.
 28     var viewportEl;
 29     if (SC.browser.isMobileSafari) {
 30       viewportEl = $("[name='viewport']")[0];
 31 
 32       viewportEl.setAttribute('content','initial-scale=0.8, minimum-scale=0.5, maximum-scale=1.2, user-scalable=yes, width=device-height');
 33     }
 34 
 35     pane = SC.MainPane.create({
 36       childViews: [
 37         SC.View.extend({
 38           classNames: 'upper'.w(),
 39           layout: { top: 20, left: 20, width: 100, height: 100 },
 40           childViews: [
 41             SC.View.extend({
 42               classNames: 'upper-inner'.w(),
 43               layout: { top: 10, left: 10, width: 20, height: 20 }
 44             })]
 45         }),
 46         SC.View.extend({
 47           classNames: 'lower'.w(),
 48           layout: { top: 1200, left: 20, width: 100, height: 100 },
 49           childViews: [
 50             SC.View.extend({
 51               classNames: 'lower-inner'.w(),
 52               layout: { top: 10, left: 10, width: 20, height: 20 }
 53             })]
 54         })]
 55 
 56       // Useful for debugging in iOS
 57       // /** Allow default touch events */
 58       //  touchStart: function(touch) {
 59       //    if (SC.browser.isMobileSafari) touch.allowDefault();
 60       //  },
 61       //
 62       //  touchesDragged: function(evt, touches) {
 63       //    if (SC.browser.isMobileSafari) evt.allowDefault();
 64       //  },
 65       //
 66       //  touchEnd: function(touch) {
 67       //    if (SC.browser.isMobileSafari) touch.allowDefault();
 68       //  }
 69     });
 70     pane.append();
 71     SC.RunLoop.end();
 72 
 73     view1 = pane.childViews[0];
 74     view2 = pane.childViews[1];
 75     view3 = view1.childViews[0];
 76     view4 = view2.childViews[0];
 77   },
 78 
 79   teardown: function() {
 80     // Useful for debugging in iOS
 81     // if (!SC.browser.isMobileSafari) {
 82       pane.remove();
 83       pane = view1 = view2 = view3 = view4 = null;
 84     // }
 85 
 86     clearHtmlbody();
 87   }

 91 function checkDocumentOffset(element, top, left) {
 92   var docOffset = SC.offset(element, 'document');
 93 
 94   equals(docOffset.y, top, '%@ document offset top'.fmt(element[0].className));
 95   equals(docOffset.x, left, '%@ document offset left'.fmt(element[0].className));
 96 }
 97 
 98 function checkViewportOffset(element, top, left) {
 99   var viewOffset = SC.offset(element, 'viewport');
100 
101   equals(viewOffset.y, top, '%@ viewport offset top'.fmt(element[0].className));
102   equals(viewOffset.x, left, '%@ viewport offset left'.fmt(element[0].className));
103 }
104 
105 function checkParentOffset(element, top, left) {
106   var parentOffset = SC.offset(element, 'parent');
107 
108   equals(parentOffset.y, top, '%@ parent offset top'.fmt(element[0].className));
109   equals(parentOffset.x, left, '%@ parent offset left'.fmt(element[0].className));
110 }

141   window.scrollTo(100, 100);

144   checkViewportOffset(element1, -80, -80);

154 
155   checkDocumentOffset(element4, 1210, 30);
156   checkViewportOffset(element4, 1110, -70);
157   checkParentOffset(element4, 10, 10);
158
