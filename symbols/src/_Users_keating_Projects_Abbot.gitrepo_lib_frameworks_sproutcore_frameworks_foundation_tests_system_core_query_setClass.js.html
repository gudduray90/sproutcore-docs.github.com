 11 module("CoreQuery.setClass()");
 12 
 13 test("setClass(existing, NO) should remove the class", function() {
 14   var cq = SC.$('<div class="existing"></div>') ;
 15   equals(cq.hasClass('existing'), YES, "cq has existing");
 16   
 17   cq.setClass("existing", NO) ;
 18   equals(cq.hasClass('existing'), NO, 'cq should not have existing');
 19 });
 20 
 21 test("setClass(existing, YES) should do nothing", function() {
 22   var cq = SC.$('<div class="existing"></div>') ;
 23   equals(cq.hasClass('existing'), YES, "cq has existing");
 24   
 25   cq.setClass("existing", YES) ;
 26   equals(cq.hasClass('existing'), YES, 'cq should have existing');
 27 });
 28 
 29 test("setClass(new, YES) should add the class", function() {
 30   var cq = SC.$('<div class="existing"></div>') ;
 31   equals(cq.hasClass('new'), NO, "cq does not have new");
 32   
 33   cq.setClass("new", YES) ;
 34   equals(cq.hasClass('new'), YES, 'cq should have new');
 35 });
 36 
 37 test("setClass(new, NO) should do nothing", function() {
 38   var cq = SC.$('<div class="existing"></div>') ;
 39   equals(cq.hasClass('new'), NO, "cq does not have new");
 40   
 41   cq.setClass("new", NO) ;
 42   equals(cq.hasClass('new'), NO, "cq does not have new");
 43 });

 45 test("setClass(mixed, YES) should work across multiple", function() {
 46   var cq = SC.$('<div class="root">\
 47     <div class="match"></div>\
 48     <div class="match"></div>\
 49   </div>').find('.match');
 50   equals(cq.length, 2, 'should have two items');
 51 
 52   var allHaveMixed = YES;
 53   cq.each(function(el){
 54     if (!SC.$(el).hasClass('mixed')) allHaveMixed = NO;
 55   });
 56   equals(allHaveMixed, NO, "should not all have mixed class");
 57 
 58   cq.setClass("mixed", YES);
 59 
 60   allHaveMixed = YES;
 61   cq.each(function(el){
 62     if (!SC.$(el).hasClass('mixed')) allHaveMixed = NO;
 63   });
 64   equals(cq.hasClass("mixed"), YES, "now all should have mixed class") ;
 65 }) ;
