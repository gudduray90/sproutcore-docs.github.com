 13 // Sets up and tears down our pane and event.
 14 module("SC.ScrollView touch", {
 15   setup: function() {
 16 
 17     // Initialize our counters.
 18     scrollStart = scrollDragged = scrollEnd = scrollCancel = 0;
 19     innerStart = innerDragged = innerEnd = innerCancel = 0;
 20 
 21     // Create our pane.
 22     pane = SC.MainPane.extend({
 23       layout: { height: 80 },
 24       childViews: ['contentView'],
 25       contentView: SC.ScrollView.extend({
 26 
 27         touchStart: function() {
 28           scrollStart++;
 29           return sc_super();
 30         },
 31         touchesDragged: function() {
 32           scrollDragged++;
 33           return sc_super();
 34         },
 35         touchEnd: function() {
 36           scrollEnd++;
 37           return sc_super();
 38         },
 39         touchCancelled: function() {
 40           scrollCancel++;
 41           return sc_super();
 42         },
 43 
 44         contentView: SC.View.extend({
 45           layout: { height: 100 },
 46           touchStart: function() {
 47             innerStart++;
 48           },
 49           touchesDragged: function(evt, touchesForView) {
 50             innerDragged++;
 51             // If we've scrolled more than 15 pixels, pass back to the scroller.
 52             if (Math.abs(evt.startY - evt.pageY) > 15) {
 53               touchesForView.invoke('restoreLastTouchResponder');
 54             }
 55           },
 56           touchEnd: function() {
 57             innerEnd++;
 58           },
 59           touchCancelled: function() {
 60             innerCancel++;
 61           }
 62         })
 63       })
 64     });
 65     // (Actually create it, in a run loop.)
 66     SC.run(function() {
 67       pane = pane.create().append();
 68     });
 69 
 70     // Set up our pertinent reused variables.
 71     scrollView = pane.contentView;
 72     inner = scrollView.contentView;
 73     targetLayer = inner.get('layer');
 74 
 75     evt = SC.Event.simulateEvent(targetLayer, 'touchstart');
 76     evt.touches = [evt];
 77     evt.identifier = 4;
 78     evt.pageX = 50;
 79     evt.pageY = 50;
 80     evt.changedTouches = [evt];
 81   },
 82 
 83   teardown: function() {
 84     SC.run(pane.destroy, pane);
 85 
 86     pane = scrollView = inner = targetLayer = evt = null;
 87   }
 88 });

345 /**
346   Some quick earned wisdom on testing touches. When you use SC.Event.trigger to trigger a touch event,
347   the touch event's `touches` property must be an array with all of the currently operational touches
348   (i.e. those previously started and not yet ended); its `changedTouches` property must be all the touches
349   whose changes you want the root responder to acknowledge.
350 
351   `touchs` and `changedTouches` only need to be present on the event that you're passing to SC.Event.trigger;
352   you trigger changes to multiple touches in one event by including them in that touch's `changedTouches`
353   property. All `changedTouches` will be used to trigger the same event (for example, 'touchStart'), which
354   can lead to testing issues if you pass the incorrect touches.
355 */

357 test("Basic touch scale", function() {
358   equals(scrollView.get('verticalScrollOffset'), 0, "PRELIM: Vertical offset starts at");
359   equals(scrollView.get('horizontalScrollOffset'), 0, "PRELIM: Horizontal offset starts at");
360   equals(scrollView.get('scale'), 1, "PRELIM: Horizontal offset starts at");
361 
362   // Start touches.
363   evt.touches = [];
364   evt.changedTouches = [evt, evt2];
365   SC.Event.trigger(targetLayer, 'touchstart', evt);
366 
367   equals(SC.RootResponder.responder.touchesForView(scrollView).length, 2, "Two touches should result in two touches");
368 
369   // Pinch out to 2x.
370   evt.pageX = evt.pageY -= 100;
371   evt2.pageX = evt2.pageY += 100;
372   evt.touches = [evt, evt2];
373   // evt.changedTouches = [touch1, touch2];
374   SC.Event.trigger(targetLayer, 'touchmove', evt);
375 
376   // SC.ScrollView's touch-pinching depends heavily on SC.RootResponder#averagedTouchesForView. If these tests
377   // are misbehaving, first verify that SC.RootResponder's touch tests are passing.
378   equals(scrollView.get('scale'), 2, "A 2x pinch gesture should double the scroll's scale");
379   equals(scrollView.get('horizontalScrollOffset'), 500, "A centered pinch gesture should move the horizontal offset by half the content view's change in width");
380   equals(scrollView.get('verticalScrollOffset'), 500, "A centered pinch gesture should move the vertical offset by half the content view's change in height");
381 
382   // Move the gesture.
383   evt.pageX = evt.pageY += 100;
384   evt2.pageX = evt2.pageY += 100;
385   SC.Event.trigger(targetLayer, 'touchmove', evt);
386 
387   equals(scrollView.get('scale'), 2, "A gesture change with no change in distance should not change scale");
388   equals(scrollView.get('horizontalScrollOffset'), 400, "Gesture change in position by 100 should move offsets by 100");
389   equals(scrollView.get('verticalScrollOffset'), 400, "Gesture change in position by 100 should move offsets by 100");
390 
391   // Move and pinch (back to 1x) in the same gesture.
392   evt.pageX = evt.pageY = 400;
393   evt2.pageX = evt2.pageY = 600;
394   SC.Event.trigger(targetLayer, 'touchmove', evt);
395 
396   equals(scrollView.get('scale'), 1, "A pinch + move gesture should change the scale");
397   equals(scrollView.get('horizontalScrollOffset'), 0, "A pinch + move gesture should update the horizontal offset correctly");
398   equals(scrollView.get('verticalScrollOffset'), 0, "A pinch + move gesture should update the vertical offset correctly");
399 });
