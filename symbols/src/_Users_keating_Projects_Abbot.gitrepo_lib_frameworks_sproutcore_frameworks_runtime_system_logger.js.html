 89 /** @class
 90 
 91   Object to allow for safe logging actions, such as using the browser console.
 92   In addition to being output to the console, logs can be optionally recorded
 93   in memory, to be accessed by your application as appropriate.
 94 
 95   This class also adds in the concept of a “current log level”, which allows
 96   your application to potentially determine a subset of logging messages to
 97   output and/or record.  The order of levels is:
 98 
 99     -  debug        SC.LOGGER_LEVEL_DEBUG
100     -  info         SC.LOGGER_LEVEL_INFO
101     -  warn         SC.LOGGER_LEVEL_WARN
102     -  error        SC.LOGGER_LEVEL_ERROR
103 
104   All messages at the level or “above” will be output/recorded.  So, for
105   example, if you set the level to 'info', all 'info', 'warn', and 'error'
106   messages will be output/recorded, but no 'debug' messages will be.  Also,
107   there are two separate log levels:  one for output, and one for recording.
108   You may wish to only output, say, 'warn' and above, but record everything
109   from 'debug' on up.  (You can also limit the number log messages to record.)
110 
111   This mechanism allows your application to avoid needless output (which has a
112   non-zero cost in many browsers) in the general case, but turning up the log
113   level when necessary for debugging.  Note that there can still be a
114   performance cost for preparing log messages (calling {@link String.fmt},
115   etc.), so it’s still a good idea to be selective about what log messages are
116   output even to 'debug', especially in hot code.
117 
118   Similarly, you should be aware that if you wish to log objects without
119   stringification — using the {@link SC.Logger.debugWithoutFmt} variants — and
120   you enable recording, the “recorded messages” array will hold onto a
121   reference to the arguments, potentially increasing the amount of memory
122   used.
123 
124   As a convenience, this class also adds some shorthand methods to SC:
125 
126     -  SC.debug()   ==>   SC.Logger.debug()
127     -  SC.info()    ==>   SC.Logger.info()
128     -  SC.warn()    ==>   SC.Logger.warn()
129     -  SC.error()   ==>   SC.Logger.error()
130 
131   …although note that no shorthand versions exist for the less-common
132   functions, such as defining groups.
133 
134   The FireFox plugin Firebug was used as a function reference. Please see
135   [Firebug Logging Reference](http://getfirebug.com/logging.html)
136   for further information.
137 
138   @author Colin Campbell
139   @author Benedikt Böhm
140   @author William Kakes
141   @extends SC.Object
142   @since SproutCore 1.0
143   @see <a href="http://getfirebug.com/logging.html">Firebug Logging Reference</a>
144 */

306   /**

308     array, provided the respective log levels are set appropriately.

311     arguments, it is assumed to be a format string.  Thus, you can (and

322     String.fmt() call will never actually be performed.

688   /**
689     This method will output all recorded log messages to the reporter.  This
690     provides a convenient way to see the messages “on-demand” without having
691     to have them always output.  The timestamp of each message will be
692     included as a prefix if you specify 'includeTimestamps' as YES, although
693     in some browsers the native group indenting can make the timestamp
694     formatting less than ideal.
695 
696     @param {Boolean}  (optional)  Whether to include timestamps in the output
697   */
698   outputRecordedLogMessages: function(includeTimestamps) {
699     // If we have no reporter, there's nothing we can do.
700     if (!this.get('exists')) return;
701 
702     var reporter        = this.get('reporter'),
703         entries         = this.get('recordedLogMessages'),
704         indentation     = 0,
705         timestampFormat = SC.LOGGER_RECORDED_LOG_TIMESTAMP_PREFIX,
706         i, iLen, entry, type, timestampStr, message, originalArguments,
707         output, title, newIndentation, disparity, j, jLen;
708 
709     if (entries) {
710 711       for (i = 0, iLen = entries.length;  i < iLen;  ++i) {
712         entry        = entries[i];
713         type         = entry.type;
714 
715         if (includeTimestamps) {
716           timestampStr = timestampFormat.fmt(entry.timestamp.toUTCString());
717         }
718 
719         // Is this a message or a group directive?
720         message = entry.message;
721         if (message) {
722           // It's a message entry.  Were the original arguments stored?  If
723           // so, we need to use those instead of the message.
724           originalArguments = entry.originalArguments;
725           this._outputMessage(type, timestampStr, indentation, message, originalArguments);
726         }
727         else {
728           // It's a group directive.  Update our indentation appropriately.
729           newIndentation = entry.indentation;
730           title          = entry.title;
731 732           disparity      = newIndentation - indentation;
733 
734           // If the reporter implements group() and the indentation level
735           // changes by more than 1, that implies that some earlier “begin
736           // group” / “end group” directives were pruned from the beginning of
737           // the buffer and we need to insert empty groups to compensate.
738           if (reporter.group) {
739             if (Math.abs(disparity) > 1) {
740               for (j = 0, jLen = (disparity - 1);  j < jLen;  ++j) {
741                 if (disparity > 0) {
742                   reporter.group();
743                 }
744                 else {
745                   reporter.groupEnd();
746                 }
747               }
748             }
749 
750             if (disparity > 0) {
751               output = timestampStr ? timestampStr : "";
752               output += title;
753               reporter.group(output);
754             }
755             else {
756               reporter.groupEnd();
757             }
758           }
759           else {
760             // The reporter doesn't implement group()?  Then simulate it using
761             // log(), assuming it implements that.
762             if (disparity > 0) {
763               // We're beginning a group.  Output the header at an indentation
764               // that is one smaller.
765               this._outputGroup(type, timestampStr, newIndentation - 1, title);
766             }
767             // else {}  (There is no need to simulate a group ending.)
768           }
769 
770           // Update our indentation.
771           indentation = newIndentation;
772         }
773       }
774     }

779     This method will return a string representation of all recorded log
780     messages to the reporter, which can be convenient for saving logs and so
781     forth.  The timestamp of each message will be included in the string.
782 
783     If there are no recorded log messages, an empty string will be returned
784     (as opposed to null).
785 
786     @returns {String}
787   */

789     var ret           = "",
790         entries       = this.get('recordedLogMessages'),
791         indentation   = 0,
792         timestampFormat = SC.LOGGER_RECORDED_LOG_TIMESTAMP_PREFIX,
793         prefixMapping = this._LOG_FALLBACK_PREFIX_MAPPING,
794         groupHeader   = SC.LOGGER_LOG_GROUP_HEADER,
795         i, iLen, entry, type, message, originalArguments, prefix, line,
796         title, newIndentation, disparity;
797 
798     if (entries) {
799       for (i = 0, iLen = entries.length;  i < iLen;  ++i) {
800         entry = entries[i];
801         type  = entry.type;
802

804         prefix = timestampFormat.fmt(entry.timestamp.toUTCString());

816         }
817         else {

819           // appropriately.  Also, if it's the beginning of the group and it

832         }
833

835         ret += line + "\n";
836       }

1099     Outputs and/or records the specified message of the specified type if the
1100     respective current log levels allow for it.  Assuming
1101     'automaticallyFormat' is specified, then String.fmt() will be called
1102     automatically on the message, but only if at least one of the log levels
1103     is such that the result will be used.
1104 
1105     @param {String}               type                 Expected to be SC.LOGGER_LEVEL_DEBUG, etc.
1106     @param {Boolean}              automaticallyFormat  Whether or not to treat 'message' as a format string if there are additional arguments
1107     @param {String}               message              Expected to a string format (for String.fmt()) if there are other arguments
1108     @param {String}   (optional)  originalArguments    All arguments passed into debug(), etc. (which includes 'message'; for efficiency, we don’t copy it)
1109   */
1110   _handleMessage: function(type, automaticallyFormat, message, originalArguments) {
1111     // Are we configured to show this type?
1112     var shouldOutput = this._shouldOutputType(type),
1113         shouldRecord = this._shouldRecordType(type),
1114         hasOtherArguments, i, len, args, output, entry;
1115 
1116     // If we're neither going to output nor record the message, then stop now.
1117     if (!(shouldOutput || shouldRecord)) return;
1118 
1119     // Do we have arguments other than 'message'?  (Remember that
1120     // 'originalArguments' contains the message here, too, hence the > 1.)
1121     hasOtherArguments = (originalArguments  &&  originalArguments.length > 1);
1122 
1123     // If we're automatically formatting and there is no message (or it is
1124     // not a string), then don't automatically format after all.
1125     if (automaticallyFormat  &&  (SC.none(message)  ||  (typeof message !== "string"))) {
1126       automaticallyFormat = NO;
1127     }
1128 
1129     // If we should automatically format, and the client specified other
1130     // arguments in addition to the message, then we'll call .fmt() assuming
1131     // that the message is a format string.
1132     if (automaticallyFormat) {
1133       if (hasOtherArguments) {
1134         args = [];
1135         for (i = 1, len = originalArguments.length;  i < len;  ++i) {
1136           args.push(originalArguments[i]);
1137         }
1138         message = message.fmt.apply(message, args);
1139       }
1140     }
1141

1143       // We only want to pass the original arguments to _outputMessage() if we
1144       // didn't format the message ourselves.
1145       args = automaticallyFormat ? null : originalArguments;
1146       this._outputMessage(type, null, this._outputIndentationLevel, message, args);
1147     }
1148 
1149     // If we're recording the log, append the message now.
1150     if (shouldRecord) {
1151       entry = {
1152         type:      type,
1153         message:   message ? message : YES,
1154         timestamp: new Date()
1155       };
1156 
1157       // If we didn't automatically format, and we have other arguments, then
1158       // be sure to record them, too.
1159       if (!automaticallyFormat  &&  hasOtherArguments) {
1160         entry.originalArguments = originalArguments;
1161       }
1162

1313   /** @private
1314     Outputs the specified message to the current reporter.  If the reporter
1315     does not handle the specified type of message, it will fall back to using
1316     log() if possible.
1317 
1318     @param {Constant}               type
1319     @param {String}                 timestampStr       An optional timestamp prefix for the line, or null for none
1320     @param {Number}                 indentation        The current indentation level
1321     @param {String}                 message
1322     @param {Arguments}  (optional)  originalArguments  If specified, the assumption is that the message was not automatically formatted
1323   */
1324   _outputMessage: function(type, timestampStr, indentation, message, originalArguments) {
1325     if (!this.get('exists')) return;
1326 
1327     // Do we have reporter[type] defined as a function?  If not, we'll fall
1328     // back to reporter.log if that exists.
1329     var reporter = this.get('reporter'),
1330         output, shouldIndent, func, prefix, args, arg;
1331 
1332     // If the reporter doesn't support group(), then we need to manually
1333     // include indentation for the group.  (It it does, we'll assume that
1334     // we're currently at the correct group level.)
1335     shouldIndent = !reporter.group;
1336 
1337     // Note:  Normally we wouldn't do the hash dereference twice, but
1338     //        storing the result like this:
1339     //
1340     //          var nativeFunction = console[type];
1341     //          nativeFunction(output);
1342     //
1343     //        …doesn't work in Safari 4, and:
1344     //
1345     //          nativeFunction.call(console, output);
1346     //
1347     //        …doesn't work in IE8 because the console.* methods are
1348     //       reported as being objects.
1349     func = reporter[type];
1350     if (func) {
1351       // If we formatted, just include the message.  Otherwise, include all
1352       // the original arguments.
1353       if (!originalArguments) {
1354         output = "";
1355         if (timestampStr) output = timestampStr;
1356         if (shouldIndent) output =+ this._indentation(indentation);
1357         output += message;
1358         reporter[type](output);
1359       }
1360       else {
1361         // We have arguments?  Then pass them along to the reporter function
1362         // so that it can format them appropriately.  We'll use the timestamp
1363         // string (if there is one) and the indentation as the first
1364         // arguments.
1365         args = this._argumentsToArray(originalArguments);
1366         prefix = "";
1367         if (timestampStr) prefix = timestampStr;
1368         if (shouldIndent) prefix += this._indentation(indentation);
1369         if (prefix) args.splice(0, 0, prefix);
1370         
1371         if (func.apply) {
1372           func.apply(reporter, args);
1373         }
1374         else {
1375           // In IE8, passing the arguments as an array isn't ideal, but it's
1376           // pretty much all we can do because we can't call apply().
1377           reporter[type](args);            
1378         }
1379       }
1380     }
1381     else {
1382       // The reporter doesn't support the requested function?  If it at least
1383       // support log(), fall back to that.
1384       if (reporter.log) {
1385         prefix = "";
1386         if (timestampStr) prefix = timestampStr;
1387         prefix += this._LOG_FALLBACK_PREFIX_MAPPING[type] || "";
1388         if (shouldIndent) prefix += this._indentation(indentation);
1389 
1390         // If we formatted, just include the message.  Otherwise, include
1391         // all the original arguments.
1392         if (!originalArguments) {
1393           reporter.log(prefix + message);
1394         }
1395         else {
1396           args = this._argumentsToArray(originalArguments);
1397           if (prefix) args.splice(0, 0, prefix);
1398           reporter.log(args);
1399         }
1400       }
1401     }
1402   },

1405   /** @private
1406     Outputs the specified “begin group” directive to the current reporter.  If
1407     the reporter does not handle the group() method, it will fall back to
1408     simulating using log() if possible.
1409 
1410     @param {Constant}               type
1411     @param {String}                 timestampStr  An optional timestamp prefix for the line, or null for none
1412     @param {Number}                 indentation   The current indentation level, not including what the group will set it to
1413     @param {String}     (optional)  title
1414   */
1415   _outputGroup: function(type, timestampStr, indentation, title) {
1416     if (!this.get('exists')) return;
1417 
1418     // Do we have reporter.group defined as a function?  If not, we'll fall
1419     // back to reporter.log if that exists.  (Thankfully, we can avoid the IE8
1420     // special-casing we have in _outputMessage() because IE8 doesn't support
1421     // console.group(), anyway.)
1422     var reporter = this.get('reporter'),
1423         func     = reporter.group,
1424         output;
1425 
1426     if (func) {
1427       output = timestampStr ? timestampStr : "";
1428       output += title;
1429       func.call(reporter, output);
1430     }
1431     else if (reporter.log) {
1432 1433       // The reporter doesn't support group()?  Then simulate with log().
1434       // (We'll live with the duplicitous dereference rather than using
1435       // apply() to work around the IE8 issue described in _outputMessage().)
1436       output = "";
1437       if (timestampStr) output = timestampStr;
1438       output += this._LOG_FALLBACK_PREFIX_MAPPING[type] || "";
1439       output += this._indentation(indentation);
1440       output += SC.LOGGER_LOG_GROUP_HEADER.fmt(title);
1441       reporter.log(output);
1442     }
1443   },

1446   /** @private
1447     This method will add the specified entry to the recorded log messages
1448     array and also prune array as necessary according to the current values of
1449     'recordedLogMessagesMaximumLength' and
1450     'recordedLogMessagesPruningMinimumLength'.
1451   */
1452   _addRecordedMessageEntry: function(entry) {
1453     var recordedMessages = this.get('recordedLogMessages'),
1454         len;
1455 
1456     // Lazily create the array.
1457     if (!recordedMessages) {
1458       recordedMessages = [];
1459       this.set('recordedLogMessages', recordedMessages);
1460     }
1461 
1462     recordedMessages.push(entry);
1463 
1464     // Have we exceeded the maximum size?  If so, do some pruning.
1465     len = recordedMessages.length;
1466     if (len > this.get('recordedLogMessagesMaximumLength')) {
1467       recordedMessages.splice(0, (len - this.get('recordedLogMessagesPruningMinimumLength')));
1468     }
1469     
1470     // Notify that the array content changed.
1471     recordedMessages.enumerableContentDidChange();
1472   },

1478     copies the elements of arguments to a blank array.

1480     @param {Array} arguments The arguments property of a function

1486

1488       for (i = 0, len = args.length;  i < len;  ++i) {
1489         ret[i] = args[i];

1491     }
