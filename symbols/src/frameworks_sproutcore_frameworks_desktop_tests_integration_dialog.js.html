 11 var pane ;
 12 
 13 module("A dialog with some basic controls and buttons", {
 14   teardown: function(){
 15     if (pane) pane.remove();
 16   }
 17 });
 18 
 19 test("adding dialog to screen", function() {
 20 
 21 
 22   var delegate = SC.Object.create({
 23 
 24     couldNotSend: function() {
 25       pane = SC.AlertPane.warn({
 26         message: "Email could not be sent",
 27         description: 'There might be a problem with the server or with your internet connection.  Try again in a few minutes.',
 28         buttons: [
 29           { title: "Try Again" },
 30           { title: "Cancel" },
 31           { title: 'Report Problem...' }
 32         ],
 33         delegate: this
 34       });
 35     },
 36 
 37     showMoreInfo: function() {
 38       pane = SC.AlertPane.info({
 39         message: "Sending Email",
 40         description: "Sometimes email doesn't make it.  It's a fact of life.  We all love email, but hey that's how it goes.\n" +
 41                       "Anyway, the nice thing is that we can provide this helpful dialog message, with multiple paragraphs and everything because of SproutCore.\n" +
 42                       "Email is OK, AlertPanes are great. So just deal m'kay?  Bye bye.",
 43         delegate: delegate
 44       });
 45     },
 46 
 47     alertPaneDidDismiss: function(alert, status) {
 48       switch(status) {
 49         case SC.OK_STATUS:
 50           this.invokeLater(this.couldNotSend, 1000);
 51           break;
 52         case SC.EXTRA_STATUS:
 53           this.showMoreInfo();
 54           break;
 55       }
 56     }
 57   });
 58 
 59   SC.RunLoop.begin();
 60   delegate.couldNotSend();
 61   SC.RunLoop.end();
 62 }) ;
