 25 test("basic VERTICAL_STACK", function () {
 26   SC.run(function() {
 27     view = SC.View.create({
 28 
 29       childViewLayout: SC.View.VERTICAL_STACK,
 30 
 31       childViewLayoutOptions: {
 32         paddingBefore: 10,
 33         paddingAfter: 20,
 34         spacing: 5
 35       },
 36       childViews: ['sectionA', 'sectionB', 'sectionC'],
 37       layout: { left: 10, right: 10, top: 20 },
 38       sectionA: SC.View.design({
 39         layout: { height: 100 }
 40       }),
 41 
 42       sectionB: SC.View.design({
 43         layout: { border: 1, height: 50 }
 44       }),
 45 
 46       sectionC: SC.View.design({
 47         layout: { left: 10, right: 10, height: 120 }
 48       })
 49 
 50     });
 51   });
 52 
 53   equals(view.sectionA.layout.top, 10, "sectionA top should be 10");
 54   equals(view.sectionB.layout.top, 115, "sectionB top should be 115");
 55   equals(view.sectionC.layout.top, 170, "sectionC top should be 170");
 56   equals(view.layout.height, 310, "view height should be 310");
 57 
 58 });

 93 test("HORIZONTAL_STACK - with fillRatio", function () {
 94   var view = null;
 95   SC.run(function() {
 96     view = SC.View.create({
 97       layout: { height: 10, width: 200 },
 98       childViewLayout: SC.View.HORIZONTAL_STACK,
 99       childViewLayoutOptions: {
100         resizeToFit: NO,
101         paddingBefore: 10,
102         paddingAfter: 20,
103         spacing: 10
104       },
105       childViews: [ "c1", "c2", "c3", "c4", "c5" ],
106 
107       c1: SC.View.create({
108         layout: { height: 10, width: 10 },
109       }),
110       c2: SC.View.create({
111         layout: { height: 10 },
112         fillRatio: 1
113       }),
114       c3: SC.View.create({
115         layout: { height: 10, width: 10 },
116       }),
117       c4: SC.View.create({
118         layout: { height: 10 },
119         fillRatio: 2
120       }),
121       c5: SC.View.create({
122         layout: { height: 10 },
123         border: 1,
124         isVisible: NO
125       }),
126     });
127   });
128 
129   equals(view.c1.layout.left, 10, "c1 left should be 10");
130   equals(view.c2.layout.left, 30, "c2 left should be 10");
131   equals(view.c2.get( "borderFrame" ).width, 40, "c2 width should be 40");
132   equals(view.c3.layout.left, 80, "c3 left should be 10");
133   equals(view.c4.layout.left, 100, "c4 left should be 10");
134   equals(view.c4.get( "borderFrame" ).width, 80, "c4 width should be 80");
135 
136   SC.run(function() {
137     view.c2.set( "isVisible", NO );
138     view.c4.set( "isVisible", NO );
139     view.c5.set( "isVisible", YES );
140   });
141 
142   equals(view.c1.layout.left, 10, "c1 left should be 10");
143   equals(view.c3.layout.left, 30, "c3 left should be 30");
144   equals(view.c5.layout.left, 50, "c5 left should be 50");
145   equals(view.c5.get( "borderFrame" ).width, 130, "being the last child, c5 will extend to fill the available space, width should be 130");
146 });
