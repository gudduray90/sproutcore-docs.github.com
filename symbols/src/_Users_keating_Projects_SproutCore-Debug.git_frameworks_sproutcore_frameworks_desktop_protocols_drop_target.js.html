  8 sc_require('system/drag');

 10 /**
 11   @namespace
 12   
 13   Implement the DropTarget protocol in your view to be able to accept drop events. You 
 14   should define the methods below as needed to handle accepting of events.
 15   
 16   See the method descriptions for more information on what you need to implement.
 17   
 18   The general call sequence for all drop targets is (in pseudo-Ragel, regex
 19   format):
 20   
 21       dragStarted
 22       (
 23         computeDragOperations+
 24         (
 25           dragEntered
 26           dragUpdated
 27           ( computeDragOperations | dragUpdated )*
 28           ( acceptDragOperation performDragOperation? )? // mouseUp
 29           dragExited
 30         )*
 31       )*
 32       dragEnded
 33   
 34   Thus, every drop target will have its dragStarted and dragEnded methods called 
 35   once during every drag session. computeDragOperations, if called at all, may be 
 36   called more than once before the dragEntered method is called. Once dragEntered 
 37   is called, you are at guaranteed that both dragUpdated and dragExited will be 
 38   called at some point, followed by either dragEnded or additional 
 39   computeDragOperation calls.
 40 */
 41 SC.DropTarget = {
 42   
 43   /**
 44     Must be true when your view is instantiated.
 45     
 46     Drop targets must be specially registered in order to receive drop
 47     events.  SproutCore knows to register your view when this property
 48     is true on view creation.
 49     
 50     @type Boolean
 51     @default YES
 52     @constant
 53   */  
 54   isDropTarget: YES,
 55 
 56   /**
 57     Called when the drag is started, regardless of where or not your drop
 58     target is current. You can use this to highlight your drop target
 59     as "eligible".
 60     
 61     The default implementation does nothing.
 62     
 63     @param {SC.Drag} drag The current drag object.
 64     @param {SC.Event} evt The most recent mouse move event.  Use to get location 
 65   */
 66   dragStarted: function(drag, evt) {},
 67   
 68   /**
 69     Called when the drag first enters the droppable area, if it returns a
 70     drag operations other than `SC.DRAG_NONE`.
 71     
 72     The default implementation does nothing.
 73     
 74     @param drag {SC.Drag} The current drag object.
 75     @param evt {SC.Event} The most recent mouse move event.  Use to get location
 76   */
 77   dragEntered: function(drag, evt) {},
 78   
 79   /**
 80     Called periodically when a drag is over your droppable area.
 81     
 82     Override this method this to update various elements of the drag state, 
 83     including the location of ghost view.  You should  use this method to 
 84     implement snapping.
 85     
 86     This method will be called periodically, even if the user is not moving
 87     the drag.  If you perform expensive operations, be sure to check the
 88     mouseLocation property of the drag to determine if you actually need to
 89     update anything before doing your expensive work.
 90     
 91     The default implementation does nothing.
 92     
 93     @param {SC.Drag} drag The current drag object.
 94     @param {SC.Event} evt The most recent mouse move event. Use to get location
 95   */
 96   dragUpdated: function(drag, evt) {},
 97   
 98   /**
 99     Called when the user exits your droppable area or the drag ends
100     and you were the last targeted droppable area.
101     
102     Override this method to perform any clean up on your UI such as hiding 
103     a special highlight state or removing insertion points.
104     
105     The default implementation does nothing.
106     
107     @param {SC.Drag} drag The current drag object
108     @param {SC.Event}   evt  The most recent mouse move event. Use to get location.
109   */
110   dragExited: function(drag, evt) {},
111   
112   /**
113     Called on all drop targets when the drag ends.  
114     
115     For example, the user might have dragged the view off the screen and let 
116     go or they might have hit escape.  Override this method to perform any 
117     final cleanup.  This will be called instead of dragExited.
118     
119     The default implementation does nothing.
120     
121     @param {SC.Drag} drag The current drag object
122     @param {SC.Event}   evt  The most recent mouse move event. Use to get location.
123   */
124   dragEnded: function(drag, evt) {},
125   
126   /**
127     Called when the drag needs to determine which drag operations are
128     valid in a given area.
129     
130     Override this method to return an OR'd mask of the allowed drag 
131     operations.  If the user drags over a droppable area within another 
132     droppable area, the drag will latch onto the deepest view that returns one 
133     or more available operations.
134     
135     The default implementation returns `SC.DRAG_NONE`
136     
137     @param {SC.Drag} drag The current drag object
138     @param {SC.Event} evt The most recent mouse move event.  Use to get 
139       location 
140     
141     @returns {DragOps} A mask of all the drag operations allowed or 
142       SC.DRAG_NONE
143   */
144   computeDragOperations: function(drag, evt) {
145     return SC.DRAG_NONE;
146   },
147   
148   /**
149     Called when the user releases the mouse.
150     
151     This method gives your drop target one last opportunity to choose to 
152     accept the proposed drop operation.  You might use this method to
153     perform fine-grained checks on the drop location, for example.
154     Return true to accept the drop operation.
155     
156     The default implementation returns `YES`.
157     
158     @param {SC.Drag} drag The drag instance managing this drag
159     @param {DragOp} op The proposed drag operation. A drag constant
160     
161     @return {Boolean} YES if operation is OK, NO to cancel.
162   */  
163   acceptDragOperation: function(drag, op) {
164     return YES;
165   },
166   
167   /**
168     Called to actually perform the drag operation.
169     
170     Override this method to actually perform the drag operation.  This method
171     is only called if you returned `YES` in `acceptDragOperation()`. 
172     
173     Return the operation that was actually performed or `SC.DRAG_NONE` if the 
174     operation was aborted.
175     
176     The default implementation returns `SC.DRAG_NONE`
177     
178     @param {SC.Drag} drag The drag instance managing this drag
179     @param {DragOp} op The proposed drag operation. A drag constant.
180     
181     @return {DragOp} Drag Operation actually performed
182   */
183   performDragOperation: function(drag, op) {
184     return SC.DRAG_NONE;
185   }
186   
187 };
