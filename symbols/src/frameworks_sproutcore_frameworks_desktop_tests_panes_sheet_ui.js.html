 10 module("SC.SheetPane UI");
 11 
 12 test("verify sheet pane slide down works", function() {
 13   var slidePane = SC.SheetPane.create({
 14     layout: { width: 400, height: 200, centerX: 0 },
 15     contentView: SC.LabelView.extend({
 16       escapeHTML: NO,
 17       value: '<h1>Slide Down!</h1>'
 18     })
 19   });
 20 
 21   var layout = slidePane.get('layout');
 22   var pt = layout.top;
 23   var pl = layout.left;
 24   var pw = layout.width;
 25   var ph = layout.height;
 26   var ret = slidePane.layoutStyle();
 27 
 28   slidePane.append();
 29   var f = function() {
 30     // make sure all fo the timers have had an opportunity to fire
 31     SC.RunLoop.begin().end();
 32     equals(slidePane.get('layout').top, 0, 'pane should be displayed at default position top after animating');
 33     ok(slidePane.get('isVisibleInWindow'), 'pane.isVisibleInWindow should be YES');
 34     ok(slidePane.$().hasClass('sc-sheet'), 'pane should have sc-sheet class');
 35     ok(slidePane.childViews[0].get('isVisibleInWindow'), 'pane.div.isVisibleInWindow should be YES');
 36     ok(slidePane.childViews[0].$().hasClass('sc-view'), 'pane.div should have sc-view class');
 37     SC.RunLoop.begin();
 38     slidePane.remove();
 39     slidePane.destroy();
 40     SC.RunLoop.end();
 41 
 42     window.start();
 43   };
 44 
 45   stop(800);
 46 
 47   setTimeout(f, 400);
 48 
 49 });
