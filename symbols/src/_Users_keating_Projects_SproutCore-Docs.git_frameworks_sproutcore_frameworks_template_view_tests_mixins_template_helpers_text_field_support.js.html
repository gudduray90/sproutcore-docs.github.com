  7 (function() {
  8   var TestObject, textFieldView, pane;
  9 
 10   module("Text Field Support", {
 11     setup: function() {
 12       TestObject = window.TestObject = SC.Object.create({
 13         value: null
 14       });
 15 
 16       textFieldView = SC.TemplateView.create(SC.TextFieldSupport, {
 17         template: SC.Handlebars.compile('<input type="text">')
 18       });
 19 
 20       pane = SC.MainPane.create({
 21         childViews: [textFieldView]
 22       });
 23       pane.append();
 24     },
 25 
 26     teardown: function() {
 27       pane.remove();
 28       TestObject = window.TestObject = textFieldView = pane = null;
 29     }
 30   });
 31 
 32   test("value property mirrors input value", function() {
 33     textFieldView.$('input').val('foo bar');
 34 
 35     equals(textFieldView.get('value'), 'foo bar', "gets value property from DOM");
 36 
 37     textFieldView.set('value', "afterlife");
 38     equals(textFieldView.$('input').val(), "afterlife", "sets value of DOM to value property");
 39   });
 40 
 41   // Not really sure how to test this without doing something like adding a selection then checking 
 42   // to see if it's still there after setting value
 43   test("only update DOM if value changed");
 44 
 45   test("value binding works properly for inputs that haven't been created", function() {
 46     var view = SC.TemplateView.create(SC.TextFieldSupport, {
 47       template: SC.Handlebars.compile('<input type="text">'),
 48       valueBinding: 'TestObject.value'
 49     });
 50 
 51     equals(view.get('value'), null, "precond - default value is null");
 52     equals(view.$('input').length, 0, "precond - view doesn't have its layer created yet, thus no input element");
 53 
 54     SC.run(function() { TestObject.set('value', 'ohai'); });
 55 
 56     equals(view.get('value'), 'ohai', "value property was properly updated");
 57 
 58     SC.run(function() { pane.appendChild(view); });
 59 
 60     equals(view.get('value'), 'ohai', "value property remains the same once the view has been appended");
 61     equals(view.$('input').val(), 'ohai', "value is reflected in the input element once it is created");
 62   });
 63 
 64   test("listens for focus and blur events", function() {
 65     var focusCalled = 0;
 66     var blurCalled = 0;
 67 
 68     textFieldView.focus = function() {
 69       focusCalled++;
 70     };
 71     textFieldView.blur = function() {
 72       blurCalled++;
 73     };
 74 
 75     equals(focusCalled+blurCalled, 0, "precond - no callbacks called yet");
 76 
 77     textFieldView.$('input').focus();
 78     equals(focusCalled, 1, "focus called after field receives focus");
 79 
 80     textFieldView.$('input').blur();
 81     equals(blurCalled, 1, "blur called after field blurs");
 82   });
 83 
 84   test("calls correct method for key events", function() {
 85     var insertNewlineCalled = 0;
 86     var cancelCalled = 0;
 87 
 88     textFieldView.insertNewline = function() {
 89       insertNewlineCalled++;
 90       return YES;
 91     };
 92     textFieldView.cancel = function() {
 93       cancelCalled++;
 94       return YES;
 95     };
 96 
 97     textFieldView.$('input').focus();
 98     equals(insertNewlineCalled+cancelCalled, 0, "precond - no callbacks called yet");
 99 
100     SC.RootResponder.responder.keyup(new SC.Event({ type: 'keyup', keyCode: 13 }));
101     equals(insertNewlineCalled, 1, "calls insertNewline after hitting return");
102 
103     SC.RootResponder.responder.keyup(new SC.Event({ type: 'keyup', keyCode: 27 }));
104     equals(cancelCalled, 1, "calls cancel after pressing escape key");
105 
106   });
