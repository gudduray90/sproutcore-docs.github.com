  5 /*globals Greenhouse */
  6 /**
  7   prevents drag events from hitting iframe
  8 
  9   thanks to Jonathan Lewis
 10   @extends SC.View
 11 */
 12 Greenhouse.EventBlocker = SC.View.extend(
 13 /** @scope Greenhouse.EventBlocker.prototype */ {
 14 
 15   isVisible: NO,
 16 
 17   dragStarted: function(drag, evt) {
 18     this.set('isVisible', YES);
 19   },
 20   dragEnded: function(drag, evt) {
 21     this.set('isVisible', NO);
 22   },
 23 
 24   isDropTarget: YES,
 25 
 26   mouseMoved: function(evt){
 27     return this.get('isVisible');
 28   },
 29   mouseDragged: function(evt){
 30     return this.get('isVisible');
 31   }
 32 });
