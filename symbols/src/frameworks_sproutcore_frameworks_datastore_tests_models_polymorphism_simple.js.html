  7 
  8 (function() {
  9 
 10   var store, Person, Male, Female, colin, maggie;
 11 
 12   module("Polymorphic SC.Record - Simple", {
 13     setup: function() {
 14       SC.RunLoop.begin();
 15 
 16       store = SC.Store.create();
 17 
 18       Person = SC.Record.extend();
 19       Person.isPolymorphic = YES;
 20 
 21       Male = Person.extend({
 22         isMale: YES
 23       });
 24 
 25       Female = Person.extend({
 26         isFemale: YES
 27       });
 28 
 29       colin = store.createRecord(Male, {
 30         guid: '1'
 31       });
 32 
 33       maggie = store.createRecord(Female, {
 34         guid: '2'
 35       });
 36     },
 37 
 38     teardown: function() {
 39       store = Person = Male = Female = colin = maggie = null;
 40       SC.RunLoop.end();
 41     }
 42   });
 43 
 44   test("Adding isPolymorphic to extend() hash applies it to record class", function() {
 45     var Test = SC.Record.extend({
 46       isPolymorphic: YES
 47     });
 48     var test = store.createRecord(Test, {});
 49 
 50     ok(Test.isPolymorphic, "Record class should have gotten passed isPolymorphic value");
 51     ok(test.isPolymorphic === null || test.isPolymorphic === undefined, "Created record instance should not have isPolymorphic property");
 52   });
 53 
 54   test("SC.Store#find works with abstract record type", function() {
 55     var person1 = store.find(Person, '1'),
 56         person2 = store.find(Person, '2');
 57 
 58     equals(person1, colin, "find on Person record type with guid 1 should return male record");
 59     ok(SC.kindOf(person1, Male) && person1.isMale, "returned record should be of type Male");
 60 
 61     equals(person2, maggie, "find on Person record type with guid 2 should return female record");
 62     ok(SC.kindOf(person2, Female) && person2.isFemale, "returned record should be of type Female");
 63   });
 64 
 65   test("Creating a record of a different concrete type with the same id errors", function() {
 66     expect(1);
 67 
 68     try {
 69       store.createRecord(Female, {
 70         guid: '1'
 71       });
 72     } catch (e) {
 73       ok(true, "Error occurred when trying to create type with same guid");
 74     }
 75   });
 76 
 77   test("Changing the 'id' updates the storeKeys for all types of the same record", function () {
 78     var person1 = store.find(Person, '1'),
 79       person2 = store.find(Person, '2');
 80 
 81     equals(person1, colin, "find on Person record type with guid 1 should return male record");
 82     colin.set('id', 'x');
 83     person1 = store.find(Person, '1');
 84 
 85     ok(person1 !== colin, "find on Person record type with guid 1 should not work anymore");
 86 
 87     person1 = store.find(Person, 'x');
 88     equals(person1, colin, "find on Person record type with guid x should return male record");
 89   });
 90 })();
