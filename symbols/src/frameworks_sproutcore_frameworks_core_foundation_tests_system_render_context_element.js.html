  8 /*global module test equals context ok */
  9 
 10 var context = null;
 11 
 12 module("SC.RenderContext#element", {
 13   setup: function() {
 14     context = SC.RenderContext() ;
 15   },
 16   
 17   teardown: function() {
 18     context = null;
 19   }
 20 });
 21 
 22 test("converts context to a DOM element and returns root element if there is one", function() {
 23   context.id('foo');
 24   var elem = context.element();
 25   ok(elem, 'elem not null');
 26   equals(elem.tagName.toString().toLowerCase(), 'div', 'is div');
 27   equals(elem.id.toString(), 'foo', 'is id=foo');
 28   elem = null ;
 29 });
 30 
 31 test("returns null if context does not generate valid element", function() {
 32   context = SC.RenderContext(null);
 33   var elem = context.element();
 34   equals(elem, null, 'should be null');
 35   elem = null;
 36 });
 37 
 38 test("returns first element if context renders multiple element", function() {
 39   context.tag('div').tag('span');
 40   var elem = context.element();
 41   ok(elem, 'elem not null');
 42   equals(elem.tagName.toString().toLowerCase(), 'div', 'is div');
 43   elem = null;
 44 });
