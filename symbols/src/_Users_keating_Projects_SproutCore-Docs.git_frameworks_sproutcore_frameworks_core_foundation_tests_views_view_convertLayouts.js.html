 15 var parent, child;
 16 
 17 /**
 18  19   Helper setup that creates a parent and child view so that you can do basic
 20   tests.
 21 */
 22 var commonSetup = {
 23   setup: function() {
 24     
 25     // create basic parent view
 26     parent = SC.View.create({
 27       layout: { top: 0, left: 0, width: 500, height: 500 }
 28     });
 29     
 30     // create child view to test against.
 31     child = SC.View.create();
 32   },
 33   
 34  35   teardown: function() {
 36     parent = child = null ;
 37   }
 38 };
 39 
 40 // ..........................................................
 41 // TEST LAYOUT WITH BASIC LAYOUT CONVERSION
 42 // 
 43 
 44 module('BASIC LAYOUT CONVERSION', commonSetup);
 45 
 46 test("layout {top, left, width, height}", function() {
 47   var layout = { top: 10, left: 10, width: 50, height: 50 };
 48   var cl = SC.View.convertLayoutToAnchoredLayout(layout, parent.get('frame'));
 49   same(cl, layout, 'conversion is equal');
 50 }) ;
 51 
 52 test("layout {top, left, bottom, right}", function() {
