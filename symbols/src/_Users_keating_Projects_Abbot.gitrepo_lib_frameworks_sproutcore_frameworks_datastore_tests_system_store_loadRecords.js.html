  7 
  8 (function() {
  9   var store, people, places, Person, Place;
 10 
 11   module("SC.Store#loadRecords", {
 12     setup: function() {
 13       Person = SC.Record.extend({
 14         first: SC.Record.attr(String, { isRequired: YES}),
 15         last: SC.Record.attr(String),
 16         age: SC.Record.attr(Number),
 17         isAlive: SC.Record.attr(Boolean)
 18       });
 19 
 20       Place = SC.Record.extend({
 21         name: SC.Record.attr(String)
 22       });
 23 
 24       SC.RunLoop.begin();
 25 
 26       store = SC.Store.create();
 27 
 28       people = [ 
 29         Person.create({ 
 30           guid: 1,
 31           first: "John",
 32           last: "Sproutish",
 33           age: 35,
 34           isAlive: YES
 35         }),
 36         Person.create({
 37           guid: 2,
 38           first: "Sarah",
 39           last: "Coop",
 40           age: 28,
 41           isAlive: YES
 42         })
 43       ];
 44 
 45       places = [
 46         Place.create({
 47           guid: 3,
 48           name: "San Francisco"
 49         }),
 50         Place.create({
 51           guid: 4,
 52           name: "St. John's"
 53         })
 54       ];
 55 
 56       SC.RunLoop.end();
 57     },
 58     teardown: function() {
 59       store = people = places = Person = Place = null;
 60     }
 61   });
 62 
 63   test("loadRecords with single Record type loads new records in store", function() {
 64     var storeKeys = store.loadRecords(Person, people),
 65         isStatusCorrect;
 66 
 67     ok(SC.isArray(storeKeys), "An array of store keys is returned");
 68 
 69     storeKeys.forEach(function(storeKey, index) {
 70       equals(store.idFor(storeKeys[index]), people[index].get('guid'), "The storeKey resolves to the correct Primary Key for index %@".fmt(index));
 71 
 72       ok(store.peekStatus(storeKey) & SC.Record.READY_CLEAN, "Record is in SC.Record.READY_CLEAN state after loading into store for index %@".fmt(index));
 73     });
 74   });
 75 
 76   test("loadRecords with multiple Record types loads new records in store", function() {
 77     var things = [],
 78         types = [Person, Person, Place, Place],
 79         storeKeys, record;
 80 
 81     things.pushObjects(people);
 82     things.pushObjects(places);
 83 
 84     things.forEach(function(thing, index) {
 85       ok(SC.kindOf(thing, types[index]), "precond - types array contains correct record type for index %@".fmt(index));
 86     });
 87 
 88     storeKeys = store.loadRecords(types, things);
 89 
 90     ok(SC.isArray(storeKeys), "An array of store keys is returned");
 91 
 92     storeKeys.forEach(function(storeKey, index) {
 93       record = store.materializeRecord(storeKey);
 94       
 95       equals(store.idFor(storeKeys[index]), things[index].get('guid'), "The storeKey resolves to the correct Primary Key for index %@".fmt(index));
 96       ok(SC.kindOf(record, types[index]), "store returns a record of the correct type for index %@".fmt(index));
 97       ok(store.peekStatus(storeKey) & SC.Record.READY_CLEAN, "Record is in SC.Record.READY_CLEAN state after loading into store for index %@".fmt(index));
 98     });
 99   });
100 })();
