  6 /*global module, test, ok, equals*/
  7 
  8 
  9 var containerView,
 10   pane,
 11   view1, view2;
 12 
 13 module("SC.ContainerView Transitions", {
 14   setup: function () {
 15     SC.run(function () {
 16       view1 = SC.View.create({
 17         toString: function () { return 'View 1'; }
 18       });
 19 
 20       view2 = SC.View.create({
 21         toString: function () { return 'View 2'; }
 22       });
 23 
 24       containerView = SC.ContainerView.create({
 25         nowShowing: view1
 26       });
 27 
 28       pane = SC.Pane.create({
 29         layout: { width: 200, height: 200, left: 0, top: 0 },
 30         childViews: [containerView]
 31       }).append();
 32     });
 33 
 34     SC.run(function () {
 35       containerView.awake();
 36     });
 37   },
 38 
 39   teardown: function () {
 40     pane.remove();
 41     containerView = pane = view1 = view2 = null;
 42   }
 43 });
 44 
 45 
 46 test("Test assumptions on the initial state of the container and views.", function () {
 47   ok(!containerView.get('isTransitioning'), "Container view should not indicate that it is transitioning.");
 48   ok(containerView.get('childViews').contains(view1), "View 1 should be a child view of container.");
 49   ok(!containerView.get('childViews').contains(view2), "View 2 should not be a child view of container.");
 50 });
 51 
 52 
 53 test("Test that the default transition (null) simply swaps the views.", function () {
 54   containerView.set('nowShowing', view2);
 55 
 56   equals(containerView.get('contentView'), view2, "Container's contentView should be");
 57   ok(!containerView.get('childViews').contains(view1), "View 1 should no longer be a child view of container.");
 58 });
 59 
 60 test("Test that the isTransitioning property of container view updates accordingly.", function () {
 61   // Pause the test execution.
 62   window.stop(2000);
 63 
 64   SC.run(function () {
 65     containerView.set('transitionSwap', SC.ContainerView.PUSH);
 66     containerView.set('nowShowing', view2);
 67   });
 68 
 69   ok(containerView.get('isTransitioning'), "Container view should indicate that it is transitioning.");
 70 
 71   setTimeout(function () {
 72     ok(!containerView.get('isTransitioning'), "Container view should not indicate that it is transitioning.");
 73 
 74     window.start();
 75   }, 1000);
 76 });
 77 
 78 test("Test that the container view calls the proper transition plugin methods.", function () {
 79   var willBuildInToViewCalled = 0,
 80     buildInToViewCalled = 0,
 81     buildInDidCancelCalled = 0,
 82     didBuildInToViewCalled = 0,
 83     willBuildOutFromViewCalled = 0,
 84     buildOutFromViewCalled = 0,
 85     buildOutDidCancelCalled = 0,
 86     didBuildOutFromViewCalled = 0,
 87     plugin;
 88 
 89   // Pause the test execution.
 90   window.stop(2000);
 91 
 92   plugin = {
 93     willBuildInToView: function () { willBuildInToViewCalled++; },
 94     buildInToView: function (statechart) {
 95       buildInToViewCalled++;
 96 
 97       setTimeout(function () {
 98         statechart.entered();
 99       }, 200);
100     },
101     buildInDidCancel: function () { buildInDidCancelCalled++; },
102     didBuildInToView: function () { didBuildInToViewCalled++; },
103     willBuildOutFromView: function () { willBuildOutFromViewCalled++; },
104     buildOutFromView: function (statechart) {
105       buildOutFromViewCalled++;
106 
107       setTimeout(function () {
108         statechart.exited();
109       }, 200);
110     },
111     buildOutDidCancel: function () { buildOutDidCancelCalled++; },
112     didBuildOutFromView: function () { didBuildOutFromViewCalled++; }
113   };
114 
115   containerView.set('transitionSwap', plugin);
116   containerView.set('nowShowing', view2);
117   equals(willBuildInToViewCalled, 1, "willBuildInToViewCalled() should have been called this many times");
118   equals(willBuildOutFromViewCalled, 1, "willBuildOutFromViewCalled() should have been called this many times");
119   equals(buildInToViewCalled, 1, "buildInToViewCalled() should have been called this many times");
120   equals(buildOutFromViewCalled, 1, "buildOutFromViewCalled() should have been called this many times");
121   equals(buildInDidCancelCalled, 0, "buildInDidCancelCalled() should have been called this many times");
122   equals(buildOutDidCancelCalled, 0, "buildOutDidCancelCalled() should have been called this many times");
123   equals(didBuildInToViewCalled, 0, "didBuildInToViewCalled() should have been called this many times");
124   equals(didBuildOutFromViewCalled, 0, "didBuildOutFromViewCalled() should have been called this many times");
125 
126   SC.run(function () {
127     setTimeout(function () {
128       equals(buildInDidCancelCalled, 0, "buildInDidCancelCalled() should have been called this many times");
129       equals(buildOutDidCancelCalled, 0, "buildOutDidCancelCalled() should have been called this many times");
130       equals(didBuildInToViewCalled, 0, "didBuildInToViewCalled() should have been called this many times");
131       equals(didBuildOutFromViewCalled, 0, "didBuildOutFromViewCalled() should have been called this many times");
132     }, 100);
133   });
134 
135   setTimeout(function () {
136     equals(buildInDidCancelCalled, 0, "buildInDidCancelCalled() should have been called this many times");
137     equals(buildOutDidCancelCalled, 0, "buildOutDidCancelCalled() should have been called this many times");
138     equals(didBuildInToViewCalled, 1, "didBuildInToViewCalled() should have been called this many times");
139     equals(didBuildOutFromViewCalled, 1, "didBuildOutFromViewCalled() should have been called this many times");
140 
141     window.start();
142   }, 300);
143 });
