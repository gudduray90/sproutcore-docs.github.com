 14 /**
 15   @class
 16 
 17   A TreeController manages a tree of model objects that you might want to
 18   display in the UI using a collection view.  For the most part, you should
 19   work with a TreeController much like you would an ObjectController, except
 20   that the TreeController will also provide an arrangedObjects property that
 21   can be used as the content of a CollectionView.
 22 
 23   @extends SC.ObjectController
 24   @extends SC.SelectionSupport
 25   @since SproutCore 1.0
 26 */
 27 SC.TreeController = SC.ObjectController.extend(SC.SelectionSupport,
 28  29 /** @scope SC.TreeController.prototype */ {
 30 
 31   // ..........................................................
 32   // PROPERTIES
 33   //
 34 
 35   /**
 36     Set to YES if you want the top-level items in the tree to be displayed as
 37     group items in the collection view.
 38 
 39     @type Boolean
 40     @default NO
 41   */
 42   treeItemIsGrouped: NO,
 43 
 44   /**
 45     If your content support expanding and collapsing of content, then set this
 46     property to the name of the key on your model that should be used to
 47     determine the expansion state of the item.  The default is
 48     "treeItemIsExpanded"
 49 
 50     @type String
 51     @default "treeItemIsExpanded"
 52   */
 53   treeItemIsExpandedKey: "treeItemIsExpanded",
 54 
 55   /**
 56     Set to the name of the property on your content object that holds the
 57     children array for each tree node.  The default is "treeItemChildren".
 58 
 59     @type String
 60     @default "treeItemChildren"
 61   */
 62   treeItemChildrenKey: "treeItemChildren",
 63 
 64   /**
 65     Returns an SC.Array object that actually will represent the tree as a
 66     flat array suitable for use by a CollectionView.  Other than binding this
 67     property as the content of a CollectionView, you generally should not
 68     use this property directly.  Instead, work on the tree content using the
 69     TreeController like you would any other ObjectController.
 70 
 71     @property {SC.Array}
 72   */
 73   arrangedObjects: function() {
 74  75     var ret, content = this.get('content');
 76     if (content) {
 77       ret = SC.TreeItemObserver.create({ item: content, delegate: this });
 78     } else ret = null; // empty!
 79     this._sctc_arrangedObjects = ret ;
 80 
 81     return ret ;
 82   }.property().cacheable(),
 83 
 84   // ..........................................................
 85   // PRIVATE
 86   //
 87 
 88   /**
 89     @private
 90 
 91     Manually invalidate the arrangedObjects cache so that we can teardown
 92     any existing value.  We do it via an observer so that this will fire
 93     immediately instead of waiting on some other component to get
 94     arrangedObjects again.
 95   */
 96   _sctc_invalidateArrangedObjects: function() {
 97     this.propertyWillChange('arrangedObjects');
 98 
 99     var ret = this._sctc_arrangedObjects;
100     if (ret) ret.destroy();
101     this._sctc_arrangedObjects = null;
102 
103     this.propertyDidChange('arrangedObjects');
104   }.observes('content', 'treeItemIsExpandedKey', 'treeItemChildrenKey', 'treeItemIsGrouped'),
105 
106   _sctc_arrangedObjectsContentDidChange: function() {
107     this.updateSelectionAfterContentChange();
108   }.observes('*arrangedObjects.[]'),
109 
110   canSelectGroups: NO,
111 
112   /**
113     @private
114 
115     Returns the first item in arrangeObjects that is not a group.  This uses
116     a brute force approach right now; we assume you probably don't have a lot
117     of groups up front.
118   */
119   firstSelectableObject: function() {
120     var objects = this.get('arrangedObjects'),
121         indexes, len, idx     = 0;
122 
123     if (!objects) return null; // fast track
124 
125     // other fast track. if you want something fancier use collectionViewDelegate
126     if(this.get('canSelectGroups')) return objects.get('firstObject');
127 
128     indexes = objects.contentGroupIndexes(null, objects);
129     len = objects.get('length');
130     while(indexes.contains(idx) && (idx<len)) idx++;
131     return idx>=len ? null : objects.objectAt(idx);
132   }.property()
133 
134 });
