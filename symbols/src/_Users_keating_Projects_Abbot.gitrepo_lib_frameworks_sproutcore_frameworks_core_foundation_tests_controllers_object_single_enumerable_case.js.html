  8 var src, content, controller;

 10 // ..........................................................
 11 // SINGLE OBSERVABLE OBJECT IN SET
 12 // 
 13 
 14 module("SC.ObjectController - single_enumerable_case - OBSERVABLE OBJECT", {
 15   setup: function() {
 16     src        = SC.Object.create({ foo: "foo1", bar: "bar1" });
 17     content    = SC.Set.create().add(src); // use generic enumerable
 18     controller = SC.ObjectController.create({ 
 19       content: content,
 20       allowsMultipleContent: NO 
 21     });
 22   },
 23   
 24   teardown: function() {
 25     controller.destroy();
 26   }
 27 });
 28 
 29 test("getting any unknown value should pass through to object", function() {
 30   equals(controller.get("foo"), "foo1", 'controller.get(foo)');
 31   equals(controller.get("bar"), "bar1", 'controller.get(bar)');
 32 });
 33 
 34 test("setting any unknown value should pass through", function() {
 35   equals(controller.set("foo", "EDIT"), controller, 'controller.set(foo, EDIT) should return self');  
 36   equals(controller.set("bar", "EDIT"), controller, 'controller.set(bar, EDIT) should return self');
 37   equals(controller.set("baz", "EDIT"), controller, 'controller.set(baz, EDIT) should return self');
 38   
 39   equals(src.get("foo"), "EDIT", 'src.get(foo)');
 40   equals(src.get("bar"), "EDIT", 'src.get(bar)');
 41   equals(src.get("baz"), "EDIT", 'src.get(bar)');
 42 });
 43 
 44 test("changing a property on the content", function() {
 45   var callCount = 0;
 46   controller.addObserver("foo", function() { callCount++; });
 47 
 48   equals(controller.get("foo"), "foo1", "controller.get(foo) before edit should have original value");
 49 
 50   src.set("foo", "EDIT");
 51   equals(controller.get("foo"), "EDIT", "controller.get(foo) after edit should have updated value");
 52   equals(callCount, 1, 'observer on controller should have fired');
 53 });
 54 
 55 test("changing the content from one to another", function() {
 56   var callCount = 0 ;
 57   var src2 = SC.Object.create({ foo: "foo2", bar: "bar2" });
 58   var content2 = [src2]; // use another type of enumerable
 59   controller.addObserver("foo", function() { callCount++; });
 60 
 61   equals(controller.get("foo"), "foo1", "controller.get(foo) before edit should have original value");
 62 
 63   controller.set("content", content2);
 64 
 65   equals(controller.get("foo"), "foo2", "controller.get(foo) after content should contain new content");
 66   equals(callCount, 1, 'observer on controller should have fired');
 67 
 68   callCount = 0 ;
 69   src2.set("foo", "EDIT");
 70   equals(controller.get("foo"), "EDIT", "controller.get(foo) after edit should have updated value");
 71   equals(callCount, 1, 'observer on controller should have fired');
 72   
 73   callCount = 0;
 74   content.set("foo", "BAR");
 75   equals(controller.get("foo"), "EDIT", "controller.get(foo) after edit of non-content object should not change value");
 76   equals(callCount, 0, 'observer on controller should NOT have fired');
 77 });
 78 
 79 test("hasContent", function() {
 80   equals(controller.get("hasContent"), YES, 'should have content');
 81   
 82   var callCount = 0;
 83   controller.addObserver("hasContent", function() { callCount++; });
 84   
 85   controller.set("content", null);
 86   equals(controller.get("hasContent"), NO, "hasContent should == NO after setting to null");
 87   ok(callCount > 0, 'hasContent observer should fire when setting to null');
 88   
 89   callCount = 0;
 90   controller.set("content", content);
 91   equals(controller.get("hasContent"), YES, "hasContent should == YES after setting back to content");
 92   ok(callCount > 0, "hasContent observer should fire");
 93 });
