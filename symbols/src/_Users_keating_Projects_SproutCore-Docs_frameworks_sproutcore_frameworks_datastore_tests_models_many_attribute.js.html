 12 module("SC.ManyAttribute core methods", {
 13   setup: function() {
 14     SC.RunLoop.begin();
 15     MyApp = SC.Object.create({
 16       store: SC.Store.create()
 17     });
 18     
 19     MyApp.Foo = SC.Record.extend({
 20       
 21       // test simple reading of a pass-through prop
 22       firstName: SC.Record.attr(String),
 23 
 24       // test mapping to another internal key
 25       otherName: SC.Record.attr(String, { key: "firstName" }),
 26       
 27       // test mapping Date
 28       date: SC.Record.attr(Date),
 29       
 30       // used to test default value
 31       defaultValue: SC.Record.attr(String, {
 32         defaultValue: "default"
 33       }),
 34       
 35       // test toMany relationships
 36       fooMany: SC.Record.toMany('MyApp.Foo'),
 37 
 38       // test toMany relationships with different key
 39       fooManyKeyed: SC.Record.toMany('MyApp.Foo', {
 40         key: 'fooIds'
 41       }),
 42 
 43       // test many-to-many relationships with inverse
 44       barToMany: SC.Record.toMany('MyApp.Bar', {
 45         inverse: 'fooToMany', isMaster: YES, orderBy: 'name'
 46       }),
 47       
 48       // test many-to-one relationships with inverse
 49       barToOne: SC.Record.toMany('MyApp.Bar', {
 50         inverse: 'fooToOne', isMaster: NO
 51       })
 52       
 53     });
 54     
 55     MyApp.Bar = SC.Record.extend({
 56       
 57       // test many-to-many
 58       fooToMany: SC.Record.toMany('MyApp.Foo', {
 59         inverse: 'barToMany', isMaster: NO
 60       }),
 61       
 62       // test many-to-one
 63       fooToOne: SC.Record.toOne('MyApp.Foo', {
 64         inverse: 'barToOne', isMaster: YES
 65       })
 66     });
 67     
 68     storeKeys = MyApp.store.loadRecords(MyApp.Foo, [
 69       { guid: 1, 
 70         firstName: "John", 
 71         lastName: "Doe",
 72         barToMany: ['bar1'],
 73         barToOne:  ['bar1', 'bar2'] 
 74       },
 75       
 76       { guid: 2, 
 77         firstName: "Jane", 
 78         lastName: "Doe",
 79         barToMany: ['bar1', 'bar2'],
 80         barToOne:  [] 
 81       },
 82       
 83       { guid: 3, 
 84         firstName: "Emily", 
 85         lastName: "Parker", 
 86         fooMany: [1,2],
 87         barToMany: ['bar2'],
 88         barToOne: [] 
 89       },
 90       
 91       { guid: 4,
 92         firstName: "Johnny",
 93         lastName: "Cash",
 94         fooIds: [1,2]
 95       }
 96     ]);
 97     
 98     MyApp.store.loadRecords(MyApp.Bar, [
 99       { guid: "bar1", name: "A", fooToMany: [1,2], fooToOne: 1 },
100       { guid: "bar2", name: "Z", fooToMany: [2,3], fooToOne: 1 },
101       { guid: "bar3", name: "C" }
102     ]);
103     
104     foo1 = rec = MyApp.store.find(MyApp.Foo, 1);
105     foo2 = rec2 = MyApp.store.find(MyApp.Foo, 2);
106     foo3 = rec3 = MyApp.store.find(MyApp.Foo, 3);
107     rec4 = MyApp.store.find(MyApp.Foo, 4);
108     equals(rec.storeKey, storeKeys[0], 'should find record');
109     
110     bar1 = MyApp.store.find(MyApp.Bar, "bar1");
111     bar2 = MyApp.store.find(MyApp.Bar, 'bar2');
112     bar3 = MyApp.store.find(MyApp.Bar, 'bar3');
113     
114     SC.RunLoop.end();
115   },
116   
117   teardown: function() {
118     MyApp = rec = rec2 = rec3 = 
119     foo1 = foo2 = foo3 = bar1 = bar2 = null;
120   }

244 test("modifying a toMany array within a nested store", function() {
245 
246   var child = MyApp.store.chain() ; // get a chained store
247   var parentFooMany = rec3.get('fooMany'); // base foo many
248   
249   var childRec3 = child.find(rec3); 
250   var childFooMany = childRec3.get('fooMany'); // get the nested fooMany
251   
252   // save store keys before modifying for easy testing
253   var expected = parentFooMany.getEach('storeKey');
254   
255   // now trying modifying...
256   var childRec4 = child.find(rec4);
257   equals(childFooMany.get('length'), 2, 'precond - childFooMany should be like parent');
258   childFooMany.pushObject(childRec4);
259   equals(childFooMany.get('length'), 3, 'childFooMany should have 1 more item');
260   
261   SC.RunLoop.end(); // allow notifications to process, if there were any...
262   
263   same(parentFooMany.getEach('storeKey'), expected, 'parent.fooMany should not have changed yet');
264   equals(rec3.get('status'), SC.Record.READY_CLEAN, 'parent rec3 should still be READY_CLEAN');
265   
266   expected = childFooMany.getEach('storeKey'); // update for after commit
267 
268   SC.RunLoop.begin();
269   child.commitChanges();
270   SC.RunLoop.end();
271   
272   // NOTE: not getting fooMany from parent again also tests changing an array
273   // underneath.  Does it clear caches, etc?
274   equals(parentFooMany.get('length'), 3, 'parent.fooMany length should have changed');
275   same(parentFooMany.getEach('storeKey'), expected, 'parent.fooMany should now have changed form child store');
276   equals(rec3.get('status'), SC.Record.READY_DIRTY, 'parent rec3 should now be READY_DIRTY');
277   
278 });

292 function checkAllClean() {
293   SC.A(arguments).forEach(function(r) {
294     equals(r.get('status'), SC.Record.READY_CLEAN, 'PRECOND - %@.status should be READY_CLEAN'.fmt(r.get('id')));
295   }, this);
