  9 var r, r2, sender, pane, pane2, barView, fooView, defaultResponder;
 10 var keyPane, mainPane, globalResponder, globalResponderContext, actionSender, actionContext;
 11 
 12 var CommonSetup = {
 13   setup: function() {
 14 
 15     actionSender = null; // use for sendAction tests
 16     actionContext = null;
 17     var action = function (sender, context) {
 18       actionSender = sender;
 19       actionContext = context;
 20     };
 21 
 22     sender = SC.Object.create();
 23 
 24     // default responder for each pane
 25     defaultResponder = SC.Object.create({
 26       defaultAction: action
 27     });
 28 
 29     // global default responder set on RootResponder
 30     globalResponder = SC.Object.create({
 31       globalAction: action
 32     });
 33 
 34     // global default responder as a responder context
 35     // set on RootResponder
 36     globalResponderContext = SC.Object.create(SC.ResponderContext, {
 37       globalAction: action
 38     });
 39 
 40     // explicit pane
 41     pane = SC.Pane.create({
 42       acceptsKeyPane: YES,
 43       defaultResponder: defaultResponder,
 44       childViews: [SC.View.extend({
 45         bar: action,  // implement bar action
 46         childViews: [SC.View.extend({
 47           foo: action // implement foo action
 48         })]
 49       })],
 50 
 51       paneAction: action
 52 
 53     });
 54 
 55     pane2 = SC.Pane.create({
 56       acceptsKeyPane: YES,
 57       defaultResponder: defaultResponder,
 58       childViews: [SC.View.extend({
 59         bar: action,  // implement bar action
 60         childViews: [SC.View.extend({
 61           foo: action // implement foo action
 62         })]
 63       })],
 64 
 65       paneAction: action,
 66 
 67       keyAction: action,
 68       mainAction: action,
 69       globalAction: action
 70     });
 71 
 72     keyPane = SC.Pane.create({
 73       acceptsKeyPane: YES,
 74       keyAction: action
 75     });
 76     keyPane.firstResponder = keyPane ;
 77 
 78     mainPane = SC.Pane.create({
 79       acceptsKeyPane: YES,
 80       mainAction: action
 81     });
 82     mainPane.firstResponder = mainPane ;
 83 
 84     r = SC.RootResponder.create({
 85       mainPane: mainPane,
 86       keyPane:  keyPane,
 87       defaultResponder: globalResponder
 88     });
 89 
 90     r2 = SC.RootResponder.create({
 91       mainPane: mainPane,
 92       keyPane: keyPane,
 93       defaultResponder: globalResponderContext
 94     });
 95 
 96     barView = pane.childViews[0];
 97     ok(barView.bar, 'barView should implement bar');
 98 
 99     fooView = barView.childViews[0];
100     ok(fooView.foo, 'fooView should implement foo');
101 
102     // setup dummy namespace
103     window.Dummy = {
104       object: SC.Object.create({ foo: action }),
105       hash: { foo: action }
106     };
107 
108   },
109 
110   teardown: function() {
111     r = r2 = sender = pane = window.Dummy = barView = fooView = null;
112     defaultResponder = keyPane = mainPane = globalResponder = null;
113     globalResponderContext = null;
114   }
115 };
