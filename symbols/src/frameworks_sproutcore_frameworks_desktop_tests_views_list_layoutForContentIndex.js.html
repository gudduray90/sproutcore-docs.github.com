  6 /*global module, test, same, ok, equals*/
  7 
  8 
  9 var content, view;
 10 
 11 module("SC.ListView.layoutForContentIndex", {
 12   setup: function () {
 13     content = "1 2 3 4 5 6 7 8 9 0".w().map(function (x) {
 14       return SC.Object.create({ value: x });
 15     }, this);
 16 
 17     view = SC.ListView.create({
 18       content: content,
 19       rowSize: 50
 20     });
 21   },
 22 
 23   teardown: function () {
 24     view.destroy();
 25     content = view = null;
 26   }
 27 });
 28 
 29 test("Expected layout objects for each content index in vertical mode.", function () {
 30   for (var i = 0, len = content.length; i < len; i++) {
 31     same(view.layoutForContentIndex(i), { left: 0, right: 0, height: 50, top: i * 50 }, "The layout object at index %@ should be".fmt(i));
 32   }
 33 });
 34 
 35 test("Expected layout objects for each content index in horizontal mode.", function () {
 36   view.set('layoutDirection', SC.LAYOUT_HORIZONTAL);
 37 
 38   for (var i = 0, len = content.length; i < len; i++) {
 39     same(view.layoutForContentIndex(i), { top: 0, bottom: 0, width: 50, left: i * 50 }, "The layout object at index %@ should be".fmt(i));
 40   }
 41 });
