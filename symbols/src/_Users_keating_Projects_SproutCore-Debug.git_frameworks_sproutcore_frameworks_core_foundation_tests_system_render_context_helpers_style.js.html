 10 var context = null;
 11 
 12 // ..........................................................
 13 // styles
 14 //
 15 module("SC.RenderContext#styles", {
 16   setup: function() {
 17  18     context = SC.RenderContext() ;
 19   }
 20 });
 21 
 22  23 test("returns empty hash if no current styles", function() {
 24   same(context.styles(), {}, 'styles') ;
 25 });
 26 
 27 test("styles(hash) replaces styles", function() {
 28   var styles = { foo: 'bar' };
 29   equals(context.styles(styles), context, "returns receiver");
 30   same(context.styles(), styles, 'Styles');
 31 });
 32 
 33 test("returns styles if set", function() {
 34   var styles = { foo: 'bar' };
 35   context.styles(styles);
 36   equals(context.styles(), styles, 'styles');
 37 });
 38 
 39 test("clone on next retrieval if styles(foo) set with cloneOnModify=YES", function() {
 40   var styles = { foo: 'bar' };
 41   context.styles(styles, YES);
 42 
 43   var result = context.styles();
 44   ok(result !== styles, "styles is NOT same instance");
 45   same(result, styles, "but styles are equivalent");
 46 
 47   equals(result, context.styles(), "2nd retrieval is same instance");
 48 });
 49 
 50 test("extracts styles from element on first retrieval", function() {
 51   var elem = document.createElement('div');
 52   SC.$(elem).attr('style', 'color: black; height: 20px; border-top: 1px solid hotpink; -webkit-column-count: 3');
 53   context = SC.RenderContext(elem);
 54 
 55   var result = context.styles();
 56 
 57   if(SC.browser.isIE){
 58     same(result, { color: 'black', height: '20px', borderTop: 'hotpink 1px solid', WebkitColumnCount: '3' }, 'extracted style. This is failing in IE8 because it return styles like cOLOR.');
 59   }else{
 60     same(result, { color: 'black', height: '20px', borderTop: '1px solid hotpink', WebkitColumnCount: '3' }, 'extracted style. This is failing in IE8 because it return styles like cOLOR.');
 61   }
 62   equals(context.styles(), result, "should reuse same instance thereafter");
 63 });

 68 module("SC.RenderContext#addStyle", {
 69   setup: function() {
 70     context = SC.RenderContext().styles({ foo: 'foo' }) ;
 71   }
 72 });
 73 
 74 test("should add passed style name to value", function() {
 75   context.addStyle('bar', 'bar');
 76   equals('bar', context.styles().bar, 'verify style name');
 77 });
 78 
 79 test("should replace passed style name  value", function() {
 80   context.addStyle('foo', 'bar');
 81   equals('bar', context.styles().foo, 'verify style name');
 82 });
 83 
 84 test("should return receiver", function() {
 85   equals(context, context.addStyle('foo', 'bar'));
 86 });
 87 
 88 test("should create styles hash if needed", function() {
 89   context = SC.RenderContext();
 90   equals(context._styles, null, 'precondition - has no styles');
 91 
 92   context.addStyle('foo', 'bar');
 93   equals('bar', context.styles().foo, 'has styles');
 94 });
 95 
 96 test("should assign all styles if a hash is passed", function() {
 97   context.addStyle({ foo: 'bar', bar: 'bar' });
 98   same(context.styles(), { foo: 'bar', bar: 'bar' }, 'has same styles');
 99 });
100 
101 test("css() should be an alias for addStyle()", function() {
102   equals(SC.RenderContext.fn.css, SC.RenderContext.fn.addStyle, 'methods');
103 });
104 
105 test("addStyle should remove properties that are part of combo properties", function(){
106   SC.COMBO_STYLES = { foo: 'fooSub'.w() };
107   context.styles({ foo: 'foo', fooSub: 'bar' });
108   equals(context.styles().fooSub, 'bar', 'proper starting values');
109   context.addStyle('foo', 'bar');
110   equals(context.styles().foo, 'bar', 'foo has new value');
111   equals(context.styles().fooSub, undefined, 'fooSub has no value');
112 });
