 14 /**
 15   @namespace
 16 
 17   Normal SproutCore views are absolutely positioned--parent views have relatively
 18   little input on where their child views are placed.
 19 
 20   This mixin makes a view layout its child views itself, flowing left-to-right
 21   or up-to-down, and, optionally, wrapping.
 22 
 23   Child views with useAbsoluteLayout===YES will be ignored in the layout process.
 24   This mixin detects when child views have changed their size, and will adjust accordingly.
 25   It also observes child views' isVisible and calculatedWidth/Height properties, and, as a
 26   flowedlayout-specific extension, isHidden.
 27 
 28   These properties are observed through `#js:observeChildLayout` and `#js:unobserveChildLayout`;
 29   you can override the method to add your own properties. To customize isVisible behavior,
 30   you will also want to override shouldIncludeChildInFlow.
 31 
 32   This relies on the children's frames or, if specified, calculatedWidth and calculatedHeight
 33   properties.
 34 
 35   This view mixes very well with animation. Further, it is able to automatically mix
 36   in to child views it manages, created or not yet created, allowing you to specify
 37   settings such as animation once only, and have everything "just work".
 38 
 39   Like normal views, you simply specify child views--everything will "just work."
 40 
 41   @since SproutCore 1.0
 42 */

155   /** @private
156     Overridden to only update if it is a view we do not manage, or the width or height has changed
157     since our last record of it.
158   */
159   layoutDidChangeFor: function(c) {
160     // now, check if anything has changed
161     var l = c._scfl_lastLayout, cl = c.get('layout'), f = c.get('frame');
162     if (!l) return sc_super();
163 
164     var same = YES;
165 
166     // in short, if anything interfered with the layout, we need to
167     // do something about it.
168     if (l.left && l.left !== cl.left) same = NO;
169     else if (l.top && l.top !== cl.top) same = NO;
170     else if (!c.get('fillWidth') && l.width && l.width !== cl.width) same = NO;
171     else if (!l.width && !c.get('fillWidth') && f.width !== c._scfl_lastFrame.width) same = NO;
172     else if (!c.get('fillHeight') && l.height && l.height !== cl.height) same = NO;
173     else if (!l.height && !c.get('fillHeight') && f.height !== c._scfl_lastFrame.height) same = NO;
174 
175     if (same) {
176       return sc_super();
177     }
178 
179     // nothing has changed. This is where we do something
180     this._scfl_tileOnce();
181     sc_super();
182   },

236   /**
237     Returns the flow spacings for a given view. By default, returns the view's flowSpacing,
238     and if they don't exist, the defaultFlowSpacing for this view.
239 
240     @field
241     @type Hash
242   */
243   flowSpacingForChild: function(idx, view) {
244     var spacing = view.get('flowSpacing');
245     if (SC.none(spacing)) spacing = this.get('defaultFlowSpacing');
246     if (SC.none(spacing)) spacing = 0;
247 
248     if (SC.typeOf(spacing) === SC.T_NUMBER) {
249       spacing = { left: spacing, right: spacing, bottom: spacing, top: spacing };
250     } else {
251       spacing['left'] = spacing['left'] || 0;
252       spacing['right'] = spacing['right'] || 0;
253       spacing['top'] = spacing['top'] || 0;
254       spacing['bottom'] = spacing['bottom'] || 0;
255     }
256 
257     return spacing;
258   },

260   /**
261     Returns the flow size for a given view, excluding spacing. The default version
262     checks the view's calculatedWidth/Height, then its frame.
263 
264     For spacers, this returns an empty size.
265 
266     @field
267     @type Hash
268     @default {width: 0, height: 0}
269   */
270   flowSizeForChild: function(idx, view) {
271     var cw = view.get('calculatedWidth'), ch = view.get('calculatedHeight'),
272     layoutDirection = this.get('layoutDirection'),
273     calc = {}, f = view.get('frame'), l = view.get('layout');
274     view._scfl_lastFrame = f;
275 
276     // if there is a calculated width, use that. NOTE: if calculatedWidth === 0,
277     // it is invalid. This is the practice in other views.
278     if (cw) {
279       calc.width = cw;
280     } else {
281       // we should use the layout width if available to avoid breaking layouts
282       // that have borders
283       calc.width = l.width || f.width;
284     }
285 
286     // same for calculated height
287     if (ch) {
288       calc.height = ch;
289     } else {
290       // we should use the layout width if available to avoid breaking layouts
291       // that have borders
292       calc.height = l.height || f.height;
293     }
294 
295     // if it is a spacer, we must set the dimension that it
296     // expands in to 0.
297     if (view.get('isSpacer')) {
298       calc.maxSpacerLength = view.get('maxSpacerLength');
299 
300       if (layoutDirection === SC.LAYOUT_HORIZONTAL) {
301         calc.width = l.minWidth || 0;
302       } else {
303         calc.height = l.minHeight || 0;
304       }
305     }
306 
307     // if it has a fillWidth/Height, clear it for later
308     if (layoutDirection === SC.LAYOUT_HORIZONTAL && view.get('fillHeight')) {
309       calc.height = l.minHeight || 0;
310     } else if (layoutDirection === SC.LAYOUT_VERTICAL && view.get('fillWidth')) {
311       calc.width = l.minWidth || 0;
312     }
313 
314     return calc;
315   },

352   /**
353     @private
354     Creates a plan, initializing all of the basic properties in it, but not
355     doing anything further.
356 
357     Other methods should be called to do this:
358 
359     - _scfl_distributeChildrenIntoRows distributes children into rows.
360     - _scfl_positionChildrenInRows positions the children within the rows.
361       - this calls _scfl_positionChildrenInRow
362     - _scfl_positionRows positions and sizes rows within the plan.
363 
364     The plan's structure is defined inside the method.
365 
366     Some of these methods may eventually be made public and/or delegate methods.
367   */
368   _scfl_createPlan: function() {
369     var layoutDirection = this.get('layoutDirection'),
370         flowPadding = this.get('_scfl_validFlowPadding'),
371         frame = this.get('frame');
372 
373     var isVertical = (layoutDirection === SC.LAYOUT_VERTICAL);
374 
375     // A plan hash contains general information about the layout, and also,
376     // the collection of rows.
377     //
378     // This method only fills out a subset of the properties in a plan.
379     //
380     var plan = {
381       // The rows array starts empty. It will get filled out by the method
382       // _scfl_distributeChildrenIntoRows.
383       rows: undefined,
384 
385 
386       // the maximum row length where all collapsible items are collapsed.
387       maximumCollapsedRowLength: 0,
388 
389       // the total sizes of all rows when collapsed (With flex-height rows
390       // at minimum size)
391       totalCollapsedRowSize: 0,
392 
393       // These properties are calculated once here, but later used by
394       // the various methods.
395       isVertical: layoutDirection === SC.LAYOUT_VERTICAL,
396       isHorizontal: layoutDirection === SC.LAYOUT_HORIZONTAL,
397 
398       flowPadding: flowPadding,
399 
400       planStartPadding: flowPadding[isVertical ? 'left' : 'top'],
401       planEndPadding: flowPadding[isVertical ? 'right' : 'bottom'],
402 
403       rowStartPadding: flowPadding[isVertical ? 'top' : 'left'],
404       rowEndPadding: flowPadding[isVertical ? 'bottom' : 'right'],
405 
406       maximumRowLength: undefined, // to be calculated below
407 
408       // if any rows need to fit to fill, this is the size to fill
409       fitToPlanSize: undefined,
410 
411 
412       align: this.get('align')
413     };
414 
415     if (isVertical) {
416       plan.maximumRowLength = frame.height - plan.rowStartPadding - plan.rowEndPadding;
417       plan.fitToPlanSize = frame.width - plan.planStartPadding - plan.planEndPadding;
418     } else {
419       plan.maximumRowLength = frame.width - plan.rowStartPadding - plan.rowEndPadding;
420       plan.fitToPlanSize = frame.height - plan.planStartPadding - plan.planEndPadding;
421     }
422 
423     return plan;
424   },

426   /** @private */
427   _scfl_distributeChildrenIntoRows: function(plan) {
428     var children = this.get('childViews'), child, idx, len = children.length,
429         isVertical = plan.isVertical, rows = [], lastIdx;
430 
431     lastIdx = -1; idx = 0;
432     while (idx < len && idx !== lastIdx) {
433       lastIdx = idx;
434 
435       var row = {
436         // always a reference to the plan
437         plan: plan,
438 
439         // the combined size of the items in the row. This is used, for instance,
440         // in justification or right-alignment.
441         rowLength: undefined,
442 
443         // the size of the row. When flowing horizontally, this is the height;
444         // it is the opposite dimension of rowLength. It is calculated
445         // both while positioning items in the row and while positioning the rows
446         // themselves.
447         rowSize: undefined,
448 
449         // whether this row should expand to fit any available space. In this case,
450         // the size is the row's minimum size.
451         shouldExpand: undefined,
452 
453         // to be decided by _scfl_distributeItemsIntoRows
454         items: undefined,
455 
456         // to be decided by _scfl_positionRows
457         position: undefined
458       };
459 
460       idx = this._scfl_distributeChildrenIntoRow(children, idx, row);
461       rows.push(row);
462     }
463 
464     plan.rows = rows;
465   },

467   /**
468     @private
469     Distributes as many children as possible into a single row, stating
470     at the given index, and returning the index of the next item, if any.
471   */
472   _scfl_distributeChildrenIntoRow: function(children, startingAt, row) {
473     var idx, len = children.length, plan = row.plan, child, childSize, spacing,
474         items = [], itemOffset = 0, isVertical = plan.isVertical, itemSize, itemLength,
475         maxSpacerLength,
476         canWrap = this.get('canWrap'),
477         newRowPending = NO,
478         maxItemLength = 0,
479         max = row.plan.maximumRowLength;
480 
481     for (idx = startingAt; idx < len; idx++) {
482       child = children[idx];
483 
484       // this must be set before we check if the child is included because even
485       // if it isn't included, we need to remember that there is a line break
486       // for later
487       newRowPending = newRowPending || (items.length > 0 && child.get('startsNewRow'));
488 
489       if (!this.shouldIncludeChildInFlow(idx, child)) continue;
490 
491       childSize = this.flowSizeForChild(idx, child);
492       spacing = this.flowSpacingForChild(idx, child);
493 
494       childSize.width += spacing.left + spacing.right;
495       childSize.height += spacing.top + spacing.bottom;
496 
497       itemLength = childSize[isVertical ? 'height' : 'width'];
498       if(!SC.none(childSize.maxSpacerLength)) maxSpacerLength = childSize.maxSpacerLength + (isVertical ? spacing.top + spacing.bottom : spacing.left + spacing.right);
499       itemSize = childSize[isVertical ? 'width' : 'height'];
500 
501       // there are two cases where we must start a new row: if the child or a
502       // previous child in the row that wasn't included has
503       // startsNewRow === YES, and if the item cannot fit. Neither applies if there
504       // is nothing in the row yet.
505       if ((newRowPending || (canWrap && itemOffset + itemLength > max)) && items.length > 0) {
506         break;
507       }
508 
509       var item = {
510         child: child,
511 
512         itemLength: itemLength,
513         maxSpacerLength: maxSpacerLength,
514         itemSize: itemSize,
515 
516         spacing: spacing,
517 
518         // The position in the row.
519         //
520         // note: in one process or another, this becomes left or top.
521         // but before that, it is calculated.
522         position: undefined,
523 
524         // whether this item should attempt to fill to the row's size
525         fillRow: isVertical ? child.get('fillWidth') : child.get('fillHeight'),
526 
527         // whether this item is a spacer, and thus should be resized to its itemLength
528         isSpacer: child.get('isSpacer'),
529 
530         // these will get set if necessary during the positioning code
531         left: undefined, top: undefined,
532         width: undefined, height: undefined
533       };
534 
535 
536       items.push(item);
537       itemOffset += itemLength;
538       maxItemLength = Math.max(itemLength, maxItemLength);
539     }
540 
541     row.rowLength = itemOffset;
542 
543     // if the row cannot wrap, then the minimum size for the row (and therefore collapsed size)
544     // is the same as the current row length: it consists of the minimum size of all items.
545     //
546     // If the row can wrap, then the longest item will determine the size of a fully
547     // collapsed (one item per row) layout.
548     var minRowLength = canWrap ? maxItemLength : row.rowLength;
549     row.plan.maximumCollapsedRowLength = Math.max(minRowLength, row.plan.maximumCollapsedRowLength);
550     row.items = items;
551     return idx;
552   },

563   /**
564     @private
565     Positions items within a row. The items are already in the row, this just
566     modifies the 'position' property.
567 
568     This also marks a tentative size of the row, and whether it should be expanded
569     to fit in any available extra space. Note the term 'size' rather than 'length'...
570   */
571   _scfl_positionChildrenInRow: function(row) {
572     var items = row.items, len = items.length, idx, item, position, rowSize = 0,
573         spacerCount = 0, spacerSize, align = row.plan.align, shouldExpand = YES,
574         leftOver = 0, noMaxWidth = NO;
575 
576     //
577     // STEP ONE: DETERMINE SPACER SIZE + COUNT
578     //
579     for (idx = 0; idx < len; idx++) {
580       item = items[idx];
581       if (item.isSpacer) {
582         spacerCount += item.child.get('spaceUnits') || 1;
583       }
584     }
585 
586     // justification is like adding a spacer between every item. We'll actually account for
587     // that later, but for now...
588     if (align === SC.ALIGN_JUSTIFY) spacerCount += len - 1;
589 
590     // calculate spacer size
591     spacerSize = Math.max(0, row.plan.maximumRowLength - row.rowLength) / spacerCount;
592 
593     // determine individual spacer sizes using spacerSize and limited by
594     // each spacer's maxWidth (if they have one)
595     while(spacerSize > 0) {
596       for (idx = 0; idx < len; idx++) {
597         item = items[idx];
598 
599         if (item.isSpacer) {
600           item.itemLength += spacerSize * (item.child.get('spaceUnits') || 1);
601           if(item.itemLength > item.maxSpacerLength) {
602             leftOver +=  item.itemLength - item.maxSpacerLength;
603             item.itemLength = item.maxSpacerLength;
604           }
605           else {
606             noMaxWidth = YES;
607           }
608         }
609       }
610 
611       // if none of the spacers can expand further, stop
612       if(!noMaxWidth) break;
613 
614       spacerSize = Math.round(leftOver / spacerCount);
615       leftOver = 0;
616     }
617 
618     //
619     // STEP TWO: ADJUST FOR ALIGNMENT
620     // Note: if there are spacers, this has no effect, because they fill all available
621     // space.
622     //
623     position = 0;
624     if (spacerCount === 0 && (align === SC.ALIGN_RIGHT || align === SC.ALIGN_BOTTOM)) {
625       position = row.plan.maximumRowLength - row.rowLength;
626     } else if (spacerCount === 0 && (align === SC.ALIGN_CENTER || align === SC.ALIGN_MIDDLE)) {
627       position = (row.plan.maximumRowLength / 2) - (row.rowLength / 2);
628     }
629 
630     position += row.plan.rowStartPadding;
631     //
632     // STEP TWO: LOOP + POSITION
633     //
634     for (idx = 0; idx < len; idx++) {
635       item = items[idx];
636 
637       // if this item has fillWidth or fillHeight set, the row should expand
638       // laterally
639       if(!item.fillRow) shouldExpand = NO;
640 
641       // if the item is not a fill-row item, this row has a size that all fill-row
642       // items should expand to
643       rowSize = Math.max(item.itemSize, rowSize);
644 
645       item.position = position;
646 
647       position += item.itemLength;
648 
649       // if justification is on, we have one more spacer
650       // note that we check idx because position is used to determine the new rowLength.
651       if (align === SC.ALIGN_JUSTIFY && idx < len - 1) position += spacerSize;
652     }
653 
654     row.shouldExpand = len > 0 ? shouldExpand : NO;
655     row.rowLength = position - row.plan.rowStartPadding; // row length does not include padding
656     row.rowSize = rowSize;
657 
658     row.plan.totalCollapsedRowSize += row.rowSize;
659 
660   },

662   /** @private */
663   _scfl_positionRows: function(plan) {
664     var rows = plan.rows, len = rows.length, idx, row, position,
665         fillRowCount = 0, planSize = 0, fillSpace;
666 
667     // first, we need a count of rows that need to fill, and the size they
668     // are filling to (the combined size of all _other_ rows).
669     for (idx = 0; idx < len; idx++) {
670       if (rows[idx].shouldExpand) fillRowCount++;
671       planSize += rows[idx].rowSize;
672     }
673 
674     fillSpace = plan.fitToPlanSize - planSize;
675 
676     // now, position+size the rows
677     position = plan.planStartPadding;
678     for (idx = 0; idx < len; idx++) {
679       row = rows[idx];
680 
681       if (row.shouldExpand && fillSpace > 0) {
682         row.rowSize += fillSpace / fillRowCount;
683         fillRowCount--;
684       }
685 
686       row.position = position;
687       position += row.rowSize;
688     }
689   },
690 
691   /**
692     @private
693     Positions all of the child views according to the plan.
694   */
695   _scfl_applyPlan: function(plan) {
696     var rows = plan.rows, rowIdx, rowsLen, row, longestRow = 0, totalSize = 0,
697         items, itemIdx, itemsLen, item, layout, itemSize,
698 
699         isVertical = plan.isVertical;
700 
701     rowsLen = rows.length;
702     for (rowIdx = 0; rowIdx < rowsLen; rowIdx++) {
703       row = rows[rowIdx];
704       longestRow = Math.max(longestRow, row.rowLength);
705       totalSize += row.rowSize;
706 
707       items = row.items; itemsLen = items.length;
708 
709       for (itemIdx = 0; itemIdx < itemsLen; itemIdx++) {
710         item = items[itemIdx];
711         item.child.beginPropertyChanges();
712 
713         itemSize = item.fillRow ? row.rowSize : item.itemSize;
714 
715         layout = {
716           left: item.spacing.left + (isVertical ? row.position : item.position),
717           top: item.spacing.top + (isVertical ? item.position : row.position),
718           width: isVertical ? itemSize : item.itemLength,
719           height: isVertical ? item.itemLength : itemSize
720         };
721 
722         layout.width -= item.spacing.left + item.spacing.right;
723         layout.height -= item.spacing.top + item.spacing.bottom;
724 
725         this.applyPlanToView(item.child, layout);
726         item.child._scfl_lastLayout = layout;
727 
728         item.child.endPropertyChanges();
729       }
730     }
731 
732     totalSize += plan.planStartPadding + plan.planEndPadding;
733     longestRow += plan.rowStartPadding + plan.rowEndPadding;
734 
735     this.beginPropertyChanges();
736 
737     this.set('calculatedHeight', isVertical ? longestRow : totalSize);
738     this.set('calculatedWidth', isVertical ? totalSize : longestRow);
739     this.set('_scfl_maximumCollapsedRowLength', plan.maximumCollapsedRowLength);
740     this.set('_scfl_totalCollapsedRowSize', plan.totalCollapsedRowSize);
741 
742     this.endPropertyChanges();
743   },
744 
745   /**
746     Applies the given layout to the view.
747     Override this if you would like your view to, for example, animate to a new position.
748   */
749   applyPlanToView: function(view, layout) {
750     view.adjust(layout);
751   },
752 
753   /** @private */
754   _scfl_tileOnce: function() {
755     this.invokeLast(this._scfl_tile);
756   },
757 
758   _scfl_tile: function() {
759     // short circuit when hidden
760     if(!this.get('isVisibleInWindow')) return;
761 
762     // first, do the plan
763     var plan = this._scfl_createPlan();
764     this._scfl_distributeChildrenIntoRows(plan);
765     this._scfl_positionChildrenInRows(plan);
766     this._scfl_positionRows(plan);
767     this._scfl_applyPlan(plan);
768 
769     // save so it can be observed
770     this.setIfChanged('numberOfRows', plan.rows.length);
771 
772     // second, observe all children, and stop observing any children we no longer
773     // should be observing.
774     var previouslyObserving = this._scfl_isObserving || SC.CoreSet.create(),
775         nowObserving = SC.CoreSet.create();
776 
777     var children = this.get('childViews'), len = children.length, idx, child;
778     for (idx = 0; idx < len; idx++) {
779       child = children[idx];
780 
781       if (!previouslyObserving.contains(child)) {
782         this.observeChildLayout(child);
783       } else {
784         previouslyObserving.remove(child);
785       }
786 
787       nowObserving.add(child);
788     }
789 
790     len = previouslyObserving.length;
791     for (idx = 0; idx < len; idx++) {
792       this.unobserveChildLayout(previouslyObserving[idx]);
793     }
794   },
795 
796   /** @private */
797   _scfl_frameDidChange: function() {
798     var frame = this.get("frame"), lf = this._scfl_lastFrameSize || {};
799     this._scfl_lastFrameSize = SC.clone(frame);
800 
801     if (lf.width == frame.width && lf.height == frame.height) {
802       return;
803     }
804 
805     this._scfl_tileOnce();
806   }.observes('frame'),
807 
808   /** @private */
809   destroyMixin: function() {
810     var isObserving = this._scfl_isObserving;
811     if (!isObserving) return;
812 
813     var len = isObserving.length, idx;
814     for (idx = 0; idx < len; idx++) {
815       this.unobserveChildLayout(isObserving[idx]);
816     }
817   },
818 
819   /** @private
820     Reorders childViews so that the passed views are at the beginning in the order they are passed. Needed because childViews are laid out in the order they appear in childViews.
821   */
822   reorder: function(views) {
823     if(!SC.typeOf(views) === SC.T_ARRAY) views = arguments;
824 
825     var i = views.length, childViews = this.childViews, view;
826 
827     // childViews.[] should be observed
828     this.beginPropertyChanges();
829 
830     while(i-- > 0) {
831       view = views[i];
832 
833       if(SC.typeOf(view) === SC.T_STRING) view = this.get(view);
834 
835       childViews.removeObject(view);
836       childViews.unshiftObject(view);
837     }
838 
839     this.endPropertyChanges();
840 
841     this._scfl_childViewsDidChange();
842 
843     return this;
844   }
