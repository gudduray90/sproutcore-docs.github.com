  8 SC.mixin ( /** @scope SC */ {
  9 
 10   /** Returns hex color from hsv value */
 11   convertHsvToHex: function (h, s, v) {
 12     var r = 0, g = 0, b = 0;
 13 
 14     if (v > 0) {
 15       var i = (h == 1) ? 0 : Math.floor(h * 6),
 16           f = (h == 1) ? 0 : (h * 6) - i,
 17           p = v * (1 - s),
 18           q = v * (1 - (s * f)),
 19           t = v * (1 - (s * (1 - f))),
 20           rgb = [[v,t,p],[q,v,p],[p,v,t],[p,q,v],[t,p,v],[v,p,q]];
 21       r = Math.round(255 * rgb[i][0]);
 22       g = Math.round(255 * rgb[i][1]);
 23  24       b = Math.round(255 * rgb[i][2]);
 25     }
 26     return this.parseColor('rgb(' + r + ',' + g + ',' + b + ')');
 27   },
 28 
 29   /** Returns hsv color from hex value */
 30   convertHexToHsv: function (hex) {
 31     var rgb = this.expandColor(hex),
 32         max = Math.max(Math.max(rgb[0], rgb[1]), rgb[2]),
 33         min = Math.min(Math.min(rgb[0], rgb[1]), rgb[2]),
 34         s = (max === 0) ? 0 : (1 - min/max),
 35         v = max/255,
 36         h = (max == min) ? 0 : ((max == rgb[0]) ? ((rgb[1]-rgb[2])/(max-min)/6) : ((max == rgb[1]) ? ((rgb[2]-rgb[0])/(max-min)/6+1/3) : ((rgb[0]-rgb[1])/(max-min)/6+2/3)));
 37     h = (h < 0) ? (h + 1) : ((h > 1)  ? (h - 1) : h);
 38     return [h, s, v];
 39   },
 40 
 41   /** regular expression for parsing color: rgb, hex */
 42   PARSE_COLOR_RGBRE: /^rgb\(\s*(\d+)\s*,\s*(\d+)\s*,\s*(\d+)\s*\)$/i,
 43   PARSE_COLOR_HEXRE: /^\#([0-9a-fA-F]{3}|[0-9a-fA-F]{6})$/,
 44 
 45   // return an array of r,g,b colour
 46   expandColor: function(color) {
 47     var hexColor, red, green, blue;
 48     hexColor = this.parseColor(color);
 49     if (hexColor) {
 50       red = parseInt(hexColor.slice(1, 3), 16);
 51       green = parseInt(hexColor.slice(3, 5), 16);
 52       blue = parseInt(hexColor.slice(5, 7), 16);
 53       return [red,green,blue];
 54     }
 55   },
 56 
 57   // parse rgb color or 3-digit hex color to return a properly formatted 6-digit hex colour spec, or false
 58   parseColor: function(string) {
 59     var i=0, color = '#', match, part;
 60     if(match = this.PARSE_COLOR_RGBRE.exec(string)) {
 61       for (i=1; i<=3; i++) {
 62         part = Math.max(0, Math.min(255, parseInt(match[i],0)));
 63         color += this.toColorPart(part);
 64       }
 65       return color;
 66     }
 67     if (match = this.PARSE_COLOR_HEXRE.exec(string)) {
 68       if(match[1].length == 3) {
 69         for (i=0; i<3; i++) {
 70           color += match[1].charAt(i) + match[1].charAt(i);
 71         }
 72         return color;
 73       }
 74       return '#' + match[1];
 75     }
 76     return false;
 77   },
 78 
 79   // convert one r,g,b number to a 2 digit hex string
 80   toColorPart: function(number) {
 81     if (number > 255) number = 255;
 82     var digits = number.toString(16);
 83     if (number < 16) return '0' + digits;
 84     return digits;
 85   }
 86 
 87 
 88 });
