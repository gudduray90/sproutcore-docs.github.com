 22 test("Checking fieldValueForObject" ,function(){
 23  var obj = {};
 24  var value;
 25  value = SC.Validator.Number.fieldValueForObject(obj,'','');
 26  equals(SC.T_HASH,SC.typeOf(value),'Returns an object');
 27  value = SC.Validator.Number.fieldValueForObject(5354534545312.999,'','');
 28  equals(5354534545313,value,'Value rounded to closest value');
 29  value = SC.Validator.Number.fieldValueForObject(null,'','');
 30  equals('',value,'for null');
 31  value = SC.Validator.Number.fieldValueForObject(undefined,'','');
 32  equals('',value,'for undefined');
 33 });
