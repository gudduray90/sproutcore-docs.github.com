  9 module("SC.IndexSet#create");
 10 
 11 test("create with no params", function() {
 12   var set = SC.IndexSet.create();
 13   equals(set.get('length'), 0, 'should have no indexes');
 14 });
 15 
 16 test("create with just index", function() {
 17   var set = SC.IndexSet.create(4);
 18   equals(set.get('length'),1, 'should have 1 index');
 19   equals(set.contains(4), YES, 'should contain index');
 20   equals(set.contains(5), NO, 'should not contain 5');
 21 });
 22 
 23 test("create with index and length", function() {
 24   var set = SC.IndexSet.create(4, 2);
 25   equals(set.get('length'),2, 'should have 2 indexes');
 26   equals(set.contains(4), YES, 'should contain 4');
 27   equals(set.contains(5), YES, 'should contain 5');
 28 });
 29 
 30 test("create with other set", function() {
 31   var first = SC.IndexSet.create(4,2);
 32 
 33   var set = SC.IndexSet.create(first);
 34   equals(set.get('length'),2, 'should have same number of indexes (2)');
 35   equals(set.contains(4), YES, 'should contain 4, just like first');
 36   equals(set.contains(5), YES, 'should contain 5, just like first');
 37 });
