  8 sc_require('system/core_query');
  9 sc_require('system/ready');
 10 sc_require('system/root_responder');
 11 sc_require('system/platform');
 12 
 13 SC.PORTRAIT_ORIENTATION = 'portrait';
 14 SC.LANDSCAPE_ORIENTATION = 'landscape';
 15 SC.NO_ORIENTATION = 'desktop'; // value 'desktop' for backwards compatibility
 16 
 17 /**
 18   The device object allows you to check device specific properties such as 
 19   orientation and if the device is offline, as well as observe when they change 
 20   state.
 21   
 22   ## Orientation
 23   When a touch device changes orientation, the orientation property will be
 24   set accordingly which you can observe
 25   
 26   ## Offline support
 27   In order to build a good offline-capable web application, you need to know 
 28   when your app has gone offline so you can for instance queue your server 
 29   requests for a later time or provide a specific UI/message.
 30   
 31   Similarly, you also need to know when your application has returned to an 
 32   'online' state again, so that you can re-synchronize with the server or do 
 33   anything else that might be needed.
 34   
 35   By observing the 'isOffline' property you can be notified when this state
 36   changes. Note that this property is only connected to the navigator.onLine
 37   property, which is available on most modern browsers.
 38   
 39 */
 40 SC.device = SC.Object.create({
 41   
 42   /**
 43     Sets the orientation for touch devices, either SC.LANDSCAPE_ORIENTATION
 44     or SC.PORTRAIT_ORIENTATION. Will be SC.NO_ORIENTATION in the case of
 45     non-touch devices that are also not simulating touch events.
 46   
 47     @property {String}
 48     @default SC.NO_ORIENTATION
 49   */
 50   orientation: SC.NO_ORIENTATION,
 51   
 52   /**
 53     Indicates whether the device is currently online or offline. For browsers
 54     that do not support this feature, the default value is NO.
 55     
 56     Is currently inverse of the navigator.onLine property. Most modern browsers
 57     will update this property when switching to or from the browser's Offline 
 58     mode, and when losing/regaining network connectivity.
 59     
 60     @property {Boolean}
 61     @default NO
 62   */
 63   isOffline: NO,
 64 
 65   /**
 66     Returns a Point containing the last known X and Y coordinates of the
 67     mouse, if present.
 68 
 69     @property {Point}
 70   */
 71   mouseLocation: function() {
 72     var responder = SC.RootResponder.responder,
 73         lastX = responder._lastMoveX,
 74         lastY = responder._lastMoveY;
 75 
 76     if (SC.empty(lastX) || SC.empty(lastY)) {
 77       return null;
 78     }
 79 
 80     return { x: lastX, y: lastY };
 81   }.property(),
 82 
 83   /**
 84     Initialize the object with some properties up front
 85   */
 86   init: function() {
 87     sc_super();
 88     
 89     if (navigator && navigator.onLine === false) {
 90       this.set('isOffline', YES);
 91     }
 92   },
 93   
 94   /**
 95     As soon as the DOM is up and running, make sure we attach necessary
 96     event handlers
 97   */
 98   setup: function() {
 99     var responder = SC.RootResponder.responder;
100     responder.listenFor(['online', 'offline'], window, this);
101     
102     this.orientationHandlingShouldChange();
103   },
104   
105   // ..........................................................
106   // ORIENTATION HANDLING
107   //
108   
109   /**
110     Determines which method to use for orientation changes.
111     Either detects orientation changes via the current size
112     of the window, or by the window.onorientationchange event.
113   */
114   orientationHandlingShouldChange: function() {
115     if (SC.platform.windowSizeDeterminesOrientation) {
116       SC.Event.remove(window, 'orientationchange', this, this.orientationchange);
117       this.windowSizeDidChange(SC.RootResponder.responder.get('currentWindowSize'));
118     } else if (SC.platform.supportsOrientationChange) {
119       SC.Event.add(window, 'orientationchange', this, this.orientationchange);
120       this.orientationchange();
121     }
122   },
123   
124   /**
125     @param {Hash} newSize The new size of the window
126     @returns YES if the method altered the orientation, NO otherwise
127   */
128   windowSizeDidChange: function(newSize) {
129     if (SC.platform.windowSizeDeterminesOrientation) {
130       if (!SC.browser.iOS) {
131         // in any browser other than iOS, use height vs. width test
132         SC.run(function() {
133           if (SC.platform.touch) {
134             if (newSize.height >= newSize.width) {
135               SC.device.set('orientation', SC.PORTRAIT_ORIENTATION);
136             } else {
137               SC.device.set('orientation', SC.LANDSCAPE_ORIENTATION);
138             }
139           } else {
140             SC.device.set('orientation', SC.NO_ORIENTATION);
141           }
142         });
143       } else {
144         // in mobile safari, because some of its chrome can make the
145         // above match landscape falsely, we compare to screen.width
146         SC.run(function() {
147           if (newSize.width === window.screen.width) {
148             SC.device.set('orientation', SC.PORTRAIT_ORIENTATION);
149           } else {
150             SC.device.set('orientation', SC.LANDSCAPE_ORIENTATION);
151           }
152         });
153       }
154       return YES;
155     }
156     return NO;
157   },
158   
159   /**
160     Called when the window.onorientationchange event is fired.
161   */
162   orientationchange: function(evt) {
163     SC.run(function() {
164       if (window.orientation === 0 || window.orientation === 180) {
165         SC.device.set('orientation', SC.PORTRAIT_ORIENTATION);
166       } else {
167         SC.device.set('orientation', SC.LANDSCAPE_ORIENTATION);
168       }
169     });
170   },
171   
172   orientationObserver: function(){
173     var body = SC.$(document.body),
174         orientation = this.get('orientation');
175     
176     if (orientation === SC.PORTRAIT_ORIENTATION) {
177       body.addClass('portrait');
178     } else {
179       body.removeClass('portrait');
180     }
181     
182     if (orientation === SC.LANDSCAPE_ORIENTATION) {
183       body.addClass('landscape');
184     } else {
185       body.removeClass('landscape');
186     }
187   }.observes('orientation'),
188   
189   
190   // ..........................................................
191   // CONNECTION HANDLING
192   // 
193   
194   online: function(evt) {
195     this.set('isOffline', NO);
196   },
197   
198   offline: function(evt) {
199     this.set('isOffline', YES);
200   }
201 
202 });
203 
204 /*
205   Invoked when the document is ready, but before main is called.  Creates
206   an instance and sets up event listeners as needed.
207 */
208 SC.ready(function() {
209   SC.device.setup() ;
210 });
