  7 sc_require("panes/pane");
  8 
  9 SC.Pane.reopen(
 10   /** @scope SC.Pane.prototype */ {
 11 
 12   /**
 13     Last known window size.
 14 
 15     @property {Rect}
 16   */
 17   currentWindowSize: null,
 18 
 19   /**
 20     The parent dimensions are always the last known window size.
 21 
 22     @returns {Rect} current window size
 23   */
 24   computeParentDimensions: function(frame) {
 25     if(this.get('designer') && SC.suppressMain) { return sc_super(); }
 26 
 27     var wDim = {x: 0, y: 0, width: 1000, height: 1000},
 28         layout = this.get('layout');
 29 
 30     // There used to be a whole bunch of code right here to calculate
 31     // based first on a stored window size, then on root responder, then
 32     // from document... but a) it is incorrect because we don't care about
 33     // the window size, but instead, the clientWidth/Height of the body, and
 34     // b) the performance benefits are not worth complicating the code that much.
 35     if (document && document.body) {
 36       wDim.width = document.body.clientWidth;
 37       wDim.height = document.body.clientHeight;
 38 
 39       // IE7 is the only browser which reports clientHeight _including_ scrollbar.
 40       if (SC.browser.isIE && SC.browser.compareVersion("7.0") === 0) {
 41         var scrollbarSize = SC.platform.get('scrollbarSize');
 42         if (document.body.scrollWidth > wDim.width) {
 43           wDim.width -= scrollbarSize;
 44         }
 45         if (document.body.scrollHeight > wDim.height) {
 46           wDim.height -= scrollbarSize;
 47         }
 48       }
 49     }
 50 
 51     // If there is a minWidth or minHeight set on the pane, take that
 52     // into account when calculating dimensions.
 53  54 
    if (layout.minHeight || layout.minWidth) {
 55       if (layout.minHeight) {
 56         wDim.height = Math.max(wDim.height, layout.minHeight);
 57       }
 58       if (layout.minWidth) {
 59         wDim.width = Math.max(wDim.width, layout.minWidth);
 60       }
 61     }
 62     return wDim;
 63   },
 64 
 65   /** @private Disable caching due to an known bug in SC. */
 66   frame: function() {
 67     if(this.get('designer') && SC.suppressMain) { return sc_super(); }
 68     return this.computeFrameWithParentFrame(null) ;
 69   }.property(),
 70 
 71   /**
 72     Invoked by the root responder whenever the window resizes.  This should
 73     simply begin the process of notifying children that the view size has
 74     changed, if needed.
 75 
 76     @param {Rect} oldSize the old window size
 77     @param {Rect} newSize the new window size
 78     @returns {SC.Pane} receiver
 79   */
 80   windowSizeDidChange: function(oldSize, newSize) {
 81     this.set('currentWindowSize', newSize) ;
 82     this.parentViewDidResize(); // start notifications.
 83     return this ;
 84   },
 85 
 86   /** @private */
 87   paneLayoutDidChange: function() {
 88     this.invokeOnce(this.updateLayout);
 89   }.observes('layout'),
 90 
 91   recomputeDependentProperties: function(original) {
 92     this.set('currentWindowSize', this.rootResponder.computeWindowSize()) ;
 93     original();
 94   }.enhance()
 95 });
