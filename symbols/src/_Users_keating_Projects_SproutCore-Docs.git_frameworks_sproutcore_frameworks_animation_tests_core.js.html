  6 var view, base, inherited, pane, computed;
  7 module("Animatable", {
  8   setup: function() {
  9     SC.RunLoop.begin();
 10     pane = SC.Pane.create({
 11       layout: { top: 0, right: 0, width: 200, height: 200 }
 12     });
 13     pane.append();
 14     
 15     view = SC.View.create(SC.Animatable, { 
 16       layout: { left: 100, top: 100, height: 100, width: 100 },
 17       style: { transform: 'scale(1)' },
 18       transitions: {
 19         left: 0.25,
 20         top: { duration: 0.35 },
 21         width: { duration: 0.2, timing: SC.Animatable.TRANSITION_EASE_IN_OUT },
 22         style: { transform: 1 }
 23       }
 24     });
 25     pane.appendChild(view);
 26     
 27     // identical to normal view above
 28     base = SC.View.extend(SC.Animatable, { 
 29       layout: { left: 100, top: 100, height: 100, width: 100 },
 30       transitions: {
 31         left: 0.25,
 32         top: { duration: 0.35 },
 33         width: { duration: 0.2, timing: SC.Animatable.TRANSITION_EASE_IN_OUT }
 34       }
 35     });
 36     
 37     // concatenate this
 38     inherited = base.create({
 39       transitions: {
 40         left: 0.99
 41       }
 42     });
 43     
 44     // computed
 45     computed = base.create({
 46       _test_left: 5,
 47       layout: function() {
 48         return { left: this._test_left, top: 5, width: 50, height: 50 };
 49       }.property("_test_left").cacheable()
 50     });
 51     
 52     SC.RunLoop.end();
 53   },
 54 
 55   teardown: function(){
 56     pane.remove();
 57   }
 58 });

 60 test("animatable should have init-ed correctly", function(){
 61   var left = view.transitions["left"];
 62   var top = view.transitions["top"];
 63   var width = view.transitions["width"];
 64   
 65   equals(left["duration"], 0.25, "left duration is .25");
 66   equals(top["duration"], 0.35, "top duration is .35");
 67   equals(width["duration"], 0.2, "width duration is .2");
 68   
 69   equals(left["timing"], undefined, "No timing for left.");
 70   equals(top["timing"], undefined, "No timing for top.");
 71   equals(width["timing"], SC.Animatable.TRANSITION_EASE_IN_OUT, "SC.Animatable.TRANSITION_EASE_IN_OUT for width.");
 72 });

 89 test("animatable handler for layer update should ensure both layout and styles are set in the 'current style'.", function() {
 90   var original_transition_enabled = SC.platform.supportsCSSTransitions;
 91   SC.platform.supportsCSSTransitions = NO;
 92   
 93   // we should have a style (it is inside a pane)
 94   var current = view.getCurrentJavaScriptStyles();
 95   ok(!SC.none(current), "There now SHOULD be a current JS style.");
 96   
 97   // and now, make sure we have both style AND layout set properly.
 98   equals(current["transform"], 'scale(1)', "transform should be scale(1)");
 99   equals(current["left"], 100, "left should be 100");
100   
101   // go back to the beginning
102   SC.platform.supportsCSSTransitions = original_transition_enabled;
103 });

105 test("animatable callbacks work in general", function(){
106   SC.RunLoop.begin();
107   view.transitions["left"] = {
108     duration: 0.25,
109     action: function() {
110       start();
111       ok(true, "Callback was called.");
112     }
113   };
114   view.updateLayout();
115   view.adjust("left", 0).updateLayout();
116   SC.RunLoop.end();
117   stop();
118   
119   setTimeout(function(){
120     start();
121     ok(false, "Timeout! Callback was not called.");
122   }, 1000);
123   
124 });

126 test("animation should not interfere with layout property observers or frame", function(){
127   SC.RunLoop.begin();
128   
129   // sanity check
130   equals(view.get("layout").left, 100, "layout left before adjust");
131   equals(view.get("frame").x, 100, "frame x before adjust");
132   
133   // adjust left
134   view.adjust("left", 0).updateLayout();
135 
136   // check values
137   equals(view.get("layout").left, 0, "layout left after adjust");
138   equals(view.get("frame").x, 0, "frame x after adjust");
139   SC.RunLoop.end();
140 });

142 test("animation should not interfere with computed layout properties.", function() {
143   equals(computed.get("layout").left, 5, "layout left before change and animation");
144   computed.adjust("left", 20);
145   equals(computed.get("layout").left, 5, "layout left after change (shouldn't change)");
146   computed.updateLayout();
147   equals(computed.get("layout").left, 5, "layout left after updateLayout");
148   computed.set("_test_left", 21);
149   equals(computed.get("layout").left, 21, "layout left after setting _test_left (make sure computed property is undisturbed)");
150 });
