  7 /*global module, test, equals, ok */
  8 module("SC.RootResponder#makeMainPane");
  9 
 10 test("returns receiver", function() {
 11   var p1 = SC.Pane.create();
 12   var r = SC.RootResponder.create();
 13   equals(r.makeMainPane(p1), r, 'returns receiver');
 14 });
 15 
 16 test("changes mainPane to new pane", function() {
 17   var p1 = SC.Pane.create(), p2 = SC.Pane.create();
 18   var r = SC.RootResponder.create();
 19   
 20   r.makeMainPane(p1);
 21   equals(r.get('mainPane'), p1, 'mainPane should be p1');
 22   ok(p1.get('isMainPane'), 'p1 should know');
 23   ok(!p2.get('isMainPane'), 'p2 should have no illusions');
 24 
 25   r.makeMainPane(p2);
 26   equals(r.get('mainPane'), p2, 'mainPane should change to p2');
 27   ok(p2.get('isMainPane'), 'p2 should know about its promotion');
 28   ok(!p1.get('isMainPane'), 'p1 should know about its demotion');  
 29   
 30 });
 31 
 32 test("if current mainpane is also keypane, automatically make new main pane key also", function() {
 33   // acceptsKeyPane is required to allow keyPane to change
 34   var p1 = SC.Pane.create({ acceptsKeyPane: YES });
 35   var p2 = SC.Pane.create({ acceptsKeyPane: YES });
 36 
 37   var r= SC.RootResponder.create({ mainPane: p1, keyPane: p1 });
 38   r.makeMainPane(p2);
 39   ok(r.get('keyPane') === p2, 'should change keyPane(%@) p1 = %@ - p2 = %@'.fmt(r.get('keyPane'), p1, p2));
 40 });

 42 test("should call blurMainTo() on current pane, passing new pane", function() {
 43   var callCount = 0;
 44   var p2 = SC.Pane.create();
 45   var p1 = SC.Pane.create({ 
 46     blurMainTo: function(pane) { 
 47       callCount++ ;
 48       equals(pane, p2, 'should pass new pane');
 49     }
 50   });
 51   
 52   var r= SC.RootResponder.create({ mainPane: p1 });
 53   r.makeMainPane(p2);
 54   equals(callCount, 1, 'should invoke callback');
 55 });

 58 test("should call focusMainFrom() on new pane, passing old pane", function() {
