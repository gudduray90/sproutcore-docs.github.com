 72     createBindingPopup: SC.State.design({
 73 
 74       parallelStatechart: 'modals',
 75 
 76       enterState: function(){
 77         Greenhouse.set("newBindingFromKey", null);
 78         Greenhouse.set("newBindingToKey", null);
 79         var modal = Greenhouse.dialogPage.get('modal');
 80         modal.set('contentView', Greenhouse.dialogPage.get('createBindingView'));
 81         modal.set('layout', {centerX: 0, centerY: 0, width: 200, height: 180});
 82         modal.append();
 83       },
 84       exitState: function(){
 85         var modal = Greenhouse.dialogPage.get('modal');
 86         modal.remove();
 87         Greenhouse.set("newBindingFromKey", null);
 88         Greenhouse.set("newBindingToKey", null);
 89         this.set('newItem', null);
 90       },
 91       cancel: function(){
 92         this.gotoState('modalReady');
 93       },
 94 
 95       create: function(){
 96         var fromKey = Greenhouse.get("newBindingFromKey"),
 97             toKey = Greenhouse.get("newBindingToKey"),
 98             newItem = this.get('newItem'),
 99             view = Greenhouse.designController.get('view'),
100             c = Greenhouse.designController.get('content');
101 
102         if(view && c){
103           Greenhouse.designController.propertyWillChange('content');
104           var designAttrs = c.get('designAttrs');
105           if(designAttrs) designAttrs = designAttrs[0];
106           newItem.addItem(fromKey, toKey, designAttrs);
107           Greenhouse.designController.propertyDidChange('content');
108         }
109 
110         this.gotoState('modalReady');
111       }
112

159     newPage: SC.State.design({
160       parentState: 'projectPicker',
161       parallelStatechart: 'modals',
162 
163       enterState: function(){
164         var modal = Greenhouse.dialogPage.get('modal');
165         modal.set('contentView', Greenhouse.dialogPage.get('pageFile'));
166         modal.set('layout', {centerX: 0, centerY: 0, width: 350, height: 300});
167 
168         Greenhouse.set('newFileName', null);
169         Greenhouse.set('newFilePath', Greenhouse.fileController.get('path'));
170         Greenhouse.set('newPageName', null);
171 
172         modal.append();
173       },
174       exitState: function(){
175         var modal = Greenhouse.dialogPage.get('modal');
176         modal.remove();
177         Greenhouse.set('newFileName', null);
178         Greenhouse.set('newFilePath', null);
179         Greenhouse.set('newPageName', null);
180       },
181 
182       cancel: function(){
183         this.gotoState('projectPicker');
184       },
185 
186       create: function(){
187         var f = Greenhouse.fileController.get('content'), ret, child, page = Greenhouse.get('newPageName'),
188             fileName = Greenhouse.get('newFileName'), filePath = Greenhouse.get('newFilePath') + "/";
189 
190         if(!fileName.match(/\.js/)) fileName = fileName + ".js";
191 
192         ret = ['// SproutCore ViewBuilder Design Format v1.0',
193           '// WARNING: This file is automatically generated.  DO NOT EDIT.  Changes you',
194           '// make to this file will be lost.', '',
195           '%@ = SC.Page.design({});'.fmt(page),''].join("\n");
196 
197         var contents = f.get('contents');
198 
199         contents.pushObject({type: 'File', dir: filePath, name: fileName, body:ret});
200         child = contents.objectAt(contents.get('length') - 1);
201         child.commitRecord();
202 
203         this.gotoState('projectPicker');
204       }
205     }),

207     editProperties: SC.State.design({
208 
209       parallelStatechart: 'modals',
210 
211       enterState: function(){
212         var picker = Greenhouse.dialogPage.get('propertyPicker');
213         var cv = Greenhouse.dialogPage.get('propertyEditor');
214         picker.set('contentView', cv);
215         var list = Greenhouse.inspectorsPage.getPath('propertiesInspector.list.contentView');
216         var content = Greenhouse.propertyController.get('content');
217 
218         //TODO: I should probably popup this picker in the plist item view....
219         picker.popup(list.itemViewForContentObject(content));
220         picker.becomeFirstResponder();
221 
222         //TODO: copy correct here?
223         Greenhouse.propertyEditorController.set('content', SC.copy(content));
224       },
225       exitState: function(){
226         var picker = Greenhouse.dialogPage.get('propertyPicker');
227         picker.remove();
228         Greenhouse.propertyEditorController.set('content', null);
229       },
230 
231       cancel: function(){
232         this.gotoState('modalReady');
233       },
234 
235       update: function(){
236         var val = Greenhouse.propertyEditorController.get('value'),
237             view = Greenhouse.propertyEditorController.get('view'),
238             key = Greenhouse.propertyEditorController.get('key'),
239             origKey = Greenhouse.propertyController.get('key'),
240             content = Greenhouse.designController.get('content'), designAttrs;
241 
242 
243 
244         // designAttrs = content.get('designAttrs');
245         //  if(designAttrs) designAttrs = designAttrs[0];
246 
247         if(key !== origKey){
248           view[origKey] = undefined;
249           delete view[origKey];
250           view.designer.designProperties.removeObject(origKey);
251           view.designer.designProperties.pushObject(key);
252           view.designer.propertyDidChange('editableProperties');
253           //delete designAttrs[origKey];
254         }
255 
256         view[key] = eval(val);
257         view.propertyDidChange(key);
258         if(view.displayDidChange) view.displayDidChange();
259 
260         Greenhouse.propertyController.set('key',key);
261         Greenhouse.propertyController.set('value', val);
262 
263         this.gotoState('modalReady');
264       }
