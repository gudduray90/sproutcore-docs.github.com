  8 var store, Foo, json, foo ;
  9 module("SC.Record#writeAttribute", {
 10   setup: function() {
 11     SC.RunLoop.begin();
 12     store = SC.Store.create();
 13     Foo = SC.Record.extend();
 14     json = {
 15       foo: "bar",
 16       number: 123,
 17       bool: YES,
 18       array: [1,2,3],
 19       guid: 1
 20     };
 21 
 22     foo = store.createRecord(Foo, json);
 23     store.writeStatus(foo.storeKey, SC.Record.READY_CLEAN);
 24     SC.RunLoop.end();
 25   },
 26 
 27   teardown: function () {
 28     store.destroy();
 29     Foo = store = json = foo = null;
 30   }
 31 });
 32 
 33 test("returns receiver", function() {
 34   equals(foo.writeAttribute("bar", "baz"), foo, 'should return receiver');
 35 });
 36 
 37 test("first time writing should mark record as dirty", function() {
 38   // precondition
 39   equals(foo.get('status'), SC.Record.READY_CLEAN, 'precond - start clean');
 40 
 41   SC.RunLoop.begin();
 42   // action
 43   foo.writeAttribute("bar", "baz");
 44   SC.RunLoop.end();
 45 
 46   // evaluate
 47   equals(foo.get('status'), SC.Record.READY_DIRTY, 'should make READY_DIRTY after write');
 48 });
 49 
 50 test("state change should be deferred if writing inside of a beginEditing()/endEditing() pair", function() {
 51 
 52   // precondition
 53   equals(foo.get('status'), SC.Record.READY_CLEAN, 'precond - start clean');
 54 
 55   SC.RunLoop.begin();
 56   // action
 57   foo.beginEditing();
 58 
 59   foo.writeAttribute("bar", "baz");
 60 
 61   equals(foo.get('status'), SC.Record.READY_CLEAN, 'should not change state yet');
 62 
 63   foo.endEditing();
 64 
 65   SC.RunLoop.end();
 66 
 67   // evaluate
 68   equals(foo.get('status'), SC.Record.READY_DIRTY, 'should make READY_DIRTY after write');
 69 
 70 }) ;
 71 
 72 test("raises exception if you try to write an attribute before an attribute hash has been set", function() {
 73   store.removeDataHash(foo.storeKey);
 74   equals(store.readDataHash(foo.storeKey), null, 'precond - should not have store key');
 75 
 76   var cnt=0 ;
 77   try {
 78     foo.writeAttribute("foo", "bar");
 79   } catch(e) {
 80     equals(e.message, SC.Record.BAD_STATE_ERROR.toString(), 'should throw BAD_STATE_ERROR');
 81     cnt++;
 82   }
 83   equals(cnt, 1, 'should raise exception');
 84 });

 87 test("Writing to an attribute in chained store sets correct status", function() {
 88 
 89   var chainedStore = store.chain() ;
 90 
 91   var chainedRecord = chainedStore.find(Foo, foo.get('id'));
 92   equals(chainedRecord.get('status'), SC.Record.READY_CLEAN, 'precon - status should be READY_CLEAN');
 93 
 94   SC.RunLoop.begin();
 95   chainedRecord.writeAttribute('foo', 'newValue');
 96   SC.RunLoop.end();
 97   //chainedRecord.set('foo', 'newValue');
 98 
 99   equals(chainedRecord.get('status'), SC.Record.READY_DIRTY, 'status should be READY_DIRTY');
100 
101 });

106   foo.set('guid', 2);

110 test("Writing primaryKey of 'id'", function(){
111   var PrimaryKeyId = SC.Record.extend({ primaryKey: 'id' });
112   var foo2 = store.createRecord(PrimaryKeyId, { id: 1 });
113 
114   equals(foo2.get('id'), 1, 'foo2.id should be 1');
115   foo2.set('id', 2);
116   equals(foo2.get('id'), 2, 'foo2.id should be 2');
