 10 var content, controller, extra;
 11 
 12 var TestObject = SC.Object.extend({
 13   title: "test",  
 14   toString: function() { return "TestObject(%@)".fmt(this.get("title")); }
 15 });
 16 
 17 
 18 // ..........................................................
 19 // NULL VALUE
 20 // 
 21 
 22 module("SC.ArrayController - null_case", {
 23   setup: function() {
 24     content = null;
 25     controller = SC.ArrayController.create({ content: content });
 26     extra = TestObject.create({ title: "FOO" });
 27   },
 28   
 29   teardown: function() {
 30     controller.destroy();
 31   }
 32 });
 33 
 34 test("state properties", function() {
 35   equals(controller.get("hasContent"), NO, 'c.hasContent');
 36   equals(controller.get("canRemoveContent"), NO, "c.canRemoveContent");
 37   equals(controller.get("canReorderContent"), NO, "c.canReorderContent");
 38   equals(controller.get("canAddContent"), NO, "c.canAddContent");
 39 });
 40 
 41 test("addObject", function() {
 42   should_throw(function() {
 43     controller.addObject(extra);
 44   }, Error, "controller.addObject should throw exception");
 45 });
 46 
 47 test("removeObject", function() {
 48   should_throw(function() {
 49     controller.removeObject(extra);
 50   }, Error, "controller.addObject should throw exception");
 51 });
 52 
 53 test("basic array operations", function() {
 54   equals(controller.get("length"), 0, 'length should be empty');
 55   equals(controller.objectAt(0), undefined, "objectAt() should return undefined");
 56   
 57   should_throw(function() {
 58     controller.replace(0,1,[extra]);
 59   }, Error, 'replace() should throw an error since it is not editable');
 60 });
 61 
 62 test("arrangedObjects", function() {
 63   equals(controller.get("arrangedObjects"), controller, 'c.arrangedObjects should return receiver');
 64 });
