  8 /**
  9   Renders and updates DOM representations of progress bars.
 10   
 11   Parameters
 12   --------------------------
 13   Expects these properties on the data source:
 14   
 15    - `isIndeterminate`
 16    - `isRunning`
 17    - `isEnabled`
 18    - `value`
 19   
 20   Theme Constants
 21   -------------------------------------
 22   Ace's `progressRenderDelegate`'s rendering process is not affected by
 23   any theme constants.
 24 */
 25 SC.BaseTheme.progressRenderDelegate = SC.RenderDelegate.create({
 26   className: 'progress',
 27   
 28   render: function(dataSource, context) {
 29     this.addSizeClassName(dataSource, context);
 30 
 31     var theme = dataSource.get('theme'),
 32         valueMax = dataSource.get('maximum'),
 33         valueMin = dataSource.get('minimum'),
 34         valueNow = dataSource.get('ariaValue');
 35 
 36     var value;
 37     if (dataSource.get('isIndeterminate')) {
 38       value = 120;
 39     } else {
 40       value = dataSource.get('value');
 41     }
 42 
 43     // make accessible
 44     context.attr('aria-valuemax', valueMax);
 45     context.attr('aria-valuemin', valueMin);
 46     context.attr('aria-valuenow', valueNow);
 47     context.attr('aria-valuetext', valueNow);
 48 
 49  50     
 51     context.setClass({
 52       indeterminate: dataSource.get('isIndeterminate'),
 53       running: dataSource.get('isRunning'),
 54       disabled: !dataSource.get('isEnabled'),
 55       'sc-empty': (value <= 0),
 56       'sc-complete': (value >= 100)
 57     });
 58     
 59     context = context.begin('div').addClass('track');
 60     this.includeSlices(dataSource, context, SC.THREE_SLICE);
 61     context = context.end();
 62     
 63     context = context.begin('div').addClass('content');
 64     context.css('width', (value * 100) + "%");
 65     this.includeSlices(dataSource, context, SC.THREE_SLICE);
 66     context = context.end();
 67  68   },
 69   
 70   update: function(dataSource, $) {
 71  72     this.updateSizeClassName(dataSource, $);
 73 
 74     var theme = dataSource.get('theme'),
 75         valueMax = dataSource.get('maximum'),
 76         valueMin = dataSource.get('minimum'),
 77         valueNow = dataSource.get('ariaValue');
 78 
 79     // make accessible
 80     $.attr('aria-valuemax', valueMax);
 81     $.attr('aria-valuemin', valueMin);
 82     $.attr('aria-valuenow', valueNow);
 83     $.attr('aria-valuetext', valueNow);
 84 
 85 
 86     var value;
 87     if (dataSource.get('isIndeterminate')) {
 88       value = 120;
 89     } else {
 90       value = dataSource.get('value');
 91     }
 92 
 93     $.setClass({
 94       indeterminate: dataSource.get('isIndeterminate'),
 95       running: dataSource.get('isRunning'),
 96       disabled: !dataSource.get('isEnabled'),
 97       'sc-empty': (value <= 0),
 98       'sc-complete': (value >= 100)
 99     });
100     
101     $.find('.content').css('width', (value * 100) + "%");
102   }
103 });
