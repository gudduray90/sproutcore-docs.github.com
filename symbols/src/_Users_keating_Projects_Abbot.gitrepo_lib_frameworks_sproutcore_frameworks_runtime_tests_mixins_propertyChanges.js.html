  7 var ObjectA;
  8 module("object.propertyChanges()", {	
  9 	setup: function() {
 10 		ObjectA = SC.Object.create({
 11 			normal: 'value',
 12 			normal1: 'zeroValue',
 13 						
 14 			action: function() {
 15 				this.normal1= 'newValue';
 16 			}.observes('normal'),
 17 			
 18 			normal2: 'dependentValue',
 19 			normal3: 'notifiedValue',
 20 			
 21 			notifyAction: function() {
 22 				this.normal3= 'newDependentValue';
 23 			}.observes('normal2'),
 24 			
 25 			notifyAllAction: function() {
 26 				this.normal2= 'newZeroValue';
 27 			}.observes('normal1')			
 28 		});
 29   }
 30 });

 46     obj.beginPropertyChanges() // nested!
