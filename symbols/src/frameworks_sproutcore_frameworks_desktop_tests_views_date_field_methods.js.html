 11 var pane, view0, view1, view2;
 12 
 13 module("SC.DateFieldView", {
 14   setup: function() {
 15     SC.RunLoop.begin();
 16     pane = SC.MainPane.create({
 17       childViews: [
 18       SC.DateFieldView.extend({
 19         hint: 'dd/mm/yyyy',
 20         value: SC.DateTime.create({ day: 1, month: 1, year: 2010})
 21       }),
 22       SC.DateFieldView.extend({
 23         hint: 'dd/mm/yyyy hh:mm AM/PM',
 24         value: SC.DateTime.create({ day: 1, month: 1, year: 2010, hour: 11, minute: 20 }),
 25         showTime: YES,
 26         isEnabled: NO
 27       }),
 28       SC.DateFieldView.extend({
 29         hint: 'hh:mm AM/PM',
 30         value: SC.DateTime.create({ hour: 11, minute: 20 }),
 31         showDate: NO,
 32         showTime: YES
 33       })]
 34     });
 35     pane.append(); // make sure there is a layer...
 36     SC.RunLoop.end();
 37 
 38     view0 = pane.childViews[0];
 39     view1 = pane.childViews[1];
 40     view2 = pane.childViews[2];
 41   },
 42 
 43   teardown: function() {
 44     pane.remove();
 45     pane.destroy();
 46     pane = view0 = view1 = view2 = null;
 47   }
 48 });
 49 
 50 test("renders an text field input tag with appropriate attributes",
 51 function() {
 52   equals(view0.get('fieldValue'), '01/01/2010', 'Field value should be equal - Date Only');
 53   equals(view1.get('fieldValue'), '01/01/2010 11:20 AM', 'Field value should be equal - Date & Time');
 54   equals(view2.get('fieldValue'), '11:20 AM', 'Field value should be equal - Time Only');
 55   equals(view0.get('isEnabled'), YES, 'field enabled');
 56   equals(view1.get('isEnabled'), NO, 'field not enabled');
 57   var q = Q$('input', view0.get('layer'));
 58   equals(q.attr('type'), 'text', 'should have type as text');
 59   equals(q.attr('name'), view0.get('layerId'), 'should have name as view_layerid');
 60 });
 61 
 62 test("isEnabled=NO should add disabled class",
 63 function() {
 64   SC.RunLoop.begin();
 65   view0.set('isEnabled', NO);
 66   SC.RunLoop.end();
 67   ok(view0.$().hasClass('disabled'), 'should have disabled class');
 68 });
 69 
 70 test("isEnabled=NO isEditable=NO should add disabled attribute", function() {
 71   SC.RunLoop.begin();
 72   view0.set('isEnabled', NO);
 73   view0.set('isEditable', NO);
 74   SC.RunLoop.end();
 75   ok(view0.$input().attr('disabled'), 'should have disabled attribute');
 76   ok(!view0.$input().attr('readOnly'), 'should not have readOnly attribute');
 77 });
 78 
 79 test("isEnabled=NO isEditable=YES should add disabled attribute", function() {
 80   SC.RunLoop.begin();
 81   view0.set('isEnabled', NO);
 82   view0.set('isEditable', YES);
 83   SC.RunLoop.end();
 84   ok(view0.$input().attr('disabled'), 'should have disabled attribute');
 85   ok(!view0.$input().attr('readOnly'), 'should not have readOnly attribute');
 86 });
 87 
 88 test("isEnabled=YES isEditable=NO should add readOnly attribute", function() {
 89   SC.RunLoop.begin();
 90   view0.set('isEnabled', YES);
 91   view0.set('isEditable', NO);
 92   SC.RunLoop.end();
 93   ok(!view0.$input().attr('disabled'), 'should not have disabled attribute');
 94   ok(view0.$input().attr('readOnly'), 'should have readOnly attribute');
 95 });
 96 
 97 test("isEnabled=YES isEditable=YES should not add disable or readOnly attribute", function() {
 98   SC.RunLoop.begin();
 99   view0.set('isEnabled', YES);
100   view0.set('isEditable', YES);
101   SC.RunLoop.end();
102   ok(!view0.$input().attr('disabled'), 'should not have disabled attribute');
103   ok(!view0.$input().attr('readOnly'), 'should not have readOnly attribute');
104 });
