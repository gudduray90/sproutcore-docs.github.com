 13 module("SC.DateFieldView", {
 14   setup: function() {
 15     SC.RunLoop.begin();
 16     pane = SC.MainPane.create({
 17       childViews: [
 18       SC.DateFieldView.extend({
 19         hint: 'dd/mm/yyyy',
 20         value: SC.DateTime.create({ day: 1, month: 1, year: 2010})
 21       }),
 22       SC.DateFieldView.extend({
 23         hint: 'dd/mm/yyyy hh:mm AM/PM',
 24         value: SC.DateTime.create({ day: 1, month: 1, year: 2010, hour: 11, minute: 20 }),
 25         showTime: YES,
 26         isEnabled: NO
 27       }),
 28       SC.DateFieldView.extend({
 29         hint: 'hh:mm AM/PM',
 30         value: SC.DateTime.create({ hour: 11, minute: 20 }),
 31         showDate: NO,
 32         showTime: YES
 33       }),
 34       SC.DateFieldView.extend({
 35         formatDateTime: '%y-%m-%d %I:%M:%S',
 36         value: SC.DateTime.create({ day: 1, month: 1, year: 2010, hour: 11, minute: 20, second: 10 }),
 37         showTime: YES,
 38       })]
 39     });
 40     pane.append(); // make sure there is a layer...
 41     SC.RunLoop.end();
 42 
 43     view0 = pane.childViews[0];
 44     view1 = pane.childViews[1];
 45     view2 = pane.childViews[2];
 46     view3 = pane.childViews[3];
 47   },
 48 
 49   teardown: function() {
 50     pane.remove();
 51     pane.destroy();
 52     pane = view0 = view1 = view2 = null;
 53   }
 54 });

 56 test("renders an text field input tag with appropriate attributes",
 57 function() {
 58   equals(view0.get('fieldValue'), '01/01/2010', 'Field value should be equal - Date Only');
 59   equals(view1.get('fieldValue'), '01/01/2010 11:20 AM', 'Field value should be equal - Date & Time');
 60   equals(view2.get('fieldValue'), '11:20 AM', 'Field value should be equal - Time Only');
 61   equals(view0.get('isEnabled'), YES, 'field enabled');
 62   equals(view1.get('isEnabled'), NO, 'field not enabled');
 63   var q = Q$('input', view0.get('layer'));
 64   equals(q.attr('type'), 'text', 'should have type as text');
 65   equals(q.attr('name'), view0.get('layerId'), 'should have name as view_layerid');
 66 });

112 test("Verify the format property", function () {
113   SC.run(function () {
114     view0.set('showDate', true);
115     view0.set('showTime', true);
116     equals(view0.get('format'), view0.get('formatDateTime'), "The format should match");
117     view0.set('formatDateTime', '%d-%m-%Y @ %I:%M %p');
118     equals(view0.get('format'), '%d-%m-%Y @ %I:%M %p', "The format should match");
119 
120     view0.set('showTime', false);
121     equals(view0.get('format'), view0.get('formatDate'), "The format should match");
122     view0.set('formatDate', '%d-%m-%Y');
123     equals(view0.get('format'), '%d-%m-%Y', "The format should match");
124 
125     view0.set('showDate', false);
126     view0.set('showTime', true);
127     equals(view0.get('format'), view0.get('formatTime'), "The format should match");
128     view0.set('formatTime', '@ %I:%M %p');
129     equals(view0.get('format'), '@ %I:%M %p', "The format should match");
130   });
131 });
