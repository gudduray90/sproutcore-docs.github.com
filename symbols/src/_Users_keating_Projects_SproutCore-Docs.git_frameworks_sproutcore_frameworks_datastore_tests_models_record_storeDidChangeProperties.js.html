  8 var store, child, Foo, json, foo ;
  9 module("SC.Record#storeDidChangeProperties", {
 10   setup: function() {
 11     SC.RunLoop.begin();
 12     
 13     store = SC.Store.create();
 14     Foo = SC.Record.extend({
 15       
 16       // record diagnostic change
 17       statusCnt: 0,
 18       statusDidChange: function() {
 19         this.statusCnt++;
 20       }.observes('status'),
 21       
 22       fooCnt: 0,
 23       fooDidChange: function() {
 24         this.fooCnt++;
 25       }.observes('foo')
 26       
 27     });
 28     
 29     
 30     json = { 
 31       foo: "bar", 
 32       number: 123,
 33       bool: YES,
 34       array: [1,2,3] 
 35     };
 36     
 37     foo = store.createRecord(Foo, json);
 38     store.writeStatus(foo.storeKey, SC.Record.READY_CLEAN);
 39     
 40     SC.RunLoop.end();
 41   }
 42 });
 43 
 44 function checkPreconditions() {
 45   equals(foo.statusCnt, 0, 'precond - statusCnt');
 46   equals(foo.fooCnt, 0, 'precond - fooCnt');
 47 }
 48 
 49 function expect(fooObject, expectedStatusCnt, expectedFooCnt) {
 50   equals(fooObject.statusCnt, expectedStatusCnt, 'status should have changed');
 51   equals(fooObject.fooCnt, expectedFooCnt, 'foo should have changed');
 52 }
 53

 55 // BASIC BEHAVIORS

 57 
 58 test("should change status only if statusOnly=YES", function() {
 59   checkPreconditions();
 60   foo.storeDidChangeProperties(YES);
 61   expect(foo,1,0);
 62 });
 63 
 64 
 65 test("should change attrs  & status if statusOnly=NO", function() {
 66   checkPreconditions();
 67   foo.storeDidChangeProperties(NO);
 68   expect(foo,1,1);
 69 });

 75 test("editing a clean record should change all", function() {
 76   checkPreconditions();
 77   
 78   SC.RunLoop.begin();
 79   foo.writeAttribute("foo", "baz"); // NB: Must be different from "foo"
 80   SC.RunLoop.end();
 81   
 82   expect(foo,2,1);
 83 });
 84 
 85 test("editing an attribute to same value should do nothing", function() {
 86   checkPreconditions();
 87   
 88   SC.RunLoop.begin();
 89   foo.writeAttribute("foo", "bar"); // NB: Must be "bar"
 90   SC.RunLoop.end();
 91   
 92   expect(foo,0,0);
 93 });
 94 
 95 test("destroying a record should change all", function() {
 96   checkPreconditions();
 97   SC.RunLoop.begin();
 98   foo.destroy();
 99   expect(foo,1,0); // calling destroy() should specifically change status
100   SC.RunLoop.end();
101   expect(foo,2,1);
102 });
103 
104 test("refreshing a record should change status", function() {
105   checkPreconditions();
106   SC.RunLoop.begin();
107   foo.refresh();
108   SC.RunLoop.end();
109   expect(foo,1,0);
110 });
111 
112 test("committing attribute changes from nested store should change attrs", function() {
113   checkPreconditions();
114   
115   SC.RunLoop.begin();
116   var child = store.chain();
117   var foo2 = child.materializeRecord(foo.storeKey);
118 
119   foo2.writeAttribute('foo', 'baz'); // must not be 'bar'
120   SC.RunLoop.end();
121   // no changes should happen yet on foo.
122   expect(foo,0,0);
123   
124   SC.RunLoop.begin();
125   // commit
126   child.commitChanges();
127 
128   // now changes
129   expect(foo,1,1);
130 });
131 
132 test("changing attributes on a parent store should notify child store if inherited", function() {
133   var child = store.chain();
134   var oldfoo = foo;
135   var parentfoo = store.materializeRecord(foo.storeKey);
136   var childfoo = child.materializeRecord(foo.storeKey);
137   equals(child.storeKeyEditState(foo.storeKey), SC.Store.INHERITED, 'precond - foo should be inherited from parent store');
138   
139   SC.RunLoop.begin();
140   parentfoo.writeAttribute('foo', 'baz'); // must not be bar
141   SC.RunLoop.end();
142   
143   expect(childfoo,1,1); // should reflect on child
144 });
145 
146 test("changing attributes on a parent store should NOT notify child store if locked", function() {
147   
148   var child = store.chain();
149   var oldfoo = foo;
150   var parentfoo = store.materializeRecord(foo.storeKey);
151   var childfoo = child.materializeRecord(foo.storeKey);
152   childfoo.readAttribute('foo');
153   equals(child.storeKeyEditState(foo.storeKey), SC.Store.EDITABLE, 'precond - foo should be locked from parent store');
154    
155   SC.RunLoop.begin();
156   parentfoo.writeAttribute('foo', 'baz'); // must not be bar
157   SC.RunLoop.end();
158   expect(childfoo,0,0); // should not reflect on child
159   expect(parentfoo,2,1);
160   // discarding changes should update
161 
162   // NOTE: recourds should change immediately on commit/discard changes.
163   // test results here BEFORE run loop ends
164   SC.RunLoop.begin();
165   child.discardChanges(); // make it match parent again
166   expect(childfoo,1,1); //the childfoo record is reset to whatever the parentValue is.
167   SC.RunLoop.end();
168 
169 });
