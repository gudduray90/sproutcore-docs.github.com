  8 /*
  9   Tests SplitView Thumb views. Specifically, event handling.
 10   
 11   Tests:
 12   
 13   - mouse events
 14   - movesChild calculation
 15   - that mouse events move movesChild
 16   - cursor style calculation
 17 
 18   This ONLY tests that adjustPositionForChild is called; we do _not_ test
 19   the positioning logic itself.
 20 */
 21 
 22 var splitView, left, divider1, middle, divider2, right, thumb,
 23     adjustedChild, adjustedPosition;
 24 module("SplitView Thumb", {
 25   setup: function() {
 26     SC.RunLoop.begin();
 27     splitView = SC.SplitView.create({
 28       childViews: [ 'left', 'middle', 'right' ],
 29 
 30       left:  SC.View.extend(SC.SplitChild, SC.SplitThumb, { name: 'left', size: 100 }),
 31       middle: SC.View.extend(SC.SplitChild, SC.SplitThumb, {
 32         name: 'middle', size: 300,
 33         childViews: 'thumb'.w(),
 34         thumb: SC.View.extend(SC.SplitThumb, { layout: { left: 0, width: 100, top:0, height: 100}})
 35       }),
 36       right: SC.View.extend(SC.SplitChild, SC.SplitThumb, { name: 'right', size: 100 }),
 37 
 38       layout: { 
 39         left: 0, top: 0, 
 40         width: 520,
 41         height: 300
 42       },
 43  44 
      splitDividerView: SC.SplitDividerView.design({ size: 10, autoResizeStyle: SC.FIXED_SIZE }),
 45 
 46       adjustPositionForChild: function(child, position) {
 47         adjustedChild = child;
 48         adjustedPosition = position;
 49       }
 50 
 51     });
 52     SC.RunLoop.end();
 53 
 54     left = splitView.childViews[0];
 55     divider1 = splitView.childViews[1];
 56     middle = splitView.childViews[2];
 57     divider2 = splitView.childViews[3];
 58     right = splitView.childViews[4];
 59 
 60     thumb = middle.thumb;
 61   }
 62 });
 63 
 64 // event fakers
 65 function md(view, x, y) {
 66   SC.RunLoop.begin();
 67   equals(view.mouseDown({ pageX: x, pageY: y }), YES, "Event returns YES");
 68   SC.RunLoop.end();
 69 }
 70 
 71 function mm(view, x, y) {
 72   SC.RunLoop.begin();
 73   equals(view.mouseDragged({ pageX: x, pageY: y }), YES, "Event returns YES");
 74   SC.RunLoop.end();
 75 }
 76 
 77 function mu(view, x, y) {
 78   SC.RunLoop.begin();
 79   equals(view.mouseUp({ pageX: x, pageY: y }), YES, "Event returns YES");
 80   SC.RunLoop.end();
 81 }
 82 
 83 test("Mouse events trigger adjustment", function() {
 84   // we always start at 0, even though it is not in-bounds;
 85   // the code always just checks the difference
 86   equals(adjustedChild || adjustedPosition, undefined, "Positions not yet adjusted.");
 87 
 88   md(divider1, 0, 0);
 89 
 90   mm(divider1, 10, 0);
 91   equals(adjustedChild, divider1, "Moved the divider.");
 92   equals(adjustedPosition, 110, "Divider moved by 10px");
 93 
 94   mu(divider1, 20, 0);
 95   equals(adjustedChild, divider1, "Moved the divider.");
 96   equals(adjustedPosition, 120, "Divider moved by 20px");
 97 
 98 });
 99 
100 test("movesChild is calculated correctly", function() {
101   equals(divider1.get('movesSibling'), SC.MOVES_CHILD, "movesSibling is MOVES_CHILD for divider1");
102   equals(divider2.get('movesSibling'), SC.MOVES_CHILD, "movesSibling is MOVES_CHILD for divider2");
103 
104   equals(left.get('movesSibling'), SC.MOVES_AUTOMATIC_CHILD, "movesSibling is SC.MOVES_AUTOMATIC_CHILD for left");
105   equals(middle.get('movesSibling'), SC.MOVES_AUTOMATIC_CHILD, "movesSibling is SC.MOVES_AUTOMATIC_CHILD for middle");
106 
107   equals(divider1.get('movesChild'), divider1, "Divider 1 moves itself.");
108   equals(divider2.get('movesChild'), divider2, "Divider 2 moves itself");
109 
110   equals(left.get('movesChild'), divider1, "Left moves child to right.");
111   equals(middle.get('movesChild'), middle, "Middle moves itself");
112   equals(right.get('movesChild'), divider2, "Right moves child to left.");
113 
114   equals(thumb.get('movesChild'), middle, "Thumb moves middle");
115 
116   // now test as we change the thumb's movesSibling
117   thumb.set('movesSibling', SC.MOVES_CHILD);
118   equals(thumb.get('movesChild'), middle, "Thumb moves middle");
119 
120   thumb.set('movesSibling', SC.MOVES_PREVIOUS_CHILD);
121   equals(thumb.get('movesChild'), divider1, "Thumb moves previous child (divider1");
122 
123   thumb.set('movesSibling', SC.MOVES_NEXT_CHILD);
124   equals(thumb.get('movesChild'), divider2, "Thumb moves next child (divider2");
125 
126 
127 });
128 
129 test("mouse events move the movesChild", function() {
130   // set directly to override computed property
131   // we are completely short-circuiting it.
132   divider1.movesChild = divider2;
133 
134   md(divider1, 0, 0);
135   mm(divider1, 10, 0);
136   equals(adjustedChild, divider2, "Attempts to move divider2.");
137 
138   // note: it _attempts_ to move to 420; doesn't neccessarily succeed.
139   equals(adjustedPosition, 420, "Attempts to move divider2 to 420.");
140 });
141 
142 test("Cursor style is calculated properly", function() {
143   // we test that, as dragging occurs, the cursor style is set on splitView's splitChildCursorStyle
144   //
145   equals(splitView.get('splitChildCursorStyle'), null, "splitChildCursorStyle should have initial null value.");
146 
147   md(divider1, 0, 0);
148   equals(splitView.get('splitChildCursorStyle'), 'ew-resize', "should now be ew-resize");
149 
150   SC.RunLoop.begin();
151   splitView.set('layoutDirection', SC.LAYOUT_VERTICAL);
152   SC.RunLoop.end();
153 
154   equals(splitView.get('splitChildCursorStyle'), 'ns-resize', "orientation change means ns-resize");
155 });
