 14 test("default implementation invokes renderChildViews if firstTime = YES", function() {
 15 
 16   var rendered = 0, updated = 0, parentRendered = 0, parentUpdated = 0 ;
 17   var view = SC.View.create({
 18     displayProperties: ["triggerRenderProperty"],
 19     childViews: ["child"],
 20 
 21     render: function(context) {
 22       parentRendered++;
 23     },
 24 
 25     update: function(jquery) {
 26       parentUpdated++;
 27     },
 28 
 29     child: SC.View.create({
 30       render: function(context) {
 31         rendered++;
 32       },
 33 
 34       update: function(jquery) {
 35         updated++;
 36       }
 37     })
 38   });
 39 
 40   view.createLayer();
 41   equals(rendered, 1, 'rendered the child');
 42   equals(parentRendered, 1);
 43 
 44   view.updateLayer(true);
 45   equals(rendered, 1, 'didn\'t call render again');
 46   equals(parentRendered, 1, 'didn\'t call the parent\'s render again');
 47   equals(parentUpdated, 1, 'called the parent\'s update');
 48   equals(updated, 0, 'didn\'t call the child\'s update');
 49 
 50   // Clean up.
 51   view.destroy();
 52 });

142 test("creates a context and then invokes renderToContext or updateLayer on each childView", function() {
143 
144   var runCount = 0, curContext, curFirstTime ;
145 
146   var ChildView = SC.View.extend({
147     renderToContext: function(context) {
148       equals(context.prevObject, curContext, 'passed child context of curContext');
149 
150       equals(context.tagName(), this.get('tagName'), 'context setup with current tag name');
151 
152       runCount++; // record run
153     },
154 
155     updateLayer: function() {
156       runCount++;
157     }
158   });
159 
160   var view = SC.View.create({
161     childViews: [
162       ChildView.extend({ tagName: 'foo' }),
163       ChildView.extend({ tagName: 'bar' }),
164       ChildView.extend({ tagName: 'baz' })
165     ]
166   });
167 
168   // VERIFY: firstTime= YES
169   curContext = view.renderContext('div');
170   curFirstTime= YES ;
171   equals(view.renderChildViews(curContext, curFirstTime), curContext, 'returns context');
172   equals(runCount, 3, 'renderToContext() invoked for each child view');
173 
174 
175   // VERIFY: firstTime= NO
176   runCount = 0 ; //reset
177   curContext = view.renderContext('div');
178   curFirstTime= NO ;
179   equals(view.renderChildViews(curContext, curFirstTime), curContext, 'returns context');
180   equals(runCount, 3, 'updateLayer() invoked for each child view');
181 
182   // Clean up.
183   view.destroy();
184 });
