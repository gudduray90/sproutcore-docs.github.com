  9 var pane, r, view ;
 10 CommonSetup = {
 11   setup: function() {
 12     pane = SC.Pane.create({
 13       childViews: [SC.View]
 14     });
 15     view = pane.childViews[0];
 16     pane.makeFirstResponder(view);
 17     
 18     pane.append(); // make visible so it will have root responder
 19     r = pane.get('rootResponder');
 20     ok(r, 'has root responder');
 21   },
 22   teardown: function() {
 23     pane.remove();
 24     pane = r = view = null ;
 25   }
 26 };
 27 
 28 // ..........................................................
 29 // becomeKeyPane()
 30 // 
 31 module("SC.Pane#becomeKeyPane", CommonSetup);
 32 
 33 test("should become keyPane if not already keyPane", function() {
 34   ok(r.get('keyPane') !== pane, 'precond - pane is not currently key');
 35   
 36   pane.becomeKeyPane();
 37   equals(r.get('keyPane'), pane, 'pane should be keyPane');
 38   equals(pane.get('isKeyPane'), YES, 'isKeyPane should be set to YES');
 39 });
 40 
 41 test("should do nothing if acceptsKeyPane is NO", function() {
 42   ok(r.get('keyPane') !== pane, 'precond - pane is not currently key');
 43   
 44   pane.acceptsKeyPane = NO ;
 45   pane.becomeKeyPane();
 46   ok(r.get('keyPane') !== pane, 'pane should not be keyPane');
 47   equals(pane.get('isKeyPane'), NO, 'isKeyPane should be NO');
 48 });
 49 
 50 test("should invoke willBecomeKeyPane & didBecomeKeyPane", function() {
 51   var callCount = 0 ;
 52   pane.willBecomeKeyPaneFrom = pane.didBecomeKeyPaneFrom = function() { 
 53     callCount++; 
 54   };
 55   
 56   pane.becomeKeyPane();
 57   equals(callCount, 2, 'should invoke both callbacks');
 58   
 59   callCount = 0;
 60   pane.becomeKeyPane();
 61   equals(callCount, 0, 'should not invoke callbacks if already key pane');
 62 });
 63 
 64 test("should invoke callbacks and update isKeyResponder state on firstResponder", function() {
 65   var callCount = 0;
 66   view.willBecomeKeyResponderFrom = view.didBecomeKeyResponderFrom = 
 67     function() { callCount++; };
 68   
 69   equals(view.get('isKeyResponder'), NO, 'precond - view is not keyResponder');
 70   equals(view.get('isFirstResponder'), YES, 'precond - view is firstResponder');
 71   
 72   pane.becomeKeyPane();
 73   equals(callCount, 2, 'should invoke both callbacks');
 74   equals(view.get('isKeyResponder'), YES, 'should be keyResponder');
 75 });
 76 
 77 // ..........................................................
 78 // resignKeyPane()
 79 // 
 80 module("SC.Pane#resignKeyPane", CommonSetup);
 81 
 82 test("should resign keyPane if keyPane", function() {
 83   pane.becomeKeyPane();
 84   ok(r.get('keyPane') === pane, 'precond - pane is currently key');
 85   
 86   pane.resignKeyPane();
 87   ok(r.get('keyPane') !== pane, 'pane should NOT be keyPane');
 88   equals(pane.get('isKeyPane'), NO, 'isKeyPane should be set to NO');
 89 });
 90 
 91 // technically this shouldn't happen, but someone could screw up and change 
 92 // acceptsKeyPane to NO once the pane has already become key
 93 test("should still resign if acceptsKeyPane is NO", function() {
 94   pane.becomeKeyPane();
 95   ok(r.get('keyPane') === pane, 'precond - pane is currently key');
 96   
 97   pane.acceptsKeyPane = NO ;
 98   pane.resignKeyPane();
 99   ok(r.get('keyPane') !== pane, 'pane should NOT be keyPane');
100   equals(pane.get('isKeyPane'), NO, 'isKeyPane should be set to NO');
101 });
102 
103 test("should invoke willLoseKeyPaneTo & didLoseKeyPaneTo", function() {
104   pane.becomeKeyPane();
105   ok(r.get('keyPane') === pane, 'precond - pane is currently key');
106 
107   var callCount = 0 ;
108   pane.willLoseKeyPaneTo = pane.didLoseKeyPaneTo = function() { 
109     callCount++; 
110   };
111   
112   pane.resignKeyPane();
113   equals(callCount, 2, 'should invoke both callbacks');
114   
115   callCount = 0;
116   pane.resignKeyPane();
117   equals(callCount, 0, 'should not invoke callbacks if already key pane');
118 });
119 
120 test("should invoke callbacks and update isKeyResponder state on firstResponder", function() {
121   var callCount = 0;
122   view.willLoseKeyResponderTo = view.didLoseKeyResponderTo = 
123     function() { callCount++; };
124   
125   pane.becomeKeyPane();
126   equals(view.get('isKeyResponder'), YES, 'precond - view is keyResponder');
127   equals(view.get('isFirstResponder'), YES, 'precond - view is firstResponder');
128   
129   pane.resignKeyPane();
130   equals(callCount, 2, 'should invoke both callbacks');
131   equals(view.get('isKeyResponder'), NO, 'should be keyResponder');
132 });
