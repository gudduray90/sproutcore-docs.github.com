 11 var items = [
 12   { title: 'Menu Item', keyEquivalent: 'ctrl_shift_n' },
 13   { title: 'Checked Menu Item', isChecked: YES, keyEquivalent: 'ctrl_a' },
 14   { title: 'Selected Menu Item', keyEquivalent: 'backspace' },
 15   { separator: YES },
 16   { title: 'Menu Item with Icon', icon: 'inbox', keyEquivalent: 'ctrl_m' },
 17   { title: 'Menu Item with Icon', icon: 'folder', keyEquivalent: 'ctrl_p' },
 18   { separator: YES },
 19   { title: 'Selected Menu Item…', isChecked: YES, keyEquivalent: 'ctrl_shift_o' },
 20   { title: 'Item with Submenu', subMenu: [{ title: 'Submenu item 1' }, { title: 'Submenu item 2'}] },
 21   { title: 'Disabled Menu Item', isEnabled: NO }//,
 22   // { isSeparator: YES },
 23   // { groupTitle: 'Menu Label', items: [{ title: 'Nested Item' }, { title: 'Nested Item' }] }
 24 ];
 25 
 26 var menu, anchor;
 27 
 28 module('SC.MenuPane Methods', {
 29   setup: function() {
 30     menu = SC.MenuPane.create({
 31       layout: { width: 206 },
 32       items: items,
 33 
 34       displayItemsCount: 0,
 35       displayItemsDidChange: function() {
 36         this.displayItemsCount++;
 37       }.observes('displayItems')
 38  39     });
 40 
 41     anchor = SC.Pane.create({
 42       layout: { top: 15, left: 15, width: 100, height: 100 }
 43     });
 44   },
 45 
 46   teardown: function() {
 47     menu.remove();
 48     anchor.remove();
 49     menu.destroy();
 50     menu = null;
 51   }
 52 });
 53 
 54 test('popup() without anchor', function(){
 55   var layout;
 56 
 57   menu.popup();
 58   layout = menu.get('layout');
 59   equals(layout.centerX, 0, 'menu should be horizontally centered');
 60   equals(layout.centerY, 0, 'menu should be vertically centered');
 61   equals(layout.width, 206, 'menu should maintain the width specified');
 62   equals(layout.height, 184, 'menu height should resize based on item content');
 63   menu.remove();
 64 });
 65 
 66 test('popup() with anchor', function(){
 67   var layout;
 68 
 69   anchor.append();
 70   menu.popup(anchor);
 71   layout = menu.get('layout');
 72   equals(layout.left, 16, 'menu should be aligned to the left of the anchor');
 73   equals(layout.top, 119, 'menu should be positioned below the anchor');
 74   equals(layout.width, 206, 'menu should maintain the width specified');
 75   equals(layout.height, 184, 'menu height should resize based on item content');
 76   menu.remove();
 77 });

 79 test('displayItems', function() {
 80   var strings = ['Alpha', 'Beta', 'Gaga'], output, count;
 81 
 82   menu.menuHeight = 1;
 83   menu.set('items', strings);
 84   equals(1, menu.displayItemsCount, 'displayItems should change when items array changes');
 85   ok(menu.get('menuHeight') > 1, 'menuHeight should be recalculated when displayItems changes');
 86 
 87   output = menu.get('displayItems')[0];
 88   equals(SC.typeOf(output), SC.T_OBJECT, 'strings should be transformed into objects');
 89   equals(output.title, 'Alpha', 'title property of transformed object should match original string');
 90   equals(output.value, 'Alpha', 'value property of transformed object should match original string');
 91   equals(output.isEnabled, YES, 'isEnabled property of transformed object should be YES');
 92 
 93   var hashes = [
 94     { title: 'Yankee' },
 95     { title: 'Hotel' },
 96     { title: 'Foxtrot' } ];
 97   menu.set('items', hashes);
 98 
 99   output = menu.get('displayItems')[0];
100   equals(SC.typeOf(output), SC.T_OBJECT, 'displayItems should convert hashes to objects');
101   equals(output.get('title'), 'Yankee', 'object properties should correspond to hash properties');
102 
103   var objects = [
104     SC.Object.create({ title: 'Whiskey' }),
105     SC.Object.create({ title: 'Mystics' }),
106     SC.Object.create({ title: 'Men' })
107   ];
108   menu.set('items', objects);
109 
110   output = menu.get('displayItems')[0];
111   equals(SC.typeOf(output), SC.T_OBJECT, 'displayItems should not convert objects');
112   equals(SC.guidFor(output), SC.guidFor(objects[0]), 'objects should be identical to provided objects');
113 });

115 test('displayItems - Edge Cases', function() {
116   menu.set('items', [null, null, false, 0, 'Real Item']);
117 
118   var output = menu.get('displayItems');
119   equals(output.get('length'), 1, 'displayItems should strip out invalid items');
120 
121   menu.set('items', ['Yellow', { title: 'Country' }, SC.Object.create({ title: 'Teeth' })]);
122   output = menu.get('displayItems');
123 
124   ok(output[0].title === 'Yellow' && output[1].title === 'Country' && output[2].title === 'Teeth',
125      'displayItems should accept a mix of supported item types');
126 
127   menu.set('items', []);
128   equals(menu.getPath('displayItems.length'), 0, 'displayItems should be empty if items is empty');
129   menu.set('items', null);
130   equals(menu.get('displayItems'), null, 'displayItems should be null if items is null');
131 });
