 13 /**
 14   SC.Module is responsible for dynamically loading in JavaScript and other
 15   resources. These packages of code and resources, called bundles, can be
 16   loaded by your application once it has finished loading, allowing you to
 17   reduce the time taken for it to launch.
 18 
 19   You can explicitly load a module by calling SC.Module.loadModule(), or you
 20   can mark a module as prefetched in your Buildfile. In those cases,
 21   SproutCore will automatically start to load the bundle once the application
 22   has loaded and the user has remained idle for more than one second.
 23 */
 24 
 25 SC.Module = SC.Object.create(/** @scope SC.Module */ {

 27   /**
 28     Returns YES if the module is ready; NO if it is not loaded or its
 29     dependencies have not yet loaded.
 30 
 31     @param {String} moduleName the name of the module to check
 32     @returns {Boolean}
 33   */
 34   isModuleReady: function (moduleName) {
 35     var moduleInfo = SC.MODULE_INFO[moduleName];
 36     return moduleInfo ? !!moduleInfo.isReady : NO;
 37   },
 38 
 39   /**
 40     Asynchronously loads a module if it is not already loaded. If you pass
 41     a function, or a target and action, it will be called once the module
 42     has finished loading.
 43 
 44     If the module you request has dependencies (as specified in the Buildfile)
 45     that are not yet loaded, it will load them first before executing the
 46     requested module.
 47 
 48     @param moduleName {String}
 49     @param target {Function}
 50     @param method {Function}
 51     @returns {Boolean} YES if already loaded, NO otherwise
 52   */
 53   loadModule: function (moduleName, target, method) {
 54     var module = SC.MODULE_INFO[moduleName], callbacks, targets,
 55         args   = SC.A(arguments).slice(3),
 56         log    = SC.LOG_MODULE_LOADING,
 57         idx, len;
 58 
 59     // Treat the first parameter as the callback if the target is a function and there is
 60     // no method supplied.
 61     if (method === undefined && SC.typeOf(target) === SC.T_FUNCTION) {
 62       method = target;
 63       target = null;
 64     }
 65 
 66     if (log) SC.debug("SC.Module: Attempting to load '%@'", moduleName);
 67 
 68     // If we couldn't find anything in the SC.MODULE_INFO hash, we don't have any record of the
 69     // requested module.
 70     if (!module) {
 71       throw new Error("SC.Module: could not find module '%@'".fmt(moduleName));
 72     }
 73 
 74     // If this module was in the middle of being prefetched, we now need to
 75     // execute it immediately when it loads.
 76     module.isPrefetching = NO;
 77 
 78     // If the module is already loaded, execute the callback immediately if SproutCore is loaded,
 79     // or else as soon as SC has finished loading.
 80     if (module.isLoaded && !module.isWaitingForRunLoop) {
 81       if (log) SC.debug("SC.Module: Module '%@' already loaded.", moduleName);
 82 
 83       // we can't just eval it if its dependencies have not been met...
 84       if (!this._dependenciesMetForModule(moduleName)) {
 85         // we can't let it return normally here, because we need the module to wait until the end of the run loop.
 86         // This is because the module may set up bindings.
 87         this._addCallbackForModule(moduleName, target, method, args);
 88 
 89         this._loadDependenciesForModule(moduleName);
 90 
 91         return NO;
 92       }
 93 
 94       // If the module has finished loading and we have the string
 95       // representation, try to evaluate it now.
 96       if (module.source) {
 97         if (log) SC.debug("SC.Module: Evaluating JavaScript for module '%@'.", moduleName);
 98         this._evaluateStringLoadedModule(module);
 99 
100         // we can't let it return normally here, because we need the module to wait until the end of the run loop.
101         // This is because the module may set up bindings.
102         this._addCallbackForModule(moduleName, target, method, args);
103 
104         this.invokeLast(function () {
105           module.isReady = YES;
106           this._moduleDidBecomeReady(moduleName);
107         });
108 
109         return NO;
110       }
111 
112       if (method) {
113         if (SC.isReady) {
114           SC.Module._invokeCallback(moduleName, target, method, args);
115         } else {
116           // Queue callback for when SC has finished loading.
117           SC.ready(SC.Module, function () {
118             SC.Module._invokeCallback(moduleName, target, method, args);
119           });
120         }
121       }
122 
123       return YES;
124     }
125 
126     // The module has loaded, but is waiting for the end of the run loop before it is "ready";
127     // we just need to add the callback.
128     else if (module.isWaitingForRunLoop) {
129       this._addCallbackForModule(moduleName, target, method, args);
130     }
131     // The module is not yet loaded, so register the callback and, if necessary, begin loading
132     // the code.
133     else {
134       if (log) SC.debug("SC.Module: Module '%@' is not loaded, loading now.", moduleName);
135 
136       // If this method is called more than once for the same module before it is finished
137       // loading, we might have multiple callbacks that need to be executed once it loads.
138       this._addCallbackForModule(moduleName, target, method, args);
139 
140       // If this is the first time the module has been requested, determine its dependencies
141       // and begin loading them as well as the JavaScript for this module.
142       if (!module.isLoading) {
143         this._loadDependenciesForModule(moduleName);
144         this._loadCSSForModule(moduleName);
145         this._loadJavaScriptForModule(moduleName);
146         module.isLoading = YES;
147       }
148 
149       return NO;
150     }
151   },
152 
153   _addCallbackForModule: function (moduleName, target, method, args) {
154     var module = SC.MODULE_INFO[moduleName];
155 
156     // Retrieve array of callbacks from MODULE_INFO hash.
157     var callbacks = module.callbacks || [];
158 
159     if (method) {
160       callbacks.push(function () {
161         SC.Module._invokeCallback(moduleName, target, method, args);
162       });
163     }
164 
165     module.callbacks = callbacks;
166   },

168   /**
169     @private
170 
171     Loads a module in string form. If you prefetch a module, its source will
172     be held as a string in memory until SC.Module.loadModule() is called, at
173     which time its JavaScript will be evaluated.
174 
175     You shouldn't call this method directly; instead, mark modules as
176     prefetched in your Buildfile. SproutCore will automatically prefetch those
177     modules once your application has loaded and the user is idle.
178 
179     @param {String} moduleName the name of the module to prefetch
180   */
181   prefetchModule: function (moduleName) {
182     var module = SC.MODULE_INFO[moduleName];
183 
184     if (module.isLoading || module.isLoaded) return;
185 
186     if (SC.LOG_MODULE_LOADING) SC.debug("SC.Module: Prefetching module '%@'.", moduleName);
187     this._loadDependenciesForModule(moduleName);
188     this._loadCSSForModule(moduleName);
189     this._loadJavaScriptForModule(moduleName);
190     module.isLoading = YES;
191     module.isPrefetching = YES;
192   },

198   /** @private
199     If a module is marked for lazy instantiation, this method will execute the closure and call
200     any registered callbacks.
201   */
202   _executeLazilyInstantiatedModule: function (moduleName, targetName, methodName) {
203     var lazyInfo =  SC.LAZY_INSTANTIATION[moduleName];
204     var target;
205     var method;
206     var idx, len;
207 
208     if (SC.LOG_MODULE_LOADING) {
209       SC.debug("SC.Module: Module '%@' is marked for lazy instantiation, instantiating it now…", moduleName);
210     }
211 
212     len = lazyInfo.length;
213     for (idx = 0; idx < len; idx++) {
214       // Iterate through each function associated with this module, and attempt to execute it.
215       try {
216         lazyInfo[idx]();
217       } catch (e) {
218         SC.Logger.error("SC.Module: Failed to lazily instatiate entry for  '%@'".fmt(moduleName));
219       }
220     }
221 
222     // Free up memory containing the functions once they have been executed.
223     delete SC.LAZY_INSTANTIATION[moduleName];
224 
225     // Now that we have executed the functions, try to find the target and action for the callback.
226     target = this._targetForTargetName(targetName);
227     method = this._methodForMethodNameInTarget(methodName, target);
228 
229     if (!method) {
230       throw new Error("SC.Module: could not find callback for lazily instantiated module '%@'".fmt(moduleName));
231     }
232   },

234   /**
235     Evaluates a module's JavaScript if it is stored in string format, then
236     deletes that code from memory.
237 
238     @param {Hash} module the module to evaluate
239   */
240   _evaluateStringLoadedModule: function (module) {
241     var moduleSource = module.source;
242 
243     // so, force a run loop.
244     jQuery.globalEval(moduleSource);
245 
246     delete module.source;
247 
248     if (module.cssSource) {
249       var el = document.createElement('style');
250       el.setAttribute('type', 'text/css');
251       if (el.styleSheet) {
252         el.styleSheet.cssText = module.cssSource;
253       } else {
254         var content = document.createTextNode(module.cssSource);
255         el.appendChild(content);
256       }
257 
258       document.getElementsByTagName('head')[0].appendChild(el);
259     }
260 
261     module.isReady = YES;
262   },

264   /**
265     @private
266 
267     Creates <link> tags for every CSS resource in a module.
268 
269     @param {String} moduleName the name of the module whose CSS should be loaded
270   */
271   _loadCSSForModule: function (moduleName) {
272     var head = document.getElementsByTagName('head')[0];
273     var module = SC.MODULE_INFO[moduleName];
274     var styles = module.styles || [];
275     var len = styles.length;
276     var url;
277     var el;
278     var idx;
279 
280     if (!head) head = document.documentElement; // fix for Opera
281     len = styles.length;
282 
283     for (idx = 0; idx < len; idx++) {
284       url = styles[idx];
285 
286       if (url.length > 0) {
287         if (SC.LOG_MODULE_LOADING) SC.debug("SC.Module: Loading CSS file in '%@' -> '%@'", moduleName, url);
288 
289         // if we are on a retina device lets load the retina style sheet instead
290         if (window.devicePixelRatio > 1 || window.location.search.indexOf("2x") > -1) {
291           url = url.replace('.css', '@2x.css');
292         }
293 
294         el = document.createElement('link');
295         el.setAttribute('href', url);
296         el.setAttribute('rel', "stylesheet");
297         el.setAttribute('type', "text/css");
298         head.appendChild(el);
299       }
300     }
301 
302     el = null;
303   },
304 
305   _loadJavaScriptForModule: function (moduleName) {
306     var module = SC.MODULE_INFO[moduleName];
307     var el;
308     var url;
309     var dependencies = module.dependencies;
310     var dependenciesAreLoaded = YES;
311 
312     // If this module has dependencies, determine if they are loaded.
313     if (dependencies && dependencies.length > 0) {
314       dependenciesAreLoaded = this._dependenciesMetForModule(moduleName);
315     }
316 
317     // If the module is prefetched, always load the string representation.
318     if (module.isPrefetched) {
319       url = module.stringURL;
320     } else {
321       if (dependenciesAreLoaded) {
322         // Either we have no dependencies or they've all loaded already,
323         // so just execute the code immediately once it loads.
324         url = module.scriptURL;
325       } else {
326         // Because the dependencies might load after this module, load the
327         // string representation so we can execute it once all dependencies
328         // are in place.
329         url = module.stringURL;
330       }
331     }
332 
333     if (url.length > 0) {
334       if (SC.LOG_MODULE_LOADING) SC.debug("SC.Module: Loading JavaScript file in '%@' -> '%@'", moduleName, url);
335 
336       el = document.createElement('script');
337       el.setAttribute('type', "text/javascript");
338       el.setAttribute('src', url);
339 
340       if (el.addEventListener) {
341         el.addEventListener('load', function () {
342           SC.run(function () {
343             SC.Module._moduleDidLoad(moduleName);
344           });
345         }, false);
346       } else if (el.readyState) {
347         el.onreadystatechange = function () {
348           if (this.readyState === 'complete' || this.readyState === 'loaded') {
349             SC.run(function () {
350               SC.Module._moduleDidLoad(moduleName);
351             });
352           }
353         };
354       }
355 
356       document.body.appendChild(el);
357     }
358   },

360   /**
361     @private
362 
363     Returns YES if all of the dependencies for a module are ready.
364 
365     @param {String} moduleName the name of the module being checked
366     @returns {Boolean} whether the dependencies are loaded
367   */
368   _dependenciesMetForModule: function (moduleName) {
369     var dependencies = SC.MODULE_INFO[moduleName].dependencies || [];
370     var idx, len = dependencies.length;
371     var dependencyName;
372     var module;
373 
374     for (idx = 0; idx < len; idx++) {
375       dependencyName = dependencies[idx];
376       module = SC.MODULE_INFO[dependencyName];
377 
378       if (!module) throw new Error("SC.loadModule: Unable to find dependency %@ for module %@.".fmt(dependencyName, moduleName));
379 
380       if (!module.isReady) {
381         return NO;
382       }
383     }
384 
385     return YES;
386   },

388   /**
389     Loads all unloaded dependencies for a module, then creates the <script> and <link> tags to
390     load the JavaScript and CSS for the module.
391   */
392   _loadDependenciesForModule: function (moduleName) {
393     // Load module's dependencies first.
394     var moduleInfo      = SC.MODULE_INFO[moduleName];
395     var log             = SC.LOG_MODULE_LOADING;
396     var dependencies    = moduleInfo.dependencies || [];
397     var dependenciesMet = YES;
398     var len             = dependencies.length;
399     var idx;
400     var requiredModuleName;
401     var requiredModule;
402     var dependents;
403 
404     for (idx = 0; idx < len; idx++) {
405       requiredModuleName = dependencies[idx];
406       requiredModule = SC.MODULE_INFO[requiredModuleName];
407 
408       // Try to find dependent module in MODULE_INFO
409       if (!requiredModule) {
410         throw new Error("SC.Module: could not find required module '%@' for module '%@'".fmt(requiredModuleName, moduleName));
411       } else {
412 
413         // Required module has been requested but hasn't loaded yet.
414         if (requiredModule.isLoading) {
415           dependenciesMet = NO;
416 
417           dependents = requiredModule.dependents;
418           if (!dependents) requiredModule.dependents = dependents = [];
419           dependents.push(moduleName);
420         }
421 
422         // Required module has already been loaded and evaluated, no need to worry about it.
423         else if (requiredModule.isReady) {
424           continue;
425         }
426         // Required module has not been loaded nor requested yet.
427         else {
428           dependenciesMet = NO;
429 
430           // Register this as a dependent module (used by SC._moduleDidLoad()...)
431           dependents = requiredModule.dependents;
432           if (!dependents) requiredModule.dependents = dependents = [];
433 
434           dependents.push(moduleName);
435 
436           if (log) SC.debug("SC.Module: '%@' depends on '%@', loading dependency…", moduleName, requiredModuleName);
437 
438           // Load dependencies
439           SC.Module.loadModule(requiredModuleName);
440         }
441       }
442     }
443   },

445   /**
446     @private
447 
448     Calls an action on a target to notify the target that a module has loaded.
449   */
450   _invokeCallback: function (moduleName, targetName, methodName, args) {
451     var method;
452     var target;
453 
454     target = this._targetForTargetName(targetName);
455     method = this._methodForMethodNameInTarget(methodName, target);
456 
457     // If we weren't able to find the callback, this module may be lazily instantiated and
458     // the callback won't exist until we execute the closure that it is wrapped in.
459     if (!method) {
460       if (SC.LAZY_INSTANTIATION[moduleName]) {
461         this._executeLazilyInstantiatedModule(moduleName, targetName, methodName);
462 
463         target = this._targetForTargetName(targetName);
464         method = this._methodForMethodNameInTarget(methodName, target);
465       } else {
466         throw new Error("SC.Module: could not find callback for '%@'".fmt(moduleName));
467       }
468     }
469 
470     if (!args) {
471       args = [];
472     }
473 
474     // The first parameter passed to the callback is the name of the module.
475     args.unshift(moduleName);
476 
477     // Invoke the callback. Wrap it in a run loop if we are not in a runloop already.
478     var needsRunLoop = !!SC.RunLoop.currentRunLoop;
479     if (needsRunLoop) {
480       SC.run(function () {
481         method.apply(target, args);
482       });
483     } else {
484       method.apply(target, args);
485     }
486   },

488   /** @private
489     Given a module name, iterates through all registered callbacks and calls them.
490   */
491   _invokeCallbacksForModule: function (moduleName) {
492     var moduleInfo = SC.MODULE_INFO[moduleName], callbacks;
493     if (!moduleInfo) return; // shouldn't happen, but recover anyway
494 
495     if (SC.LOG_MODULE_LOADING) SC.debug("SC.Module: Module '%@' has completed loading, invoking callbacks.", moduleName);
496 
497     callbacks = moduleInfo.callbacks || [];
498 
499     for (var idx = 0, len = callbacks.length; idx < len; ++idx) {
500       callbacks[idx]();
501     }
502   },
503 
504   _evaluateAndInvokeCallbacks: function (moduleName) {
505     var moduleInfo = SC.MODULE_INFO;
506     var module = moduleInfo[moduleName];
507     var log = SC.LOG_MODULE_LOADING;
508 
509     if (log) SC.debug("SC.Module: Evaluating and invoking callbacks for '%@'.", moduleName);
510 
511     if (module.source) {
512       this._evaluateStringLoadedModule(module);
513     }
514 
515     // this is ugly, but a module evaluated late like this won't be done instantiating
516     // until the end of a run loop. Also, the code here is not structured in a way that makes
517     // it easy to "add a step" before saying a module is ready. And finally, invokeLater doesn't
518     // accept arguments; hence, the closure.
519     module.isWaitingForRunLoop = YES;
520     this.invokeLast(function () {
521       module.isReady = YES;
522       this._moduleDidBecomeReady(moduleName);
523     });
524   },
525 
526   _moduleDidBecomeReady: function (moduleName) {
527     var moduleInfo = SC.MODULE_INFO;
528     var module = moduleInfo[moduleName];
529     var log = SC.LOG_MODULE_LOADING;
530 
531     module.isWaitingForRunLoop = NO;
532 
533     if (SC.isReady) {
534       SC.Module._invokeCallbacksForModule(moduleName);
535       delete module.callbacks;
536     } else {
537       SC.ready(SC, function () {
538         SC.Module._invokeCallbacksForModule(moduleName);
539         delete module.callbacks;
540       });
541     }
542 
543     // for each dependent module, try and load them again...
544     var dependents = module.dependents || [];
545     var dependentName, dependent;
546 
547     for (var idx = 0, len = dependents.length; idx < len; idx++) {
548       dependentName = dependents[idx];
549       dependent = moduleInfo[dependentName];
550       if (dependent.isLoaded && this._dependenciesMetForModule(dependentName)) {
551         if (log) SC.debug("SC.Module: Now that %@ has loaded, all dependencies for a dependent %@ are met.", moduleName, dependentName);
552         this._evaluateAndInvokeCallbacks(dependentName);
553       }
554     }
555 
556   },

558   /** @private
559     Called when the JavaScript for a module finishes loading.
560 
561     Any pending callbacks are called (if SC.isReady), and any dependent
562     modules which were waiting for this module to load are notified so they
563     can continue loading.
564 
565     @param moduleName {String} the name of the module that just loaded
566   */
567   _moduleDidLoad: function (moduleName) {
568     var module = SC.MODULE_INFO[moduleName];
569     var log    = SC.LOG_MODULE_LOADING;
570     var dependenciesMet;
571     var callbacks, targets;
572 
573     if (log) SC.debug("SC.Module: Module '%@' finished loading.", moduleName);
574 
575     if (!module) {
576       if (log) SC.debug("SC._moduleDidLoad() called for unknown module '@'.", moduleName);
577       module = SC.MODULE_INFO[moduleName] = { isLoaded: YES, isReady: YES };
578       return;
579     }
580 
581     if (module.isLoaded) {
582       if (log) SC.debug("SC._moduleDidLoad() called more than once for module '%@'. Skipping.", moduleName);
583       return;
584     }
585 
586     // Remember that we're loaded.
587     delete module.isLoading;
588     module.isLoaded = YES;
589 
590     if (!module.isPrefetching) {
591       dependenciesMet = this._dependenciesMetForModule(moduleName);
592       if (dependenciesMet) {
593         this._evaluateAndInvokeCallbacks(moduleName);
594       } else {
595         if (log) SC.debug("SC.Module: Dependencies for '%@' not met yet, waiting to evaluate.", moduleName);
596       }
597     } else {
598       delete module.isPrefetching;
599       if (log) SC.debug("SC.Module: Module '%@' was prefetched, not evaluating until needed.", moduleName);
600     }
601   },

636   /**
637     A list of the methods to temporarily disable (and buffer calls for) when we are suspended.
638   */
639   methodsForSuspend: "loadModule _moduleDidLoad prefetchModule _moduleDidBecomeReady".w(),
640 
641   /**
642     Call this in order to prevent expensive tasks from occurring at inopportune times.
643   */
644   suspend: function () {
645 
646     //Increment the suspension count, to support nested suspend()/resume() pairs.
647     //We only do anything if the suspend count ends up at 1, as that implies it's
648     //the first suspend() call.
649     this._suspendCount = (this._suspendCount || 0) + 1;
650     if (this._suspendCount !== 1) return;
651 
652     //Yummy variables.
653     var methods = this.get('methodsForSuspend'),
654         replaceKey, saveKey, key, i;
655 
656     //Now we go through the list of methods to suspend, and overwrite them with
657     //versions that will buffer their calls in a _bufferedCalls array.
658     for (i = 0; (key = methods[i]); i++) {
659       //jshint loopfunc: true
660       //Ensure the replacement function exists at a key where it'll be cached.
661       if (!this[replaceKey = "__replacement_" + key + "__"]) {
662         (this[replaceKey] = function () {
663           (this._bufferedCalls || (this._bufferedCalls = [])).push({
664             method: arguments.callee.methodName,
665             arguments: arguments
666           });
667         }).methodName = key;
668       }
669 
670       //Ensure the original function exists at a key where it'll be cached.
671       if (!this[saveKey = "__saved_" + key + "__"]) this[saveKey] = this[key];
672 
673       //Ensure that the replacement function exists where the rest of the
674       //code expects the original.
675       this[key] = this[replaceKey];
676     }
677   },
678 
679   /**
680     Call this in order to resume normal behavior of the methods here, and to
681     finally perform any calls that may have occurred during suspension. Calls
682     will run in the order they were received.
683   */
684   resume: function () {
685 
686     //First, we need to decrement the suspension count, and warn if the suspension
687     //count implied that we weren't already suspended. Furthermore, if the suspend
688     //count is not zero, then we haven't tackled the last suspend() call with a resume(),
689     //and should therefore not resume.
690     this._suspendCount = (this._suspendCount || 0) - 1;
691     if (this._suspendCount < 0) {
692       SC.warn("SC.Module.resume() was called without SC.Module having been in a suspended state. Call aborted.");
693       this._suspendCount = 0;
694       return;
695     }
696     if (this._suspendCount > 0) return;
697 
698     //Yummy variables.
699     var methods = this.get('methodsForSuspend'),
700         calls = this._bufferedCalls,
701         key, i, method, call;
702 
703     //Restore the original methods to where they belong for normal functionality.
704     for (i = 0; (key = methods[i]); i++) this[key] = this["__saved_" + key + "__"];
705 
706     //Perform any buffered calls that built up during the suspended period.
707     for (i = 0; (call = calls) && calls[i]; i++) this[call.method].apply(this, call.arguments);
708 
709     //Clear the list of calls, so subsequent resume() calls won't flush them again.
710     if (calls) calls.length = 0;
711   }
712 });

714 /**
715 Inspect the list of modules and, for every prefetched module, create a
716 background task to load the module when the user remains idle.
717 */
718 SC.ready(function () {
719   var moduleInfo = SC.MODULE_INFO;
720   var moduleName;
721   var module;
722   var task;
723 
724   // Iterate through all known modules and look for those that are marked
725   // as prefetched.
726   for (moduleName in moduleInfo) {
727     module = moduleInfo[moduleName];
728 
729     if (module.isPrefetched) {
730       // Create a task that will load the module, and then register it with
731       // the global background task queue.
732       task = SC.Module.PrefetchModuleTask.create({ prefetchedModuleName: moduleName });
733       SC.backgroundTaskQueue.push(task);
734     }
735   }
736 });
737 
738 SC.Module.PrefetchModuleTask = SC.Task.extend({
739   prefetchedModuleName: null,
740   run: function () {
741     SC.Module.prefetchModule(this.prefetchedModuleName);
742   }
743 });
