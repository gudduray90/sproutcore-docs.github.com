 25     this.set('content', root);

 36   /**
 37     Called by the collection view just before it starts a drag so that 
 38     you can provide the data types you would like to support in the data.
 39 
 40     You can implement this method to return an array of the data types you
 41     will provide for the drag data.
 42 
 43     If you return null or an empty array, can you have set canReorderContent
 44     to YES on the CollectionView, then the drag will go ahead but only 
 45     reordering will be allowed.  If canReorderContent is NO, then the drag
 46     will not be allowed to start.
 47 
 48     If you simply want to control whether a drag is allowed or not, you
 49     should instead implement collectionViewShouldBeginDrag().
 50 
 51     The default returns an empty array.
 52 
 53     @param view {SC.CollectionView} the collection view to begin dragging.
 54     @returns {Array} array of supported data types.
 55   */
 56   collectionViewDragDataTypes: function(view) { return ['SC.Object']; },
 57 
 58   /**
 59     Called by a collection view when a drag concludes to give you the option
 60     to provide the drag data for the drop.
 61 
 62     This method should be implemented essentially as you would implement the
 63     dragDataForType() if you were a drag data source.  You will never be asked
 64     to provide drag data for a reorder event, only for other types of data.
 65 
 66     The default implementation returns null.
 67 
 68     @param view {SC.CollectionView} 
 69       the collection view that initiated the drag
 70 
 71     @param dataType {String} the data type to provide
 72     @param drag {SC.Drag} the drag object
 73     @returns {Object} the data object or null if the data could not be provided.
 74   */
 75   collectionViewDragDataForType: function(view, drag, dataType) {
 76     var ret = (dataType === 'SC.Object') ? this.get('selection').firstObject() : null;
 77     return ret ;
 78   },

 83     Renders a drag view for the passed content indexes. If you return null

115   init: function(){

121   _searchDidChange: function(){
122     this._runSearch();

125  _sanitizeSearchString: function(str){
126    var specials = [
127        '/', '.', '*', '+', '?', '|',
128        '(', ')', '[', ']', '{', '}', '\\'
129    ];
130    var s = new RegExp('(\\' + specials.join('|\\') + ')', 'g');
131    return str.replace(s, '\\$1');
132  },
133 
134  _runSearch: function(){
135    var searchResults = [];
136    var search = this.get('search');
137    var c = this.get('content');
138    if(search === null || search === '' || search === undefined){ 
139      this.set('searchResults', c);
140    }
141    else {
142      search = this._sanitizeSearchString(search).toLowerCase();
143      var searchRegex = new RegExp(search,'i');
144      var searchKey = this.get('searchKey');
145      this._iconKey = this.get('iconKey');
146      this._nameKey = this.get('nameKey');
147      searchResults = this._runSearchOnItem(c, search, searchRegex, searchKey);
148 
149      // create the root search tree
150      var searchedTree = SC.Object.create({
151        treeItemIsExpanded: YES,
152        treeItemChildren: searchResults
153      });
154      this.set('searchResults', searchedTree);
155    }
156  },
157 
158  /** 
159    @private
160    Returns a flat list of matches for the foldered tree item.
161  */
162  _runSearchOnItem: function(treeItem, search, searchRegex, searchKey) {
163    var searchMatches = [], iconKey = this.get('iconKey'),
164        searchedList, key, searchLen, 
165        children, nameKey = this._nameKey, that;
166 
167    if (SC.none(treeItem)) return searchMatches;
168 
169    children = treeItem.get('treeItemChildren');
170    if (!children) children = treeItem.get('children');
171    that = this;
172    children.forEach( function(child){      
173      if (child.treeItemChildren) {
174        var searchedList = that._runSearchOnItem(child, search, searchRegex, searchKey);
175        searchedList.forEach( function(m){ searchMatches.push(m); });
176      }
177 
178      if (searchKey && child.get(searchKey)) {
179        key = child.get(searchKey).toLowerCase();
180        if(key.match(searchRegex)){
181          var match = SC.Object.create({});
182          match[searchKey]  = child.get(searchKey);
183          match[nameKey]    = child.get(nameKey);
184          match.treeItem    = child;
185          match.icon        = child.get(this._iconKey);
186          searchMatches.push(match);
187        } 
188      }
189    });
190 
191    return searchMatches;
192  }
