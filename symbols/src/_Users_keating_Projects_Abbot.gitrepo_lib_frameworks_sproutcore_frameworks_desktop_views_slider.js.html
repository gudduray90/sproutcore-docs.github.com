  8 /** @class
  9 
 10   A SliderView shows a horizontal slider control that you can use to set 
 11   variable values.
 12 
 13   You can use a slider view much like you would any other control.  Simply
 14   set the value or content/contentValueKey to whatever value you want to 
 15   display.  You can also set the maximumValue and minValue properties to 
 16   determine the mapping of the control to its children.
 17   
 18   @extends SC.View
 19   @extends SC.Control
 20   @since SproutCore 1.0
 21   @test in progress
 22 */
 23 SC.SliderView = SC.View.extend(SC.Control,
 24 /** @scope SC.SliderView.prototype */ {
 25   
 26   classNames: 'sc-slider-view',
 27 
 28   /**
 29     The WAI-ARIA role for slider view. This property's value should not be
 30     changed.
 31 
 32     @property {String}
 33   */
 34   ariaRole: 'slider',
 35 
 36   /**
 37     Bind this to the current value of the progress bar.  Note that by default 
 38     an empty value will disable the progress bar and a multiple value too make 
 39     it indeterminate.
 40   */
 41   value: 0.50,
 42   valueBindingDefault: SC.Binding.single().notEmpty(),
 43   
 44   /**
 45     The minimum value of the progress.
 46   */ 
 47   minimum: 0,
 48   minimumBindingDefault: SC.Binding.single().notEmpty(),
 49 
 50   /**
 51     Optionally specify the key used to extract the minimum progress value 
 52     from the content object.  If this is set to null then the minimum value
 53     will not be derived from the content object.
 54     
 55     @property {String}
 56   */
 57   contentMinimumKey: null,
 58   
 59   /**
 60     The maximum value of the progress bar.
 61   */
 62   maximum: 1.0,
 63   maximumBindingDefault: SC.Binding.single().notEmpty(),
 64 
 65   /**
 66     Optionally specify the key used to extract the maximum progress value 
 67     from the content object.  If this is set to null then the maximum value
 68     will not be derived from the content object.
 69     
 70     @property {String}
 71   */
 72   contentMaximumKey: null,
 73   
 74   /**
 75     Optionally set to the minimum step size allowed.
 76     
 77     All values will be rounded to this step size when displayed.
 78   */
 79   step: 0.1,
 80 
 81   // ..........................................................
 82   // INTERNAL PROPERTIES
 83   // 
 84   
 85   displayProperties: ['displayValue', 'ariaValue', 'minimum', 'maximum', 'step', 'frame'],
 86 
 87   /**
 88    @property
 89    The raw, unchanged value to be provided to screen readers and the like.
 90   */
 91   ariaValue: function() {
 92     return this.get('value');
 93   }.property('value').cacheable(),
 94 
 95   // The name of the render delegate which is creating and maintaining
 96   // the DOM associated with instances of this view
 97   renderDelegateName: 'sliderRenderDelegate',
 98   
 99   displayValue: function() {
100     var min = this.get('minimum'),
101         max = this.get('maximum'),
102         value = this.get('value'),
103         step = this.get('step');
104 
105     // determine the constrained value.  Must fit within min & max
106     value = Math.min(Math.max(value, min), max);
107 
108     // limit to step value
109     if (!SC.none(step) && step !== 0) {
110       value = Math.round(value / step) * step;
111     }
112     
113     // determine the percent across
114     if(value!==0) value = Math.floor((value - min) / (max - min) * 100);
115     
116     return value;
117   }.property('value', 'minimum', 'maximum', 'step').cacheable(),
118   
119   _isMouseDown: NO,
120   
121   mouseDown: function(evt) {
122     if (!this.get('isEnabled')) return YES; // nothing to do...
123     this.set('isActive', YES);
124     this._isMouseDown = YES ;
125     return this._triggerHandle(evt, YES);
126   },
127   
128   // mouseDragged uses same technique as mouseDown.
129   mouseDragged: function(evt) { 
130     return this._isMouseDown ? this._triggerHandle(evt) : YES; 
131   },
132   
133   // remove active class
134   mouseUp: function(evt) {
135     if (this._isMouseDown) this.set('isActive', NO);
136     var ret = this._isMouseDown ? this._triggerHandle(evt) : YES ;
137     this._isMouseDown = NO;
138     return ret ;
139   },
140   
141   mouseWheel: function(evt) {
142     if (!this.get('isEnabled')) return YES;
143     var min = this.get('minimum'),
144         max = this.get('maximum'),
145         newVal = this.get('value')+((evt.wheelDeltaX+evt.wheelDeltaY)*0.01),
146         step = this.get('step'),
147         value = Math.round(newVal / step) * step ;
148     if (newVal< min) this.setIfChanged('value', min);
149     else if (newVal> max) this.setIfChanged('value', max);
150     else this.setIfChanged('value', newVal);
151     return YES ;  
152   },
153   
154   touchStart: function(evt){
155     return this.mouseDown(evt);
156   },
157   
158   touchEnd: function(evt){
159     return this.mouseUp(evt);
160   },
161   
162   touchesDragged: function(evt){
163     return this.mouseDragged(evt);
164   },
165   
166   /** @private
167     Updates the handle based on the mouse location of the handle in the
168     event.
169   */
170   _triggerHandle: function(evt, firstEvent) {
171     var width = this.get('frame').width,
172         min = this.get('minimum'), max=this.get('maximum'),
173         step = this.get('step'), v=this.get('value'), loc;
174         
175     if(firstEvent){    
176       loc = this.convertFrameFromView({ x: evt.pageX }).x;
177       this._evtDiff = evt.pageX - loc;
178     }else{
179       loc = evt.pageX-this._evtDiff;
180     }
181     
182     // convert to percentage
183     loc = Math.max(0, Math.min(loc / width, 1));
184     
185     // if the location is NOT in the general vicinity of the slider, we assume
186     // that the mouse pointer or touch is in the center of where the knob should be.
187     // otherwise, if we are starting, we need to do extra to add an offset
188     if (firstEvent) {
189       var value = this.get("value");
190       value = (value - min) / (max - min);
191       
192       // if the value and the loc are within 16px
193       if (Math.abs(value * width - loc * width) < 16) this._offset = value - loc;
194       else this._offset = 0;
195     }
196     
197     // add offset and constrain
198     loc = Math.max(0, Math.min(loc + this._offset, 1));
199 
200     // convert to value using minimum/maximum then constrain to steps
201     loc = min + ((max-min)*loc);
202     if (step !== 0) loc = Math.round(loc / step) * step ;
203     
204     // if changes by more than a rounding amount, set v.
205     if (Math.abs(v-loc)>=0.01) {
206       this.set('value', loc); // adjust 
207     }
208     
209     return YES ;
210   },
211   
212   /** tied to the isEnabled state */
213   acceptsFirstResponder: function() {
214     if (SC.FOCUS_ALL_CONTROLS) { return this.get('isEnabled'); }
215     return NO;
216   }.property('isEnabled'),
217   
218   keyDown: function(evt) {
219 
220      // handle tab key
221      if (evt.which === 9 || evt.keyCode === 9) {
222        var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
223        if(view) view.becomeFirstResponder();
224        else evt.allowDefault(); 
225        return YES ; // handled
226      }
227      
228      if (evt.which >= 37 && evt.which <= 40){
229        var min = this.get('minimum'),max=this.get('maximum'),
230           step = this.get('step'),
231           size = max-min, val=0, calculateStep;
232      
233        if (evt.which === 37 || evt.which === 38 ){
234          if(step === 0){
235            if(size<100){
236              val = this.get('value')-1;
237            }else{
238              calculateStep = Math.abs(size/100);
239              if(calculateStep<2) calculateStep =2;
240              val = this.get('value')-Math.abs(size/100);
241            }
242          }else{
243            val = this.get('value')-step;
244          }
245        }
246        if (evt.which === 39 || evt.which === 40 ){
247            if(step === 0){
248               if(size<100){
249                 val = this.get('value') + 2;
250               }else{
251                 calculateStep = Math.abs(size/100);
252                 if(calculateStep<2) calculateStep =2;
253                 val = this.get('value')+calculateStep;
254               }
255             }else{
256               val = this.get('value')+step;
257             }       
258        }
259        if(val>=min && val<=max) this.set('value', val);
260      }else{
261        evt.allowDefault();
262        return NO;
263      }
264      return YES; 
265    },
266 
267   contentPropertyDidChange: function(target, key) {
268     var content = this.get('content');
269     this.beginPropertyChanges()
270       .updatePropertyFromContent('value', key, 'contentValueKey', content)
271       .updatePropertyFromContent('minimum', key, 'contentMinimumKey', content)
272       .updatePropertyFromContent('maximum', key, 'contentMaximumKey', content)
273       .updatePropertyFromContent('isIndeterminate', key, 'contentIsIndeterminateKey', content)
274     .endPropertyChanges();
275   }  
276   
277 });
