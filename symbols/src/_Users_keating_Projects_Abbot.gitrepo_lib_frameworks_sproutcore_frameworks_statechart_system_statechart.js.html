 12 /**
 13   @class
 14 
 15   The startchart manager mixin allows an object to be a statechart. By becoming a statechart, the
 16   object can then be manage a set of its own states.
 17   
 18   This implemention of the statechart manager closely follows the concepts stated in D. Harel's 
 19   original paper "Statecharts: A Visual Formalism For Complex Systems" 
 20   (www.wisdom.weizmann.ac.il/~harel/papers/Statecharts.pdf). 
 21   
 22   The statechart allows for complex state heircharies by nesting states within states, and 
 23   allows for state orthogonality based on the use of concurrent states.
 24   
 25   At minimum, a statechart must have one state: The root state. All other states in the statechart
 26   are a decendents (substates) of the root state.
 27   
 28   The following example shows how states are nested within a statechart:
 29   
 30       MyApp.Statechart = SC.Object.extend(SC.StatechartManager, {
 31         rootState: SC.State.design({
 32           initialSubstate: 'stateA',
 33 
 34           stateA: SC.State.design({
 35             // ... can continue to nest further states
 36           }),
 37         
 38           stateB: SC.State.design({
 39             // ... can continue to nest further states
 40           })
 41         })
 42       });
 43   
 44   Note how in the example above, the root state as an explicit initial substate to enter into. If no
 45   initial substate is provided, then the statechart will default to the the state's first substate.
 46   
 47   You can also defined states without explicitly defining the root state. To do so, simply create properties
 48   on your object that represents states. Upon initialization, a root state will be constructed automatically
 49   by the mixin and make the states on the object substates of the root state. As an example:
 50   
 51       MyApp.Statechart = SC.Object.extend(SC.StatechartManager, {
 52         initialState: 'stateA',
 53 
 54         stateA: SC.State.design({
 55           // ... can continue to nest further states
 56         }),
 57       
 58         stateB: SC.State.design({
 59           // ... can continue to nest further states
 60         })
 61       });
 62   
 63   If you liked to specify a class that should be used as the root state but using the above method to defined
 64   states, you can set the rootStateExample property with a class that extends from SC.State. If the 
 65   rootStateExample property is not explicitly assigned the then default class used will be SC.State.
 66   
 67   To provide your statechart with orthogonality, you use concurrent states. If you use concurrent states,
 68   then your statechart will have multiple current states. That is because each concurrent state represents an
 69   independent state structure from other concurrent states. The following example shows how to provide your
 70   statechart with concurrent states:
 71   
 72       MyApp.Statechart = SC.Object.extend(SC.StatechartManager, {
 73         rootState: SC.State.design({
 74           substatesAreConcurrent: YES,
 75 
 76           stateA: SC.State.design({
 77             // ... can continue to nest further states
 78           }),
 79         
 80           stateB: SC.State.design({
 81             // ... can continue to nest further states
 82           })
 83         })
 84       });
 85   
 86   Above, to indicate that a state's substates are concurrent, you just have to set the substatesAreConcurrent to 
 87   YES. Once done, then stateA and stateB will be independent of each other and each will manage their
 88   own current substates. The root state will then have more then one current substate.
 89   
 90   To define concurrent states directly on the object without explicitly defining a root, you can do the 
 91   following:
 92   
 93       MyApp.Statechart = SC.Object.extend(SC.StatechartManager, {
 94         statesAreConcurrent: YES,
 95 
 96         stateA: SC.State.design({
 97           // ... can continue to nest further states
 98         }),
 99     
100         stateB: SC.State.design({
101           // ... can continue to nest further states
102         })
103       });
104   
105   Remember that a startchart can have a mixture of nested and concurrent states in order for you to 
106   create as complex of statecharts that suite your needs. Here is an example of a mixed state structure:
107   
108       MyApp.Statechart = SC.Object.extend(SC.StatechartManager, {
109         rootState: SC.State.design({
110           initialSubstate: 'stateA',
111 
112           stateA: SC.State.design({
113             substatesAreConcurrent: YES,
114 
115             stateM: SC.State.design({ ... })
116             stateN: SC.State.design({ ... })
117             stateO: SC.State.design({ ... })
118           }),
119         
120           stateB: SC.State.design({
121             initialSubstate: 'stateX',
122 
123             stateX: SC.State.design({ ... })
124             stateY: SC.State.desgin({ ... })
125           })
126         })
127       });
128   
129   Depending on your needs, a statechart can have lots of states, which can become hard to manage all within
130   one file. To modularize your states and make them easier to manage and maintain, you can plug-in states
131   into other states. Let's say we are using the statechart in the last example above, and all the code is 
132   within one file. We could update the code and split the logic across two or more files like so:
133 
134       // state_a.js
135 
136       MyApp.StateA = SC.State.extend({
137         substatesAreConcurrent: YES,
138 
139         stateM: SC.State.design({ ... })
140         stateN: SC.State.design({ ... })
141         stateO: SC.State.design({ ... })
142       });
143 
144       // state_b.js
145 
146       MyApp.StateB = SC.State.extend({
147         substatesAreConcurrent: YES,
148 
149         stateM: SC.State.design({ ... })
150         stateN: SC.State.design({ ... })
151         stateO: SC.State.design({ ... })
152       });
153 
154       // statechart.js
155 
156       MyApp.Statechart = SC.Object.extend(SC.StatechartManager, {
157         rootState: SC.State.design({
158           initialSubstate: 'stateA',
159           stateA: SC.State.plugin('MyApp.StateA'),
160           stateB: SC.State.plugin('MyApp.StateB')
161         })
162       });
163 
164   Using state plug-in functionality is optional. If you use the plug-in feature you can break up your statechart
165   into as many files as you see fit.
166 
167   @author Michael Cohen
168 */

327   /**
328     Initializes the statechart. By initializing the statechart, it will create all the states and register
329     them with the statechart. Once complete, the statechart can be used to go to states and send events to.
330   */
331   initStatechart: function() {
332     if (this.get('statechartIsInitialized')) return;
333     
334     this._gotoStateLocked = NO;
335     this._sendEventLocked = NO;
336     this._pendingStateTransitions = [];
337     this._pendingSentEvents = [];
338     
339     this.sendAction = this.sendEvent;
340     
341     if (this.get('monitorIsActive')) {
342       this.set('monitor', SC.StatechartMonitor.create({ statechart: this }));
343     }
344 
345     var traceKey = this.get('statechartTraceKey');
346 
347     this.addObserver(traceKey, this, '_statechartTraceDidChange');
348     this._statechartTraceDidChange();
349 
350     var trace = this.get('allowStatechartTracing'),
351         rootState = this.get('rootState'),
352         msg;
353     
354     if (trace) this.statechartLogTrace("BEGIN initialize statechart");
355     
356     // If no root state was explicitly defined then try to construct
357     // a root state class
358     if (!rootState) {
359       rootState = this._constructRootStateClass();
360     }
361     else if (SC.typeOf(rootState) === SC.T_FUNCTION && rootState.statePlugin) {
362       rootState = rootState.apply(this);
363     }
364     
365     if (!(SC.kindOf(rootState, SC.State) && rootState.isClass)) {
366       msg = "Unable to initialize statechart. Root state must be a state class";
367       this.statechartLogError(msg);
368       throw msg;
369     }
370     
371     rootState = this.createRootState(rootState, { 
372       statechart: this, 
373       name: SC.ROOT_STATE_NAME 
374     });
375     
376     this.set('rootState', rootState);
377     rootState.initState();
378     
379     if (SC.kindOf(rootState.get('initialSubstate'), SC.EmptyState)) {
380       msg = "Unable to initialize statechart. Root state must have an initial substate explicilty defined";
381       this.statechartLogError(msg);
382       throw msg;
383     }
384     
385     if (!SC.empty(this.get('initialState'))) {
386       var key = 'initialState';
387       this.set(key, rootState.get(this.get(key)));
388     } 
389     
390     this.set('statechartIsInitialized', YES);
391     this.gotoState(rootState);
392     
393     if (trace) this.statechartLogTrace("END initialize statechart");
394   },

484   /**
485     When called, the statechart will proceed with making state transitions in the statechart starting from 
486     a current state that meet the statechart conditions. When complete, some or all of the statechart's 
487     current states will be changed, and all states that were part of the transition process will either 
488     be exited or entered in a specific order.
489     
490     The state that is given to go to will not necessarily be a current state when the state transition process
491     is complete. The final state or states are dependent on factors such an initial substates, concurrent 
492     states, and history states.
493     
494     Because the statechart can have one or more current states, it may be necessary to indicate what current state
495     to start from. If no current state to start from is provided, then the statechart will default to using
496     the first current state that it has; depending of the make up of the statechart (no concurrent state vs.
497     with concurrent states), the outcome may be unexpected. For a statechart with concurrent states, it is best
498     to provide a current state in which to start from.
499     
500     When using history states, the statechart will first make transitions to the given state and then use that
501     state's history state and recursively follow each history state's history state until there are no 
502     more history states to follow. If the given state does not have a history state, then the statechart
503     will continue following state transition procedures.
504     
505     Method can be called in the following ways:
506     
507         // With one argument. 
508         gotoState(<state>)
509       
510         // With two argument.
511         gotoState(<state>, <state | boolean | hash>)
512       
513         // With three argument.
514         gotoState(<state>, <state>, <boolean | hash>)
515         gotoState(<state>, <boolean>, <hash>)
516       
517         // With four argument.
518         gotoState(<state>, <state>, <boolean>, <hash>)
519     
520     where <state> is either a SC.State object or a string and <hash> is a regular JS hash object.
521     
522     @param state {SC.State|String} the state to go to (may not be the final state in the transition process)
523     @param fromCurrentState {SC.State|String} Optional. The current state to start the transition process from.
524     @param useHistory {Boolean} Optional. Indicates whether to include using history states in the transition process
525     @param context {Hash} Optional. A context object that will be passed to all exited and entered states
526   */
527   gotoState: function(state, fromCurrentState, useHistory, context) {
528     
529     if (!this.get('statechartIsInitialized')) {
530       this.statechartLogError("can not go to state %@. statechart has not yet been initialized".fmt(state));
531       return;
532     }
533     
534     if (this.get('isDestroyed')) {
535       this.statechartLogError("can not go to state %@. statechart is destroyed".fmt(this));
536       return;
537     }
538     
539     var args = this._processGotoStateArgs(arguments);
540 
541     state = args.state;
542     fromCurrentState = args.fromCurrentState;
543     useHistory = args.useHistory;
544     context = args.context;
545     
546     var pivotState = null,
547         exitStates = [],
548         enterStates = [],
549         trace = this.get('allowStatechartTracing'),
550         rootState = this.get('rootState'),
551         paramState = state,
552         paramFromCurrentState = fromCurrentState,
553         msg;
554     
555     state = rootState.getSubstate(state);
556     
557     if (SC.none(state)) {
558       this.statechartLogError("Can not to goto state %@. Not a recognized state in statechart".fmt(paramState));
559       return;
560     }
561     
562     if (this._gotoStateLocked) {
563       // There is a state transition currently happening. Add this requested state
564       // transition to the queue of pending state transitions. The request will
565       // be invoked after the current state transition is finished.
566       this._pendingStateTransitions.push({
567         state: state,
568         fromCurrentState: fromCurrentState,
569         useHistory: useHistory,
570         context: context
571       });
572       
573       return;
574     }
575     
576     // Lock the current state transition so that no other requested state transition 
577     // interferes. 
578     this._gotoStateLocked = YES;
579     
580     if (!SC.none(fromCurrentState)) {
581       // Check to make sure the current state given is actually a current state of this statechart
582       fromCurrentState = rootState.getSubstate(fromCurrentState);
583       if (SC.none(fromCurrentState) || !fromCurrentState.get('isCurrentState')) {
584         msg = "Can not to goto state %@. %@ is not a recognized current state in statechart";
585         this.statechartLogError(msg.fmt(paramState, paramFromCurrentState));
586         this._gotoStateLocked = NO;
587         return;
588       }
589     } 
590     else if (this.getPath('currentStates.length') > 0) {
591       // No explicit current state to start from; therefore, just use the first current state as 
592       // a default, if there is a current state.
593       fromCurrentState = this.get('currentStates')[0];
594       msg = "gotoState: fromCurrentState not explicitly provided. Using a default current state to transition from: %@";
595       this.statechartLogWarning(msg.fmt(fromCurrentState));
596     }
597         
598     if (trace) {
599       this.statechartLogTrace("BEGIN gotoState: %@".fmt(state));
600       msg = "starting from current state: %@";
601       msg = msg.fmt(fromCurrentState ? fromCurrentState : '---');
602       this.statechartLogTrace(msg);
603       msg = "current states before: %@";
604       msg = msg.fmt(this.getPath('currentStates.length') > 0 ? this.get('currentStates') : '---');
605       this.statechartLogTrace(msg);
606     }
607 
608     // If there is a current state to start the transition process from, then determine what
609     // states are to be exited
610     if (!SC.none(fromCurrentState)) {
611       exitStates = this._createStateChain(fromCurrentState);
612     }
613     
614     // Now determine the initial states to be entered
615     enterStates = this._createStateChain(state);
616     
617     // Get the pivot state to indicate when to go from exiting states to entering states
618     pivotState = this._findPivotState(exitStates, enterStates);
619 
620     if (pivotState) {
621       if (trace) this.statechartLogTrace("pivot state = %@".fmt(pivotState));
622       if (pivotState.get('substatesAreConcurrent')) {
623         this.statechartLogError("Can not go to state %@ from %@. Pivot state %@ has concurrent substates.".fmt(state, fromCurrentState, pivotState));
624         this._gotoStateLocked = NO;
625         return;
626       }
627     }
628     
629     // Collect what actions to perform for the state transition process
630     var gotoStateActions = [];
631     
632     // Go ahead and find states that are to be exited
633     this._traverseStatesToExit(exitStates.shift(), exitStates, pivotState, gotoStateActions);
634     
635     // Now go find states that are to entered
636     if (pivotState !== state) {
637       this._traverseStatesToEnter(enterStates.pop(), enterStates, pivotState, useHistory, gotoStateActions);
638     } else {
639       this._traverseStatesToExit(pivotState, [], null, gotoStateActions);
640       this._traverseStatesToEnter(pivotState, null, null, useHistory, gotoStateActions);
641     }
642     
643     // Collected all the state transition actions to be performed. Now execute them.
644     this._executeGotoStateActions(state, gotoStateActions, null, context);
645   },

662   /**
663     Resumes an active goto state transition process that has been suspended.
664   */
665   resumeGotoState: function() {
666     if (!this.get('gotoStateSuspended')) {
667       this.statechartLogError("Can not resume goto state since it has not been suspended");
668       return;
669     }
670     
671     var point = this._gotoStateSuspendedPoint;
672     this._executeGotoStateActions(point.gotoState, point.actions, point.marker, point.context);
673   },

675   /** @private */
676   _executeGotoStateActions: function(gotoState, actions, marker, context) {
677     var action = null,
678         len = actions.length,
679         actionResult = null;
680       
681     marker = SC.none(marker) ? 0 : marker;
682     
683     for (; marker < len; marker += 1) {
684       action = actions[marker];
685       switch (action.action) {
686         case SC.EXIT_STATE:
687           actionResult = this._exitState(action.state, context);
688           break;
689           
690         case SC.ENTER_STATE:
691           actionResult = this._enterState(action.state, action.currentState, context);
692           break;
693       }
694       
695       //
696       // Check if the state wants to perform an asynchronous action during
697       // the state transition process. If so, then we need to first
698       // suspend the state transition process and then invoke the 
699       // asynchronous action. Once called, it is then up to the state or something 
700       // else to resume this statechart's state transition process by calling the
701       // statechart's resumeGotoState method.
702       //
703       if (SC.kindOf(actionResult, SC.Async)) {
704         this._gotoStateSuspendedPoint = {
705           gotoState: gotoState,
706           actions: actions,
707           marker: marker + 1,
708           context: context
709         }; 
710         
711         actionResult.tryToPerform(action.state);
712         return;
713       }
714     }
715     
716     this.beginPropertyChanges();
717     this.notifyPropertyChange('currentStates');
718     this.notifyPropertyChange('enteredStates');
719     this.endPropertyChanges();
720     
721     if (this.get('allowStatechartTracing')) {
722       this.statechartLogTrace("current states after: %@".fmt(this.get('currentStates')));

825   /**
826     When called, the statechart will proceed to make transitions to the given state then follow that
827     state's history state. 
828     
829     You can either go to a given state's history recursively or non-recursively. To go to a state's history
830     recursively means to following each history state's history state until no more history states can be
831     followed. Non-recursively means to just to the given state's history state but do not recusively follow
832     history states. If the given state does not have a history state, then the statechart will just follow

834     
835     Because a statechart can have one or more current states, depending on if the statechart has any concurrent
836     states, it is optional to provided current state in which to start the state transition process from. If no
837     current state is provided, then the statechart will default to the first current state that it has; which, 
838     depending on the make up of that statechart, can lead to unexpected outcomes. For a statechart with concurrent
839     states, it is best to explicitly supply a current state.
840     
841     Method can be called in the following ways:
842     
843         // With one arguments. 
844         gotoHistorytate(<state>)
845       
846         // With two arguments. 
847         gotoHistorytate(<state>, <state | boolean | hash>)
848       
849         // With three arguments.
850         gotoHistorytate(<state>, <state>, <boolean | hash>)
851         gotoHistorytate(<state>, <boolean>, <hash>)
852       
853         // With four argumetns
854         gotoHistorytate(<state>, <state>, <boolean>, <hash>)
855     
856     where <state> is either a SC.State object or a string and <hash> is a regular JS hash object.
857     
858     @param state {SC.State|String} the state to go to and follow it's history state
859     @param fromCurrentState {SC.State|String} Optional. the current state to start the state transition process from
860     @param recursive {Boolean} Optional. whether to follow history states recursively.

864       this.statechartLogError("can not go to state %@'s history state. Statechart has not yet been initialized".fmt(state));
865       return;
866     }
867     
868     var args = this._processGotoStateArgs(arguments);
869     
870     state = args.state;
871     fromCurrentState = args.fromCurrentState;
872     recursive = args.useHistory;
873     context = args.context;
874     
875     state = this.getState(state);
876   
877     if (!state) {
878       this.statechartLogError("Can not to goto state %@'s history state. Not a recognized state in statechart".fmt(state));
879       return;
880     }
881     
882     var historyState = state.get('historyState');
883     
884     if (!recursive) { 
885       if (historyState) {
886         this.gotoState(historyState, fromCurrentState, context);
887       } else {

889       }
890     } else {

910       return;
911     }

998   /** @private
999     
1000     Recursively follow states that are to be exited during a state transition process. The exit
1001     process is to start from the given state and work its way up to when either all exit
1002     states have been reached based on a given exit path or when a stop state has been reached.
1003     
1004     @param state {State} the state to be exited
1005     @param exitStatePath {Array} an array representing a path of states that are to be exited
1006     @param stopState {State} an explicit state in which to stop the exiting process
1007   */
1008   _traverseStatesToExit: function(state, exitStatePath, stopState, gotoStateActions) {    
1009     if (!state || state === stopState) return;
1010     
1011     var trace = this.get('allowStatechartTracing');
1012     
1013 1014     // This state has concurrent substates. Therefore we have to make sure we
1015     // exit them up to this state before we can go any further up the exit chain.
1016     if (state.get('substatesAreConcurrent')) {
1017       var i = 0,
1018           currentSubstates = state.get('currentSubstates'),
1019           len = currentSubstates.length,
1020           currentState = null;
1021       
1022       for (; i < len; i += 1) {
1023         currentState = currentSubstates[i];
1024         if (currentState._traverseStatesToExit_skipState === YES) continue;
1025         var chain = this._createStateChain(currentState);
1026         this._traverseStatesToExit(chain.shift(), chain, state, gotoStateActions);
1027       }
1028     }
1029     
1030     gotoStateActions.push({ action: SC.EXIT_STATE, state: state });
1031     if (state.get('isCurrentState')) state._traverseStatesToExit_skipState = YES;
1032     this._traverseStatesToExit(exitStatePath.shift(), exitStatePath, stopState, gotoStateActions);
1033   },

1035   /** @private
1036   
1037     Recursively follow states that are to be entred during the state transition process. The
1038     enter process is to start from the given state and work its way down a given enter path. When
1039     the end of enter path has been reached, then continue entering states based on whether 
1040     an initial substate is defined, there are concurrent substates or history states are to be
1041     followed; when none of those condition are met then the enter process is done.
1042     
1043     @param state {State} the sate to be entered
1044     @param enterStatePath {Array} an array representing an initial path of states that are to be entered
1045     @param pivotState {State} The state pivoting when to go from exiting states to entering states
1046     @param useHistory {Boolean} indicates whether to recursively follow history states 
1047   */
1048   _traverseStatesToEnter: function(state, enterStatePath, pivotState, useHistory, gotoStateActions) {
1049     if (!state) return;
1050     
1051     var trace = this.get('allowStatechartTracing');
1052     
1053     // We do not want to enter states in the enter path until the pivot state has been reached. After
1054     // the pivot state has been reached, then we can go ahead and actually enter states.
1055     if (pivotState) {
1056       if (state !== pivotState) {
1057         this._traverseStatesToEnter(enterStatePath.pop(), enterStatePath, pivotState, useHistory, gotoStateActions);
1058       } else {
1059         this._traverseStatesToEnter(enterStatePath.pop(), enterStatePath, null, useHistory, gotoStateActions);
1060       }
1061     }
1062     
1063     // If no more explicit enter path instructions, then default to enter states based on 
1064     // other criteria
1065     else if (!enterStatePath || enterStatePath.length === 0) {
1066       var gotoStateAction = { action: SC.ENTER_STATE, state: state, currentState: NO };
1067       gotoStateActions.push(gotoStateAction);
1068       
1069       var initialSubstate = state.get('initialSubstate'),
1070           historyState = state.get('historyState');
1071       
1072       // State has concurrent substates. Need to enter all of the substates
1073       if (state.get('substatesAreConcurrent')) {
1074         this._traverseConcurrentStatesToEnter(state.get('substates'), null, useHistory, gotoStateActions);
1075       }
1076       
1077       // State has substates and we are instructed to recursively follow the state's
1078       // history state if it has one.
1079       else if (state.get('hasSubstates') && historyState && useHistory) {
1080         this._traverseStatesToEnter(historyState, null, null, useHistory, gotoStateActions);
1081       }
1082       
1083       // State has an initial substate to enter
1084       else if (initialSubstate) {
1085         if (SC.kindOf(initialSubstate, SC.HistoryState)) {
1086           if (!useHistory) useHistory = initialSubstate.get('isRecursive');
1087           initialSubstate = initialSubstate.get('state');
1088         }
1089         this._traverseStatesToEnter(initialSubstate, null, null, useHistory, gotoStateActions);  
1090       } 
1091       
1092       // Looks like we hit the end of the road. Therefore the state has now become
1093       // a current state of the statechart.
1094       else {
1095         gotoStateAction.currentState = YES;
1096       }
1097     }
1098     
1099     // Still have an explicit enter path to follow, so keep moving through the path.
1100     else if (enterStatePath.length > 0) {
1101       gotoStateActions.push({ action: SC.ENTER_STATE, state: state });
1102       var nextState = enterStatePath.pop();
1103       this._traverseStatesToEnter(nextState, enterStatePath, null, useHistory, gotoStateActions); 
1104       
1105       // We hit a state that has concurrent substates. Must go through each of the substates
1106       // and enter them
1107       if (state.get('substatesAreConcurrent')) {
1108         this._traverseConcurrentStatesToEnter(state.get('substates'), nextState, useHistory, gotoStateActions);
1109       }
1110     }
1111   },

1156   /**
1157     Used to invoke a method on current states. If the method can not be executed
1158     on a current state, then the state's parent states will be tried in order
1159     of closest ancestry.
1160     
1161     A few notes: 
1162     
1163      1. Calling this is not the same as calling sendEvent or sendAction.
1164         Rather, this should be seen as calling normal methods on a state that 
1165         will *not* call gotoState or gotoHistoryState.
1166      2. A state will only ever be invoked once per call. So if there are two 
1167         or more current states that have the same parent state, then that parent 
1168         state will only be invoked once if none of the current states are able
1169         to invoke the given method.
1170     
1171     When calling this method, you are able to supply zero ore more arguments
1172     that can be pass onto the method called on the states. As an example
1173     
1174         invokeStateMethod('render', context, firstTime);
1175     
1176     The above call will invoke the render method on the current states
1177     and supply the context and firstTime arguments to the method. 
1178     
1179     Because a statechart can have more than one current state and the method 
1180     invoked may return a value, the addition of a callback function may be provided 
1181     in order to handle the returned value for each state. As an example, let's say
1182     we want to call a calculate method on the current states where the method
1183     will return a value when invoked. We can handle the returned values like so:
1184     
1185         invokeStateMethod('calculate', value, function(state, result) {
1186           // .. handle the result returned from calculate that was invoked
1187           //    on the given state
1188         })
1189     
1190     If the method invoked does not return a value and a callback function is
1191     supplied, then result value will simply be undefined. In all cases, if
1192     a callback function is given, it must be the last value supplied to this
1193     method.
1194     
1195     invokeStateMethod will return a value if only one state was able to have 
1196     the given method invoked on it, otherwise no value is returned. 
1197     
1198     @param methodName {String} methodName a method name
1199     @param args {Object...} Optional. any additional arguments
1200     @param func {Function} Optional. a callback function. Must be the last
1201            value supplied if provided.
1202            
1203     @returns a value if the number of current states is one, otherwise undefined
1204              is returned. The value is the result of the method that got invoked
1205              on a state.
1206 1207   */
1208   invokeStateMethod: function(methodName, args, func) {
1209     if (methodName === 'unknownEvent') {
1210       this.statechartLogError("can not invoke method unkownEvent");
1211       return;
1212     }
1213     
1214     args = SC.A(arguments); args.shift();
1215     
1216     var len = args.length, 
1217         arg = len > 0 ? args[len - 1] : null,
1218         callback = SC.typeOf(arg) === SC.T_FUNCTION ? args.pop() : null,
1219         currentStates = this.get('currentStates'), 
1220         i = 0, state = null, checkedStates = {},
1221         method, result = undefined, calledStates = 0;
1222         
1223     len = currentStates.get('length');
1224 1225     
1226     for (; i < len; i += 1) {
1227       state = currentStates.objectAt(i);
1228       while (state) {
1229         if (checkedStates[state.get('fullPath')]) break;
1230         checkedStates[state.get('fullPath')] = YES;
1231         method = state[methodName];
1232         if (SC.typeOf(method) === SC.T_FUNCTION && !method.isEventHandler) {
1233           result = method.apply(state, args);
1234           if (callback) callback.call(this, state, result);
1235           calledStates += 1;  
1236           break;
1237         }
1238         state = state.get('parentState');
1239       }
1240     }
1241     
1242     return calledStates === 1 ? result : undefined;
1243   },

1367   _constructRootStateClass: function() {
1368     var rsExampleKey = 'rootStateExample',
1369         rsExample = this.get(rsExampleKey),
1370         initialState = this.get('initialState'),
1371         statesAreConcurrent = this.get('statesAreConcurrent'),
1372         stateCount = 0,
1373         key, value, valueIsFunc, attrs = {};
1374     
1375     if (SC.typeOf(rsExample) === SC.T_FUNCTION && rsExample.statePlugin) {
1376       rsExample = rsExample.apply(this);
1377     }
1378 
1379     if (!(SC.kindOf(rsExample, SC.State) && rsExample.isClass)) {
1380       this._logStatechartCreationError("Invalid root state example");
1381       return null;
1382     }
1383     
1384     if (statesAreConcurrent && !SC.empty(initialState)) {
1385       this._logStatechartCreationError("Can not assign an initial state when states are concurrent");
1386     } else if (statesAreConcurrent) {
1387       attrs.substatesAreConcurrent = YES;
1388     } else if (SC.typeOf(initialState) === SC.T_STRING) {
1389       attrs.initialSubstate = initialState;
1390     } else {
1391 1392       this._logStatechartCreationError("Must either define initial state or assign states as concurrent");
1393       return null;
1394     }
1395     
1396     for (key in this) {
1397       if (key === rsExampleKey) continue;
1398       
1399       value = this[key];
1400       valueIsFunc = SC.typeOf(value) === SC.T_FUNCTION;
1401       
1402       if (valueIsFunc && value.statePlugin) {
1403         value = value.apply(this);
1404       }
1405       
1406       if (SC.kindOf(value, SC.State) && value.isClass && this[key] !== this.constructor) {
1407         attrs[key] = value;
1408         stateCount += 1;
1409       }
1410     }
1411     
1412     if (stateCount === 0) {
1413       this._logStatechartCreationError("Must define one or more states");
1414       return null;
1415     }
1416     
1417     return rsExample.extend(attrs);
1418   },
