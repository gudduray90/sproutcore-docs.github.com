 28 /** @class
 29   A Pane is like a regular view except that it does not need to live within a
 30   parent view.  You usually use a Pane to form the root of a view hierarchy in
 31   your application, such as your main application view or for floating
 32   palettes, popups, menus, etc.
 33 
 34   Usually you will not work directly with the SC.Pane class, but with one of
 35   its subclasses such as SC.MainPane, SC.Panel, or SC.PopupPane.
 36 
 37   ## Showing a Pane
 38 
 39   To make a pane visible, you need to add it to your HTML document.  The
 40   simplest way to do this is to call the append() method:
 41 
 42       myPane = SC.Pane.create();
 43       myPane.append(); // adds the pane to the document
 44 
 45   This will insert your pane into the end of your HTML document body, causing
 46   it to display on screen.  It will also register your pane with the
 47   SC.RootResponder for the document so you can start to receive keyboard,
 48   mouse, and touch events.
 49 
 50   If you need more specific control for where you pane appears in the
 51   document, you can use several other insertion methods such as appendTo(),
 52   prependTo(), before() and after().  These methods all take a an element to
 53   indicate where in your HTML document you would like you pane to be inserted.
 54 
 55   Once a pane is inserted into the document, it will be sized and positioned
 56   according to the layout you have specified.  It will then automatically
 57   resize with the window if needed, relaying resize notifications to children
 58   as well.
 59 
 60   ## Hiding a Pane
 61 
 62   When you are finished with a pane, you can hide the pane by calling the
 63   remove() method.  This method will actually remove the Pane from the
 64   document body, as well as deregistering it from the RootResponder so that it
 65   no longer receives events.
 66 
 67   The isVisibleInWindow method will also change to NO for the Pane and all of
 68   its childViews and the views will no longer have their updateDisplay methods
 69   called.
 70 
 71   You can readd a pane to the document again any time in the future by using
 72   any of the insertion methods defined in the previous section.
 73 
 74   ## Receiving Events
 75 
 76   Your pane and its child views will automatically receive any mouse or touch
 77   events as long as it is on the screen.  To receive keyboard events, however,
 78   you must focus the keyboard on your pane by calling makeKeyPane() on the
 79   pane itself.  This will cause the RootResponder to route keyboard events to
 80   your pane.  The pane, in turn, will route those events to its current
 81   keyView, if there is any.
 82 
 83   Note that all SC.Views (anything that implements SC.ClassicResponder,
 84   really) will be notified when it is about or gain or lose keyboard focus.
 85   These notifications are sent both when the view is made keyView of a
 86   particular pane and when the pane is made keyPane for the entire
 87   application.
 88 
 89   You can prevent your Pane from becoming key by setting the acceptsKeyPane
 90   to NO on the pane.  This is useful when creating palettes and other popups
 91   that should not steal keyboard control from another view.
 92 
 93   @extends SC.View
 94   @extends SC.ResponderContext
 95   @since SproutCore 1.0
 96 */

116   /**
117     The rootResponder for this pane.  Whenever you add a pane to a document,
118     this property will be set to the rootResponder that is now forwarding
119     events to the pane.
120 
121     @property {SC.Responder}

125   /**
126     Attempts to send the event down the responder chain for this pane.  If you
127     pass a target, this method will begin with the target and work up the
128     responder chain.  Otherwise, it will begin with the current rr
129     and walk up the chain looking for any responder that implements a handler
130     for the passed method and returns YES when executed.
131 
132     @param {String} action
133     @param {SC.Event} evt
134     @param {Object} target
135     @returns {Object} object that handled the event
136   */
137   sendEvent: function (action, evt, target) {
138     var handler;
139 
140     // walk up the responder chain looking for a method to handle the event
141     if (!target) target = this.get('firstResponder');
142     while (target) {
143       if (action === 'touchStart') {
144         // first, we must check that the target is not already touch responder
145         // if it is, we don't want to have "found" it; that kind of recursion is sure to
146         // cause really severe, and even worse, really odd bugs.
147         if (evt.touchResponder === target) {
148           target = null;
149           break;
150         }
151 
152         // now, only pass along if the target does not already have any touches, or is
153         // capable of accepting multitouch.
154         if (!target.get("hasTouch") || target.get("acceptsMultitouch")) {
155           if (target.tryToPerform("touchStart", evt)) break;
156         }
157       } else if (action === 'touchEnd' && !target.get("acceptsMultitouch")) {
158         if (!target.get("hasTouch")) {
159           if (target.tryToPerform("touchEnd", evt)) break;
160         }
161       } else {
162         if (target.tryToPerform(action, evt)) break;
163       }
164 
165       // even if someone tries to fill in the nextResponder on the pane, stop
166       // searching when we hit the pane.
167       target = (target === this) ? null : target.get('nextResponder');
168     }
169 
170     // if no handler was found in the responder chain, try the default
171     if (!target && (target = this.get('defaultResponder'))) {
172       if (typeof target === SC.T_STRING) {
173         target = SC.objectForPropertyPath(target);
174       }
175 
176       if (!target) target = null;
177       else target = target.tryToPerform(action, evt) ? target : null;
178     }
179 
180     // if we don't have a default responder or no responders in the responder
181     // chain handled the event, see if the pane itself implements the event
182     else if (!target && !(target = this.get('defaultResponder'))) {
183       target = this.tryToPerform(action, evt) ? this : null;
184     }
185 
186     return evt.mouseHandler || target;
187   },

254   /**
255     Makes the passed view (or any object that implements SC.Responder) into
256     the new firstResponder for this pane.  This will cause the current first
257     responder to lose its responder status and possibly keyResponder status as
258     well.
259 
260     @param {SC.View} view
261     @param {Event} evt that cause this to become first responder
262     @returns {SC.Pane} receiver
263   */
264   makeFirstResponder: function (original, view, evt) {
265     // firstResponder should never be null
266     if (!view) view = this;
267 
268     var current = this.get('firstResponder'),
269       isKeyPane = this.get('isKeyPane');
270 
271     if (current === view) return this; // nothing to do
272     if (SC.platform.touch && view && view.kindOf(SC.TextFieldView) && !view.get('focused')) return this;
273 
274     // if we are currently key pane, then notify key views of change also
275     if (isKeyPane) {
276       if (current) { current.tryToPerform('willLoseKeyResponderTo', view); }
277       if (view) {
278         view.tryToPerform('willBecomeKeyResponderFrom', current);
279       }
280     }
281 
282     if (current) {
283       current.beginPropertyChanges();
284       current.set('isKeyResponder', NO);
285     }
286 
287     if (view) {
288       view.beginPropertyChanges();
289       view.set('isKeyResponder', isKeyPane);
290     }
291 
292     original(view, evt);
293 
294     if (current) current.endPropertyChanges();
295     if (view) view.endPropertyChanges();
296 
297     // and notify again if needed.
298     if (isKeyPane) {
299       if (view) {
300         view.tryToPerform('didBecomeKeyResponderFrom', current);
301       }
302       if (current) {
303         current.tryToPerform('didLoseKeyResponderTo', view);
304       }
305     }
306 
307     return this;
308   }.enhance(),

549   /** @private */
550   _addIntercept: function () {
551     if (this.get('hasTouchIntercept')) {
552       var div = document.createElement("div");
553       var divStyle = div.style;
554       divStyle.position = "absolute";
555       divStyle.left = "0px";
556       divStyle.top = "0px";
557       divStyle.right = "0px";
558       divStyle.bottom = "0px";
559       divStyle.webkitTransform = "translateZ(0px)";
560       divStyle.zIndex = this.get("zIndex") + this.get("touchZ");
561       div.className = "touch-intercept";
562       div.id = "touch-intercept-" + SC.guidFor(this);
563       this._touchIntercept = div;
564       document.body.appendChild(div);
565     }
