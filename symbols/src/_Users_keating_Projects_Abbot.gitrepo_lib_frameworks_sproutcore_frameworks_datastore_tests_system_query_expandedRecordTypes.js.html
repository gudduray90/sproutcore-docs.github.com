  8 // test parsing of query string
  9 var store, storeKey, rec1, rec2, rec3, rec4, rec5, MyApp, q;
 10 
 11 module("SC.Query#containsRecordTypes", {
 12   setup: function() {
 13     MyApp = SC.Object.create();
 14     
 15     MyApp.Contact  = SC.Record.extend();
 16     MyApp.Person   = MyApp.Contact.extend(); // person is a type of contact
 17     MyApp.Group    = SC.Record.extend() ; // NOT a subclass
 18     MyApp.Foo      = SC.Record.extend();
 19     
 20   },
 21   
 22   teardown: function() { 
 23     MyApp = null ; 
 24     SC.Record.subclasses = SC.Set.create(); // reset subclasses
 25   }
 26 });
 27 
 28 test("single recordType with no subclasses", function() {
 29   var q = SC.Query.local(MyApp.Foo),
 30       expected = SC.CoreSet.create().add(MyApp.Foo);
 31       
 32   same(q.get('expandedRecordTypes'), expected, 'should have only MyApp.Foo');
 33 });
 34 
 35 test("multiple recordTypes with no subclasses", function() {
 36   var q = SC.Query.local([MyApp.Foo, MyApp.Group]),
 37       expected = SC.CoreSet.create().add(MyApp.Foo).add(MyApp.Group);
 38       
 39   same(q.get('expandedRecordTypes'), expected, 'should have MyApp.Foo, MyApp.Group');
 40 });
 41 
 42 test("base SC.Record", function() {
 43   var q = SC.Query.local(),
 44       expected = SC.CoreSet.create().addEach([SC.Record, MyApp.Foo, MyApp.Group, MyApp.Contact, MyApp.Person]);
 45       
 46   same(q.get('expandedRecordTypes'), expected, 'should have all defined types');
 47 });
 48 
 49 test("type with subclass", function() {
 50   var q = SC.Query.local(MyApp.Contact),
 51       expected = SC.CoreSet.create().addEach([MyApp.Contact, MyApp.Person]);
 52       
 53   same(q.get('expandedRecordTypes'), expected, 'should have all Contact and Person');
 54 });
 55 
 56 test("adding new type should invalidate property", function() {
 57   var q = SC.Query.local(MyApp.Contact),
 58       expected = SC.CoreSet.create().addEach([MyApp.Contact, MyApp.Person]);    
 59   same(q.get('expandedRecordTypes'), expected, 'precond - should have all Contact and Person');
 60 
 61   var Bar = MyApp.Person.extend(); // add a new record
 62   expected.add(Bar);
 63   same(q.get('expandedRecordTypes'), expected, 'should have all Contact, Person, and Bar');
 64 });
