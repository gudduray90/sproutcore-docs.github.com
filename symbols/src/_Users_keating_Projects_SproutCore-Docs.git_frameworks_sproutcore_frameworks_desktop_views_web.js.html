  8 /** @class
  9 
 10   Used to display an iframe. The source, (specified by the value property) of
 11   the iFrame should be from the same domain. (i.e. the src / value should be
 12   from the same domain) if you want to access the contents of the iframe.
 13 
 14   @extends SC.View
 15   @extends SC.Control
 16   @since SproutCore 1.0
 17 */
 18 SC.WebView = SC.View.extend(SC.Control, {/** @scope SC.WebView.prototype */
 19   classNames: 'sc-web-view',
 20 
 21   /**
 22     @type Array
 23     @default ['value', 'shouldAutoResize']
 24     @see SC.View#displayProperties
 25   */
 26   displayProperties: ['value', 'shouldAutoResize'],
 27 
 28   /**
 29     The content of the iframe can be bigger than the size specified when creating
 30     the view. If you want the view to be auto-resized to the dimensions of the
 31     iframe, then set the value of this property to YES.
 32     The web view can be auto resized only if the contents are from the same
 33     domain as the parent domain.
 34 
 35     @type Boolean
 36     @default NO
 37   */
 38   shouldAutoResize: NO,
 39 
 40   /**
 41     @param {SC.RenderContext} context
 42     @param {Boolean} firstTime
 43   */
 44   render: function(context, firstTime) {
 45     var src = this.get('value'), iframe;
 46 
 47     if (firstTime) {
 48       context.push('<iframe src="' + src +
 49       '" style="position: absolute; width: 100%; height: 100%; border: 0px; margin: 0px; padding: 0px;"></iframe>');
 50     }
 51     else if(src!==this._lastSrc) {
 52       iframe = this.$('iframe');
 53       // clear out the previous src, to force a reload
 54       iframe.attr('src', 'javascript:;');
 55       iframe.attr('src', src);
 56     }
 57 
 58     this._lastSrc = src;
 59   },
 60 
 61   /**
 62     Called when the layer gets created.
 63   */
 64   didCreateLayer: function() {
 65     var f = this.$('iframe');
 66     // Attach an onload event to the iframe.
 67     SC.Event.add(f, 'load', this, this.iframeDidLoad);
 68   },
 69 
 70   /**
 71     Called before the layer gets destroyed.
 72   */
 73   willDestroyLayer: function() {
 74     var f = this.$('iframe');
 75     // Remove the onload event so that the iframe can be released
 76     SC.Event.remove(f, 'load', this, this.iframeDidLoad);
 77   },
 78 
 79   /**  @private
 80     Called when iframe onload event is fired.
 81     1. Resizes the view to fit the contents of the iframe using the
 82     scroll width and scroll height of the contents of the iframe
 83 
 84     The iframe contents can be accessed only when the src is from the same
 85     domain as the parent document
 86   */
 87   iframeDidLoad: function() {
 88 
 89     //fit the iframe to size of the contents.
 90     if (this.get('shouldAutoResize') === YES) {
 91       var contentWindow;
 92       var iframeElt = this.$('iframe')[0];
 93       if(iframeElt && iframeElt.contentWindow) {
 94         contentWindow = iframeElt.contentWindow;
 95         if(contentWindow && contentWindow.document && contentWindow.document.documentElement){
 96           var docElement = contentWindow.document.documentElement;
 97           // setting the width before the height gives more accurate results..
 98           // at least for the test iframe content i'm using.
 99           //TODO: try out document flows other than top to bottom.
100           if (!SC.browser.isIE) {
101             this.$().width(docElement.scrollWidth);
102             this.$().height(docElement.scrollHeight);
103           }
104           else {
105             this.$().width(docElement.scrollWidth + 12);
106             this.$().height(docElement.scrollHeight + 5);
107           }
108 109         }
110       }
111     }
112   }
113 
114 });
