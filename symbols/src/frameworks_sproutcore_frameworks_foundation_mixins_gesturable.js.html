  8 /**
  9   @namespace
 10 
 11   You can mix in SC.Gesturable to your views to add some support for recognizing
 12   gestures.
 13 
 14   SproutCore views have built-in touch events. However, sometimes you may want
 15   to recognize gestures like tap, pinch, swipe, etc. This becomes tedious if you
 16   need to do this often, and more so if you need to check for multiple possible
 17   gestures on the same view.
 18 
 19   SC.Gesturable allows you to define a collection of gestures (SC.Gesture objects)
 20   that your view should recognize. When a gesture is recognized, methods will be
 21   called on the view:
 22 
 23     - [gestureName](gesture, args...): called when the gesture has occurred. This is
 24       useful for event-style gestures, where you aren't interested in when it starts or
 25       ends, but just that it has occurred. SC.SwipeGesture triggers this after the
 26       swipe has moved a minimum amount—40px by default.
 27     - [gestureName]Start(gesture, args...): called when the gesture is first recognized.
 28       For instance, a swipe gesture may be recognized after the finger has moved a
 29       minimum distance in a horizontal.
 30     - [gestureName]Changed(gesture, args...): called when some property of the gesture
 31       has changed. For instance, this may be called continuously as the user swipes as
 32       the swipe's distance changes.
 33     - [gestureName]Cancelled(gesture, args...): called when a gesture, for one reason
 34       or another, is no longer recognized. For instance, a horizontal swipe gesture
 35       could cancel if the user moves too far in a vertical direction.
 36     - [gestureName]End(gesture, args...): called when a gesture ends. A swipe would end
 37       when the user lifts their finger.
 38 
 39   Each of these methods is passed the gesture instance, in addition to any arguments
 40   the gesture sends for your convenience. The default swipe gesture sends an SC.Touch
 41   instance, the swipe direction, and the distance the swipe has moved in that direction.
 42 
 43   Using SC.Gesturable
 44   -------------------
 45 
 46   To make your view recognize gestures, mix in Gesturable and add items to the 'gestures'
 47   property:
 48 
 49       SC.View.extend(SC.Gesturable, {
 50         gestures: [SC.PinchGesture, 'mySwipeGesture'],
 51 
 52         // specifying as a string allows you to configure it:
 53         mySwipeGesture: SC.SwipeGesture.extend({
 54           direction: SC.SWIPE_VERTICAL,
 55           startDistance: 3,
 56           swipeDistance: 20
 57         }),
 58 
 59         // handle the swipe action
 60         swipe: function(touch, direction) {
 61           console.error("Swiped! In direction: " + direction);
 62         },
 63 
 64         swipeStart: function(touch, direction, delta) {
 65           console.error("Swipe started in direction: " + direction + "; dist: " + delta);
 66         },
 67 
 68         swipeChanged: function(touch, direction, delta) {
 69           console.error("Swipe continued in direction: " + direction + "; dist: " + delta);
 70         },
 71 
 72         swipeEnd: function(touch, direction, delta) {
 73           console.error("Completed swipe in direction: " + direction + "; dist: " + delta);
 74         }
 75 
 76       })
 77 
 78   @extends SC.ObjectMixinProtocol
 79 */
 80 SC.Gesturable = {
 81 
 82   /**
 83     @type Array
 84     @default ['gestures']
 85     @see SC.Object#concatenatedProperties
 86   */
 87   concatenatedProperties: ['gestures'],
 88 
 89   /**
 90     The gestures that the view will support. This property must be set on the consumer of
 91     `SC.Gesturable` before it is initialized.
 92 
 93     These gestures should be objects that extend the `SC.Gesture` class. You can use SproutCore's
 94     pre-built gestures or create your own. If you create your own, you can use a property name
 95     in the list of gestures to refer to the actual gesture class, similar to how the childViews
 96     array works. For example,
 97 
 98         gestures: [SC.PinchGesture, 'mySwipeGesture'],
 99 
100         // Specifying the Gesture by property name allows you to configure it.
101         mySwipeGesture: SC.SwipeGesture.extend({
102           direction: SC.SWIPE_VERTICAL,
103           startDistance: 3,
104           swipeDistance: 20
105         }),
106 
107     Note that `gestures` is a *concatenated property*, which means that it will not be overwritten
108     by subclasses. So for example, if the base class lists gestures as `[SC.PinchGesture]` and its
109     subclass lists gestures as `[SC.TapGesture]`, the actual gestures supported by the subclass will
110     be `[SC.PinchGesture, SC.TapGesture]`.
111 
112     @type Array
113     @default null
114   */
115   gestures: null,
116 
117   /**
118     When SC.Gesturable initializes, any gestures named on the view are instantiated.
119 
120     @see SC.ObjectMixinProtocol#initMixin
121   */
122   initMixin: function() {
123     //@if(debug)
124     if (SC.none(this.gestures)) {
125       SC.error("Developer Error: When mixing in SC.Gesturable, you must define a list of gestures to use.");
126     }
127     //@endif
128     this.createGestures();
129   },
130 
131   /** @private  Instantiates the gestures. */
132   createGestures: function() {
133     var gestures = this.get("gestures"),
134         len = gestures.length,
135         instantiatedGestures = [],
136         idx;
137 
138     // loop through all gestures
139     for (idx = 0; idx < len; idx++) {
140       var gesture;
141 
142       // get the proper gesture
143       if (SC.typeOf(gestures[idx]) === SC.T_STRING) {
144         gesture = this.get(gestures[idx]);
145       } else {
146         gesture = gestures[idx];
147       }
148 
149       // if it was not found, well, that's an error.
150       if (!gesture) {
151         throw new Error("Developer Error: Could not find gesture named '" + gestures[idx] + "' on view.");
152       }
153 
154       // if it is a class, instantiate (it really ought to be a class...)
155       if (gesture.isClass) {
156         gesture = gesture.create({
157           view: this
158         });
159       }
160 
161       // and set the gesture instance and add it to the array.
162       if (SC.typeOf(gestures[idx]) === SC.T_STRING) this[gestures[idx]] = gesture;
163       instantiatedGestures.push(gesture);
164     }
165 
166     this.set("gestures", instantiatedGestures);
167   },
168 
169   /**
170     Handles touch start by handing it to the gesture recognizing code.
171 
172     If you override touchStart, you will need to call gestureTouchStart to
173     give the gesture system control of the touch. You will continue to get
174     events until if and when a gesture decides to take "possession" of a touch—
175     at this point, you will get a [gestureName]Start event.
176 
177     You do not have to call gestureTouchStart immediately; you can call it
178     at any time. This allows you to avoid passing control until _after_ you
179     have determined your own touchStart, touchesDragged, and touchEnd methods
180     are not going to handle it.
181   */
182   touchStart: function(touch) {
183     this.gestureTouchStart(touch);
184   },
185 
186   /**
187     Tells the gesture recognizing code about touches moving.
188 
189     If you override touchesDragged, you will need to call gestureTouchesDragged
190     (at least for any touches you called gestureTouchStart for in touchStart) to
191     allow the gesture system to update.
192   */
193   touchesDragged: function(evt, touches) {
194     this.gestureTouchesDragged(evt, touches);
195   },
196 
197   /**
198     Tells the gesture recognizing code about a touch ending.
199 
200     If you override touchEnd, you will need to call gestureTouchEnd
201     for any touches you called touchStart for.
202   */
203   touchEnd: function(touch) {
204     this.gestureTouchEnd(touch);
205   },
206 
207   /**
208     Tells the gesture recognizing system about a new touch.
209 
210     This informs all gestures that a new touch, "unassigned" to any gesture,
211     has been located. Later, each gesture has an opportunity to claim the touch.
212 
213     Once they have claimed the touch, further events will go _directly_ to them—
214     this view will cease receiving the touchesDragged and will not receive a touchEnd.
215   */
216   gestureTouchStart: function(touch) {
217     touch.isInteresting = 0;
218 
219     var gestures = this.get("gestures"), idx, len = gestures.length, g;
220     for (idx = 0; idx < len; idx++) {
221       g = gestures[idx];
222       g.unassignedTouchDidStart(touch);
223     }
224   },
225 
226   /**
227     Tells the gesture recognition system that some touches have moved.
228 
229     This informs all gestures that these touches have changed. All such touches
230     are "unassigned" because all "assigned" touches already get sent directly
231     to the gesture.
232   */
233   gestureTouchesDragged: function(evt, touches) {
234     var gestures = this.get("gestures"), idx, len = gestures.length, g;
235     for (idx = 0; idx < len; idx++) {
236       g = gestures[idx];
237       g.unassignedTouchesDidChange(evt, touches);
238     }
239   },
240 
241   /**
242     Tells the gesture recognition system that a touch have ended.
243 
244     This informs all of the gestures that the touch ended. The touch is
245     an unassigned touch as, if it were assigned to a gesture, it would have
246     been sent directly to the gesture, bypassing this view.
247   */
248   gestureTouchEnd: function(touch) {
249     var gestures = this.get("gestures"), idx, len = gestures.length, g;
250     for (idx = 0; idx < len; idx++) {
251       g = gestures[idx];
252       g.unassignedTouchDidEnd(touch);
253     }
254   }
255 };
