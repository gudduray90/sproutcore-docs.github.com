1 // ==========================================================================
  2 // SC Unit Test
  3 // ==========================================================================
  4 /*globals SC */
  5 
  6 var sc, root, stateA, stateB, stateX, stateY;
  7 
  8 module("SC.State: addSubstate method Tests", {
  9   
 10   setup: function() {
 11     
 12     sc = SC.Statechart.create({
 13       
 14       trace: YES,
 15       
 16       initialState: 'a',
 17     
 18       a: SC.State.design(),
 19       
 20       b: SC.State.design({
 21         
 22         substatesAreConcurrent: YES,
 23         
 24         x: SC.State.design(),
 25         
 26         y: SC.State.design()
 27         
 28       })
 29       
 30     });
 31     
 32     sc.initStatechart();
 33     root = sc.get('rootState');
 34     stateA = sc.getState('a');
 35     stateB = sc.getState('b');
 36     stateX = sc.getState('x');
 37     stateY = sc.getState('y');
 38   },
 39   
 40   teardown: function() {
 41     sc = root = stateA = stateB = stateX = stateY = null;
 42   }
 43   
 44 });
 45 
 46 test("add a substate to the statechart's root state", function() {
 47   ok(!root.getSubstate('z'), "root state should not have a state z");
 48   
 49   var state = root.addSubstate('z');
 50   
 51   ok(SC.kindOf(state, SC.State) && state.isObject, "addState should return a state object");
 52   equals(root.getSubstate('z'), state, "root state should have a state z after adding state");
 53   ok(state.get('stateIsInitialized'), "state z should be initialized");
 54   ok(!state.get('isEnteredState'), "state z should not be entered");
 55   ok(!state.get('isCurrentState'), "state z should not be current");
 56   
 57   sc.gotoState('z');
 58   
 59   ok(state.get('isEnteredState'), "state z should be entered after transitioning to it");
 60   ok(state.get('isCurrentState'), "state z should be current after transitioning to it");
 61 });
 62 
 63 test("add a substate to state A", function() {
 64   ok(!stateA.getSubstate('z'), "state A should not have a state z");
 65   ok(!stateA.get('initialSubstate'), "state A should not have an initial substate");
 66 
 67   var state = stateA.addSubstate('z');
 68   
 69   ok(SC.kindOf(state, SC.State) && state.isObject, "addState should return a state object");
 70   equals(stateA.getSubstate('z'), state, "state A should have a state z after adding state");
 71   ok(SC.kindOf(stateA.get('initialSubstate'), SC.EmptyState), "state A's initial substate should be an empty state after adding state");
 72   ok(!state.get('isEnteredState'), "state z should not be entered");
 73   ok(!state.get('isCurrentState'), "state z should not be current");
 74   ok(stateA.get('isCurrentState'), "state A should be current");
 75   ok(stateA.get('isEnteredState'), "state A should be entered");
 76   
 77   console.log('reentering state A');
 78   stateA.set('initialSubstate', state);
 79   stateA.reenter();
 80   
 81   ok(state.get('isEnteredState'), "state z should be entered after reentering state A");
 82   ok(state.get('isCurrentState'), "state z should be current after reentering state A");
 83   ok(stateA.get('isEnteredState'), "state A should be entered after reentering");
 84   ok(!stateA.get('isCurrentState'), "state A should not be current after reentering");
 85 });
 86 
 87 test("add a substate to state B", function() {
 88   ok(!stateB.getSubstate('z'), "state B should not have a state z");
 89   sc.gotoState('b');
 90 
 91   var state = stateB.addSubstate('z');
 92   
 93   ok(SC.kindOf(state, SC.State) && state.isObject, "addState should return a state object");
 94   equals(stateB.getSubstate('z'), state, "state B should have a state z after adding state");
 95   ok(!state.get('isEnteredState'), "state z should not be entered");
 96   ok(!state.get('isCurrentState'), "state z should not be current");
 97   ok(!stateB.get('isCurrentState'), "state B should be current");
 98   ok(!stateB.get('initialSubstate'), "state B's initial substate should not be set");
 99   ok(stateB.get('isEnteredState'), "state B should be entered");
100   equals(stateB.getPath('currentSubstates.length'), 2, "state B should have 2 current substates");
101   
102   stateB.reenter();
103   
104   ok(state.get('isEnteredState'), "state z should be entered after reentering state B");
105   ok(state.get('isCurrentState'), "state z should be current after reentering state B");
106   ok(stateB.get('isEnteredState'), "state B should be entered");
107   equals(stateB.getPath('currentSubstates.length'), 3, "state B should have 3 current substates");
108 });
