 38 test("Verify find() loads all fixture data", function() {
 39 
 40   var result = store.find(Sample.File),
 41       rec, storeKey, dataHash;
 42 
 43   ok(result, 'should return a result');
 44   equals(result.get('length'), Sample.File.FIXTURES.get('length'), 'should return records for each item in FIXTURES');
 45 
 46   // verify storeKeys actually return Records
 47   var idx, len = result.get('length'), expected = [];
 48   for(idx=0;idx<len;idx++) {
 49     rec = result.objectAt(idx);
 50     storeKey = rec ? rec.get('storeKey') : null;
 51     dataHash = storeKey ? store.readDataHash(storeKey) : null;
 52 
 53     ok(!!dataHash, 'storeKey at result[%@] (%@) should return dataHash'.fmt(idx, storeKey));
 54 
 55     expected.push(rec); // save record for later test
 56   }
 57 
 58   // verify multiple calls to find() returns SAME data
 59   result = store.find(Sample.File);
 60 
 61   equals(result.get('length'), expected.length, 'second result should have same length as first');
 62   len = result.get('length');
 63   for(idx=0;idx<len;idx++) {
 64     rec = result.objectAt(idx);
 65     equals(rec, expected[idx], 'record returned at index %@ should be same as previous'.fmt(idx));
 66   }
