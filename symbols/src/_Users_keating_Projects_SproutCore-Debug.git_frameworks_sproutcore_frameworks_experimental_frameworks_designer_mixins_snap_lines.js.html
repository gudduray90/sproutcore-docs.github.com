 10  11 sc_require('views/drawing');
 12 /**
 13   @mixin
 14   @author Mike Ball
 15   
 16   Add this Mixin to any View and it gives you an API to draw snap lines for
 17   all the child views
 18 */
 19 
 20 //the number of pixles that will cause a snap line (factor of 2?)
 21 SC.SNAP_ZONE = 2;
 22 
 23 SC.SNAP_LINE = {
 24   shape: SC.LINE,
 25   start: {x: 0, y: 0},
 26   end: {x: 0, y: 0},
 27   style: {
 28     width: 0.5,
 29     color: '#00c6ff'
 30     //transparency: 0.2
 31   }
 32 };
 33 
 34 
 35 SC.SnapLines = {
 36   
 37   hasSnapLines: YES,
 38   
 39   
 40   
 41   /*
 42     This method will setup the data structure required to draw snap lines
 43     it should be called in dragStarted if using with an `SC.Drag` or on
 44     `mouseDown` if using it with a move
 45   
 46     @param {Array} ignoreViews array of views to not include
 47     sets up the data structure used for the line drawing
 48   */
 49   setupData: function(ignoreViews){
 50     if(!ignoreViews) ignoreViews = [];
 51     this.removeLines(); //can't have any existing lines
 52     this._xPositions = {};
 53     this._yPositions = {}; 
 54     
 55     var xPositions = this._xPositions, yPositions = this._yPositions, children = this.get('childViews'), 
 56         that = this, parentView, frame, minX, midX, maxX, minY, midY, maxY, factor = (SC.SNAP_ZONE*2);
 57     
 58     
 59     // little insert utility
 60     var insert = function(min, mid, max, child, positions){
 61       var origMin = min, origMid = mid, origMax = max;
 62       min = Math.floor(min/factor);
 63       mid = Math.floor(mid/factor);
 64       max = Math.floor(max/factor);
 65       if(positions[min]){
 66         positions[min].push({value: origMin, child: child});
 67       }
 68       else{
 69         positions[min] = [{value: origMin, child: child}];
 70       }
 71       
 72       if(positions[mid]){
 73         positions[mid].push({value: origMid, child: child});
 74       }
 75       else{
 76         positions[mid] = [{value: origMid, child: child}];
 77       }
 78       
 79       if(positions[max]){
 80         positions[max].push({value: origMax, child: child});
 81       }
 82       else{
 83         positions[max] = [{value: origMax, child: child}];
 84       }
 85     };
 86 
 87     parent = this;    
 88     children.forEach(function(child){
 89       if(ignoreViews.indexOf(child) < 0){
 90         frame = parent ? parent.convertFrameToView(child.get('frame'), null) : child.get('frame');
 91       
 92         minX = frame.x;
 93         midX = SC.midX(frame);
 94         maxX = frame.x + frame.width;
 95         insert(minX, midX, maxX, child, xPositions);
 96       
 97       
 98         minY = frame.y;
 99         midY = SC.midY(frame);
100         maxY = frame.y + frame.height;
101         insert(minY, midY, maxY, child, yPositions);
102       }
103     });
104 
105     //add the parent
106     parent = this.get('parentView');
107     frame = parent ? parent.convertFrameToView(this.get('frame'), null) : this.get('frame');
108     this._globalFrame = frame;
109     minX = frame.x;
110     midX = SC.midX(frame);
111     maxX = frame.x + frame.width;
112     insert(minX, midX, maxX, this, xPositions);
113     
114     
115     minY = frame.y;
116     midY = SC.midY(frame);
117     maxY = frame.y + frame.height;
118     insert(minY, midY, maxY, this, yPositions);
119     
120     
121   },
122   
123   /**
124     This method will check the passed views position with the other child views
125     and draw any lines.  It should be called in `dragUpdated` if using `SC.Drag`
126     or in `mouseMoved` if using a move.  it will also return a hash of the
127     snapped coords in local and global coordinates
128     
129   */
130   drawLines: function(view, eventX, eventY, mouseDownX, mouseDownY){
131     if(!this._drawingView){
132       this._drawingView = this.createChildView(SC.DrawingView.design({
133         shapes: []
134       }));
135       this.appendChild(this._drawingView);
136     }
137     var factor = (SC.SNAP_ZONE*2), shapes = [], xline, yline, frame, parent, rMinX, rMidX, rMaxX,
138         rMinY, rMidY, rMaxY, rMinXMod, rMidXMod, rMaxXMod, rMinYMod, rMidYMod, rMaxYMod, xHit, yHit,
139         moveDirection = this._dragDirection(eventX, eventY, mouseDownX, mouseDownY), xValues, yValues, 
140         that = this, xHitVals, yHitVals, ret;
141     //get the frame and all the relevant points of interest
142     parent = view.get('parentView');
143     frame = parent ? parent.convertFrameToView(view.get('frame'), null) : view.get('frame');
144     rMinX = SC.minX(frame);
145     rMidX = SC.midX(frame);
146     rMaxX = SC.maxX(frame);
147     rMinY = SC.minY(frame);
148     rMidY = SC.midY(frame);
149     rMaxY = SC.maxY(frame);
150     rMinXMod = Math.floor(rMinX/factor);
151     rMidXMod = Math.floor(rMidX/factor);
152     rMaxXMod = Math.floor(rMaxX/factor);
153     rMinYMod = Math.floor(rMinY/factor);
154     rMidYMod = Math.floor(rMidY/factor);
155     rMaxYMod = Math.floor(rMaxY/factor);
156     
157     //array of tuples containing the mod and the value you need to add to the resulting position
158     xValues = moveDirection.UP ? [{mod: rMinXMod, val: 0}, {mod: rMidXMod, val: frame.width/2}, {mod: rMaxXMod, val: frame.width}] : [{mod: rMaxXMod, val: frame.width}, {mod: rMidXMod, val: frame.width/2}, {mod: rMinXMod, val: 0}];
159     //compute the three possible line positions
160     xValues.forEach(function(xVal){
161       if(that._xPositions[xVal.mod]){
162         xHitVals = xVal;
163         xHit = that._xPositions[xVal.mod][0].value - that._globalFrame.x;
164         return;
165       }
166     });
167     if(!SC.none(xHit)){
168       xline = SC.copy(SC.SNAP_LINE);
169       xline.start = {x: xHit, y: 0};
170       xline.end = {x: xHit, y: this._globalFrame.height};
171       shapes.push(xline);
172     }
173     
174     yValues = moveDirection.LEFT ? [{mod: rMinYMod, val: 0}, {mod: rMidYMod, val: frame.height/2}, {mod: rMaxYMod, val: frame.height}] : [{mod: rMaxYMod, val: frame.height}, {mod: rMidYMod, val: frame.height/2}, {mod: rMinYMod, val: 0}];
175     //compute the three possible line positions
176     yValues.forEach(function(yVal){
177       if(that._yPositions[yVal.mod]){
178         yHitVals = yVal;
179         yHit = that._yPositions[yVal.mod][0].value - that._globalFrame.y;
180         return;
181       }
182     });
183     if(!SC.none(yHit)){
184       yline = SC.copy(SC.SNAP_LINE);
185       yline.start = {y: yHit, x: 0};
186       yline.end = {y: yHit, x: this._globalFrame.width};
187       shapes.push(yline);
188     }
189     this._drawingView.set('shapes', shapes);
190     ret = {pageX: xHit + this._globalFrame.x, pageY: yHit + this._globalFrame.y, frameX: xHit, frameY: yHit};
191     if(xHitVals){
192       ret.pageX -= xHitVals.val;
193       ret.frameX -= xHitVals.val;
194     }
195     if(yHitVals){
196       ret.pageY -= yHitVals.val;
197       ret.frameY -= yHitVals.val;
198     }
199     return ret;
200   },
201   
202   /*
203     called to cleanup the lines...
204     This method should be called in `mouseUp` if doing a move and in
205     `dragEnded` if using a `SC.Drag`.
206   */
207   removeLines: function() {
208     this._xPositions = null;
209     this._yPositions = null;
210     this._globalFrame = null;
211     if(this._drawingView) {
212       this.removeChild(this._drawingView);
213       this._drawingView = null;
214     }
215   },
216   
217   /*
218     takes the event x, y and mouseDown x, y and computes a direction
219   */
220   _dragDirection: function(eventX, eventY, mouseDownX, mouseDownY){
221     var deltaX = eventX - mouseDownX, deltaY = eventY - mouseDownY, ret = {};
222     ret.UP = deltaX > 0 ? NO : YES;
223     ret.DOWN = deltaX > 0 ? YES : NO;
224     ret.LEFT = deltaY > 0 ? NO : YES;
225     ret.RIGHT = deltaY > 0 ? YES : NO;
226     return ret;
227   }
228 };
