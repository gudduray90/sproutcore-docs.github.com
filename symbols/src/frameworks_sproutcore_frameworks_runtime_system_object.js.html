 25 SC._detect_base = function _detect_base(func, parent, name) {
 26   return function invoke_superclass_method() {
 27     var base = parent[name], args;
 28 
 29     if (!base) {
 30       throw new Error("No '" + name + "' method was found on the superclass");
 31     }
 32 
 33     // NOTE: It is possible to cache the base, so that the first
 34     // call to sc_super will avoid doing the lookup again. However,
 35     // since the cost of the extra method dispatch is low and is
 36     // only incurred on sc_super, but also creates another possible
 37     // weird edge-case (when a class is enhanced after first used),
 38     // we'll leave it off for now unless profiling demonstrates that
 39     // it's a hotspot.
 40     //if(base && func === base) { func.base = function () {}; }
 41     //else { func.base = base; }
 42 
 43     if (func.isEnhancement) {
 44       args = Array.prototype.slice.call(arguments, 1);
 45     } else {
 46       args = arguments;
 47     }
 48 
 49     return base.apply(this, args);
 50   };
 51 };

 53 /** @private
 54   Augments a base object by copying the properties from the extended hash.
 55   In addition to simply copying properties, this method also performs a
 56   number of optimizations that can make init'ing a new object much faster
 57   including:
 58 
 59   - concatenating concatenatedProperties
 60   - prepping a list of bindings, observers, and dependent keys
 61   - caching local observers so they don't need to be manually constructed.
 62 
 63   @param {Hash} base hash
 64   @param {Hash} extension
 65   @returns {Hash} base hash
 66 */
 67 SC._object_extend = function _object_extend(base, ext, proto) {
 68   if (!ext) { throw new Error("SC.Object.extend expects a non-null value.  Did you forget to 'sc_require' something?  Or were you passing a Protocol to extend() as if it were a mixin?"); }
 69   // set _kvo_cloned for later use
 70   base._kvo_cloned = null;
 71 
 72   // get some common vars
 73   var key, idx, cur,
 74     cprops = base.concatenatedProperties,
 75     K = SC.K,
 76     p1, p2;
 77 
 78   // first, save any concat props.  use old or new array or concat
 79   idx = (cprops) ? cprops.length : 0;
 80   var concats = (idx > 0) ? {} : null;
 81   while (--idx >= 0) {
 82     key = cprops[idx];
 83     p1 = base[key];
 84     p2 = ext[key];
 85 
 86     if (p1) {
 87       if (!(p1 instanceof Array)) p1 = SC.$A(p1);
 88       concats[key] = (p2) ? p1.concat(p2) : p2;
 89     } else {
 90       if (!(p2 instanceof Array)) p2 = SC.$A(p2);
 91       concats[key] = p2;
 92     }
 93   }
 94 
 95   // setup arrays for bindings, observers, and properties.  Normally, just
 96   // save the arrays from the base.  If these need to be changed during
 97   // processing, then they will be cloned first.
 98   var bindings = base._bindings, clonedBindings = NO,
 99       observers = base._observers, clonedObservers = NO,
100       properties = base._properties, clonedProperties = NO,
101       paths, pathLoc, local, value;
102 
103   // outlets are treated a little differently because you can manually
104   // name outlets in the passed in hash. If this is the case, then clone
105   // the array first.
106   var outlets = base.outlets, clonedOutlets = NO;
107   if (ext.outlets) {
108     outlets = (outlets || SC.EMPTY_ARRAY).concat(ext.outlets);
109     clonedOutlets = YES;
110   }
111 
112   // now copy properties, add superclass to func.
113   for (key in ext) {
114 
115     if (key === '_kvo_cloned') continue; // do not copy
116 
117     // avoid copying builtin methods
118     if (!ext.hasOwnProperty(key)) continue;
119 
120     // get the value.  use concats if defined
121     value = (concats.hasOwnProperty(key) ? concats[key] : null) || ext[key];
122 
123     // Possibly add to bindings.
124     if (key.length > 7 && key.slice(-7) === "Binding") {
125       if (!clonedBindings) {
126         bindings = (bindings || SC.EMPTY_ARRAY).slice();
127         clonedBindings = YES;
128       }
129 
130       if (bindings === null) bindings = (base._bindings || SC.EMPTY_ARRAY).slice();
131       //@if(debug)
132       // Add some developer support.
133 
134       // If a property binding is set on a Class and that Class is extended and
135       // the same property binding is set in the extend, two instances of the
136       // same Binding will exist on the object leading to strange behavior.
137       for (var i = bindings.length - 1; i >= 0; i--) {
138         if (bindings[i] === key) {
139           // There is already a binding for this key!
140           SC.warn("Developer Warning: '%@' was defined twice on the same class, likely because it was defined on both the parent and its subclass.  See the initial line of the following trace:".fmt(key));
141           SC.Logger.trace();
142         }
143       }
144       //@endif
145       bindings[bindings.length] = key;
146 
147     // Also add observers, outlets, and properties for functions...
148     } else if (value && (value instanceof Function)) {
149 
150       // add super to funcs.  Be sure not to set the base of a func to
151       // itself to avoid infinite loops.
152       if (!value.superclass && (value !== (cur = base[key]))) {
153         value.superclass = cur || K;
154         value.base = proto ? SC._detect_base(value, proto, key) : cur || K;
155       }
156 
157       // handle regular observers
158       if (value.propertyPaths) {
159         if (!clonedObservers) {
160           observers = (observers || SC.EMPTY_ARRAY).slice();
161           clonedObservers = YES;
162         }
163         observers[observers.length] = key;
164 
165       // handle local properties
166       }
167 
168       paths = value.localPropertyPaths;
169       if (paths) {
170         pathLoc = paths.length;
171         while (--pathLoc >= 0) {
172           local = base._kvo_for(SC.keyFor('_kvo_local', paths[pathLoc]), SC.CoreSet);
173           local.add(key);
174           base._kvo_for('_kvo_observed_keys', SC.CoreSet).add(paths[pathLoc]);
175         }
176 
177       // handle computed properties
178       }
179 
180       if (value.dependentKeys) {
181         if (!clonedProperties) {
182           properties = (properties || SC.EMPTY_ARRAY).slice();
183           clonedProperties = YES;
184         }
185         properties[properties.length] = key;
186 
187       // handle outlets
188       }
189 
190       if (value.autoconfiguredOutlet) {
191         if (!clonedOutlets) {
192           outlets = (outlets || SC.EMPTY_ARRAY).slice();
193           clonedOutlets = YES;
194         }
195         outlets[outlets.length] = key;
196       }
197 
198       if (value.isEnhancement) {
199         value = SC._enhance(base[key] || K, value);
200       }
201     }
202 
203     // copy property
204     base[key] = value;
205   }
206 
207   // Manually set base on toString() because some JS engines (such as IE8) do
208   // not enumerate it
209   if (ext.hasOwnProperty('toString')) {
210     key = 'toString';
211     // get the value.  use concats if defined
212     value = (concats.hasOwnProperty(key) ? concats[key] : null) || ext[key];
213     if (!value.superclass && (value !== (cur = base[key]))) {
214       value.superclass = value.base = cur || K;
215     }
216     // copy property
217     base[key] = value;
218   }
219 
220 
221   // copy bindings, observers, and properties
222   base._bindings = bindings || [];
223   base._observers = observers || [];
224   base._properties = properties || [];
225   base.outlets = outlets || [];
226 
227   return base;
228 };

231 SC._enhance = function (originalFunction, enhancement) {
232   return function () {
233     var args = Array.prototype.slice.call(arguments, 0),
234         self = this;
235 
236     args.unshift(function () { return originalFunction.apply(self, arguments); });
237     return enhancement.apply(this, args);
238   };

241 /** @class
242 
243   Root object for the SproutCore framework.  SC.Object is the root class for
244   most classes defined by SproutCore.  It builds on top of the native object
245   support provided by JavaScript to provide support for class-like
246   inheritance, automatic bindings, properties observers, and more.
247 
248   Most of the classes you define in your application should inherit from
249   SC.Object or one of its subclasses.  If you are writing objects of your
250   own, you should read this documentation to learn some of the details of
251   how SC.Object's behave and how they differ from other frameworks.
252 
253   About SproutCore Classes
254   ===
255 
256   JavaScript is not a class-based language.  Instead it uses a type of
257   inheritance inspired by self called "prototypical" inheritance.
258   ...
259 
260   Using SproutCore objects with other JavaScript object.
261   ===
262 
263   You can create a SproutCore object just like any other object...
264   obj = new SC.Object();
265 
266   @extends SC.Observable
267   @since SproutCore 1.0
268 */
269 SC.Object = function (props) {
270   this.__sc_super__ = SC.Object.prototype;
271   return this._object_init(props);

946 /** @private
947   This is a way of performing brute-force introspection.  This searches
948   through all the top-level properties looking for classes.  When it finds
949   one, it saves the class path name.
950 */
951 function findClassNames() {
952 
953   if (SC._object_foundObjectClassNames) return;
954   SC._object_foundObjectClassNames = true;
955 
956   var seen = [],
957       detectedSC = false;
958   var searchObject = function (root, object, levels) {
959 
960     var path, value, type;
961     levels--;
962 
963     // not the fastest, but safe
964     if (seen.indexOf(object) >= 0) return;
965     seen.push(object);
966 
967     for (var key in object) {
968       if (key == '__scope__') continue;
969       if (key == 'superclass') continue;
970       if (key == '__SC__') key = 'SC';
971       if (!key.match(/^[A-Z0-9]/)) continue;
972       if (key == 'SC') {
973         if (detectedSC) continue;
974         detectedSC = true;
975       }
976 
977       path = (root) ? [root, key].join('.') : key;
978       value = object[key];
979 
980       try {
981         type = SC.typeOf(value);
982       } catch (e) {
983         // Firefox gives security errors when trying to run typeOf on certain objects
984         break;
985       }
986 
987       switch (type) {
988       case SC.T_CLASS:
989         if (!value._object_className) value._object_className = path;
990         if (levels >= 0) searchObject(path, value, levels);
991         break;
992 
993       case SC.T_OBJECT:
994         if (levels >= 0) searchObject(path, value, levels);
995         break;
996 
997       case SC.T_HASH:
998         if (((root) || (path === 'SC')) && (levels >= 0)) searchObject(path, value, levels);
999         break;
1000 
1001       default:
1002         break;
1003       }
1004     }
1005   };
1006 
1007   // Fix for IE 7 and 8 in order to detect the SC global variable. When you create
1008   // a global variable in IE, it is not added to the window object like in other
1009   // browsers. Therefore the searchObject method will not pick it up. So we have to
1010   // update the window object to have a reference to the global variable. And
1011   // doing window['SC'] does not work since the global variable already exists. For
1012   // any object that you create that is used act as a namespace, be sure to create it
1013   // like so:
1014   //
1015   //   window.MyApp = window.MyApp || SC.Object.create({ ... })
1016   //
1017   window.__SC__ = SC;
1018   searchObject(null, window, 2);
1019 }

1046 /** @private
1047   Returns the name of this class.  If the name is not known, triggers
1048   a search.  This can be expensive the first time it is called.
1049 
1050   This method is used to allow classes to determine their own name.
1051 */
1052 SC._object_className = function (obj) {
1053   if (SC.isReady === NO) return ''; // class names are not available until ready
1054   if (!obj._object_className) findClassNames();
1055   if (obj._object_className) return obj._object_className;
1056 
1057   // if no direct classname was found, walk up class chain looking for a
1058   // match.
1059   var ret = obj;
1060   while (ret && !ret._object_className) ret = ret.superclass;
1061   return (ret && ret._object_className) ? ret._object_className : 'Anonymous';
1062 };
