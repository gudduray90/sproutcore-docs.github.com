110   test("changing nowShowing", function() {
111     var view = pane.view('basic');
112     // Set nowShowing to an instantiated object.
113     var viewToAdd = SC.LabelView.create({value: 'View1'});
114     view.set('nowShowing', viewToAdd);
115     equals(view.get('contentView').get('value'), 'View1', 'contentView changes as intended when an instantiated view is passed to nowShowing');
116     
117     // Set nowShowing to an uninstantiated object.
118     viewToAdd = SC.LabelView.design({value: 'View2'});
119     view.set('nowShowing', viewToAdd);
120     equals(view.get('contentView').get('value'), 'View2', 'contentView changes as intended when an uninstantiated view (class) is passed to nowShowing');
121     
122     // Set nowShowing to an SC.CoreView
123     viewToAdd = SC.CoreView.design({value: 'View5'});
124     view.set('nowShowing', viewToAdd);
125     equals(view.get('contentView').get('value'), 'View5', 'contentView instantiates and inserts an SC.CoreView');
126 
127     // Set nowShowing to a non-view object.
128     viewToAdd = SC.Object;
129     view.set('nowShowing', viewToAdd);
130     equals(view.get('contentView'), null, 'contentView changes to null when nowShowing is set to a non-view');
131     
132     // Set nowShowing to a string.
133     var viewForString = SC.LabelView.create({value: 'View3'});
134     view.set('label', viewForString);
135     view.set('nowShowing', 'label');
136     equals(view.get('contentView').get('value'), 'View3', 'contentView changes as intended when an instantiated view is passed to nowShowing');
137     
138     // Set nowShowing to a nonexistent string.
139     viewToAdd = 'NonexistentNamespace.NonexistentViewClass';
140     view.set('nowShowing', viewToAdd);
141     equals(view.get('contentView'), null, 'contentView changes to null when nowShowing is set to a string pointing at nothing');
142     
143     // Set nowShowing to null.
144     viewToAdd = null;
145     view.set('nowShowing', viewToAdd);
146     equals(view.get('contentView'), null, 'contentView changes to null when nowShowing is set to null');
147     
148   });
