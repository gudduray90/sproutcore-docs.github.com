  9 var pane, a, aa ;
 10 module("SC.View#clippingFrame", {
 11   setup: function() {
 12     htmlbody('<style> .sc-view { border: 1px blue solid; position: absolute;  overflow: hidden; }</style>');
 13     SC.RunLoop.begin();
 14     pane = SC.Pane.design()
 15       .layout({ top: 0, left: 0, width: 200, height: 200 })
 16       .childView(SC.View.design()
 17         .layout({ top: 50, left: 50, width: 100, height: 100 })
 18         .childView(SC.View.design()
 19           .layout({ top: 20, left: 20, width: 40, height: 40 })))
 20       .create();
 21     pane.append();
 22     a = pane.childViews[0];
 23     aa = a.childViews[0];
 24   },
 25 
 26   teardown: function() {
 27     pane.remove();
 28     pane.destroy();
 29     pane = a = aa = null ;
 30     SC.RunLoop.end();
 31     clearHtmlbody();
 32   }
 33 });

 35 test("clippingFrame === frame if view not occluded or scaled.", function() {
 36   var result = pane.get('clippingFrame'), expected = pane.get('frame');
 37   expected.x = expected.y = 0 ;
 38   ok(SC.rectsEqual(result, expected), 'pane');
 39 
 40   result = a.get('clippingFrame'); expected = a.get('frame');
 41   expected.x = expected.y = 0 ;
 42   ok(SC.rectsEqual(result, expected), 'child');
 43 
 44   result = aa.get('clippingFrame'); expected = aa.get('frame');
 45   expected.x = expected.y = 0 ;
 46   ok(SC.rectsEqual(result, expected), 'nested child');
 47 });

101 test("clippingFrame accounts for scale when not occluded.", function() {
102   var frame, clippingFrame, expectedFrame, expectedClippingFrame;
103 
104   a.adjust('scale', 2);
105 
106   frame = a.get('frame');
107   clippingFrame = a.get('clippingFrame');
108   expectedFrame = { x: 0, y: 0, height: 200, width: 200 };
109   expectedClippingFrame = { x: 0, y: 0, height: 100, width: 100 };
110 
111   ok(SC.rectsEqual(frame, expectedFrame), "PRELIM: childView's frame is scaled to the parentView's scale.");
112 
113   ok(SC.rectsEqual(clippingFrame, expectedClippingFrame), "childView's clippingFrame is scaled correctly into its own scale.");
114 });
115 
116 test("clippingFrame correctly scales clipped portion when scaled and occluded.", function() {
117   var frame, clippingFrame, expectedFrame, expectedClippingFrame;
118 
119   a.adjust({ scale: 2, top: -50 });
120 
121   frame = a.get('frame');
122   clippingFrame = a.get('clippingFrame');
123   expectedFrame = { x: 0, y: -100, height: 200, width: 200 };
124   expectedClippingFrame = { x: 0, y: 50, height: 50, width: 100 };
125 
126   ok(SC.rectsEqual(frame, expectedFrame), "PRELIM: childView's frame is scaled to the parentView's scale.");
127 
128   ok(SC.rectsEqual(clippingFrame, expectedClippingFrame), "childView's clippingFrame is scaled correctly into its own scale and correctly reflects the occluded portion in that scale.");
129 });

131 test("notifies receiver and each child if parent clipping frame changes", function() {
132   var callCount = 0;
133 
134   // setup observers
135   function observer() { callCount++; }
136   a.addObserver('clippingFrame', observer);
137   aa.addObserver('clippingFrame', observer);
138 
139   // now, adjust layout of child so that clipping frame will change...
140   a.adjust('top', -50);
141 
142   // IMPORTANT:  If this test fails because the callCount is > 2 it means that
143   // when you set the layout, the frame is getting invalidated more than once.
144   // This should not happen.  If this is the case, fix the view code so that
145   // it does not invalidate frame more than once before you change this
146   // number.
147   equals(callCount, 2, 'should invoke observer on child and nested child');
148 });
