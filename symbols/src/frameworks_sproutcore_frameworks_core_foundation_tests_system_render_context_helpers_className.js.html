 10 var context = null;
 11 
 12 // ..........................................................
 13 // classes()
 14 //
 15 module("SC.RenderContext#classes", {
 16   setup: function() {
 17     context = SC.RenderContext() ;
 18   }
 19 });
 20 
 21 test("returns empty array if no current class names", function() {
 22   same(context.classes(), [], 'classes') ;
 23 });
 24 
 25 test("addClass(array) updates class names", function() {
 26   var cl = 'bar baz'.w();
 27   equals(context.addClass(cl), context, "returns receiver");
 28   same(context.classes(), cl, 'class names');
 29 });
 30 
 31 test("returns classes if set", function() {
 32   context.addClass('bar');
 33   same(context.classes(), ['bar'], 'classNames');
 34 });
 35 
 36 test("clone on retrieval if addClass(array) set", function() {
 37   var cl = 'foo bar'.w();
 38   context.addClass(cl);
 39 
 40   var result = context.classes();
 41   ok(result !== cl, "class name is NOT same instance");
 42   same(result, cl, "but arrays are equivalent");
 43 
 44   equals(result, context.classes(), "2nd retrieval is same instance");
 45 });
 46 
 47 test("extracts class names from element on first retrieval", function() {
 48   var elem = document.createElement('div');
 49   SC.$(elem).attr('class', 'foo bar');
 50   context = SC.RenderContext(elem);
 51 
 52   var result = context.classes();
 53   same(result, ['foo', 'bar'], 'extracted class names');
 54 });

 59 module("SC.RenderContext#hasClass", {
 60   setup: function() {
 61     context = SC.RenderContext().addClass('foo bar'.w()) ;
 62   }
 63 });
 64 
 65 test("should return true if context classNames has class name", function() {
 66   equals(YES, context.hasClass('foo'), 'should have foo');
 67 });
 68 
 69 test("should return false if context classNames does not have class name", function() {
 70   equals(NO, context.hasClass('imaginary'), "should not have imaginary");
 71 });
 72 
 73 test("should return false if context has no classNames", function() {
 74   context = context.begin('div');
 75   ok(context.classes().length === 0, 'precondition - context has no classNames');
 76   equals(NO, context.hasClass('foo'), 'should not have foo');
 77 });

 82 module("SC.RenderContext#addClass", {
 83   setup: function() {
 84     context = SC.RenderContext().addClass('foo') ;
 85   }
 86 });
 87 
 88 test("should return receiver", function() {
 89   equals(context.addClass('foo'), context, "receiver");
 90 });
 91 
 92 test("should add class name to existing classNames array on currentTag", function() {
 93   context.addClass('bar');
 94   same(context.classes(), ['foo', 'bar'], 'has classes');
 95   equals(context._classesDidChange, YES, "note did change");
 96 });
 97 
 98 test("should only add class name once - does nothing if name already in array", function() {
 99   same(context.classes(), ['foo'], 'precondition - has foo classname');
100   context._classesDidChange = NO; // reset  to pretend once not modified
101 
102   context.addClass('foo');
103   same(context.classes(), ['foo'], 'no change');
104   equals(context._classesDidChange, NO, "note did not change");
105 });

107 // ..........................................................
108 // removeClass()
109 //
110 module("SC.RenderContext#removeClass", {
111   setup: function() {
112     context = SC.RenderContext().addClass(['foo', 'bar']) ;
113   }
114 });
115 
116 test("should remove class if already in classNames array", function() {
117   ok(context.classes().indexOf('foo')>=0, "precondition - has foo");
118 
119   context.removeClass('foo');
120   ok(context.classes().indexOf('foo')<0, "does not have foo");
121 });
122 
123 test('should return receiver', function() {
124   equals(context.removeClass('foo'), context, 'receiver');
125 });
126 
127 test("should do nothing if class name not in array", function() {
128   context._classesDidChange = NO; // reset to pretend not modified
129   context.removeClass('imaginary');
130   same(context.classes(), 'foo bar'.w(), 'did not change');
131   equals(context._classesDidChange, NO, "note did not change");
132 });
133 
134 test("should do nothing if there are no class names", function() {
135   context = context.begin();
136   same(context.classes(), [], 'precondition - no class names');
137   context._classesDidChange = NO; // reset to pretend not modified
138 
139   context.removeClass('foo');
140   same(context.classes(), [], 'still no class names -- and no errors');
141   equals(context._classesDidChange, NO, "note did not change");
142 });

144 // ..........................................................
145 // setClass
146 //
147 module("SC.RenderContext#setClass", {
148   setup: function() {
149     context = SC.RenderContext().addClass('foo') ;
150   }
151 });
152 
153 test("should add named class if shouldAdd is YES", function() {
154   ok(!context.hasClass("bar"), "precondition - does not have class bar");
155   context.setClass("bar", YES);
156   ok(context.hasClass("bar"), "now has bar");
157 });
158 
159 test("should remove named class if shouldAdd is NO", function() {
160   ok(context.hasClass("foo"), "precondition - has class foo");
161   context.setClass("foo", NO);
162   ok(!context.hasClass("foo"), "should not have foo ");
163 });
164 
165 test("should return receiver", function() {
166   equals(context, context.setClass("bar", YES), "returns receiver");
167 });
168 
169 test("should add/remove all classes if a hash of class names is passed", function() {
170   ok(context.hasClass("foo"), "precondition - has class foo");
171   ok(!context.hasClass("bar"), "precondition - does not have class bar");
172 
173   context.setClass({ foo: NO, bar: YES });
174 
175   ok(context.hasClass("bar"), "now has bar");
176   ok(!context.hasClass("foo"), "should not have foo ");
177 });
