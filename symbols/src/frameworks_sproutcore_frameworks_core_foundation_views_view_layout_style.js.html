 10 // The scale transform must be last in order to decompose the transformation matrix.
 11 SC.CSS_TRANSFORM_NAMES = ['rotateX', 'rotateY', 'rotateZ', 'scale'];
 12 
 13 SC.CSS_TRANSFORM_MAP = {
 14   rotate: function (val) {
 15     if (SC.typeOf(val) === SC.T_NUMBER) { val += 'deg'; }
 16     return 'rotate(' + val + ')';
 17   },
 18 
 19   rotateX: function (val) {
 20     if (SC.typeOf(val) === SC.T_NUMBER) { val += 'deg'; }
 21     return 'rotateX(' + val + ')';
 22   },
 23 
 24   rotateY: function (val) {
 25     if (SC.typeOf(val) === SC.T_NUMBER) { val += 'deg'; }
 26     return 'rotateY(' + val + ')';
 27   },
 28 
 29   rotateZ: function (val) {
 30     if (SC.typeOf(val) === SC.T_NUMBER) { val += 'deg'; }
 31     return 'rotateZ(' + val + ')';
 32   },
 33 
 34   scale: function (val) {
 35     if (SC.typeOf(val) === SC.T_ARRAY) { val = val.join(', '); }
 36     return 'scale(' + val + ')';
 37   }

 46     return value === undefined ? null : value;

 49   /** @private */
 50   _prepareStyle: function (layout) {
 51     /*jshint eqnull:true */
 52     // It's important to provide null defaults to reset any previous style when
 53     // this is applied.
 54     var commonBorder = this._valueOrNull(layout.border),
 55       style = {
 56         marginLeft: null,
 57         marginTop: null
 58       };
 59 
 60     // Position and size.
 61     style.bottom = layout.bottom;
 62     style.right = layout.right;
 63     style.left = layout.left;
 64     style.top = layout.top;
 65     style.centerX = layout.centerX;
 66     style.centerY = layout.centerY;
 67     style.height = layout.height;
 68     style.width = layout.width;
 69 
 70     // Borders.
 71     style.borderTopWidth = layout.borderTop !== undefined ? layout.borderTop : commonBorder;
 72     style.borderRightWidth = layout.borderRight !== undefined ? layout.borderRight : commonBorder;
 73     style.borderBottomWidth = layout.borderBottom !== undefined ? layout.borderBottom : commonBorder;
 74     style.borderLeftWidth = layout.borderLeft !== undefined ? layout.borderLeft : commonBorder;
 75 
 76     // Minimum and maximum size.
 77     style.maxHeight = this._valueOrNull(layout.maxHeight);
 78     style.maxWidth = this._valueOrNull(layout.maxWidth);
 79     style.minWidth = this._valueOrNull(layout.minWidth);
 80     style.minHeight = this._valueOrNull(layout.minHeight);
 81 
 82     // the toString here is to ensure that it doesn't get px added to it
 83     style.zIndex  = (layout.zIndex != null) ? layout.zIndex.toString() : null;
 84     style.opacity = (layout.opacity != null) ? layout.opacity.toString() : null;
 85 
 86     style.backgroundPosition = this._valueOrNull(layout.backgroundPosition);
 87 
 88     // Handle transforms (including reset).
 89     if (SC.platform.supportsCSSTransforms) {
 90       var transformAttribute = SC.browser.experimentalStyleNameFor('transform'),
 91         transforms = [],
 92         transformMap = SC.CSS_TRANSFORM_MAP;
 93 
 94       // The order of the transforms is important so that we can decompose them
 95       // from the transformation matrix later if necessary.
 96       for (var i = 0, len = SC.CSS_TRANSFORM_NAMES.length; i < len; i++) {
 97         var transformName = SC.CSS_TRANSFORM_NAMES[i],
 98           layoutTransform = layout[transformName];
 99 
100         if (layoutTransform != null) {
101           // normalizing transforms like rotateX: 5 to rotateX(5deg)
102           transforms.push(transformMap[transformName](layoutTransform));
103         }
104       }
105 
106       style[transformAttribute] = transforms.length > 0 ? transforms.join(' ') : null;
107 
108       // Transform origin.
109       var transformOriginAttribute = SC.browser.experimentalStyleNameFor('transformOrigin'),
110         originX = layout.transformOriginX,
111         originY = layout.transformOriginY;
112       if (originX == null && originY == null) {
113         style[transformOriginAttribute] = null;
114       } else {
115         if (originX == null) originX = 0.5;
116         if (originY == null) originY = 0.5;
117         style[transformOriginAttribute] = (originX * 100) + '% ' + (originY * 100) + '%';
118       }
119     }
120 
121     // Reset any transitions.
122     if (SC.platform.supportsCSSTransitions) {
123       style[SC.browser.experimentalStyleNameFor('transition')] = null;
124     }
125 
126     // for ie, we will NOT use alpha. It is just a source of pain.
127     // a) it will not affect absolutely positioned child elements, and is therefore
128     //    useless for most SC purposes.
129     //
130     // b) It completely breaks semitransparent background images (PNGs with opacity)
131     //
132     // If users want to use alpha, they should do it on their own.
133 
134     // if(!SC.none(layout.opacity)) style.filter = "alpha(opacity=%@)".fmt(layout.opacity * 100);
135 
136     return style;
137   },

139   /** @private */
140   _prepareState: function (style) {
141     /*jshint eqnull:true */
142     var state = {};
143 
144     state.hasBottom = (style.bottom != null);
145     state.hasRight = (style.right != null);
146     state.hasLeft = (style.left != null);
147     state.hasTop = (style.top != null);
148     state.hasCenterX = (style.centerX != null);
149     state.hasCenterY = (style.centerY != null);
150     state.hasHeight = (style.height != null);
151     state.hasWidth = (style.width != null);
152     state.hasMaxWidth = (style.maxWidth != null);
153     state.hasMaxHeight = (style.maxHeight != null);
154 
155     return state;
156   },

159   // handles the case where you do width:auto or height:auto and are not using "staticLayout"
160   _invalidAutoValue: function (view, property) {
161     var error = SC.Error.desc("%@.layout() you cannot use %@:auto if staticLayout is disabled".fmt(view, property), "%@".fmt(view), -1);
162     SC.Logger.error(error.toString());
163     throw error;

166   /** @private */
167   _calculatePosition: function (style, state, direction) {
168     var start, finish, size,
169       hasStart, hasFinish, hasSize, hasMaxSize,
170       startBorder,
171       finishBorder;
172 
173     if (direction === 'X') {
174       start      = 'left';
175       finish     = 'right';
176       size       = 'width';
177       startBorder  = 'borderLeftWidth';
178       finishBorder = 'borderRightWidth';
179       hasStart   = state.hasLeft;
180       hasFinish  = state.hasRight;
181       hasSize    = state.hasWidth;
182       hasMaxSize = state.hasMaxWidth;
183     } else {
184       start      = 'top';
185       finish     = 'bottom';
186       size       = 'height';
187       startBorder  = 'borderTopWidth';
188       finishBorder = 'borderBottomWidth';
189       hasStart   = state.hasTop;
190       hasFinish  = state.hasBottom;
191       hasSize    = state.hasHeight;
192       hasMaxSize = state.hasMaxHeight;
193     }
194 
195     style[start]  = this._cssNumber(style[start]);
196     style[finish] = this._cssNumber(style[finish]);
197 
198     var startBorderVal = this._cssNumber(style[startBorder]),
199       finishBorderVal = this._cssNumber(style[finishBorder]),
200       sizeNum = style[size];
201 
202     style[startBorder] = startBorderVal;
203     style[finishBorder] = finishBorderVal;
204 
205     // This is a normal number
206     if (sizeNum >= 1) { sizeNum -= (startBorderVal + finishBorderVal); }
207     style[size] = this._cssNumber(sizeNum);
208 
209     if (hasStart) {
210       // top, bottom, height -> top, bottom
211       if (hasFinish && hasSize)  { style[finish] = null; }
212     } else {
213       // bottom aligned
214       if (!hasFinish || (hasFinish && !hasSize && !hasMaxSize)) {
215         // no top, no bottom
216         style[start] = 0;
217       }
218     }
219 
220     if (!hasSize && !hasFinish) { style[finish] = 0; }
221   },

224   /** @private */
225   _calculateCenter: function (style, direction) {
226     var size, center, start, finish, margin,
227         startBorder,
228         finishBorder;
229 
230     if (direction === 'X') {
231       size   = 'width';
232       center = 'centerX';
233       start  = 'left';
234       finish = 'right';
235       margin = 'marginLeft';
236       startBorder  = 'borderLeftWidth';
237       finishBorder = 'borderRightWidth';
238     } else {
239       size   = 'height';
240       center = 'centerY';
241       start  = 'top';
242       finish = 'bottom';
243       margin = 'marginTop';
244       startBorder  = 'borderTopWidth';
245       finishBorder = 'borderBottomWidth';
246     }
247 
248     style[start] = "50%";
249 
250     var startBorderVal = this._cssNumber(style[startBorder]),
251       finishBorderVal = this._cssNumber(style[finishBorder]),
252       sizeValue   = style[size],
253       centerValue = style[center],
254       sizeIsPercent = SC.isPercentage(sizeValue),
255       value;
256 
257     style[startBorder] = startBorderVal;
258     style[finishBorder] = finishBorderVal;
259 
260     // Calculate the margin offset used to center the value along this axis.
261     if (SC.none(sizeValue)) {
262       //@if(debug)
263       // This error message happens whenever width or height is not set.
264       SC.warn("Developer Warning: When setting '" + center + "' in the layout, you must also set '" + size + "'.");
265       //@endif
266       style[margin] = "50%";
267     } else {
268       value = centerValue - sizeValue / 2;
269       style[margin] = (sizeIsPercent) ? Math.floor(value * 100) + "%" : Math.floor(value);
270     }
271 
272     // If > 1 then it's a pixel value, in which case we shrink it to accommodate the borders.
273     if (sizeValue > 1) { sizeValue -= (startBorderVal + finishBorderVal); }
274 
275     style[size] = this._cssNumber(sizeValue) || 0;
276     style[finish] = style[center] = null;
277   },

279   /** @private */
280   // return "auto" for "auto", null for null, converts 0.XY into "XY%".
281   // otherwise returns the original number, rounded down
282   _cssNumber: function (val) {
283     /*jshint eqnull:true*/
284     if (val == null) { return null; }
285     else if (val === SC.LAYOUT_AUTO) { return SC.LAYOUT_AUTO; }
286     else if (SC.isPercentage(val)) { return (val * 100) + "%"; }
287     else { return Math.floor(val); }
288   },

290   /** @private
291     Calculate the layout style for the given view, making adjustments to allow
292     for flexible positioning, animation and accelerated transforms.
293 
294     @return {Object} Layout style hash.
295   */
296   calculate: function (view) {
297     var layout = view.get('layout'),
298       animations = view._activeAnimations,
299       state,
300       useStaticLayout = view.get('useStaticLayout'),
301       style;
302 
303     // Fast path!
304     // If the developer sets useStaticLayout and doesn't provide a unique `layout` property, we
305     // should not insert the styles "left: 0px; right: 0px; top: 0px; bottom: 0px" as they could
306     // conflict with the developer's intention.  However, if they do provide a unique `layout`,
307     // use it.
308     if (useStaticLayout && layout === SC.View.prototype.layout) { return {}; }
309 
310     style = this._prepareStyle(layout);
311     state = this._prepareState(style);
312 
313     // handle invalid use of auto in absolute layouts
314     if (!useStaticLayout) {
315       if (style.width === SC.LAYOUT_AUTO) { this._invalidAutoValue(view, "width"); }
316       if (style.height === SC.LAYOUT_AUTO) { this._invalidAutoValue(view, "height"); }
317     }
318 
319     // X DIRECTION
320     if (state.hasLeft || state.hasRight || !state.hasCenterX) {
321       this._calculatePosition(style, state, "X");
322     } else {
323       this._calculateCenter(style, "X");
324     }
325 
326     // Y DIRECTION
327     if (state.hasTop || state.hasBottom || !state.hasCenterY) {
328       this._calculatePosition(style, state, "Y");
329     } else {
330       this._calculateCenter(style, "Y");
331     }
332 
333     this._calculateAnimations(style, animations, view.get('hasAcceleratedLayer'));
334 
335     // convert any numbers into a number + "px".
336     for (var key in style) {
337       var value = style[key];
338       if (typeof value === SC.T_NUMBER) { style[key] = (value + "px"); }
339     }
340 
341     return style;
342   },

344   /** @private Calculates animation styles. */
345   _calculateAnimations: function (style, animations, hasAcceleratedLayer) {
346     /*jshint eqnull:true*/
347     var key,
348       shouldTranslate;
349 
350     // Handle transforms
351     if (hasAcceleratedLayer) {
352       shouldTranslate = YES;
353 
354       // If we're animating other transforms at different speeds, don't use acceleratedLayer
355       if (animations && (animations.top || animations.left)) {
356         for (key in animations) {
357           if (SC.CSS_TRANSFORM_MAP[key] &&
358               ((animations.top &&
359                 animations.top.duration !== animations[key].duration) ||
360                (animations.left &&
361                 animations.left.duration !== animations[key].duration))) {
362             shouldTranslate = NO;
363           }
364         }
365       }
366 
367       if (shouldTranslate) {
368         var transformAttribute = SC.browser.experimentalStyleNameFor('transform'),
369           curValue = style[transformAttribute],
370           newValue;
371 
372         newValue = 'translateX(' + style.left + 'px) translateY(' + style.top + 'px)';
373 
374         // double check to make sure this is needed
375         if (SC.platform.supportsCSS3DTransforms) { newValue += ' translateZ(0px)'; }
376 
377         // Append any current transforms.
378         // NOTE: The order of transforms is important. If we scale before translate, our translations
379         // will be scaled and incorrect.
380         if (curValue) { newValue += ' ' + curValue; }
381         style[transformAttribute] = newValue;
382 
383         // Set the absolute left & top style to 0,0 (will be translated from there).
384         style.left = 0;
385         style.top = 0;
386       }
387     }
388 
389     // Handle animations
390     if (animations) {
391       if (SC.platform.supportsCSSTransitions) {
392         var transitions = [];
393 
394         for (key in animations) {
395           var animation = animations[key],
396             isTransformProperty = !!SC.CSS_TRANSFORM_MAP[key],
397             isTurboProperty = shouldTranslate && (key === 'top' || key === 'left');
398 
399           if (SC.platform.supportsCSSTransforms && (isTurboProperty || isTransformProperty)) {
400             // Untrack the un-transformed property name.
401             delete animations[key];
402 
403             // The key will be either 'transform' or one of '-webkit-transform', '-ms-transform', '-moz-transform', '-o-transform'
404             key = SC.browser.experimentalCSSNameFor('transform');
405 
406             var curTransformAnimation = animations[key];
407 
408             // Because multiple transforms actually share one CSS property, we can't animate multiple transforms
409             // at different speeds. So, to handle that case, we just force them to all have the same length.
410             if (curTransformAnimation) {
411               //@if(debug)
412               if (curTransformAnimation.duration !== animation.duration || curTransformAnimation.timing !== animation.timing || curTransformAnimation.delay !== animation.delay) {
413                 SC.Logger.warn("Developer Warning: Can't animate transforms with different durations, timings or delays! Using the first options specified.");
414               }
415               //@endif
416               animation = curTransformAnimation;
417             } else {
418               // Track the transformed property name.
419               animations[key] = animation;
420             }
421           }
422 
423           // Fix up the centerX & centerY properties.
424           if (key === 'centerX') { key = 'margin-left'; }
425           if (key === 'centerY') { key = 'margin-top'; }
426 
427           // We're actually storing the css for the animation on layout.animate[key].css
428           animation.css = key + " " + animation.duration + "s " + animation.timing + " " + animation.delay + "s";
429 
430           // If there are multiple transform properties, we only need to set this key once.
431           // We already checked before to make sure they have the same duration.
432           // if (!pendingAnimations[key]) {
433           if (transitions.indexOf(animation.css) < 0) {
434             transitions.push(animation.css);
435           }
436         }
437 
438         style[SC.browser.experimentalStyleNameFor('transition')] = transitions.join(", ");
439       } else {
440         // TODO: Do it the JS way
441       }
442     }
443   }
