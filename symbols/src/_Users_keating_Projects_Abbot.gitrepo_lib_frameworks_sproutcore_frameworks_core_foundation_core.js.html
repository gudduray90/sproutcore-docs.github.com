 92 SC.mixin(/** @lends SC */ {
 93   
 94   /**
 95     Reads or writes data from a global cache.  You can use this facility to
 96     store information about an object without actually adding properties to
 97     the object itself.  This is needed especially when working with DOM,
 98     which can leak easily in IE.
 99     
100     To read data, simply pass in the reference element (used as a key) and
101     the name of the value to read.  To write, also include the data.
102     
103     You can also just pass an object to retrieve the entire cache.
104     
105     @param elem {Object} An object or Element to use as scope
106     @param name {String} Optional name of the value to read/write
107     @param data {Object} Optional data.  If passed, write.
108     @returns {Object} the value of the named data
109   */
110   data: $.data,
111   
112   /**
113     Removes data from the global cache.  This is used throughout the
114     framework to hold data without creating memory leaks.
115     
116     You can remove either a single item on the cache or all of the cached 
117     data for an object.
118     
119     @param elem {Object} An object or Element to use as scope
120     @param name {String} optional name to remove. 
121     @returns {Object} the value or cache that was removed
122   */
123   removeData: $.removeData,
124 
125   // ..........................................................
126   // LOCALIZATION SUPPORT
127   //
128 
129   /**
130     Known loc strings
131 
132     @type Hash
133   */
134   STRINGS: {},
135 
136   /**
137     This is a simplified handler for installing a bunch of strings.  This
138     ignores the language name and simply applies the passed strings hash.
139 
140     @param {String} lang the language the strings are for
141     @param {Hash} strings hash of strings
142     @returns {SC} The receiver, useful for chaining calls to the same object.
143   */
144   stringsFor: function(lang, strings) {
145     SC.mixin(SC.STRINGS, strings);
146     return this ;
147   }
