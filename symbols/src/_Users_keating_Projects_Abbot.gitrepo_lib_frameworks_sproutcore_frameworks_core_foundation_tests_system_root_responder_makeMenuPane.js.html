  9 var responder, menu;
 10 
 11 module("SC.RootResponder#makeMenuPane", {
 12   setup: function() {
 13     responder = SC.RootResponder.create();
 14     menu = SC.Pane.create({
 15       acceptsMenuPane: YES
 16     });
 17   },
 18   
 19   teardown: function() {
 20     menu.remove();
 21     menu = responder = null;
 22   }
 23 });
 24 
 25 test("Returns receiver", function() {
 26   equals(responder.makeMenuPane(menu), responder, 'returns receiver');
 27 });
 28 
 29 test("Sets RootResponder's menuPane", function() {
 30   equals(responder.get('menuPane'), null, 'precond - menuPane should be null by default');
 31   responder.makeMenuPane(menu);
 32   equals(responder.get('menuPane'), menu, 'menuPane should be passed menu');
 33 });
 34 
 35 test("menuPane does not affect keyPane", function() {
 36   var p2 = SC.Pane.create();
 37   responder.makeKeyPane(p2);
 38   equals(responder.get('keyPane'), p2, 'precond - pane should be key pane');
 39   responder.makeMenuPane(menu);
 40   equals(responder.get('menuPane'), menu, 'menuPane should be set');
 41   equals(responder.get('keyPane'), p2, 'key pane should not change');
 42 });
 43 
 44 test("Pane should not become menu pane if acceptsMenuPane is not YES", function() {
 45   menu.set('acceptsMenuPane', NO);
 46   responder.makeMenuPane(menu);
 47   equals(responder.get('menuPane'), null, 'menuPane should remain null');
 48 });
