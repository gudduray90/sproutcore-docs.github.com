 12 var revMatches = NO , ObjectA;
 13 
 14 module("object.propertyChanges", {  
 15   setup: function() {
 16     ObjectA = SC.Object.create({
 17       foo  : 'fooValue',
 18       prop : 'propValue',
 19             
 20       action: function() {
 21         this.prop= 'changedPropValue';
 22       }.observes('foo'),
 23       
 24       newFoo : 'newFooValue',
 25       newProp: 'newPropValue',
 26       
 27       notifyAction: function() {
 28         this.newProp = 'changedNewPropValue';
 29       }.observes('newFoo'),
 30       
 31       notifyAllAction: function() {
 32         this.newFoo = 'changedNewFooValue';
 33       }.observes('prop'),
 34 
 35       starProp: null,
 36       starObserver: function(target, key, value, rev) {
 37         revMatches = (rev === target.propertyRevision) ;
 38         this.starProp = key;
 39       }
 40       
 41     });
 42     }
 43 });

 46 test("should observe the changes within the nested begin / end property changes", function() {
 47     
 48   //start the outer nest
 49   ObjectA.beginPropertyChanges();
 50     // Inner nest
 51     ObjectA.beginPropertyChanges();
 52         ObjectA.set('foo', 'changeFooValue');
 53       equals(ObjectA.prop, "propValue") ;
 54       ObjectA.endPropertyChanges();
 55     
 56     //end inner nest
 57     ObjectA.set('prop', 'changePropValue');
 58     equals(ObjectA.newFoo, "newFooValue") ;
 59   //close the outer nest
 60   ObjectA.endPropertyChanges();
 61   
 62   equals(ObjectA.prop, "changedPropValue") ;
 63   equals(ObjectA.newFoo, "changedNewFooValue") ;
 64   
 65 });

 86 test("should indicate that the property of an object has just changed", function() {
 87   // inidicate that proprty of foo will change to its subscribers
 88   ObjectA.propertyWillChange('foo') ;
 89   
 90   //Value of the prop is unchanged yet as this will be changed when foo changes
 91   equals(ObjectA.prop, 'propValue' ) ;
 92   
 93   //change the value of foo.
 94   ObjectA.foo = 'changeFooValue';
 95   
 96   // Indicate the subscribers of foo that the value has just changed
 97   ObjectA.propertyDidChange('foo', null) ;
 98   
 99   // Values of prop has just changed
100   equals(ObjectA.prop,'changedPropValue') ;
101 });
102 
103 test("should notify that the property of an object has changed", function() {
104   // Notify to its subscriber that the values of 'newFoo' will be changed. In this
105   // case the observer is "newProp". Therefore this will call the notifyAction function
106   // and value of "newProp" will be changed.
107   ObjectA.notifyPropertyChange('newFoo','fooValue');
108   
109   //value of newProp changed.
110   equals(ObjectA.newProp,'changedNewPropValue') ;
111 });
112 
113 test("should notify all observers that their property might have changed", function() {
114   //When this function is called, all the subscribers are notified that something has
115   //Changed. So when allPropertiesDidChange() is called, all the subscribers get invoked. 
116   ObjectA.allPropertiesDidChange();
117   
118   //All the values changed.
119   equals(ObjectA.prop,'changedPropValue') ;
120   equals(ObjectA.newProp,'changedNewPropValue') ;
121   equals(ObjectA.newFoo,'changedNewFooValue') ;
122 });

140 test("should invalidate function property cache when notifyPropertyChange is called", function() {
141   
142   var a = SC.Object.create({
143     _b: null,
144     b: function(key, value) {
145       if (value !== undefined) {
146         this._b = value;
147         return this;
148       }
149       return this._b;
150     }.property()
151   });
152   
153   a.set('b', 'foo');
154   equals(a.get('b'), 'foo', 'should have set the correct value for property b');
155   
156   a._b = 'bar';
157   a.notifyPropertyChange('b');
158   a.set('b', 'foo');
159   equals(a.get('b'), 'foo', 'should have invalidated the cache so that the newly set value is actually set');
160   
161 });
