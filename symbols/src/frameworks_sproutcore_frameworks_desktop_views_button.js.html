497   /**
498     Handle a key equivalent if set.  Trigger the default action for the
499     button.  Depending on the implementation this may vary.
500 
501     @param {String} keystring
502     @param {SC.Event} evt
503     @returns {Boolean}  YES if handled, NO otherwise
504   */
505   performKeyEquivalent: function(keystring, evt) {
506     //If this is not visible
507     if (!this.get('isVisibleInWindow')) return NO;
508 
509     if (!this.get('isEnabledInPane')) return NO;
510     var equiv = this.get('keyEquivalent');
511 
512     // button has defined a keyEquivalent and it matches!
513     // if triggering succeeded, true will be returned and the operation will
514     // be handled (i.e performKeyEquivalent will cease crawling the view
515     // tree)
516     if (equiv) {
517       if (equiv === keystring) return this.triggerAction(evt);
518 
519     // should fire if isDefault OR isCancel.  This way if isDefault AND
520     // isCancel, responds to both return and escape
521     } else if ((this.get('isDefault') && (keystring === 'return')) ||
522         (this.get('isCancel') && (keystring === 'escape'))) {
523           return this.triggerAction(evt);
524     }
525 
526     return NO; // did not handle it; keep searching
527   },

533   /**
534     This is the standard logic to compute a proposed isSelected state for a
535     new value.  This takes into account the `toggleOnValue`/`toggleOffValue`
536     properties, among other things.  It may return `YES`, `NO`, or
537     `SC.MIXED_STATE`.
538 
539     @param {Object} value
540     @returns {Boolean} return state
541   */
542   computeIsSelectedForValue: function(value) {
543     var targetValue = this.get('toggleOnValue'), state, next ;
544 
545     if (SC.typeOf(value) === SC.T_ARRAY) {
546 
547       // treat a single item array like a single value
548       if (value.length === 1) {
549         state = (value[0] == targetValue) ;
550 
551       // for a multiple item array, check the states of all items.
552       } else {
553         state = null;
554         value.find(function(x) {
555           next = (x == targetValue) ;
556           if (state === null) {
557             state = next ;
558           } else if (next !== state) state = SC.MIXED_STATE ;
559           return state === SC.MIXED_STATE ; // stop when we hit a mixed state.
560         });
561       }
562 
563     // for single values, just compare to the toggleOnValue...use truthiness
564     } else {
565       if(value === SC.MIXED_STATE) state = SC.MIXED_STATE;
566       else state = (value === targetValue) ;
567     }
568     return state ;
569   },

571   /** @private

575     var value = this.get('value'),

577     this.set('isSelected', state) ; // set new state...

579     // value acts as a display property
580     this.displayDidChange();

583   /** @private
584     Whenever the selected state is changed, make sure the button value is
585     also updated.  Note that this may be called because the value has just
586     changed.  In that case this should do nothing.
587   */
588   _button_isSelectedDidChange: function() {
589     var newState = this.get('isSelected'),
590         curState = this.computeIsSelectedForValue(this.get('value'));
591 
592     // fix up the value, but only if computed state does not match.
593     // never fix up value if isSelected is set to MIXED_STATE since this can
594     // only come from the value.
595     if ((newState !== SC.MIXED_STATE) && (curState !== newState)) {
596       var valueKey = (newState) ? 'toggleOnValue' : 'toggleOffValue' ;
597       this.set('value', this.get(valueKey));
598     }

739   /** @private */
740   keyDown: function(evt) {
741     // handle tab key
742      if(!this.get('isEnabledInPane')) return YES;
743     if (evt.which === 9 || evt.keyCode === 9) {
744       var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');

760   /** @private
761     Perform an action based on the behavior of the button.
762 
763      - toggle behavior: switch to on/off state
764      - on behavior: turn on.
765      - off behavior: turn off.
766      - otherwise: invoke target/action
767   */
768   _action: function(evt, skipHoldRepeat) {
769     switch(this.get('buttonBehavior')) {
770 
771     // When toggling, try to invert like values. i.e. 1 => 0, etc.
772     case SC.TOGGLE_BEHAVIOR:
773       var sel = this.get('isSelected') ;
774       if (sel) {
775         this.set('value', this.get('toggleOffValue')) ;
776       } else {
777         this.set('value', this.get('toggleOnValue')) ;
778       }
779       break ;
780 
781     // set value to on.  change 0 => 1.
782     case SC.TOGGLE_ON_BEHAVIOR:
783       this.set('value', this.get('toggleOnValue')) ;
784       break ;
785 
786     // set the value to false. change 1 => 0
787     case SC.TOGGLE_OFF_BEHAVIOR:
788       this.set('value', this.get('toggleOffValue')) ;
789       break ;
790 
791     case SC.HOLD_BEHAVIOR:
792       this._runHoldAction(evt, skipHoldRepeat);
793       break ;
794 
795     // otherwise, just trigger an action if there is one.
796     default:
797       //if (this.action) this.action(evt);
798       this._runAction(evt);
799     }
800   },
801 
802   /** @private */
803   _runAction: function(evt) {
804     var action = this.get('action'),
805         target = this.get('target') || null,
806         rootResponder;
807 
808     if (action) {
809       if (action && (SC.typeOf(action) === SC.T_FUNCTION)) {
810         this.action(evt);
811         return;
812       } else {
813         rootResponder = this.getPath('pane.rootResponder');
814         if (rootResponder) {
815           // newer action method + optional target syntax...
816           rootResponder.sendAction(action, target, this, this.get('pane'), null, this);
817         }
818       }
819     }
820   },
821 
822   /** @private */
823   _runHoldAction: function(evt, skipRepeat) {
824     if (this.get('isActive')) {
825       this._runAction();
826 
827       if (!skipRepeat) {
828         // This run loop appears to only be necessary for testing
829         SC.RunLoop.begin();
830         this.invokeLater('_runHoldAction', this.get('holdInterval'), evt);
831         SC.RunLoop.end();
832       }
833     }
834   },
835

837   /** @private */
838   didBecomeKeyResponderFrom: function(keyView) {
839     // focus the text field.
840     if (!this._isFocused) {
841       this._isFocused = YES ;
842       this.becomeFirstResponder();
843       if (this.get('isVisibleInWindow')) {
844         this.$().focus();
845       }
846     }
847   },
848 
849   /** @private */
850   willLoseKeyResponderTo: function(responder) {
851     if (this._isFocused) this._isFocused = NO ;
852   },

854   /** @private */
855   didAppendToDocument: function() {
856     if(SC.browser.isIE &&
857         SC.browser.compare(SC.browser.engineVersion, '7') === 0 &&
858         this.get('useStaticLayout')){
859       var layout = this.get('layout'),
860           elem = this.$(), w=0;
861       if(elem && elem[0] && (w=elem[0].clientWidth) && w!==0 && this._labelMinWidthIE7===0){
862         var label = this.$('.sc-button-label'),
863             paddingRight = parseInt(label.css('paddingRight'),0),
864             paddingLeft = parseInt(label.css('paddingLeft'),0),
865             marginRight = parseInt(label.css('marginRight'),0),
866             marginLeft = parseInt(label.css('marginLeft'),0);
867         if(marginRight=='auto') SC.Logger.log(marginRight+","+marginLeft+","+paddingRight+","+paddingLeft);
868         if(!paddingRight && isNaN(paddingRight)) paddingRight = 0;
869         if(!paddingLeft && isNaN(paddingLeft)) paddingLeft = 0;
870         if(!marginRight && isNaN(marginRight)) marginRight = 0;
871         if(!marginLeft && isNaN(marginLeft)) marginLeft = 0;
872 
873         this._labelMinWidthIE7 = w-(paddingRight + paddingLeft)-(marginRight + marginLeft);
874         label.css('minWidth', this._labelMinWidthIE7+'px');
875       }else{
876         this.invokeLater(this.didAppendToDocument, 1);
877       }
878     }
879   }
