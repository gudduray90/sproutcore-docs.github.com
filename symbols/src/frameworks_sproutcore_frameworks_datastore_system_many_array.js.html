  8 /**
  9   @class
 10 
 11   A `ManyArray` is used to map an array of record ids back to their
 12   record objects which will be materialized from the owner store on demand.
 13 
 14   Whenever you create a `toMany()` relationship, the value returned from the
 15   property will be an instance of `ManyArray`.  You can generally customize the
 16   behavior of ManyArray by passing settings to the `toMany()` helper.
 17 
 18   @extends SC.Enumerable
 19   @extends SC.Array
 20   @since SproutCore 1.0
 21 */
 22 
 23 SC.ManyArray = SC.Object.extend(SC.Enumerable, SC.Array,
 24   /** @scope SC.ManyArray.prototype */ {
 25 
 26   /**
 27     `recordType` will tell what type to transform the record to when
 28     materializing the record.
 29 
 30     @default null
 31     @type String
 32   */
 33   recordType: null,
 34 
 35   /**
 36     If set, the record will be notified whenever the array changes so that
 37     it can change its own state

 40     @type SC.Record
 41   */

 43 
 44   /**
 45     If set will be used by the many array to get an editable version of the

 58     @type SC.ManyAttribute

 62   /**
 63     The store that owns this record array.  All record arrays must have a
 64     store to function properly.
 65 
 66     @type SC.Store
 67     @property
 68   */

 70     return this.get('record').get('store');

 78     @property
 79   */
 80   storeKey: function() {
 81     return this.get('record').get('storeKey');
 82   }.property('record').cacheable(),

 86     Returns the `storeId`s in read-only mode.  Avoids modifying the record
 87     unnecessarily.
 88 
 89     @type SC.Array
 90     @property
 91   */
 92   readOnlyStoreIds: function() {
 93     return this.get('record').readAttribute(this.get('propertyName'));
 94   }.property(),
 95 
 96 
 97   /**
 98     Returns an editable array of `storeId`s.  Marks the owner records as
 99     modified.
100 
101     @type {SC.Array}
102     @property
103   */
104   editableStoreIds: function() {
105     var store    = this.get('store'),
106         storeKey = this.get('storeKey'),
107         pname    = this.get('propertyName'),
108         ret, hash;
109 
110     ret = store.readEditableProperty(storeKey, pname);
111     if (!ret) {
112       hash = store.readEditableDataHash(storeKey);
113       ret = hash[pname] = [];
114     }
115 
116     if (ret !== this._prevStoreIds) this.recordPropertyDidChange();
117     return ret ;
118   }.property(),

177   /** @private
178     Returned length is a pass-through to the `storeIds` array.
179 
180     @type Number
181     @property
182   */
183   length: function() {
184     var storeIds = this.get('readOnlyStoreIds');
185     return storeIds ? storeIds.get('length') : 0;
186   }.property('readOnlyStoreIds'),
187 
188   /** @private
189     Looks up the store id in the store ids array and materializes a
190     records.
191   */
192   objectAt: function(idx) {
193     var recs      = this._records,
194         storeIds  = this.get('readOnlyStoreIds'),
195         store     = this.get('store'),
196         recordType = this.get('recordType'),
197         storeKey, ret, storeId ;
198 
199     if (!storeIds || !store) return undefined; // nothing to do
200     if (recs && (ret=recs[idx])) return ret ; // cached
201 
202     // not in cache, materialize
203     if (!recs) this._records = recs = [] ; // create cache
204     storeId = storeIds.objectAt(idx);
205     if (storeId) {
206 
207       // if record is not loaded already, then ask the data source to
208       // retrieve it
209       storeKey = store.storeKeyFor(recordType, storeId);
210 
211       if (store.readStatus(storeKey) === SC.Record.EMPTY) {
212         store.retrieveRecord(recordType, null, storeKey);
213       }
214 
215       recs[idx] = ret = store.materializeRecord(storeKey);
216     }
217     return ret ;
218   },
219 
220   /** @private
221     Pass through to the underlying array.  The passed in objects must be
222     records, which can be converted to `storeId`s.
223   */
224   replace: function(idx, amt, recs) {
225 
226     if (!this.get('isEditable')) {
227       throw new Error("%@.%@[] is not editable".fmt(this.get('record'), this.get('propertyName')));
228     }
229 
230     var storeIds = this.get('editableStoreIds'),
231         len      = recs ? (recs.get ? recs.get('length') : recs.length) : 0,
232         record   = this.get('record'),
233         pname    = this.get('propertyName'),
234         i, ids, toRemove, inverse, attr, inverseRecord;
235 
236     // map to store keys
237     ids = [] ;
238     for(i=0;i<len;i++) ids[i] = recs.objectAt(i).get('id');
239 
240     // if we have an inverse - collect the list of records we are about to
241     // remove
242     inverse = this.get('inverse');
243     if (inverse && amt>0) {
244       toRemove = SC.ManyArray._toRemove;
245       if (toRemove) SC.ManyArray._toRemove = null; // reuse if possible
246       else toRemove = [];
247 
248       for(i=0;i<amt;i++) toRemove[i] = this.objectAt(idx + i);
249     }
250 
251     // pass along - if allowed, this should trigger the content observer
252     storeIds.replace(idx, amt, ids);
253 
254     // ok, notify records that were removed then added; this way reordered
255     // objects are added and removed
256     if (inverse) {
257 
258       // notify removals
259       for(i=0;i<amt;i++) {
260         inverseRecord = toRemove[i];
261         attr = inverseRecord ? inverseRecord[inverse] : null;
262         if (attr && attr.inverseDidRemoveRecord) {
263           attr.inverseDidRemoveRecord(inverseRecord, inverse, record, pname);
264         }
265       }
266 
267       if (toRemove) {
268         toRemove.length = 0; // cleanup
269         if (!SC.ManyArray._toRemove) SC.ManyArray._toRemove = toRemove;
270       }
271 
272       // notify additions
273       for(i=0;i<len;i++) {
274         inverseRecord = recs.objectAt(i);
275         attr = inverseRecord ? inverseRecord[inverse] : null;
276         if (attr && attr.inverseDidAddRecord) {
277           attr.inverseDidAddRecord(inverseRecord, inverse, record, pname);
278         }
279       }
280 
281     }
282 
283     // only mark record dirty if there is no inverse or we are master
284     if (record && (!inverse || this.get('isMaster'))) {
285       record.recordDidChange(pname);
286     }
287 
288     this.enumerableContentDidChange(idx, amt, len - amt);
289 
290     return this;
291   },

297   /**
298     Called by the `ManyAttribute` whenever a record is removed on the inverse
299     of the relationship.
300 
301     @param {SC.Record} inverseRecord the record that was removed
302     @returns {SC.ManyArray} receiver
303   */
304   removeInverseRecord: function(inverseRecord) {
305 
306     if (!inverseRecord) return this; // nothing to do
307     var id = inverseRecord.get('id'),
308         storeIds = this.get('editableStoreIds'),
309         idx      = (storeIds && id) ? storeIds.indexOf(id) : -1,
310         record;
311 
312     if (idx >= 0) {
313       storeIds.removeAt(idx);
314       if (this.get('isMaster') && (record = this.get('record'))) {
315         record.recordDidChange(this.get('propertyName'));
316       }
317     }
318 
319     return this;
320   },
321 
322   /**
323     Called by the `ManyAttribute` whenever a record is added on the inverse
324     of the relationship.
325 
326     @param {SC.Record} inverseRecord the record this array is a part of
327     @returns {SC.ManyArray} receiver
328   */
329   addInverseRecord: function (inverseRecord) {
330     // Fast path!
331     if (!inverseRecord) return this;
332 
333     var storeIds = this.get('editableStoreIds'),
334         orderBy  = this.get('orderBy'),
335         len      = storeIds.get('length'),
336         idx, record;
337 
338     // find idx to insert at.
339     if (orderBy) {
340       idx = this._findInsertionLocation(inverseRecord, 0, len, orderBy);
341     } else idx = len;
342 
343     storeIds.insertAt(idx, inverseRecord.get('id'));
344     if (this.get('isMaster') && (record = this.get('record'))) {
345       record.recordDidChange(this.get('propertyName'));
346     }
347 
348     return this;
349   },
350 
351   /** @private binary search to find insertion location */
352   _findInsertionLocation: function (rec, min, max, orderBy) {
353     var idx   = min + Math.floor((max - min) / 2),
354         cur   = this.objectAt(idx),
355         order = this._compare(rec, cur, orderBy);
356 
357     if (order < 0) {
358       // The location is before the first index.
359       if (idx === 0) return idx;
360 
361       // The location is in the lower subset.
362       else return this._findInsertionLocation(rec, 0, idx - 1, orderBy);
363     } else if (order > 0) {
364       // The location is after the current index.
365       if (idx >= max) return idx + 1;
366 
367       // The location is in the upper subset.
368       else return this._findInsertionLocation(rec, idx + 1, max, orderBy);
369     } else {
370       // The location is the current index.
371       return idx;
372     }
373   },
374 
375   /** @private function to compare two objects*/
376   _compare: function (a, b, orderBy) {
377     var t = SC.typeOf(orderBy),
378         ret, idx, len;
379 
380     if (t === SC.T_FUNCTION) ret = orderBy(a, b);
381     else if (t === SC.T_STRING) ret = SC.compare(a,b);
382     else {
383       len = orderBy.get('length');
384       ret = 0;
385       for(idx=0;(ret===0) && (idx<len);idx++) ret = SC.compare(a,b);
386     }
387 
388     return ret ;
389   },

395   /** @private
396     Invoked whenever the `storeIds` array changes.  Observes changes.
397   */
398   recordPropertyDidChange: function(keys) {
399 
400     if (keys && !keys.contains(this.get('propertyName'))) return this;
401 
402     var storeIds = this.get('readOnlyStoreIds'), oldLen, newLen;
403     var prev = this._prevStoreIds, f = this._storeIdsContentDidChange;
404 
405     if (storeIds === prev) return this; // nothing to do
406 
407     if (prev) {
408       prev.removeArrayObservers({
409         target: this,
410         willChange: this.arrayContentWillChange,
411         didChange: f
412       });
413 
414       oldLen = prev.get('length');
415     } else {
416       oldLen = 0;
417     }
418 
419     if (storeIds) {
420       storeIds.addArrayObservers({
421         target: this,
422         willChange: this.arrayContentWillChange,
423         didChange: f
424       });
425 
426       newLen = storeIds.get('length');
427     } else {
428       newLen = 0;
429     }
430 
431     this.arrayContentWillChange(0, oldLen, newLen);
432     this._prevStoreIds = storeIds;
433     this._storeIdsContentDidChange(0, oldLen, newLen);
434   },
435 
436   /** @private
437     Invoked whenever the content of the storeIds array changes.  This will
438     dump any cached record lookup and then notify that the enumerable content
439     has changed.
440   */
441   _storeIdsContentDidChange: function(start, removedCount, addedCount) {
442     this._records = null ; // clear cache
443     this.arrayContentDidChange(start, removedCount, addedCount);
444   },
445 
446   /** @private */
447   unknownProperty: function(key, value) {
448     var ret;
449     if (SC.typeOf(key) === SC.T_STRING) ret = this.reducedProperty(key, value);
450     return ret === undefined ? sc_super() : ret;
451   },
452 
453   /** @private */
454   init: function() {
455     sc_super();
456     this.recordPropertyDidChange();
457   }
