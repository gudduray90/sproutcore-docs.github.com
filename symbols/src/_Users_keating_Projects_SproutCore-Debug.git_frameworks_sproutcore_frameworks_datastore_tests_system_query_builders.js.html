 15 var TestRecord = SC.Record.extend();
 16 var TestRecord2 = SC.Record.extend();
 17 
 18 function queryEquals(q, location, recordType, conditions, extra, desc) {
 19   if (desc===undefined && typeof extra === 'string') {
 20     desc = extra;  extra = undefined ;
 21   }
 22   if (!desc) desc = '';
 23 
 24   ok(!!q, desc + ': should have a query');
 25   equals(q.get('isFrozen'), YES, desc + ": should be frozen");
 26 
 27   if (q) {
 28     if (location) {
 29       equals(q.get('location'), location, desc + ": should have location");
 30     }
 31 
 32     if (recordType && recordType.isEnumerable) {
 33       same(q.get('recordTypes'), recordType, desc + ': should have recordTypes (plural)');
 34     } else {
 35       equals(q.get('recordType'), recordType, desc + ': should have recordType (singular)');
 36     }
 37 
 38     equals(q.get('conditions'), conditions, desc + ': should have conditions');
 39 
 40     if (extra) {
 41       for (var key in extra) {
 42         if (!extra.hasOwnProperty(key)) continue;
 43         equals(q.get(key), extra[key], desc + ': should have extra key ' + key);
 44       }
 45     }
 46   }
 47 }

 53 // The local() and remote() builder methods are very similar.  This will
 54 // perform the same basic tests on both of them.  If you add a builder for a
 55 // new type of location, you can just call this function again with your new
 56 // location
 57 function performBasicTests(methodName, loc) {
 58 
 59   module("SC.Query.%@()".fmt(methodName), {
 60     setup: function() {
 61       window.TestRecord = TestRecord;
 62       window.TestRecord2 = TestRecord2;
 63     },
 64 
 65     teardown: function() {
 66       window.TestRecord = window.TestRecord2 = null; // cleanup
 67     }
 68   });
 69 
 70   function invokeWith() {
 71     return SC.Query[methodName].apply(SC.Query, arguments);
 72   }
 73 
 74   test("basic query with just record type", function() {
 75     var q, q1, q2, q3, q4;
 76 
 77     // local
 78     q = invokeWith(TestRecord);
 79     queryEquals(q, loc, TestRecord, null, 'first query');
 80 
 81     q1 = invokeWith(TestRecord);
 82     equals(q1, q, 'second call should return cached value');
 83 
 84     // using string for record type name should work
 85     q2 = invokeWith("TestRecord");
 86     equals(q2, q, 'queryFor with string should return cached value');
 87 
 88     // using an array of a single item should be treated as a single item
 89     q3 = invokeWith([TestRecord]);
 90     equals(q3, q, 'queryFor([TestRecord]) should return cached value');
 91 
 92     // ditto w/ strings
 93     q4 = invokeWith(['TestRecord']);
 94     equals(q4, q, 'queryFor(["TestRecord"]) with string should return cached value');
 95 
 96   });
 97 
 98   test("query with multiple recordtypes", function() {
 99 
100     var types = [TestRecord, TestRecord2],
101         q1, q2, a3, q4, q5, set;
102 
103     // create first query
104     q1 = invokeWith(types);
105     queryEquals(q1, loc, types, null, 'first query');
106 
107     // try again - should get cache
108     q2 = invokeWith(types);
109     equals(q2, q1, 'second queryFor call should return cached value');
110 
111     // try again - different order
112     q3 = invokeWith([TestRecord2, TestRecord]);
113     equals(q3, q1, 'queryFor with different order of record types should return same cached value');
114 
115     // try again - using a set
116     set = SC.Set.create().add(TestRecord).add(TestRecord2);
117     q4  = invokeWith(set);
118     equals(q4, q1, 'should return cached query even if using an enumerable for types');
119 
120     // try again using strings
121     q5 = invokeWith('TestRecord TestRecord2'.w());
122     equals(q5, q1, 'should return cached query even if string record names are used');
123   });
124 
125   test("query with record type and conditions", function() {
126 
127     var q1, q2, q3, q4, q5, q6, q7;
128 
129     q1 = invokeWith(TestRecord, 'foobar');
130     queryEquals(q1, loc, TestRecord, 'foobar', 'first query');
131 
132     q2 = invokeWith(TestRecord, 'foobar');
133     equals(q2, q1, 'second call to queryFor(TestRecord, foobar) should return cached instance');
134 
135     q3 = invokeWith(TestRecord2, 'foobar');
136     queryEquals(q3, loc, TestRecord2, 'foobar', 'query(TestRecord2, foobar)');
137     ok(q3 !== q1, 'different recordType same conditions should return new query');
138 
139     q4 = invokeWith(TestRecord, 'baz');
140     queryEquals(q4, loc, TestRecord, 'baz', 'query(TestRecord2, baz)');
141     ok(q4 !== q1, 'different conditions should return new query');
142 
143     q5 = invokeWith(TestRecord, 'baz');
144     equals(q5, q4, 'second call for different conditions should return cache');
145   });
146 
147   test("query with record types and conditions hash", function() {
148 
149     var q = invokeWith([TestRecord, TestRecord2], {});
150     queryEquals(q, loc, [TestRecord, TestRecord2], null, 'first query');
151 
152   });
153 
154   test("query with no record type and with conditions", function() {
155     var q1, q2;
156 
157     q1 = invokeWith(null, 'foobar');
158     queryEquals(q1, loc, SC.Record, 'foobar', 'first query');
159 
160     q2 = invokeWith(null, 'foobar');
161     equals(q2, q1, 'should return cached value');
162   });
163 
164   test("query with recordtype, conditions, and parameters hash", function() {
165     var opts  = { opt1: 'bar', opt2: 'baz' },
166         q1, q2;
167 
168     q1 = invokeWith(TestRecord, 'foo', opts);
169     queryEquals(q1, loc, TestRecord, 'foo', { parameters: opts }, 'first query');
170 
171     q2 = invokeWith(TestRecord, 'foo', opts);
172     ok(q1 !== q2, 'second call to queryFor with opts cannot be cached');
173     queryEquals(q1, loc, TestRecord, 'foo', { parameters: opts }, 'second query');
174   });
175 
176   test("query with recordtype, conditions, and parameters array", function() {
177     var opts  = ['foo', 'bar'],
178         q1, q2;
179 
180     q1 = invokeWith(TestRecord, 'foo', opts);
181     queryEquals(q1, loc, TestRecord, 'foo', { parameters: opts }, 'first query should include parameters prop');
182 
183     q2 = invokeWith(TestRecord, 'foo', opts);
184     ok(q1 !== q2, 'second call to queryFor with opts cannot be cached');
185     queryEquals(q1, loc, TestRecord, 'foo', { parameters: opts }, 'second query');
186   });
187 
188   test("passing query object", function() {
189 
190     var local = SC.Query.local(TestRecord),
191         remote = SC.Query.remote(TestRecord),
192         q;
193 
194     q = invokeWith(local);
195     if (loc === SC.Query.LOCAL) {
196       equals(q, local, 'invoking with local query should return same query');
197     } else {
198       ok(q !== local, 'invoke with local query should return new instance');
199     }
200     equals(q.get('location'), loc, 'query should have expected location');
201 
202     q = invokeWith(remote);
203     if (loc === SC.Query.REMOTE) {
204       equals(q, remote, 'invoking with remote query should return same query');
205     } else {
206       ok(q !== remote, 'invoke with remote query should return new instance');
207     }
208     equals(q.get('location'), loc, 'query should have expected location');
209   });
210 
211   test("no options (matches everything)", function() {
212     var q1, q2;
213 
214     q1 = invokeWith();
215     queryEquals(q1, loc, SC.Record, null, 'first query - matches everything');
216 
217     q2 = invokeWith();
218     equals(q2, q1, 'should return same cached query');
219 
220   });
221 
222 
223 }

225 performBasicTests('local', SC.Query.LOCAL);
