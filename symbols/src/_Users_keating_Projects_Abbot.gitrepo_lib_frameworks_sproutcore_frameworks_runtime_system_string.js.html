  8 /**
  9   @class
 10   
 11   Implements support methods useful when working with strings in SproutCore
 12   applications.
 13 */
 14 SC.String = /** @scope SC.String.prototype */ {
 15 
 16   // Interpolate string. looks for %@ or %@1; to control the order of params.
 17  18   /**
 19     Apply formatting options to the string.  This will look for occurrences
 20     of %@ in your string and substitute them with the arguments you pass into
 21     this method.  If you want to control the specific order of replacement,
 22     you can add a number after the key as well to indicate which argument
 23     you want to insert.
 24 
 25     Ordered insertions are most useful when building loc strings where values
 26     you need to insert may appear in different orders.
 27 
 28     Examples
 29     -----
 30 
 31         "Hello %@ %@".fmt('John', 'Doe') => "Hello John Doe"
 32         "Hello %@2, %@1".fmt('John', 'Doe') => "Hello Doe, John"
 33 
 34     @param {Object...} args optional arguments
 35     @returns {String} formatted string
 36   */
 37   fmt: function(str, formats) {
 38     // first, replace any ORDERED replacements.
 39     var idx  = 0; // the current index for non-numerical replacements
 40     return str.replace(/%@([0-9]+)?/g, function(s, argIndex) {
 41       argIndex = (argIndex) ? parseInt(argIndex,0) - 1 : idx++ ;
 42       s = formats[argIndex];
 43       return ((s === null) ? '(null)' : (s === undefined) ? '' : s).toString();
 44     }) ;
 45   },
 46   
 47   /**
 48     Splits the string into words, separated by spaces. Empty strings are
 49     removed from the results.
 50 
 51     @returns {Array} An array of non-empty strings
 52   */
 53   w: function(str) {
 54     var ary = [], ary2 = str.split(' '), len = ary2.length, string, idx=0;
 55     for (idx=0; idx<len; ++idx) {
 56       string = ary2[idx] ;
 57       if (string.length !== 0) ary.push(string) ; // skip empty strings
 58     }
 59     return ary ;
 60   }
 61 };
