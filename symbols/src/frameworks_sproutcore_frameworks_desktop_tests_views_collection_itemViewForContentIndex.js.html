 11 module("SC.CollectionView.itemViewForContentIndex", {
 12   setup: function() {
 13     content = "a b c".w().map(function(x) {
 14       return SC.Object.create({ title: x });
 15     });
 16 
 17     del = {
 18       fixture: {
 19         isEnabled: YES,
 20         isSelected: YES,
 21         outlineLevel: 3,
 22         disclosureState: SC.LEAF_NODE
 23       },
 24 
 25       contentIndexIsEnabled: function() {
 26         return this.fixture.isEnabled;
 27       },
 28 
 29       contentIndexIsSelected: function() {
 30         return this.fixture.isSelected;
 31       },
 32 
 33       contentIndexOutlineLevel: function() {
 34         return this.fixture.outlineLevel;
 35       },
 36 
 37       contentIndexDisclosureState: function() {
 38         return this.fixture.disclosureState ;
 39       }
 40     };
 41 
 42     // NOTE: delegate methods above are added here.
 43     SC.run(function () {
 44     view = SC.CollectionView.create(del, {
 45       content: content,
 46 
 47       layoutForContentIndex: function(contentIndex) {
 48         return this.fixtureLayout ;
 49       },
 50 
 51       fixtureLayout: { left: 0, right: 0, top:0, bottom: 0 },
 52 
 53       groupExampleView: SC.View.extend(), // custom for testing
 54 
 55       exampleView: SC.View.extend({
 56         isReusable: false
 57       }), // custom for testing
 58 
 59       testAsGroup: NO,
 60 
 61       contentIndexIsGroup: function() { return this.testAsGroup; },
 62 
 63       contentGroupIndexes: function() {
 64         if (this.testAsGroup) {
 65           return SC.IndexSet.create(0, this.get('length'));
 66         } else return null ;
 67       },
 68 
 69       fixtureNowShowing: SC.IndexSet.create(0,3),
 70 
 71       computeNowShowing: function() {
 72         return this.fixtureNowShowing;
 73       }
 74 
 75     });
 76     });
 77 
 78     // add in delegate mixin
 79     del = SC.mixin({}, SC.CollectionContent, del);
 80 
 81   }
 82 });

 84 function shouldMatchFixture(itemView, fixture) {
 85   var key;
 86   for(key in fixture) {
 87     if (!fixture.hasOwnProperty(key)) continue;
 88     equals(itemView.get(key), fixture[key], 'itemView.%@ should match delegate value'.fmt(key));
 89   }
 90 }

100   equals(itemView.get('content'), content.objectAt(1), 'itemView.content should be set to content item');

103   equals(itemView.get('displayDelegate'), view, 'itemView.displayDelegate should be collection view');
104   equals(itemView.get('parentView'), view, 'itemView.parentView should be collection view');
105 
106   // test data from delegate

150 // Tests support for the addition of designModes to SC.Pane and SC.View.  Since
151 // SC.CollectionView doesn't use child views and thus doesn't call
152 // SC.View:insertBefore, it needs to pass the designMode down to its item views
153 // itself.
154 test("set designMode on item views", function() {
155   var itemView,
156     updateDesignModeCount = 0;
157 
158   view.set('exampleView', SC.View.extend({
159     updateDesignMode: function () {
160       updateDesignModeCount++;
161 
162       sc_super();
163     }
164   }));
165 
166   // Initial designMode before creating the item view.
167   view.set('designMode', 'small');
168   itemView = view.itemViewForContentIndex(1);
169   equals(itemView.get('designMode'), 'small', "itemView.designMode should be set to match the current value of the collection");
170   equals(updateDesignModeCount, 1, "updateDesignMode should have been called once on the itemView");
171 
172   // Changes to designMode after creating the item view.
173   view.updateDesignMode('small', 'large');
174   equals(itemView.get('designMode'), 'large', "itemView.designMode should be set to match the current value of the collection");
175   equals(updateDesignModeCount, 2, "updateDesignMode should have been called once more on each itemView");
176 });

341 test("after making an item visible then invisible again", function() {
342 
343   view.isVisibleInWindow = YES ;
344 
345   // STEP 1- setup with some nowShowing
346   SC.run(function() {
347     view.set('fixtureNowShowing', SC.IndexSet.create(1));

349   });
350   equals(view.get('childViews').length, 1, 'precond - should have a child view');
351 
352   var itemView = view.itemViewForContentIndex(1);
353   equals(itemView.get('parentView'), view, 'itemView has parent view after some nowShowing');
354 
355   // STEP 2- setup with NONE visible
356   SC.run(function() {
357     view.set('fixtureNowShowing', SC.IndexSet.create());
358     view.notifyPropertyChange('nowShowing');
359   });
360   equals(view.get('childViews').length, 0, 'precond - should have no childview');
361 
362   itemView = view.itemViewForContentIndex(1);
363   equals(itemView.get('parentView'), view, 'itemView has parent view after none visible');
364 
365 
366   // STEP 3- go back to nowShowing
367   SC.run(function() {
368     view.set('fixtureNowShowing', SC.IndexSet.create(1));
369     view.notifyPropertyChange('nowShowing');
370   });
371   equals(view.get('childViews').length, 1, 'precond - should have a child view');
372 
373   itemView = view.itemViewForContentIndex(1);
