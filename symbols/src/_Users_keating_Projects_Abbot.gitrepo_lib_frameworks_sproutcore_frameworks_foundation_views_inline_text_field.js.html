 13 /**
 14   @class
 15   
 16   The inline text editor is used to display an editable area for controls 
 17   that are not always editable such as label views and source list views.
 18   
 19   You generally will not use the inline editor directly but instead will
 20   invoke beginEditing() and endEditing() on the views you are 
 21   editing. If you would like to use the inline editor for your own views, 
 22   you can do that also by using the editing API described here.
 23   
 24   ## Using the Inline Editor in Your Own Views
 25 
 26   To use the inlineEditor on a custom view you should mixin SC.InlineEditable on
 27   it. SC.InlineTextFieldView is the default editor so you do not need to do any
 28   other setup. The class methods beginEditing, commitEditing, and discardEditing
 29   still exist for backwards compatibility but should not be used on new views.
 30 
 31       MyProject.MyView = SC.View.extend(SC.InlineEditable, {
 32       });
 33   
 34   ### Starting the Editor
 35   
 36   The inline editor works by positioning itself over the top of your view 
 37   with the same offset, width, and font information.
 38 
 39   To start it simply call beginEditing on your view.
 40 
 41       myView.beginEditing();
 42   
 43   By default, if the inline editor is currently in use elsewhere, it will automatically
 44   close itself over there and begin editing for your view instead. This behavior
 45   is defined by the inlineEditorDelegate of your view, and can be changed by using
 46   one other than the default.
 47 
 48   ## Customizing the editor
 49 
 50   The editor has several parameters that can be used to customize it to your
 51   needs. These options should be set on the editor passed to your delegate's (or
 52   view's) inlineEditorWillBeginEditing method:
 53 
 54    - `exampleFrame` -- The editors initial frame in viewport coordinates.
 55    - `value` -- Initial value of the edit field.
 56    - `exampleElement` -- A DOM element to use when copying styles.
 57    - `multiline` -- If YES then the hitting return will add to the value instead
 58      of exiting the inline editor.
 59    - `commitOnBlur` -- If YES then blurring will commit the value, otherwise it
 60      will discard the current value.  Defaults to YES.
 61    - `validator` -- Validator to be attached to the field.
 62 
 63   For backwards compatibility, calling the class method beginEditing with an
 64   options hash will translate the values in the hash to the correct settings on
 65   the editor.
 66 
 67   ## Committing or Discarding Changes
 68   
 69   Normally the editor will automatically commit or discard its changes 
 70   whenever the user exits the edit mode by pressing enter, escape, or clicking
 71   elsewhere on the page. If you need to force the editor to end editing, you can
 72   do so by calling commitEditing() or discardEditing():
 73   
 74       myView.commitEditing();
 75       myView.discardEditing();
 76   
 77   Both methods will try to end the editing context and will call the 
 78   relevant delegate methods on the inlineEditorDelegate set on your view.
 79   
 80   Note that it is possible an editor may not be able to commit editing 
 81   changes because either the delegate disallowed it or because its validator
 82   failed.  In this case commitEditing() will return NO.  If you want to
 83   end editing anyway, you can discard the editing changes instead by calling
 84   discardEditing().  This method will generally succeed unless your delegate
 85   refuses it as well.
 86   
 87   @extends SC.TextFieldView
 88   @since SproutCore 1.0
 89 */

170   /*
171   * @private
172   * @method
173   *
174   * Scans the given element for styles and copies them into a style element in
175   * the head. This allows the styles to be overriden by css matching classNames
176   * on the editor.
177   *
178   * @params {element} the dom element to copy
179   */
180 	updateStyle: function(exampleElement) {
181     if(exampleElement.length) exampleElement = exampleElement[0];
182 
183     // the styles are placed into a style element so that they can be overridden
184     // by your css based on the editor className

213   /*
214   * @method
215   *
216   * Positions the editor over the passed view.
217   *
218   * If you want to tweak the positioning of the editor, you may pass a custom
219   * frame for it to position itself on.
220   *
221   * Additionally, if your view is a member of a collectionView, the isCollection
222   * flag should be set to YES.
223   *
224   * @param {SC.View} the view to be positioned over
225   * @param {Hash} optional custom frame
226   * @param {Boolean} if the view is a member of a collection
227   */
228 	positionOverTargetView: function(target, isCollection, pane, frame, elem) {
229     if(!pane) pane = target.get('pane');
230 
231     if(!elem) elem = target.$()[0];
232 
233     // if we weren't given a frame, build one from the target
234     if(!frame) {
235       var tempFrame = target.get('frame');
236 
237       frame = SC.offset(elem);
238 
239       frame.height = tempFrame.height;
240       frame.width = tempFrame.width;
241     }
242 
243     var layout={},
244 		paneElem = pane.$()[0],
245 		tarLayout = target.get('layout');
246 
247     layout.height = frame.height;
248     layout.width = frame.width;
249 
250     if (isCollection && tarLayout.left) {
251       layout.left=frame.x-tarLayout.left-paneElem.offsetLeft-1;
252       if(SC.browser.msie==7) layout.left--;
253     } else {
254       layout.left=frame.x-paneElem.offsetLeft-1;
255       if(SC.browser.msie==7) layout.left--;
256     }
257 
258     if (isCollection && tarLayout.top) {
259       layout.top=frame.y-tarLayout.top-paneElem.offsetTop;
260       if(SC.browser.msie==7) layout.top=layout.top-2;
261     } else {
262       layout.top=frame.y-paneElem.offsetTop;
263       if(SC.browser.msie==7) layout.top=layout.top-2;
264     }
265 
266     this.set('layout', layout);
267 	},

494 /*
495 * These class methods allow you to manage an editor without implementing
496 * SC.Editable. They exist for backwards compatibility and should no longer be
497 * used.
498 */
499 SC.mixin(SC.InlineTextFieldView, {
500   inlineEditorDelegate: SC.InlineTextFieldDelegate,
501 
502   label: null,
503   editor: null,
504 
505   /*
506   * @method
507   *
508   * This method creates a singleton editor editing your view. Delegate methods
509   * will be called on your view as normal.
510   *
511   * To trigger compatibility mode, call this method with a hash of options. The
512   * required options are:
513   *
514         SC.InlineTextFieldView.beginEditing({
515           delegate: myView,
516           frame: myView.get('frame'),
517           exampleElement: myView.$()
518         }) ;
519   *
520      - `delegate` -- The view that should be notified of events on the editor.
521      - `frame` -- The frame of the view that you want the editor to position itself over.
522      - `exampleElement` -- A DOM element to use when copying styles.
523 
524   * Other options may be passed to further customize the editor:
525   *
526      - `multiline` -- If YES then the hitting return will add to the value instead of exiting the inline editor.
527      - `commitOnBlur` -- If YES then blurring will commit the value, otherwise it will discard the current value.  Defaults to YES.
528      - `validator` -- Validator to be attached to the field.
529 
530     The editor expects your source view to implement the InlineTextFieldViewDelegate protocol.
531 
532     @params {Object} a hash of options or the view to edit
533     @returns {Boolean} whether editing began successfully
534   */
535   beginEditing: function(label) {
536     var del, editor, options, value, labelProxy;
537 
538     // for backwards compatibility, we allow you to pass an options hash with options that will be set on the editor
539     if(SC.typeOf(label) === SC.T_HASH) {
540       options = label;
541 
542       if(!options.delegate || !options.exampleElement || !options.frame) {
543         SC.error("Delegate, exampleElement, and frame options are required.");
544         return NO;
545       }
546 
547       label = options.delegate;
548       value = options.value;
549     }
550 
551     else {
552       value = label.get('value');
553     }
554 
555     labelProxy = SC.beget(label);
556 
557     // these functions may have side effects, so they need to have their
558     // this reference assigned to the original object before proxying
559     labelProxy.mixin({
560       inlineEditorWillBeginEditing: function() {
561         if(label.inlineEditorWillBeginEditing) label.inlineEditorWillBeginEditing.apply(label, arguments);
562       },
563 
564       inlineEditorDidBeginEditing: function() {
565         if(label.inlineEditorDidBeginEditing) label.inlineEditorDidBeginEditing.apply(label, arguments);
566       },
567 
568       inlineEditorWillCommitEditing: function(editor, value, editable) {
569         if(label.inlineEditorWillCommitEditing) label.inlineEditorWillCommitEditing(editor, value, editable);
570         if(label.inlineEditorWillEndEditing) label.inlineEditorWillEndEditing(editor, value);
571       },
572 
573       inlineEditorDidCommitEditing: function(editor, value, editable) {
574         if(label.inlineEditorDidCommitEditing) label.inlineEditorDidCommitEditing(editor, value, editable);
575         if(label.inlineEditorDidEndEditing) label.inlineEditorDidEndEditing(editor, value);
576 
577         SC.InlineTextFieldView._endEditing();
578       },
579 
580       inlineEditorWillDiscardEditing: function(editor, editable) {
581         if(label.inlineEditorWillDiscardEditing) label.inlineEditorWillDiscardEditing(editor, editable);
582         if(label.inlineEditorWillEndEditing) label.inlineEditorWillEndEditing(editor, this.get('value'));
583       },
584 
585       inlineEditorDidDiscardEditing: function(editor, editable) {
586         if(label.inlineEditorDidDiscardEditing) label.inlineEditorDidDiscardEditing(editor, editable);
587         if(label.inlineEditorDidEndEditing) label.inlineEditorDidEndEditing(editor, this.get('value'));
588 
589         SC.InlineTextFieldView._endEditing();
590       }
591     });
592 
593     if(label.inlineEditorShouldBeginEditing && !label.inlineEditorShouldBeginEditing(label, value)) return NO;
594 
595     this.editor = editor = this.inlineEditorDelegate.acquireEditor(label);
596 
597     editor.set('value', value);
598 
599     if(options) {
600       editor.set('exampleElement', options.exampleElement);
601       editor.set('exampleFrame', options.frame);
602       editor.set('multiline', options.multiline);
603       editor.set('escapeHTML', options.escapeHTML);
604       editor.set('isCollection', options.isCollection);
605       editor.set('commitOnBLur', options.commitOnBlur);
606       editor.set('validator', options.validator);
607     }
608 
609     if(editor) return editor.beginEditing(labelProxy);
610     else return NO;
611   },
612 
613   /*
614   * @method
615   *
616   * Ends editing on the current editor and saves the value back to the
617   * view being edited.
618   *
619   * @returns {Boolean} whether the editor was allowed to commit successfully
620   */
621   commitEditing: function() {
622     return this.inlineEditorDelegate.editor ? this.inlineEditorDelegate.editor.commitEditing() : NO;
623   },
624 
625   /*
626   * @method
627   *
628   * Ends editing on the current editor without saving the value.
629   *
630   * @returns {Boolean} whether the editor was allowed to discard successfully
631   */
632   discardEditing: function() {
633     return this.inlineEditorDelegate.editor ? this.inlineEditorDelegate.editor.discardEditing() : NO;
634   },
635 
636   /*
637   * @private
638   * @method
639   *
640   * Cleans up the current editor and editing context.
641   */
642   _endEditing: function() {
643     this.inlineEditorDelegate.releaseEditor(this.editor);
644 
645     this.editor = null;
646   }
