 10 /**
 11   @class
 12 
 13   SegmentedView is a special type of button that can display multiple
 14   segments.  Each segment has a value assigned to it.  When the user clicks
 15   on the segment, the value of that segment will become the new value of
 16   the control.
 17 
 18   You can also optionally configure a target/action that will fire whenever
 19   the user clicks on an item.  This will give your code an opportunity to take
 20   some action depending on the new value.  (of course, you can always bind to
 21   the value as well, which is generally the preferred approach.)
 22 
 23   # Defining Your Segments
 24 
 25   You define your segments by providing a items array, much like you provide
 26   to a RadioView.  Your items array can be as simple as an array of strings
 27   or as complex as full model objects.  Based on how you configure your
 28   itemKey properties, the segmented view will read the properties it needs
 29   from the array and construct the button.
 30 
 31   You can define the following properties on objects you pass in:
 32 
 33     - *itemTitleKey* - the title of the button
 34     - *itemValueKey* - the value of the button
 35     - *itemWidthKey* - the preferred width. if omitted, it autodetects
 36     - *itemIconKey*  - an icon
 37     - *itemActionKey* - an optional action to fire when pressed
 38     - *itemTargetKey* - an optional target for the action
 39     - *itemLayerIdKey* - an optional target for the action 
 40     - *segmentViewClass* - class to be used for creating segments
 41 
 42   @extends SC.View
 43   @extends SC.Control
 44   @since SproutCore 1.0
 45 */

331   /** @private
332     Called whenever the number of items changes.  This method populates SegmentedView's childViews, taking
333     care to re-use existing childViews if possible.
334   */
335   itemsDidChange: function() {
336     var items = this.get('items') || [],
337         item,
338         localItem,                        // Used to avoid altering the original items
339         childViews = this.get('childViews'),
340         childView,
341         overflowView = childViews.lastObject(),
342         value = this.get('value'),        // The value can change if items that were once selected are removed
343         isSelected,
344         itemKeys = this.get('itemKeys'),
345         itemKey,
346         viewKeys = this.get('viewKeys'),
347         viewKey,
348         segmentViewClass = this.get('segmentViewClass'),
349         i, j;
350 
351     // Update childViews
352     if (childViews.get('length') - 1 > items.get('length')) {   // We've lost segments (ie. childViews)
353 
354       // Remove unneeded segments from the end back
355       for (i = childViews.get('length') - 2; i >= items.get('length'); i--) {
356         childView = childViews.objectAt(i);
357 
358         // If a selected childView has been removed then update our value
359         if (SC.isArray(value)) {
360           value.removeObject(childView.get('value'));
361         } else if (value === childView.get('value')) {
362           value = null;
363         }
364 
365         this.removeChild(childView);
366       }
367 
368       // Update our value which may have changed
369       this.set('value', value);
370 
371     } else if (childViews.get('length') - 1 < items.get('length')) {  // We've gained segments
372 
373       // Create the new segments
374       for (i = childViews.get('length') - 1; i < items.get('length'); i++) {
375 
376         // We create a default SC.ButtonView-like object for each segment
377         childView = segmentViewClass.create({
378           controlSize: this.get('controlSize'),
379           localize: this.get('localize'),
380           layoutDirection: this.get('layoutDirection')
381         });
382 
383         // Attach the child
384         this.insertBefore(childView, overflowView);
385       }
386     }
387 
388     // Because the items array can be altered with insertAt or removeAt, we can't be sure that the items
389     // continue to match 1-to-1 the existing views, so once we have the correct number of childViews,
390     // simply update them all
391     childViews = this.get('childViews');
392 
393     for (i = 0; i < items.get('length'); i++) {
394       localItem = items.objectAt(i);
395       childView = childViews.objectAt(i);
396 
397       // Skip null/undefined items (but don't skip empty strings)
398       if (SC.none(localItem)) continue;
399 
400       // Normalize the item (may be a String, Object or SC.Object)
401       if (SC.typeOf(localItem) === SC.T_STRING) {
402 
403         localItem = SC.Object.create({
404           'title': localItem.humanize().titleize(),
405           'value': localItem
406         });
407 
408         // Update our keys accordingly
409         this.set('itemTitleKey', 'title');
410         this.set('itemValueKey', 'value');
411       } else if (SC.typeOf(localItem) === SC.T_HASH) {
412 
413         localItem = SC.Object.create(localItem);
414       } else if (localItem instanceof SC.Object)  {
415 
416         // We don't need to make any changes to SC.Object items, but we can observe them
417         for (j = itemKeys.get('length') - 1; j >= 0; j--) {
418           itemKey = this.get(itemKeys.objectAt(j));
419 
420           if (itemKey) {
421             localItem.removeObserver(itemKey, this, this.itemContentDidChange);
422             localItem.addObserver(itemKey, this, this.itemContentDidChange, i);
423           }
424         }
425       } else {
426         SC.Logger.error('SC.SegmentedView items may be Strings, Objects (ie. Hashes) or SC.Objects only');
427       }
428 
429       // Determine whether this segment is selected based on the view's existing value(s)
430       isSelected = NO;
431       if (SC.isArray(value) ? value.indexOf(localItem.get(this.get('itemValueKey'))) >= 0 : value === localItem.get(this.get('itemValueKey'))) {
432         isSelected = YES;
433       }
434       childView.set('isSelected', isSelected);
435 
436       // Assign segment specific properties based on position
437       childView.set('index', i);
438       childView.set('isFirstSegment', i === 0);
439       childView.set('isMiddleSegment',  i < items.get('length') - 1 && i > 0);
440       childView.set('isLastSegment', i === items.get('length') - 1);
441 
442       // Be sure to update the view's properties for the (possibly new) matched item
443       childView.updateItem(this, localItem);
444     }
445 
446     // Force a segment remeasure to check overflow
447     this.invokeLast(this.remeasure);
448   }.observes('*items.[]'),

490   /** @private
491     Calling this method forces the segments to be remeasured and will also adjust the
492     segments for overflow if necessary.
493   */
494   remeasure: function() {
495     var renderDelegate = this.get('renderDelegate'),
496         childViews = this.get('childViews'),
497         overflowView;
498 
499     if (this.get('isVisibleInWindow')) {
500       // Make all the views visible so that they can be measured
501       overflowView = childViews.lastObject();
502       overflowView.set('isVisible', YES);
503 
504       for (var i = childViews.get('length') - 1; i >= 0; i--){
505         childViews.objectAt(i).set('isVisible', YES);
506       }
507 
508 
509       this.cachedDims = this.segmentDimensions();
510       this.cachedOverflowDim = this.overflowSegmentDim();
511 
512       this.adjustOverflow();
513     }
514   },

516   /** @private
517     This method is called to adjust the segment views to see if we need to handle for overflow.
518    */
519   adjustOverflow: function() {
520     var childViews = this.get('childViews'),
521         childView,
522         value = this.get('value'),
523         overflowView = childViews.lastObject(),
524         isHorizontal = this.get('layoutDirection') === SC.LAYOUT_HORIZONTAL,
525         visibleDim = isHorizontal ? this.$().width() : this.$().height(),  // The inner width/height of the div
526         curElementsDim = 0,
527         dimToFit,
528         length, i;
529 
530     // This variable is useful to optimize when we are overflowing
531     this.isOverflowing = NO;
532     overflowView.set('isSelected', NO);
533 
534     // Clear out the overflow items (these are the items not currently visible)
535     this.overflowItems = [];
536 
537     length = this.cachedDims.length;
538     for (i=0; i < length; i++) {
539       childView = childViews.objectAt(i);
540       curElementsDim += this.cachedDims[i];
541 
542       // check for an overflow (leave room for the overflow segment except for with the last segment)
543       dimToFit = (i === length - 1) ? curElementsDim : curElementsDim + this.cachedOverflowDim;
544 
545       if (dimToFit > visibleDim) {
546         // Add the localItem to the overflowItems
547         this.overflowItems.pushObject(childView.get('localItem'));
548 
549         // Record that we're now overflowing
550         this.isOverflowing = YES;
551 
552         childView.set('isVisible', NO);
553 
554         // If the first item is already overflowed, make the overflowView first segment
555         if (i === 0) overflowView.set('isFirstSegment', YES);
556 
557         // If the overflowed segment was selected, show the overflowView as selected instead
558         if (SC.isArray(value) ? value.indexOf(childView.get('value')) >= 0 : value === childView.get('value')) {
559           overflowView.set('isSelected', YES);
560         }
561       } else {
562         childView.set('isVisible', YES);
563 
564         // If the first item is not overflowed, don't make the overflowView first segment
565         if (i === 0) overflowView.set('isFirstSegment', NO);
566       }
567     }
568 
569     // Show/hide the overflow view if we have overflowed
570     if (this.isOverflowing) overflowView.set('isVisible', YES);
571     else overflowView.set('isVisible', NO);
572 
573     // Store the minimum dimension (height/width) before overflow
574     this.cachedMinimumDim = curElementsDim + this.cachedOverflowDim;
575   },
576 
577   /**
578     Return the dimensions (either heights or widths depending on the layout direction) of the DOM 
579     elements of the segments.  This will be measured by the view to determine which segments should
580     be overflowed.
581 
582     It ignores the last segment (the overflow segment).
583   */
584   segmentDimensions: function() {
585     var cv = this.get('childViews'),
586         v, f,
587         dims = [],
588         isHorizontal = this.get('layoutDirection') === SC.LAYOUT_HORIZONTAL;
589 
590     for (var i = 0, length = cv.length; i < length - 1; i++) {
591       v = cv[i];
592       f = v.get('frame');
593       dims[i] = isHorizontal ? f.width : f.height;
594     }
595 
596     return dims;
597   },
598 
599   /**
600     Return the dimension (height or width depending on the layout direction) over the overflow segment.
601   */
602   overflowSegmentDim: function() {
603     var cv = this.get('childViews'),
604         v, f, 
605         isHorizontal = this.get('layoutDirection') === SC.LAYOUT_HORIZONTAL;
606 
607     v = cv.length && cv[cv.length - 1];
608     if (v) {
609       f = v.get('frame');
610       return isHorizontal ? f.width : f.height;
611     }
612 
613     return 0;
614   },

616   /**
617     Return the index of the segment view that is the target of the mouse click.
618   */
619   indexForClientPosition: function(x, y) {
620     var cv = this.get('childViews'),
621         length, i,
622         v, rect,
623         point;
624 
625     point = {x: x, y: y};
626     for (i = 0, length = cv.length; i < length; i++) {
627       v = cv[i];
628       
629       rect = v.get('layer').getBoundingClientRect();
630       rect = { 
631         x: rect.left, 
632         y: rect.top, 
633         width: (rect.right-rect.left), 
634         height: (rect.bottom - rect.top) 
635       };
636       
637       // Return the index early if found
638       if (SC.pointInRect(point, rect)) return i;
639     }
640 
641     // Default not found
642     return -1;
643   },

674   /** @private */
675   keyDown: function(evt) {
676     var childViews,
677         childView,
678         i, length,
679         value, isArray;
680 
681     // handle tab key
682     if (evt.which === 9 || evt.keyCode === 9) {
683       var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
684       if(view) view.becomeFirstResponder();
685       else evt.allowDefault();
686       return YES ; // handled
687     }
688 
689     // handle arrow keys
690     if (!this.get('allowsMultipleSelection')) {
691       childViews = this.get('childViews');
692 
693       length = childViews.get('length');
694       value = this.get('value');
695       isArray = SC.isArray(value);
696 
697       // Select from the left to the right
698       if (evt.which === 39 || evt.which === 40) {
699 
700         if (value) {
701           for(i = 0; i < length - 2; i++){
702             childView = childViews.objectAt(i);
703             if ( isArray ? (value.indexOf(childView.get('value'))>=0) : (childView.get('value')===value)){
704               this.triggerItemAtIndex(i + 1);
705             }
706           }
707         } else {
708           this.triggerItemAtIndex(0);
709         }
710         return YES ; // handled
711 
712       // Select from the right to the left
713       } else if (evt.which === 37 || evt.which === 38) {
714 
715         if (value) {
716           for(i = 1; i < length - 1; i++) {
717             childView = childViews.objectAt(i);
718             if ( isArray ? (value.indexOf(childView.get('value'))>=0) : (childView.get('value')===value)){
719               this.triggerItemAtIndex(i - 1);
720             }
721           }
722         } else {
723           this.triggerItemAtIndex(length - 2);
724         }
725 
726         return YES; // handled
727       }
728     }
729 
730     return NO;
731   },

733   /** @private */
734   mouseDown: function(evt) {
735     var childViews = this.get('childViews'),
736         childView,
737         overflowIndex = childViews.get('length') - 1,
738         index;
739 
740     if (!this.get('isEnabled')) return YES; // nothing to do
741 
742     index = this.displayItemIndexForEvent(evt);
743 
744     if (index >= 0) {
745       childView = childViews.objectAt(index);
746       childView.set('isActive', YES);
747       this.activeChildView = childView;
748 
749       // if mouse was pressed on the overflow segment, popup the menu
750       if (index === overflowIndex) this.showOverflowMenu();
751       else this._isMouseDown = YES;
752     }
753 
754     return YES;
755   },
756 
757   /** @private */
758   mouseUp: function(evt) {
759     var activeChildView,
760         index;
761 
762     index = this.displayItemIndexForEvent(evt);
763 
764     if (this._isMouseDown && (index >= 0)) {
765 
766       this.triggerItemAtIndex(index);
767 
768       // Clean up
769       activeChildView = this.activeChildView;
770       activeChildView.set('isActive', NO);
771       this.activeChildView = null;
772 
773       this._isMouseDown = NO;
774     }
775 
776     return YES;
777   },

779   /** @private */
780   mouseMoved: function(evt) {
781     var childViews = this.get('childViews'),
782         overflowIndex = childViews.get('length') - 1,
783         activeChildView,
784         childView,
785         index;
786 
787     if (this._isMouseDown) {
788       // Update the last segment

968   /** @private
969     Simulates the user clicking on the segment at the specified index. This
970     will update the value if possible and fire the action.
971   */
972   triggerItemAtIndex: function(index) {
973     var childViews = this.get('childViews'),
974         childView,
975         sel, value, val, empty, mult;
976 
977     childView = childViews.objectAt(index);
978 
979     if (!childView.get('isEnabled')) return this; // nothing to do!
980 
981     empty = this.get('allowsEmptySelection');
982     mult = this.get('allowsMultipleSelection');
983 
984     // get new value... bail if not enabled. Also save original for later.
985     sel = childView.get('value');
986     value = val = this.get('value') ;
987 
988     if (SC.empty(value)) {
989       value = [];
990     } else if (!SC.isArray(value)) {
991       value = [value]; // force to array
992     }
993 
994     // if we do not allow multiple selection, either replace the current
995     // selection or deselect it
996     if (!mult) {
997       // if we allow empty selection and the current value is the same as
998       // the selected value, then deselect it.
999       if (empty && (value.get('length')===1) && (value.objectAt(0)===sel)) {
1000         value = [];
1001 
1002       // otherwise, simply replace the value.
1003       } else value = [sel] ;
1004 
1005     // if we do allow multiple selection, then add or remove item to the array.
1006     } else {
1007       if (value.indexOf(sel) >= 0) {
1008         if (value.get('length')>1 || (value.objectAt(0)!==sel) || empty) {
1009           value = value.without(sel);
1010         }
1011       } else value = value.concat([sel]) ;
1012     }
1013 
1014     // normalize back to non-array form
1015     switch(value.get('length')) {
1016       case 0:
1017         value = null;
1018         break;
1019       case 1:
1020         value = value.objectAt(0);
1021         break;
1022       default:
1023         break;
1024     }
1025 
1026     // also, trigger target if needed.
1027     var actionKey = this.get('itemActionKey'),
1028         targetKey = this.get('itemTargetKey'),
1029         action, target = null,
1030         resp = this.getPath('pane.rootResponder'),
1031         item;
1032 
1033     if (actionKey && (item = this.get('items').objectAt(index))) {
1034       // get the source item from the item array.  use the index stored...
1035       action = item.get ? item.get(actionKey) : item[actionKey];
1036       if (targetKey) {
1037         target = item.get ? item.get(targetKey) : item[targetKey];
1038       }
1039       if (resp) resp.sendAction(action, target, this, this.get('pane'),value);
1040     }
1041 
1042     if(val !== undefined && (!action || this.get('selectSegmentWhenTriggeringAction'))) {
1043       this.set('value', value);
1044     }
1045 
1046     // if an action/target is defined on self use that also
1047     action =this.get('action');
1048     if (action && resp) {
1049       resp.sendAction(action, this.get('target'), this, this.get('pane'),value);
1050     }
1051   },

1071   showOverflowMenu: function() {

1082

1113       modalPaneDidClick: function() {

1115

1118

1121       }

1124     var layer = this.get('layer');
1125     var overflowElement = layer.childNodes[layer.childNodes.length - 1];
1126     menu.popup(overflowElement);
1127
