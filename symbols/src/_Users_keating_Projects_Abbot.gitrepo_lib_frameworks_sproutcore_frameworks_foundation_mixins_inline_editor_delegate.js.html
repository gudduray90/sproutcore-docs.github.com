  8 /**
  9   @namespace
 10   
 11   This delegate is consulted by view implementing SC.InlineEditable and
 12   SC.InlineEditor and controls the lifecycle of the editor as well as begin
 13   notified of events in the editing process.
 14   
 15   By default it edits an SC.LabelView using an SC.InlineTextFieldView.
 16 
 17   All methods will be attempted to be called on the inlineEditorDelegate of the
 18   inlineEditor or inlineEditable first and then the target view if it didn't exist
 19   on the delegate. This allows you to implement default delegate handlers on your
 20   editable view.
 21   
 22  23   @since SproutCore 1.0
 24 */
 25 SC.InlineEditorDelegate = {
 26 
 27   // ..........................................................
 28   // Required Functions
 29   // 
 30 
 31   /**
 32     Acquires an editor for the view. This may simply create one and return it,
 33     or you may implement more complex lifecycle management like pooling of
 34     editors.
 35 
 36     May return null if for some reason an editor could not be required.
 37 
 38     You must override this function.
 39 
 40     @params {SC.InlineEditable} editable the view that is begin edited
 41     @returns {SC.InlineEditor} an editor for the view
 42   */
 43   acquireEditor:function(editable) {},
 44 
 45   /**
 46     Releases an editor. This may simply remove it from its parent and dispose of
 47     it, or you may implement more complex lifecycle management like pooling of
 48     editors.
 49 
 50     You must override this function.
 51 
 52     @params {SC.InlineEditor} editor the editor being released
 53     @returns {Boolean} YES if it was successfully released
 54   */
 55   releaseEditor:function(editor) {},
 56 
 57 
 58   // ..........................................................
 59   // Optional Functions
 60   // 
 61 
 62   /**
 63     Determines if the view should be allowed to begin editing and returns YES if
 64     so. Isn't passed the editor because it hasn't been created yet. If this
 65     method is not defined the editor will assume it is always allowed to begin
 66     editing.
 67 
 68     @params {SC.InlineEditable} editable the view that is attempting to begin editing
 69     @params {Object} value the current value of the view
 70     @returns {Boolean} YES if the view is allowed to edit
 71   */
 72   inlineEditorShouldBeginEditing: function(editable, value) {},
 73 
 74   /**
 75     Notifies the delegate that the view was allowed to begin editing and the
 76     editor has been acquired, but hasn't actually done any setup. Most views will
 77     set their current value as the starting value of the editor here, and
 78     depending on the editor other configuration options may be available.
 79 
 80     Since the editor's value hasn't been configured with, the value passed here will be
 81     the default value of the editor.
 82 
 83     @params {SC.InlineEditable} the view being edited
 84     @params {SC.InlineEditor} the editor for the view
 85     @params {Object} the value the editor will start with
 86   */
 87   inlineEditorWillBeginEditing:function(editor, value, editable) {},
 88 
 89   /**
 90     Notifies the delegate that the editor has finished setting up itself and is
 91     now editing.
 92 
 93     @params {SC.InlineEditable} the view being edited
 94     @params {SC.InlineEditor} the editor for the view
 95     @params {Object} the value the editor started with
 96   */
 97   inlineEditorDidBeginEditing:function(editor, value, editable) {},
 98 
 99   /**
100     Determines if the editor is allowed to end editing and store its value back
101     to the view being edited. If this method is not defined the editor will
102     assume it is always allowed to commit.
103 
104     @params {SC.InlineEditable} the view being edited
105     @params {SC.InlineEditor} the editor for the view
106     @params {Object} the value the editor is attempting to commit
107   */
108   inlineEditorShouldCommitEditing:function(editor, value, editable) {},
109 
110   /**
111     Notifies the delegate that the editor was allowed to commit and is going to
112     commit but hasn't actually performed any cleanup yet.
113 
114     @params {SC.InlineEditable} the view being edited
115     @params {SC.InlineEditor} the editor for the view
116     @params {Object} the value the editor ended with
117   */
118   inlineEditorWillCommitEditing:function(editor, value, editable) {},
119 
120   /**
121     Notifies the delegate that the editor was allowed to commit and finished
122     performing any cleanup necessary. This is where you should save the final
123     value back to your view after performing any necessary transforms to it.
124 
125     @params {SC.InlineEditable} the view being edited
126     @params {SC.InlineEditor} the editor for the view
127     @params {Object} the value the editor ended with
128   */
129   inlineEditorDidCommitEditing:function(editor, value, editable) {},
130 
131   /**
132     Determines if the editor is allowed to discard its current value and end
133     editing. If this method is undefined the editor will assume it is always
134     allowed to discard.
135 
136     @params {SC.InlineEditable} the view being edited
137     @params {SC.InlineEditor} the editor for the view
138   */
139   inlineEditorShouldDiscardEditing:function(editor, editable) {},
140 
141   /**
142     Notifies the delegate that the view was allowed to discard editing but
143     hasn't performed any cleanup yet.
144 
145     @params {SC.InlineEditable} the view being edited
146     @params {SC.InlineEditor} the editor for the view
147   */
148   inlineEditorWillDiscardEditing:function(editor, editable) {},
149 
150   /**
151     Notifies the delegate that the editor has finished cleaning up after
152     discarding editing.
153 
154     @params {SC.InlineEditable} the view being edited
155     @params {SC.InlineEditor} the editor for the view
156   */
157   inlineEditorDidDiscardEditing:function(editor, editable) {},
158 
159 
160   // ..........................................................
161   // Backwards Compatibility
162   // 
163 
164   /**
165     @private
166 
167     Notifies the delegate that the editor will end editing but hasn't cleaned up
168     yet. This can be caused by both commit or discard. If it was a discard, the
169     value will be the same as the current value of the editable view. Otherwise,
170     it was a commit and the value will be the value of the editor.
171 
172     This method is for backwards compatibility and should not be used.
173 
174     @params {SC.InlineEditable} the view being edited
175     @params {SC.InlineEditor} the editor for the view
176     @params {Object} the final value of the edit
177   */
178   inlineEditorWillEndEditing: function(editor, value, editable) {},
179 
180   /**
181     @private
182 
183     Notifies the delegate that the editor has cleaned up after editing. This can
184     be caused by both commit or discard. If it was a discard, the value will be
185     the same as the current value of the editable view. Otherwise, it was a
186 187     commit and the value will be the value of the editor.
188 
189     This method is for backwards compatibility and should not be used.
190 
191     @params {SC.InlineEditable} the view being edited
192     @params {SC.InlineEditor} the editor for the view
193     @params {Object} the final value of the edit
194   */
195   inlineEditorDidEndEditing: function(editor, value, editable) {}
196 };
