 13 var initModels = function () {
 14   NestedRecord.Directory = SC.Record.extend({
 15     /** Child Record Namespace */
 16     nestedRecordNamespace: NestedRecord,
 17     name: SC.Record.attr(String),
 18     contents: SC.Record.toMany('SC.Record', { isNested: true })
 19   });
 20 
 21   NestedRecord.File = SC.Record.extend({
 22     name: SC.Record.attr(String)
 23   });
 24 
 25 };

 30 module("Data Store Tests for Nested Records", {
 31 
 32   setup: function () {
 33     SC.RunLoop.begin();
 34     NestedRecord = SC.Object.create({
 35       store: SC.Store.create()
 36     });
 37     store = NestedRecord.store;
 38     initModels();
 39 
 40     storeKeys = store.loadRecords([NestedRecord.Directory, NestedRecord.File], [
 41       {
 42         type: 'Directory',
 43         name: 'Dir 1',
 44         guid: 1,
 45         contents: [
 46           {
 47             type: 'Directory',

 49             guid: 2,
 50             contents: [
 51               {
 52                 type: 'File',
 53                 guid: 3,
 54                 name: 'File 1'
 55               },
 56               {
 57                 type: 'File',
 58                 guid: 4,
 59                 name: 'File 2'
 60               }
 61             ]
 62           }
 63         ]
 64       },
 65       {
 66         type: 'File',
 67         id: 5,
 68         name: 'File 3'
 69       }
 70     ]);
 71     SC.RunLoop.end();
 72   },
 73 
 74   teardown: function () {
 75     delete NestedRecord.Directory;
 76     delete NestedRecord.File;
 77     NestedRecord = null;
 78     store = null;
 79   }

 98 test("Test that reset() clears out the store for re-use", function () {
 99   var isEmptyObject = function ( obj ) {
100     var name;
101 
102     for (name in obj) {
103       return false;
104     }
105     return true;
106   };
107 
108   var parent, child;
109   SC.run(function() { parent = store.materializeRecord(storeKeys[0]); });
110   SC.run(function() { child = parent.get('contents').firstObject(); });
111 
112   ok(!isEmptyObject(store.parentRecords), "We expect there to be values in store.parentRecords");
113   ok(!isEmptyObject(store.childRecords), "We expect there to be values in store.childRecords");
114 
115   store.reset();
116 
117   ok(isEmptyObject(store.parentRecords), "We expect there to no longer be any values in store.parentRecords");
118   ok(isEmptyObject(store.childRecords), "We expect there to no longer be any values in store.childRecords");
119 });

178 test("Use in Nested Store", function () {
179   var nstore, dir, c, file,
180       pk, id, nFile, nDir;
181 
182   // First, find the first file
183   SC.run(function() { dir = store.find(NestedRecord.Directory, 1); });
184   ok(dir, "Directory id:1 exists");
185   equals(dir.get('name'), 'Dir 1', "Directory id:1 has a name of 'Dir 1'");
186   c = dir.get('contents');
187   ok(c, "Content of Directory id:1 exists");
188   SC.run(function() {dir = c.objectAt(0);});
189   ok(dir, "Directory id:2 exists");
190   equals(dir.get('name'), 'Dir 2', "Directory id:2 has a name of 'Dir 2'");
191   c = dir.get('contents');
192   ok(c, "Content of Directory id:2 exists");
193   SC.run(function() {file = c.objectAt(0);});
194   ok(file, "File id:1 exists");
195   equals(file.get('name'), 'File 1', "File id:1 has a name of 'File 1'");
196 
197   // Second, create nested store
198   nstore = store.chain();
199   SC.RunLoop.begin();
200   pk = file.get('primaryKey');
201   id = file.get(pk);
202   nFile = nstore.find(NestedRecord.File, id);
203   SC.RunLoop.end();
204   ok(nFile, "Nested > File id:1 exists");
205   equals(nFile.get('name'), 'File 1', "Nested > File id:1 has a name of 'File 1'");
206 
207   // Third, change the name of the nested store and see what happens
208   SC.run(function(){nFile.set('name', 'Change Name');});
209   equals(nFile.get('name'), 'Change Name', "Nested > File id:1 has changed the name to 'Changed Name'");
210   equals(file.get('name'), 'File 1', "Base > File id:1 still has the name of 'File 1'");
211   nDir = nstore.find(NestedRecord.Directory, 1);
212 
213   // Fourth, commit the changes
214   SC.run(function(){
215     nstore.commitChanges();
216     nstore.destroy();
217   });
218   nstore = null;
219   equals(file.get('name'), 'Change Name', "Base > File id:1 has changed to name of 'Changed Name'");
220 
221   // Fifth, double check that the change exists
222   dir = store.find(NestedRecord.Directory, 1);
223   SC.run(function() {
224      file = dir.get('contents').objectAt(0).get('contents').objectAt(0);
225   });
226   equals(dir.get('status'), SC.Record.READY_DIRTY, 'Base > Directory id:1 has a READY_DIRTY State');
227   equals(file.get('status'), SC.Record.READY_DIRTY, 'Base > File id:1 has a READY_DIRTY State');
228   equals(file.get('name'), 'Change Name', "Base > File id:1 has actually changed to name of 'Changed Name'");
229 
230 });
