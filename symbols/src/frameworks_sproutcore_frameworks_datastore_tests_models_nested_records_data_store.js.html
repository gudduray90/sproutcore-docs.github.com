 11 var NestedRecord, store, storeKeys;
 12 
 13 var initModels = function () {
 14   NestedRecord.Directory = SC.Record.extend({
 15     /** Child Record Namespace */
 16     nestedRecordNamespace: NestedRecord,
 17     name: SC.Record.attr(String),
 18     contents: SC.Record.toMany('SC.Record', { isNested: true })
 19   });
 20 
 21   NestedRecord.File = SC.Record.extend({
 22     name: SC.Record.attr(String)
 23   });
 24 
 25 };
 26 
 27 // ..........................................................
 28 // Basic SC.Record Stuff
 29 //
 30 module("Data Store Tests for Nested Records", {
 31 
 32   setup: function () {
 33     NestedRecord = SC.Object.create({
 34       store: SC.Store.create()
 35     });
 36     store = NestedRecord.store;
 37     initModels();
 38     SC.RunLoop.begin();
 39     storeKeys = store.loadRecords([NestedRecord.Directory, NestedRecord.File], [
 40       {
 41         type: 'Directory',
 42         name: 'Dir 1',
 43         guid: 1,
 44         contents: [
 45           {
 46             type: 'Directory',
 47             name: 'Dir 2',
 48             guid: 2,
 49             contents: [
 50               {
 51                 type: 'File',
 52                 guid: 3,
 53                 name: 'File 1'
 54               },
 55               {
 56                 type: 'File',
 57                 guid: 4,
 58                 name: 'File 2'
 59               }
 60             ]
 61           }
 62         ]
 63       },
 64       {
 65         type: 'File',
 66         id: 5,
 67         name: 'File 3'
 68       }
 69     ]);
 70     SC.RunLoop.end();
 71   },
 72 
 73   teardown: function () {
 74     delete NestedRecord.Directory;
 75     delete NestedRecord.File;
 76     NestedRecord = null;
 77     store = null;
 78   }
 79 });
 80 
 81 test("Proper Initialization", function () {
 82   var first, second;
 83   equals(storeKeys.get('length'), 2, "number of primary store keys should be 2");
 84 
 85   // First
 86   first = store.materializeRecord(storeKeys[0]);
 87   ok(SC.kindOf(first, SC.Record), "first record is a kind of a SC.Record Object");
 88   ok(SC.instanceOf(first, NestedRecord.Directory), "first record is a instance of a NestedRecord.Directory Object");
 89 
 90   // Second
 91   second = store.materializeRecord(storeKeys[1]);
 92   ok(SC.kindOf(second, SC.Record), "second record is a kind of a SC.Record Object");
 93   ok(SC.instanceOf(second, NestedRecord.File), "second record is a instance of a NestedRecord.File Object");
 94 });
 95 
 96 test("Proper Status", function () {
 97   var first, second;
 98 
 99   // First
100   first = store.materializeRecord(storeKeys[0]);
101   equals(first.get('status'), SC.Record.READY_CLEAN, 'first record has a READY_CLEAN State');
102 
103   // Second
104   second = store.materializeRecord(storeKeys[1]);
105   equals(second.get('status'), SC.Record.READY_CLEAN, 'second record has a READY_CLEAN State');
106 });

109   var first, contents;

133 test("Use in Nested Store", function () {
134   var nstore, dir, c, file,
135       pk, id, nFile, nDir;
136 
137   // First, find the first file
138   dir = store.find(NestedRecord.Directory, 1);
139   ok(dir, "Directory id:1 exists");
140   equals(dir.get('name'), 'Dir 1', "Directory id:1 has a name of 'Dir 1'");
141   c = dir.get('contents');
142   ok(c, "Content of Directory id:1 exists");
143   dir = c.objectAt(0);
144   ok(dir, "Directory id:2 exists");
145   equals(dir.get('name'), 'Dir 2', "Directory id:2 has a name of 'Dir 2'");
146   c = dir.get('contents');
147   ok(c, "Content of Directory id:2 exists");
148   file = c.objectAt(0);
149   ok(file, "File id:1 exists");
150   equals(file.get('name'), 'File 1', "File id:1 has a name of 'File 1'");
151 
152   // Second, create nested store
153   nstore = store.chain();
154   SC.RunLoop.begin();
155   pk = file.get('primaryKey');
156   id = file.get(pk);
157   nFile = nstore.find(NestedRecord.File, id);
158   SC.RunLoop.end();
159   ok(nFile, "Nested > File id:1 exists");
160   equals(nFile.get('name'), 'File 1', "Nested > File id:1 has a name of 'File 1'");
161 
162   // Third, change the name of the nested store and see what happens
163   nFile.set('name', 'Change Name');
164   equals(nFile.get('name'), 'Change Name', "Nested > File id:1 has changed the name to 'Changed Name'");
165   equals(file.get('name'), 'File 1', "Base > File id:1 still has the name of 'File 1'");
166   nDir = nstore.find(NestedRecord.Directory, 1);
167 
168   // Fourth, commit the changes
169   nstore.commitChanges();
170   nstore.destroy();
171   nstore = null;
172   equals(file.get('name'), 'Change Name', "Base > File id:1 has changed to name of 'Changed Name'");
173 
174   // Fifth, double check that the change exists
175   dir = store.find(NestedRecord.Directory, 1);
176   file = dir.get('contents').objectAt(0).get('contents').objectAt(0);
177   equals(dir.get('status'), SC.Record.READY_DIRTY, 'Base > Directory id:1 has a READY_DIRTY State');
178   equals(file.get('status'), SC.Record.READY_DIRTY, 'Base > File id:1 has a READY_DIRTY State');
179   equals(file.get('name'), 'Change Name', "Base > File id:1 has actually changed to name of 'Changed Name'");
180 
181 });

183 test("Store#pushRetrieve for parent updates the child records", function () {
184   var parent = store.materializeRecord(storeKeys[0]),
185     nr = parent.get('contents').firstObject(),
186     newDataHash = {
187       type: 'Directory',
188       name: 'Dir 1 Changed',
189       guid: 1,
190       contents: [
191         {
192           type: 'Directory',
193           name: 'Dir 2 Changed',
194           guid: 2,
195           contents: [
196             {
197               type: 'File',
198               guid: 3,
199               name: 'File 1'
200             },
201             {
202               type: 'File',
203               guid: 4,
204               name: 'File 2'
205             }
206           ]
207         }
208       ]
209     };
210 
211   ok(nr, "Got nested record");
212   equals(nr.get('name'), 'Dir 2', "Dir id:2 has correct name");
213 
214   SC.run(function () {
215     store.pushRetrieve(null, null, newDataHash, storeKeys[0]);
216     store.flush();
217   });
218   equals(parent.get('name'), 'Dir 1 Changed', 'Dir id:1 name was changed');
219   equals(nr.get('name'), 'Dir 2 Changed', "Dir id:2 name was changed");
220 });
