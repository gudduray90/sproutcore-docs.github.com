 14 test("verify panel content container is visible at correct location with right size", function() {
 15   pane = SC.PanelPane.create({
 16     layout: { width: 400, height: 200, centerX: 0, centerY: 0 },
 17     contentView: SC.View.extend({
 18     })
 19   });
 20   pane.append();
 21 
 22   ok(pane.get('isVisibleInWindow'), 'pane.isVisibleInWindow should be YES');
 23   ok(pane.$().hasClass('sc-panel'), 'pane should have sc-panel class');
 24   ok(pane.childViews[0].get('isVisibleInWindow'), 'pane.div.isVisibleInWindow should be YES');
 25   ok(pane.childViews[0].$().hasClass('sc-view'), 'pane.div should have sc-view class');
 26 
 27   var pw = pane.layout.width;
 28   var ph = pane.layout.height;
 29   var ret = pane.layoutStyle();
 30 
 31   equals(ret.top, '50%', 'pane should center vertically');
 32   equals(ret.left, '50%', 'pane should center horizontally');
 33   equals(ret.width, '400px', 'pane should have width 400px');
 34   equals(ret.height, '200px', 'pane should have height 200px');
 35   equals(ret.marginLeft, -pw/2+'px', 'pane should shift-left %@ px'.fmt(-pw/2));
 36   equals(ret.marginTop, -ph/2+'px', 'pane should shift-top %@ px'.fmt(-ph/2));
 37 
 38   pane.remove();
 39   pane.destroy();
 40 }) ;

 56 test("Verify panel pane has aria-label attribute set, when ariaLabel is provided", function() {
 57   var pane = SC.PanelPane.create({
 58     layout: { width: 400, height: 200, centerX: 0, centerY: 0 },

 60     }),
 61     ariaLabel: "Panel is labelled by this value for voiceover"

 63   pane.append();
