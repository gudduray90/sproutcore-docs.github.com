 14 test("verify panel content container is visible at correct location with right size", function() {
 15   pane = SC.PanelPane.create({
 16     layout: { width: 400, height: 200, centerX: 0, centerY: 0 },
 17     contentView: SC.View
 18   });
 19   SC.run(function () {
 20     pane.append();
 21   });
 22 
 23   ok(pane.get('isVisibleInWindow'), 'pane.isVisibleInWindow should be YES');
 24   ok(pane.$().hasClass('sc-panel'), 'pane should have sc-panel class');
 25   ok(pane.childViews[0].get('isVisibleInWindow'), 'pane.div.isVisibleInWindow should be YES');
 26   ok(pane.childViews[0].$().hasClass('sc-view'), 'pane.div should have sc-view class');
 27 
 28   var pw = pane.layout.width;
 29   var ph = pane.layout.height;
 30   var ret = pane.layoutStyle();
 31 
 32   equals(ret.top, '50%', 'pane should center vertically');
 33   equals(ret.left, '50%', 'pane should center horizontally');
 34   equals(ret.width, '400px', 'pane should have width 400px');
 35   equals(ret.height, '200px', 'pane should have height 200px');
 36   equals(ret.marginLeft, -pw/2+'px', 'pane should shift-left %@ px'.fmt(-pw/2));
 37   equals(ret.marginTop, -ph/2+'px', 'pane should shift-top %@ px'.fmt(-ph/2));
 38 
 39   SC.run(function () {
 40     pane.destroy();
 41   });
 42 });

 77 test("Verify panel pane's modal pane lifecycle", function() {
 78   var pane = SC.PanelPane.create({
 79     contentView: SC.View,
 80     modalPane: SC.ModalPane,
 81     isModal: YES
 82   });
 83   ok(pane.get('modalPane').isClass, "Modal pane class is not instantiated before is panel is appended.");
 84   SC.run(function () {
 85     pane.append();
 86   });
 87   var modal = pane.get('modalPane');
 88   ok(!modal.isClass, "Modal pane class is instantiated when the panel is appended.");
 89   ok(modal.get('viewState') === SC.CoreView.ATTACHED_SHOWN, "Modal pane appends when panel pane is appended.");
 90   SC.run(function () {
 91     pane.remove();
 92   });
 93   ok(modal.get('viewState') & SC.CoreView.UNATTACHED, "Modal pane is removed when panel pane is removed.");
 94   SC.run(function () {
 95     pane.destroy();
 96   });
 97   ok(modal.get('isDestroyed'), "Modal pane is destroyed when its panel pane is destroyed.");
 98 
 99 });

117     pane.destroy();
