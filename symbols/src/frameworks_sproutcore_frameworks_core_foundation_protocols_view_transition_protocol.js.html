  8 /** @namespace
  9   This protocol defines the allowable transition plugin methods.
 10 
 11   SC.View uses transition plugins to setup, execute and cleanup the
 12   swapping between views and expects the given transition plugin object
 13   to implement the methods in this protocol.
 14 */
 15 SC.ViewTransitionProtocol = {
 16 
 17   /**
 18     This optional method is called to set up the entrance transition (i.e.
 19     transitionIn or transitionShow).
 20 
 21     Use this method to adjust the layout of the view so that it may be properly
 22     animated.  For example, you may need to adjust the content from a flexible
 23     layout (i.e. { left: 0, top: 0, right: 0, bottom: 0 }) to a fixed layout
 24     (i.e. { left: 0, top: 0, width: 100, height: 200 }) so that it can be
 25     moved.
 26 
 27     @param {SC.View} view The view being transitioned.
 28     @param {Object} options Options to modify the transition.  As set by transitionShowOptions or transitionInOptions.
 29     @param {Boolean} inPlace Whether the transition should start with the current layout of the view, because a previous transition was cancelled in place.
 30   */
 31   setup: function (view, options, inPlace) {},
 32 
 33   /**
 34     This method is called to transition the view in or visible (i.e.
 35     transitionIn or transitionShow).
 36 
 37     When the transition completes, this function *must* call `didTransitionIn()`
 38     on the view, passing this object and the original options as
 39     arguments.
 40 
 41     @param {SC.View} view The view being transitioned.
 42     @param {Object} options Options to modify the transition.  As set by transitionShowOptions or transitionInOptions.
 43     @param {Object} finalLayout The final layout of the view, which may be different than the starting layout of the view if a previous transition was cancelled in place.
 44     @param {Object} finalFrame The final frame of the view, which may be different than the starting frame of the view if a previous transition was cancelled in place.
 45   */
 46   run: function (view, options, finalLayout, finalFrame) {}
 47 
 48 };
