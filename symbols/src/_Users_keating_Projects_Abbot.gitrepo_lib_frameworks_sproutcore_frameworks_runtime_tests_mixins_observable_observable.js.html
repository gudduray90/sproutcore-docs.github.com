283       computed: function(key, value) {
284         this.computedCalls.push(value);
285         return 'computed';
286       }.property(),
287 
288       computedCachedCalls: [],
289       computedCached: function(key, value) {
290         this.computedCachedCalls.push(value);
291         return 'computedCached';
292       }.property().cacheable(),
293 
294 
295       // DEPENDENT KEYS
296 
297       changer: 'foo',
298 
299       dependentCalls: [],
300       dependent: function(key, value) {
301         this.dependentCalls.push(value);
302         return 'dependent';
303       }.property('changer'),
304 
305       dependentCachedCalls: [],
306       dependentCached: function(key, value) {
307         this.dependentCachedCalls.push(value);
308         return 'dependentCached';
309       }.property('changer').cacheable(),
310 
311       // everytime it is recomputed, increments call
312       incCallCount: 0,
313       inc: function() {
314         return this.incCallCount++;
315       }.property('changer').cacheable(),
316 
317       // depends on cached property which depends on another property...
318       nestedIncCallCount: 0,
319       nestedInc: function(key, value) {
320         return this.nestedIncCallCount++;
321       }.property('inc').cacheable(),
322 
323       // two computed properties that depend on a third property
324       state: 'on',
325       isOn: function(key, value) {
326         if (value !== undefined) this.set('state', 'on');
327         return this.get('state') === 'on';
328       }.property('state'),
329 
330       isOff: function(key, value) {
331         if (value !== undefined) this.set('state', 'off');
332         return this.get('state') === 'off';
333       }.property('state')

335     }) ;
336   }
