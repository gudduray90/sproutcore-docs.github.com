363   /**
364     The current selection of the text field, returned as an SC.TextSelection
365     object.
366 
367     Note that if the selection changes a new object will be returned -- it is
368     not the case that a previously-returned SC.TextSelection object will
369     simply have its properties mutated.
370 
371     @field
372     @type SC.TextSelection
373    */
374   selection: function (key, value) {
375     var element = this.$input()[0],
376         range, start, end;
377 
378     // Are we being asked to set the value, or return the current value?
379     if (value === undefined) {
380       // The client is retrieving the value.
381       if (element) {
382         start = null;
383         end = null;
384 
385         if (!element.value) {
386           start = end = 0 ;
387         }
388         else {
389           // In IE8, input elements don't have hasOwnProperty() defined.
390           try {
391             if ('selectionStart' in element) {
392               start = element.selectionStart ;
393             }
394             if ('selectionEnd' in element) {
395               end = element.selectionEnd ;
396             }
397           }
398           // In Firefox when you ask the selectionStart or End of a hidden
399           // input, sometimes it throws a weird error.
400           // Adding this to just ignore it.
401           catch (e){
402             return null;
403           }
404 
405           // Support Internet Explorer.
406           if (start === null  ||  end === null ) {
407             var selection = document.selection ;
408             if (selection) {
409               var type = selection.type ;
410               if (type  &&  (type === 'None'  ||  type === 'Text')) {
411                 range = selection.createRange() ;
412 
413                 if (!this.get('isTextArea')) {
414                   // Input tag support.  Figure out the starting position by
415                   // moving the range's start position as far left as possible
416                   // and seeing how many characters it actually moved over.
417                   var length = range.text.length ;
418                   start = Math.abs(range.moveStart('character', 0 - (element.value.length + 1))) ;
419                   end = start + length ;
420                 }
421                 else {
422                   // Textarea support.  Unfortunately, this case is a bit more
423                   // complicated than the input tag case.  We need to create a
424                   // "dummy" range to help in the calculations.
425                   var dummyRange = range.duplicate() ;
426                   dummyRange.moveToElementText(element) ;
427                   dummyRange.setEndPoint('EndToStart', range) ;
428                   start = dummyRange.text.length ;
429                   end = start + range.text.length ;
430                 }
431               }
432             }
433           }
434         }
435         return SC.TextSelection.create({ start:start, end:end }) ;
436       }
437       else {
438         return null;
439       }
440     }
441     else {
442       // The client is setting the value.  Make sure the new value is a text
443       // selection object.
444       if (!value  ||  !value.kindOf  ||  !value.kindOf(SC.TextSelection)) {
445         throw "When setting the selection, you must specify an SC.TextSelection instance.";
446       }
447 
448       if (element) {
449         if (element.setSelectionRange) {
450           element.setSelectionRange(value.get('start'), value.get('end')) ;
451         }
452         else {
453           // Support Internet Explorer.
454           range = element.createTextRange() ;
455           start = value.get('start') ;
456           range.move('character', start) ;
457           range.moveEnd('character', value.get('end') - start) ;
458           range.select() ;
459         }
460       }
461       return value;
462     }
463 
464     // Implementation note:
465     // There are certain ways users can add/remove text that we can't identify
466     // via our key/mouse down/up handlers (such as the user choosing Paste
467     // from a menu).  So that's why we need to update our 'selection' property
468     // whenever the field's value changes.
469   }.property('fieldValue').cacheable(),

486   accessoryViewObserver: function () {
487     var classNames,
488         viewProperties = ['leftAccessoryView', 'rightAccessoryView'],
489         len = viewProperties.length , i, viewProperty, previousView,
490         accessoryView;
491 
492     for (i=0; i<len; i++) {
493       viewProperty = viewProperties[i] ;
494 
495       // Is there an accessory view specified?
496       previousView = this['_'+viewProperty] ;
497       accessoryView = this.get(viewProperty) ;
498 
499       // If the view is the same, there's nothing to do.  Otherwise, remove
500       // the old one (if any) and add the new one.
501       if (! (previousView &&
502              accessoryView &&
503              (previousView === accessoryView) ) ) {
504 
505         // If there was a previous previous accessory view, remove it now.
506         if (previousView) {
507           // Remove the "sc-text-field-accessory-view" class name that we had
508           // added earlier.
509           classNames = previousView.get('classNames') ;
510           classNames = classNames.without('sc-text-field-accessory-view') ;
511           previousView.set('classNames', classNames) ;
512           this.removeChild(previousView) ;
513           previousView = null ;
514           this['_'+viewProperty] = null ;
515         }
516 
517         // If there's a new accessory view to add, do so now.
518         if (accessoryView) {
519           // If the user passed in a class rather than an instance, create an
520           // instance now.
521           if (accessoryView.isClass) {
522             accessoryView = accessoryView.create({
523               layoutView: this
524             }) ;
525           }
526 
527           // Add in the "sc-text-field-accessory-view" class name so that the
528           // z-index gets set correctly.
529           classNames = accessoryView.get('classNames') ;
530           var className = 'sc-text-field-accessory-view' ;
531           if (classNames.indexOf(className) < 0) {
532             classNames = SC.clone(classNames);
533             classNames.push(className) ;
534             accessoryView.set('classNames', classNames);
535           }
536 
537           // Actually add the view to our hierarchy and cache a reference.
538           this.appendChild(accessoryView) ;
539           this['_'+viewProperty] = accessoryView ;
540         }
541       }
542     }
543   }.observes('leftAccessoryView', 'rightAccessoryView'),
544 
545   layoutChildViewsIfNeeded: function (isVisible) {
546     // For the right accessory view, adjust the positioning such that the view
547     // is right-justified, unless 'right' is specified.
548     if (!isVisible) isVisible = this.get('isVisibleInWindow') ;
549     if (isVisible && this.get('childViewsNeedLayout')) {
550       var rightAccessoryView = this.get('rightAccessoryView') ;
551       if (rightAccessoryView  &&  rightAccessoryView.get) {
552         var layout = rightAccessoryView.get('layout') ;
553         if (layout) {
554           // Clear out any 'left' value.
555           layout.left = null;
556 
557           // Unless the user specified a 'right' value, specify a default to
558           // right-justify the view.
559           if (!layout.right) layout.right = 0 ;
560 
561           rightAccessoryView.adjust({ layout: layout }) ;
562         }
563       }
564     }
565     sc_super() ;
566   },

568   render: function (context, firstTime) {
569     sc_super() ;
570     var v, accessoryViewWidths, leftAdjustment, rightAdjustment;
571 
572     // always have at least an empty string
573     v = this.get('fieldValue');
574     if (SC.none(v)) v = '';
575     v = String(v);
576 
577     // update layer classes always
578     context.setClass('not-empty', v.length > 0);
579 
580     // If we have accessory views, we'll want to update the padding on the
581     // hint to compensate for the width of the accessory view.  (It'd be nice
582     // if we could add in the original padding, too, but there's no efficient
583     // way to do that without first rendering the element somewhere on/off-
584     // screen, and we don't want to take the performance hit.)
585     accessoryViewWidths = this._getAccessoryViewWidths() ;
586     leftAdjustment  = accessoryViewWidths['left'] ;
587     rightAdjustment = accessoryViewWidths['right'] ;
588 
589     if (leftAdjustment)  leftAdjustment  += 'px' ;
590     if (rightAdjustment) rightAdjustment += 'px' ;
591 
592     this._renderField(context, firstTime, v, leftAdjustment, rightAdjustment);
593   },

607   /** @private */
608   _renderField: function (context, firstTime, value, leftAdjustment, rightAdjustment) {
609     // TODO:  The cleanest thing might be to create a sub- rendering context
610     //        here, but currently SC.RenderContext will render sibling
611     //        contexts as parent/child.
612 
613     var hint = this.get('formattedHint'),
614         hintOnFocus = this.get('hintOnFocus'),
615         hintString = '',
616         maxLength = this.get('maxLength'),
617         isTextArea = this.get('isTextArea'),
618         isEnabled = this.get('isEnabled'),
619         isEditable = this.get('isEditable'),
620         autoCorrect = this.get('autoCorrect'),
621         autoCapitalize = this.get('autoCapitalize'),
622         isBrowserFocusable = this.get('isBrowserFocusable'),
623         spellCheckString='', autocapitalizeString='', autocorrectString='',
624         name, adjustmentStyle, type, hintElements, element, paddingElementStyle,
625         fieldClassNames, isOldSafari, activeState, browserFocusable;
626 
627     context.setClass('text-area', isTextArea);
628 
629     //Adding this to differentiate between older and newer versions of safari
630     //since the internal default field padding changed
631     isOldSafari= SC.browser.isWebkit &&
632         SC.browser.compare(SC.browser.engineVersion, '532')<0;
633     context.setClass('oldWebKitFieldPadding', isOldSafari);
634 
635 
636     if (firstTime || this._forceRenderFirstTime) {
637       this._forceRenderFirstTime = NO;
638       activeState = isEnabled ? (isEditable ? '' : 'readonly="readonly"') : 'disabled="disabled"' ;
639       name = this.get('layerId');
640 
641       spellCheckString = this.get('spellCheckEnabled') ? ' spellcheck="true"' : ' spellcheck="false"';
642 
643       if (autoCorrect != null) {
644         autocorrectString = ' autocorrect=' + (!autoCorrect ? '"off"' : '"true"');
645       }
646 
647       if (autoCorrect != null) {
648         autocapitalizeString = ' autocapitalize=' + (!autoCapitalize ? '"off"' : '"true"');
649       }
650 
651       if (isBrowserFocusable) {
652         browserFocusable = 'tabindex="-1"';
653       }
654         // if hint is on and we don't want it to show on focus, create one
655       if (SC.platform.input.placeholder && !hintOnFocus) {
656         hintString = ' placeholder="' + hint + '"';
657       }
658 
659       if (this.get('shouldRenderBorder')) context.push('<div class="border"></div>');
660 
661       // Render the padding element, with any necessary positioning
662       // adjustments to accommodate accessory views.
663       adjustmentStyle = '' ;
664       if (leftAdjustment || rightAdjustment) {
665         adjustmentStyle = 'style="' ;
666         if (leftAdjustment)  adjustmentStyle += 'left:'  + leftAdjustment  + ';' ;
667         if (rightAdjustment) adjustmentStyle += 'right:' + rightAdjustment + ';' ;
668         adjustmentStyle += '"' ;
669       }
670       context.push('<div class="padding" '+adjustmentStyle+'>');
671 
672       value = this.get('escapeHTML') ? SC.RenderContext.escapeHTML(value) : value;
673       if (this._hintON && !SC.platform.input.placeholder && (!value || (value && value.length===0))) {
674         value = hint;
675         context.setClass('sc-hint', YES);
676       }
677 
678       if (hintOnFocus) {
679         var hintStr = '<div aria-hidden="true" class="hint '+
680                       (isTextArea ? '':'ellipsis')+'%@">'+ hint + '</div>';
681         context.push(hintStr.fmt(value ? ' sc-hidden': ''));
682       }
683 
684       fieldClassNames = "field";
685 
686       // Render the input/textarea field itself, and close off the padding.
687       if (isTextArea) {
688         context.push('<textarea aria-label="' + hint + '" aria-multiline="true" class="'+fieldClassNames+'" name="'+ name+
689                       '" '+ activeState + hintString +
690                       spellCheckString + autocorrectString + browserFocusable +
691                       autocapitalizeString + ' maxlength="'+ maxLength+ '">'+
692                       value+ '</textarea></div>') ;
693       }
694       else {
695         type = this.get('type');
696 
697         // Internet Explorer won't let us change the type attribute later
698         // so we force it to password if needed now, or if the value is not the hint
699         if (this.get('isPassword')) {
700           // @if (debug)
701           SC.Logger.warn("SC.TextFieldView#isPassword is deprecated. Please set SC.TextFieldView#type to password instead.");
702           // @endif
703 
704           type = 'password';
705         }
706 
707         context.push('<input aria-label="' + hint + '" class="'+fieldClassNames+'" type="'+ type+
708                       '" name="'+ name + '" '+ activeState + ' value="'+ value + '"' +
709                       hintString + spellCheckString+ browserFocusable +
710                       ' maxlength="'+ maxLength+ '" '+autocorrectString+' ' +
711                       autocapitalizeString+'/></div>') ;
712       }
713     }
714     else {
715       var input= this.$input(),
716           elem = input[0],
717           val = this.get('value');
718 
719       if (hintOnFocus) this.$('.hint')[0].innerHTML = hint;
720       else if (!hintOnFocus) elem.placeholder = hint;
721 
722       // IE8 has problems aligning the input text in the center
723       // This is a workaround for centering it.
724       if (SC.browser.name === SC.BROWSER.ie && SC.browser.version <= 8 && !isTextArea) {
725         input.css('line-height', this.get('frame').height + 'px');
726       }
727 
728       if (!val || (val && val.length === 0)) {
729         if (this.get('isPassword')) { elem.type = 'password'; }
730 
731         if (!SC.platform.input.placeholder && this._hintON) {
732           if (!this.get('isFirstResponder')) {
733           // Internet Explorer doesn't allow you to modify the type afterwards
734           // jQuery throws an exception as well, so set attribute directly
735 
736           context.setClass('sc-hint', YES);
737           input.val(hint);
738           } else {
739           // Internet Explorer doesn't allow you to modify the type afterwards
740           // jQuery throws an exception as well, so set attribute directly
741             context.setClass('sc-hint', NO);
742             input.val('');
743           }
744         }
745       }
746 
747       if (autoCorrect != null) {
748         input.attr('autoCorrect', !autoCorrect ? 'off' : 'true');
749       } else {
750         input.attr('autoCorrect', null);
751       }
752

1127   /**

1131   keyDown: function (evt) {

1133         keyCode = evt.keyCode,
1134         maxLengthReached = false,
1135         value, view;
1136 
1137     // Handle return and escape.  this way they can be passed on to the
1138     // responder chain.
1139     // If the event is triggered by a return while entering IME input,
1140     // don't got through this path.
1141     if ((which === SC.Event.KEY_RETURN && !evt.isIMEInput) && !this.get('isTextArea')) { return NO; }
1142     if (which === SC.Event.KEY_ESC) { return NO; }
1143 
1144     // handle tab key
1145     if ((which === SC.Event.KEY_TAB || keyCode === SC.Event.KEY_TAB) && this.get('defaultTabbingEnabled')) {
1146       view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
1147       if (view) view.becomeFirstResponder();
1148       else evt.allowDefault();
1149       return YES ; // handled
1150     }
1151 
1152     // maxlength for textareas
1153     if (!SC.platform.input.maxlength && this.get('isTextArea')) {
1154       var val = this.get('value');
1155 
1156       // This code is nasty. It's thanks gecko .keycode table that has charters like & with the same keycode as up arrow key
1157       if (val && ((!SC.browser.isMozilla && which>47) ||
1158                   (SC.browser.isMozilla && ((which>32 && which<43) || which>47) && !(keyCode>36 && keyCode<41))) &&
1159           (val.length >= this.get('maxLength'))) {
1160         maxLengthReached = true;
1161       }
1162     }
1163     // validate keyDown...
1164     // do not validate on touch, as it prevents return.
1165     if ((this.performValidateKeyDown(evt) || SC.platform.touch) && !maxLengthReached) {
1166       this._isKeyDown = YES;
1167       evt.allowDefault();
1168     } else {
1169       evt.stop();
1170     }
1171 
1172     if (this.get('applyImmediately')) {
1173       // There used to be an invokeLater here instead of setTimeout. What we
1174       // really need is setTimeout.
1175       var self = this;
1176       setTimeout(function () {
1177         self.fieldValueDidChange();
1178       }, 10);
1179     }
1180 
1181     return YES;
1182   },
