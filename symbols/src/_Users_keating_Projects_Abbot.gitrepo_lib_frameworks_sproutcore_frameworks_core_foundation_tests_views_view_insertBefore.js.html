  9 var parent, child;
 10 module("SC.View#insertBefore", {
 11 	setup: function() {
 12 	  child = SC.View.create();
 13 	  parent = SC.View.create({
 14 	    childViews: [SC.View]
 15 	  });		
 16 	}
 17 });
 18 
 19 test("returns receiver", function() {
 20   equals(parent.insertBefore(child, null), parent, 'receiver');
 21 });
 22 
 23 test("makes set child.parentView = to new parent view", function() {
 24 	ok(child.get('parentView')!==parent, 'precond - parent is not child.parentView yet');
 25 	
 26 	// add observer to make sure property change triggers 
 27 	var callCount = 0;
 28 	child.addObserver('parentView', function() {
 29 	  callCount++;
 30 	});
 31 	
 32 	parent.insertBefore(child, null);
 33 	equals(child.get('parentView'), parent, 'parent is child.parentView');
 34 	equals(callCount, 1, 'observer did fire');
 35 });
 36 
 37 test("insertBefore(child, null) appends child to end of parent.childView's array", function() {
 38 	parent.insertBefore(child, null);
 39 	equals(parent.childViews[parent.childViews.length-1], child, 'child is last childView');
 40 });
 41 
 42 test("insertBefore(child, otherChild) inserts child before other child view", function() {
 43   
 44   var otherChild = parent.childViews[0]; // get current first child
 45   ok(otherChild, 'precond - otherChild is not null');
 46   parent.insertBefore(child, otherChild);
 47   equals(parent.childViews[0], child, 'child inserted before other child');
 48 });
 49 
 50 test("if child belongs to another parent view, child will be removed from other parent when added to new parent", function() {
 51 
 52 	// create child view that belongs to a parent already
 53   var oldParent = SC.View.create({ childViews:[SC.View] });
 54   child = oldParent.childViews[0];
 55   
 56 	ok(child, 'precond - got child');
 57   ok(child.get('parentView'), 'precond - child has a parent view');  
 58   
 59   parent.insertBefore(child, null);
 60   ok(oldParent.childViews.indexOf(child)<0, 'oldParent no longer has childView');
 61   equals(child.get('parentView'), parent, 'child has new parent view');
 62 });
 63 
 64 test("invokes willAddChild() on receiver if defined before adding child" ,function() {
 65 
 66   // monkey patch to test
 67   var callCount = 0;
 68   var otherChild = parent.childViews[0];
 69   parent.willAddChild = function(newChild, beforeView) {
 70   
 71   	// verify params
 72   	equals(newChild, child, 'passed newChild');
 73   	equals(beforeView, otherChild, 'passed beforeView');
 74   	
 75   	// verify this is called BEFORE the view is added
 76   	ok(parent.childViews.indexOf(child)<0, 'should not have child yet');
 77   	ok(child.get('parentView')!==parent, 'childView not changed yet either');
 78   	callCount++;
 79   };
 80 
 81 	  
 82   parent.insertBefore(child, otherChild);
 83   equals(callCount, 1, 'invoked');
 84 });
