 12 module("SC.Range");

 14 test("to find the maxRange() and minRange() values of a given range",function(){
 15 	var obj = {start:15,length:75};
 16 	equals(YES,SC.minRange(obj) == 15,'Minimum range');
 17     equals(YES,SC.maxRange(obj) == 90,'Maximum range');
 18 });
 19 
 20 test("unionRanges() to find the union of two ranges",function(){
 21 	var obj = {start:15,length:75};
 22 	var obj1 = {start:5,length:50};
 23 	var c = SC.unionRanges(obj,obj1);
 24 	equals(obj1.start,SC.minRange(c),'Minimum range');
 25 	equals(85,c.length,'Maximum range');
 26  27 });
 28 
 29 test("rangesEqual() to find if the given ranges are equal",function(){
 30     var obj = {start:15,length:75};
 31 	var obj1 = {start:15,length:75};
 32 	var obj2 = {start:5,length:50};
 33 	var c = SC.rangesEqual(obj,obj1);
 34 	var d = SC.rangesEqual(obj1,obj2);
 35 	equals(true,c,'Equal ranges');
 36 	equals(false,d,'Unequal ranges');	
 37 });
 38 
 39 test("cloneRange() to clone the given range",function(){
 40 	var obj = {start:15,length:75};
 41 	var c = SC.cloneRange(obj);
 42 	equals(obj.start,SC.minRange(c),'Minimum range');
 43 	equals(75,c.length,'Maximum range');
 44 });
 45 
 46 test("valueInRange() to find if a given value is in range",function(){
 47 	var obj = {start:15,length:75};
 48 	var c = SC.valueInRange(25,obj);
 49 	var d = SC.valueInRange(10,obj);
 50 	equals(true,c,'In range');
 51 	equals(false,d,'Not in range');
 52 });

 62 test("intersectRanges() to get the intersection of 2 ranges",function(){
 63 	var obj1 = {start:15,length:75};
 64 	var obj2 = {start:5,length:50};
 65 	
 66 	var c = SC.intersectRanges(obj1,obj2);
 67 	equals(SC.minRange(obj1),SC.minRange(c),'Minimum Intersection Range');
 68 	equals(40,c.length,'Maximum Intersection Range');
 69 });
