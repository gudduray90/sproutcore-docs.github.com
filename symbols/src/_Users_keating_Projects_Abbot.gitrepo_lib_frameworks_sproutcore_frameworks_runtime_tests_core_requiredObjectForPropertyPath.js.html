 10 /*globals module test ok same equals expects */
 11 
 12 module("SC.requiredObjectForPropertyPath") ;
 13 
 14 test("should be able to resolve an object on the window", function() {
 15   var myLocal = (window.myGlobal = { test: 'this '}) ;
 16 
 17   same(myLocal, { test: 'this '}) ;
 18   same(window.myGlobal, { test: 'this '}) ;
 19 
 20   // verify we can resolve our binding path
 21   same(SC.requiredObjectForPropertyPath('myGlobal'), { test: 'this '}) ;
 22 
 23   window.myGlobal = null;
 24 });
 25 
 26 test("should throw error when object can't be found", function() {
 27   should_throw(function(){ SC.requiredObjectForPropertyPath('notExistingObject'); },
 28                   Error, "notExistingObject could not be found");
 29 });
