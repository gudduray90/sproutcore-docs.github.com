  9   OK: 200,

 16 SC.RESOURCE_ROUTE_LEVEL = {
 17   COLLECTION: 0,
 18   MEMBER: 1
 19 };
 20 
 21 SC.RESOURCE_ROUTE_FORMAT = {
 22   XML: 'xml',
 23   JSON: 'json'
 24 };
 25 
 26 SC.ResourceRoute = SC.Object.extend({
 27 
 28   /** properties **/
 29 
 30   format: SC.RESOURCE_ROUTE_FORMAT.JSON,
 31  32 
  isResourceRoute: YES,
 33 
 34   httpMethod: SC.HTTP_METHOD.GET,
 35 
 36   level: SC.RESOURCE_ROUTE_LEVEL.COLLECTION,
 37 
 38   successStatus: SC.HTTP_STATUS_CODE.OK,
 39 
 40   /** functions **/
 41   didComplete: function(response, params) {
 42     if (response.get('status') === this.get('successStatus')) {
 43       if (params.storeKey) {
 44         this.onSuccess(params.store, params.storeKey, response.get('body'));
 45       } else {
 46         this.onSuccess(params.store, response.get('body'));
 47       }
 48     } else {
 49       this.onFailure(response.get('body'));
 50     }
 51   },
 52 
 53   willSendRequest: function(params) {
 54     
 55   },
 56   
 57   onSuccess: function(store, body) {
 58     SC.Logger.warn("SC.ResourceRoute: onSuccess must be provided by the subclass");
 59   },
 60 
 61   onFailure: function(body) {
 62     SC.Logger.warn("SC.ResourceRoute: failed to load resource.");
 63   }
 64 });
