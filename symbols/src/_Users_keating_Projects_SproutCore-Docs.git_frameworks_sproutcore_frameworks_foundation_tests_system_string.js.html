 11 test("Pluralizing a string", function() {
 12   expect(8);
 13   same('Goat'.pluralize(), 'Goats', "standard pluralization");
 14   same('There are many goat'.pluralize(), 'There are many goats', "standard pluralization of a multi-word string");
 15   same('Bunny'.pluralize(), 'Bunnies', "non-standard pluralization");
 16   same('I like bunny'.pluralize(), 'I like bunnies', "non-standard pluralization of a multi-word string");
 17   same('child'.pluralize(), 'children', "irregular pluralization");
 18   same('I have three child'.pluralize(), 'I have three children', "irregular pluralization of a multi-word string");
 19   same('sheep'.pluralize(), 'sheep', "uncountable pluralization");
 20   same('Please hold this sheep'.pluralize(), 'Please hold this sheep', "uncountable pluralization of a multi-word string");
 21 });

 23 test("Singularizing a string", function() {
 24   expect(8);
 25   same('Vegetables'.singularize(), 'Vegetable', "standard singularization");
 26   same('Broccoli is a vegetables'.singularize(), 'Broccoli is a vegetable', "standard singularization of a multi-word string");
 27   same('Properties'.singularize(), 'Property', "non-standard singularization");
 28   same('Buy a properties'.singularize(), 'Buy a property', "non-standard singularization of a multi-word string");
 29   same('people'.singularize(), 'person', "irregular singularization");
 30   same('The Village People'.singularize(), 'The Village Person', "irregular singularization of a multi-word string");
 31   same('money'.singularize(), 'money', "uncountable singularization");
 32   same('Gotta git da money'.singularize(), 'Gotta git da money', "uncountable singularization of a multi-word string");
 33 });
