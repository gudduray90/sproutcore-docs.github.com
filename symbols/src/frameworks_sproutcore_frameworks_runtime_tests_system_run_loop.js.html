  7 module("System:run_loop() - chained binding", {
  8   setup: function() {
  9     first = SC.Object.create({
 10 		output: 'first'
 11 	}) ;
 12 
 13 	second = SC.Object.create({
 14       input: 'second',
 15       output: 'second',
 16 
 17       inputDidChange: function() {
 18         this.set("output", this.get("input")) ;
 19       }.observes("input")
 20 
 21     }) ;
 22 
 23     third = SC.Object.create({
 24 		input: "third"
 25 	}) ;
 26   }
 27 });
 28 
 29 test("Should propagate bindings after the RunLoop completes (using SC.RunLoop)", function() {
 30 	SC.RunLoop.begin();
 31 		//Binding of output of first object to input of second object
 32   		binding1 = SC.Binding.from("output", first).to("input", second).connect() ;
 33 
 34 		//Binding of output of second object to input of third object
 35 		binding2 = SC.Binding.from("output", second).to("input", third).connect() ;
 36 
 37 		SC.Binding.flushPendingChanges() ; // actually sets up the connection
 38 
 39 		//Based on the above binding if you change the output of first object it should
 40 		//change the all the variable of first,second and third object
 41 		first.set("output", "change") ;
 42 
 43 		//Changes the output of the first object
 44 		equals(first.get("output"), "change") ;
 45 
 46 		//since binding has not taken into effect the value still remains as change.
 47 		equals(second.get("output"), "first") ;
 48 	SC.RunLoop.end(); // allows bindings to trigger...
 49 
 50 	//Value of the output variable changed to 'change'
 51 	equals(first.get("output"), "change") ;
 52 
 53 	//Since binding triggered after the end loop the value changed to 'change'.
 54 	equals(second.get("output"), "change") ;
 55 });

 86 test("Should propagate bindings after the RunLoop completes (checking invokeOnce() function)", function() {
 87 	SC.RunLoop.begin();
 88 		//Binding of output of first object to input of second object
 89   		binding1 = SC.Binding.from("output", first).to("input", second).connect() ;
 90 
 91 		//Binding of output of second object to input of third object
 92 		binding2 = SC.Binding.from("output", second).to("input", third).connect() ;
 93 
 94 		SC.Binding.flushPendingChanges() ; // actually sets up the connection
 95 
 96 		//Based on the above binding if you change the output of first object it should
 97 		//change the all the variable of first,second and third object
 98 		first.set("output", "change") ;
 99 
100 		//Changes the output of the first object
101 		equals(first.get("output"), "change") ;
102 
103 		//since binding has not taken into effect the value still remains as change.
104 		equals(second.get("output"), "first") ;
105 
106 		// Call the invokeOnce function to set the function which needs to be called once
107 		second.invokeOnce('second','inputDidChange');
108 
109 	SC.RunLoop.end(); // allows bindings to trigger...
110 
111 	//Value of the output variable changed to 'change'
112 	equals(first.get("output"), "change") ;
113 
114 	//Since binding triggered after the end loop the value changed to 'change'.
115 	equals(second.get("output"), "change") ;
116 
117 	//Set the output for the first so that the 'inputDidChange' function in the second object is called again
118 	first.set("output", "againChanged") ;
119 
120 	//Value of the output variable changed to 'change'
121 	equals(first.get("output"), "againChanged") ;
122 
123 	//Since the invoker function is called only once the value of output did not change.
124 	equals(second.get("output"), "change") ;
125

128 test("Should flush the invoke last queue at the end of the current run loop", function () {
129   var iCalled = 0;
130   SC.RunLoop.begin();
131 
132   SC.RunLoop.currentRunLoop.invokeLast(function () {
133     iCalled++;
134   });
135 
136   ok(!iCalled, "should not have flushed the invokeLast queue");
137   SC.RunLoop.end();
138   ok(iCalled, "should have flushed the invokeLast queue");
139 
140   SC.RunLoop.begin();
141   SC.RunLoop.end();
142   equals(iCalled, 1, "should have flushed the invokeLast queue only once");
143 });
144 
145 test("Should repeatedly flush the invoke last queue until there are no more items", function () {
146   var iCalled = 0;
147   SC.RunLoop.begin();
148 
149   SC.RunLoop.currentRunLoop.invokeLast(function () {
150     iCalled++;
151     if (iCalled < 5) {
152       SC.RunLoop.currentRunLoop.invokeLast(arguments.callee);
153     }
154   });
155 
156   ok(!iCalled, "should not have flushed the invokeLast queue");
157   SC.RunLoop.end();
158   ok(iCalled, "should have flushed the invokeLast queue");
159   equals(iCalled, 5, "should have flushed the invokeLast queue 5 times");
160 });
161 
162 test("Should flush the invoke next queue at the beginning of the next run loop", function () {
163   var iCalled = 0;
164   SC.RunLoop.begin();
165 
166   SC.RunLoop.currentRunLoop.invokeNext(function () {
167     iCalled++;
168   });
169 
170   SC.RunLoop.end();
171   ok(!iCalled, "should not have flushed the invokeNext queue");
172 
173   SC.RunLoop.begin();
174   ok(iCalled, "should have flushed the invokeNext queue");
175 
176   SC.RunLoop.end();
177   equals(iCalled, 1, "should have flushed the invokeNext queue only once");
178 });
179 
180 test("Calling invokeNext inside an invokeNext invocation will schedule for the function to run in the next run loop", function () {
181   var iCalled = 0;
182   SC.RunLoop.begin();
183 
184   SC.RunLoop.currentRunLoop.invokeNext(function () {
185     if (iCalled < 5) {
186       SC.RunLoop.currentRunLoop.invokeNext(arguments.callee);
187     }
188     iCalled++;
189   });
190 
191   for (var i = 0; i < 5; i++) {
192     equals(iCalled, i, "should have flushed the invokeNext queue " + i + " time" + (i !== 1 ? 's' : ''));
193     SC.RunLoop.end();
194     SC.RunLoop.begin();
195   }
196   equals(iCalled, 5, "should have flushed the invokeNext queue 5 times");
197   SC.RunLoop.end();
198 });

200 /**
201   The previous implementation of invokeNext never actually caused the next run
202   of the run loop and would only work if another run loop happened to occur.
203   So if you were using invokeNext to do rendering, it would not really work
204   unless the mouse moved or some other timer was firing.
205 */
206 test("Using invokeNext should result in an additional run of the run loop immediately after it ends.", function () {
207   stop(2000);
208 
209   expect(1);
210   SC.run(function () {
211     SC.RunLoop.currentRunLoop.invokeNext(function () {
212       ok(YES, "should have started another run of the run loop");
213       start();
214     });
215   });
216 });

218 /**
219   There was a regression where unscheduling a run loop failed to clear out the
220   internal state and therefore further scheduling of the run loop failed.
221 */
222 test("Using unscheduleRunLoop should not prevent scheduleRunLoop from working.", function () {
223   SC.run(function () {
224     var now = new Date(),
225       newTimeoutID,
226       timeoutID;
227 
228     timeoutID = SC.RunLoop.currentRunLoop.scheduleRunLoop(now);
229     SC.RunLoop.currentRunLoop.unscheduleRunLoop();
230     newTimeoutID = SC.RunLoop.currentRunLoop.scheduleRunLoop(now);
231 
232     ok(timeoutID !== newTimeoutID, "should not be the same timeout id after unscheduling");
233   });
234 });
