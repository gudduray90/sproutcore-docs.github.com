  9 /**
 10   Renders and updates the HTML representation of a button.
 11 */
 12 SC.BaseTheme.buttonRenderDelegate = SC.RenderDelegate.create({
 13   className: 'button',
 14 
 15   //
 16   // SIZE DEFINITIONS
 17   //
 18   'sc-small-size': {
 19     height: 18,
 20     autoResizePadding: 15
 21   },
 22 
 23   'sc-regular-size': {
 24     height: 24,
 25     autoResizePadding: 20
 26   },
 27 
 28   'sc-huge-size': {
 29     height: 30,
 30     autoResizePadding: 30
 31   },
 32 
 33   'sc-jumbo-size': {
 34     height: 44,
 35     autoResizePadding: 50
 36   },
 37 
 38 
 39   //
 40   // RENDERING LOGIC
 41   //
 42 
 43   /**
 44     Called when we need to create the HTML that represents the button.
 45 
 46     @param {SC.Object} dataSource the object containing the information on how to render the button
 47     @param {SC.RenderContext} context the render context instance
 48   */
 49   render: function(dataSource, context) {
 50     this.addSizeClassName(dataSource, context);
 51 
 52     var labelContent,
 53         toolTip     = dataSource.get('toolTip'),
 54         isSelected  = dataSource.get('isSelected') || NO,
 55         isActive    = dataSource.get('isActive') || NO,
 56         labelId     = SC.guidFor(dataSource) + '-label';
 57 
 58     context.setClass({
 59       'icon': !!dataSource.get('icon') || NO,
 60       'def': dataSource.get('isDefault'),
 61       'cancel': dataSource.get('isCancel'),
 62       'active': isActive,
 63       'sel': isSelected
 64     });
 65 
 66     if (toolTip) {
 67       context.attr('title', toolTip);
 68       context.attr('alt', toolTip);
 69     }
 70 
 71     this.includeSlices(dataSource, context, SC.THREE_SLICE);
 72 
 73     // accessibility
 74     context.attr('aria-pressed', isActive.toString());
 75     context.attr('aria-labelledby', labelId);
 76 
 77     // Create the inner label element that contains the text and, optionally,
 78     // an icon.
 79     context = context.begin('label').addClass('sc-button-label').id(labelId);
 80     dataSource.get('theme').labelRenderDelegate.render(dataSource, context);
 81     context = context.end();
 82 
 83     if (dataSource.get('supportFocusRing')) {
 84       context = context.begin('div').addClass('focus-ring');
 85       this.includeSlices(dataSource, context, SC.THREE_SLICE);
 86       context = context.end();
 87     }
 88   },
 89 
 90   /**
 91     Called when one or more display properties have changed and we need to
 92     update the HTML representation with the new values.
 93 
 94     @param {SC.Object} dataSource the object containing the information on how to render the button
 95     @param {SC.RenderContext} jquery the jQuery object representing the HTML representation of the button
 96   */
 97   update: function(dataSource, jquery) {
 98     this.updateSizeClassName(dataSource, jquery);
 99 
100     if (dataSource.get('isActive')) {
101       jquery.addClass('active');
102     }
103 
104     jquery.attr('aria-pressed', dataSource.get('isActive').toString());
105 
106     jquery.setClass('icon', !!dataSource.get('icon') || NO);
107     jquery.setClass('def', dataSource.get('isDefault') || NO);
108     jquery.setClass('cancel', dataSource.get('isCancel') || NO);
109 
110     dataSource.get('theme').labelRenderDelegate.update(dataSource, jquery.find('label'));
111   }
112 });
