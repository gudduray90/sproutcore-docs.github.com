  8 sc_require('views/segmented');
  9 
 10 /**
 11   @static
 12   @type String
 13   @constant
 14 */
 15 SC.TOP_LOCATION = 'top';
 16 
 17 /**
 18   @static
 19   @type String
 20   @constant
 21 */
 22 SC.TOP_TOOLBAR_LOCATION = 'top-toolbar';
 23 
 24 /**
 25   @static
 26   @type String
 27   @constant
 28 */
 29 SC.BOTTOM_LOCATION = 'bottom';
 30 
 31 /**
 32   @class
 33 
 34   Incorporates a segmented view and a container view to display the selected
 35   tab.  Provide an array of items, which will be passed onto the segmented
 36   view.
 37 
 38   @extends SC.View
 39   @since SproutCore 1.0
 40 */
 41 SC.TabView = SC.View.extend(
 42 /** @scope SC.TabView.prototype */ {
 43 
 44   /**
 45     @type Array
 46     @default ['sc-tab-view']
 47     @see SC.View#classNames
 48   */
 49   classNames: ['sc-tab-view'],
 50 
 51   /**
 52     @type Array
 53     @default ['nowShowing']
 54     @see SC.View#displayProperties
 55   */
 56   displayProperties: ['nowShowing'],
 57 
 58   // ..........................................................
 59   // PROPERTIES
 60   //
 61 
 62  /**
 63     Set nowShowing with the page you want to display.
 64 
 65     @type String
 66     @default null
 67   */
 68   nowShowing: null,
 69 
 70   /**
 71     @type Array
 72     @default []
 73   */
 74   items: [],
 75 
 76   /**
 77     @type Boolean
 78     @default YES
 79   */
 80   isEnabled: YES,
 81 
 82   /**
 83     @type String
 84     @default null
 85   */
 86   itemTitleKey: null,
 87 
 88   /**
 89     @type String
 90     @default null
 91   */
 92   itemValueKey: null,
 93 
 94   /**
 95     @type String
 96     @default null
 97   */
 98   itemIsEnabledKey: null,
 99 
100   /**
101     @type String
102     @default null
103   */
104   itemIconKey: null,
105 
106   /**
107     @type String
108     @default null
109   */
110   itemWidthKey: null,
111 
112   /**
113     @type String
114     @default null
115   */
116   itemToolTipKey: null,
117 
118   /**
119     @type Number
120     @default SC.REGULAR_BUTTON_HEIGHT
121   */
122   tabHeight: SC.REGULAR_BUTTON_HEIGHT,
123 
124   /**
125     Possible values:
126 
127       - SC.TOP_LOCATION
128       - SC.TOP_TOOLBAR_LOCATION
129       - SC.BOTTOM_LOCATION
130 
131     @type String
132     @default SC.TOP_LOCATION
133   */
134   tabLocation: SC.TOP_LOCATION,
135 
136   /**
137     If set, then the tab location will be automatically saved in the user
138     defaults.  Browsers that support localStorage will automatically store
139     this information locally.
140 
141     @type String
142     @default null
143   */
144   userDefaultKey: null,
145 
146 
147   // ..........................................................
148   // FORWARDING PROPERTIES
149   //
150 
151   // forward important changes on to child views
152   /** @private */
153   _tab_nowShowingDidChange: function() {
154     var v = this.get('nowShowing');
155     this.get('containerView').set('nowShowing',v);
156     this.get('segmentedView').set('value',v);
157     return this ;
158   }.observes('nowShowing'),
159 
160   /** @private */
161   _tab_saveUserDefault: function() {
162     // if user default is set, save also
163     var v = this.get('nowShowing');
164     var defaultKey = this.get('userDefaultKey');
165     if (defaultKey) {
166       SC.userDefaults.set([defaultKey,'nowShowing'].join(':'), v);
167     }
168   }.observes('nowShowing'),
169 
170   /** @private */
171   _tab_itemsDidChange: function() {
172     this.get('segmentedView').set('items', this.get('items'));
173     return this ;
174   }.observes('items'),
175 
176   /** @private
177     Restore userDefault key if set.
178   */
179   init: function() {
180     sc_super();
181     this._tab_nowShowingDidChange()._tab_itemsDidChange();
182   },
183 
184   /** @private */
185   awake: function() {
186     sc_super();
187     var defaultKey = this.get('userDefaultKey');
188     if (defaultKey) {
189       defaultKey = [defaultKey,'nowShowing'].join(':');
190       var nowShowing = SC.userDefaults.get(defaultKey);
191       if (!SC.none(nowShowing)) this.set('nowShowing', nowShowing);
192     }
193 
194   },
195 
196   /** @private */
197   createChildViews: function() {
198     var childViews  = [], view, containerView, layout,
199         tabLocation = this.get('tabLocation'),
200         tabHeight   = this.get('tabHeight'),
201         controlSize = this.get('controlSize');
202 
203     if (tabLocation === SC.TOP_LOCATION) {
204       layout = { top: tabHeight/2+1, left: 0, right: 0, bottom: 0, border: 1 };
205     } else if (tabLocation === SC.TOP_TOOLBAR_LOCATION) {
206       layout = { top: tabHeight+1, left: 0, right: 0, bottom: 0, border: 1 };
207     } else {
208       layout = { top: 0, left: 0, right: 0, bottom: (tabHeight/2) - 1, border: 1 };
209     }
210 
211     containerView = this.containerView.extend({
212       layout: layout,
213       //adding the role
214       ariaRole: 'tabpanel'
215     });
216 
217     view = this.containerView = this.createChildView(containerView) ;
218     childViews.push(view);
219 
220     //  The segmentedView managed by this tab view.  Note that this TabView uses
221     //  a custom segmented view.  You can access this view but you cannot change
222     // it.
223     layout = (tabLocation === SC.TOP_LOCATION ||
224               tabLocation === SC.TOP_TOOLBAR_LOCATION) ?
225              { height: tabHeight, left: 0, right: 0, top: 0 } :
226              { height: tabHeight, left: 0, right: 0, bottom: 0 } ;
227 
228     this.segmentedView = this.get('segmentedView').extend({
229       layout: layout,
230 
231       controlSize: controlSize,
232 
233       /** @private
234         When the value changes, update the parentView's value as well.
235       */
236       _sc_tab_segmented_valueDidChange: function() {
237         var pv = this.get('parentView');
238         if (pv) pv.set('nowShowing', this.get('value'));
239       }.observes('value'),
240 
241       /** @private */
242       init: function() {
243         // before we setup the rest of the view, copy key config properties
244         // from the owner view...
245         var pv = this.get('parentView');
246         if (pv) {
247           SC._TAB_ITEM_KEYS.forEach(function(k) { this[k] = pv.get(k); }, this);
248         }
249         return sc_super();
250       }
251     });
252 
253     view = this.segmentedView = this.createChildView(this.segmentedView) ;
254     childViews.push(view);
255 
256     this.set('childViews', childViews);
257     return this;
258   },
259 
260   // ..........................................................
261   // COMPONENT VIEWS
262   //
263 
264   /**
265     The containerView managed by this tab view.  Note that TabView uses a
266     custom container view.  You can access this view but you cannot change
267     it.
268 
269     @type SC.View
270     @default SC.ContainerView
271     @readOnly
272   */
273   containerView: SC.ContainerView.extend({ renderDelegateName: 'wellRenderDelegate' }),
274 
275   /**
276     @type SC.View
277     @default SC.SegmentedView
278   */
279   segmentedView: SC.SegmentedView
280 
281 }) ;
282 
283 SC._TAB_ITEM_KEYS = ['itemTitleKey', 'itemValueKey', 'itemIsEnabledKey', 'itemIconKey', 'itemWidthKey', 'itemToolTipKey', 'itemActionKey', 'itemTargetKey'];
