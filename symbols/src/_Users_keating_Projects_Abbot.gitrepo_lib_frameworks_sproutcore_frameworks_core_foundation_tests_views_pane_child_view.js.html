  7 module("SC.Pane - childViews");
  8 
  9 test("SC.Pane should not attempt to recompute visibility on child views that do not have visibility support", function() {
 10   var pane = SC.Pane.create({
 11     childViews: ['noVisibility'],
 12 
 13     noVisibility: SC.CoreView
 14   });
 15 
 16   // tomdale insists on slowing down the tests with extra scope chain traversals
 17   var errored = NO;
 18 
 19   try {
 20     pane.append();
 21   } catch(e) {
 22     errored = YES;
 23   }
 24 
 25   ok(!errored, "appending a pane with child views without visibility does not result in an error");
 26 });
