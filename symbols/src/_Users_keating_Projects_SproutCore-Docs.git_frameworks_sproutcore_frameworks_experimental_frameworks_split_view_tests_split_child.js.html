  8 /*
  9   Tests SplitView Child logic. This covers properties which are interpreted
 10   by the SplitChild mixin itself—not the properties used by the SplitView.
 11   
 12   As such, we look at:
 13   
 14   - layout
 15   - positionOffset
 16   - sizeOffset
 17   - splitChildLayoutDidChange
 18   - splitView calculation
 19   - splitViewLayoutDirection property
 20   
 21   All tests are run twice: once in horizontal, once in vertical.
 22 */
 23 
 24 
 25 var splitView;
 26 
 27 function setupSuite(layoutDirection) {
 28 module("SplitView - SplitChild (" + (layoutDirection === SC.LAYOUT_HORIZONTAL ? "HORIZONTAL" : "VERTICAL") + ")", {
 29   setup: function() {
 30     SC.RunLoop.begin();
 31     splitView = SC.SplitView.create({
 32       childViews: [ 'left', 'middle', 'right' ],
 33       
 34       left:  SC.View.extend(SC.SplitChild, { name: 'left', size: 100 }),
 35       middle: SC.View.extend(SC.SplitChild, { name: 'middle', size: 300, positionOffset: -10, sizeOffset: 20 }),
 36       right: SC.View.extend(SC.SplitChild, { name: 'right', size: 100 }),
 37       
 38       layout: { 
 39         left: 0, top: 0, 
 40         width: layoutDirection === SC.LAYOUT_HORIZONTAL ? 500 : 300, 
 41         height: layoutDirection === SC.LAYOUT_HORIZONTAL ? 300 : 500 
 42       },
 43       
 44       layoutDirection: layoutDirection,
 45       
 46       splitDividerView: null // in most of these tests, we are not checking the behavior of split dividers 
 47     });
 48     SC.RunLoop.end();
 49   }
 50 });
 51 
 52 function verifyChildren(view) {
 53   var pos = 0;
 54   for (var i = 1; i < arguments.length; i++) {
 55     var child = view.childViews[i - 1];
 56     equals(child.get('position'), pos, "Correct position for view " + i);
 57     equals(child.get('size'), arguments[i], "Correct size for view " + i);
 58     
 59     var cpos = pos + child.get('positionOffset');
 60     var csize = arguments[i] + child.get('sizeOffset');
 61     var layout = child.get('layout');
 62     
 63     // check layout, and consider offset
 64     if (splitView.get('layoutDirection') === SC.LAYOUT_HORIZONTAL) {
 65       equals(layout.left, cpos, "Correct left for view " + i);      
 66       equals(layout.width, csize, "Correct width for view " + i);
 67 
 68       equals(layout.top, 0, "Top is 0 in LAYOUT_HORIZONTAL");
 69       equals(layout.bottom, 0, "Bottom is 0 in LAYOUT_HORIZONTAL");
 70     } else {
 71       equals(layout.top, cpos, "Correct top for view " + i);
 72       equals(layout.height, csize, "Correct height for view " + i);
 73 
 74       equals(layout.left, 0, "Left is 0 in LAYOUT_VERTICAL");
 75       equals(layout.right, 0, "Right is 0 in LAYOUT_VERTICAL");
 76     }
 77     
 78     pos += arguments[i];
 79   }
 80 }
 81 
 82 test("Layout gets applied correctly for child views.", function() {
 83   verifyChildren(splitView, 100, 300, 100);
 84   
 85   // double-check the position/sizeOffset
 86   var cv = splitView.get('childViews');
 87   equals(cv[1].get('positionOffset'), -10, "Middle child has proper position offset");
 88   equals(cv[1].get('sizeOffset'), 20, "Middle child has proper size offset");
 89   
 90   if (layoutDirection === SC.LAYOUT_HORIZONTAL) {
 91     equals(cv[1].get('layout').left, 90, "Middle child has proper (offsetted) position in layout");
 92     equals(cv[1].get('layout').width, 320, "Middle child has proper (offsetted) size in layout");
 93   } else {
 94     equals(cv[1].get('layout').top, 90, "Middle child has proper (offsetted) position in layout");
 95     equals(cv[1].get('layout').height, 320, "Middle child has proper (offsetted) size in layout");
 96   }
 97   
 98 });
 99 
100 test("Check that layout adjusts after adjusting child view position", function() {
101   SC.RunLoop.begin();
102   splitView.adjustPositionForChild(splitView.childViews[1], 200);
103   SC.RunLoop.end();
104   
105   verifyChildren(splitView, 200, 200, 100);
106 });
107 
108 test("Check that changing orientation changes layouts.", function() {
109   equals(splitView.get('layoutDirection'), layoutDirection);
110 
111   var childLayoutDirection = splitView.childViews[0].get('splitViewLayoutDirection');
112   equals(childLayoutDirection, layoutDirection, "Child has correct layout direction before orientation change.");
113   
114   var newLayoutDirection = layoutDirection === SC.LAYOUT_HORIZONTAL ? SC.LAYOUT_VERTICAL : SC.LAYOUT_HORIZONTAL;
115   
116   SC.RunLoop.begin();
117   splitView.set(
118     'layoutDirection', 
119     newLayoutDirection
120   );
121   SC.RunLoop.end();
122   
123   equals(splitView.get('layoutDirection'), newLayoutDirection);
124 
125   childLayoutDirection = splitView.childViews[0].get('splitViewLayoutDirection');
126   equals(childLayoutDirection, newLayoutDirection, "Child has correct layout direction after orientation change.");  
127   
128   // height is different, so layout should have changed to 100, 100, 100
129   verifyChildren(splitView, 100, 100, 100);
130 });
131 
132 }
133 
134 setupSuite(SC.LAYOUT_HORIZONTAL);
135 setupSuite(SC.LAYOUT_VERTICAL);
