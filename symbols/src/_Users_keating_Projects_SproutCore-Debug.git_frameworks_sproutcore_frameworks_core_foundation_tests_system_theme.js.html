 11 var Ace, Dark, Capsule, DarkCapsule, AceOnly;
 12 
 13 module("SC.Theme", {
 14   setup: function() {
 15     // make and register Ace
 16     Ace = SC.Theme.create({
 17       name: 'ace',
 18       "classNames": ["ace"]
 19     });
 20     SC.Theme.addTheme(Ace);
 21     
 22     // Dark
 23     Dark = Ace.subtheme("dark");
 24     
 25     // Capsule
 26     Capsule = Ace.subtheme("capsule");
 27   },
 28   
 29   teardown: function() {
 30     
 31   }
 32 });
 33 
 34 // use this utility to check themes
 35 function themeIs(themeInstance, shouldBe, classNames) {
 36   ok(themeInstance === shouldBe, "Correct theme");
 37   if (!themeInstance) return;
 38 
 39   var isOk = same(themeInstance.classNames, SC.Set.create(classNames), "Correct class names.");
 40 }
 41 
 42 test("Calling SC.Theme.find finds proper theme.", function(){
 43   var ace = SC.Theme.find("ace");
 44   themeIs(ace, Ace, ["ace"]);
 45 });
 46 
 47 test("There is no proliferation of theme registration (that is, subthemes are not registered globally)", function(){
 48   var dark = SC.Theme.find("dark");
 49   ok(!dark, "Theme should not be found.");
 50 });
 51 
 52 test("Calling find on a subtheme class finds proper theme.", function(){
 53   var dark = Ace.find("dark");
 54 
 55   // child themes are specialized
 56   themeIs(dark.baseTheme, Dark, ["ace", "dark"]);
 57 });
 58 
 59 test("Calling find on a theme instance finds proper theme.", function(){
 60   var ace = SC.Theme.find("ace");
 61   var dark = ace.find("dark");
 62 
 63  64   // child themes are specialized (that means derived by the parent theme)
 65   themeIs(dark.baseTheme, Dark, ["ace", "dark"]);
 66 });
 67 
 68 test("Calling find on a subtheme instance finds the proper theme.", function(){
 69   var dark = Ace.find("dark");
 70   var capsule = dark.find("capsule");
 71 
 72   // child themes are specialized (that is, derived by the parent)
 73   themeIs(capsule.baseTheme, Capsule, ["ace", "capsule"]);
 74 
 75   // and now we are testing said specialization
 76   themeIs(capsule, capsule, ["ace", "dark", "capsule"]);
 77 });
