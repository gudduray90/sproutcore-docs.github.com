1 var button, rootResponder, pane, expectedAction, targetObject, sendActionSpy;
  2 
  3 module("SC.Button", {
  4   setup: function() {
  5     expectedAction = 'myAction';
  6     targetObject = SC.Object.create({

  8       }
  9     });
 10

 13 
 14     pane = SC.Object.create({

 23   }
 24 });

 26 test("#mouseUp - triggers the action when mouse is over the button", function() {
