  8 // This file tests both SC.DateTime which is in the foundation framework and
  9 // SC.RecordAttribute which is in the datastore framework. The desktop
 10 // framework might not be the best place for it but it works because the
 11 // desktop framework requires both datestore and foundation frameworks.
 12 
 13 var sprocket, nullSprocket, nullUnixTime, d1, d2, d3;
 14 
 15 module('SC.DateTime transform', {
 16 
 17   setup: function() {
 18 
 19     d1 = SC.DateTime.create({ year: 2009, month: 3, day: 1, hour: 20, minute: 30, timezone: 480 });
 20     d2 = SC.DateTime.create({ year: 2009, month: 3, day: 1, hour: 20, minute: 30, timezone: SC.DateTime.timezone });
 21     d3 = SC.DateTime.create({ year: 2009, month: 3, day: 1, hour: 20, minute: 30, timezone: 0 });
 22 
 23     var MyApp = SC.Object.create({
 24       store: SC.Store.create()
 25     });
 26 
 27     MyApp.Sprocket = SC.Record.extend({
 28       createdAt: SC.Record.attr(SC.DateTime),
 29       frenchCreatedAt: SC.Record.attr(SC.DateTime, { format: '%d/%m/%Y %H:%M:%S' }),
 30       unixTimeCreatedAt: SC.Record.attr(SC.DateTime, { useUnixTime: YES })
 31     });
 32 
 33     SC.RunLoop.begin();
 34     MyApp.store.loadRecords(MyApp.Sprocket, [
 35       {
 36         guid: '1',
 37         createdAt: '2009-03-01T20:30:00-08:00',
 38         frenchCreatedAt: '01/03/2009 20:30:00',
 39         unixTimeCreatedAt: 1235939400
 40       },
 41       {
 42         guid: '2',
 43         createdAt: null,
 44         frenchCreatedAt: null,
 45         unixTimeCreatedAt: 'invalidValue'
 46       },
 47       {
 48         guid: '3',
 49         createdAt: null,
 50         frenchCreatedAt: null,
 51         unixTimeCreatedAt: null
 52       }
 53     ]);
 54     SC.RunLoop.end();
 55 
 56     sprocket = MyApp.store.find(MyApp.Sprocket, '1');
 57     nullSprocket = MyApp.store.find(MyApp.Sprocket, '2');
 58     nullUnixTime = MyApp.store.find(MyApp.Sprocket, '3');
 59   }
 60 
 61 });
 62 
 63 test("reading a DateTime should successfully parse the underlying string value", function() {
 64   equals(sprocket.get('createdAt'), d1, 'reading a DateTime should return the correct SC.DateTime object');
 65   equals(sprocket.get('frenchCreatedAt'), d2, 'reading a DateTime with a custom format should return the correct SC.DateTime object');
 66 });
 67 
 68 test("writing a DateTime should successfully format the value into a string", function() {
 69   d1 = d1.advance({ year: 1, hour: 2, minute: 28 });
 70   d2 = d2.advance({ month: -2, minute: 16 });
 71 
 72   sprocket.set('createdAt', d1);
 73   sprocket.set('frenchCreatedAt', d2);
 74 
 75   equals(sprocket.readAttribute('createdAt'), '2010-03-01T22:58:00-08:00', 'writing a DateTime should successfully format the value into the a string');
 76   equals(sprocket.readAttribute('frenchCreatedAt'), '01/01/2009 20:46:00', 'writing a DateTime with a custom format should successfully format the value into the a string');
 77 });
 78 
 79 test("reading or writing null values should work", function() {
 80   sprocket.set('createdAt', null);
 81   equals(sprocket.readAttribute('createdAt'), null, 'Setting a date attribute to null should work');
 82 
 83   equals(nullSprocket.get('createdAt'), null, 'Reading a null date attribute should work');
 84   equals(nullUnixTime.get('unixTimeCreatedAt'), null, 'Reading a null date attribute with useUnixTime: YES should work');
 85 });
 86 
 87 test("reading and writing a DateTime should successfully convert to/from unix time", function() {
 88   // Reading test
 89   equals(sprocket.get('unixTimeCreatedAt'), d3, 'reading a DateTime stored in unix time format should return the correct SC.DateTime object');
 90 
 91   // Writing test
 92   d3 = d3.advance({ year: 2, month: 7, day: 14, hour: -3 });
 93   sprocket.set('unixTimeCreatedAt', d3);
 94   equals(sprocket.readAttribute('unixTimeCreatedAt'), 1318699800, 'writing a DateTime attribute that stored in unix time format should store the correct attribute');
 95 });
 96 
 97 test("unix time should default to 0 ms when invalid value is provided", function() {
 98   equals(nullSprocket.get('unixTimeCreatedAt'), SC.DateTime.create({ milliseconds: 0, timezone: 0 }), 'reading a DateTime store in unix time should default to 0ms when an invalid value is provided');
 99 });
