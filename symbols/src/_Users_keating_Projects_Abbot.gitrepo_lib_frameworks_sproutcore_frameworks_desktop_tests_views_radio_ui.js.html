109 pane.verifyButtons = function verifyButtons(view, items) {
110   var radioButtons = view.$('.sc-radio-button');
111   equals(radioButtons.length, items.length, 'number of radio buttons should be %@'.fmt(items.length));
112   
113   var i = 0;
114   radioButtons.forEach(function(radioInput) {
115     var theInput = SC.$(radioInput),
116       idx = parseInt(theInput.attr('index'),0),
117       buttonValue = theInput.attr('value');
118     
119     equals(idx, i, 'radio button #%@ should have field value %@'.fmt(idx, i));
120     
121     equals(theInput.attr('aria-checked'), 'false', 'radio button #%@ should not be checked'.fmt(idx));
122     ok(!theInput.hasClass('disabled'), 'radio button #%@ should not be disabled'.fmt(idx));
123     
124     i++;
125   });
126 };

154 test("basic", function() {
155   
156   var view = pane.view('basic');
157   
158   ok(!view.$().hasClass('disabled'), 'should not have disabled class');
159   pane.verifyButtons(view, itemList);
160   pane.verifyLabels(view, itemList);
161   
162   // Modify the items array in place
163   itemList.replace(0, {title: "Hue", value: "hue", enabled: YES });
164   itemList.replace(1, { title: "Saturation", value: "saturation" });
165   itemList.replace(2, { title: "Brightness", value: "brightness"});
166   
167   // Allow the radio view a chance to update
168   SC.RunLoop.begin().end();
169   
170   pane.verifyButtons(view, itemList);
171   pane.verifyLabels(view, itemList);
172   
173   // Swap out the items array
174   view.set('items', itemList2);
175   
176   // Allow the radio view a chance to update
177   SC.RunLoop.begin().end();
178   
179   pane.verifyButtons(view, itemList2);
180   pane.verifyLabels(view, itemList2);
181   
182   // Reset the items array before moving on
183   itemList.replace(0, { title: "Red", value: "red", enabled: YES });
184   itemList.replace(1, { title: "Green", value: "green" });
185   itemList.replace(2, { title: "Blue", value: "blue" });
186 });
