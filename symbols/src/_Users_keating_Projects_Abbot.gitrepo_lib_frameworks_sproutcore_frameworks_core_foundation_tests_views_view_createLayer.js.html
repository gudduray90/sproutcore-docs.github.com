 12 module("SC.View#createLayer");

 14 test("returns the receiver", function() {
 15   var v= SC.View.create();
 16   equals(v.createLayer(), v, 'returns receiver');
 17 });
 18 
 19 test("calls renderToContext() and sets layer to resulting element", function() {
 20   var v= SC.View.create({
 21     tagName: 'span',
 22     
 23     renderToContext: function(context, firstTime) {
 24       context.push("foo");
 25     }
 26   });
 27   
 28   equals(v.get('layer'), null, 'precondition - has no layer');
 29   v.createLayer();
 30   
 31   var elem = v.get('layer');
 32   ok(!!elem, 'has element now');
 33   equals(elem.innerHTML, 'foo', 'has innerHTML from context');
 34   equals(elem.tagName.toString().toLowerCase(), 'span', 'has tagName from view');
 35   elem = null ;
 36 });
 37 
 38 test("invokes didCreateLayer() on receiver and all child views", function() {
 39   var callCount = 0, mixinCount = 0;
 40   var v= SC.View.create({
 41     
 42     didCreateLayer: function() { callCount++; },
 43     didCreateLayerMixin: function() { mixinCount++; },
 44     
 45     childViews: [SC.View.extend({
 46       didCreateLayer: function() { callCount++; },
 47       childViews: [SC.View.extend({
 48         didCreateLayer: function() { callCount++; },
 49         didCreateLayerMixin: function() { mixinCount++; }
 50       }), SC.View.extend({ /* no didCreateLayer */ })]
 51     })]
 52   });
 53   
 54   // verify setup...
 55   ok(v.didCreateLayer, 'precondition - has root');
 56   ok(v.childViews[0].didCreateLayer, 'precondition - has firstChild');
 57   ok(v.childViews[0].childViews[0].didCreateLayer, 'precondition - has nested child');
 58   ok(!v.get('layer'), 'has no layer');
 59 
 60   v.createLayer();
 61   equals(callCount, 3, 'did invoke all methods');
 62   equals(mixinCount, 2, 'did invoke all mixin methods');
 63 });

 65 test("generated layer include HTML from child views as well", function() {
 66   var v = SC.View.create({
 67     childViews: [ SC.View.extend({ layerId: "foo" })]
 68   });
 69  70   
 71   v.createLayer();
 72   ok(Q$('#foo', v.get('layer')).get(0), 'has element with child layerId');
 73 });

 75 test("does NOT assign layer to child views immediately", function() {

 77     childViews: [ SC.View.extend({ layerId: "foo" })]
 78   });
