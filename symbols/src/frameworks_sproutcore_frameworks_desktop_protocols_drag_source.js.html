  8 sc_require('system/drag') ;
  9 
 10 /**
 11   @namespace
 12 
 13   If you implement the SC.DragSource protocol on your drag's source, it will
 14   receive a series of callbacks throughout the course of the drag, and be consulted
 15   about what operations to allow on a particular candidate drop target. All methods
 16   are optional.
 17 
 18   (Note that when you initiate a drag you must also provide an object implementing
 19   SC.DragDataSource, which includes some required methods. A single object may
 20   serve as both the drag's source and its data source.)
 21 */
 22 SC.DragSource = {
 23 
 24   /**
 25     Return a bitwise OR'd mask of the drag operations allowed on the
 26     specified target.  If you don't care about the target, just return a
 27     constant value. If a drag's source does not implement this method,
 28     it will assume that any drag operation (SC.DRAG_ANY) is allowed.
 29     
 30     @param {SC.Drag} drag The SC.Drag instance managing this drag.
 31     @param {SC.View} dropTarget The proposed target of the drop.
 32   */
 33   dragSourceOperationMaskFor: function(drag, dropTarget) {
 34     return SC.DRAG_ANY;
 35   },
 36   
 37   /**
 38     If this property is set to `NO` or is not implemented, then the user may
 39     modify the drag operation by changing the modifier keys they have 
 40     pressed.
 41     
 42     @type Boolean
 43     @default NO
 44   */
 45   ignoreModifierKeysWhileDragging: NO,
 46     
 47   /**
 48     This method is called when the drag begins. You can use this to do any
 49     visual highlighting to indicate that the receiver is the source of the 
 50     drag.
 51     
 52     @param {SC.Drag} drag The Drag instance managing this drag.
 53     @param {Point} loc The point in *window* coordinates where the drag 
 54       began.  You can use convertOffsetFromView() to convert this to local 
 55       coordinates.
 56   */
 57   dragDidBegin: function(drag, loc) {},
 58   
 59   /**
 60     This method is called whenever the drag image is moved.  This is
 61     similar to the `dragUpdated()` method called on drop targets.
 62     
 63     @param {SC.Drag} drag The Drag instance managing this drag.
 64     @param {Point} loc  The point in *window* coordinates where the drag 
 65       mouse is.  You can use convertOffsetFromView() to convert this to local 
 66       coordinates.
 67   */
 68   dragDidMove: function(drag, loc) {},
 69 
 70   /**
 71     This method is called if the drag ends and is successfully handled by a
 72     drop target (i.e. the drop target returns any operation other than
 73     SC.DRAG_NONE).
 74 
 75     @param {SC.Drag} drag The drag instance managing the drag.
 76     @param {Point} loc The point in WINDOW coordinates where the drag
 77       ended.
 78     @param {DragOp} op The drag operation that was performed. One of
 79       SC.DRAG_COPY, SC.DRAG_MOVE, or SC.DRAG_LINK.
 80   */
 81   dragDidSucceed: function(drag, loc, op) {},
 82 
 83   /**
 84     This method is called if the drag ends without being handled, or if a drop
 85     target handles it but returns SC.DRAG_NONE.
 86 
 87     @param {SC.Drag} drag The drag instance managing the drag.
 88     @param {Point} loc The point in WINDOW coordinates where the drag
 89       ended.
 90     @param {DragOp} op Provided for consistency. Always SC.DRAG_NONE.
 91   */
 92   dragDidCancel: function(drag, loc, op) {},
 93   
 94   /**
 95     This method is called when the drag ended, regardless of whether it succeeded
 96     or not. You can use this to do any cleanup.
 97     
 98     @param {SC.Drag} drag The drag instance managing the drag.
 99     @param {Point} loc The point in WINDOW coordinates where the drag
100       ended.
101     @param {DragOp} op The drag operation that was performed. One of
102       SC.DRAG_COPY, SC.DRAG_MOVE, SC.DRAG_LINK, or SC.DRAG_NONE.
103   */
104   dragDidEnd: function(drag, loc, op) {},
105 
106   /**
107     If a drag is canceled or not handled, and has its slideBack property set
108     to YES, then the drag's ghost view will slide back to its initial location.
109     dragDidEnd is called immediately upon mouseUp; dragSlideBackDidEnd is called
110     after the slide-back animation completes.
111     
112     @param {SC.Drag} drag The drag instance managing the drag.
113   */
114   dragSlideBackDidEnd: function(drag) {}
115 
116 };
