  8 var store, child, storeKey, json;
  9 module("SC.Store#commitChangesFromNestedStore", {
 10   setup: function() {
 11     SC.RunLoop.begin();
 12 
 13     store = SC.Store.create();
 14 
 15     json = {
 16       string: "string",
 17       number: 23,
 18       bool:   YES
 19     };
 20 
 21     storeKey = SC.Store.generateStoreKey();
 22 
 23     child = store.chain();  // test multiple levels deep
 24 
 25     // write basic status
 26     child.writeDataHash(storeKey, json, SC.Record.READY_DIRTY);
 27     child.dataHashDidChange(storeKey);
 28     child.changelog = SC.Set.create();
 29     child.changelog.add(storeKey);
 30 
 31     SC.RunLoop.end();
 32   }
 33 });
 34 
 35 test("copies changed data hashes, statuses, and revisions", function() {
 36 
 37   SC.RunLoop.begin();
 38 
 39   // verify preconditions
 40   equals(store.readDataHash(storeKey), null, 'precond - should not have data yet');
 41   ok(child.chainedChanges.contains(storeKey), 'precond - child changes should include storeKey');
 42 
 43   // perform action
 44   equals(store.commitChangesFromNestedStore(child, child.chainedChanges, NO), store, 'should return receiver');
 45 
 46   // verify new status
 47   equals(store.readDataHash(storeKey), json, 'now should have json');
 48   equals(store.readStatus(storeKey), SC.Record.READY_DIRTY, 'now should have status');
 49   equals(store.revisions[storeKey], child.revisions[storeKey], 'now shoulave have revision from child');
 50 
 51   SC.RunLoop.end();
 52 });

 54 test("adds items in changelog to receiver changelog", function() {

 87 function createConflict(force) {
 88   var json2 = { kind: "json2" };
 89   var json3 = { kind: "json3" };
 90 
 91   // create a lock conflict.  use a new storeKey since the old one has been
 92   // setup in a way that won't work for this.
 93   storeKey = SC.Store.generateStoreKey();
 94 
 95   // step 1: add data to root store
 96   store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
 97   store.dataHashDidChange(storeKey);
 98 
 99   // step 2: read data in chained store.  this will create lock
100   child.readDataHash(storeKey);
101   ok(child.locks[storeKey], 'child store should now have lock');
102 
103   // step 3: modify root store again
104   store.writeDataHash(storeKey, json2, SC.Record.READY_CLEAN);
105   store.dataHashDidChange(storeKey);
106 
107   // step 4: modify data in chained store so we have something to commit.
108   child.writeDataHash(storeKey, json3, SC.Record.READY_DIRTY);
109   child.dataHashDidChange(storeKey);
110 
111   // just to make sure verify that the lock and revision in parent do not
112   // match
113   ok(child.locks[storeKey] !== store.revisions[storeKey], 'child.lock (%@) should !== store.revision (%@)'.fmt(child.locks[storeKey], store.revisions[storeKey]));
114 
115   // step 5: now try to commit changes from child store.  This should throw
116   // an exception.
117   var errorCount = 0;
118   try {
119     child.commitChanges(force);
120   } catch(e) {
121     equals(e, SC.Store.CHAIN_CONFLICT_ERROR, 'should throw CHAIN_CONFLICT_ERROR');
122     errorCount++;
123   }
124 
125   return errorCount ;
126 }
127 
128 test("throws exception if any record fails optimistic locking test", function() {
129   var errorCount = createConflict(NO);
130   equals(errorCount, 1, 'should have raised error');
131 });
132 
133 test("does not throw exception if optimistic locking fails but force option is passed", function() {
134   var errorCount = createConflict(YES);
135   equals(errorCount, 0, 'should not raise error');
136 });
