  8 sc_require('render_delegates/render_delegate');
  9 
 10 SC.RenderDelegate.reopen({
 11   /**
 12     A list of size names to look for when automatically determining
 13     control size. By default, this has all of the SproutCore control sizes.
 14   */
 15   sizes: [
 16     SC.TINY_CONTROL_SIZE, SC.SMALL_CONTROL_SIZE,
 17     SC.REGULAR_CONTROL_SIZE, SC.LARGE_CONTROL_SIZE,
 18     SC.HUGE_CONTROL_SIZE, SC.JUMBO_CONTROL_SIZE
 19   ],
 20 
 21   /**
 22     Determines the correct size for the given data source, and returns the
 23     hash, if any, representing it.
 24    
 25     The hashes to choose from are properties on the render delegate. You define
 26     them with the same name as you would use for styling. For example,
 27     SC.REGULAR_CONTROL_SIZE uses a property name 'sc-regular-size':
 28    
 29         SC.RenderDelegate.create({
 30           'sc-regular-size': {
 31             // my properties here
 32           }
 33    
 34     If no matching size is found, the hash (if any) for SC.REGULAR_CONTROL_SIZE
 35  36     will be returned.
 37    
 38     @param {DataSource} dataSource The data source in which to find `controlSize`
 39     or `frame` and to determine the size for.
 40    
 41     @returns {Hash undefined}
 42   */
 43   sizeFor: function(dataSource) {
 44     var controlSize = dataSource.get('controlSize'), size, idx, len;
 45 
 46     // if there is a control size set on the control
 47     // then we need to use it, and give an error if we
 48     // don't have it.
 49     if (controlSize) {
 50       if (!this[controlSize]) {
 51         // create a hash for the control size
 52         this[controlSize] = {};
 53       }
 54 
 55       size = this[controlSize];
 56 
 57       // make sure there's a name on the size for use as class name
 58       if (!size.name) {
 59         size.name = controlSize;
 60       }
 61 
 62       return size;
 63     }
 64 
 65     // try to determine control size for the supplied frame
 66     // TODO: cache this in dataSource.renderState
 67     var frame = dataSource.get('frame');
 68     if (!frame) {
 69       size = this['sc-regular-size'];
 70 
 71       // create the size hash if needed
 72       if (!size) { size = this['sc-regular-size'] = {}; }
 73       if (!size.name) { size.name = 'sc-regular-size'; }
 74       return size;
 75     }
 76 
 77     // loop to automatically find size
 78     for (idx = 0; idx < len; idx++) {
 79       key = sizes[idx];
 80       size = this[key];
 81 
 82       // when the size is not defined, skip it.
 83       if (!size) {
 84         continue;
 85       }
 86 
 87       if (
 88         // if no auto-size-selection params are supplied, then we cannot
 89         // automatically select a size...
 90         (
 91           size.width === undefined && size.height === undefined && 
 92           size.minHeight === undefined && size.minWidth === undefined &&
 93           size.maxHeight === undefined && size.maxWidth === undefined
 94         ) ||
 95 
 96         // otherwise, if any are defined and are non-equal
 97         (size.width !== undefined && frame.width !== size.width) ||
 98         (size.minWidth !== undefined && frame.width < size.minWidth) ||
 99         (size.maxWidth !== undefined && frame.width > size.maxWidth) ||
100 
101         (size.height !== undefined && frame.height !== size.height) ||
102         (size.minHeight !== undefined && frame.height < size.minHeight) ||
103         (size.maxHeight !== undefined && frame.height < size.maxHeight)
104       ) {
105         continue;
106       }
107 
108       // the size needs a name to use as a class name. If one is not already
109       // present, set it to the key.
110       if (!size.name) {
111         size.name = key;
112       }
113 
114       return size;
115     }
116 
117     // hardcoded to return regular size if defined
118     size = this['sc-regular-size'];
119 
120     // create the size hash if needed
121     if (!size) { size = this['sc-regular-size'] = {}; }
122     if (!size.name) { size.name = 'sc-regular-size'; }
123 
124 
125     return size;
126   },
127 
128   /**
129     Determines the proper size for the dataSource, and then renders the class
130     name corresponding to that size.
131   */
132   addSizeClassName: function(dataSource, context) {
133     var size = this.sizeFor(dataSource);
134     if (size) {
135       context.addClass(size.name);
136     }
137   },
138 
139   /**
140     Determines the proper size for the dataSource, and then updates
141     the DOM to include that size's class name.
142   */
143   updateSizeClassName: function(dataSource, jquery) {
144     var size = this.sizeFor(dataSource);
145     if (size) {
146       jquery.addClass(size.name);
147     }
148   },
149 
150   /**
151     Retrieves the given property for the specified data source. This property
152     may be static, or may be computed specifically for this data source. This
153     version fo `getPropertyFor` will check in your size hashes to see if any
154     properties have been overriden.
155     
156     @param {DataSource} dataSource The data source to get the property
157     for. Some properties may differ based on the data source; for instance,
158     some may have different values depending on size.
159     @param {String} propertyName The name of the property to retrieve.
160   */
161   getPropertyFor: function(dataSource, propertyName) {
162     var size = this.sizeFor(dataSource);
163     if (size) {
164       if (size[propertyName + 'For']) {
165         return size[propertyName + 'For'](dataSource, propertyName);
166       } else if (size[propertyName] !== undefined) {
167         return size[propertyName];
168       }
169     }
170 
171     if (this[propertyName + 'For']) {
172       return this[propertyName + 'For'];
173     }
174 
175     return this[propertyName];
176   }
177 });
