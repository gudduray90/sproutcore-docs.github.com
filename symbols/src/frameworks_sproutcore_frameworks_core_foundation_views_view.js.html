567   /**
568     @private
569 
570     Iterates over the view's `classNameBindings` array, inserts the value
571     of the specified property into the `classNames` array, then creates an
572     observer to update the view's element if the bound property ever changes
573     in the future.
574   */
575   _applyClassNameBindings: function () {
576     var classBindings = this.get('classNameBindings'),
577         classNames = this.get('classNames'),
578         dasherizedClass;
579 
580     if (!classBindings) { return; }
581 
582     // Loop through all of the configured bindings. These will be either
583     // property names ('isUrgent') or property paths relative to the view
584     // ('content.isUrgent')
585     classBindings.forEach(function (property) {
586 
587       // Variable in which the old class value is saved. The observer function
588       // closes over this variable, so it knows which string to remove when
589       // the property changes.
590       var oldClass;
591 
592       // Set up an observer on the context. If the property changes, toggle the
593       // class name.
594       var observer = function () {
595         // Get the current value of the property
596         var newClass = this._classStringForProperty(property);
597         var elem = this.$();
598 
599         // If we had previously added a class to the element, remove it.
600         if (oldClass) {
601           elem.removeClass(oldClass);
602           classNames.removeObject(oldClass);
603         }
604 
605         // If necessary, add a new class. Make sure we keep track of it so
606         // it can be removed in the future.
607         if (newClass) {
608           elem.addClass(newClass);
609           classNames.push(newClass);
610           oldClass = newClass;
611         } else {
612           oldClass = null;
613         }
614       };
615 
616       this.addObserver(property.split(':')[0], this, observer);
617 
618       // Get the class name for the property at its current value
619       dasherizedClass = this._classStringForProperty(property);
620 
621       if (dasherizedClass) {
622         // Ensure that it gets into the classNames array
623         // so it is displayed when we render.
624         classNames.push(dasherizedClass);
625 
626         // Save a reference to the class name so we can remove it
627         // if the observer fires. Remember that this variable has
628         // been closed over by the observer.
629         oldClass = dasherizedClass;
630       }
631 
632     }, this);
633   },

635   /**
636     Iterates through the view's attribute bindings, sets up observers for each,
637     then applies the current value of the attributes to the passed render buffer.
638 
639     @param {SC.RenderBuffer} buffer
640   */
641   _applyAttributeBindings: function (context) {
642     var attributeBindings = this.get('attributeBindings'),
643         attributeValue, elem, type;
644 
645     if (!attributeBindings) { return; }
646 
647     attributeBindings.forEach(function (attribute) {
648       // Create an observer to add/remove/change the attribute if the
649       // JavaScript property changes.
650       var observer = function () {
651         elem = this.$();
652         var currentValue = elem.attr(attribute);
653         attributeValue = this.get(attribute);
654 
655         type = typeof attributeValue;
656 
657         if ((type === 'string' || type === 'number') && attributeValue !== currentValue) {
658           elem.attr(attribute, attributeValue);
659         } else if (attributeValue && type === 'boolean') {
660           elem.attr(attribute, attribute);
661         } else if (attributeValue === NO) {
662           elem.removeAttr(attribute);
663         }
664       };
665 
666       this.addObserver(attribute, this, observer);
667 
668       // Determine the current value and add it to the render buffer
669       // if necessary.
670       attributeValue = this.get(attribute);
671       type = typeof attributeValue;
672 
673       if (type === 'string' || type === 'number') {
674         context.setAttr(attribute, attributeValue);
675       } else if (attributeValue && type === 'boolean') {
676         // Apply boolean attributes in the form attribute="attribute"
677         context.setAttr(attribute, attribute);
678       }
679     }, this);
680   },

682   /**
683     @private
684 
685     Given a property name, returns a dasherized version of that
686     property name if the property evaluates to a non-falsy value.
687 
688     For example, if the view has property `isUrgent` that evaluates to true,
689     passing `isUrgent` to this method will return `"is-urgent"`.
690   */
691   _classStringForProperty: function (property) {
692     var split = property.split(':'), className = split[1];
693     property = split[0];
694 
695     var val = SC.getPath(this, property);
696 
697     // If value is a Boolean and true, return the dasherized property
698     // name.
699     if (val === YES) {
700       if (className) { return className; }
701 
702       // Normalize property path to be suitable for use
703       // as a class name. For exaple, content.foo.barBaz
704       // becomes bar-baz.
705       return SC.String.dasherize(property.split('.').get('lastObject'));
706 
707     // If the value is not NO, undefined, or null, return the current
708     // value of the property.
709     } else if (val !== NO && val !== undefined && val !== null) {
710       return val;
711 
712     // Nothing to display. Return null so that the old class is removed
713     // but no new class is added.
714     } else {
715       return null;
716     }

1860 /**
1861   @class
1862 
1863   Base class for managing a view.  Views provide two functions:
1864 
1865    1. They translate state and events into drawing instructions for the
1866      web browser and
1867    2. They act as first responders for incoming keyboard, mouse, and
1868      touch events.
1869 
1870   View Initialization
1871   ====
1872 
1873   When a view is setup, there are several methods you can override that
1874   will be called at different times depending on how your view is created.
1875   Here is a guide to which method you want to override and when:
1876 
1877    - `init` -- override this method for any general object setup (such as
1878      observers, starting timers and animations, etc) that you need to happen
1879      every time the view is created, regardless of whether or not its layer
1880      exists yet.
1881    - `render` -- override this method to generate or update your HTML to reflect
1882      the current state of your view.  This method is called both when your view
1883      is first created and later anytime it needs to be updated.
1884    - `didCreateLayer` -- the render() method is used to generate new HTML.
1885      Override this method to perform any additional setup on the DOM you might
1886      need to do after creating the view.  For example, if you need to listen
1887      for events.
1888    - `willDestroyLayer` -- if you implement didCreateLayer() to setup event
1889      listeners, you should implement this method as well to remove the same
1890      just before the DOM for your view is destroyed.
1891    - `updateLayer` -- Normally, when a view needs to update its content, it will
1892      re-render the view using the render() method.  If you would like to
1893      override this behavior with your own custom updating code, you can
1894      replace updateLayer() with your own implementation instead.
1895    - `didAppendToDocument` -- in theory all DOM setup could be done
1896      in didCreateLayer() as you already have a DOM element instantiated.
1897      However there is cases where the element has to be first appended to the
1898      Document because there is either a bug on the browser or you are using
1899      plugins which objects are not instantiated until you actually append the
1900      element to the DOM. This will allow you to do things like registering
1901      DOM events on flash or quicktime objects.
1902

1904   @extends SC.DelegateSupport
1905   @since SproutCore 1.0
1906 
1907 */
