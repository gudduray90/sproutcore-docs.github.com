 10 var content, controller, extra;

 12 var TestObject = SC.Object.extend({
 13   title: "test",  
 14   xFactor: "THETA",
 15   toString: function() { return "TestObject(%@)".fmt(this.get("title")); }

 21 // 
 22 
 23 module("SC.ArrayController - enum_case - EMPTY SET", {
 24   setup: function() {
 25     content = SC.Set.create();
 26     controller = SC.ArrayController.create({ 
 27       content: content, orderBy: "title" 
 28     });
 29     extra = TestObject.create({ title: "FOO", xFactor: "ZED" });
 30   },
 31   
 32   teardown: function() {
 33     controller.destroy();
 34   }
 35 });
 36 
 37 test("state properties", function() {
 38   equals(controller.get("hasContent"), YES, 'c.hasContent');
 39   equals(controller.get("canRemoveContent"), YES, "c.canRemoveContent");
 40   equals(controller.get("canReorderContent"), NO, "c.canReorderContent");
 41   equals(controller.get("canAddContent"), YES, "c.canAddContent");
 42 });
 43 
 44 // addObject should append to end of array + notify observers on Array itself
 45 test("addObject", function() {
 46   var callCount = 0;
 47   controller.addObserver('[]', function() { callCount++; });
 48   
 49   SC.run(function() { controller.addObject(extra); });
 50   
 51   var expected = SC.Set.create().add(extra);
 52   same(content, expected, 'addObject(extra) should work');
 53   equals(callCount, 1, 'should notify observer that content has changed');
 54   equals(content.get('length'), 1, 'should update length of controller');
 55 });
 56 
 57 test("removeObject", function() {
 58   var callCount = 0;
 59   controller.addObserver('[]', function() { callCount++; });
 60   
 61   SC.run(function() { controller.removeObject(extra); });
 62   
 63   same(content, SC.Set.create(), 'removeObject(extra) should have no effect');
 64   equals(callCount, 0, 'should not notify observer since content did not change');
 65 });
 66 
 67 test("basic array READ operations", function() {
 68   equals(controller.get("length"), 0, 'length should be empty');
 69   equals(controller.objectAt(0), undefined, "objectAt() should return undefined");
 70 });
 71 
 72 test("basic array WRITE operations", function() {
 73   var callCount = 0;
 74   controller.addObserver('[]', function() { callCount++; });
 75 
 76   should_throw(function() {
 77     controller.replace(0,1,[extra]);
 78   }, Error, "calling replace on an enumerable should throw");
 79 });
 80 
 81 test("arrangedObjects", function() {
 82   equals(controller.get("arrangedObjects"), controller, 'c.arrangedObjects should return receiver');
 83 });

 89 module("SC.ArrayController - enum_case - NON-EMPTY SET", {
 90   setup: function() {
 91     content = SC.Set.create();
 92     "1 2 3 4 5".w().forEach(function(x) {
 93       content.add(TestObject.create({ title: x, xFactor: (5-x) }));
 94     });
 95     
 96     controller = SC.ArrayController.create({ 
 97       content: content, orderBy: "title" 
 98     });
 99     extra = TestObject.create({ title: "FOO", xFactor: 0 });
100   },
101   
102   teardown: function() {
103     controller.destroy();
104   }
105 });
106 
107 test("state properties", function() {
108   equals(controller.get("hasContent"), YES, 'c.hasContent');
109   equals(controller.get("canRemoveContent"), YES, "c.canRemoveContent");
110   equals(controller.get("canReorderContent"), NO, "c.canReorderContent");
111   equals(controller.get("canAddContent"), YES, "c.canAddContent");
112 });
113 
114 // addObject should regenerate ordered + notify observers on Array itself
115 test("addObject", function() {
116   var expected = content.copy();
117   expected.add(extra);
118   
119   var callCount = 0;
120   controller.addObserver('[]', function() { callCount++; });
121   
122   SC.run(function() { controller.addObject(extra); });
123   
124   same(content, expected, 'addObject(extra) should work');
125   equals(callCount, 1, 'should notify observer that content has changed');
126   equals(content.get('length'), expected.length, 'should update length of controller');
127   
128   var idx, len = controller.get('length');
129   expected = SC.A(expected).sort(function(a,b) { 
130     return SC.compare(a.get('title'), b.get('title')); 
131   });
132 
133   for(idx=0;idx<len;idx++) {
134     equals(controller.objectAt(idx), expected[idx], "controller.objectAt(%@) should be match ordered array %@".fmt(idx,idx));
135   }
136   
137 });
138 
139 test("removeObject", function() {
140   var expected = content.copy(), obj = expected.pop();
141   
142   var callCount = 0;
143   controller.addObserver('[]', function() { callCount++; });
144   
145   equals(controller.get('length'), content.get('length'), 'precond - controller should have same length as content to start');
146   
147   SC.run(function() { controller.removeObject(obj); });
148   
149   same(content, expected, 'removeObject(extra) should remove object');
150   equals(callCount, 1, 'should notify observer that content has changed');
151   equals(content.get('length'), expected.length, 'should update length of content');
152   equals(controller.get('length'), expected.length, 'should update length of controller as well');
153   
154   var idx, len = controller.get('length');
155   expected = SC.A(expected).sort(function(a,b) { 
156     return SC.compare(a.get('title'), b.get('title')); 
157   });
158 
159   for(idx=0;idx<len;idx++) {
160     equals(controller.objectAt(idx), expected[idx], "controller.objectAt(%@) should be match ordered array %@".fmt(idx,idx));
161   }
162 });
163 
164 test("basic array READ operations", function() {
165   equals(controller.get("length"), content.length, 'length should be empty');
166   
167   var expected = SC.A(content).sort(function(a,b) { 
168     return SC.compare(a.get('title'), b.get('title')); 
169   });
170   var loc = expected.length+1; // verify 1 past end as well
171 
172   while(--loc>=0) {
173     equals(controller.objectAt(loc), expected[loc], "objectAt(%@) should return same value at content[%@]".fmt(loc, loc));
174   }
175 });
176 
177 test("basic array WRITE operations", function() {
178   var callCount = 0;
179   controller.addObserver('[]', function() { callCount++; });
180 
181   should_throw(function() {
182     controller.replace(0,1,[extra]);
183   }, Error, "calling replace on an enumerable should throw");
184 });
185 
186 test("arrangedObjects", function() {
187   equals(controller.get("arrangedObjects"), controller, 'c.arrangedObjects should return receiver');
188 });
189 
190 
191 test("modifying orderBy should build order", function() {
192   
193   var cnt = 0 ;
194   controller.addObserver('[]', this, function() { cnt++; });
195   same(controller.getEach('title'), '1 2 3 4 5'.w(), 'initially should be ordered by title');
196   
197   cnt = 0;
198   controller.set('orderBy', 'xFactor');
199   equals(cnt, 1, 'should have fired observer on enumerable');
200   same(controller.getEach('title'), '5 4 3 2 1'.w(), 'should be ordered reverse');
201 });
