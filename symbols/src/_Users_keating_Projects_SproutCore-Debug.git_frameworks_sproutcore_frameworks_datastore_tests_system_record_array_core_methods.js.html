  8 // test core array-mapping methods for RecordArray
  9 var store, storeKey, json, rec, storeKey2, json2, rec2, storeKeys, recs;
 10 module("SC.RecordArray core methods", {
 11   setup: function() {
 12     // setup dummy store
 13     store = SC.Store.create();
 14 
 15     storeKey = SC.Record.storeKeyFor('foo');
 16     json = { guid: "foo", foo: "bar" };
 17     store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
 18 
 19     storeKey2 = SC.Record.storeKeyFor('baz');
 20     json2 = { guid: "baz", baz: "bash" };
 21     store.writeDataHash(storeKey2, json2, SC.Record.READY_CLEAN);
 22 
 23     // get records
 24     rec = store.materializeRecord(storeKey);
 25     equals(rec.get('foo'), 'bar', 'precond - record should have json');
 26     rec2 = store.materializeRecord(storeKey2);
 27     equals(rec2.get('baz'), 'bash', 'precond - record 2 should have json');
 28 
 29     // get record array.
 30     storeKeys = [storeKey, storeKey2];
 31     recs = SC.RecordArray.create({ store: store, storeKeys: storeKeys });
 32   }
 33 });
 34 
 35 test("initial status", function() {
 36   equals(recs.get('status'), SC.Record.EMPTY, 'status should be SC.Record.EMPTY');
 37 });

 70 test("modifying the underlying storeKey should change the returned materialized record", function() {
 71   // read record once to make it materialized
 72   equals(recs.objectAt(0), rec, 'recs.objectAt(0) should materialize record');
 73 
 74   // create a new record.
 75   SC.RunLoop.begin();
 76   var rec3 = store.createRecord(SC.Record, { foo: "rec3" });
 77   SC.RunLoop.end();
 78 
 79   var storeKey3 = rec3.get('storeKey');
 80 
 81   // add to beginning of storeKey array
 82   storeKeys.unshiftObject(storeKey3);
 83   equals(recs.get('length'), 3, 'should now have length of 3');
 84   equals(recs.objectAt(0), rec3, 'objectAt(0) should return new record');
 85   equals(recs.objectAt(1), rec, 'objectAt(1) should return old record 1');
 86   equals(recs.objectAt(2), rec2, 'objectAt(2) should return old record 2');
 87 });

 89 test("reading a record not loaded in store should trigger retrieveRecord", function() {
 90   var callCount = 0;
 91 
 92   // patch up store to record a call and to make it look like data is not
 93   // loaded.
 94   store.removeDataHash(storeKey, SC.Record.EMPTY);
 95   store.retrieveRecord = function() { callCount++; };
 96 
 97   equals(store.peekStatus(storeKeys.objectAt(0)), SC.Record.EMPTY, 'precond - storeKey must not be loaded');
 98 
 99   var rec = recs.objectAt(0);
100   equals(callCount, 1, 'store.retrieveRecord() should have been called');
101 });

107 test("adding a record to the record array should pass through storeKeys", function() {
108   // read record once to make it materialized
109   equals(recs.objectAt(0), rec, 'recs.objectAt(0) should materialize record');
110 
111   // create a new record.
112   SC.RunLoop.begin();
113   var rec3 = store.createRecord(SC.Record, { foo: "rec3" });
114   SC.RunLoop.end();
115 
116   var storeKey3 = rec3.get('storeKey');
117 
118   // add record to beginning of record array
119   recs.unshiftObject(rec3);
120 
121   // verify record array
122   equals(recs.get('length'), 3, 'should now have length of 3');
123   equals(recs.objectAt(0), rec3, 'recs.objectAt(0) should return new record');
124   equals(recs.objectAt(1), rec, 'recs.objectAt(1) should return old record 1');
125   equals(recs.objectAt(2), rec2, 'recs.objectAt(2) should return old record 2');
126 
127   // verify storeKeys
128   equals(storeKeys.objectAt(0), storeKey3, 'storeKeys[0] should return new storeKey');
129   equals(storeKeys.objectAt(1), storeKey, 'storeKeys[1] should return old storeKey 1');
130   equals(storeKeys.objectAt(2), storeKey2, 'storeKeys[2] should return old storeKey 2');
131 });

137 test("changing the underlying storeKeys should notify observers of records", function() {
138 
139   // setup observer
140   var obj = SC.Object.create({
141     cnt: 0,
142     observer: function() { this.cnt++; }
143   });
144   recs.addObserver('[]', obj, obj.observer);
145 
146   // now modify storeKeys
147   storeKeys.pushObject(SC.Store.generateStoreKey());
148   equals(obj.cnt, 1, 'observer should have fired after changing storeKeys');
149 });
150 
151 test("swapping storeKey array should change recordArray and observers", function() {
152 
153   // setup alternate storeKeys
154   SC.RunLoop.begin();
155   var rec2 = store.createRecord(SC.Record, { foo: "rec2" });
156   SC.RunLoop.end();
157 
158   var storeKey2 = rec2.get('storeKey');
159   var storeKeys2 = [storeKey2];
160 
161   // setup observer
162   var obj = SC.Object.create({
163     cnt: 0,
164     observer: function() { this.cnt++; }
165   });
166   recs.addObserver('[]', obj, obj.observer);
167 
168   // read record once to make it materialized
169   equals(recs.objectAt(0), rec, 'recs.objectAt(0) should materialize record');
170 
171   // now swap storeKeys
172   obj.cnt = 0 ;
173   recs.set('storeKeys', storeKeys2);
174 
175   // verify observer fired and record changed
176   equals(obj.cnt, 1, 'observer should have fired after swap');
177   equals(recs.objectAt(0), rec2, 'recs.objectAt(0) should return new rec');
178 
179   // modify storeKey2, make sure observer fires and content changes
180   obj.cnt = 0;
181   storeKeys2.unshiftObject(storeKey);
182   equals(obj.cnt, 1, 'observer should have fired after edit');
183   equals(recs.get('length'), 2, 'should reflect new length');
184   equals(recs.objectAt(0), rec, 'recs.objectAt(0) should return pushed rec');
185 
186 });
