 10 var pane = SC.ControlTestPane.design()
 11   .add("aria-role", SC.InlineTextFieldView.extend({
 12       localize: YES
 13     })
 14   )
 15 
 16   .add("aria-multiline", SC.InlineTextFieldView.extend({
 17       isTextArea: YES
 18     })
 19   )
 20   
 21   .add("aria-disabled", SC.InlineTextFieldView.extend({
 22       isEnabled: NO
 23     })
 24   )
 25   
 26   .add("aria-invalid", SC.InlineTextFieldView.extend({
 27       value: SC.Error.create({errorValue:'Error Message'})
 28     })
 29   );
 30 pane.show();
 31 
 32 module('SC.InlineTextFieldView ui', {
 33   setup: function() {
 34     htmlbody('<style> .sc-static-layout { border: 1px red dotted; } </style>');
 35     var view = pane.view('aria-role');
 36   },
 37   teardown: function(){
 38     clearHtmlbody();
 39   }
 40 });
 41 
 42 test("should have role as textbox", function() {
 43   var view = pane.view('aria-role');
 44   var label = view.$();  
 45   equals(label.attr('role'), 'textbox', 'role should be textbox');
 46 });
 47 
 48 test("should have aria-multiline as YES", function() {
 49   var view = pane.view('aria-multiline');
 50   var label = view.$();  
 51   equals(label.attr('aria-multiline'), 'true', 'aria-multiline should be true');
 52 });
 53 
 54 test("should have aria-disabled as YES", function() {

 58 });
 59

 64 });
