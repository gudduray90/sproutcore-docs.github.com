 13 module("SC.ScrollView", {
 14   setup: function () {
 15     SC.run(function () {
 16       pane = SC.MainPane.create({
 17         layout: { height: 114, width: 114 }, // Gives us 100x100 container views without worrying about scrollers (or resorting to horrible, counterproductive hacks)
 18         childViews: [
 19           // ScrollView with 4000x4000 contentView. "view" below.
 20           SC.ScrollView.extend({
 21             contentView: SC.ImageView.design({ value: appleURL, layout: { height: 4000, width: 4000 }}),
 22             horizontalAlign: SC.ALIGN_LEFT
 23           }),
 24           // ScrollView with 2000x2000 contentView. "view2" below.
 25           SC.ScrollView.extend({
 26             contentView: SC.ImageView.design({ value: appleURL, layout: { height: 2000, width: 2000 }}),
 27             horizontalAlign: SC.ALIGN_LEFT
 28           }),
 29           // ScrollView (view3 below) with nested ScrollView (view4 below).
 30           SC.ScrollView.extend({
 31             layout: { height: 400, width: 400 },
 32             horizontalAlign: SC.ALIGN_LEFT,
 33             contentView: SC.View.design({
 34               layout: { height: 500, width: 500 },
 35               childViews: [
 36                 SC.ScrollView.design({
 37                   layout: { height: 200, width: 200, centerX: 0, centerY: 0 },
 38                   contentView: SC.ImageView.design({ value: appleURL, layout: { height: 300, width: 300 }}),
 39                   horizontalAlign: SC.ALIGN_LEFT
 40                 })
 41               ]
 42             })
 43           })
 44         ],
 45 
 46         expectedVertLine: function (line) {
 47           var ret = view.get('verticalLineScroll') * line;
 48           var alt = view.get('maximumVerticalScrollOffset');
 49           ret = (ret > alt) ? alt : ret;
 50 
 51           return ret;
 52         },
 53 
 54         expectedHorzLine: function (line) {
 55           var ret = view.get('horizontalLineScroll') * line;
 56           var alt = view.get('maximumHorizontalScrollOffset');
 57           ret = (ret > alt) ? alt : ret;
 58 
 59           return ret;
 60         },
 61 
 62         expectedVertPage: function (page) {
 63           var ret = view.get('verticalPageScroll') * page;
 64           var alt = view.get('maximumVerticalScrollOffset');
 65           ret = (ret > alt) ? alt : ret;
 66 
 67           return ret;
 68         },
 69 
 70         expectedHorzPage: function (page) {
 71           var ret = view.get('horizontalPageScroll') * page;
 72           var alt = view.get('maximumHorizontalScrollOffset');
 73           ret = (ret > alt) ? alt : ret;
 74 
 75           return ret;
 76         }
 77       });
 78 
 79       pane.append(); // make sure there is a layer...
 80     });
 81 
 82     view = pane.childViews[0];
 83 
 84     view2 = pane.childViews[1];
 85 
 86     view3 = pane.childViews[2];
 87     view4 = view3.get('contentView').get('childViews')[0];
 88   },
 89 
 90   teardown: function () {
 91     SC.run(function () {
 92       pane.destroy();
 93     });
 94     pane = view = view2 = view3 = view4 = null;
 95   }
 96 });

701   // Exceeding max
702   SC.run(function() {
703     view.set('scale', 100);
