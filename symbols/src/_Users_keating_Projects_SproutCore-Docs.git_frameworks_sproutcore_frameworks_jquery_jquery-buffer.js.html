  9   var Buffer = function(elem) {
 10     if (elem) this.assign(elem);
 11 
 12     // the internal buffer
 13     this._bufferedCommandList = [];
 14     this._bufferedCommands = {};
 15   };
 16 
 17   // the set of buffers
 18   Buffer._buffers = [];
 19   Buffer._pool = [];
 20 
 21   /**
 22     If there is already a buffer for the element, returns that. Otherwise, creates a new one.
 23   */
 24   Buffer.bufferForElement = function(elem) {
 25     if (elem._jquery_buffer) {
 26       return elem._jquery_buffer;
 27     }
 28     return this.bufferFromPool().assign(elem);
 29   };
 30 
 31   Buffer.bufferFromPool = function() {
 32     var buffer = null;
 33     if (this._pool.length === 0) {
 34       buffer = new Buffer();
 35     } else {
 36       buffer = this._pool.pop();
 37     }
 38 
 39     // add buffer
 40     Buffer._buffers.push(buffer);
 41     if (!this.flushingScheduled) this.scheduleFlushing();
 42     return buffer;
 43   };
 44 
 45   Buffer.returnToPool = function(buffer) {
 46     buffer.unassign();
 47     this._pool.push(buffer);
 48   };
 49 
 50   Buffer.scheduleFlushing = function() {
 51     this.flushingScheduled = true;
 52   };
 53 
 54   /**
 55     Flushes all of the buffers.
 56   */
 57   Buffer.flush = function() {
 58     // get the buffers
 59     var buffers = this._buffers, idx, len = buffers.length;
 60     for (idx = 0; idx < len; idx++) {
 61       buffers[idx].flush();
 62       this.returnToPool(buffers[idx]);
 63     }
 64     this._buffers = [];
 65     this.flushingScheduled = false;
 66   };
 67 
 68   /**
 69     Assigns an element to a buffer.
 70   */
 71   Buffer.prototype.assign = function(elem) {
 72     if (this._el) this.unassign();
 73 
 74     this._el = elem;
 75     this._el._jquery_buffer = this;
 76     return this;
 77   };
 78 
 79   Buffer.prototype.unassign = function() {
 80     if (!this._el) return;
 81     this._el._jquery_buffer = undefined;
 82     this._el = undefined;
 83     return this;
 84   };
 85 
 86   /**
 87     Flushes a buffer.
 88   */
 89   Buffer.prototype.flush = function() {
 90     var commands = this._bufferedCommandList, len = commands.length, idx, c;
 91     for (idx = 0; idx < len; idx++) {
 92       // get command name
 93       c = commands[idx];
 94 
 95       // run command
 96       this[c](this._bufferedCommands[c]);
 97 
 98       // remove from set now that we have run it.
 99       delete this._bufferedCommands[c];
100     }
101 
102     this._bufferedCommandList.length = 0;
103     this.unassign();
104   };

106   Buffer.prototype.$ = function(selector, context) {
107     if (!context) context = this._el;
108     if (selector === "" || selector === undefined) {
109       selector = context; context = undefined;
110     }
111     return jQuery(selector, context);
112   };

114   /**
115     Gets the buffered command, adding it to a buffer if needed.
116   */
117   Buffer.prototype.bufferedCommand = function(command) {
118     // creates the buffered command if needed
119     if (!this._bufferedCommands[command]) {
120       // sets up the hash for the command
121       this._bufferedCommands[command] = {};
122       this._bufferedCommandList.push(command);
123     }
124 
125     // return the buffered commands
126     return this._bufferedCommands[command];
127   };
128 
129   Buffer.prototype.hasBufferedCommand = function(command) {
130     return !!this._bufferedCommands[command];
131   };

136   Buffer.prototype.html = function(value) {
137     var context = this.bufferedCommand("flushContent");
138     if (value === undefined) return context.text || context.html || this.$().html();
139 
140     context.text = undefined;
141     context.html = value;
142   };

144   Buffer.prototype.text = function(value) {
145     var context = this.bufferedCommand("flushContent");
146     if (value === undefined) return context.text || context.html || this.$().text();
147 
148     context.text = value;
149     context.html = undefined;
150   };

152   Buffer.prototype.flushContent = function(context) {
153     if (context.text !== undefined) this.$().text(context.text);
154     else if (context.html !== undefined) this.$().html(context.html);
155   };

159   /**
160     Handles attribute setting.
161   */
162   Buffer.prototype.attr = function(key, value) {
163     // first handle the key-as-object scenario
164     if (typeof key === "object") {
165       for (var k in key) this.attr(k, key[k]);
166       return;
167     }
168 
169     // now, if it is a special key, handle it specially.
170     if (key === "class") {
171       // note: setClass will return the value if "value" is undefined.
172       if (value === undefined) return this.setClass().join(' ');
173       else {
174         // if we have a string or array, we need to reset class names altogether
175         if (typeof value === "string" || jQuery.isArray(value)) {
176           var classNamesContext = this.bufferedCommand("flushClassNames");
177           classNamesContext.classNames = this._hashFromClassNames(value);
178           return;
179         } else {
180           // otherwise, hand it to setClass
181           return this.setClass(value);
182         }
183       }
184     } else if (key === "html") {
185       return this.html(value);
186 187     } else if (key === "text") {
188       return this.text(value);
189     } else if (key === 'style') {
190       return this.resetStyles(value);
191     }
192 
193     var context = this.bufferedCommand("flushAttributes");
194     if (!context.attr) context.attr = {};
195     context.attr[key] = value;
196   };
197 
198   Buffer.prototype.flushAttributes = function(context) {
199     var attr = context.attr, cq = this.$(), v;
200     for (var key in attr) {
201       if (!attr.hasOwnProperty(key)) continue;
202       v = attr[key];
203       if (v !== null) cq.attr(key, v);
204       else cq.removeAttr(key);
205     }
206   };

212   Buffer.prototype._STYLE_REGEX = /-?\s*([^:\s]+)\s*:\s*([^;]+)\s*;?/g;
213 
214   Buffer.prototype._camelizeStyleName = function(name) {
215     // IE wants the first letter lowercase so we can allow normal behavior
216     var needsCap = name.match(/^-(webkit|moz|o)-/),
217         camelized = name.camelize();
218 
219     if (needsCap) {
220       return camelized.substr(0,1).toUpperCase() + camelized.substr(1);
221     } else {
222       return camelized;
223     }
224   };
225 
226   Buffer.prototype._dasherizeStyleName = function(name) {
227     /*
228       TODO This is in jQuery framework but its relying on dasherize,
229             which is defined in Core Foundation
230     */
231     var dasherized = SC.String.dasherize(name);
232     if (dasherized.match(/^(webkit|moz|ms|o)-/)) dasherized = '-'+dasherized;
233     return dasherized;
234   };

236   Buffer.prototype._loadStyles = function(attr) {
237     // parse style...
238     if (!attr) attr = this.$().attr('style');
239 
240     if (attr && (attr = attr.toString()).length>0) {
241       if(SC.browser.isIE){
242         attr = attr.toLowerCase();
243       }
244       var styles = {};
245 
246       var regex = this._STYLE_REGEX, match;
247       regex.lastIndex = 0;
248 
249       while(match = regex.exec(attr)) styles[this._camelizeStyleName(match[1])] = match[2];
250 
251       return styles;
252     } else {
253       return { };
254     }
255   };
256 
257   Buffer.prototype.resetStyles = function(styles) {
258     var context = this.bufferedCommand("flushStyles");
259     context._styles = this._loadStyles(styles || " ");
260   };
261 
262   Buffer.prototype.styles = function() {
263     var context = this.bufferedCommand("flushStyles");
264     if (!context._styles) context._styles = this._loadStyles();
265     return context._styles;
266   };
267 
268   Buffer.prototype.css = function(key, value) {
269     if (typeof key === "object") {
270       for (var k in key) this.css(k, key[k]);
271       return;
272     }
273 
274     var context = this.bufferedCommand("flushStyles");
275     if (!context._styles) context._styles = this._loadStyles();
276 
277     context._styles[key] = value;
278   };
279 
280   Buffer.prototype.flushStyles = function(context) {
281     var styles = context._styles;
282     var str = "";
283 
284     var key, value, props = [], idx = 0;
285     for (key in styles) {
286       if (!styles.hasOwnProperty(key)) continue;
287       value = styles[key];
288       if (value === null) continue;
289       if (typeof value === "number" && key !== 'zIndex' && key !== "fontWeight" && key !== "opacity") value += "px";
290 
291       props[idx++] = this._dasherizeStyleName(key) + ": " + value;
292     }
293 
294     this.$().attr("style", props.join("; "));
295   };

297   //
298   // SUPPORT FOR CLASS NAMES
299   //
300   Buffer.prototype._hashFromClassNames = function(classNames) {
301     // split if needed
302     if (typeof classNames === "string") classNames = classNames.split(" ");
303 
304     // and continue
305     var idx, len = classNames.length, ret = {};
306     for (idx = 0; idx < len; idx++) {
307       ret[classNames[idx]] = true;
308     }
309     return ret;
310   };

312   Buffer.prototype.setClass = function(value, on) {
313     var context = this.bufferedCommand("flushClassNames"), key;
314 
315     // if there is no value, that means we are trying to actually _get_ the class names.
316     if (value === undefined) {
317       if (!context.classNames) context.classNames = this._hashFromClassNames(this._el.className);
318 
319       var classNames = context.classNames, v = [];
320       for (key in classNames) if (key && classNames[key]) v.push(key);
321       return v;
322     }
323 
324     // Hashes of key/values
325     if (typeof value === "object") {
326       if (!context.classNames) context.classNames = this._hashFromClassNames(this._el.className);
327 
328       // loop over class names and set it properly.
329       for (key in value) {
330         context.classNames[key] = value[key];
331       }
332 
333       return;
334     }
335 
336     // value is not a hash of class names and values, so "on" must either be
337 338     // a boolean, or undefined meaning NO.
339     if (!context.classNames) {
340       context.classNames = this._hashFromClassNames(this._el.className);
341     }
342 
343     context.classNames[value] = on || false;
344   };
345 
346   Buffer.prototype.hasClass = function(className) {
347     var context = this.bufferedCommand("flushClassNames");
348     if (!context.classNames) context.classNames = this._hashFromClassNames(this._el.className);
349     return !!context.classNames[className];
350   };
351 
352   Buffer.prototype.addClass = function(value) {
353     if (!value) return;
354 
355     var context = this.bufferedCommand("flushClassNames");
356     if (!context.classNames) context.classNames = this._hashFromClassNames(this._el.className);
357 
358     if (typeof value === "string") value = value.split(' ');
359 
360     var idx, len = value.length;
361     for (idx = 0; idx < len; idx++) context.classNames[jQuery.trim(value[idx])] = true;
362   };
363 
364   Buffer.prototype.removeClass = function(value) {
365     var context = this.bufferedCommand("flushClassNames");
366     if (!context.classNames) context.classNames = this._hashFromClassNames(this._el.className);
367     context.classNames[value] = false;
368   };
369 
370   Buffer.prototype.resetClassNames = function(value) {
371     var context = this.bufferedCommand("flushClassNames");
372     context.classNames = {};
373   };
374 
375   Buffer.prototype.flushClassNames = function(context) {
376     var classNames = [];
377     var c = context.classNames, k;
378     for (k in c) if (c[k]) classNames.push(k)
379 
380     this.$().attr("class", classNames.join(" "));
381   };
