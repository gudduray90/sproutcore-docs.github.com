  8 // NOTE: The test below are based on the Data Hashes state chart.  This models
  9 // the "write" event in the NestedStore portion of the diagram.
 10 
 11 var store, child, storeKey, json;
 12 module("SC.Store#writeDataHash", {
 13   setup: function() {
 14     store = SC.Store.create();
 15     
 16     json = {
 17       string: "string",
 18       number: 23,
 19       bool:   YES
 20     };
 21     
 22     storeKey = SC.Store.generateStoreKey();
 23     child = store.chain();  // test multiple levels deep
 24   }
 25 });
 26 
 27 // ..........................................................
 28 // BASIC STATE TRANSITIONS
 29 // 
 30 
 31 // The transition from each base state performs the same operation, so just
 32 // run the same test on each state.
 33 function testWriteDataHash() {
 34   var oldrev = store.revisions[storeKey];
 35   
 36   // perform test
 37   var json2 = { foo: "bar" };
 38   equals(store.writeDataHash(storeKey, json2, SC.Record.READY_NEW), store, 'should return receiver');
 39   
 40   // verify
 41   equals(store.storeKeyEditState(storeKey), SC.Store.EDITABLE, 'new edit state should be editable');
 42   
 43   equals(store.readDataHash(storeKey), json2, 'should have new json data hash');
 44   equals(store.readStatus(storeKey), SC.Record.READY_NEW, 'should have new status');
 45 
 46   equals(store.revisions[storeKey], oldrev, 'should not change revision');
 47   if (!SC.none(oldrev)) {
 48     ok(store.revisions.hasOwnProperty(storeKey), 'should clone reference to revision');
 49   }
 50 }
 51 
 52 
 53 test("edit state=LOCKED - also writes a NEW hash", function() {
 54   
 55   // test preconditions
 56   equals(store.storeKeyEditState(storeKey), SC.Store.LOCKED, 'precond - edit state should be locked');
 57   
 58   testWriteDataHash();
 59 });
 60 
 61 test("edit state=EDITABLE - also overwrites an EXISTING hash", function() {
 62   
 63   // test preconditions
 64   store.writeDataHash(storeKey, { foo: "bar" });
 65   equals(store.storeKeyEditState(storeKey), SC.Store.EDITABLE, 'precond - edit state should be editable');
 66   
 67   testWriteDataHash();
 68 
 69 });
 70 
 71 // ..........................................................
 72 // PROPOGATING TO NESTED STORES
 73 // 
 74 
 75 test("change should propogate to child if child edit state = INHERITED", function() {
 76 
 77   // verify preconditions
 78   equals(child.storeKeyEditState(storeKey), SC.Store.INHERITED, 'precond - child edit state should be INHERITED');
 79 
 80   // perform change
 81   var json2 = { version: 2 };
 82   store.writeDataHash(storeKey, json2, SC.Record.READY_NEW);
 83   
 84   // verify
 85   same(child.readDataHash(storeKey), json2, 'child should pick up change');
 86   equals(child.readStatus(storeKey), SC.Record.READY_NEW, 'child should pick up new status');
 87 });
 88 
 89 
 90 function testLockedOrEditableChild() {
 91   // perform change
 92   var json2 = { version: 2 };
 93   store.writeDataHash(storeKey, json2, SC.Record.READY_NEW);
 94   
 95   // verify
 96   same(child.readDataHash(storeKey), json, 'child should NOT pick up change');
 97   equals(child.readStatus(storeKey), SC.Record.READY_CLEAN, 'child should pick up new status');
 98 }
 99 
100 
101 test("change should not propogate to child if child edit state = LOCKED", function() {
102   store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
103   store.editables = null ; // clear to simulate locked mode.
104   
105   // verify preconditions
106   child.readDataHash(storeKey);
107   equals(child.storeKeyEditState(storeKey), SC.Store.LOCKED, 'precond - child edit state should be LOCKED');
108 
109   testLockedOrEditableChild();
110 });
111 
112 test("change should not propogate to child if child edit state = EDITABLE", function() {
113   store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
114   store.editables = null ; // clear to simulate locked mode.
115 
116   // verify preconditions
117   child.readEditableDataHash(storeKey);
118   equals(child.storeKeyEditState(storeKey), SC.Store.EDITABLE, 'precond - child edit state should be EDITABLE');
119 
120   testLockedOrEditableChild();
121 });
