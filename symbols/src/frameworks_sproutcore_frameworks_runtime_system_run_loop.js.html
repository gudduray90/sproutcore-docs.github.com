 20 /**
 21   @class
 22 
 23   The run loop provides a universal system for coordinating events within
 24   your application.  The run loop processes timers as well as pending
 25   observer notifications within your application.
 26 
 27   To use a RunLoop within your application, you should make sure your event
 28   handlers always begin and end with SC.RunLoop.begin() and SC.RunLoop.end()
 29 
 30   The RunLoop is important because bindings do not fire until the end of
 31   your run loop is reached.  This improves the performance of your
 32   application.
 33 
 34   Example:
 35 
 36   This is how you could write your mouseup handler in jQuery:
 37 
 38         $('#okButton').on('click', function () {
 39           SC.run(function () {
 40 
 41             // handle click event...
 42 
 43           }); // allows bindings to trigger...
 44         });
 45 
 46   @extends SC.Object
 47   @since SproutCore 1.0
 48 */
 49 SC.RunLoop = SC.Object.extend(
 50   /** @scope SC.RunLoop.prototype */ {
 51 
 52   /**
 53     Call this method whenver you begin executing code.
 54 
 55     This is typically invoked automatically for you from event handlers and
 56     the timeout handler.  If you call setTimeout() or setInterval() yourself,
 57     you may need to invoke this yourself.
 58 
 59     @returns {SC.RunLoop} receiver
 60   */
 61   beginRunLoop: function () {
 62     this._start = new Date().getTime(); // can't use Date.now() in runtime
 63     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 64       SC.Logger.log("-- SC.RunLoop.beginRunLoop at %@".fmt(this._start));
 65     }
 66     this._runLoopInProgress = YES;
 67     this._flushinvokeNextQueue();
 68     return this;
 69   },
 70 
 71   /**
 72     YES when a run loop is in progress
 73 
 74     @property
 75     @type Boolean
 76   */
 77   isRunLoopInProgress: function () {
 78     return this._runLoopInProgress;
 79   }.property(),
 80 
 81   /**
 82     Call this method whenever you are done executing code.
 83 
 84     This is typically invoked automatically for you from event handlers and
 85     the timeout handler.  If you call setTimeout() or setInterval() yourself
 86     you may need to invoke this yourself.
 87 
 88     @returns {SC.RunLoop} receiver
 89   */
 90   endRunLoop: function () {
 91     // at the end of a runloop, flush all the delayed actions we may have
 92     // stored up.  Note that if any of these queues actually run, we will
 93     // step through all of them again.  This way any changes get flushed
 94     // out completely.
 95 
 96     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 97       SC.Logger.log("-- SC.RunLoop.endRunLoop ~ flushing application queues");
 98     }
 99 
100     this.flushAllPending();
101 
102     this._start = null;
103 
104     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
105       SC.Logger.log("-- SC.RunLoop.endRunLoop ~ End");
106     }
107 
108     SC.RunLoop.lastRunLoopEnd = Date.now();
109     this._runLoopInProgress = NO;
110 
111     // If there are members in the invokeNextQueue, be sure to schedule another
112     // run of the run loop.
113     var queue = this._invokeNextQueue;
114     if (queue && queue.getMembers().length) {
115       this._invokeNextTimeout = this.scheduleRunLoop(SC.RunLoop.lastRunLoopEnd);
116     }
117 
118     return this;
119   },
120 
121   /**
122     Repeatedly flushes all bindings, observers, and other queued functions until all queues are empty.
123   */
124   flushAllPending: function () {
125     var didChange;
126 
127     do {
128       didChange = this.flushApplicationQueues();
129       if (!didChange) didChange = this._flushinvokeLastQueue();
130     } while (didChange);
131   },

134   /**
135     Invokes the passed target/method pair once at the end of the runloop.
136     You can call this method as many times as you like and the method will
137     only be invoked once.
138 
139     Usually you will not call this method directly but use invokeOnce()
140     defined on SC.Object.
141 
142     Note that in development mode only, the object and method that call this
143     method will be recorded, for help in debugging scheduled code.
144 
145     @param {Object} target
146     @param {Function} method
147     @returns {SC.RunLoop} receiver
148   */
149   invokeOnce: function (target, method) {
150     //@if(debug)
151     // Calling invokeOnce outside of a run loop causes problems when coupled
152     // with time dependent methods invokeNext and invokeLater, because the
153     // time dependent methods execute at the start of the next run of the run
154     // loop and may fire before the invokeOnce code in this case.
155     var isRunLoopInProgress = this.get('isRunLoopInProgress');
156     if (!isRunLoopInProgress) {
157       SC.warn("Developer Warning: invokeOnce called outside of the run loop, which may cause unexpected problems. Check the stack trace below for what triggered this, and see http://blog.sproutcore.com/1-10-upgrade-invokefoo/ for more.");
158       console.trace();
159     }
160     //@endif
161     // normalize
162     if (method === undefined) {
163       method = target;
164       target = this;
165     }
166 
167     var deferredCallLoggingInfo;      // Used only in debug mode
168 
169     //@if (debug)
170     // When in debug mode, SC.Object#invokeOnce() will pass in the originating
171     // method, target, and stack.  That way, we'll record the interesting parts
172     // rather than having most of these calls seemingly coming from
173     // SC.Object#invokeOnce().
174     //
175     // If it was not specified, we'll record the originating function ourselves.
176     var shouldLog = SC.LOG_DEFERRED_CALLS;
177     if (shouldLog) {
178       var originatingTarget = arguments[2],
179           originatingMethod = arguments[3],
180           originatingStack  = arguments[4];
181 
182       if (!originatingTarget) originatingTarget = null;   // More obvious when debugging
183       if (!originatingMethod) {
184         originatingStack  = SC._getRecentStack();
185         originatingMethod = originatingStack[0];
186       }
187 
188       deferredCallLoggingInfo = {
189         originatingTarget: originatingTarget,
190         originatingMethod: originatingMethod,
191         originatingStack:  originatingStack
192       };
193     }
194     //@endif
195 
196     if (typeof method === "string") method = target[method];
197     if (!this._invokeQueue) this._invokeQueue = SC.ObserverSet.create();
198     if (method) this._invokeQueue.add(target, method, undefined, deferredCallLoggingInfo);
199     return this;
200   },
201 
202   /**
203     Invokes the passed target/method pair at the very end of the run loop,
204     once all other delayed invoke queues have been flushed.  Use this to
205     schedule cleanup methods at the end of the run loop once all other work
206     (including rendering) has finished.
207 
208     If you call this with the same target/method pair multiple times it will
209     only invoke the pair only once at the end of the runloop.
210 
211     Usually you will not call this method directly but use invokeLast()
212     defined on SC.Object.
213 
214     Note that in development mode only, the object and method that call this
215     method will be recorded, for help in debugging scheduled code.
216 
217     @param {Object} target
218     @param {Function} method
219     @returns {SC.RunLoop} receiver
220   */
221   invokeLast: function (target, method) {
222     //@if(debug)
223     // Calling invokeLast outside of a run loop causes problems when coupled
224     // with time dependent methods invokeNext and invokeLater, because the
225     // time dependent methods execute at the start of the next run of the run
226     // loop and may fire before the invokeLast code in this case.
227     var isRunLoopInProgress = this.get('isRunLoopInProgress');
228     if (!isRunLoopInProgress) {
229       SC.warn("Developer Warning: invokeLast called outside of the run loop, which may cause unexpected problems. Check the stack trace below for what triggered this, and see http://blog.sproutcore.com/1-10-upgrade-invokefoo/ for more.");
230       console.trace();
231     }
232     //@endif
233 
234     // normalize
235     if (method === undefined) {
236       method = target;
237       target = this;
238     }
239 
240     var deferredCallLoggingInfo;      // Used only in debug mode
241 
242     //@if (debug)
243     // When in debug mode, SC.Object#invokeOnce() will pass in the originating
244     // method, target, and stack.  That way, we'll record the interesting parts
245     // rather than having most of these calls seemingly coming from
246     // SC.Object#invokeOnce().
247     //
248     // If it was not specified, we'll record the originating function ourselves.
249     var shouldLog = SC.LOG_DEFERRED_CALLS;
250     if (shouldLog) {
251       var originatingTarget = arguments[2],
252           originatingMethod = arguments[3],
253           originatingStack  = arguments[4];
254 
255       if (!originatingTarget) originatingTarget = null;   // More obvious when debugging
256       if (!originatingMethod) {
257         originatingStack  = SC._getRecentStack();
258         originatingMethod = originatingStack[0];
259       }
260 
261       deferredCallLoggingInfo = {
262         originatingTarget: originatingTarget,
263         originatingMethod: originatingMethod,
264         originatingStack:  originatingStack
265       };
266     }
267     //@endif
268 
269 
270     if (typeof method === "string") method = target[method];
271     if (!this._invokeLastQueue) this._invokeLastQueue = SC.ObserverSet.create();
272     this._invokeLastQueue.add(target, method, undefined, deferredCallLoggingInfo);
273     return this;
274   },

276   /**
277     Invokes the passed target/method pair once at the beginning of the next
278     run of the run loop, before any other methods (including events) are
279     processed.  Use this to defer painting to make views more responsive.
280 
281     If you call this with the same target/method pair multiple times it will
282     only invoke the pair only once at the beginning of the next runloop.
283 
284     Usually you will not call this method directly but use invokeNext()
285     defined on SC.Object.
286 
287     @param {Object} target
288     @param {Function} method
289     @returns {SC.RunLoop} receiver
290   */
291   invokeNext: function (target, method) {
292     // normalize
293     if (method === undefined) {
294       method = target;
295       target = this;
296     }
297 
298     if (typeof method === "string") method = target[method];
299     if (!this._invokeNextQueue) this._invokeNextQueue = SC.ObserverSet.create();
300     this._invokeNextQueue.add(target, method);
301     return this;
302   },
303 
304   /**
305     Executes any pending events at the end of the run loop.  This method is
306     called automatically at the end of a run loop to flush any pending
307     queue changes.
308 
309     The default method will invoke any one time methods and then sync any
310     bindings that might have changed.  You can override this method in a
311     subclass if you like to handle additional cleanup.
312 
313     This method must return YES if it found any items pending in its queues
314     to take action on.  endRunLoop will invoke this method repeatedly until
315     the method returns NO.  This way if any if your final executing code
316     causes additional queues to trigger, then can be flushed again.
317 
318     @returns {Boolean} YES if items were found in any queue, NO otherwise
319   */
320   flushApplicationQueues: function () {
321     var hadContent = NO,
322         // execute any methods in the invokeQueue.
323         queue = this._invokeQueue;
324     if (queue && queue.getMembers().length) {
325       this._invokeQueue = null; // reset so that a new queue will be created
326       hadContent = YES; // needs to execute again
327       queue.invokeMethods();
328     }
329 
330     // flush any pending changed bindings.  This could actually trigger a
331     // lot of code to execute.
332     return SC.Binding.flushPendingChanges() || hadContent;
333   },
334 
335   _flushinvokeLastQueue: function () {
336     var queue = this._invokeLastQueue, hadContent = NO;
337     if (queue && queue.getMembers().length) {
338       this._invokeLastQueue = null; // reset queue.
339       hadContent = YES; // has targets!
340       if (hadContent) queue.invokeMethods();
341     }
342     return hadContent;
343   },
344 
345   _flushinvokeNextQueue: function () {
346     var queue = this._invokeNextQueue, hadContent = NO;
347     if (queue && queue.getMembers().length) {
348       this._invokeNextQueue = null; // reset queue.
349       hadContent = YES; // has targets!
350       if (hadContent) queue.invokeMethods();
351     }
352     return hadContent;
353   },
354 
355   /** @private
356     Schedules the run loop to run at the given time.  If the run loop is
357     already scheduled to run earlier nothing will change, but if the run loop
358     is not scheduled or it is scheduled later, then it will be rescheduled
359     to the value of nextTimeoutAt.
360 
361     @returns timeoutID {Number} The ID of the timeout to start the next run of the run loop
362   */
363   scheduleRunLoop: function (nextTimeoutAt) {
364     /*jshint eqnull:true*/
365     // If there is no run loop scheduled or if the scheduled run loop is later, reschedule.
366     if (this._timeoutAt == null || this._timeoutAt > nextTimeoutAt) {
367       // clear existing...
368       if (this._timeout) { clearTimeout(this._timeout); }
369 
370       // reschedule
371       var delay = Math.max(0, nextTimeoutAt - Date.now());
372       this._timeout = setTimeout(this._timeoutDidFire, delay);
373       this._timeoutAt = nextTimeoutAt;
374     }
375 
376     return this._timeout;
377   },
378 
379   /** @private
380     Invoked when a timeout actually fires.  Simply cleanup, then begin and end
381     a runloop. Note that this function will be called with 'this' set to the
382     global context, hence the need to lookup the current run loop.
383   */
384   _timeoutDidFire: function () {
385     var rl = SC.RunLoop.currentRunLoop;
386     rl._timeout = rl._timeoutAt = rl._invokeNextTimeout = null; // cleanup
387     SC.run();  // begin/end runloop to trigger timers.
388   },
389 
390   /** @private Unschedule the run loop that is scheduled for the given timeoutID */
391   unscheduleRunLoop: function () {
392     // Don't unschedule if the timeout is shared with an invokeNext timeout.
393     if (!this._invokeNextTimeout) {
394       clearTimeout(this._timeout);
395       this._timeout = this._timeoutAt = null; // cleanup
396     }
397   }
398 
399 });

402 //@if(debug)
403  /**
404   Will return the recent stack as a hash with numerical keys, for nice output
405   in some browsers’ debuggers.  The “recent” stack is capped at 6 entries.
406 
407   This is used by, amongst other places, SC.LOG_DEFERRED_CALLS.
408 
409   @returns {Hash}
410 */
411 SC._getRecentStack = function () {
412   var currentFunction = arguments.callee.caller,
413       i               = 0,
414       stack           = {},
415       first           = YES,
416       functionName;
417 
418   while (currentFunction  &&  i < 10) {
419     // Skip ourselves!
420     if (first) {
421       first = NO;
422     } else {
423       functionName = currentFunction.displayName || currentFunction.toString().substring(0, 40);
424       stack[i++]   = functionName;
425     }
426     currentFunction = currentFunction.caller;
427   }
428 
429   return stack;
430 };

435 /**
436   The current run loop.  This is created automatically the first time you
437   call begin().
438 
439   @type SC.RunLoop
440 */
441 SC.RunLoop.currentRunLoop = null;
442 
443 /**
444   The default RunLoop class.  If you choose to extend the RunLoop, you can
445   set this property to make sure your class is used instead.
446 
447   @type Class
448 */
449 SC.RunLoop.runLoopClass = SC.RunLoop;
450 
451 /**
452   Begins a new run loop on the currentRunLoop.  If you are already in a
453   runloop, this method has no effect.
454 
455   @returns {SC.RunLoop} receiver
456 */
457 SC.RunLoop.begin = function () {
458   var runLoop = this.currentRunLoop;
459   if (!runLoop) runLoop = this.currentRunLoop = this.runLoopClass.create();
460   runLoop.beginRunLoop();
461   return this;
462 };
463 
464 /**
465   Ends the run loop on the currentRunLoop.  This will deliver any final
466   pending notifications and schedule any additional necessary cleanup.
467 
468   @returns {SC.RunLoop} receiver
469 */
470 SC.RunLoop.end = function () {
471   var runLoop = this.currentRunLoop;
472   if (!runLoop) {
473     throw new Error("SC.RunLoop.end() called outside of a runloop!");
474   }
475   runLoop.endRunLoop();
476   return this;
477 };

480 /**
481   Call this to kill the current run loop--stopping all propagation of bindings
482   and observers and clearing all timers.
483 
484   This is useful if you are popping up an error catcher: you need a run loop
485   for the error catcher, but you don't want the app itself to continue
486   running.
487 */
488 SC.RunLoop.kill = function () {
489   this.currentRunLoop = this.runLoopClass.create();
490   return this;
491 };

503 /**
504   Executes a passed function in the context of a run loop. If called outside a
505   runloop, starts and ends one. If called inside an existing runloop, is
506   simply executes the function unless you force it to create a nested runloop.
507 
508   If an exception is thrown during execution, we give an error catcher the
509   opportunity to handle it before allowing the exception to bubble again.
510 
511   @param {Function} callback callback to execute
512   @param {Object} target context for callback
513   @param {Boolean} if YES, starts/ends a new runloop even if one is already running
514 */
515 SC.run = function (callback, target, forceNested) {
516   var alreadyRunning = SC.RunLoop.isRunLoopInProgress();
517 
518   // Only use a try/catch block if we have an ExceptionHandler
519   // since in some browsers try/catch causes a loss of the backtrace
520   if (SC.ExceptionHandler && SC.ExceptionHandler.enabled) {
521     try {
522       if (forceNested || !alreadyRunning) SC.RunLoop.begin();
523       if (callback) callback.call(target);
524       if (forceNested || !alreadyRunning) SC.RunLoop.end();
525     } catch (e) {
526       var handled = SC.ExceptionHandler.handleException(e);
527 
528       // If the exception was not handled, throw it again so the browser
529       // can deal with it (and potentially use it for debugging).
530       // (We don't throw it in IE because the user will see two errors)
531       if (!handled && !SC.browser.isIE) {
532         throw e;
533       }
534     }
535   } else {
536     if (forceNested || !alreadyRunning) SC.RunLoop.begin();
537     if (callback) callback.call(target);
538     if (forceNested || !alreadyRunning) SC.RunLoop.end();
539   }
540 };

542 /**
543   Wraps the passed function in code that ensures a run loop will
544   surround it when run.
545 */
546 SC.RunLoop.wrapFunction = function (func) {
547   var ret = function () {
548     var alreadyRunning = SC.RunLoop.isRunLoopInProgress();
549     if (!alreadyRunning) SC.RunLoop.begin();
550     var ret = arguments.callee.wrapped.apply(this, arguments);
551     if (!alreadyRunning) SC.RunLoop.end();
552     return ret;
553   };
554   ret.wrapped = func;
555   return ret;
556 };
