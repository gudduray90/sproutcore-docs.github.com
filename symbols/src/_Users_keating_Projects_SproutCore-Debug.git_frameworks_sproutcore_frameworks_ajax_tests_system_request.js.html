168 /**
169   There are two ways to be notified of request changes:
170 
171     - Implementing a didReceive function on the SC.Request object
172     - Registering a listener using notify()
173 
174   The following two tests test the timeout functionality for each of these.
175 */
176 test("Timeouts - SC.Request didReceive callback", function() {
177   var message;
178 
179   // Sanity check - Should throw an error if we try to set a timeout of 0s.
180   try {
181     SC.Request.getUrl(url).set('timeout', 0).send();
182   }
183   catch (e) {
184     message = e;
185   }
186   ok(message && message.indexOf("The timeout value must either not be specified or must be greater than 0") !== -1, 'An error should be thrown when the timeout value is 0 ms');
187 
188   // Sanity check 2 - Can't set timeouts on synchronous XHR requests
189   try {
190     SC.Request.getUrl(url).set('isAsynchronous', NO).set('timeout', 10).send();
191   }
192   catch (e2) {
193     message = e2;
194   }
195   ok(message && message.indexOf("Timeout values cannot be used with synchronous requests") !== -1, 'An error should be thrown when trying to use a timeout with a synchronous request');
196 
197 
198   // Make sure timeouts actually fire, and fire when expected.
199   // Point to the server itself so that the tests will work even when offline
200   var timeoutRequest = SC.Request.getUrl("/"),
201       checkstop;
202 
203   var now = Date.now();
204 
205   // Set timeout as short as possible so that it will always timeout before
206   // the request returns.
207   // This test will fail should the response time drop to
208   // below 10ms.
209   timeoutRequest.set('timeout', 10);
210 
211   timeoutRequest.set('didReceive', function(request, response) {
212     // Test runner is paused after the request is sent; resume unit testing
213     // once we receive a response.
214     start();
215     clearTimeout(checkstop);
216 
217     // If this response was caused by a timeout…
218     if (response.get('timedOut')) {
219       equals(response.get('status'), 0,
220              'Timed out responses should have status 0');
221 
222       // We should never be called before the timeout we specified
223       var elapsed = Date.now()-now;
224       ok(elapsed >= 10,
225         'timeout must not fire earlier than 10msec - actual %@'.fmt(elapsed));
226     } else {
227       // We received a response from the server, which should never happen
228       ok(false, 'timeout did not fire before response was received.  should have fired after 10msec.  response time: %@msec'.fmt(Date.now() - now));
229     }
230   });
231 
232   SC.RunLoop.begin();
233   timeoutRequest.send();
234   SC.RunLoop.end();
235 
236   // Stop the test runner and wait for a timeout or a response.
237   stop();
238 
239   // In case we never receive a timeout, just start unit testing again after
240   // 500ms.
241   checkstop = setTimeout(function() {
242     window.start();
243     ok(false, 'timeout did not fire at all');
244   }, 500);
245 });

247 test("Timeouts - Status listener callback", function() {
248   // Make sure timeouts actually fire, and fire when expected.
249   // Point to local server so test works offline
250   var timeoutRequest = SC.Request.getUrl("/"),
251       checkstop;
252 
253   // make the timeout as short as possible so that it will always happen
254   timeoutRequest.timeoutAfter(10).notify(this, function(response) {
255     start();
256     clearTimeout(checkstop);
257 
258     equals(response.get('status'), 0, "Status code should be zero");
259     equals(response.get('timedOut'), YES, "Should have timedOut property set to YES");
260     // timeout did fire...just resume...
261 
262     return YES;
263   });
264 
265   SC.RunLoop.begin();
266   timeoutRequest.send();
267   SC.RunLoop.end();
268 
269   stop() ; // stops the test runner
270 
271   // in case nothing works
272   checkstop = setTimeout(function() {
273     ok(false, 'timeout did not fire at all');
274     window.start();
275   }, 500);
276 });
