 10 var context = null, elem = null;
 11 
 12 module("SC.RenderContext#update", {
 13   setup: function() {
 14     elem = document.createElement('div');
 15     context = SC.RenderContext(elem) ;
 16   },
 17 
 18   teardown: function() {
 19     elem = context = null; // avoid memory leaks
 20   }
 21 });
 22 
 23 test("should replace innerHTML of DIV if strings were pushed", function() {
 24   elem.innerHTML = "initial";
 25   context.push("changed").update();
 26   equals(elem.innerHTML, "changed", "innerHTML did change");
 27 });
 28 
 29 test("should NOT replace innerHTML of DIV if no strings were pushed", function() {
 30   elem.innerHTML = "initial";
 31   context.update();
 32   equals(elem.innerHTML, "initial", "innerHTML did NOT change");
 33 });
 34 
 35 test("returns receiver if no prevObject", function() {
 36   equals(context.update(), context, "return value");
 37 });
 38 
 39 test("returns previous context if there is one", function() {
 40   var c2 = context.begin(elem);
 41   equals(c2.update(), context, "returns prev context");
 42 });
 43 
 44 test("clears internal _elem to avoid memory leaks on update", function() {
 45   ok(!!context._elem, 'precondition - has element')  ;
 46   context.update();
 47   ok(!context._elem, "no longer an element");
 48 });

157 // Style Editing

159 module("SC.RenderContext#update - style", {
160   setup: function() {
161     elem = document.createElement('div');
162     SC.$(elem).attr("style", "color: red;");
163     context = SC.RenderContext(elem);
164   },
165 
166   teardown: function() {
167     elem = context = null ;
168   }
169 });
170 
171 test("does not change styles if retrieved but not edited", function() {
172   context.styles();
173   context.update();
174   var style = SC.$(elem).attr("style");
175   if (!style.match(/;$/)) style += ';' ;
176 
177   equals(style.toLowerCase(), "color: red;", "style");
178 });
179 
180 test("replaces style name if styles edited", function() {
181   context.styles({ color: "black" });
182   context.update();
183 
184   // Browsers return single attribute styles differently, sometimes with a trailing ';'
185   // sometimes, without one. Normalize it here.
186   var style = SC.$(elem).attr("style");
187   if (!style.match(/;$/)) style += ';' ;
188 
189   equals(style.toLowerCase(), "color: black;", "attribute");
190 });
191 
192 
193 test("set styles override style attr", function() {
194   context.attr("style", "color: green");
195   context.styles({ color: "black" });
196   context.update();
197 
198   // Browsers return single attribute styles differently, sometimes with a trailing ';'
199   // sometimes, without one. Normalize it here.
200   var style = SC.$(elem).attr("style");
201   if (!style.match(/;$/)) style += ';' ;
202 
203   equals(style.toLowerCase(), "color: black;", "attribute");
204 });
205 
206 test("set styles handle custom browser attributes", function() {
207   context.styles({ mozColumnCount: '3', webkitColumnCount: '3', oColumnCount: '3', msColumnCount: '3' });
208   context.update();
209 
210   // Browsers return single attribute styles differently, sometimes with a trailing ';'
211   // sometimes, without one. Normalize it here.
212   var style = SC.$(elem).attr("style");
213   if (!style.match(/;$/)) style += ';' ;
214 
215   // Older Gecko completely ignores css attributes that it doesn't understand.
216   if(SC.browser.isMozilla &&
217       SC.browser.compare(SC.browser.engineVersion, '2.0') < 0) equals(style, "-moz-column-count: 3;");
218   else if (SC.browser.isIE) equals(style, "-webkit-column-count: 3; -moz-column-count: 3; -o-column-count: 3; -ms-column-count: 3;");
219   else equals(style, "-moz-column-count: 3; -webkit-column-count: 3; -o-column-count: 3; -ms-column-count: 3;");
220 });
