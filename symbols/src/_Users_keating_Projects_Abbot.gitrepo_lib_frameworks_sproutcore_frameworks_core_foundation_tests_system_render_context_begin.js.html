  8 /*global module test equals context ok */
  9 
 10 var context = null;
 11 
 12 module("SC.RenderContext#begin", {
 13   setup: function() {
 14     context = SC.RenderContext();
 15   }
 16 });
 17 
 18 test("should return a new context with parent context as prevObject", function() {
 19   var c2 = context.begin();
 20   ok(c2 !== context, "new context");
 21   equals(c2.prevObject, context, 'previous context');
 22 });
 23 
 24 test("should set offset for new context equal to length of previous context", function() {
 25   context.push("line1");
 26   var expected = context.length ;
 27   var c2 = context.begin();
 28   equals(c2.offset, expected, "offset");
 29 });
 30 
 31 test("should copy same strings array to new child context", function() {
 32   context.push("line1");
 33   var c2 =context.begin();
 34   equals(c2.strings, context.strings);
 35 });
 36 
 37 test("should start new context with length of 1 (reserving a space for opening tag)", function() {
 38   context.push("line1");
 39   var c2 = context.begin() ;
 40   equals(c2.length, 1, 'has empty line');
 41   equals(c2.strings.length, 3, "parent empty line + parent line + empty line");
 42 });
 43 
 44 test("should assign passed tag name to new context", function() {
 45   var c2 = context.begin('foo');
 46   equals(c2.tagName(), 'foo', 'tag name');
 47 });
