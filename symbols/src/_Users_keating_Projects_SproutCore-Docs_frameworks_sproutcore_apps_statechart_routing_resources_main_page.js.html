  5 /*globals Test */

  7 Test.mainPage = SC.Page.create({
  8 
  9   fooView: SC.outlet('Test.fooPage.mainView'),
 10   
 11   barView: SC.outlet('Test.barPage.mainView'),
 12   
 13   emptyView: SC.View.design(),
 14 
 15   mainPane: SC.MainPane.design({
 16     
 17     childViews: 'headerView containerView'.w(),
 18     
 19     headerView: SC.View.design({
 20       layout: { top: 0, left: 0, right: 0, height: 50 },
 21       childViews: 'switchModeView'.w(),
 22       switchModeView: SC.SegmentedView.design({
 23         layout: { centerX: 0, centerY: 0, height: 24, width: 100 },
 24         items: [
 25           { title: 'Foo', value: Test.MODE_FOO, action: 'switchToFooMode' },
 26           { title: 'Bar', value: Test.MODE_BAR, action: 'switchToBarMode' }
 27         ],
 28         itemTitleKey: 'title',
 29         itemValueKey: 'value',
 30         itemActionKey: 'action',
 31         valueBinding: SC.Binding.oneWay('Test.mainController.mode')
 32       })
 33     }),
 34     
 35     containerView: SC.ContainerView.design({
 36       layout: { top: 50, left: 0, right: 0, bottom: 0 },
 37       contentViewBinding: SC.Binding.transform(function(mode) {
 38         if (mode === Test.MODE_FOO) return Test.fooPage.get('mainView');
 39         if (mode === Test.MODE_BAR) return Test.barPage.get('mainView');
 40         return Test.mainPage.get('emptyView');
 41       }).oneWay('Test.mainController.mode')
 42     })
 43     
 44   })
 45 
 46 });
