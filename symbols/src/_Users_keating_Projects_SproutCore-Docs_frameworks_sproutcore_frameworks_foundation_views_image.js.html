 29 /**
 30   @class
 31 
 32   Displays an image in the browser.
 33 
 34   The ImageView can be used to efficiently display images in the browser.
 35   It includes a built in support for a number of features that can improve
 36   your page load time if you use a lot of images including a image loading
 37   cache and automatic support for CSS spriting.
 38 
 39   Note that there are actually many controls that will natively include
 40   images using an icon property name.
 41 
 42   @extends SC.View
 43   @extends SC.Control
 44   @extends SC.InnerFrame
 45   @since SproutCore 1.0
 46 */
 47 SC.ImageView = SC.View.extend(SC.Control, SC.InnerFrame,
 48 /** @scope SC.ImageView.prototype */ {
 49 
 50   classNames: 'sc-image-view',
 51 
 52   // Don't apply this role until each image view can assign a non-empty string value for @aria-label <rdar://problem/9941887>
 53   // ariaRole: 'img',
 54 
 55   displayProperties: ['frame', 'image', 'innerFrame', 'toolTip', 'imageValue', 'type'],
 56 
 57   renderDelegateName: function() {
 58     return (this.get('useCanvas') ? 'canvasImage' : 'image') + "RenderDelegate";
 59   }.property('useCanvas').cacheable(),
 60 
 61   tagName: function() {
 62     return this.get('useCanvas') ? 'canvas' : 'div';
 63   }.property('useCanvas').cacheable(),
 64 
 65 
 66   // ..........................................................
 67   // Properties
 68   //
 69 
 70   /**
 71     If YES, this image can load in the background.  Otherwise, it is treated
 72     as a foreground image.  If the image is not visible on screen, it will
 73     always be treated as a background image.
 74   */
 75   canLoadInBackground: NO,
 76 
 77   /**
 78     @property {Image}
 79     @default SC.BLANK_IMAGE
 80   */
 81   image: SC.BLANK_IMAGE,
 82 
 83   /**
 84     @property {String}
 85     @default null
 86   */
 87   imageValue: function() {
 88     var value = this.get('value');
 89     return value && value.isEnumerable ? value.firstObject() : value;
 90   }.property('value').cacheable(),
 91 
 92   /**
 93     The frame for the inner img element or for the canvas to draw within, altered according to the scale
 94     and align properties provided by SC.InnerFrame.
 95 
 96     @property {Object}
 97   */
 98   innerFrame: function() {
 99     var image = this.get('image'),
100         imageWidth = image.width,
101         imageHeight = image.height,
102         frame = this.get('frame');
103 
104     if (SC.none(frame)) return {x: 0, y: 0, width: 0, height: 0};  // frame is 'null' until rendered when useStaticLayout === YES
105 
106     return this.innerFrameForSize(imageWidth, imageHeight, frame.width, frame.height);
107   }.property('align', 'image', 'scale', 'frame').cacheable(),
108 
109   /**
110     If YES, any specified toolTip will be localized before display.
111 
112     @property {Boolean}
113     @default YES
114   */
115   localize: YES,
116 
117   /**
118     Current load status of the image.
119 
120     This status changes as an image is loaded from the server.  If spriting
121     is used, this will always be loaded.  Must be one of the following
122     constants: SC.IMAGE_STATE_NONE, SC.IMAGE_STATE_LOADING,
123     SC.IMAGE_STATE_LOADED, SC.IMAGE_STATE_FAILED
124 
125     @property {String}
126   */
127   status: SC.IMAGE_STATE_NONE,
128 
129   /**
130     Will be one of the following constants: SC.IMAGE_TYPE_URL or
131     SC.IMAGE_TYPE_CSS_CLASS
132 
133     @property {String}
134     @observes imageValue
135   */
136   type: function() {
137     var imageValue = this.get('imageValue');
138     if (SC.ImageView.valueIsUrl(imageValue)) return SC.IMAGE_TYPE_URL;
139     else if (!SC.none(imageValue)) return SC.IMAGE_TYPE_CSS_CLASS;
140     return SC.IMAGE_TYPE_NONE;
141   }.property('imageValue').cacheable(),
142 
143   /**
144     The canvas element is more performant than the img element, since we can
145     update the canvas image without causing browser reflow.  As an additional
146     benefit, canvas images are less easily copied, which is generally in line
147     with acting as an 'application'.
148 
149     @property {Boolean}
150     @default YES if supported
151     @since SproutCore 1.5
152   */
153   useCanvas: function() {
154     return SC.platform.supportsCanvas && !this.get('useStaticLayout') && this.get('type') !== SC.IMAGE_TYPE_CSS_CLASS;
155   }.property('useStaticLayout', 'type').cacheable(),
156 
157   /**
158     If YES, image view will use the SC.imageQueue to control loading.  This
159     setting is generally preferred.
160 
161     @property {Boolean}
162     @default YES
163   */
164   useImageQueue: YES,
165 
166   /**
167     A url or CSS class name.
168 
169     This is the image you want the view to display.  It should be either a
170     url or css class name.  You can also set the content and
171     contentValueKey properties to have this value extracted
172     automatically.
173 
174     If you want to use CSS spriting, set this value to a CSS class name.  If
175     you need to use multiple class names to set your icon, separate them by
176     spaces.
177 
178     Note that if you provide a URL, it must contain at least one '/' as this
179     is how we autodetect URLs.
180 
181     @property {String}
182   */
183   value: null,
184 
185   /**
186     Recalculate our innerFrame if the outer frame has changed.
187 
188     @returns {void}
189   */
190   // Note: SC.View's updateLayer() will call viewDidResize() if useStaticLayout is true.  The result of this
191   // is that since our display depends on the frame, when the view or parent view resizes, viewDidResize
192   // notifies that the frame has changed, so we update our view, which calls viewDidResize, which notifies
193   // that the frame has changed, so we update our view, etc. in an infinite loop.
194   viewDidResize: function() {
195     // 'frame' as a property is cached and won't yet be updated, however calling notifyPropertyChange on 'frame'
196     // causes the aforementioned infinite loop.  Instead, measure the frame ourselves and only notify if it has
197     // changed width or height
198     var layer = this.get('layer'),
199         width,
200         height;
201 
202     if (layer) {
203       width = layer.offsetWidth;
204       height = layer.offsetHeight;
205 
206       if (this._cachedWidth !== width || this._cachedHeight !== height) {
207         this.notifyPropertyChange('frame');
208         this._cachedWidth = width;
209         this._cachedHeight = height;
210       }
211     }
212   },

226     }

241     if (this.get('useCanvas')) this.set('layerNeedsUpdate', YES);

249   /** @private
250     Whenever the value changes, update the image state and possibly schedule
251     an image to load.
252   */
253   _image_valueDidChange: function() {
254     var value = this.get('imageValue'),
255         type = this.get('type');
256 
257     // check to see if our value has changed
258     if (value !== this._iv_value) {
259       this._iv_value = value;
260 
261       this.set('image', SC.BLANK_IMAGE);
262 
263       if (type !== SC.IMAGE_TYPE_CSS_CLASS) {
264         // While the new image is loading use SC.BLANK_IMAGE as a placeholder
265         this.set('status', SC.IMAGE_STATE_LOADING);
266 
267         // order: image cache, normal load
268         if (!this._loadImageUsingCache()) {
269           this._loadImage();
270         }
271       }
272     }
273   }.observes('imageValue'),
274 
275   /** @private
276     Tries to load the image value using the SC.imageQueue object. If the imageValue is not
277     a URL, it won't attempt to load it using this method.
278 
279     @returns YES if loading using SC.imageQueue, NO otherwise
280   */
281   _loadImageUsingCache: function() {
282     var value = this.get('imageValue'),
283         type = this.get('type');
284 
285     // now update local state as needed....
286     if (type === SC.IMAGE_TYPE_URL && this.get('useImageQueue')) {
287       var isBackground = this.get('isVisibleInWindow') || this.get('canLoadInBackground');
288 
289       SC.imageQueue.loadImage(value, this, this._loadImageUsingCacheDidComplete, isBackground);
290       return YES;
291     }
292 
293     return NO;
294   },
295 
296   _loadImageUsingCacheDidComplete: function(url, image) {
297     var value = this.get('imageValue');
298 
299     if (value === url) {
300       if (SC.ok(image)) {
301         this.didLoad(image);
302       } else {
303         // if loading it using the cache didn't work, it's useless to try loading the image normally
304         this.didError(image);
305       }
306     }
307   },
308 
309   /** @private
310     Loads an image using a normal Image object, without using the SC.imageQueue.
311 
312     @returns YES if it will load, NO otherwise
313   */
314   _loadImage: function() {
315     var value = this.get('imageValue'),
316         type = this.get('type'),
317         that = this,
318         image,
319         jqImage;
320 
321     if (type === SC.IMAGE_TYPE_URL) {
322       image = new Image();
323 
324       var errorFunc = function() {
325         SC.run(function() {
326           that._loadImageDidComplete(value, SC.$error("SC.Image.FailedError", "Image", -101));
327         });
328       };
329 
330       var loadFunc = function() {
331         SC.run(function() {
332           that._loadImageDidComplete(value, image);
333         });
334       };
335 
336       // Don't grab the jQuery object repeatedly
337       jqImage = $(image);
338 
339       // Using bind here instead of setting onabort/onerror/onload directly
340       // fixes an issue with images having 0 width and height
341       jqImage.bind('error', errorFunc);
342       jqImage.bind('abort', errorFunc);
343       jqImage.bind('load', loadFunc);
344 
345       image.src = value;
346       return YES;
347     }
348 
349     return NO;
350   },
351 
352   _loadImageDidComplete: function(url, image) {
353     var value = this.get('imageValue');
354 
355     if (value === url) {
356       if (SC.ok(image)) {
357         this.didLoad(image);
358       } else {
359         this.didError(image);
360       }
361     }
362   },
363 
364   didLoad: function(image) {
365     this.set('status', SC.IMAGE_STATE_LOADED);
366     if (!image) image = SC.BLANK_IMAGE;
367     this.set('image', image);
368   },
369 
370   didError: function(error) {
371     this.set('status', SC.IMAGE_STATE_FAILED);
372     this.set('image', SC.BLANK_IMAGE);
373   }
374

377 /**
378   Returns YES if the passed value looks like an URL and not a CSS class
379   name.
380 */
381 SC.ImageView.valueIsUrl = function(value) {
382   return value && SC.typeOf(value)===SC.T_STRING ? value.indexOf('/') >= 0 : NO ;
383 } ;
