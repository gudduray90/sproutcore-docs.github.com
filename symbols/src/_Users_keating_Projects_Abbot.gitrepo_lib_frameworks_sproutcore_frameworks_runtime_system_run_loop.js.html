 11 /**
 12   @class
 13 
 14   The run loop provides a universal system for coordinating events within
 15   your application.  The run loop processes timers as well as pending
 16   observer notifications within your application.
 17 
 18   To use a RunLoop within your application, you should make sure your event
 19   handlers always begin and end with SC.RunLoop.begin() and SC.RunLoop.end()
 20 
 21   The RunLoop is important because bindings do not fire until the end of
 22   your run loop is reached.  This improves the performance of your
 23   application.
 24 
 25   Example:
 26 
 27   This is how you could write your mouseup handler in jQuery:
 28 
 29         $('#okButton').on('click', function() {
 30           SC.RunLoop.begin();
 31 
 32           // handle click event...
 33 
 34           SC.RunLoop.end(); // allows bindings to trigger...
 35         });
 36 
 37   @extends SC.Object
 38   @since SproutCore 1.0
 39 */
 40 SC.RunLoop = SC.Object.extend(/** @scope SC.RunLoop.prototype */ {
 41 
 42   /**
 43     Call this method whenver you begin executing code.
 44 
 45     This is typically invoked automatically for you from event handlers and
 46     the timeout handler.  If you call setTimeout() or setInterval() yourself,
 47  48     you may need to invoke this yourself.
 49 
 50     @returns {SC.RunLoop} receiver
 51   */
 52   beginRunLoop: function() {
 53     this._start = new Date().getTime() ; // can't use Date.now() in runtime
 54     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 55       SC.Logger.log("-- SC.RunLoop.beginRunLoop at %@".fmt(this._start));
 56     }
 57     this._runLoopInProgress = YES;
 58     this._flushinvokeNextQueue();
 59     return this ;
 60   },
 61 
 62   /**
 63     YES when a run loop is in progress
 64 
 65     @property
 66     @type Boolean
 67   */
 68   isRunLoopInProgress: function() {
 69     return this._runLoopInProgress;
 70   }.property(),
 71 
 72   /**
 73     Call this method whenever you are done executing code.
 74 
 75     This is typically invoked automatically for you from event handlers and
 76     the timeout handler.  If you call setTimeout() or setInterval() yourself
 77     you may need to invoke this yourself.
 78 
 79     @returns {SC.RunLoop} receiver
 80   */
 81   endRunLoop: function() {
 82     // at the end of a runloop, flush all the delayed actions we may have
 83     // stored up.  Note that if any of these queues actually run, we will
 84     // step through all of them again.  This way any changes get flushed
 85     // out completely.
 86 
 87     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 88       SC.Logger.log("-- SC.RunLoop.endRunLoop ~ flushing application queues");
 89     } 
 90     
 91     this.flushAllPending();
 92     
 93     this._start = null ;
 94 
 95     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 96       SC.Logger.log("-- SC.RunLoop.endRunLoop ~ End");
 97     }
 98 
 99     SC.RunLoop.lastRunLoopEnd = Date.now();
100     this._runLoopInProgress = NO;
101 
102     return this ;
103   },
104 
105   /**
106     Repeatedly flushes all bindings, observers, and other queued functions until all queues are empty.
107   */
108   flushAllPending: function() {
109     var didChange ;
110     
111     do {
112       didChange = this.flushApplicationQueues() ;
113       if (!didChange) didChange = this._flushinvokeLastQueue() ; 
114     } while(didChange) ;
115   },
116   
117   
118   /**
119     Invokes the passed target/method pair once at the end of the runloop.
120     You can call this method as many times as you like and the method will
121     only be invoked once.
122 
123     Usually you will not call this method directly but use invokeOnce()
124     defined on SC.Object.
125 
126     Note that in development mode only, the object and method that call this
127     method will be recorded, for help in debugging scheduled code.
128 
129     @param {Object} target
130     @param {Function} method
131     @returns {SC.RunLoop} receiver
132   */
133   invokeOnce: function(target, method) {
134     // normalize
135     if (method === undefined) {
136       method = target; target = this ;
137     }
138 
139     if (typeof method === "string") method = target[method];
140     if (!this._invokeQueue) this._invokeQueue = SC.ObserverSet.create();
141     if ( method ) this._invokeQueue.add(target, method);
142     return this ;
143   },
144 
145   /**
146     Invokes the passed target/method pair at the very end of the run loop,
147     once all other delayed invoke queues have been flushed.  Use this to
148     schedule cleanup methods at the end of the run loop once all other work
149     (including rendering) has finished.
150 
151     If you call this with the same target/method pair multiple times it will
152     only invoke the pair only once at the end of the runloop.
153 
154     Usually you will not call this method directly but use invokeLast()
155     defined on SC.Object.
156 
157     Note that in development mode only, the object and method that call this
158     method will be recorded, for help in debugging scheduled code.
159 
160     @param {Object} target
161     @param {Function} method
162     @returns {SC.RunLoop} receiver
163   */
164   invokeLast: function(target, method) {
165     // normalize
166     if (method === undefined) {
167       method = target; target = this ;
168     }
169 
170     if (typeof method === "string") method = target[method];
171     if (!this._invokeLastQueue) this._invokeLastQueue = SC.ObserverSet.create();
172     this._invokeLastQueue.add(target, method);
173     return this ;
174 175   },
176 
177   /**
178     Invokes the passed target/method pair once at the beginning of the next
179     runloop, before any other methods (including events) are processed.
180     Use this to defer painting to make views more responsive.
181 
182     If you call this with the same target/method pair multiple times it will
183     only invoke the pair only once at the beginning of the next runloop.
184 
185     Usually you will not call this method directly but use invokeNext()
186     defined on SC.Object.
187 
188     Note that in development mode only, the object and method that call this
189     method will be recorded, for help in debugging scheduled code.
190 
191     @param {Object} target
192     @param {Function} method
193     @returns {SC.RunLoop} receiver
194   */
195   invokeNext: function(target, method) {
196     // normalize
197     if (method === undefined) {
198       method = target; target = this ;
199     }
200 
201     if (typeof method === "string") method = target[method];
202     if (!this._invokeNextQueue) this._invokeNextQueue = SC.ObserverSet.create();
203     this._invokeNextQueue.add(target, method);
204     return this ;
205   },
206 
207   /**
208     Executes any pending events at the end of the run loop.  This method is
209     called automatically at the end of a run loop to flush any pending
210     queue changes.
211 
212     The default method will invoke any one time methods and then sync any
213     bindings that might have changed.  You can override this method in a
214     subclass if you like to handle additional cleanup.
215 
216     This method must return YES if it found any items pending in its queues
217     to take action on.  endRunLoop will invoke this method repeatedly until
218     the method returns NO.  This way if any if your final executing code
219     causes additional queues to trigger, then can be flushed again.
220 
221     @returns {Boolean} YES if items were found in any queue, NO otherwise
222   */
223   flushApplicationQueues: function() {
224     var hadContent = NO,
225         // execute any methods in the invokeQueue.
226         queue = this._invokeQueue;
227 228     if ( queue && queue.getMembers().length ) {
229       this._invokeQueue = null; // reset so that a new queue will be created
230       hadContent = YES ; // needs to execute again
231       queue.invokeMethods();
232     }
233 
234     // flush any pending changed bindings.  This could actually trigger a
235     // lot of code to execute.
236     return SC.Binding.flushPendingChanges() || hadContent ;
237   },
238 
239   _flushinvokeLastQueue: function() {
240     var queue = this._invokeLastQueue, hadContent = NO ;
241     if (queue && queue.getMembers().length ) {
242       this._invokeLastQueue = null; // reset queue.
243       hadContent = YES; // has targets!
244       if (hadContent) queue.invokeMethods();
245     }
246     return hadContent ;
247   },
248 
249   _flushinvokeNextQueue: function() {
250     var queue = this._invokeNextQueue, hadContent = NO ;
251     if (queue && queue.getMembers().length ) {
252       this._invokeNextQueue = null; // reset queue.
253       hadContent = YES; // has targets!
254 255       if (hadContent) queue.invokeMethods();
256     }
257     return hadContent ;
258   }
259 
260 });

262 /**
263   The current run loop.  This is created automatically the first time you
264   call begin().
265 
266   @type SC.RunLoop
267 */
268 SC.RunLoop.currentRunLoop = null;
269 
270 /**
271   The default RunLoop class.  If you choose to extend the RunLoop, you can
272   set this property to make sure your class is used instead.
273 
274   @type Class
275 */
276 SC.RunLoop.runLoopClass = SC.RunLoop;
277 
278 /**
279   Begins a new run loop on the currentRunLoop.  If you are already in a
280   runloop, this method has no effect.
281 
282   @returns {SC.RunLoop} receiver
283 */
284 SC.RunLoop.begin = function() {
285   var runLoop = this.currentRunLoop;
286   if (!runLoop) runLoop = this.currentRunLoop = this.runLoopClass.create();
287   runLoop.beginRunLoop();
288   return this ;
289 };
290 
291 /**
292   Ends the run loop on the currentRunLoop.  This will deliver any final
293   pending notifications and schedule any additional necessary cleanup.
294 
295   @returns {SC.RunLoop} receiver
296 */
297 SC.RunLoop.end = function() {
298   var runLoop = this.currentRunLoop;
299   if (!runLoop) {
300     throw "SC.RunLoop.end() called outside of a runloop!";
301   }
302   runLoop.endRunLoop();
303   return this ;
304 } ;
305 
306 /**
307   Returns YES when a run loop is in progress
308 
309   @return {Boolean}
310 */
311 SC.RunLoop.isRunLoopInProgress = function() {
312   if(this.currentRunLoop) return this.currentRunLoop.get('isRunLoopInProgress');
313   return NO;
314 };
315 
316 /**
317   Executes a passed function in the context of a run loop. If called outside a 
318   runloop, starts and ends one. If called inside an existing runloop, is 
319   simply executes the function unless you force it to create a nested runloop.
320   
321   If an exception is thrown during execution, we give an error catcher the
322   opportunity to handle it before allowing the exception to bubble again.
323   
324   @param {Function} callback callback to execute
325   @param {Object} target context for callback
326   @param {Boolean} if YES, starts/ends a new runloop even if one is already running
327 */
328 SC.run = function(callback, target, forceNested) {
329   var alreadyRunning = SC.RunLoop.isRunLoopInProgress();
330 
331   // Only use a try/catch block if we have an ExceptionHandler
332   // since in some browsers try/catch causes a loss of the backtrace
333   if (SC.ExceptionHandler && SC.ExceptionHandler.enabled) {
334     try {
335       if(forceNested || !alreadyRunning) SC.RunLoop.begin();
336       if (callback) callback.call(target);
337       if(forceNested || !alreadyRunning) SC.RunLoop.end();
338     } catch (e) {
339 340       var handled = SC.ExceptionHandler.handleException(e);
341 
342       // If the exception was not handled, throw it again so the browser
343       // can deal with it (and potentially use it for debugging).
344       // (We don't throw it in IE because the user will see two errors)
345       if (!handled && !SC.browser.msie) {
346         throw e;
347       }
348     }
349   } else {
350     if(forceNested || !alreadyRunning) SC.RunLoop.begin();
351     if (callback) callback.call(target);
352     if(forceNested || !alreadyRunning) SC.RunLoop.end();
353   }
354 };
