400 test("Writing over a child record should remove caches in the store.", function() {
401   // Test Child Record creation
402   var cr, key, store = testParent.get('store'), cacheLength, storeKeys = [], ids = [];
403 
404   // Get the child record once before setting it in order to test that this child
405   // doesn't become abandoned in the store.
406   cr = testParent.get('info');
407   storeKeys.push(cr.get('storeKey'));
408   ids.push(cr.get('id'));
409   ids = ids.uniq();
410 
411   // Once we get the child record, certain caches are created in the store.
412   // Verify the cache lengths to prove that there are no leaked objects.
413   cacheLength = 0;
414   for (key in store.parentRecords) { cacheLength += 1; }
415   equals(cacheLength, 1, 'there should only be one parent record registered in the store');
416 
417   cacheLength = 0;
418   for (key in store.childRecords) { cacheLength += 1; }
419   equals(cacheLength, 1, 'there should only be one child record registered in the store');
420 
421   cacheLength = 0;
422   for (key in store.records) { cacheLength += 1; }
423   equals(cacheLength, 4, 'there should be four records cached in the store');
424 
425   cacheLength = 0;
426   for (key in store.dataHashes) { if (store.dataHashes[key] !== null) cacheLength += 1; }
427   equals(cacheLength, 4, 'there should be four non-null datahashes in the store');
428 
429   // Overwrite the child record with a new child record with the same guid.
430   testParent.set('info', {type: 'ChildRecordTest', name: 'New Child Name', value: 'Red Goo', guid: '5001'});
431   cr = testParent.get('info');
432   storeKeys.push(cr.get('storeKey'));
433   ids.push(cr.get('id'));
434   ids = ids.uniq();
435 
436   // Verify the cache lengths to prove that there are no leaked objects.
437   cacheLength = 0;
438   for (key in store.parentRecords) { cacheLength += 1; }
439   equals(cacheLength, 1, 'there should only be one parent record registered in the store after replacing child record once');

441   cacheLength = 0;
