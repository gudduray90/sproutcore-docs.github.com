  8 sc_require('system/drag') ;

 10 /**
 11   @namespace
 12 
 13   The DragSource protocol is used to dynamically generate multiple types of
 14   data from a single object.  You must implement this protocol if you want to
 15   provide the data for a drag event.
 16 */
 17 SC.DragSource = {
 18 
 19   /**
 20     This method must be overridden for drag operations to be allowed. 
 21     Return a bitwise OR'd mask of the drag operations allowed on the
 22     specified target.  If you don't care about the target, just return a
 23     constant value.
 24     
 25     The default implementation returns `SC.DRAG_NONE`
 26     
 27     @param {SC.View} dropTarget The proposed target of the drop.
 28     @param {SC.Drag} drag The SC.Drag instance managing this drag.
 29   */
 30   dragSourceOperationMaskFor: function(drag, dropTarget) {
 31     return SC.DRAG_NONE;
 32   },
 33   
 34   /**
 35     If this property is set to `NO` or is not implemented, then the user may
 36     modify the drag operation by changing the modifier keys they have 
 37     pressed.
 38     
 39     @type Boolean
 40     @default NO
 41   */
 42   ignoreModifierKeysWhileDragging: NO,
 43     
 44   /**
 45     This method is called when the drag begins. You can use this to do any
 46     visual highlighting to indicate that the receiver is the source of the 
 47     drag.
 48     
 49     @param {SC.Drag} drag The Drag instance managing this drag.
 50     @param {Point} loc The point in *window* coordinates where the drag 
 51       began.  You can use convertOffsetFromView() to convert this to local 
 52       coordinates.
 53   */
 54   dragDidBegin: function(drag, loc) {},
 55   
 56   /**
 57     This method is called whenever the drag image is moved.  This is
 58     similar to the `dragUpdated()` method called on drop targets.
 59     
 60     @param {SC.Drag} drag The Drag instance managing this drag.
 61     @param {Point} loc  The point in *window* coordinates where the drag 
 62       mouse is.  You can use convertOffsetFromView() to convert this to local 
 63       coordinates.
 64   */
 65   dragDidMove: function(drag, loc) {},
 66   
 67   /**
 68     This method is called when the drag ended. You can use this to do any
 69     cleanup.  The operation is the actual operation performed on the drag.
 70     
 71     @param {SC.Drag} drag The drag instance managing the drag.
 72     @param {Point} loc The point in WINDOW coordinates where the drag
 73       ended.
 74     @param {DragOp} op The drag operation that was performed. One of
 75       SC.DRAG_COPY, SC.DRAG_MOVE, SC.DRAG_LINK, or SC.DRAG_NONE.
 76   */
 77   dragDidEnd: function(drag, loc, op) {}
 78 
 79 };
