 43 var parent, child, parentDom, childDom ;
 44 module("SC.View#layer - autodiscovery", {
 45   setup: function() {
 46 
 47     parent = SC.View.create({
 48        childViews: [ SC.View.extend({
 49          // redefine this method in order to isolate testing of layer prop.
 50          // simple version just returns firstChild of parentLayer.
 51          findLayerInParentLayer: function(parentLayer) {
 52            return parentLayer.firstChild;
 53          }
 54        }) ]
 55     });
 56     child = parent.childViews[0];
 57 
 58     // setup parent/child dom
 59     parentDom = document.createElement('div');
 60     childDom = document.createElement('div');
 61     parentDom.appendChild(childDom);
 62 
 63     // set parent layer...
 64     parent.set('layer', parentDom);
 65   },
 66 
 67   teardown: function() {
 68     parent = child = parentDom = childDom = null ;
 69   }
 70 });
 71 
 72 test("discovers layer if has no layer but parent view does have layer", function() {
 73   equals(parent.get('layer'), parentDom, 'precond - parent has layer');
 74   ok(!!parentDom.firstChild, 'precond - parentDom has first child');
 75 
 76   equals(child.get('layer'), childDom, 'view discovered child');
 77 });

 81   parentDom.removeChild(childDom) ;
 82 
 83   equals(child.get('layer'), childDom, 'view kept layer cached (i.e. did not do a discovery again)');

146 var parentView;
147 
148 module("Notifies that layer has changed whenever re-render", {
149 
150   setup: function () {
151     parentView = SC.View.create({
152       childViews: ['a', 'b', SC.View],
153 
154       containerLayer: function () {
155         return this.$('._wrapper')[0];
156       }.property('layer').cacheable(),
157 
158       a: SC.View,
159       b: SC.View,
160 
161       render: function (context) {
162         context = context.begin().addClass('_wrapper');
163         this.renderChildViews(context);
164         context = context.end();
165       }
166     });
167   },
168 
169   teardown: function () {
170     parentView.destroy();
171     parentView = null;
172   }
173 
174 });
175 
176 /**
177   If the containerLayer property is cached, then when the view re-renders and the original container layer
178   element is lost, the containerLayer property will be invalid.  Instead, whenever the view updates,
179   we have to indicate that the 'layer' property has changed.
180   */
181 test("containerLayer should be able to be dependent on layer so that it invalidates when updated", function () {
182   var containerLayer;
183 
184   // Render the parent view and attach.
185   parentView.createLayer();
186   parentView._doAttach(document.body);
187 
188   // Get the container layer.
189   containerLayer = parentView.get('containerLayer');
190 
191   // Rerender the view.
192   SC.run(function () {
193     parentView.displayDidChange();
194   });
195 
196   ok(containerLayer !== parentView.get('containerLayer'), "The container layer should not be the same anymore.");
197 });
