 10 var pane, view ;
 11 
 12 module("SC.SplitView",{
 13   setup: function() {
 14     SC.RunLoop.begin();
 15 	pane = SC.MainPane.create({
 16 	  childViews: [ SC.SplitView.extend() ]
 17 	});
 18 	pane.append(); // make sure there is a layer...	    
 19 	SC.RunLoop.end();
 20 	
 21 	view = pane.childViews[0];
 22   },
 23     	
 24   teardown: function() {
 25     pane.remove();
 26     pane = view = null ;
 27   }		
 28 });
 29 
 30 test("the views are collapsible", function() {
 31 	equals(YES,view.canCollapseView(view.get('topLeftView')),'the top left view is collapsable');
 32 	equals(YES,view.canCollapseView(view.get('bottomRightView')),'the bottom right view is collapsable');	
 33 	equals(YES,view.splitViewCanCollapse(view,view.get('topLeftView')),'should return true');
 34 	view.set('canCollapseViews','NO');
 35 });
 36 
 37 test("the thickness of the views",function(){
 38 	ok(view.thicknessForView(view.get('topLeftView')),'thickness of the topLeftView');
 39 	ok(view.thicknessForView(view.get('bottomRightView')),'thickness of the bottomRightView');
 40 });
 41 
 42 test("Layout direction is Horizontal",function() {
 43   view.set('layoutDirection', SC.LAYOUT_HORIZONTAL) ;
 44   equals(view.getPath('thumbViewCursor.cursorStyle'),"ew-resize",'The Cursor is');
 45 });
 46 
 47 test("Layout direction is Vertical",function() {
 48   view.set('layoutDirection', SC.LAYOUT_VERTICAL) ;
 49   equals(view.getPath('thumbViewCursor.cursorStyle'),"ns-resize",'The Cursor is');
 50 });
