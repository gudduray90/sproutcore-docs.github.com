 40

 43         name: 'b'

 46         name: 'c'
 47       })],

 51       name: 'a'
 52     })

 54 
 55   view.createGestures();

124

132           equals(aTouch._isSeen, 0, "The value of _isSeen is set on the touch to");
133 
134           // Bump up _isSeen to assert the order of the gestures is correct.
135           aTouch._isSeen = 1;
136         },
137 
138         touchAddedToSession: function (aTouch, touchesAlreadyInSession) {
139           equals(aTouch, testTouch2, "The touch is passed to the gesture. The touch param is");

141         }
142       }),
143       SC.Gesture.extend({
144         name: 'b',
145 
146         touchSessionStarted: function (aTouch) {
147           equals(aTouch, testTouch, "The touch is passed to the gesture. The touch param is");
148           equals(aTouch._isSeen, 1, "The value of _isSeen is set on the touch to");
149         },
150 
151         touchAddedToSession: function (aTouch, touchesAlreadyInSession) {
