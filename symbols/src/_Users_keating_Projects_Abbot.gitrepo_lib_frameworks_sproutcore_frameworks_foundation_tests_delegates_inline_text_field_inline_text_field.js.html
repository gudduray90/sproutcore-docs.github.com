  7 
  8 (function() {
  9 var testPane = SC.ControlTestPane.design(), customEditor, commitEditor, discardEditor, failEditor;
 10 
 11 testPane.add('label', SC.LabelView, {
 12   value: "i am a test label"
 13 });
 14 
 15 customEditor = SC.View.extend(SC.InlineEditor);
 16 
 17 commitEditor = SC.View.extend(SC.InlineEditor, {
 18   didCommit: NO,
 19 
 20   commitEditing: function() {
 21     this.didCommit = YES;
 22     return YES;
 23   }
 24 });
 25 
 26 discardEditor = SC.View.extend(SC.InlineEditor, {
 27   didCommit: NO,
 28 
 29   commitEditing: function() {
 30     this.didCommit = YES;
 31     return NO;
 32   },
 33 
 34   didDiscard: NO,
 35 
 36   discardEditing: function() {
 37     this.didDiscard = YES;
 38     return YES;
 39   }
 40 });
 41 
 42 failEditor = SC.View.extend(SC.InlineEditor, {
 43   didCommit: NO,
 44 
 45   commitEditing: function() {
 46     this.didCommit = YES;
 47     return NO;
 48   },
 49 
 50   didDiscard: NO,
 51 
 52   discardEditing: function() {
 53     this.didDiscard = YES;
 54     return NO;
 55   }
 56 });
 57 
 58 testPane.show();
 59 
 60 module("SC.InlineTextFieldDelegate basic", testPane.standardSetup());
 61 
 62 test("basic acquire and release", function() {
 63   var label = testPane.view('label');
 64   var editor = SC.InlineTextFieldDelegate.acquireEditor(label);
 65 
 66   ok(editor.kindOf(SC.InlineTextFieldView), "acquired an inlineTextFieldView");
 67   same(editor.get('pane'), label.get('pane'), "editor created in the correct pane");
 68 
 69   SC.InlineTextFieldDelegate.releaseEditor(editor);
 70 
 71   same(editor.get('pane'), null, "editor removed from pane after release");
 72 });
 73 
 74 test("acquire custom editor", function() {
 75   var label = testPane.view('label');
 76 
 77   label.exampleEditor = customEditor;
 78 
 79   var editor = SC.InlineTextFieldDelegate.acquireEditor(label);
 80 
 81   ok(editor.kindOf(customEditor), "acquired a custom editor");
 82   same(editor.get('pane'), label.get('pane'), "editor created in the correct pane");
 83 
 84   SC.InlineTextFieldDelegate.releaseEditor(editor);
 85 
 86   same(editor.get('pane'), null, "editor removed from pane after release");
 87 });
 88 
 89 test("if second editor is requested, commit the first", function() {
 90   var label = testPane.view('label');
 91 
 92   label.exampleEditor = commitEditor;
 93 
 94   var first = SC.InlineTextFieldDelegate.acquireEditor(label);
 95 
 96   ok(first, "first editor was acquired");
 97 
 98   first.isEditing = YES;
 99 
100   var second = SC.InlineTextFieldDelegate.acquireEditor(label);
101 
102   ok(first.didCommit, "first editor was committed");
103   ok(!first.didDiscard, "first editor was not discarded");
104 
105   ok(second, "second editor was acquired");
106 
107   SC.InlineTextFieldDelegate.releaseEditor(first);
108   SC.InlineTextFieldDelegate.releaseEditor(second);
109 });
110 
111 test("if second editor is requested, commit the first, and discard if commit fails", function() {
112   var label = testPane.view('label');
113 
114   label.exampleEditor = discardEditor;
115 
116   var first = SC.InlineTextFieldDelegate.acquireEditor(label);
117 
118   first.isEditing = YES;
119 
120   var second = SC.InlineTextFieldDelegate.acquireEditor(label);
121 
122   ok(first.didCommit, "first editor was committed");
123   ok(first.didDiscard, "commit failed so discard was called");
124 
125   ok(second, "second editor was created");
126 
127   SC.InlineTextFieldDelegate.releaseEditor(first);
128   SC.InlineTextFieldDelegate.releaseEditor(second);
129 });
130 
131 test("if second editor is requested, fail to create second editor if commit and discard fail", function() {
132   var label = testPane.view('label');
133 
134   label.exampleEditor = failEditor;
135 
136   var first = SC.InlineTextFieldDelegate.acquireEditor(label);
137 
138   first.isEditing = YES;
139 
140   var second = SC.InlineTextFieldDelegate.acquireEditor(label);
141 
142   ok(first.didCommit, "first editor was committed");
143   ok(first.didDiscard, "commit failed so discard was called");
144 
145   equals(second, null, "second editor was not created");
146 
147   SC.InlineTextFieldDelegate.releaseEditor(first);
148 });
149 })();
