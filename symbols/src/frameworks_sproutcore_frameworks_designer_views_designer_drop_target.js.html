 13 SC.DesignerDropTarget = SC.ContainerView.extend(
 14   /** @scope SC.DesignerDropTarget.prototype */ {
 15

 17

 27   keyUp: function(evt) {
 28     return YES;
 29   },
 30 
 31   deleteForward: function(evt){
 32     var c = SC.designsController.getPath('page.designController');
 33     if(c) c.deleteSelection();
 34     return YES;
 35   },
 36 
 37   deleteBackward: function(evt){
 38     var c = SC.designsController.getPath('page.designController');
 39     if(c) c.deleteSelection();
 40     return YES;
 41   },
 42 
 43   moveLeft: function(sender, evt) {
 44     return YES;
 45   },
 46 
 47   moveRight: function(sender, evt) {
 48     return YES;
 49   },
 50 
 51   moveUp: function(sender, evt) {
 52     return YES;
 53   },
 54 
 55   moveDown: function(sender, evt) {
 56     return YES;
 57   },
 58 
 59   // ..........................................................
 60   // Drag and drop code
 61   //
 62   isDropTarget: YES,
 63 
 64   targetIsInIFrame: YES,
 65 
 66   dragStarted: function(drag, evt) {
 67   },
 68 
 69   dragEntered: function(drag, evt) {
 70   },
 71 
 72   dragUpdated: function(drag, evt) {},
 73 
 74   dragExited: function(drag, evt) {},
 75 
 76   dragEnded: function(drag, evt) {},
 77 
 78 
 79   computeDragOperations: function(drag, evt) {
 80     return SC.DRAG_ANY;
 81   },
 82 
 83 
 84   acceptDragOperation: function(drag, op) {
 85     var data = drag.dataForType('SC.Object'),
 86         scClass = eval(data.get('scClass'));
 87     return scClass.kindOf(SC.View);
 88   },
 89 
 90   /**
 91     Called to actually perform the drag operation.
 92 
 93     Override this method to actually perform the drag operation.  This method
 94     is only called if you returned `YES` in `acceptDragOperation()`.
 95 
 96     Return the operation that was actually performed or `SC.DRAG_NONE` if the
 97     operation was aborted.
 98 
 99     The default implementation returns `SC.DRAG_NONE`
100 
101     @param {SC.Drag} drag The drag instance managing this drag
102     @param {DragOp} op The proposed drag operation. A drag constant.
103 
104     @return {DragOp} Drag Operation actually performed
105   */
106   performDragOperation: function(drag, op) {
107     var data = drag.dataForType('SC.Object'),
108         cv = this.get('contentView'),
109         loc = drag.get('location'),
110         iframeOffset = drag.globalTargetOffset,
111         design, size, newView, defaults, layout;
112     var page = cv.get('page');
113     var designController = page.get('designController'),
114         rootDesigner = designController.get('rootDesigner');
115     var rootDesignerFrame = rootDesigner.get('frame');
116     //TODO: [MB] should we move most of this into the designer's addView?
117     //size and location
118     size = data.get('size');
119     loc.x = loc.x - iframeOffset.x - rootDesignerFrame.x;
120     loc.y = loc.y - iframeOffset.y - rootDesignerFrame.y;
121     //setup design (use eval to make sure code comes from iframe)
122     //TODO use new Function("return "+data.get('scClass))() ?...
123     design = eval(data.get('scClass'));
124     defaults = data.get('defaults') || {};
125     layout = defaults.layout || {};
126     layout = SC.merge(layout, {top: loc.y, left: loc.x});
127     //pull width and height from ghost if none exists form defaults
128     if(!layout.width) layout.width = drag.getPath('ghostView.layout').width;
129     if(!layout.height) layout.height = drag.getPath('ghostView.layout').height;
130     defaults.layout = layout;
131     design = design.design(defaults);
132     //drop it in the root designer
133     newView = design.create({page: page});
134     if(rootDesigner && newView){
135       rootDesigner.addView(newView);
136       //cv.appendChild(newView);
137     }
138     page.get('designController').select(newView.get('designer'));
139     return SC.DRAG_ANY;
140   }
