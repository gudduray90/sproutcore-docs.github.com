361   test("Check that overflow adds an overflow segment on view.", function() {
362     var sv = pane.view("5_items,widths,overflow");
363     var lastIsOverflow = function(sv) {
364       SC.RunLoop.begin().end(); // allow for a render to happen
365 
366       var segments=sv.$('.sc-segment-view');
367 368       var overflowEl = segments[segments.length - 1];
369       ok($(overflowEl).hasClass('sc-overflow-segment'), 'overflow segment should have .sc-overflow-segment class');
370       var overflowLabelEl = $(overflowEl).find('label')[0];
371       equals(overflowLabelEl.innerHTML, "Â»", 'there should be "Â»" text in the overflow segment');
372     };
373 
374     var lastIsSegment = function(sv, text) {
375       SC.RunLoop.begin().end(); // allow for a render to happen
376 
377       var segments=sv.$('.sc-segment-view');
378       var lastEl = segments[segments.length - 2];
379       ok(!$(lastEl).hasClass('sc-overflow-segment'), 'last segment should not have .sc-overflow-segment class');
380       var lastLabelEl = $(lastEl).find('label')[0];
381       equals(lastLabelEl.innerHTML, text, 'there should be "' + text + '" text in the last segment');
382     };
383 
384     // the last item should be an overflow segment (ie. has .sc-overflow-segment class and text "Â»")
385     lastIsOverflow(sv);
386 
387     // check that the overflowed items are stored
388 389     var overflowItems = sv.overflowItems;
390 
391     // five items, 70px wide, and a 342px wide container means 1 overflow.
392     equals(overflowItems.length, 1, "there should be 2 overflowed items");
393 
394     // 1. remove the last two items (the last item should no longer be an overflow segment)
395     var items = sv.get('items');
396     items.removeAt(items.length - 1);
397     items.removeAt(items.length - 1);
398     lastIsSegment(sv, "C");
399 
400     // 2. add an item (the last item should be an overflow segment again)
401     items.pushObject({value: 'X', width: 100});
402     lastIsOverflow(sv);
403 
404     // 3. shrink the items (the last item should no longer be an overflow segment)
405     items.invoke('set', 'width', 50);
406     lastIsSegment(sv, "X");
407 
408     // 4. grow the items (the last item should be an overflow segment again)
409     items.invoke('set', 'width', 100);
410     lastIsOverflow(sv);
411 
412     // 5. shrink the items, but then shrink the segmented view
413     items.invoke('set', 'width', 50);
414     lastIsSegment(sv, "X");
415     sv.set('layout', {left: 75, right: 75, top: 0, height: 25});
416     sv.updateLayer();
417     SC.RunLoop.begin().end(); // allow for a render to happen (update layout)
418     sv.notifyPropertyChange('frame');
419     SC.RunLoop.begin().end(); // allow for a render to happen (measure the segments)
420     lastIsOverflow(sv);
