 18   /**
 19     Show loading targets view.

    var target = TestRunner.sourceController.get('selection').firstObject();
 27     TestRunner.updateRoute(target, null, NO);
 28     if (TestRunner.get('routePending')) {

 31       else TestRunner.updateRoute(target, null, YES);
 32     } 
 33     
 34   },
