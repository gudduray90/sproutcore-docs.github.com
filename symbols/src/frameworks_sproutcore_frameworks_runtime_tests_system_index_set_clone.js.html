  7 /*global module test equals context ok same notest */
  8 var set ;
  9 module("SC.IndexSet#clone", {
 10   setup: function() {
 11     set = SC.IndexSet.create();
 12   }
 13 });
 14 
 15 test("clone should return new object with same key properties", function() {
 16   set.add(100,100).add(200,100);
 17   set.source = "foo";
 18   
 19   var set2 = set.clone();
 20   ok(set2 !== null, 'return value should not be null');
 21   ok(set2 !== set, 'cloned set should not be same instance as set');
 22   ok(set.isEqual(set2), 'set.isEqual(set2) should be true');
 23   
 24   equals(set2.get('length'), set.get('length'), 'clone should have same length');
 25   equals(set2.get('min'), set.get('min'), 'clone should have same min');
 26   equals(set2.get('max'), set.get('max'), 'clone should have same max');
 27   equals(set2.get('source'), set.get('source'), 'clone should have same source');
 28 
 29 });
 30 
 31 test("cloning frozen object returns unfrozen", function() {
 32   var set2 = set.freeze().clone();
 33   equals(set2.get('isFrozen'), NO, 'set2.isFrozen should be NO');
 34 });
 35 
 36 test("copy works like clone", function() {
 37   same(set.copy(), set, 'should return copy');
 38   ok(set.copy() !== set, 'should not return same instance');
 39   
 40   set.freeze();
 41   equals(set.frozenCopy(), set, 'should return same instance when frozen');
 42 });
