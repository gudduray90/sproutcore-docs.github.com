 10 module("SC.Record normalize method", {
 11   setup: function() {
 12 
 13     SC.RunLoop.begin();
 14  
 15     MyApp = SC.Object.create({
 16       store: SC.Store.create()
 17     });
 18     
 19     MyApp.Foo = SC.Record.extend({
 20       
 21       guid: SC.Record.attr(String, { defaultValue: function() {
 22         var i, rnum, chars = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXTZabcdefghiklmnopqrstuvwxyz",
 23           strLen = 8, ret = '';
 24     		for (i=0; i<strLen; i++) {
 25     			rnum = Math.floor(Math.random() * chars.length);
 26     			ret += chars.substring(rnum,rnum+1);
 27     		}
 28     		return ret;
 29       } }),
 30       
 31       // test simple reading of a pass-through prop
 32       firstName: SC.Record.attr(String),
 33 
 34       // test Array
 35       anArray: SC.Record.attr(Array),
 36       
 37       // used to test default value
 38       defaultValue: SC.Record.attr(String, {
 39         defaultValue: "default"
 40       }),
 41       
 42       // test toOne relationships
 43       relatedTo: SC.Record.toOne('MyApp.Foo', { defaultValue: '1' }),
 44       
 45       // test toOne relationship computed default
 46       relatedToComputed: SC.Record.toOne('MyApp.Foo', { 
 47         defaultValue: function() {
 48           var num = Math.floor(Math.random()*2+1);
 49           return 'foo' + num;
 50         }
 51       }),
 52       
 53       // test toMany relationships
 54       relatedToMany: SC.Record.toMany('MyApp.Foo')
 55  
 56     });
 57 
 58     // A parent record
 59     MyApp.FooParent = SC.Record.extend({
 60       nestedRecordNamespace: MyApp,
 61       myChild: SC.ChildAttribute.attr('MyApp.FooChild')
 62     });
 63 
 64     // A child record
 65     MyApp.FooChild = SC.Record.extend({
 66     });
 67     
 68     MyApp.Bar = SC.Record.extend({
 69       // test toOne relationships
 70       relatedTo: SC.Record.toOne('MyApp.Bar', { defaultValue: '1' })
 71     });
 72     
 73     MyApp.OneBar = SC.Record.extend({
 74       manyFoos: SC.Record.toMany('MyApp.ManyFoo', {
 75         key: 'many_foos',
 76         inverse: 'oneBar'
 77       })
 78     });
 79     
 80     MyApp.ManyFoo = SC.Record.extend({
 81       oneBar: SC.Record.toOne('MyApp.OneBar', {
 82         key: 'bar_id',
 83         inverse: 'manyFoos'
 84       })
 85     });
 86     
 87     storeKeys = MyApp.store.loadRecords(MyApp.Foo, [
 88       { 
 89         guid: 'foo1', 
 90         firstName: 123, 
 91         anArray: ['one', 'two', 'three']

 93

100       {
101         guid: 'foo3'
102       }
103       
104     ]);
105     
106     rec = MyApp.store.find(MyApp.Foo, 'foo1');
107     rec2 = MyApp.store.find(MyApp.Foo, 'foo2');
108     rec3 = MyApp.store.find(MyApp.Foo, 'foo3');
109     
110     equals(rec.storeKey, storeKeys[0], 'should find record');

124 test("normalizing a pre-populated record" ,function() {
125   
126   equals(rec.attributes()['firstName'], 123, 'hash value of firstName is 123');
127   equals(rec.get('firstName'), '123', 'get value of firstName is 123 string');
128   
129   rec.normalize();
130   
131   var sameValue = rec.attributes()['firstName'] === '123';
132   var relatedTo = rec.attributes()['relatedTo'] === '1';
133   var relatedToComputed = rec.attributes()['relatedToComputed'];
134   
135   var computedValues = ['foo1', 'foo2', 'foo3'];
136   
137   ok(sameValue, 'hash value of firstName after normalizing is 123 string');
138   ok(sameValue, 'hash value of relatedTo should be 1');
139   ok(computedValues.indexOf(relatedToComputed)!==-1, 'hash value of relatedToComputed should be either foo1, foo2 or foo3');
140   
141   equals(rec.get('firstName'), '123', 'get value of firstName after normalizing is 123 string');
142   
143 });

169 test("normalizing a new record with toOne should reflect id in data hash" ,function() {
170 
171   var recHash = { 
172     guid: 'foo4', 
173     firstName: "Jack",
174     relatedTo: 'foo1'
175   };
176 
177   var newRecord = MyApp.store.createRecord(MyApp.Foo, recHash);
178   MyApp.store.commitRecords();
179   
180   equals(newRecord.attributes()['relatedTo'], 'foo1', 'hash value of relatedTo is foo1');
181   equals(newRecord.get('relatedTo'), rec, 'get value of relatedTo is foo1');
182 
183   newRecord.normalize();
184   
185   equals(newRecord.attributes()['relatedTo'], 'foo1', 'hash value of relatedTo after normalizing is still foo1');
186   equals(newRecord.get('relatedTo'), rec, 'get value of relatedTo after normalizing remains foo1');
187
