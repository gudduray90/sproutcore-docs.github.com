  5 /*globals Greenhouse */
  6 /** @class
  7 
  8   @extends SC.PickerPane
  9 */
 10 Greenhouse.TearOffPicker = SC.PickerPane.extend(
 11 /** @scope Greenhouse.TearOffPicker.prototype */ {
 12     
 13   dragAction: '',
 14   
 15   mouseDragged: function(evt){
 16     
 17     Greenhouse.sendAction(this.get('dragAction'));
 18     this._blockedIframe = YES;
 19     Greenhouse.eventBlocker.set('isVisible', YES);
 20     
 21     return sc_super();
 22   },
 23   
 24   mouseUp: function(evt){
 25     if(this._blockedIframe){
 26       Greenhouse.eventBlocker.set('isVisible', NO);
 27       this._blockedIframe = NO;
 28     }
 29     return YES;
 30   },
 31   
 32   mouseDown: function(evt) {
 33     var f=this.get('frame');
 34     this._mouseOffsetX = f ? (f.x - evt.pageX) : 0;
 35     this._mouseOffsetY = f ? (f.y - evt.pageY) : 0;
 36     return this.modalPaneDidClick(evt);
 37   },
 38   
 39  40   modalPaneDidClick: function(evt) {
 41     var f = this.get("frame");
 42     if(!this.clickInside(f, evt)){ 
 43       Greenhouse.sendAction('cancel');
 44     }
 45     return YES ; 
 46   },
 47   
 48   computeAnchorRect: function(anchor) {
 49     var ret = SC.viewportOffset(anchor); // get x & y
 50     var cq = SC.$(anchor);
 51     var wsize = SC.RootResponder.responder.computeWindowSize() ;
 52     ret.width = cq.outerWidth();
 53     ret.height = (wsize.height-ret.y) < cq.outerHeight() ? (wsize.height-ret.y) : cq.outerHeight();
 54     ret.y = ret.y -11;
 55     return ret ;
 56   }
 57 });
