  9   /**
 10     Set to true when the item is enabled.   Note that changing this value
 11     will alter the isVisibleInWindow property for this view and any
 12     child views as well as to automatically add or remove a 'disabled' CSS
 13     class name.
 14 
 15     This property is observable and bindable.
 16 
 17     @property {Boolean}
 18   */
 19   isEnabled: YES,
 20   isEnabledBindingDefault: SC.Binding.oneWay().bool(),
 21 
 22   /**
 23     Computed property returns YES if the view and all of its parent views
 24     are enabled in the pane.  You should use this property when deciding
 25     whether to respond to an incoming event or not.
 26 
 27     This property is not observable.
 28 
 29  30     @property {Boolean}
 31   */
 32   isEnabledInPane: function() {
 33     var ret = this.get('isEnabled'), pv ;
 34     if (ret && (pv = this.get('parentView'))) { ret = pv.get('isEnabledInPane'); }
 35     return ret ;
 36   }.property('parentView', 'isEnabled'),
 37 
 38   /** @private
 39     Observes the isEnabled property and resigns first responder if set to NO.
 40     This will avoid cases where, for example, a disabled text field retains
 41     its focus rings.
 42 
 43     @observes isEnabled
 44   */
 45   _sc_view_isEnabledDidChange: function(){
 46     if(!this.get('isEnabled') && this.get('isFirstResponder')){
 47       this.resignFirstResponder();
 48     }
 49   }.observes('isEnabled'),
 50 
 51   applyAttributesToContext: function(original, context) {
 52     var isEnabled = this.get('isEnabled');
 53 
 54     original(context);
 55 
 56     context.setClass('disabled', !isEnabled);
 57     context.attr('aria-disabled', !isEnabled ? 'true' : null);
 58   }.enhance()
 59 });
