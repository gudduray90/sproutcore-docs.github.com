 69 /**
 70   @namespace
 71   
 72   A Control is a view that also implements some basic state functionality.
 73   Apply this mixin to any view that you want to have standard control
 74   functionality including showing a selected state, enabled state, focus
 75   state, etc.
 76   
 77   ## About Values and Content
 78   
 79   Controls typically are used to represent a single value, such as a number,
 80   boolean or string.  The value a control is managing is typically stored in
 81   a "value" property.  You will typically use the value property when working
 82   with controls such as buttons and text fields in a form.
 83   
 84   An alternative way of working with a control is to use it to manage some
 85   specific aspect of a content object.  For example, you might use a label
 86   view control to display the "name" property of a Contact record.  This 
 87   approach is often necessary when using the control as part of a collection
 88   view.
 89   
 90   You can use the content-approach to work with a control by setting the 
 91   "content" and "contentValueKey" properties of the control.  The 
 92   "content" property is the content object you want to manage, while the 
 93   "contentValueKey" is the name of the property on the content object 
 94   you want the control to display.
 95   
 96   The default implementation of the Control mixin will essentially map the
 97   contentValueKey of a content object to the value property of the 
 98   control.  Thus if you are writing a custom control yourself, you can simply
 99   work with the value property and the content object support will come for
100   free.  Just write an observer for the value property and update your 
101   view accordingly.
102   
103   If you are working with a control that needs to display multiple aspects
104   of a single content object (for example showing an icon and label), then
105   you can override the contentValueDidChange() method instead of observing
106   the value property.  This method will be called anytime _any_ property 
107   on the content object changes.  You should use this method to check the
108   properties you care about on the content object and update your view if 
109   anything you care about has changed.
110   
111   ## Delegate Support
112   
113   Controls can optionally get the contentDisplayProperty from a 
114   displayDelegate, if it is set.  The displayDelegate is often used to 
115   delegate common display-related configurations such as which content value
116   to show.  Anytime your control is shown as part of a collection view, the
117   collection view will be automatically set as its displayDelegate.
118   
119   @since SproutCore 1.0
120   @extends SC.ContentValueSupport
121 */
122 SC.Control = SC.mixin(SC.clone(SC.ContentValueSupport),
123 /** @scope SC.Control.prototype */{
124   
125   /**
126     Walk like a duck
127     
128     @type Boolean
129     @default YES
130     @readOnly
131   */
132   isControl: YES,
133   
134   /** 
135     The selected state of this control. Possible values:
136     
137       - `YES`
138       - `NO`
139       - SC.MIXED_STATE.
140     
141     @type Boolean
142     @default NO
143   */
144   isSelected: NO,
145   
146   /** @private */
147   isSelectedBindingDefault: SC.Binding.oneWay().bool(),
148   
149   /**
150     Set to YES when the item is currently active.  Usually this means the 
151     mouse is current pressed and hovering over the control, however the 
152     specific implementation my vary depending on the control.
153     
154     Changing this property value by default will cause the Control mixin to
155     add/remove an 'active' class name to the root element.
156     
157     @type Boolean
158     @default NO
159   */
160   isActive: NO,
161   
162   /** @private */
163   isActiveBindingDefault: SC.Binding.oneWay().bool(),
164   
165   /**
166     The name of the property this control should display if it is part of an
167     SC.FormView.
168     
169     If you add a control as part of an SC.FormView, then the form view will 
170     automatically bind the value to the property key you name here on the 
171     content object.
172     
173     @type String
174     @default null
175   */
176   fieldKey: null,
177   
178   /**
179     The human readable label you want shown for errors.  May be a loc string.
180     
181     If your field fails validation, then this is the name that will be shown
182     in the error explanation.  If you do not set this property, then the 
183     fieldKey or the class name will be used to generate a human readable name.
184     
185     @type String
186     @default null
187   */
188   fieldLabel: null,
189   
190   /**
191     The human readable label for this control for use in error strings.  This
192     property is computed dynamically using the following rules:
193     
194     If the fieldLabel is defined, that property is localized and returned.
195     Otherwise, if the keyField is defined, try to localize using the string 
196     "ErrorLabel.{fieldKeyName}".  If a localized name cannot be found, use a
197     humanized form of the fieldKey.
198     
199     Try to localize using the string "ErrorLabel.{ClassName}". Return a 
200     humanized form of the class name.
201     
202     @field
203     @type String
204     @observes 'fieldLabel'
205     @observes 'fieldKey'
206   */
207   errorLabel: function() {
208     var ret, fk, def, locFK;
209     if (ret = this.get('fieldLabel')) return ret;
210     
211 212     // if field label is not provided, compute something...
213     fk = this.get('fieldKey') || this.constructor.toString();
214     def = fk ? SC.String.capitalize(SC.String.humanize(fk)) : '';
215     locFK = SC.String.locWithDefault("FieldKey." + fk, def);
216     return SC.String.locWithDefault("ErrorLabel." + fk, locFK);
217   }.property('fieldLabel','fieldKey').cacheable(),
218 
219   /**
220     The control size.  This will set a CSS style on the element that can be 
221     used by the current theme to vary the appearance of the control.
222     
223     Some controls will default to SC.AUTO_CONTROL_SIZE, which will allow you
224     to simply size the control, and the most appropriate control size will
225     automatically be picked; be warned, though, that if you don't specify
226     a height, performance will be impacted as it must be calculated; if you do
227     this, a warning will be issued. If you don't care, use SC.CALCULATED_CONTROL_SIZE.
228     
229     @type String
230 231     @default SC.REGULAR_CONTROL_SIZE
232   */
233   controlSize: SC.REGULAR_CONTROL_SIZE,
234   
235   /** @private */
236   displayProperties: ['isEnabled', 'isSelected', 'isActive', 'controlSize'],
237   
238   /** @private */
239   _CONTROL_TMP_CLASSNAMES: {},
240   
241   /** @private
242     Invoke this method in your updateDisplay() method to update any basic 
243     control CSS classes.
244   */
245   renderMixin: function(context, firstTime) {
246     var sel = this.get('isSelected'), disabled = !this.get('isEnabled'),
247     // update the CSS classes for the control.  note we reuse the same hash
248     // to avoid consuming more memory
249     names = this._CONTROL_TMP_CLASSNAMES ; // temporary object
250     names.mixed = sel === SC.MIXED_STATE;
251     names.sel = sel && (sel !== SC.MIXED_STATE) ;
252     names.active = this.get('isActive') ;
253 
254     var controlSize = this.get("controlSize");
255     if (!controlSize) {
256       controlSize = SC.REGULAR_CONTROL_SIZE;
257     }
258 
259     if (firstTime) {
260       context.setClass(names);
261 
262       // delegates handle adding 'controlSize' on their own. We only support it
263       // here for backwards-compatibility.
264       if (!this.get('renderDelegate')) {
265         context.addClass(controlSize);
266       }
267     } else {
268       context.$().setClass(names);
269       if (!this.get('renderDelegate')) {
270         context.$().addClass(controlSize);
271       }
272     }
273 
274     // if the control implements the $input() helper, then fixup the input
275     // tags
276     if (!firstTime && this.$input) {
277       var inps = this.$input();
278       if(inps.attr('type')!=="radio"){
279         this.$input().attr('disabled', disabled);
280       }
281     }
282   }
283 
284 });
