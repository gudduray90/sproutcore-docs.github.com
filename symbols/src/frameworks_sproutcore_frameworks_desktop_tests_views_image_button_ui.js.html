  2 var pane = SC.ControlTestPane.design({height:24})
  3   .add('default', SC.ImageButtonView, {
  4     layout: { left: 0, top: 2, right: 0, bottom: 2 },
  5     image: 'start',
  6     toolTip: 'OMGHI'
  7   })
  8 
  9   .add("iconchange", SC.ImageButtonView, {
 10     layout: { left: 0, top: 2, right: 0, bottom: 2 },
 11     image: 'start'
 12   });
 13 
 14 module('SC.ImageButtonView ui', pane.standardSetup());
 15 
 16 test("Check if image class is set properly on ImageButton", function() {
 17   var viewElem = pane.view('default').$();
 18   ok(viewElem.hasClass('start'), 'Icon class set initially to "start"');
 19 });
 20 
 21 test("toolTip support", function() {
 22   var view = pane.view('default'),
 23       viewElem = view.$();
 24 
 25   equals(viewElem.attr('title'), 'OMGHI', 'title attribute is set correctly');
 26   equals(viewElem.attr('alt'), 'OMGHI', 'alt attribute is set correctly');
 27 
 28   SC.run(function() {
 29     view.set('toolTip', 'ZOMG');
 30   });
 31 
 32   equals(viewElem.attr('title'), 'ZOMG', 'title attribute is updated correctly');
 33   equals(viewElem.attr('alt'), 'ZOMG', 'alt attribute is updated correctly');
 34 
 35   SC.run(function() {
 36     view.set('toolTip', null);
 37   });
 38 
 39   equals(viewElem.attr('title'), "", 'title attribute is removed correctly');
 40   equals(viewElem.attr('alt'), "", 'alt attribute is removed correctly');
 41 });
 42 
 43 test("Check if image class is set properly on ImageButton if changed", function() {
 44   SC.RunLoop.begin();
 45   var viewElem = pane.view('iconchange');
 46   viewElem.set('image','stop');
 47   SC.RunLoop.end(); // force redraw...
 48   var newViewElem = pane.view('iconchange').$();
 49   ok(newViewElem.hasClass('stop'), 'Icon class has correctly changed to "stop"');
 50 });
