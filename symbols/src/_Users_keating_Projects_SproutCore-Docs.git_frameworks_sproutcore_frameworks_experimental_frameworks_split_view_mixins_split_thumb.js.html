  8 /**
  9   Tells the SplitThumb to automatically choose which child of the SplitView
 10   to move in response to touch or mouse events in an SC.SplitThumb.
 11 */
 12 SC.MOVES_AUTOMATIC_CHILD = 'moves-automatic-child';
 13 
 14 /**
 15   Tells the SplitThumb to move the child of the SplitView that is
 16   either the SplitThumb or a parent of it.
 17 */
 18 SC.MOVES_CHILD = 'moves-child';
 19 
 20 /**
 21   Tells the SplitThumb to move the child of the SplitView that
 22   preceeds the child that is either the SplitThumb or a parent of it.
 23 */
 24 SC.MOVES_PREVIOUS_CHILD = 'moves-previous-child';
 25 
 26 /**
 27   Tells the SplitThumb to move the child of the SplitVie that
 28  29   comes after the child that is either the SplitThumb or a parent of it.
 30 */
 31 SC.MOVES_NEXT_CHILD = 'moves-next-child';
 32 
 33 /**
 34   @namespace
 35   A SplitThumb may be placed inside any view in a SplitView, and can even
 36   be a direct child of the SplitView. It forwards its events to the SplitView
 37   to control the movement of a divider or another child of the SplitView.
 38   
 39   Using a view that mixes in SplitThumb, you can place a handle that moves the
 40   divider anywhere inside the SplitView's view tree.
 41   
 42   SplitThumb will automatically choose which divider to move. It's choice will
 43   almost always be correct for 2-pane layouts. However, in 3-pane layouts, you 
 44   may want to adjust the way it chooses.
 45   
 46   You can adjust the behavior by setting the movesSibling property to 
 47   SC.MOVES_NEXT_CHILD or SC.MOVES_PREVIOUS_CHILD. If your ThumbView is inside
 48   the middle pane, for instance, this would tell it whether the ThumbView 
 49   should move the divider to the left, or the divider to the right.
 50 */
 51 SC.SplitThumb = {
 52   
 53   /**
 54    * The child which should be moved by any mouse or touch events. Should usually
 55    * be a divider view.
 56    *
 57    * The default implementation calculates the child by first finding
 58    * the ancestor of this view that is a direct child of the SplitView,
 59    * and then either returning it or one of its immediate siblings, depending
 60    * on the value of the movesSibling property.
 61    *
 62    * @property SC.View
 63   */
 64   movesChild: function() {
 65     var view = this, child, splitView = this.get('splitView'),
 66         sibling = this.get('movesSibling');
 67     while (view && view !== splitView) {
 68       child = view;
 69       view = view.get('parentView');
 70     }
 71     
 72     if (sibling === SC.MOVES_NEXT_CHILD) return child.nextView;
 73     if (sibling === SC.MOVES_PREVIOUS_CHILD) return child.previousView;
 74 
 75     if (sibling === SC.MOVES_AUTOMATIC_CHILD) {
 76       if (!child.nextView) return child.previousView;
 77       if (!child.previousView) return child.nextView;
 78     }
 79     
 80     return child;
 81   }.property('splitView', 'movesSibling').cacheable(),
 82   
 83   /**
 84    * Decides whether an ancestor of this SplitThumb view or one of its
 85    * siblings is to be moved in response to events.
 86    *
 87    * Usually, you want events to move a divider. If the SplitThumb is inside
 88    * a view which is _not_ a divider, you probably want to move one of the
 89    * view's siblings-one that _is_ a divider.
 90    *
 91    * You can tell SC.SplitThumb to:
 92    *
 93    * - SC.MOVES_AUTOMATIC_CHILD: if the SplitView child is either first or last,
 94    *   moves the adjacent child (likely a divider). Otherwise moves the child itself.
 95    *
 96    * - SC.MOVES_CHILD: moves the child itself, not one of its siblings. Divider
 97    *   views could use this setting.
 98    *
 99    * - SC.MOVES_PREVIOUS_CHILD: moves the previous child. For instance, in a
100    *   two-pane setup, if the SplitThumb is in the rightmost child, this will
101    *   move the divider between the two children.
102    *
103    * - SC.MOVES_NEXT_CHILD: moves the next child.
104 105    *
106    * @type TYPE
107   */
108   movesSibling: SC.MOVES_AUTOMATIC_CHILD,
109   
110   /**
111    * The SplitView that contains the child views to be adjusted.
112    *
113    * This is computed to be the first SplitView found in a search
114    * up the view hierarchy. You can substitute your own SplitView
115    *
116    * @property SC.SplitView
117   */
118   splitView: function() {
119     var view = this ;
120     while (view && !view.isSplitView) view = view.get('parentView') ;
121     return view ;
122   }.property('parentView').cacheable(),
123   
124   /**
125    * The layoutDirection of the SplitView. This is observed so that we
126    * can update positioning if the layoutDirection changes but the position
127    * and size properties do not.
128    *
129    * NOTE: duplicated in SplitChild because both this and SplitChild use it.
130    *
131    * @type {LayoutDirection}
132   */
133   // NOTE: While an edge case, this is implemented because it makes it _much_
134   // easier to write the sample in the Test Controls app.
135   splitViewLayoutDirection: null,
136   splitViewLayoutDirectionBinding: SC.Binding.oneWay('*splitView.layoutDirection'),
137   
138   /**
139    * The name of the CSS cursor that should be used for splitting.
140    * The containing SplitView will adopt this cursor if and when this
141    * view is dragged.
142    *
143    * Computed based on the SplitView's layoutDirection.
144    *
145    * @type {String}
146   */
147   splitCursorStyle: function() {
148     if (this.get('splitViewLayoutDirection') === SC.LAYOUT_HORIZONTAL) {
149       return 'ew-resize';
150     } else {
151 
152       return 'ns-resize';
153     }
154   }.property('splitViewLayoutDirection').cacheable(),
155   
156   splitCursorStyleDidChange: function() {
157     if (this._isDragging) {
158       this.get('splitView').set('splitChildCursorStyle', this.get('splitCursorStyle'));
159     }
160     
161     this.$().css('cursor', this.get('splitCursorStyle'));
162   }.observes('splitCursorStyle'),
163   
164   /**
165    * @private
166    * Renders the cursor for the view as defined by this view's splitCursor
167    * property.
168   */
169   renderMixin: function(context) {
170     context.css('cursor', this.get('splitCursorStyle'));
171   },
172   
173   
174   //
175   // EVENT HANDLING
176   //
177   touchStart: function(touch) {
178     this._isDragging = YES;
179     
180     var splitView = this.get('splitView');
181     splitView.beginLiveResize();
182     
183     this._scst_mouseStartPosition = splitView.get('layoutDirection') === SC.LAYOUT_HORIZONTAL ? 
184       touch.pageX : touch.pageY;
185     
186     this._scst_childStartPosition = splitView.getPositionForChild(this.get('movesChild'));
187     
188     return YES;
189   },
190   
191   touchesDragged: function(evt) {
192     var splitView = this.get('splitView');
193     
194     var mousePosition = splitView.get('layoutDirection') === SC.LAYOUT_HORIZONTAL ? 
195       evt.pageX : evt.pageY;
196     
197     var diff = mousePosition - this._scst_mouseStartPosition,
198         start = this._scst_childStartPosition;
199     
200     splitView.adjustPositionForChild(this.get('movesChild'), start + diff);
201 
202     return YES;
203   },
204   
205   touchEnd: function(touch) {
206     this._isDragging = NO;
207     
208     var splitView = this.get('splitView');
209     
210     var mousePosition = splitView.get('layoutDirection') === SC.LAYOUT_HORIZONTAL ? 
211       touch.pageX : touch.pageY;
212     
213     var diff = mousePosition - this._scst_mouseStartPosition,
214         start = this._scst_childStartPosition;
215     
216     splitView.adjustPositionForChild(this.get('movesChild'), start + diff);
217     
218     splitView.set('splitChildCursorStyle', null);
219     splitView.endLiveResize();
220     return YES;
221   },
222 
223   mouseDown: function(evt) {
224     var splitView = this.get('splitView');
225     splitView.set('splitChildCursorStyle', this.get('splitCursorStyle'));
226     
227     return this.touchStart(evt);
228   },
229   
230   mouseDragged: function(evt) {
231     return this.touchesDragged(evt);
232   },
233   
234   mouseUp: function(evt) {
235     this.get('splitView').set('splitChildCursorStyle', null);
236     
237     return this.touchEnd(evt);
238   }
239   
240 };
