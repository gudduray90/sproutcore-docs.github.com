 15 var view, pane, originalSupportsTransitions = SC.platform.supportsCSSTransitions;

 17 function styleFor(view) {
 18   return view.get('layer').style;
 19 }
 20 
 21 function transitionFor(view){
 22   return styleFor(view)[SC.platform.domCSSPrefix+"Transition"];
 23 }

204   test("should not cancel callback when value hasn't changed", function() {
205     var callbacks = 0, wasCancelled = NO, check = 0;
206     stop(2000);
207 
208     SC.RunLoop.begin();
209     view.invokeLater(function() {
210       // this triggers the initial layoutStyle code
211       view.animate('left', 79, 0.500, function(data) {
212         callbacks++;
213         wasCancelled = data.isCancelled;
214       });
215       // this triggers a re-render, re-running the layoutStyle code
216       view.displayDidChange();
217     }, 1);
218     SC.RunLoop.end();
219 
220     setTimeout(function() {
221       // capture the callbacks value
222       check = callbacks;
223     }, 250);
224 
225     setTimeout(function() {
226       start();
227       equals(check, 0, "the callback should not have been cancelled initially");
228       equals(callbacks, 1, "the callback should have been fired");
229       equals(wasCancelled, NO, "the callback should not have been cancelled");
230     }, 1000);
231   });
