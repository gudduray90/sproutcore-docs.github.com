146   init: function() {
147     sc_super();
148 
149     // call valueDidChange to get the initial item, if any
150     this._scsv_valueDidChange();

153   /**
154     @private
155 
156     This gets the value for a specific menu item. This function allows a few different
157     forms of input:
158 
159     - A string: returns the string.
160     - A hash: returns hash[itemValueKey], using 'value' for the key if necessary.
161     - An SC.Object: returns object.get(itemValueKey), using 'value' for the key if needed.
162 
163     This method therefore accepts both the menu items as created for the menupane's displayItems
164     AND the raw items provided by the developer in `items`.
165   */
166   _scsv_getValueForMenuItem: function(item) {
167     var valueKey = this.get('itemValueKey') || 'value';
168 
169     if (SC.typeOf(item) === SC.T_STRING) {
170       return item;
171     } else if (item.get) {
172       return item.get(valueKey);
173     } else {
174       return item[valueKey];
175     }
176   },
177 
178   /**
179     * When the selected item changes, we need to update our value.
180     * @private
181   */
182   _scsv_selectedItemDidChange: function() {
183     var sel = this.get('selectedItem'),
184         last = this._scsv_lastSelection,
185         titleKey = this.get('itemTitleKey') || 'title',
186         valueKey = this.get('itemValueKey') || 'value';
187 
188     // selected item could be a menu item from SC.MenuPane's displayItems, or it could
189     // be a raw item. So, we have to use _scsv_getValueForMenuItem to resolve it.
190     if(sel) {
191       this.setIfChanged('value', this._scsv_getValueForMenuItem(sel));
192     }
193 
194     // add/remove observers for the title and value so we can invalidate.
195     if (last && last.addObserver && sel !== last) {
196       last.removeObserver(titleKey, this, this._scsv_selectedItemPropertyDidChange);
197       last.removeObserver(valueKey, this, this._scsv_selectedItemPropertyDidChange);
198     }
199     
200     if (sel && sel.addObserver && sel !== last) {
201       sel.addObserver(titleKey, this, this._scsv_selectedItemPropertyDidChange);
202       sel.addObserver(valueKey, this, this._scsv_selectedItemPropertyDidChange);
203     }
204 
205     this._scsv_lastSelection = sel;
206   }.observes('selectedItem'),
207 
208   // called when either title or value changes on the selected item
209   _scsv_selectedItemPropertyDidChange: function(item) {
210     this.notifyPropertyChange('title');
211     this.set('value', item.get(this.get('itemValueKey') || 'value'));
212   },

223   /**
224     The title of the button, derived from the selected item.
225   */
226   title: function() {
227     var sel = this.get('selectedItem');
228 
229     if (!sel) {
230       return this.get('defaultTitle');
231     } else if (sel.get) {
232       return sel.get(this.get('itemTitleKey') || 'title');
233     } else if (SC.typeOf(sel) == SC.T_HASH) {
234       return sel[this.get('itemTitleKey') || 'title'];
235     } else {
236       return sel;
237     }
238   }.property('selectedItem').cacheable(),
239 
240   /**
241     * When the value changes, we need to update selectedItem.
242     * @private
243   */
244   _scsv_valueDidChange: function() {
245     var value = this.get('value');
246 
247     if (!this.get('items')) {
248       return;
249     }
250 
251     var items = this.get('items'), len = items.length, idx;
252     for (idx = 0; idx < len; idx++) {
253       if (this._scsv_getValueForMenuItem(items[idx]) === value) {
254         this.setIfChanged('selectedItem', items[idx]);
255         return;
256       }
257     }
258 
259     // if we got here, this means no item is selected
260     this.setIfChanged('selectedItem', null);
261   }.observes('value'),
262 
263   /**
264     SelectView must set the selectView property on the menu so that the menu's

266     the SelectViewMenu mixin, which should be mixed in to any SelectView menu.

268     In addition, the initial selected item and the initial minimum menu width are set.
269     @private

273       selectView: this,

324   /**
325     The prefer matrix for menu positioning. It is calculated so that the selected
326     menu item is positioned directly over the SelectView.
327     
328     @property
329     @type Array
330     @private
331   */
332   menuPreferMatrix: function() {
333     var menu = this.get('menu'),
334         leftPosition = this.get('menuLeftOffset'),
335         topPosition = this.get('menuTopOffset');
336 
337     if (!menu) {
338       return [leftPosition, topPosition, 2];
339     }
340 
341     var idx = this.get('_selectedItemIndex'), itemViews = menu.get('menuItemViews');
342     if (idx > -1) {
343       var layout = itemViews[idx].get('layout');
344       return [leftPosition, topPosition - layout.top + (layout.height/2), 2];
345     }
346 
347     return [leftPosition, topPosition, 2];
348 
349   }.property('value', 'menu').cacheable(),
350 
351   /**
352     Used to calculate things like the menu's top position.
353 
354     @private
355   */
356   _selectedItemIndex: function() {
357     var menu = this.get('menu');
358     if (!menu) {
359       return -1;
360     }
361 
362     // We have to find the selected item, and then get its 'top' position so we
363     // can position the menu correctly.
364     var itemViews = menu.get('menuItemViews'), idx, len = itemViews.length, view;
365     for (idx = 0; idx < len; idx++) {
366       view = itemViews[idx];
367 
368       // we have to compare via value
369       var value = view.get('content').get(this.get('itemValueKey'));
370       if (value === this.get('value')) {
371         break;
372       }
373     }
374 
375     if (idx < len) {
376       return idx;
377     }
378 
379     return -1;
380   }.property('value', 'menu').cacheable(),
381 
382   /**
383     The minimum width for the child menu. For instance, this property can make the
384     menu always cover the entire SelectView--or, alternatively, cover all but the
385     arrows on the side.
386 
387     By default, it is calculated by adding the menuMinimumWidthOffset to the view's 
388     width. If you are writing a theme and want to change the width so the menu covers
389     a specific part of the select view, change your render delegate's menuMinimumWidthOffset
390     property.
391 
392     @type Number
393     @property
394   */
395   minimumMenuWidth: function() {
396     return this.get('frame').width + this.get('menuMinimumWidthOffset');
397   }.property('frame', 'menuMinimumWidthOffset').cacheable(),

416   /**
417     @private
418 
419     Pressing the Up or Down arrow key should display the menu pane
420   */
421   interpretKeyEvents: function(event) {
422     if (event) {
423       if ((event.keyCode === 38 || event.keyCode === 40)) {
424         this.showMenu();
425         return YES;
426       }
427       else if (event.keyCode === 27) {
428         this.resignFirstResponder() ;
429         return YES;
430       }
431     }
432     return sc_super();
433   },
