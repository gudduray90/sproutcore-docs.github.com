  7 /*global module test equals context ok same should_throw*/
  8 var set ;
  9 module("SC.IndexSet#add", {
 10   setup: function() {
 11     set = SC.IndexSet.create();
 12   }
 13 });
 14 
 15 function iter(s) {
 16   var ret = [];
 17   set.forEach(function(k) { ret.push(k); });
 18   return ret ;
 19 }

 25 test("add range to end of set", function() {
 26   set.add(1000,5);
 27   equals(set.get('length'), 5, 'should have correct index count');  
 28   equals(set.get('max'), 1005, 'max should return 1 past last index');
 29   same(iter(set), [1000,1001,1002,1003,1004]);
 30 });
 31 
 32 test("add range into middle of empty range", function() {
 33   set.add(100,2); // add initial set.
 34   equals(iter(set)[0], 100, 'precond - first index is 100');
 35   
 36   // now add second range
 37   set.add(10,1);
 38   equals(set.get('length'), 3, 'should have extra length');
 39   equals(set.get('max'), 102, 'max should return 1 past last index');
 40   same(iter(set), [10, 100, 101]);
 41 });
 42 
 43 test("add range overlapping front edge of range", function() {
 44   set.add(100,2); // add initial set.
 45   equals(iter(set)[0], 100, 'precond - first index is 100');
 46   
 47   // now add second range
 48   set.add(99,2);
 49   equals(set.get('length'), 3, 'should have extra length');
 50   equals(set.get('max'), 102, 'max should return 1 past last index');
 51   same(iter(set), [99, 100, 101]);
 52 });
 53 
 54 test("add range overlapping last edge of range", function() {
 55   set.add(100,2).add(200,2);
 56   same(iter(set), [100,101,200,201], 'should have two sets');
 57   
 58   // now add overlapping range
 59   set.add(101,2);
 60   equals(set.get('length'), 5, 'new set.length');
 61   equals(set.get('max'), 202, 'max should return 1 past last index');
 62   same(iter(set), [100,101,102,200,201], 'should include 101-102');
 63 });
 64 
 65 test("add range overlapping two ranges, merging into one", function() {
 66   set.add(100,2).add(110,2);
 67   same(iter(set), [100,101,110,111], 'should have two sets');
 68   
 69   // now add overlapping range
 70   set.add(101,10);
 71   equals(set.get('length'), 12, 'new set.length');
 72   equals(set.get('max'), 112, 'max should return 1 past last index');
 73   same(iter(set), [100,101,102,103,104,105,106,107,108,109,110,111], 'should include one range 100-111');
 74 });
 75 
 76 test("add range overlapping three ranges, merging into one", function() {

172   equals(set.get('max'), 4000, 'should have max');
173   equals(set.get('length'), 2000, 'should have length');
174   
175 });
