  8 /**
  9   @class
 10   Enhances SC.MenuItemView to support auto resize.
 11 */
 12 SC.MenuItemView = SC.MenuItemView; // for docs
 13 
 14 SC.MenuItemView.reopen(SC.AutoResize);
 15 SC.MenuItemView.reopen(
 16 /** @scope SC.MenuItemView.prototype */{
 17 
 18   //
 19   // For automatic resizing, if enabled (to be enabled by parent menu)
 20   //
 21   /**
 22     The item view is capable of automatic resizing.
 23     
 24     @private
 25     @property
 26     @type {Boolean}
 27   */
 28   supportsAutoResize: YES,
 29 
 30   /**
 31  32     The menu item should NOT change its own width and height.
 33     
 34     @private
 35     @property
 36     @type {Boolean}
 37   */
 38   shouldAutoResize: NO,
 39   
 40   /**
 41     If YES, the menu item will measure its width and height so that the menu
 42     can automatically resize itself. This is usually set by the parent menu.
 43     
 44     @property
 45     @type {Boolean}
 46     @default NO
 47   */
 48   shouldMeasureSize: NO,
 49 
 50   // NOTE: this property could come from the theme at some point, but MenuItemView
 51   // would have to be migrated to render delegates first. MenuPane adds the
 52   // necessary padding for now.
 53   autoResizePadding: 0,
 54   
 55   /** @private */
 56   autoResizeText: function() {
 57     return this.get('title');
 58   }.property('title'),
 59 
 60   /** @private */
 61   autoResizeLayer: function() {
 62     return this.$('.value')[0];
 63   }.property('layer').cacheable(),
 64 
 65   /**
 66    * @private
 67    * When we render, we recreate all of the DOM, including the element that gets measured.
 68    * This is a problem because our autoResizeLayer changes. So, we must invalidate that
 69    * layer whenever we re-render.
 70    *
 71    * We need to move menu item rendering into a render delegate. When this happens, there
 72    * are a few ways we could do it:
 73    *
 74    * - Give render delegate method to find clientWidth and return it: 
 75    *   getMenuItemTitleWidth(dataSource, $)
 76    *
 77    * - Make render delegate provide the autoResizeLayer:
 78    *   In this case, the autoResizeLayer might be a computed property that we invalidate
 79    *   on didUpdateLayer, and that calls a method like getAutoResizeLayer. Alternatively,
 80    *   if render delegate properties are added, we could make this one of those, but it
 81    *   would need some way to access the DOM. Maybe data sources can have $ properties or
 82    *   methods? Maybe a jQuery property/method?
 83   */
 84   didUpdateLayer: function() {
 85     this.notifyPropertyChange('autoResizeLayer');
 86     this.scheduleMeasurement();
 87   }.enhance()
 88 
 89 }) ;
