 21 test("should recognize all primitives", function() {
 22   // PROPERTY
 23   q.conditions = "what_to_do_now";
 24   q.parse();
 25   ok(q._tokenList.length == 1, 'list should have one token');
 26   equals(q._tokenList[0].tokenType, 'PROPERTY', 'type should be PROPERTY');
 27   equals(q._tokenList[0].tokenValue, 'what_to_do_now', 'value should be what_to_do_now');
 28   
 29   // PROPERTY - one character
 30   q.conditions = "a";
 31   q.parse();
 32   ok(q._tokenList.length == 1, 'list should have one token');
 33   equals(q._tokenList[0].tokenType, 'PROPERTY', 'type should be PROPERTY');
 34   equals(q._tokenList[0].tokenValue, 'a', 'value should be "a"');
 35   
 36   // BOOLEAN VALUE - false
 37   q.conditions = "false";
 38   q.parse();
 39   ok(q._tokenList.length == 1, 'list should have one token');
 40   equals(q._tokenList[0].tokenType, 'false', 'type should be false');
 41   equals(q._tokenList[0].tokenValue, 'false', 'value should be false');
 42   
 43   // BOOLEAN VALUE - true
 44   q.conditions = "true";
 45   q.parse();
 46   ok(q._tokenList.length == 1, 'list should have one token');
 47   equals(q._tokenList[0].tokenType, 'true', 'type should be true');
 48   equals(q._tokenList[0].tokenValue, 'true', 'value should be true');
 49   
 50   // NULL
 51   q.conditions = "null";
 52   q.parse();
 53   ok(q._tokenList.length == 1, 'list should have 2 tokens');
 54   equals(q._tokenList[0].tokenType, 'null', 'type should be null');
 55 
 56   // NULL
 57   q.conditions = "undefined";
 58   q.parse();
 59   ok(q._tokenList.length == 1, 'list should have 2 tokens');
 60   equals(q._tokenList[0].tokenType, 'undefined', 'type should be undefined');
 61

 81   equals(q._tokenList[0].tokenValue, -1, 'value should be -1');
 82 
 83   // NUMBER - negative float

 85   q.parse();
 86   ok(q._tokenList.length == 1, 'list should have one token');
 87   equals(q._tokenList[0].tokenType, 'NUMBER', 'type should be NUMBER');
 88   equals(q._tokenList[0].tokenValue, -1.234, 'value should be -1.234');
 89   
 90   // STRING - single quoted
 91   q.conditions = "'ultravisitor'";
 92   q.parse();
 93   ok(q._tokenList.length == 1, 'list should have one token');
 94   equals(q._tokenList[0].tokenType, 'STRING', 'type should be STRING');
 95   equals(q._tokenList[0].tokenValue, 'ultravisitor', 'value should be ultravisitor');
