1 // The second responsibility of SC.SelectViewMenu is to make sure the selected
  2 // item is checked.
  3 
  4 module("SelectViewMenu -- Check Selected");
  5 
  6 test("Selected menu item is checked", function() {
  7   SC.RunLoop.begin();
  8   var selectView = SC.SelectView.create({
  9     items: "A B C D E".w(),
 10     value: "A"
 11   });
 12   selectView.showMenu();
 13 
 14   SC.RunLoop.end();
 15 
 16   var menu = selectView.get('menu');
 17 
 18   equals(menu.get('menuItemViews')[0].get('isChecked'), YES, "First menu item is checked");
 19 
 20   SC.RunLoop.begin();
 21   selectView.set('value', 'C');
 22   SC.RunLoop.end();
 23 
 24   equals(menu.get('menuItemViews')[0].get('isChecked'), NO, "First menu item is checked");
 25   equals(menu.get('menuItemViews')[2].get('isChecked'), YES, "Third menu item is checked");
 26 
 27   SC.RunLoop.begin();
 28   selectView.hideMenu();
 29   SC.RunLoop.end();
 30 });
