 15 module("SC.View#updateLayer");
 16 
 17 test("invokes renderLayerSettings() and then updates layer element", function() {
 18   var layer = document.createElement('div');
 19 
 20   var times = 0;
 21   var view = SC.View.create({
 22     _renderLayerSettings: function() {
 23       times++;
 24       this.$().addClass('did-update-' + times);
 25     }
 26   });
 27   view.createLayer();
 28   view.updateLayer();
 29   ok(view.$().attr('class').indexOf('did-update-2')>=0, 'has class name added by prepareContext()');
 30 });

 33 // TEST: updateLayerIfNeeded()
 34 // 
 35 var view, callCount ;
 36 module("SC.View#updateLayerIfNeeded", {
 37   setup: function() {
 38     // setup a fake view class so that updateLayerIfNeeded() will call
 39     // updateLayer() if needed.  updateLayer() is faked to isolate test
 40     view = SC.View.create({
 41       isVisibleInWindow: YES,
 42       updateLayer: function() { callCount++; }
 43     });
 44     callCount = 0 ;
 45     
 46     view.createLayer();
 47     view.set("layerNeedsUpdate", YES);
 48   }
 49   
 50 });
 51 
 52 test("does not call updateLayer if layerNeedsUpdate is NO", function() {
 53   view.set('layerNeedsUpdate', NO);
 54   view.updateLayerIfNeeded();
 55   equals(callCount, 0, 'updateLayer did NOT run');
 56 });
 57 
 58 test("does not call updateLayer if isVisibleInWindow is NO", function() {
 59   view.set('isVisibleInWindow', NO);
 60   view.updateLayerIfNeeded();
 61   equals(callCount, 0, 'updateLayer did NOT run');
 62 });
 63 
 64 test("does call updateLayer() if isVisible & layerNeedsUpdate", function() {
 65   equals(view.get('isVisibleInWindow'), YES, 'precond - isVisibleInWindow');
 66   equals(view.get('layerNeedsUpdate'), YES, 'precond - layerNeedsUpdate');
 67   
 68   view.updateLayerIfNeeded();
 69   ok(callCount > 0, 'updateLayer() did run');
 70 });
 71 
 72 test("resets layerNeedsUpdate to NO if called", function() {
 73   equals(view.get('layerNeedsUpdate'), YES, 'precond - layerNeedsUpdate');
 74   view.updateLayerIfNeeded();
 75   equals(view.get('layerNeedsUpdate'), NO, 'layerNeedsUpdate reset to NO');
 76 });
 77 
 78 test("returns receiver", function() {
 79   equals(view.updateLayerIfNeeded(), view, 'returns receiver');
 80 });
 81 
 82 test("only runs updateLayer() once if called multiple times (since layerNeedsUpdate is set to NO)", function() {
 83   callCount = 0;
 84   view.updateLayerIfNeeded().updateLayerIfNeeded().updateLayerIfNeeded();
 85   equals(callCount, 1, 'updateLayer() called only once');
 86 });

 88 // ..........................................................
 89 // TEST: layerNeedsUpdate auto-trigger
 90 // 
 91 module("SC.View#layerNeedsUpdate auto-triggers", {
 92   setup: function() {
 93     // use fake method to isolate call...
 94     view = SC.View.create({
 95       updateLayerIfNeeded: function() { callCount++; }
 96     });
 97     callCount = 0;
 98   }
 99 });
100 
101 test("setting layerNeedsUpdate calls updateLayerIfNeeded at end of runloop", function() {
102   SC.RunLoop.begin();
103   view.set('layerNeedsUpdate', YES);
104   SC.RunLoop.end();
105   
106   equals(callCount, 1, 'updateLayerIfNeeded did run');  
107 });
108 
109 test("setting & resetting only triggers updateLayerIfNeeded once per runloop", function() {
110   SC.RunLoop.begin();
111   view.set('layerNeedsUpdate', YES)
112       .set('layerNeedsUpdate', NO)
113       .set('layerNeedsUpdate', YES);
114   SC.RunLoop.end();
115   
116   equals(callCount, 1, 'updateLayerIfNeeded did run');  
117 });

123 module("SC.View#updateLayer - integration");
124 
125 test("layerNeedsUpdate actually triggers updateLayer", function() {
126   var callCount = 0 ;
127   var layer = document.createElement('div');
128   var view = SC.View.create({
129     isVisibleInWindow: YES,
130     updateLayer: function() { callCount++; }
131   });
132   view.createLayer();
133   
134   SC.RunLoop.begin();
135   view.set('layerNeedsUpdate', YES);
136   SC.RunLoop.end();
137   
138   equals(callCount, 1, 'updateLayer did run b/c layerNeedsUpdate is YES');
139   callCount = 0 ;
140   
141   SC.RunLoop.begin();
142   view.set('layerNeedsUpdate', YES);
143   view.set('layerNeedsUpdate', NO);
144   SC.RunLoop.end();
145   
146   equals(callCount, 0, 'updateLayer did NOT run b/c layerNeedsUpdate is NO');
147 });
