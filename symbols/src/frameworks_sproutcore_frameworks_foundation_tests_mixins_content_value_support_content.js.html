 12 var view, content;
 13 module('SC.ContentValueSupport#contentPropertyDidChange', {
 14   setup: function () {
 15     content = SC.Object.create();
 16     view = SC.View.create(SC.ContentValueSupport);
 17   },
 18 
 19   teardown: function () {
 20     content = null;
 21     view.destroy();
 22   }
 23 });
 24 
 25 test("invoked with key = * whenever content changes", function () {
 26   view.contentPropertyDidChange = function (target, key) {
 27     ok(target === content, 'should pass content object as target target=%@'.fmt(target));
 28     equals(key, '*', 'should pass * as key');
 29   };
 30   view.set('content', content);
 31 });
 32 
 33 test("should not be invoked when arbitrary keys are changed", function () {
 34   var isTesting = NO, count = 0;
 35   view.contentPropertyDidChange = function (target, key) {
 36     if (!isTesting) return; //wait until testing should begin...
 37     count++;
 38   };
 39 
 40   view.set('content', content);
 41 
 42   isTesting = YES;
 43 
 44   content.set('foo', 'foo');
 45   content.set('bar', 'bar');
 46 
 47   equals(count, 0, "method was not invoked");
 48 });
 49 
 50 test("should no longer be invoked when a key is changed on a former content object", function () {
 51   var isTesting = NO;
 52   view.contentPropertyDidChange = function (target, key) {
 53     if (!isTesting) return; //wait until testing should begin...
 54     ok(NO, 'should not invoke contentPropertyDidChange after content is removed');
 55   };
 56 
 57   view.set('content', content);
 58   content.set('foo', 'foo');
 59   view.set('content', null);
 60 
 61   isTesting = YES;
 62   content.set('bar', 'bar');
 63 });

 65 test("should fire even on a content object set when the object is created", function () {

 78 test("should call updatePropertyFromContent for all properties when content changes", function () {
 79   var originalContent = SC.Object.create({
 80     contentFoo: 'foo1',
 81     contentBar: 'bar1'
 82   });
 83 
 84   var fooDidChangeCount = 0,
 85     barDidChangeCount = 0;
 86 
 87   var view = SC.View.create(SC.ContentValueSupport, {
 88     contentKeys: {
 89       'contentFooKey': 'foo',
 90       'contentBarKey': 'bar'
 91     },
 92     contentFooKey: 'contentFoo',
 93     contentBarKey: 'contentBar',
 94 
 95     foo: null,
 96     bar: null,
 97 
 98     fooDidChange: function () {
 99       fooDidChangeCount++;
100     }.observes('foo'),
101 
102     barDidChange: function () {
103       barDidChangeCount++;
104     }.observes('bar'),
105 
106     content: null
107   });
108 
109   equals(fooDidChangeCount, 0, "foo indicated as changed this many times");
110   equals(barDidChangeCount, 0, "bar indicated as changed this many times");
111 
112   view.set('content', originalContent);
113 
114   equals(fooDidChangeCount, 1, "foo indicated as changed this many times");
115   equals(barDidChangeCount, 1, "bar indicated as changed this many times");
116 
117   // set new content
118   var newContent = SC.Object.create({
119     contentFoo: 'foo2',
120     contentBar: 'bar2'
121   });
122 
123   view.set('content', newContent);
124 
125   equals(fooDidChangeCount, 2, "foo indicated as changed this many times");
126   equals(barDidChangeCount, 2, "bar indicated as changed this many times");
127 
128   // Clean up.
129   originalContent.destroy();
130   newContent.destroy();
131 });

158 });

207   equals(content.get('foo'), 'foo', 'should not change');

225 // updateContentWithValueObserver()
226 //
227 module("SC.ContentValueSupport#contentKeys", {
228   setup: function () {
229     this.count = 0;
230     var self = this;
231 
232     this.obj = SC.Object.create(SC.ContentValueSupport, SC.DelegateSupport, {
233       contentKeys: {'contentFooKey': 'foo'},
234       contentFooKey: 'foo',
235       content: SC.Object.create({foo: 'BAR'}),
236       contentPropertyDidChange: function (orig, target, key) {
237         equals(target, this.content, "content is target");
238         self.count++;
239 
240         return orig(target, key);
241       }.enhance()
242     });
243   },
244 
245   teardown: function () {
246     this.obj.destroy();
247 
248     this.obj = null;
249   }
250 });
251 
252 test("different contentKeys on creation are observed correctly", function () {
253   equals(this.count, 1, "Content property setup was called on init");
254 
255   this.obj.content.set('foo', 'BAR2');
256 
257   equals(this.count, 2, "observer was called again on set");
258 
259   equals(this.obj.get('foo'), 'BAR2', "value is updated correctly");
260 
261   this.obj.content.set('bar', 'ASDF');
262 
263   equals(this.count, 2, "observer was not called again on setting other keys");
264 });
265 
266 test("different contentKeys after creation are observed correctly", function () {
267   equals(this.count, 1,  "Content property setup was called on init");
268 
269   this.obj.beginPropertyChanges();
270   this.obj.set({
271     contentKeys: {'contentBarKey': 'bar'},
272     contentBarKey: 'bar'
273   });
274   this.obj.endPropertyChanges();
275 
276   equals(this.count, 2, "observer was called again when changing contentKeys");
277 
278   this.obj.content.set('bar', 'BAR2');
279 
280   equals(this.count, 3, "observer was called when changing bar");
281 
282   equals(this.obj.get('bar'), 'BAR2', "value is updated correctly");
283 
284   this.obj.content.set('asdfasf', 'asdfasd');
285 
286   equals(this.count, 3, "observer was not called again on setting other keys");
287 
288   this.obj.content.set('foo', 'asdfasd');
289 
290   equals(this.count, 3, "observer was not called again on setting old observed keys");
291 });
