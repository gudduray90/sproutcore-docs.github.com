  6 /*globals module ok equals same test Sample */
  7 
  8 var store;
  9 
 10 module("SC.CascadeDataSource", {
 11   setup: function () {
 12     SC.RunLoop.begin();
 13 
 14     var Sample = (window.Sample = SC.Object.create());
 15 
 16     Sample.FooDataSource = SC.DataSource.extend({
 17       commitRecords: function (st, createStoreKeys, updateStoreKeys, destroyStoreKeys, params) {
 18         equals(store, st, "should equal store");
 19         equals(0, createStoreKeys[0], "should equal [0]");
 20         equals(1, updateStoreKeys[0], "should equal [1]");
 21         equals(2, destroyStoreKeys[0], "should equal [2]");
 22         equals('world', params.hello, 'should equal { hello: "world" }');
 23         return YES;
 24       },
 25 
 26       retriveRecords: function () {
 27         ok(true, "retriveRecords should be handled by baz");
 28         return NO;
 29       },
 30 
 31       fetch: function (store, query) {
 32         ok(true, "fetch should be handled by bar");
 33         return NO;
 34       },
 35 
 36       cancel: function (st, storeKeys) {
 37         equals(store, st, "should equal store");
 38         equals(1, storeKeys[0], "should equal [1]");
 39         return NO;
 40       }
 41     });
 42     Sample.fooDataSource = Sample.FooDataSource.create();
 43 
 44     Sample.BarDataSource = SC.DataSource.extend({
 45       commitRecords: function () {
 46         ok(false, "should never be called, since foo handles commitRecords first");
 47         return NO;
 48       },
 49 
 50       retriveRecords: function () {
 51         ok(true, "retriveRecords should be handled by baz");
 52         return NO;
 53       },
 54 
 55       fetch: function (st, query) {
 56         equals(store, st, "should equal store");
 57         equals('query', query, "should equal 'query'");
 58         return YES;
 59       }
 60     });
 61     Sample.barDataSource = Sample.BarDataSource.create();
 62 
 63     Sample.BazDataSource = SC.DataSource.extend({
 64       commitRecords: function () {
 65         ok(false, "should never be called, since foo handles commitRecords first");
 66         return NO;
 67       },
 68 
 69       retrieveRecords: function (st, storeKeys, ids) {
 70         equals(store, st, "should equal store");
 71         equals(0, storeKeys[0], "should equal [0]");
 72         equals("id", ids[0], 'should equal ["id"]');
 73         return YES;
 74       },
 75 
 76       fetch: function () {
 77         ok(false, "should never be called, since bar handles fetch first");
 78         return NO;
 79       }
 80     });
 81     Sample.bazDataSource = Sample.BazDataSource.create();
 82 
 83     Sample.dataSource = SC.CascadeDataSource.create({
 84       dataSources: "foo bar baz".w(),
 85 
 86       foo: Sample.fooDataSource,
 87       bar: Sample.barDataSource,
 88       baz: Sample.bazDataSource
 89     });
 90 
 91     store = SC.Store.create({ dataSource: Sample.dataSource });
 92   },
 93 
 94   teardown: function () {
 95     SC.RunLoop.end();
 96   }
 97 });
 98 
 99 test("Verify dataSources points to the actual dataSource", function () {
100   var dataSource = Sample.dataSource;
101   equals(dataSource.dataSources[0], dataSource.foo, 'should equal data source foo');
102   equals(dataSource.dataSources[1], dataSource.bar, 'should equal data source bar');
103   equals(dataSource.dataSources[2], dataSource.baz, 'should equal data source baz');
104 });
105 
106 test("Verify dataSources added using 'from' are appended in order", function () {
107   Sample.dataSource = SC.CascadeDataSource.create()
108        .from(Sample.fooDataSource)
109        .from(Sample.barDataSource)
110        .from(Sample.bazDataSource);
111 
112   var dataSource = Sample.dataSource;
113   equals(dataSource.dataSources[0], Sample.fooDataSource, 'should equal data source foo');
114   equals(dataSource.dataSources[1], Sample.barDataSource, 'should equal data source bar');
115   equals(dataSource.dataSources[2], Sample.bazDataSource, 'should equal data source baz');
116 });
117 
118 test("Verify dataSource returns 'YES' when handled by a child dataSource for commitRecords", function () {
119   ok(Sample.dataSource.commitRecords(store, [0], [1], [2], { hello: "world" }),
120      "commitRecords should be handled by foo");
121 });
122 
123 test("Verify dataSource returns 'YES' when handled by a child dataSource for fetch", function () {
124   ok(Sample.dataSource.fetch(store, 'query'), "fetch should be handled by bar");
125 });
126 
127 test("Verify dataSource returns 'YES' when handled by a child dataSource for retriveRecords", function () {
128   ok(Sample.dataSource.retrieveRecords(store, [0], ['id']), "retrieveRecords should be handled by baz");
129 });
130 
131 test("Verify dataSource returns 'NO' when not handled by a child dataSource", function () {
132   ok(!Sample.dataSource.cancel(store, [1]), "cancel should be handled by no data source");
133 });
