 27 /**
 28   @namespace
 29 
 30   A RenderContext is a builder that can be used to generate HTML for views or
 31   to update an existing element.  Rather than making changes to an element
 32   directly, you use a RenderContext to queue up changes to the element,
 33   finally applying those changes or rendering the new element when you are
 34   finished.
 35 
 36   You will not usually create a render context yourself but you will be passed
 37   a render context as the first parameter of your render() method on custom
 38   views.
 39 
 40   Render contexts are essentially arrays of strings.  You can add a string to
 41   the context by calling push().  You can retrieve the entire array as a
 42   single string using join().  This is basically the way the context is used
 43   for views.  You are passed a render context and expected to add strings of
 44   HTML to the context like a normal array.  Later, the context will be joined
 45   into a single string and converted into real HTML for display on screen.
 46 
 47   In addition to the core push and join methods, the render context also
 48   supports some extra methods that make it easy to build tags.
 49 
 50   context.begin() <-- begins a new tag context
 51   context.end() <-- ends the tag context...
 52 */
 53 SC.RenderContext = SC.Builder.create(

 56   SELF_CLOSING: SC.CoreSet.create().addEach(['area', 'base', 'basefront', 'br', 'hr', 'input', 'img', 'link', 'meta']),

 58   /**
 59     When you create a context you should pass either a tag name or an element
 60     that should be used as the basis for building the context.  If you pass
 61     an element, then the element will be inspected for class names, styles
 62     and other attributes.  You can also call update() or replace() to
 63     modify the element with you context contents.
 64 
 65     If you do not pass any parameters, then we assume the tag name is 'div'.
 66 
 67     A second parameter, parentContext, is used internally for chaining.  You
 68     should never pass a second argument.
 69 
 70     @param {String|DOMElement} tagNameOrElement
 71     @returns {SC.RenderContext} receiver
 72   */
 73   init: function(tagNameOrElement, prevContext) {
 74     var strings, tagNameOrElementIsString;
 75 
 76     // if a prevContext was passed, setup with that first...
 77     if (prevContext) {
 78       this.prevObject = prevContext ;
 79       this.strings    = prevContext.strings ;
 80       this.offset     = prevContext.length + prevContext.offset ;
 81     }
 82 
 83     if (!this.strings) this.strings = [] ;
 84 
 85     // if tagName is string, just setup for rendering new tagName
 86     if (tagNameOrElement === undefined) {
 87       tagNameOrElement = 'div' ;
 88       tagNameOrElementIsString = YES ;
 89     }
 90     else if (tagNameOrElement === 'div'  ||  tagNameOrElement === 'label'  ||  tagNameOrElement === 'a') {
 91       // Fast path for common tags.
 92       tagNameOrElementIsString = YES ;
 93     }
 94     else if (SC.typeOf(tagNameOrElement) === SC.T_STRING) {
 95       tagNameOrElement = tagNameOrElement.toLowerCase() ;
 96       tagNameOrElementIsString = YES ;
 97     }
 98 
 99     if (tagNameOrElementIsString) {
100       this._tagName     = tagNameOrElement ;
101       this._needsTag    = YES ; // used to determine if end() needs to wrap tag
102       this.needsContent = YES ;
103 
104       // increase length of all contexts to leave space for opening tag
105       var c = this;
106       while(c) { c.length++; c = c.prevObject; }
107 
108       this.strings.push(null);
109       this._selfClosing = this.SELF_CLOSING.contains(tagNameOrElement);
110     }
111     else {
112       this._elem        = tagNameOrElement ;
113       this._needsTag    = NO ;
114       this.length       = 0 ;
115       this.needsContent = NO ;
116     }
117     return this ;
118   },

173   /**
174     Returns the string at the designated index.  If you do not pass anything
175     returns the string array.  This index is an offset from the start of the
176     strings owned by this context.
177 
178     @param {Number} idx the index
179     @returns {String|Array}

181   get: function(idx) {
182     var strings = this.strings || [];
183     return (idx === undefined) ? strings.slice(this.offset, this.length) : strings[idx+this.offset];

186   /**
187     Adds a string to the render context for later joining.  Note that you can
188     pass multiple arguments to this method and each item will be pushed.
189 
190     @param {String} line the liene to add to the string.
191     @returns {SC.RenderContext} receiver
192   */
193   push: function(line) {
194     var strings = this.strings, len = arguments.length;
195     if (!strings) this.strings = strings = []; // create array lazily
196 
197     if (len > 1) {
198       strings.push.apply(strings, arguments) ;
199     } else strings.push(line);
200 
201     // adjust string length for context and all parents...
202     var c = this;
203     while(c) { c.length += len; c = c.prevObject; }
204 
205     this.needsContent = YES;
206 
207     return this;
208   },

210   /**
211     Pushes the passed string onto the array, but first escapes the string
212     to ensure that no user-entered HTML is processed as HTML.
213 
214     @param {String} line one or mroe lines of text to add
215     @returns {SC.RenderContext} receiver
216   */
217   text: function(line) {
218     var len = arguments.length, idx=0;
219     for(idx=0;idx<len;idx++) {
220       this.push(SC.RenderContext.escapeHTML(arguments[idx]));
221     }
222     return this ;
223   },

236     var strings = this.strings;
237     return strings ? strings.join(joinChar || '') : '' ;

283   /**
284     If an element was set on this context when it was created, this method
285     will actually apply any changes to the element itself.  If you have not
286     written any inner html into the context, then the innerHTML of the
287     element will not be changed, otherwise it will be replaced with the new
288     innerHTML.
289 
290     Also, any attributes, id, classNames or styles you've set will be
291     updated as well.  This also ends the editing context session and cleans
292     up.
293 
294     @returns {SC.RenderContext} previous context or null if top
295   */
296   update: function() {
297     var elem = this._elem,
298         mode = this.updateMode,
299         cq, value, factory, cur, next;
300 
301     this._innerHTMLReplaced = NO;
302 
303     if (!elem) {
304       // throw "Cannot update context because there is no source element";
305       return ;
306     }
307 
308     cq = this.$();
309 
310     // console.log('%@#update() called'.fmt(this));
311     // if (this.length>0) console.log(this.join());
312     // else console.log('<no length>');
313 
314     // replace innerHTML
315     if (this.length>0) {
316       this._innerHTMLReplaced = YES;
317       if (mode === SC.MODE_REPLACE) {
318         cq.html(this.join());
319       } else {
320         factory = elem.cloneNode(false);
321         factory.innerHTML = this.join() ;
322         cur = factory.firstChild ;
323         while(cur) {
324           next = cur.nextSibling ;
325           elem.insertBefore(cur, next);
326           cur = next ;
327         }
328         cur = next = factory = null ; // cleanup
329       }
330     }
331 
332     // attributes, styles, and class naems will already have been set.
333 
334     // id="foo"
335     if (this._idDidChange && (value = this._id)) {
336       cq.attr('id', value);
337     }
338 
339     // flush jQuery buffers
340     jQuery.Buffer.flush();
341 
342 
343     // now cleanup element...
344     elem = this._elem = null ;
345     return this.prevObject || this ;
346   },

348   // these are temporary objects are reused by end() to avoid memory allocs.
349   _DEFAULT_ATTRS: {},
350   _TAG_ARRAY: [],
351   _JOIN_ARRAY: [],
352   _STYLE_PAIR_ARRAY: [],
353 
354   /**
355     Ends the current tag editing context.  This will generate the tag string
356     including any attributes you might have set along with a closing tag.
357 
358     The generated HTML will be added to the render context strings.  This will
359     also return the previous context if there is one or the receiver.
360 
361     If you do not have a current tag, this does nothing.
362 
363     @returns {SC.RenderContext}
364   */
365   end: function() {
366     // console.log('%@.end() called'.fmt(this));
367     // NOTE: If you modify this method, be careful to consider memory usage
368     // and performance here.  This method is called frequently during renders
369     // and we want it to be as fast as possible.
370 
371     // generate opening tag.
372 
373     // get attributes first.  Copy in className + styles...
374     var tag = this._TAG_ARRAY, pair, joined, key , value,
375         attrs = this._attrs, className = this._classNames,
376         id = this._id, styles = this._styles;
377 
378     // add tag to tag array
379     tag[0] = '<';  tag[1] = this._tagName ;
380 
381     // add any attributes...
382     if (attrs || className || styles || id) {
383       if (!attrs) attrs = this._DEFAULT_ATTRS ;
384       if (id) attrs.id = id ;
385       if (className) attrs['class'] = className.join(' ');
386 
387       // add in styles.  note how we avoid memory allocs here to keep things
388       // fast...
389       if (styles) {
390         joined = this._JOIN_ARRAY ;
391         pair = this._STYLE_PAIR_ARRAY;
392         for(key in styles) {
393           if(!styles.hasOwnProperty(key)) continue ;
394           value = styles[key];
395           if (value === null) continue; // skip empty styles
396           if (typeof value === SC.T_NUMBER && !SC.NON_PIXEL_PROPERTIES.contains(key)) value += "px";
397 
398           pair[0] = this._dasherizeStyleName(key);
399           pair[1] = value;
400           joined.push(pair.join(': '));
401         }
402         attrs.style = joined.join('; ') ;
403 
404         // reset temporary object.  pair does not need to be reset since it
405         // is always overwritten
406         joined.length = 0;
407       }
408 
409       // now convert attrs hash to tag array...
410       tag.push(' '); // add space for joining0
411       for(key in attrs) {
412         if (!attrs.hasOwnProperty(key)) continue ;
413         value = attrs[key];
414         if (value === null) continue ; // skip empty attrs
415         tag.push(key, '="', value, '" ');
416       }
417 
418       // if we are using the DEFAULT_ATTRS temporary object, make sure we
419       // reset.
420       if (attrs === this._DEFAULT_ATTRS) {
421         delete attrs.style;  delete attrs['class']; delete attrs.id;
422       }
423 
424     }
425 
426     // this is self closing if there is no content in between and selfClosing
427     // is not set to false.
428     var strings = this.strings;
429     var selfClosing = (this._selfClosing === NO) ? NO : (this.length === 1) ;
430     tag.push(selfClosing ? ' />' : '>') ;
431 
432     // console.log('selfClosing == %@'.fmt(selfClosing));
433     strings[this.offset] = tag.join('');
434     tag.length = 0 ; // reset temporary object
435 
436     // now generate closing tag if needed...
437     if (!selfClosing) {
438       tag[0] = '</' ;
439       tag[1] = this._tagName;
440       tag[2] = '>';
441       strings.push(tag.join(''));
442 
443       // increase length of receiver and all parents
444       var c = this;
445       while(c) { c.length++; c = c.prevObject; }
446       tag.length = 0; // reset temporary object again
447     }
448 
449     // if there was a source element, cleanup to avoid memory leaks
450     this._elem = null;
451     return this.prevObject || this ;
452   },

507   /**
508     Reads the current classNames array or sets the array if a param is passed.
509     Note that if you get the classNames array and then modify it, you MUST
510     call this method again to set the array or else it may not be copied to
511     the element.
512 
513     If you do pass a classNames array, you can also pass YES for the
514     cloneOnModify param.  This will cause the context to clone the class names
515     before making any further edits.  This is useful is you have a shared
516     array of class names you want to start with but edits should not change
517     the shared array.
518 
519     @param {Array} classNames array
520     @param {Boolean} cloneOnModify
521     @returns {Array|SC.RenderContext} classNames array or receiver
522   */
523   classNames: function(classNames, cloneOnModify) {
524     if (this._elem) {
525       if (classNames) {
526         this.$().resetClassNames().addClass(classNames);
527         return this;
528       } else {
529         return this.$().attr('class').split(' ');
530       }
531     }
532 
533     if (classNames === undefined) {
534       if (this._cloneClassNames) {
535         this._classNames = (this._classNames || []).slice();
536         this._cloneClassNames = NO ;
537       }
538 
539       // if there are no class names, create an empty array but don't modify.
540       if (!this._classNames) this._classNames = [];
541 
542       return this._classNames ;
543     } else {
544       this._classNames = classNames ;
545       this._cloneClassNames = cloneOnModify || NO ;
546       this._classNamesDidChange = YES ;
547       return this ;
548     }
549   },

565   /**
566     Adds the specified className to the current tag, if it does not already
567     exist.  This method has no effect if there is no open tag.
568 
569     If there is an element backing this RenderContext, buffered jQuery is
570     used to perform the update.
571 
572     @param {String|Array} nameOrClasses the class name or an array of classes
573     @returns {SC.RenderContext} receiver
574   */
575   addClass: function(nameOrClasses) {
576     if(nameOrClasses === undefined || nameOrClasses === null) {
577       SC.Logger.warn('You are adding an undefined or empty class'+ this.toString());
578       return this;
579     }
580 
581     if (this._elem) {
582       if (SC.typeOf(nameOrClasses) === SC.T_STRING) {
583         this.$().addClass(nameOrClasses);
584       } else {
585         var idx, len = nameOrClasses.length;
586         for (idx = 0; idx < len; idx++) this.$().addClass(nameOrClasses[idx]);
587       }
588       return this;
589     }
590 
591     var classNames = this.classNames() ; // handles cloning ,etc.
592     if(SC.typeOf(nameOrClasses) === SC.T_STRING){
593       if (classNames.indexOf(nameOrClasses)<0) {
594         classNames.push(nameOrClasses);
595         this._classNamesDidChange = YES ;
596       }
597     } else {
598       var cl;
599       for(var i = 0, iLen= nameOrClasses.length; i<iLen; i++){
600         cl = nameOrClasses[i];
601         if (classNames.indexOf(cl)<0) {
602           classNames.push(cl);
603           this._classNamesDidChange = YES ;
604         }
605       }
606     }
607 
608     return this;
609   },

611   /**
612     Removes the specified className from the current tag.  This method has
613     no effect if there is not an open tag.
614 
615     If there is an actual DOM element backing this render context,
616     the modification will be written immediately to a buffered jQuery instance.
617 
618     @param {String} className the class to add
619     @returns {SC.RenderContext} receiver
620   */
621   removeClass: function(className) {
622     if (this._elem) {
623       this.$().removeClass(className);
624       return this;
625     }
626 
627     var classNames = this._classNames, idx;
628     if (classNames && (idx=classNames.indexOf(className))>=0) {
629       if (this._cloneClassNames) {
630         classNames = this._classNames = classNames.slice();
631         this._cloneClassNames = NO ;
632       }
633 
634       // if className is found, just null it out.  This will end up adding an
635       // extra space to the generated HTML but it is faster than trying to
636       // recompact the array.
637       classNames[idx] = null;
638       this._classNamesDidChange = YES ;
639     }
640 
641     return this;
642   },

644   /**
645     Removes all classnames from the context. If the context represents an
646     element, this will be handled in CoreQuery.

648     @returns {SC.RenderContext} receiver
649   */
650   resetClassNames: function() {
651     if (this._elem) {
652       this.$().resetClassNames();
653       return this;
654     }

657     this._classNamesDidChange = YES ;
658     return this;

661   /**
662     You can either pass a single class name and a boolean indicating whether
663     the value should be added or removed, or you can pass a hash with all
664     the class names you want to add or remove with a boolean indicating
665     whether they should be there or not.
666 
667     This is far more efficient than using addClass/removeClass.
668 
669     If this context represents an element, this uses the buffered jQuery to
670     ensure all planned DOM operations stay in-sync.
671 
672     @param {String|Hash} className class name or hash of classNames + bools
673     @param {Boolean} shouldAdd for class name if a string was passed
674     @returns {SC.RenderContext} receiver
675   */
676   setClass: function(className, shouldAdd) {
677     if (this._elem) {
678       this.$().setClass(className, shouldAdd);
679       return this;
680     }
681 
682     var classNames, idx, key, didChange;
683 
684     // simple form
685     if (shouldAdd !== undefined) {
686       return shouldAdd ? this.addClass(className) : this.removeClass(className);
687     // bulk form
688     } else {
689       classNames = this._classNames ;
690       if (!classNames) classNames = this._classNames = [];
691 
692       if (this._cloneClassNames) {
693         classNames = this._classNames = classNames.slice();
694         this._cloneClassNames = NO ;
695       }
696 
697       didChange = NO;
698       for(key in className) {
699         if (!className.hasOwnProperty(key)) continue ;
700         idx = classNames.indexOf(key);
701         if (className[key]) {
702           if (idx<0) { classNames.push(key); didChange = YES; }
703         } else {
704           if (idx>=0) { classNames[idx] = null; didChange = YES; }
705         }
706       }
707       if (didChange) this._classNamesDidChange = YES;
708     }
709 
710     return this ;
711   },

718 
719   /**
720     Retrieves or sets the current styles for the outer tag.  If you retrieve
721     the styles hash to edit it, you must set the hash again in order for it
722     to be applied to the element on rendering.
723 
724     Optionally you can also pass YES to the cloneOnModify param to cause the
725     styles has to be cloned before it is edited.  This is useful if you want
726     to start with a shared style hash and then optionally modify it for each
727     context.
728 
729     @param {Hash} styles styles hash
730     @param {Boolean} cloneOnModify
731     @returns {Hash|SC.RenderContext} styles hash or receiver
732   */
733   styles: function(styles, cloneOnModify) {
734     if (this._elem) {
735       if (styles) {
736         this.$().resetStyles().css(styles);
737       }
738       return this.$().styles();
739     }
740 
741     var attr, regex, match;
742     if (styles === undefined) {
743 
744       // no styles are defined yet but we do have a source element.  Lazily
745       // extract styles from element.
746       if (!this._styles && this._elem) {
747         // parse style...
748         attr = this.$().attr('style');

754           styles = {};

757           regex.lastIndex = 0;

764         } else {
765           this._styles = {};
766         }

769       // before returning.
770       } else {
771         if (!this._styles) {
772           this._styles = {};
773         } else {
774           if (this._cloneStyles) {
775             this._styles = SC.clone(this._styles);
776             this._cloneStyles = NO ;
777           }
778         }
779       }
780 
781       return this._styles ;
782 
783     // set the styles if passed.
784     } else {
785       this._styles = styles ;
786       this._cloneStyles = cloneOnModify || NO ;
787       this._stylesDidChange = YES ;
788       return this ;
789     }

792   _deleteComboStyles: function(styles, key) {
793     var comboStyles = SC.COMBO_STYLES[key],
794         didChange = NO;

819   /**
820     Apply the passed styles to the tag.  You can pass either a single key
821     value pair or a hash of styles.  Note that if you set a style on an
822     existing element, it will replace any existing styles on the element.
823 
824     @param {String|Hash} nameOrStyles the style name or a hash of styles
825     @param {String|Number} value style value if string name was passed
826     @returns {SC.RenderContext} receiver
827   */
828   addStyle: function(nameOrStyles, value) {
829     if (this._elem) {
830       this.$().css(nameOrStyles, value);
831       return this;
832     }
833 
834     // get the current hash of styles.  This will extract the styles and
835     // clone them if needed.  This will get the actual styles hash so we can
836     // edit it directly.
837     var key, didChange = NO, styles = this.styles();
838 
839     // simple form
840     if (typeof nameOrStyles === SC.T_STRING) {
841       if (value === undefined) { // reader
842         return styles[nameOrStyles];
843       } else { // writer
844         didChange = this._deleteComboStyles(styles, nameOrStyles);
845         if (styles[nameOrStyles] !== value) {
846           styles[nameOrStyles] = value ;
847           didChange = YES ;
848         }
849         if (didChange) this._stylesDidChange = YES;
850       }
851 
852     // bulk form
853     } else {
854       for(key in nameOrStyles) {
855         if (!nameOrStyles.hasOwnProperty(key)) continue ;
856         didChange = didChange || this._deleteComboStyles(styles, key);
857         value = nameOrStyles[key];
858         if (styles[key] !== value) {
859           styles[key] = value;
860           didChange = YES;
861         }
862       }
863       if (didChange) this._stylesDidChange = YES ;
864     }
865 
866     return this ;
867   },

889     if (styles[styleName]) {
890       styles[styleName] = null;
891       this._stylesDidChange = YES ;
892     }
