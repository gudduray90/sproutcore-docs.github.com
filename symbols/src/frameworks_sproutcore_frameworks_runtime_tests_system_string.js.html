 10 // Note that these unit tests are calling SC.String.fmt directly, which has a different
 11 // signature than String.prototype.fmt does.
 12 
 13 module("String Formatting");
 14 test("Passing ordered arguments", function() {
 15   equals(SC.String.fmt("%@, %@%@", ["Hello", "World", "!"]), "Hello, World!");
 16 });
 17 
 18 test("Passing indexed arguments", function() {
 19   equals(SC.String.fmt("%@2, %@3%@1", ["!", "Hello", "World"]), "Hello, World!");
 20 });
 21 
 22 test("Passing named arguments", function() {
 23   // NOTE: usually, "str".fmt() would be called. Because we are calling String.fmt,
 24   // which takes an array of arguments, we have to pass the arguments as an array.
 25   equals(SC.String.fmt("%{first}, %{last}%{punctuation}", [
 26     { first: "Hello", last: "World", "punctuation": "!" }
 27   ]), "Hello, World!");
 28 });
 29 
 30 test("Passing named arguments with a SC.Object instance", function() {
 31   var t = SC.Object.create({
 32     prop: 'Hello',
 33     computedProp: function () {
 34       return 'World';
 35     }.property().cacheable(),
 36     unknownProperty: function (key, value) {
 37       if (key === "unknownProp") return "!";
 38     }
 39   });
 40   equals(SC.String.fmt("%{prop}, %{computedProp}%{unknownProp}", [t]), "Hello, World!");
 41 });
 42 
 43 test("Passing incomplete named arguments", function() {
 44   equals( SC.String.fmt("%{first}, %{last}%{punctuation}", [{first: 'Hello', punctuation: '!'}]), "Hello, %{last}!", "Formatting a string with an incomplete set of named arguments should leave unspecified named arguments in place." );
 45 })
 46 
 47 test("Passing arguments with formatters", function() {
 48   var F = function(value) {
 49     return "$" + value;
 50   };
 51 
 52   equals(SC.String.fmt("%{number}", [{ number: 12, numberFormatter: F }]), "$12", "Formatter was applied");
 53 });
 54 
 55 test("Passing formatting strings with formatters", function() {
 56   var F = function(value, arg) {
 57     return "$" + value + ";" + arg;
 58   };
 59 
 60   equals(SC.String.fmt("%{number:blah}", [{ number: 12, numberFormatter: F }]), "$12;blah", "Formatter was applied with argument");
 61 });
