 10 /**
 11   @class
 12   Use this mixin to make your view automatically resize based upon its value,
 13   title, or other string property. Only works for views that support automatic
 14   resizing.
 15 
 16   Supporting Automatic Resizing
 17   -------------------------------------
 18   To support automatic resizing, your view must provide these properties:
 19 
 20   - *`supportsAutoResize`.* Must be set to YES.
 21 
 22   - *`autoResizeLayer`* A DOM element to use as a template for resizing the
 23     view. Font sizes and other styles will be copied to the measuring element
 24     SproutCore uses to measure the text.
 25 
 26   - *`autoResizeText`.* The text to measure. A button view might make a proxy
 27     to its `displayTitle`, for instance.
 28 
 29   Your view may also supply:
 30 
 31   - *`autoResizePadding`.* An amount to add to the measured size. This may be either
 32     a single number to be added to both width and height, or a hash containing
 33     separate `width` and `height` properties.
 34 
 35 
 36   NOTE: these properties are not defined in the mixin itself because the supporting view,
 37   rather than the user of SC.AutoResize, will be providing the properties, and mixing
 38   SC.AutoResize into the view should not override these properties.
 39 */
 40 SC.AutoResize = {
 41   /*@scope SC.AutoResize.prototype */
 42 
 43   /**
 44     If YES, automatically resizes the view (default). If NO, only measures,
 45     setting 'measuredSize' to the measured value (you can bind to measuredSize
 46     and update size manually).
 47 
 48     @type Boolean
 49     @default YES
 50   */
 51   shouldAutoResize: YES,
 52 
 53   /**
 54     If NO, prevents SC.AutoResize from doing anything at all.
 55 
 56     @type Boolean
 57     @default YES
 58   */
 59   shouldMeasureSize: YES,
 60 
 61   /**
 62     Caches sizes for measured strings. This cache does not have a max size, so
 63     should only be used when a view has a limited number of possible values.
 64     Multiple views that have the same batchResizeId will share the same cache.
 65 
 66     @type Boolean
 67     @default NO
 68   */
 69   shouldCacheSizes: NO,
 70 
 71   /**
 72     Determines if the view's width should be resized
 73     on calculation.
 74 
 75     @type Boolean
 76     @default YES
 77   */
 78   shouldResizeWidth: YES,
 79 
 80   /**
 81     Determines if the view's height should be resized
 82     on calculation. Default is NO to retain backwards
 83     compatibility.
 84 
 85     @type Boolean
 86     @default NO
 87   */
 88   shouldResizeHeight: NO,
 89 
 90   /**
 91     The measured size of the view's content (the value of the autoResizeField).
 92     This property is observable, and, if used in conjunction with setting
 93     shouldAutoResize to NO, allows you to customize the 'sizing' part, using
 94     SC.AutoResize purely for its measuring code.
 95 
 96     @type Rect
 97   */
 98   measuredSize: { width: 0, height: 0 },
 99 
100   /**
101     If provided, will limit the maximum width to this value.
102   */
103   maxWidth: null,
104 
105   /**
106     If provided, will limit the maximum height to this value.
107   */
108   maxHeight: null,
109 
110   /**
111     If YES, the view's text will be resized to fit the view. This is applied _after_ any
112     resizing, so will only take affect if shouldAutoResize is off, or a maximum width/height
113     is set.
114 
115     You also must set a minimum and maximum font size. Any auto resizing will happen at the
116     maximum size, and then the text will be resized as necessary.
117   */
118   shouldAutoFitText: NO,
119 
120   /**
121     If NO, the calculated font size may be any size between minFontSize and
122     maxFontSize. If YES, it will only be either minFontSize or maxFontSize.
123 
124     @type Boolean
125     @default NO
126   */
127   autoFitDiscreteFontSizes: NO,
128 
129   /**
130     The minimum font size to use when automatically fitting text. If shouldAutoFitText is set,
131     this _must_ be supplied.
132 
133     Font size is in pixels.
134   */
135   minFontSize: 12,
136 
137   /**
138     The maximum font size to use when automatically fitting text. If shouldAutoFitText is set,
139     this _must_ be supplied.
140 
141     Font size is in pixels.
142   */
143   maxFontSize: 20,
144 
145   /**
146     If shouldAutoFitText is YES, this is the calculated font size.
147   */
148   calculatedFontSize: 20,
149 
150   fontPropertyDidChange: function() {
151     if(this.get('shouldAutoFitText')) this.invokeLast(this.fitTextToFrame);
152   }.observes('shouldAutoFitText', 'minFontSize', 'maxFontSize', 'measuredSize'),
153 
154   /**
155     Observes the measured size and actually performs the resize if necessary.
156   */
157   measuredSizeDidChange: function() {
158     var measuredSize = this.get('measuredSize'),
159     calculatedWidth = measuredSize.width, calculatedHeight = measuredSize.height,
160     paddingHeight, paddingWidth,
161     autoResizePadding = this.get('autoResizePadding') || 0,
162     maxWidth = this.get('maxWidth'), maxHeight = this.get('maxHeight');
163 
164     if(SC.typeOf(autoResizePadding) === SC.T_NUMBER) {
165       paddingHeight = paddingWidth = autoResizePadding;
166     } else {
167       paddingHeight = autoResizePadding.height;
168       paddingWidth = autoResizePadding.width;
169     }
170 
171     calculatedHeight += paddingHeight;
172     calculatedWidth += paddingWidth;
173 
174     if(this.get('shouldAutoResize')) {
175       // if we are allowed to autoresize, adjust the layout
176       if (this.get('shouldResizeWidth')) {
177         if (maxWidth && calculatedWidth > maxWidth) {
178           calculatedWidth = maxWidth;
179         }
180         this.set('calculatedWidth', calculatedWidth);
181 
182         this.adjust('width', calculatedWidth);
183       }
184 
185       if (this.get('shouldResizeHeight')) {
186         if (maxHeight && calculatedHeight > maxHeight) {
187           calculatedHeight = maxHeight;
188         }
189         this.set('calculatedHeight', calculatedHeight);
190         this.adjust('height', calculatedHeight);
191       }
192     }
193 
194   }.observes('shouldAutoResize', 'measuredSize', 'autoResizePadding', 'maxWidth', 'maxHeight', 'shouldResizeWidth', 'shouldResizeHeight'),
195 
196   /**
197     @private
198     Begins observing the auto resize field.
199   */
200   initMixin: function() {
201     // @if (debug)
202     if (!this.get('supportsAutoResize')) {
203       throw new Error("View `%@` does not support automatic resize. See documentation for SC.AutoResize".fmt(this));
204     }
205     // @endif
206   },
207 
208   /**
209     If this property is provided, all views that share the same value for this property will be resized as a batch for increased performance.
210 
211     @type String
212   */
213   batchResizeId: null,
214 
215   /**
216     Schedules a measurement to happen later.
217   */
218   scheduleMeasurement: function() {
219     var batchResizeId = this.get('batchResizeId');
220 
221     // only measure if we are visible, active, and the text or style actually changed
222     if (!this.get('shouldMeasureSize') || !this.get('isVisibleInWindow') || (this.get('autoResizeText') === this._lastMeasuredText && batchResizeId === this._lastMeasuredId)) return;
223 
224     // batchResizeId is allowed to be undefined; views without an id will just
225     // get measured one at a time
226     SC.AutoResizeManager.scheduleMeasurementForView(this, batchResizeId);
227   }.observes('isVisibleInWindow', 'shouldMeasureSize', 'autoResizeText', 'batchResizeId'),
228 
229   _lastMeasuredText: null,
230 
231   _cachedMetrics: function(key, value) {
232     if(!this.get('shouldCacheSizes')) return;
233 
234     // if we don't have a tag, then it is unique per view
235     // you shouldn't usually turn on caching without a tag, but it is supported
236     var cacheSlot = SC.cacheSlotFor(this.get('batchResizeId') || this),
237     autoResizeText = this.get('autoResizeText');
238 
239     if(value) cacheSlot[autoResizeText] = value;
240     else value = cacheSlot[autoResizeText];
241 
242     return value;
243   }.property('shouldCacheSizes', 'autoResizeText', 'batchResizeId').cacheable(),
244 
245   /**
246     Measures the size of the view.
247 
248     @param batch For internal use during batch resizing.
249   */
250   measureSize: function(batch) {
251     var metrics, layer = this.get('autoResizeLayer'), autoResizeText = this.get('autoResizeText'),
252         ignoreEscape = !this.get('escapeHTML'),
253         batchResizeId = this.get('batchResizeId'),
254         cachedMetrics = this.get('_cachedMetrics'),
255         maxFontSize = this.get('maxFontSize');
256 
257     if(!layer) return;
258 
259     // There are three special cases.
260     //   - size is cached: the cached size is used with no measurement
261     //     necessary
262     //   - empty: we should do nothing. The metrics are 0.
263     //   - batch mode: just call measureString.
264     //
265     // If we are in neither of those special cases, we should go ahead and
266     // resize normally.
267     //
268     if(cachedMetrics) {
269       metrics = cachedMetrics;
270     }
271 
272     else if (SC.none(autoResizeText) || autoResizeText === "") {
273       metrics = { width: 0, height: 0 };
274     }
275 
276     else if (batch) {
277       metrics = SC.measureString(autoResizeText, ignoreEscape);
278     }
279 
280     else {
281       this.prepareLayerForStringMeasurement(layer);
282 
283       metrics = SC.metricsForString(autoResizeText, layer, this.get('classNames'), ignoreEscape);
284     }
285 
286     // In any case, we set measuredSize.
287     this.set('measuredSize', metrics);
288 
289     // and update the cache if we are using it
290     if(this.get('shouldCacheSizes')) this.setIfChanged('_cachedMetrics', metrics);
291 
292     // set the measured value so we can avoid extra measurements in the future
293     this._lastMeasuredText = autoResizeText;
294     this._lastMeasuredId = batchResizeId;
295 
296     return metrics;
297   },

304   /**
305     If we are fitting text, the layer must be measured with its font size set to our
306     maximum font size.
307   */
308   prepareLayerForStringMeasurement: function(layer) {
309     var maxFontSize = this.get('maxFontSize');
310 
311     if (this.get('shouldAutoFitText') && this.get('calculatedFontSize') !== maxFontSize) {
312       layer.style.fontSize = maxFontSize + "px";
313     }
314   },
315 
316   /**
317     Whenever the view resizes, the text fitting must be reevaluated.
318   */
319   viewDidResize: function(orig) {
320     orig();
321 
322     this.fontPropertyDidChange();
323   }.enhance(),
324 
325   /**
326     Fits the text into the frame's size, minus autoResizePadding.
327   */
328   fitTextToFrame: function() {
329     // we can only fit text when we have a layer.
330     var layer = this.get('autoResizeLayer');
331     if (!layer) return;
332 
333     var maxFontSize = this.get('maxFontSize'),
334         minFontSize = this.get('minFontSize');
335 
336     // if the font size has been adjusted, reset it to the max
337     this.prepareLayerForStringMeasurement(layer);
338 
339     var frame = this.get('frame'),
340 
341         padding = this.get('autoResizePadding') || 0,
342 
343         // these need to be shrunk by 1 pixel or text that is exactly as wide as
344         // the frame will be truncated
345         width = frame.width - 1, height = frame.height - 1,
346         measured = this.get('measuredSize'),
347         mWidth = measured.width, mHeight = measured.height;
348 
349     // figure out and apply padding to the width/height
350     if(SC.typeOf(padding) === SC.T_NUMBER) {
351       width -= padding;
352       height -= padding;
353     } else {
354       width -= padding.width;
355       height -= padding.height;
356     }
357 
358     // measured size is at maximum. If there is no resizing to be done, short-circuit.
359     if (mWidth <= width && mHeight <= height) return;
360 
361     // if only discrete values are allowed, we can short circuit here and just
362     // use the minimum
363     if(this.get('autoFitDiscreteFontSizes')) {
364       actual = minFontSize;
365     }
366 
367     // otherwise we have to find the actual best font size
368     else {
369       // now, we are going to make an estimate font size. We will figure out the proportion
370       // of both actual width and actual height to the measured width and height, and then we'll
371       // pick the smaller. We'll multiply that by the maximum font size to figure out
372       // a rough guestimate of the proper font size.
373       var xProportion = width / mWidth, yProportion = height / mHeight,
374 
375           guestimate = Math.floor(maxFontSize * Math.min(xProportion, yProportion)),
376           actual,
377 
378           classNames = this.get('classNames'),
379           ignoreEscape = !this.get('escapeHTML'),
380           value = this.get('autoResizeText'),
381 
382           metrics;
383 
384 
385       guestimate = actual = Math.min(maxFontSize, Math.max(minFontSize, guestimate));
386 
387       // Now, we must test the guestimate. Based on that, we'll either loop down
388       // or loop up, depending on the measured size.
389       layer.style.fontSize = guestimate + "px";
390       metrics = SC.metricsForString(value, layer, classNames, ignoreEscape);
391 
392       if (metrics.width > width || metrics.height > height) {
393 
394         // if we're larger, we must go down until we are smaller, at which point we are done.
395         for (guestimate = guestimate - 1; guestimate >= minFontSize; guestimate--) {
396           layer.style.fontSize = guestimate + "px";
397           metrics = SC.metricsForString(value, layer, classNames, ignoreEscape);
398 
399           // always have an actual in this case; even if we can't get it small enough, we want
400           // to keep this as close as possible.
401           actual = guestimate;
402 
403           // if the new size is small enough, stop shrinking and set it for real
404           if (metrics.width <= width && metrics.height <= height) {
405             break;
406           }
407         }
408 
409       } else if (metrics.width < width || metrics.height < height) {
410         // if we're smaller, we must go up until we hit maxFontSize or get larger. If we get
411         // larger, we want to use the previous guestimate (which we know was valid)
412         //
413         // So, we'll start actual at guestimate, and only increase it while we're smaller.
414         for (guestimate = guestimate + 1; guestimate <= maxFontSize; guestimate++) {
415           layer.style.fontSize = guestimate + "px";
416           metrics = SC.metricsForString(value, layer, classNames, ignoreEscape);
417 
418           // we update actual only if it is still valid. Then below, whether valid
419           // or not, if we are at or past the width/height we leave
420           if (metrics.width <= width && metrics.height <= height) {
421             actual = guestimate;
422           }
423 
424           // we put this in a separate if statement JUST IN CASE it is ===.
425           // Unlikely, but possible, and why ruin a good thing?
426           if (metrics.width >= width || metrics.height >= height){
427             break;
428           }
429         }
430       }
431     }
432 
433     layer.style.fontSize = actual + "px";
434     this.set('calculatedFontSize', actual);
435   },

437   /**
438     Extends renderSettingsToContext to add font size if shouldAutoFitText is YES.
439   */
440   applyAttributesToContext: function(orig, context) {
441     orig(context);
442 
443     if (this.get('shouldAutoFitText')) {
444       context.setStyle('font-size', this.get('calculatedFontSize') + "px");
445     }
446   }.enhance(),

458   /** @private
459     If the view has a transitionIn property, we have to delay the transition
460     setup and execution until after we measure.  In order to prevent a brief
461     flash of the view, we ensure it is hidden while it is being measured and
462     adjusted.
463 
464     TODO: consider making the measurement state a formal SC.View state
465   */
466   _transitionIn: function (original) {
467     // In order to allow views to measure and adjust themselves on append, we
468     // can't transition until after the measurement is done.
469     var preTransitionOpacity = this.get('layout').opacity || 1;
470 
471     this.adjust('opacity', 0);
472     this.invokeNext(function () {
473       this.adjust('opacity', preTransitionOpacity);
474       original();
475     });
476   }.enhance()

480 /**
481  * @private
482  * @class
483  * Manages batch auto resizing.
484  *
485  * This used to be part of SC.AutoResize, but we shouldn't mix these
486  * methods/properties into each view.
487  */
488 SC.AutoResizeManager = {
489   /**
490     Views queued for batch resizing, but with no batch resize id.
491 
492     @property {SC.CoreSet}
493   */
494   measurementQueue: SC.CoreSet.create(),
495 
496   /**
497     Schedules a re-measurement for the specified view in the batch with the
498     given id.
499 
500     If a batch does not exist by that id, it will be created. If there is no id,
501     the view will be measured individually.
502 
503     @param view The view to measure.
504     @param id The id of the batch to measure the view in.
505   */
506   scheduleMeasurementForView: function(view) {
507     this.measurementQueue.add(view);
508 
509     SC.RunLoop.currentRunLoop.invokeLast(this.doBatchResize);
510   },
511 
512   /**
513     Cancels a scheduled measurement for a view in the named batch id.
514 
515     @param view The view that was scheduled for measurement.
516     @param id The batch id the view was scheduled in.
517   */
518   cancelMeasurementForView: function(view, id) {
519     this.measurementQueue.remove(view);
520   },
521 
522   /**
523     Processes all autoResize batches. This will automatically be invoked at the
524     end of any run loop in which measurements were scheduled.
525   */
526   doBatchResize: function() {
527     // make sure we are called from the correct scope.
528     // this will make our property references below clearer.
529     if (this !== SC.AutoResizeManager) {
530       return SC.AutoResizeManager.doBatchResize();
531     }
532 
533     var tag, view, layer, measurementQueue = this.measurementQueue, prepared, autoResizeText,
534     i, len;
535 
536     while((len = measurementQueue.get('length')) > 0) {
537       prepared = NO;
538       // save the first tag we see
539       tag = measurementQueue[len - 1].get('batchResizeId');
540 
541       // now we iterate over all the views with the same tag
542       for(i = len - 1; i >= 0; --i) {
543         view = measurementQueue[i];
544 
545         // if the view has a different tag, skip it
546         if(view.get('batchResizeId') !== tag) continue;
547 
548         // make sure the view is still qualified to be measured
549         if(view.get('isVisibleInWindow') && view.get('shouldMeasureSize') && (layer = view.get('autoResizeLayer'))) {
550           autoResizeText = view.get('autoResizeText');
551 
552           // if the text is empty or a size is cached don't bother preparing
553           if(!SC.none(autoResizeText) && autoResizeText !== "" && !view.get('_cachedMetrics') && !prepared) {
554             // this is a bit of a hack: before we can prepare string measurement, there are cases where we
555             // need to reset the font size first (specifically, if we are also fitting text)
556             //
557             // It is expected that all views in a batch will have the same font settings.
558             view.prepareLayerForStringMeasurement(layer);
559 
560             // now we can tell SC to prepare the layer with the settings from the view's layer
561             SC.prepareStringMeasurement(layer, view.get('classNames'));
562             prepared = YES;
563           }
564 
565           view.measureSize(YES);
566         }
567 
568         // it's been handled
569         measurementQueue.remove(view);
570 
571         // if the view didn't have a tag, we can't batch so just move on
572         if(!tag) break;
573       }
574 
575       // only call teardown if prepare was called
576       if(prepared) {
577         SC.teardownStringMeasurement();
578       }
579     }
580   }
