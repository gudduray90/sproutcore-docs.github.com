  8 SC.mixin(SC.browser,
  9 /** @scope SC.browser */ {
 10 
 11   /**
 12     Pass any number of arguments, and this will check them against the browser
 13     version split on ".".  If any of them are not equal, return the inequality.
 14     If as many arguments as were passed in are equal, return 0.  If something
 15     is NaN, return 0.
 16   */
 17   compareVersion: function () {
 18     if (this._versionSplit === undefined) {
 19       var coerce = function (part) {
 20         return Number(part.match(/^[0-9]+/));
 21       };
 22       this._versionSplit = SC.A(this.version.split('.')).map(coerce);
 23     }
 24 
 25     var tests = SC.A(arguments).map(Number);
 26     for (var i = 0; i < tests.length; i++) {
 27       var check = this._versionSplit[i] - tests[i];
 28       if (isNaN(check)) return 0;
 29       if (check !== 0) return check;
 30     }
 31     
 32     return 0;
 33   }
 34 
 35 });
