 29 test("parentViewDidResize should only be called when the parent's layout property changes in a manner that may affect child views.", function() {
 30   var callCount = 0 ;
 31   var view = SC.View.create({
 32     // use the callback below to detect when viewDidResize is icalled.
 33     childViews: [SC.View.extend({
 34       parentViewDidResize: function() { callCount++; }
 35     })]
 36   });
 37 
 38   SC.run(function () { view.set('layout', { top: 10, left: 20, height: 50, width: 40 }); });
 39   equals(callCount, 1, 'parentViewDidResize should invoke once');
 40 
 41   SC.run(function () { view.adjust('top', 0); });
 42   equals(callCount, 1, 'parentViewDidResize should invoke once');
 43 
 44   SC.run(function () { view.adjust('height', 60); });
 45   equals(callCount, 2, 'parentViewDidResize should invoke twice');
 46 
 47   // This is tricky, if the height increases, but the same size border is added, the effective height/width is unchanged.
 48   SC.run(function () { view.adjust({'height': 70, 'borderTop': 10 }); });
 49   equals(callCount, 2, 'parentViewDidResize should invoke twice');
 50 });

 52 test("The view's frame should only notify changes when its layout changes if the effective size or position actually change.", function () {
 53   var view2 = SC.View.create({
 54       frameCallCount: 0,
 55       frameDidChange: function() {
 56         this.frameCallCount++;
 57       }.observes('frame'),
 58       viewDidResize: CoreTest.stub('viewDidResize', SC.View.prototype.viewDidResize)
 59     }),
 60     view1 = SC.View.create({
 61       childViews: [view2],
 62       layout: { width: 200, height: 200 }
 63     });
 64 
 65   SC.run(function () { view2.set('layout', { height: 50, width: 50 }); });
 66   equals(view2.get('frameCallCount'), 1, 'frame should have notified changing once.');
 67 
 68   SC.run(function () { view2.adjust('top', 0); });
 69   equals(view2.get('frameCallCount'), 2, 'frame should have notified changing once.');
 70 
 71   SC.run(function () { view2.adjust('height', 100); });
 72   equals(view2.get('frameCallCount'), 3, 'frame should have notified changing twice.');
 73 
 74   // Tricky.
 75   SC.run(function () { view2.adjust({ 'height': 110, 'borderTop': 10, 'top': -10 }); });
 76   equals(view2.get('frameCallCount'), 4, 'frame should have notified changing twice.');
 77 
 78   SC.run(function () { view2.adjust('width', null); });
 79   equals(view2.get('frameCallCount'), 5, 'frame should have notified changing thrice.');
 80 
 81   // Tricky.
 82   SC.run(function () { view2.adjust('width', 200); });
 83   equals(view2.get('frameCallCount'), 6, 'frame should have notified changing thrice.');
 84 });

 86 test("making sure that the frame value is correct inside viewDidResize()", function() {
 87   // We want to test to be sure that when the view's viewDidResize() method is
 88   // called, its frame has been updated.  But rather than run the test inside
 89   // the method itself, we'll cache a global reference to the then-current
 90   // value and test it later.
 91   var cachedFrame;
 92 
 93   var view = SC.View.create({
 94 
 95     layout: { left:0, top:0, width:400, height:400 },
 96 
 97     viewDidResize: function() {
 98         sc_super();
 99 
100         // Set a global reference to my frame at this point so that we can
101         // test for the correct value later.
102         cachedFrame = this.get('frame');
103       }
104   });
105 
106 
107   // Access the frame once before resizing the view, to make sure that the
108   // previous value was cached.  That way, when we ask for the frame again
109   // after the resize, we can verify that the cache invalidation logic is
110   // working correctly.
111   var originalFrame = view.get('frame');
112 
113   SC.RunLoop.begin();
114   view.adjust('height', 314);
115   SC.RunLoop.end();
116 
117   // Now that we've adjusted the view, the cached view (as it was inside its
118   // viewDidResize() method) should be the same value, because the cached
119   // 'frame' value should have been invalidated by that point.
120   same(view.get('frame').height, cachedFrame.height, 'height');
121 });

129 test("When parentViewDidResize is called on a view, it should only notify on frame and cascade the call to child views if it will be affected by the parent's resize.", function() {
130   var view = SC.View.create({
131       // instrument...
132       frameCallCount: 0,
133       frameDidChange: function() {
134         this.frameCallCount++;
135       }.observes('frame'),
136       viewDidResize: CoreTest.stub('viewDidResize', SC.View.prototype.viewDidResize)
137     }),
138     parentView = SC.View.create({
139       childViews: [view],
140       layout: { height: 100, width: 100 }
141     });
142 
143   // try with fixed layout
144   view.set('layout', { top: 10, left: 10, height: 10, width: 10 });
145   view.viewDidResize.reset(); view.frameCallCount = 0;
146   parentView.adjust({ width: 90, height: 90 });
147   view.viewDidResize.expect(0);
148   equals(view.frameCallCount, 0, 'should not notify frame changed when isFixedPosition: %@ and isFixedSize: %@'.fmt(view.get('isFixedPosition'), view.get('isFixedSize')));
149 
150   // try with flexible height
151   view.set('layout', { top: 10, left: 10, bottom: 10, width: 10 });
152   view.viewDidResize.reset(); view.frameCallCount = 0;
153   parentView.adjust({ width: 80, height: 80 });
154   view.viewDidResize.expect(1);
155   equals(view.frameCallCount, 1, 'should notify frame changed when isFixedPosition: %@ and isFixedSize: %@'.fmt(view.get('isFixedPosition'), view.get('isFixedSize')));
156 
157   // try with flexible width
158   view.set('layout', { top: 10, left: 10, height: 10, right: 10 });
159   view.viewDidResize.reset(); view.frameCallCount = 0;
160   parentView.adjust({ width: 70, height: 70 });
161   view.viewDidResize.expect(1);
162   equals(view.frameCallCount, 1, 'should notify frame changed when isFixedPosition: %@ and isFixedSize: %@'.fmt(view.get('isFixedPosition'), view.get('isFixedSize')));
163 
164   // try with right align
165   view.set('layout', { top: 10, right: 10, height: 10, width: 10 });
166   view.viewDidResize.reset(); view.frameCallCount = 0;
167   parentView.adjust({ width: 60, height: 60 });
168   view.viewDidResize.expect(0);
169   equals(view.frameCallCount, 1, 'right align: should notify frame changed when isFixedPosition: %@ and isFixedSize: %@'.fmt(view.get('isFixedPosition'), view.get('isFixedSize')));
170 
171   // try with bottom align
172   view.set('layout', { left: 10, bottom: 10, height: 10, width: 10 });
173   view.viewDidResize.reset(); view.frameCallCount = 0;
174   parentView.adjust({ width: 50, height: 50 });
175   view.viewDidResize.expect(0);
176   equals(view.frameCallCount, 1, 'bottom align: should notify frame changed when isFixedPosition: %@ and isFixedSize: %@'.fmt(view.get('isFixedPosition'), view.get('isFixedSize')));
177 
178   // try with center horizontal align
179   view.set('layout', { centerX: 10, top: 10, height: 10, width: 10 });
180   view.viewDidResize.reset(); view.frameCallCount = 0;
181   parentView.adjust({ width: 40, height: 40 });
182   view.viewDidResize.expect(0);
183   equals(view.frameCallCount, 1, 'centerX: should notify frame changed when isFixedPosition: %@ and isFixedSize: %@'.fmt(view.get('isFixedPosition'), view.get('isFixedSize')));
184 
185   // try with center vertical align
186   view.set('layout', { left: 10, centerY: 10, height: 10, width: 10 });
187   view.viewDidResize.reset(); view.frameCallCount = 0;
188   parentView.adjust({ width: 30, height: 30 });
189   view.viewDidResize.expect(0);
190   equals(view.frameCallCount, 1, 'centerY: should notify frame changed when isFixedPosition: %@ and isFixedSize: %@'.fmt(view.get('isFixedPosition'), view.get('isFixedSize')));
191 });
