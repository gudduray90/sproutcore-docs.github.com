  8 var store, storeKey1, storeKey2, storeKey3, storeKey4, storeKey5, storeKey6, json;
  9 module("SC.Store#pushChanges", {
 10   setup: function () {
 11     SC.RunLoop.begin();
 12     store = SC.Store.create();
 13 
 14     json = {
 15       string: "string",
 16       number: 23,
 17       bool:   YES
 18     };
 19 
 20     storeKey1 = SC.Store.generateStoreKey();
 21     store.writeDataHash(storeKey1, json, SC.Record.EMPTY);
 22 
 23     storeKey2 = SC.Store.generateStoreKey();
 24     store.writeDataHash(storeKey2, json, SC.Record.EMPTY);
 25 
 26     storeKey3 = SC.Store.generateStoreKey();
 27     store.writeDataHash(storeKey3, json, SC.Record.EMPTY);
 28 
 29     storeKey4 = SC.Store.generateStoreKey();
 30     store.writeDataHash(storeKey4, json, SC.Record.BUSY_LOADING);
 31 
 32     storeKey5 = SC.Store.generateStoreKey();
 33     store.writeDataHash(storeKey5, json, SC.Record.BUSY_LOADING);
 34 
 35     storeKey6 = SC.Store.generateStoreKey();
 36     store.writeDataHash(storeKey6, json, SC.Record.BUSY_LOADING);
 37     SC.RunLoop.end();
 38   },
 39 
 40   teardown: function () {
 41     store.destroy();
 42     store = json = null;
 43   }
 44 });
 45 
 46 test("Do a pushRetrieve and check if there is conflicts", function () {
 47   var res = store.pushRetrieve(SC.Record, undefined, undefined, storeKey1);
 48   equals(res, storeKey1, "There is no conflict, pushRetrieve was succesful.");
 49   res = store.pushRetrieve(SC.Record, undefined, undefined, storeKey4);
 50   ok(!res, "There is a conflict, because of the state, this is expected.");
 51 });
 52 
 53 test("Do a pushDestroy and check if there is conflicts", function () {
 54   var res = store.pushDestroy(SC.Record, undefined, storeKey2);
 55   equals(res, storeKey2, "There is no conflict, pushDestroy was succesful.");
 56   res = store.pushRetrieve(SC.Record, undefined, undefined, storeKey5);
 57   ok(!res, "There is a conflict, because of the state, this is expected.");
 58 });
 59 
 60 test("Issue a pushError and check if there is conflicts", function () {
 61   var res = store.pushError(SC.Record, undefined, SC.Record.NOT_FOUND_ERROR, storeKey3);
 62   equals(res, storeKey3, "There is no conflict, pushError was succesful.");
 63   res = store.pushRetrieve(SC.Record, undefined, undefined, storeKey6);
 64   ok(!res, "There is a conflict, because of the state, this is expected.");
 65 });
 66 
 67 test("A pushRetrieve updating the id of an existing record should update the primary Key cache", function () {
 68   var tmpid, recFirst, recSecond, sK;
 69 
 70   tmpid = "@2345235asddsgfd";
 71   recFirst = { firstname: 'me', lastname: 'too', guid: tmpid };
 72   recSecond = { firstname: 'me', lastname: 'too', guid: 1 };
 73   SC.RunLoop.begin();
 74 
 75   sK = store.loadRecord(SC.Record, recFirst, tmpid);
 76   SC.RunLoop.end();
 77   equals(store.idFor(sK), tmpid); //check whether the id is indeed tmpid
 78   SC.RunLoop.begin();
 79   store.pushRetrieve(SC.Record, 1, recSecond, sK);
 80   SC.RunLoop.end();
 81   equals(store.idFor(sK), 1); // id should now have been updated
 82 });
