  9 /**
 10   This class view is for use in the Views & Controls lists.
 11 */
 12 Showcase.ViewsItemView = SC.View.extend({
 13 
 14   classNames: ['views-item-view'],
 15 
 16   content: null,
 17 
 18   isLastItem: false,
 19 
 20   /**
 21     Returns the layout for the example box.
 22   */
 23   exampleLayout: function () {
 24     var owner = this.get('owner'),
 25         exampleHeight = owner.get('exampleHeight');
 26 
 27     return { left: 60, top: 70, right: 60, height: exampleHeight }
 28   }.property().cacheable(),
 29 
 30   /** @private */
 31   render: function (context, firstTime) {
 32     var isLastItem = this.get('isLastItem');
 33 
 34     if (isLastItem) { context.addClass('last-item'); }
 35   },
 36 
 37   /** @private */
 38   update: function (jQuery) {
 39     var isLastItem = this.get('isLastItem');
 40 
 41     jQuery.toggleClass('last-item', isLastItem);
 42   },
 43 
 44   childViews: ['titleLabel', 'exampleBox', 'supportButton', 'disclosure', 'snippetField'],
 45 
 46   titleLabel: SC.LabelView.design({
 47     classNames: ['title-label'],
 48     escapeHTML: NO,
 49     layout: { left: 20, top: 20, right: 20, height: 50 },
 50     tagName: 'h2',
 51     valueBinding: SC.Binding.oneWay('.parentView.content.title')
 52   }),
 53 
 54   exampleBox: SC.ContainerView.design({
 55     classNames: ['example-box'],
 56     layoutBinding: SC.Binding.oneWay('.parentView.exampleLayout'),
 57     nowShowingBinding: SC.Binding.oneWay('.parentView.content.exampleView'),
 58     replaceContent: function(newContent) {
 59       // SC.ContainerView needs its awake function to be called to be correctly initialized.
 60       newContent.awake();
 61 
 62       sc_super();
 63     },
 64   }),
 65 
 66   supportButton: SC.ButtonView.design({
 67     actionBinding: SC.Binding.oneWay('.parentView.content.supportAction'),
 68     isVisibleBinding: SC.Binding.oneWay('.parentView.content.supportTitle').bool(),
 69     layoutBinding: SC.Binding.oneWay('.parentView.exampleLayout').transform(function(layout) {
 70       var ourLayout = { height: 24, right: 60, top: layout.top + layout.height + 10, width: 140 };
 71       return ourLayout;
 72     }),
 73     titleBinding: SC.Binding.oneWay('.parentView.content.supportTitle')
 74   }),
 75 
 76   disclosure: SC.DisclosureView.design({
 77     layoutBinding: SC.Binding.oneWay('.parentView.exampleLayout').transform(function(layout) {
 78       var ourLayout = { left: 10, height: 25, width: 150, top: layout.top + layout.height + 20 };
 79       return ourLayout;
 80     }),
 81 
 82     title: function() {
 83       return this.get('value') ? 'Hide' : 'Show Code';
 84     }.property('value').cacheable(),
 85 
 86     valueBinding: SC.Binding.from('.parentView.content.isShowingSnippet')
 87   }),
 88 
 89   snippetField: SC.TextFieldView.design({
 90     classNames: ['snippet-field'],
 91     isEditable: false,
 92     isTextArea: true,
 93     isVisibleBinding: SC.Binding.oneWay('.parentView.disclosure.value'),
 94     layoutBinding: SC.Binding.oneWay('.parentView.exampleLayout').transform(function(layout) {
 95       var ourLayout = { left: 15, bottom: 15, right: 15, top: layout.top + layout.height + 50 };
 96       return ourLayout;
 97     }),
 98     valueBinding: SC.Binding.oneWay('.parentView.content.exampleString')
 99   })
100 });
