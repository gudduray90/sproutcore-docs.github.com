  9 module("SC.Validator.Date");
 10 
 11 test("Converts into String if a number is given",function() {
 12   var date = SC.DateTime.create({milliseconds: 1234947136000 + (new Date().getTimezoneOffset() * 60000)});
 13 
 14   var c = SC.Validator.Date.fieldValueForObject(date._ms, '', '');
 15   var expected = "Feb 18, 2009 8:52:16 AM";
 16   equals(c, expected, "Number converted to date format");
 17 });
 18 
 19 test("Converts into number when date string is given", function() {
 20   var expected = 1234918336000;
 21   var date = SC.DateTime.create(expected);
 22   var d = SC.Validator.Date.objectForFieldValue(date.toFormattedString('%b %d, %Y %i:%M:%S %p'),'','');
 23   equals(d, expected, "Number of milliseconds returned is correct");
 24 });
