  7 /*global module test equals context */
  8 module("Render Context--Escaping HTML");
  9 test("Escaping HTML", function() {
 10   var input = "<p>HTML!</p><script>alert('hi');<" + "/script> & Hello, World!";
 11   var output = SC.RenderContext.escapeHTML(input);
 12   
 13   equals(output, '<p>HTML!</p><script>alert(\'hi\');</script> & Hello, World!', "Properly escapes HTML");
 14 });
