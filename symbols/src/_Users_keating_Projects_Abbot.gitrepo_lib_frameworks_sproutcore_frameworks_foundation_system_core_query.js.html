  8  SC.mixin(SC.$.fn, /** @scope SC.$.prototype */ {
  9 
 10   /**
 11     You can either pass a single class name and a boolean indicating whether
 12     the value should be added or removed, or you can pass a hash with all
 13     the class names you want to add or remove with a boolean indicating
 14     whether they should be there or not.
 15 
 16     This is far more efficient than using addClass/removeClass.
 17 
 18     @param {String|Hash} className class name or hash of classNames + bools
 19     @param {Boolean} shouldAdd for class name if a string was passed
 20  21     @returns {SC.CoreQuery} receiver
 22   */
 23   setClass: function(className, shouldAdd) {
 24     if (SC.none(className)) { return this; } //nothing to do
 25     var isHash = SC.typeOf(className) !== SC.T_STRING,
 26         fix = this._fixupClass, key;
 27 
 28     this.each(function() {
 29       if (this.nodeType !== 1) { return; } // nothing to do
 30 
 31       // collect the class name from the element and build an array
 32       var classNames = this.className.split(/\s+/), didChange = NO;
 33 
 34       // loop through hash or just fix single className
 35       if (isHash) {
 36         for(var key in className) {
 37           if (className.hasOwnProperty(key)) {
 38             didChange = fix(classNames, key, className[key]) || didChange;
 39           }
 40         }
 41       } else {
 42         didChange = fix(classNames, className, shouldAdd);
 43       }
 44 
 45       // if classNames were changed, join them and set...
 46       if (didChange) { this.className = classNames.join(' '); }
 47     });
 48     return this ;
 49   },
 50 
 51   /** @private used by setClass */
 52   _fixupClass: function(classNames, name, shouldAdd) {
 53     var indexOf = classNames.indexOf(name);
 54     // if should add, add class...
 55     if (shouldAdd) {
 56       if (indexOf < 0) { classNames.push(name); return YES ; }
 57 
 58     // otherwise, null out class name (this will leave some extra spaces)
 59     } else if (indexOf >= 0) { classNames[indexOf]=null; return YES; }
 60     return NO ;
 61   }
 62 
 63 
 64 });
