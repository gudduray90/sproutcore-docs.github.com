 28   didCreateLayerMixin: function() {

 33   /**
 34     Animate a given property using CSS animations.
 35 
 36     Takes a key, value and either a duration, or a hash of options.
 37     The options hash has the following parameters
 38 
 39      - duration: Duration of animation in seconds
 40      - callback: Callback method to run when animation completes
 41      - timing: Animation timing function
 42 
 43     @param {String|Hash} key
 44     @param {Object} value
 45     @params {Number|Hash} duration or options
 46     @returns {SC.View} receiver
 47   */
 48   animate: function(keyOrHash, valueOrOptions, optionsOrCallback, callback) {
 49     var hash, options;
 50 
 51     if (typeof keyOrHash === SC.T_STRING) {
 52       hash = {};
 53       hash[keyOrHash] = valueOrOptions;
 54       options = optionsOrCallback;
 55     } else {
 56       hash = keyOrHash;
 57       options = valueOrOptions;
 58       callback = optionsOrCallback;
 59     }
 60 
 61     var optionsType = SC.typeOf(options);
 62     if (optionsType === SC.T_NUMBER) {
 63       options = { duration: options };
 64     } else if (optionsType !== SC.T_HASH) {
 65       throw "Must provide options hash or duration!";
 66     }
 67 
 68     if (callback) { options.callback = callback; }
 69 
 70     var timing = options.timing;
 71     if (timing) {
 72       if (typeof timing !== SC.T_STRING) {
 73         options.timing = "cubic-bezier("+timing[0]+", "+timing[1]+", "+
 74                                          timing[2]+", "+timing[3]+")";
 75       }
 76     } else {
 77       options.timing = 'linear';
 78     }
 79 
 80     var layout = SC.clone(this.get('layout')), didChange = NO, value, cur, animValue, curAnim, key;
 81 
 82     if (!layout.animate) { layout.animate = {}; }
 83 
 84     // Very similar to #adjust
 85     for(key in hash) {
 86       if (!hash.hasOwnProperty(key) || !SC.ANIMATABLE_PROPERTIES[key]) { continue; }
 87       value = hash[key];
 88       cur = layout[key];
 89       curAnim = layout.animate[key];
 90 
 91       // loose comparison used instead of (value === null || value === undefined)
 92       if (value == null) { throw "Can only animate to an actual value!"; }
 93 
 94       // FIXME: We should check more than duration
 95       if (cur !== value || (curAnim && curAnim.duration !== options.duration)) {
 96         didChange = YES;
 97         layout.animate[key] = options;
 98         layout[key] = value;
 99 100       }
101     }
102 
103     // now set adjusted layout
104     if (didChange) { this.set('layout', layout) ; }
105 
106     return this ;
107   },
108 
109   /**
110   Resets animation, stopping all existing animations.
111   */
112   resetAnimation: function() {
113     var layout = this.get('layout'),
114         animations = layout.animate,
115         didChange = NO, key;
116 
117     if (!animations) { return; }
118 
119     var hasAnimations;
120 
121     for (key in animations) {
122       didChange = YES;
123       delete animations[key];
124     }
125 
126     if (didChange) {
127       this.set('layout', layout);
128       this.notifyPropertyChange('layout');
129     }
130 
131     return this;
132   },
133 
134   /**
135     Called when animation ends, should not usually be called manually

137   transitionDidEnd: function(evt){
138     // WARNING: Sometimes this will get called more than once for a property. Not sure why.
139     this.get('layoutStyleCalculator').transitionDidEnd(evt);
140   },

142   /**
143     Setting wantsAcceleratedLayer to YES will use transforms to move the
144     layer when available. On some platforms transforms are hardware accelerated.
145   */
146   wantsAcceleratedLayer: NO,
147 
148   /**
149     Specifies whether transforms can be used to move the layer.
150   */
151   hasAcceleratedLayer: function(){
152     if (this.get('wantsAcceleratedLayer') && SC.platform.supportsAcceleratedLayers) {
153       var layout = this.get('layout'),
154           animations = layout.animate,
155           AUTO = SC.LAYOUT_AUTO,
156           key;
157 
158       if (animations && (animations.top || animations.left)) {
159         for (key in animations) {
160           // If we're animating other transforms at different speeds, don't use acceleratedLayer
161           if (
162             SC.CSS_TRANSFORM_MAP[key] &&
163             ((animations.top && animations.top.duration !== animations[key].duration) ||
164              (animations.left && animations.left.duration !== animations[key].duration))
165           ) {
166             return NO;
167           }
168         }
169       }
170 
171       // loose comparison used instead of (layout.X === null || layout.X === undefined)
172       if (
173         layout.left != null && !SC.isPercentage(layout.left) && layout.left !== AUTO &&
174         layout.top != null && !SC.isPercentage(layout.top) && layout.top !== AUTO &&
175         layout.width != null && !SC.isPercentage(layout.width) && layout.width !== AUTO &&
176         layout.height != null && !SC.isPercentage(layout.height) && layout.height !== AUTO
177       ) {
178        return YES;
179       }
180     }
181     return NO;
182   }.property('wantsAcceleratedLayer').cacheable()
