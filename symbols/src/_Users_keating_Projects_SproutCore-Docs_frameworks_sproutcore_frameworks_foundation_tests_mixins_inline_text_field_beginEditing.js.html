  9 var pane, optionsForLabel1, optionsForLabel2, delegate, optionsForLabelFromView;

 11 pane = SC.ControlTestPane.design().add("label1", SC.LabelView, {
 12   value: 'Some Text',
 13   notifiedWillBegin: NO,
 14   notifiedDidBegin: NO,
 15 
 16   inlineEditorShouldBeginEditing: function(inlineEditor) {
 17     return YES;
 18   },
 19 
 20   inlineEditorWillBeginEditing: function(inlineEditor) {
 21     this.set('notifiedWillBegin', YES);
 22 
 23     // The inline editor is the last view appended to the pane
 24     var length = pane._pane.childViews.length,
 25     editor = pane._pane.childViews[length - 1];
 26 
 27     ok(!editor.get('isFirstResponder'), "should not be first responder yet");
 28 
 29     sc_super();
 30   },
 31 
 32   inlineEditorDidBeginEditing: function(inlineEditor) {
 33     this.set('notifiedDidBegin', YES);
 34 
 35     // The inline editor is the last view appended to the pane
 36     var length = pane._pane.childViews.length,
 37     editor = pane._pane.childViews[length - 1];
 38 
 39     ok(editor.get('isFirstResponder'), "should be first responder now");
 40 
 41     sc_super();
 42   }
 43 }).add("label2", SC.LabelView, {
 44   value: 'Can\'t Touch This',
 45 
 46   inlineEditorShouldBeginEditing: function(inlineEditor) {
 47     return NO;
 48   }
 49 });
 50 
 51 pane.resetView = function(view) {
 52   view.set('notifiedWillBegin', NO);
 53   view.set('notifiedDidBegin', NO);
 54 };
 55 
 56 
 57 optionsForLabelFromView = function(view) {
 58   var el = view.$(),
 59   f = SC.offset(el[0]),
 60   frameTemp = view.convertFrameFromView(view.get('frame'), null);
 61 
 62   f.width = frameTemp.width;
 63   f.height = frameTemp.height;
 64 
 65   var optionsForLabel = {
 66     frame: f,
 67     delegate: view,
 68     pane: pane._pane,
 69     exampleElement: view.$(),
 70     value: view.get('value'),
 71     multiline: view.get('isInlineEditorMultiline'),
 72     isCollection: NO,
 73     validator: view.get('validator'),
 74     exampleInlineTextFieldView: view.get('exampleInlineTextFieldView')
 75   };
 76 
 77   return optionsForLabel;
 78 };

199 
200   // Start a run loop because the notification isn't invoked until the beginning of the next run loop
