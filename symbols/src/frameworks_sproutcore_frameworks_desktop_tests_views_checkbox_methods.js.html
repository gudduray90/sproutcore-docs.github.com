 10 (function() {
 11   var pane, view, triggered = false;
 12 
 13   module("SC.Checkbox", {
 14     setup: function() {
 15       SC.RunLoop.begin();
 16 
 17       // actions must be strings, and there must be a target. So, we need this dummy.
 18       var act = SC.Object.create({
 19         action: function() {
 20           triggered = true;
 21         }
 22       });
 23 
 24       pane = SC.MainPane.create({
 25         childViews: [
 26           SC.CheckboxView.extend({
 27             layout: { right: 20, bottom: 20, width: 100, height: 23 },
 28             title: "First Name",
 29             value: YES,
 30             target: act,
 31             action: 'action'
 32           })]
 33       });
 34       pane.append(); // make sure there is a layer...
 35       SC.RunLoop.end();
 36 
 37       view = pane.childViews[0];
 38     },
 39 
 40     teardown: function() {
 41       pane.remove();
 42       pane = view = null;
 43       triggered = false;
 44     }
 45   });
 46 
 47   test("renders something that is NOT an input tag with appropriate attributes", function() {
 48     equals(view.get('value'), YES, 'precon - value should be YES');
 49 
 50     var q = view.$();
 51     equals(q.attr('role'), 'checkbox', 'should have type=checkbox');
 52     equals(q.attr('aria-checked'), 'true', 'should be checked');
 53   });
 54 
 55   test("double-check that there is NO input element. Just in case.", function(){
 56     equals(view.$("input").length, 0, "CoreQuery lookup for inputs should have length of 0.");
 57   });
 58 
 59   test("should have span with title inside", function() {
 60     var q = Q$('span', view.get('layer'));
 61     ok(view.get('displayTitle').length>0, 'precond - display title should not be empty');
 62     equals(q.text(), view.get('displayTitle'), 'should have display title');
 63   });
 64 
 65   test("changing the title should update the span", function() {
 66     var oldDisplayTitle = view.get('displayTitle');
 67     SC.RunLoop.begin();
 68     view.set('title', 'Last Name');
 69     SC.RunLoop.end();
 70 
 71     ok(view.get('displayTitle') !== oldDisplayTitle, 'precond - should have changed display title');
 72 
 73     var q = Q$('span', view.get('layer'));
 74     equals(q.text(), view.get('displayTitle'), 'should have display title');
 75   });
 76 
 77   test("isEnabled=NO should add disabled class", function() {
 78     SC.RunLoop.begin();
 79     view.set('isEnabled', NO);
 80     SC.RunLoop.end();
 81 
 82     ok(view.$().hasClass('disabled'), 'should have disabled class');
 83   });
 84 
 85 
 86   test("isSelected should alter sel classname and sync with value property", function() {
 87 
 88     // check initial render state
 89     ok(view.get('isSelected'), 'isSelected should match value');
 90     ok(view.$().hasClass('sel'), 'should have sel class');
 91 
 92     // update value -- make sure isSelected changes.
 93     SC.RunLoop.begin();
 94     view.set('value', 0); // make falsy. (but not NO exactly)
 95     SC.RunLoop.end();
 96 
 97     ok(!view.get('isSelected'), 'isSelected should now be NO');
 98     ok(!view.$().hasClass('sel'), 'should no longer have sel class');
 99     equals(view.$().attr('aria-checked'), 'false', 'view should not be checked');
100 
101     // update isSelected -- make sure it edits the value
102     SC.RunLoop.begin();
103     view.set('isSelected', YES);
104     SC.RunLoop.end();
105 
106     ok(view.get('isSelected'), 'isSelected should match value');
107     ok(view.$().hasClass('sel'), 'should have sel class');
108     equals(view.$().attr('aria-checked'), 'true', 'aria-checked should be true');
109   });
110 
111   test("mouseDown and then mouseUp anywhere in the checkbox should toggle the selection", function() {
112     var elem = view.get('layer'),
113       evt;
114 
115     ok(!triggered, 'precond - action should not have been triggered yet');
116 
117     evt = SC.Event.simulateEvent(elem, 'mousedown', { which: 1 });
118     SC.Event.trigger(elem, 'mousedown', [evt]);
119     ok(view.get('isActive'), 'view should be active');
120     ok(view.get('value'), 'value should not change yet');
121     equals(view.$().attr('aria-checked'), 'true', 'aria-checked should be true');
122 
123     // simulate mouseUp and browser-native change to control
124     SC.Event.trigger(elem,'mouseup');
125 
126     ok(!view.get('isActive'), 'view should no longer be active');
127     ok(!view.get('value'), 'value should change to NO');
128     equals(view.$().attr('aria-checked'), 'false', 'aria-checked should be false');
129 
130     ok(triggered, 'action should have been triggered');
131 
132     elem = null ;
133   });
134 
135   test("isEnabled=NO should add disabled attr to input", function() {
136     SC.RunLoop.begin();
137     view.set('isEnabled', NO);
138     SC.RunLoop.end();
139 
140     ok(view.get('value'), 'precond - value should be true');
141     var evt = SC.Event.simulateEvent(view.get('layer'), 'mousedown', { which: 1 });
142     view.mouseDown(evt);
143     view.mouseUp();
144     ok(view.get('value'), 'value should not change when clicked');
145   });
146 
147 })();
