 51 SC.View.reopen(
 52   /** @scope SC.View.prototype */ {
 53 
 54   // ------------------------------------------------------------------------
 55   // Properties
 56   //
 57 
 58   /**
 59     The current enabled state of the view.
 60 
 61     Views have a few possible enabled states:
 62 
 63     * SC.CoreView.ENABLED
 64     * SC.CoreView.DISABLED
 65     * SC.CoreView.DISABLED_BY_PARENT
 66 
 67     @type String
 68     @default SC.CoreView.ENABLED
 69     @readonly
 70   */
 71   enabledState: SC.CoreView.ENABLED,
 72 
 73   /**
 74     Set to true when the item is enabled.   Note that changing this value
 75     will alter the `isEnabledInPane` property for this view and any
 76     child views as well as to automatically add or remove a 'disabled' CSS
 77     class name.
 78 
 79     This property is observable and bindable.
 80 
 81     @type Boolean
 82   */
 83   isEnabled: YES,
 84   isEnabledBindingDefault: SC.Binding.oneWay().bool(),
 85 
 86   /**
 87     Computed property returns YES if the view and all of its parent views
 88     are enabled in the pane.  You should use this property when deciding
 89     whether to respond to an incoming event or not.
 90 
 91     @type Boolean
 92   */
 93   // The previous version used a lazy upward search method.  This has better
 94   // performance, but made isEnabledInPane non-bindable.
 95   // isEnabledInPane: function() {
 96   //   var ret = this.get('isEnabled'), pv ;
 97   //   if (ret && (pv = this.get('parentView'))) { ret = pv.get('isEnabledInPane'); }
 98   //   return ret ;
 99   // }.property('parentView', 'isEnabled'),
100   isEnabledInPane: function () {
101     return this.get('enabledState') === SC.CoreView.ENABLED;
102   }.property('enabledState').cacheable(),
103 
104   /**
105     By default, setting isEnabled to false on a view will place all of its
106     child views in a disabled state.  To block this from happening to a
107     specific child view and its children, you can set `shouldInheritEnabled`
108     to false.
109 
110     In this way you can set `isEnabled` to false on a main pane to disable all
111     buttons, collections and other controls within it, but can still keep a
112     section of it editable using `shouldInheritEnabled: false`.
113 
114     @type Boolean
115   */
116   shouldInheritEnabled: true,
117 
118   // ------------------------------------------------------------------------
119   // Actions & Events
120   //
121 
122   /** @private */
123   _doEnable: function () {
124     var handled = true,
125       enabledState = this.get('enabledState');
126 
127     if (enabledState === SC.CoreView.DISABLED) {
128       // If the view itself is disabled, then we can enable it.
129       this._callOnChildViews('_parentDidEnableInPane');
130       this._gotoEnabledState();
131     } else if (enabledState === SC.CoreView.DISABLED_AND_BY_PARENT) {
132       // The view is no longer disabled itself, but still disabled by an ancestor.
133       this._gotoDisabledByParentState();
134     } else {
135       handled = false;
136     }
137 
138     return handled;
139   },
140 
141   /** @private */
142   _doDisable: function () {
143     var handled = true,
144       enabledState = this.get('enabledState');
145 
146     // If the view is not itself disabled, then we can disable it.
147     if (enabledState === SC.CoreView.ENABLED) {
148       if (this.get('isFirstResponder')) {
149         this.resignFirstResponder();
150       }
151 
152       this._callOnChildViews('_parentDidDisableInPane');
153       this._gotoDisabledState();
154     } else if (enabledState === SC.CoreView.DISABLED_BY_PARENT) {
155       // The view is now disabled itself and disabled by an ancestor.
156       this._gotoDisabledAndByParentState();
157     } else {
158       handled = false;
159     }
160 
161     return handled;
162   },
163 
164   // ------------------------------------------------------------------------
165   // Methods
166   //
167 
168   /** @private
169     Observes the isEnabled property and resigns first responder if set to NO.
170     This will avoid cases where, for example, a disabled text field retains
171     its focus rings.
172 
173     @observes isEnabled
174   */
175   _sc_view_isEnabledDidChange: function () {
176     // Filter the input channel.
177     this.invokeOnce(this._doUpdateEnabled);
178   }.observes('isEnabled'),
179 
180   /** @private */
181   _doUpdateEnabled: function () {
182     var state = this.get('viewState');
183 
184     // Call the proper action.
185     if (this.get('isEnabled')) {
186       this._doEnable();
187     } else {
188       this._doDisable();
189     }
190 
191     // Update the display if in a visible state.
192     switch (state) {
193     case SC.CoreView.ATTACHED_SHOWN:
194     case SC.CoreView.ATTACHED_SHOWING:
195     case SC.CoreView.ATTACHED_BUILDING_IN:
196       // Update the display.
197       this._doUpdateEnabledStyle();
198       break;
199     default:
200       // Indicate that a display update is required the next time we are visible.
201       this._enabledStyleNeedsUpdate = true;
202     }
203   },
204 
205   /** @private */
206   _doUpdateEnabledStyle: function () {
207     var isEnabled = this.get('isEnabled');
208 
209     this.$().toggleClass('disabled', !isEnabled);
210     this.$().attr('aria-disabled', !isEnabled ? true : null);
211 
212     // Reset that an update is required.
213     this._enabledStyleNeedsUpdate = false;
214   },
215 
216   /** @private */
217   _parentDidEnableInPane: function () {
218     var enabledState = this.get('enabledState');
219 
220     if (this.get('shouldInheritEnabled')) {
221 
222       if (enabledState === SC.CoreView.DISABLED_BY_PARENT) { // Was enabled before.
223         this._gotoEnabledState();
224       } else if (enabledState === SC.CoreView.DISABLED_AND_BY_PARENT) { // Was disabled before.
225         this._gotoDisabledState();
226 
227         // There's no need to continue to further child views.
228         return false;
229       }
230     } else {
231       // There's no need to continue to further child views.
232       return false;
233     }
234   },
235 
236   /** @private */
237   _parentDidDisableInPane: function () {
238     var enabledState = this.get('enabledState');
239 
240     if (this.get('shouldInheritEnabled')) {
241 
242       if (enabledState === SC.CoreView.ENABLED) { // Was enabled.
243         this._gotoDisabledByParentState();
244       } else if (enabledState === SC.CoreView.DISABLED) { // Was disabled.
245         this._gotoDisabledAndByParentState();
246       } else { // Was already disabled by ancestor.
247 
248         // There's no need to continue to further child views.
249         return false;
250       }
251     } else {
252       // There's no need to continue to further child views.
253       return false;
254     }
255   },
256 
257   /** @private */
258   _gotoEnabledState: function () {
259     this.set('enabledState', SC.CoreView.ENABLED);
260   },
261 
262   /** @private */
263   _gotoDisabledState: function () {
264     this.set('enabledState', SC.CoreView.DISABLED);
265   },
266 
267   /** @private */
268   _gotoDisabledAndByParentState: function () {
269     // Update the state.
270     this.set('enabledState', SC.CoreView.DISABLED_AND_BY_PARENT);
271   },
272 
273   /** @private */
274   _gotoDisabledByParentState: function () {
275     // Update the state.
276     this.set('enabledState', SC.CoreView.DISABLED_BY_PARENT);
277   }
278 
279 });
