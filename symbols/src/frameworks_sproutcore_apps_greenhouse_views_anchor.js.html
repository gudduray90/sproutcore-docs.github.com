 13 Greenhouse.AnchorView = SC.View.extend(
 14 /** @scope Greenhouse.AnchorView.prototype */ {
 15 
 16   /**
 17     The anchor location to display
 18   */
 19   anchorLocation: null,
 20 
 21   /**
 22     Enabled/disable
 23   */
 24   isEnabled: YES,
 25 
 26   /**
 27     Set to YES while the mouse is pressed.
 28   */
 29   isActive: NO,
 30 
 31   /**
 32     Proposed anchor location.  Changes as mouse moves/drags
 33   */
 34   proposedAnchorLocation: null,
 35 
 36   displayProperties: "anchorLocation isEnabled isActive proposedAnchorLocation".w(),
 37 
 38   render: function(context, firstTime) {
 39     if (firstTime) {
 40       var f = this.get('frame');
 41       context.begin('canvas')
 42         .attr('width', f.width).attr('height', f.height)
 43         .end();
 44     }
 45   },
 46 
 47   didCreateLayer: function() {
 48     this.didUpdateLayer();
 49   },
 50 
 51   didUpdateLayer: function() {
 52     var elem   = this.$('canvas'),
 53         ctx    = elem[0].getContext("2d"),
 54         width  = this.$().width(),
 55         height = this.$().height(),
 56         loc    = this.get('anchorLocation'),
 57         ploc, color, x, y, tmp;
 58 
 59     // adjust size as needed...
 60     if (Number(elem.attr('width')) !== width) elem.attr('width', width);
 61     if (Number(elem.attr('height')) !== height) elem.attr('height', height);
 62     width--; height--; // adjust for being off 0.5
 63 
 64     // do  the drawr-ing!
 65     if (!this.get('isEnabled')) loc = null;
 66     color = loc ? 'black' : 'rgb(128,128,128)';
 67 
 68     ctx.save();
 69     ctx.lineWidth = 1;
 70     ctx.fillStyle = 'rgb(255,255,255)';
 71     ctx.strokeStyle = color;
 72     ctx.fillRect(0.5, 0.5, width, height);
 73     ctx.strokeRect(0.5, 0.5, width, height);
 74 
 75     ctx.strokeStyle = color;
 76     ctx.strokeRect(20.5, 20.5, width-40, height-40);
 77 
 78     ctx.beginPath();
 79     ctx.moveTo(Math.floor(width/2)+0.5, 20.5);
 80     ctx.lineTo(Math.floor(width/2)+0.5, Math.floor(height-20)+0.5);
 81     ctx.moveTo(20.5, Math.floor(height/2)+0.5);
 82     ctx.lineTo(Math.floor(width-20)+0.5, Math.floor(height/2)+0.5);
 83     ctx.stroke();
 84     ctx.restore();
 85 
 86     loc = this.get('anchorLocation');
 87 
 88     ploc = this.get('proposedAnchorLocation');
 89     if (ploc && ploc !== loc) {
 90       color = this.get('isActive') ? 'rgb(80,80,80)' : 'rgb(200,200,200)';
 91       this._drawAnchorAt(ploc, ctx, color, width, height);
 92     }
 93 
 94     this._drawAnchorAt(loc, ctx, 'red', width, height);
 95   },
 96 
 97   // ..........................................................
 98   // MOUSE EVENTS
 99   //
100 
101   mouseMoved: function(evt) {
102     this._updateProposedAnchorLocation(evt);
103   },
104 
105   mouseExited: function(evt) {
106     this.setIfChanged('proposedAnchorLocation', null);
107   },
108 
109   mouseDown: function(evt) {
110     if (this.get('isEnabled') && this.get('anchorLocation')) {
111       this.get('mouseDown');
112 
113       this.set('isActive', YES);
114       this._updateProposedAnchorLocation(evt);
115     }
116     return YES ;
117   },
118 
119   mouseDragged: function(evt) {
120     if (this.get('isActive')) this._updateProposedAnchorLocation(evt);
121     return YES ;
122   },
123 
124   mouseUp: function(evt) {
125     var loc;
126 
127     if (this.get('isActive')) {
128       this._updateProposedAnchorLocation(evt);
129       loc = this.get('proposedAnchorLocation');
130       if (loc) this.setIfChanged('anchorLocation', loc);
131       this.set('isActive', NO);
132     }
133 
134     return YES ;
135   },
136

142   _updateProposedAnchorLocation: function(evt) {
143     var loc = this.get('anchorLocation'),
144         pnt = this.convertFrameFromView({ x: evt.pageX, y: evt.pageY },null),
145         K   = SC.ViewDesigner,
146         rad, f, w, h, ret, centerAnchor, centerResize;
147 
148     if (!this.get('isEnabled') || !loc) ret = null;
149     else {
150       rad = 10;
151       f = SC.copy(this.get('frame'));
152 
153       // calc outside rect
154       f.x = f.y = 20;
155       f.width -= 40 ;
156       f.height -= 40;
157 
158       if (Math.abs(pnt.x - SC.minX(f))<=rad) w = K.ANCHOR_LEFT;
159       else if (Math.abs(pnt.x - SC.midX(f))<=rad) w = K.ANCHOR_CENTERX;
160       else if (Math.abs(pnt.x - SC.maxX(f))<=rad) w = K.ANCHOR_RIGHT;
161       else w = 0;
162 
163       if (Math.abs(pnt.y - SC.minY(f))<=rad) h = K.ANCHOR_TOP;
164       else if (Math.abs(pnt.y - SC.midY(f))<=rad) h = K.ANCHOR_CENTERY;
165       else if (Math.abs(pnt.y - SC.maxY(f))<=rad) h = K.ANCHOR_BOTTOM;
166       else h = 0;
167 
168       // not in a regular anchor zone; look for edges...
169       if (w===0 || h===0) {
170         rad /= 2;
171         if (Math.abs(pnt.x - SC.minX(f))<=rad) {
172           ret = K.ANCHOR_LEFT | K.ANCHOR_HEIGHT;
173 
174         } else if (Math.abs(pnt.x - SC.midX(f)) <= rad) {
175           ret = K.ANCHOR_CENTERX | K.ANCHOR_HEIGHT;
176 
177         } else if (Math.abs(pnt.x - SC.maxX(f)) <= rad) {
178           ret = K.ANCHOR_RIGHT | K.ANCHOR_HEIGHT;
179 
180         } else if (Math.abs(pnt.y - SC.minY(f)) <= rad) {
181           ret = K.ANCHOR_WIDTH | K.ANCHOR_TOP;
182 
183         } else if (Math.abs(pnt.y - SC.midY(f)) <= rad) {
184           ret = K.ANCHOR_WIDTH | K.ANCHOR_CENTERY;
185 
186         } else if (Math.abs(pnt.y - SC.maxY(f)) <= rad) {
187           ret = K.ANCHOR_WIDTH | K.ANCHOR_BOTTOM;
188         }
189 
190       } else ret = w|h;
191       if (ret === 0) ret = null;
192     }
193 
194     // alternate between center anchor/resize if options...
195     centerAnchor = K.ANCHOR_CENTERX | K.ANCHOR_CENTERY;
196     centerResize = K.ANCHOR_WIDTH | K.ANCHOR_HEIGHT;
197     if (loc===ret) {
198       if (ret===centerAnchor) ret = centerResize;
199       else if (ret===centerResize) ret = centerAnchor;
200     }
201 
202     this.setIfChanged('proposedAnchorLocation', ret);
203   },
204

206   _drawAnchorAt: function(loc, ctx, color, width, height) {
207     var x = this._xForAnchorLocation(loc, 20, width-40),
208         y = this._yForAnchorLocation(loc, 20, height-40),
209         tmp;
210 
211     // if either is zero - don't do anything
212     if (x && y) {
213       ctx.save();
214       ctx.strokeStyle = color;
215 
216       // if x|y < 0, then draw over lines to show height/width
217       if (x<0) {
218         tmp = Math.floor(Math.abs(y));
219         ctx.lineWidth = 2;
220 
221         ctx.beginPath();
222         ctx.arc(20.5, tmp, 3, 0, Math.PI*2, true);
223         ctx.lineTo(Math.floor(width-20)-3.5, tmp);
224         ctx.arc(Math.floor(width-20), tmp, 3, Math.PI, Math.PI*2, true);
225         ctx.arc(Math.floor(width-20), tmp, 3, 0, Math.PI, true);
226         ctx.stroke();
227       }
228 
229       if (y<0) {
230         tmp = Math.floor(Math.abs(x));
231         ctx.lineWidth = 2;
232 
233         ctx.beginPath();
234         ctx.arc(tmp, 20.5, 3, 0, Math.PI*2, true);
235         ctx.moveTo(tmp, 23.5);
236         ctx.lineTo(tmp, Math.floor(height-20)-3.5);
237         ctx.arc(tmp, Math.floor(height-20), 3, Math.PI*1.5, Math.PI*2, true);
238         ctx.arc(tmp, Math.floor(height-20), 3, 0, Math.PI*1.5, true);
239         ctx.stroke();
240       }
241 
242       if (x>0 && y>0) {
243         ctx.beginPath();
244         ctx.lineWidth = 2;
245         ctx.arc(x,y,10,0,Math.PI*2, true);
246         ctx.stroke();
247       }
248 
249       ctx.restore();
250     }
251   },
252 
253   _xForAnchorLocation: function(loc, left, w) {
254     var K = SC.ViewDesigner, ret ;
255 
256     if (loc & K.ANCHOR_LEFT) ret = left;
257     else if (loc & K.ANCHOR_RIGHT) ret = left+w;
258     else if (loc & K.ANCHOR_CENTERX) ret = left+Math.floor(w/2);
259     else if (loc & K.ANCHOR_WIDTH) ret = 0-(left+Math.floor(w/2)) ;
260     else ret = 0;
261 
262     return ret ;
263   },
264 
265   _yForAnchorLocation: function(loc, top, h) {
266     var K = SC.ViewDesigner, ret ;
267 
268     if (loc & K.ANCHOR_TOP) ret = top;
269     else if (loc & K.ANCHOR_BOTTOM) ret = top+h;
270     else if (loc & K.ANCHOR_CENTERY) ret = top+Math.floor(h/2);
271     else if (loc & K.ANCHOR_HEIGHT) ret = 0-(top+Math.floor(h/2)) ;
272     else ret = 0;
273 
274     return ret ;
275   }
