 12 var initModels = function() {
 13   NestedRecord.ParentRecordTest = SC.Record.extend({
 14     /** Child Record Namespace */
 15     nestedRecordNamespace: NestedRecord,
 16

 31 module("Basic SC.Record Functions w/ Parent > Child", {

 34     NestedRecord = SC.Object.create({
 35       store: SC.Store.create()
 36     });
 37     window.NestedRecord = NestedRecord;
 38     store = NestedRecord.store;
 39     initModels();
 40     SC.RunLoop.begin();
 41     // Test Parent 1
 42     testParent = store.createRecord(NestedRecord.ParentRecordTest, {
 43       name: 'Parent Name',
 44       info: {
 45         type: 'ChildRecordTest',
 46         name: 'Child Name',
 47         value: 'Blue Goo',
 48         guid: '5001'
 49       }
 50     });
 51     // Test parent 2
 52     testParent2 = NestedRecord.store.createRecord(NestedRecord.ParentRecordTest, {
 53       name: 'Parent Name 2',
 54       info: {
 55         type: 'ChildRecordTest',
 56         name: 'Child Name 2',
 57         value: 'Purple Goo',
 58         guid: '5002'
 59       }
 60     });
 61     SC.RunLoop.end();
 62     
 63     
 64     // ..........................................................
 65     // Child Data
 66     // 
 67     childData1 = {
 68       type: 'ChildRecordTest',
 69       name: 'Child Name',
 70       value: 'Green Goo',
 71       guid: '5002'
 72     };
 73   },
 74 
 75   teardown: function() {
 76     delete NestedRecord.ParentRecordTest;
 77     delete NestedRecord.ChildRecordTest;
 78     testParent = null;

148 test("Basic Read", function() {

152   equals(testParent.get('nothing'), null, "get should be correct for invalid key");

165   equals(cr, storeRef, "checking the parent reference is the same as the direct store reference");
166   
167   // Check to see if the attributes of a Child Record match the reference of the parent
168   same(storeRef.get('attributes'), testParent.readAttribute('info'), "check that the ChildRecord's attributes are the same as the ParentRecord's readAttribute for the reference");
169   
170   // Duplication check
171   var sameCR = testParent.get('info');
172   ok(sameCR, "check to see if we have an instance of a child record again");
173   var oldKey = cr.get(pm), newKey = sameCR.get(pm);
174   equals(oldKey, newKey, "check to see if the Primary Key are the same");
175   same(sameCR, cr, "check to see that it is the same child record as before");

178 test("Basic Write As a Hash", function() {
179   
180   // Test general gets
181   testParent.set('name', 'New Parent Name');
182   equals(testParent.get('name'), 'New Parent Name', "set() should change name attribute");
183   testParent.set('nothing', 'nothing');
184   equals(testParent.get('nothing'), 'nothing', "set should change non-existent property to a new property");
185   
186   // Test Child Record creation
187   var oldCR = testParent.get('info');
188   testParent.set('info', {
189     type: 'ChildRecordTest',
190     name: 'New Child Name',
191     value: 'Red Goo',
192     guid: '6001'
193   });
194   var cr = testParent.get('info');
195   // Check Model Class information
196   ok(SC.kindOf(cr, SC.Record), "set() with an object creates an actual instance that is a kind of a SC.Record Object");
197   ok(SC.instanceOf(cr, NestedRecord.ChildRecordTest), "set() with an object creates an actual instance of a ChildRecordTest Object");
198   
199   // Check reference information
200   var pm = cr.get('primaryKey');
201   var key = cr.get(pm);
202   var storeRef = store.find(NestedRecord.ChildRecordTest, key);
203   ok(storeRef, 'after a set() with an object, checking that the store has the instance of the child record with proper primary key');
204   equals(cr, storeRef, "after a set with an object, checking the parent reference is the same as the direct store reference");
205   var oldKey = oldCR.get(pm);
206   ok(!(oldKey === key), 'check to see that the old child record has a different key from the new child record');
207   
208   // Check for changes on the child bubble to the parent.
209   cr.set('name', 'Child Name Change');
210   equals(cr.get('name'), 'Child Name Change', "after a set('name', <new>) on child, checking that the value is updated");
211 212   ok(cr.get('status') & SC.Record.DIRTY, 'check that the child record is dirty');
213   ok(testParent.get('status') & SC.Record.DIRTY, 'check that the parent record is dirty');
214   var newCR = testParent.get('info');
215   same(newCR, cr, "after a set('name', <new>) on child, checking to see that the parent has received the changes from the child record");
216   same(testParent.readAttribute('info'), cr.get('attributes'), "after a set('name', <new>) on child, readAttribute on the parent should be correct for info child attributes");
217 });

234   cr = testParent.get('info');
235   // Check Model Class information
236   ok(SC.kindOf(cr, SC.Record), "set() with an object creates an actual instance that is a kind of a SC.Record Object");
237   ok(SC.instanceOf(cr, NestedRecord.ChildRecordTest), "set() with an object creates an actual instance of a ChildRecordTest Object");
238   
239   // Check reference information
240   var pm = cr.get('primaryKey');
241   var key = cr.get(pm);
242   var storeRef = store.find(NestedRecord.ChildRecordTest, key);
243   ok(storeRef, 'after a set() with an object, checking that the store has the instance of the child record with proper primary key');
244   equals(cr, storeRef, "after a set with an object, checking the parent reference is the same as the direct store reference");
245   
246   // Check for changes on the child bubble to the parent.
247   cr.set('name', 'Child Name Change');
248   equals(cr.get('name'), 'Child Name Change', "after a set('name', <new>) on child, checking that the value is updated");

250   ok(testParent.get('status') & SC.Record.DIRTY, 'check that the parent record is dirty');
251   var newCR = testParent.get('info');
252   same(newCR, cr, "after a set('name', <new>) on child, checking to see that the parent has received the changes from the child record");
253   same(testParent.readAttribute('info'), cr.get('attributes'), "after a set('name', <new>) on child, readAttribute on the parent should be correct for info child attributes");
254 
255   // Make sure you can set the child to null.
256   testParent.set('info', null);
257   equals(testParent.get('info'), null, 'should be able to set child record to null');

260 test("Child Status Changed", function() {

269   SC.RunLoop.end();
270

315   equals(cr.get('status'), testParent.get('status'), 'after setting the parent to BUSY_REFRESH, check that the child record matches');
