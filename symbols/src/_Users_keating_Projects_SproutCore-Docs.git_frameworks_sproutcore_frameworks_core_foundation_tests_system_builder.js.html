 13 var obj; //global variables
 14 
 15 // Test cases for pushStack and end functions
 16 module("Builder",{
 17 	 
 18 	  setup: function(){
 19 	  	obj = [1,2,3,4];
 20 	    
 21 	}
 22 });
 23 
 24 test("To check if the set of array elements are pushed into stack",function(){
 25     
 26     var c = SC.Builder.fn.pushStack(obj);
 27     equals(4,obj.length = c.length,'No Of elements in the stack');
 28     equals(YES,obj[0]==c[0],'First element');
 29     equals(YES,obj[1]==c[1],'Second element'); 
 30     equals(YES,obj[2]==c[2],'Third element'); 
 31     equals(YES,obj[3]==c[3],'Fourth element'); 
 32 
 33     var d = SC.Builder.fn.end();
 34     equals(YES,SC.typeOf(d) == SC.T_HASH,'Previous item returned');   
 35 });
 36 
 37 test("create a new builder subclass with any passed properties copied to the builder's 'fn' property",function(){
 38 	obj =SC.Builder.create({ name : 'Charles'});
 39 	var fn_name =obj.fn.name;
 40 	equals(fn_name,"Charles","name should match");
 41 });
 42 
 43 test("instantiate the builder, any passed args will be forwarded onto an internal init() method",function(){
 44 	obj = new SC.Builder({name : 'Charles',age :23, sex :'M'});
 45 	var objA =obj.fn.init();
 46 	equals(objA.name,obj.fn.name);
 47 	equals(objA.age,obj.fn.age);
 48 	equals(objA.sex,obj.fn.sex);
 49 });
