var url,url1,url2;
 12 
 13 module("SC.normalizeURL");
 14 
 15 test("should normalize the url passed as the parameter",function(){
 16  url = '/desktop/mydocuments/music';
 17  equals(SC.normalizeURL(url), 'http://'+window.location.host+'/desktop/mydocuments/music','Path with slash');
 18  
 19  url1 = 'desktop/mydocuments/music';
 20  equals(SC.normalizeURL(url1), '%@/desktop/mydocuments/music'.fmt(window.location.href),'Path without slash');  
 21 
 22  url2 = 'http:';
 23  equals(YES,SC.normalizeURL(url2) === url2,'Path with http:');	
 24 });
