  9 var view ;
 10 module("SC.Control#displayProperties", {
 11   setup: function() {
 12     view = SC.View.create(SC.Control, {
 13         isVisibleInWindow: YES
 14     }).createLayer();
 15     view._doAttach(document.body);
 16   },
 17 
 18   teardown: function() {
 19     view.destroy();
 20   }
 21 });
 22 
 23 test("setting isSelected to YES adds sel class name", function() {
 24   SC.RunLoop.begin();
 25   view.set('isSelected', YES);
 26   SC.RunLoop.end();
 27   ok(view.$().hasClass('sel'), 'should have css class sel');
 28 });
 29 
 30 test("setting isSelected to SC.MIXED_STATE add mixed class name, and removes sel class name", function() {
 31   SC.RunLoop.begin();
 32   view.set('isSelected', SC.MIXED_STATE);
 33   SC.RunLoop.end();
 34   ok(view.$().hasClass('mixed'), 'should have css class mixed');
 35   ok(!view.$().hasClass('sel'), 'should NOT have css class sel');
 36 });
 37 
 38 test("setting isSelected to ON removes sel class name", function() {
 39   SC.RunLoop.begin();
 40   view.set('isSelected', YES);
 41   SC.RunLoop.end();
 42   ok(view.$().hasClass('sel'), 'precond - should have sel class');
 43 
 44   SC.RunLoop.begin();
 45   view.set('isSelected', NO);
 46   SC.RunLoop.end();
 47   ok(!view.$().hasClass('sel'), 'should no longer have sel class');
 48 });
 49 
 50 test("setting isEnabled to NO adds disabled class", function() {
 51   SC.RunLoop.begin();
 52   view.set('isEnabled', NO);
 53   SC.RunLoop.end();
 54   ok(view.$().hasClass('disabled'), 'should have disabled class');
 55 
 56   SC.RunLoop.begin();
 57   view.set('isEnabled', YES);
 58   SC.RunLoop.end();
 59   ok(!view.$().hasClass('disabled'), 'should remove disabled class');
 60 });
 61 
 62 test("should gain focus class if isFirstResponder", function() {
 63   SC.RunLoop.begin();
 64   view.set('isFirstResponder', YES);
 65   SC.RunLoop.end();
 66   ok(view.$().hasClass('focus'), 'should have focus class');
 67 
 68   SC.RunLoop.begin();
 69   view.set('isFirstResponder', NO);
 70   SC.RunLoop.end();
 71   ok(!view.$().hasClass('focus'), 'should remove focus class');
 72 });
 73 
 74 test("should gain active class if isActive", function() {
 75   SC.RunLoop.begin();
 76   view.set('isActive', YES);
 77   SC.RunLoop.end();
 78   ok(view.$().hasClass('active'), 'should have active class');
 79 
 80   SC.RunLoop.begin();
 81   view.set('isActive', NO);
 82   SC.RunLoop.end();
 83   ok(!view.$().hasClass('active'), 'should remove active class');
 84 });
 85 
 86 test("should get controlSize class on init", function() {
 87   ok(view.$().hasClass(SC.REGULAR_CONTROL_SIZE), 'should have regular control size class');
 88 });
