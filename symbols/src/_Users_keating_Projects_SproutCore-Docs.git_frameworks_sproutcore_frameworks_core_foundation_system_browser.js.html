  9 SC.mixin(SC.browser,
 10 /** @scope SC.browser */ {
 11 
 12   /**
 13     Version Strings should not be compared against Numbers.  For example,
 14     the version "1.20" is greater than "1.2" and less than "1.200", but as
 15     Numbers, they are all 1.2.
 16 
 17     Pass in one of the browser versions: SC.browser.version,
 18     SC.browser.engineVersion or SC.browser.osVersion and a String to compare
 19     against.  The function will split each version on the decimals and compare
 20     the parts numerically.
 21 
 22     Examples:
 23 
 24       SC.browser.compare('1.20', '1.2') == 18
 25       SC.browser.compare('1.08', '1.8') == 0
 26       SC.browser.compare('1.1.1', '1.1.004') == -3
 27 
 28     @param version {String} One of SC.browser.version, SC.browser.engineVersion or SC.browser.osVersion
 29     @param other {String} The version to compare against.
 30     @return {Number} The difference between the versions at the first difference.
 31   */
 32   compare: function(version, other) {
 33  34     var coerce,
 35         parts,
 36         tests;
 37 
 38     // Ensure that the versions are Strings.
 39     if (typeof version === 'number' || typeof other === 'number') {
 40       SC.warn('SC.browser.compare(): Versions compared against Numbers may not provide accurate results.  Use a String of decimal separated Numbers instead.')
 41       version = String(version);
 42       other = String(other);
 43     }
 44 
 45     // This function transforms the String to a Number or NaN
 46     coerce = function (part) {
 47       return Number(part.match(/^[0-9]+/));
 48     };
 49 
 50     parts = SC.A(version.split('.')).map(coerce);
 51     tests = SC.A(other.split('.')).map(coerce);
 52 
 53     // Test each part stopping when there is a difference.
 54     for (var i = 0; i < tests.length; i++) {
 55       var check = parts[i] - tests[i];
 56       if (isNaN(check)) return 0;
 57       if (check !== 0) return check;
 58     }
 59 
 60     return 0;
 61   },
 62 
 63   /** @deprecated Since 1.7. Use SC.browser.compare(version, otherVersion) instead.
 64 
 65     Pass any number of arguments, and this will check them against the browser
 66     version split on ".".  If any of them are not equal, return the inequality.
 67     If as many arguments as were passed in are equal, return 0.  If something
 68     is NaN, return 0.
 69   */
 70 
 71   // Deprecation Note:
 72   //
 73   // This function forces the comparison against the value of
 74   // SC.browser.version, but the old value of SC.browser.version would
 75   // occasionally be the browser's version or the layout engine's version,
 76   // which could cause unexpected results.  As well, there was no way to
 77   // compare the actual browser version or OS version.
 78   compareVersion: function () {
 79     SC.warn('SC.browser.compareVersion() has been deprecated.  Please ' +
 80         'use SC.browser.compare() instead.  Example: ' +
 81         'SC.browser.compareVersion(16,0,912) < 0 becomes ' +
 82         'SC.browser.compare(SC.browser.engineVersion, \'16.0.912\').');
 83 
 84     if (this._versionSplit === undefined) {
 85       var coerce = function (part) {
 86         return Number(part.match(/^[0-9]+/));
 87       };
 88       this._versionSplit = SC.A(this.version.split('.')).map(coerce);
 89     }
 90 
 91     var tests = SC.A(arguments).map(Number);
 92     for (var i = 0; i < tests.length; i++) {
 93       var check = this._versionSplit[i] - tests[i];
 94       if (isNaN(check)) return 0;
 95       if (check !== 0) return check;
 96     }
 97 
 98     return 0;
 99   }
100 
101 });
