 13 /**
 14   @class
 15 
 16   A response represents a single response from a server request.  An instance
 17   of this class is returned whenever you call SC.Request.send().
 18 
 19   @extend SC.Object
 20   @since SproutCore 1.0

 24 
 25   /**
 26     Walk like a duck
 27 
 28     @type Boolean
 29   */
 30   isResponse: YES,
 31 
 32   /**
 33     Becomes true if there was a failure.  Makes this into an error object.
 34 
 35     @type Boolean
 36     @default NO
 37   */
 38   isError: NO,
 39 
 40   /**
 41     Always the current response
 42 
 43     @field
 44     @type SC.Response
 45     @default `this`
 46   */
 47   errorValue: function() {
 48     return this;
 49   }.property().cacheable(),
 50 
 51   /**
 52     The error object generated when this becomes an error
 53 
 54     @type SC.Error
 55     @default null
 56   */
 57   errorObject: null,
 58 
 59   /**
 60     Request used to generate this response.  This is a copy of the original
 61     request object as you may have modified the original request object since
 62     then.
 63 
 64     To retrieve the original request object use originalRequest.
 65 
 66     @type SC.Request
 67     @default null
 68   */
 69   request: null,

 71   /**
 72     The request object that originated this request series.  Mostly this is

 76     @field
 77     @type SC.Request
 78     @observes request
 79   */

 87     Type of request. Must be an HTTP method. Based on the request.

118     return this.getPath('request.isJSON') || NO;
119   }.property('request').cacheable(),
120 
121   /**
122     If set, then will attempt to automatically parse response as XML
123     regardless of headers.
124 
125     @field
126     @type Boolean
127     @default NO
128     @observes request
129   */
130   isXML: function() {
131     return this.getPath('request.isXML') || NO ;
132   }.property('request').cacheable(),

139     @observes request

142     return this.getPath('request.listeners');
143   }.property('request').cacheable(),

161   /**
162     The response body or the parsed JSON. Returns a SC.Error instance
163     if there is a JSON parsing error. If isJSON was set, will be parsed
164     automatically.
165 
166     @field
167     @type {Hash|String|SC.Error}
168   */
169   body: function() {
170     // TODO: support XML
171     var ret = this.get('encodedBody');
172     if (ret && this.get('isJSON')) {
173       try {
174         ret = SC.json.decode(ret);
175       } catch(e) {
176         return SC.Error.create({
177           message: e.name + ': ' + e.message,
178           label: 'Response',
179           errorValue: this });
180       }
181     }
182     return ret;
183   }.property('encodedBody').cacheable(),
184

190

229   /**
230     Called by the request manager when its time to actually run. This will
231     invoke any callbacks on the source request then invoke transport() to
232     begin the actual request.
233   */
234   fire: function() {
235     var req = this.get('request'),
236         source = req ? req.get('source') : null;
237 
238     // first give the source a chance to fixup the request and response
239     // then freeze req so no more changes can happen.
240     if (source && source.willSend) { source.willSend(req, this); }
241     req.freeze();
242 
243     // if the source did not cancel the request, then invoke the transport
244     // to actually trigger the request.  This might receive a response
245     // immediately if it is synchronous.
246     if (!this.get('isCancelled')) { this.invokeTransport(); }
247 
248     // If the request specified a timeout value, then set a timer for it now.
249     var timeout = req.get('timeout');
250     if (timeout) {
251       var timer = SC.Timer.schedule({
252         target: this,
253         action: 'timeoutReached',
254         interval: timeout,
255         repeats: NO
256       });
257       this.set('timeoutTimer', timer);
258     }
259 
260     // if the transport did not cancel the request for some reason, let the
261     // source know that the request was sent
262     if (!this.get('isCancelled') && source && source.didSend) {
263       source.didSend(req, this);
264     }
265   },
266 
267   /**
268     Called by `SC.Response#fire()`. Starts the transport by invoking the
269     `SC.Response#receive()` function.
270   */
271   invokeTransport: function() {
272     this.receive(function(proceed) { this.set('status', 200); }, this);
273   },
274 
275   /**
276     Invoked by the transport when it receives a response. The passed-in
277     callback will be invoked to actually process the response. If cancelled
278     we will pass NO. You should clean up instead.
279 
280     Invokes callbacks on the source request also.
281 
282     @param {Function} callback the function to receive
283     @param {Object} context context to execute the callback in
284     @returns {SC.Response} receiver
285   */
286   receive: function(callback, context) {
287     if (!this.get('timedOut')) {
288       // If we had a timeout timer scheduled, invalidate it now.
289       var timer = this.get('timeoutTimer');
290       if (timer) { timer.invalidate(); }
291       this.set('timedOut', NO);
292     }
293 
294     var req = this.get('request');
295     var source = req ? req.get('source') : null;
296 
297     SC.run(function() {
298       // invoke the source, giving a chance to fixup the response or (more
299       // likely) cancel the request.
300       if (source && source.willReceive) { source.willReceive(req, this); }
301 
302       // invoke the callback.  note if the response was cancelled or not
303       callback.call(context, !this.get('isCancelled'));
304 
305       // if we weren't cancelled, then give the source first crack at handling
306       // the response.  if the source doesn't want listeners to be notified,
307       // it will cancel the response.
308       if (!this.get('isCancelled') && source && source.didReceive) {
309         source.didReceive(req, this);
310       }
311 
312       // notify listeners if we weren't cancelled.
313       if (!this.get('isCancelled')) { this.notify(); }
314     }, this);
315 
316     // no matter what, remove from inflight queue
317     SC.Request.manager.transportDidClose(this);
318     return this;
319   },
320 
321   /**
322     Default method just closes the connection. It will also mark the request
323     as cancelled, which will not call any listeners.
324   */
325   cancel: function() {
326     if (!this.get('isCancelled')) {
327       this.set('isCancelled', YES) ;
328       this.cancelTransport() ;
329       SC.Request.manager.transportDidClose(this) ;
330     }
331   },
332 
333   /**
334     Default method just closes the connection.
335 
336     @returns {Boolean} YES if this response has not timed out yet, NO otherwise
337   */
338   timeoutReached: function() {
339     // If we already received a response yet the timer still fired for some
340     // reason, do nothing.
341     if (this.get('timedOut') === null) {
342       this.set('timedOut', YES);
343       this.cancelTransport();
344 
345       // Invokes any relevant callbacks and notifies registered listeners, if
346       // any. In the event of a timeout, we set the status to 0 since we
347       // didn't actually get a response from the server.
348       this.receive(function(proceed) {
349         if (!proceed) { return; }
350 
351         // Set our value to an error.
352         var error = SC.$error("HTTP Request timed out", "Request", 0) ;
353         error.set("errorValue", this) ;
354         this.set('isError', YES);
355         this.set('errorObject', error);
356         this.set('status', 0);
357       }, this);
358 
359       return YES;
360     }
361 
362     return NO;
363   },
364 
365   /**
366     Override with concrete implementation to actually cancel the transport.
367   */
368   cancelTransport: function() {},
369 
370   /**
371     @private
372 
373     Will notify each listener. Returns true if any of the listeners handle.
374   */
375   _notifyListeners: function(listeners, status) {
376     var notifiers = listeners[status], params, target, action;
377     if (!notifiers) { return NO; }
378 
379     var handled = NO;
380     var len = notifiers.length;
381 
382     for (var i = 0; i < len; i++) {
383       var notifier = notifiers[i];
384       params = (notifier.params || []).copy();
385       params.unshift(this);
386 
387       target = notifier.target;
388       action = notifier.action;
389       if (SC.typeOf(action) === SC.T_STRING) { action = target[action]; }
390 
391       handled = action.apply(target, params);
392     }
393 
394     return handled;
395   },
396 
397   /**
398     Notifies any saved target/action. Call whenever you cancel, or end.
399 
400     @returns {SC.Response} receiver
401   */
402   notify: function() {
403     var listeners = this.get('listeners'),
404         status = this.get('status'),
405         baseStat = Math.floor(status / 100) * 100,
406         handled = NO;
407 
408     if (!listeners) { return this; }
409 
410     handled = this._notifyListeners(listeners, status);
411     if (!handled && baseStat !== status) { handled = this._notifyListeners(listeners, baseStat); }
412     if (!handled && status !== 0) { handled = this._notifyListeners(listeners, 0); }
413 
414     return this ;
415   },
416 
417   /**
418     String representation of the response object
419 
420     @returns {String}
421   */
422   toString: function() {
423     var ret = sc_super();
424     return "%@<%@ %@, status=%@".fmt(ret, this.get('type'), this.get('address'), this.get('status'));
425   }

429 /**
430   @class
431 
432   Concrete implementation of SC.Response that implements support for using
433   XHR requests.
434 
435   @extends SC.Response
436   @since SproutCore 1.0
437 */
438 SC.XHRResponse = SC.Response.extend(
439 /** @scope SC.XHRResponse.prototype */{
440 
441   /**
442     Implement transport-specific support for fetching all headers
443   */
444   headers: function() {
445     var xhr = this.get('rawRequest'),
446         str = xhr ? xhr.getAllResponseHeaders() : null,
447         ret = {};
448 
449     if (!str) { return ret; }
450 
451     str.split("\n").forEach(function(header) {
452       var idx = header.indexOf(':'),
453           key, value;
454 
455       if (idx >= 0) {
456         key = header.slice(0,idx);
457         value = header.slice(idx + 1).trim();
458         ret[key] = value;
459       }
460     }, this);
461 
462     return ret;
463   }.property('status').cacheable(),
464 
465   /**
466     Returns a header value if found.
467 
468     @param {String} key The header key
469     @returns {String}
470   */
471   header: function(key) {
472     var xhr = this.get('rawRequest');
473     return xhr ? xhr.getResponseHeader(key) : null;
474   },
475 
476   /**
477     Implement transport-specific support for fetching tasks
478 
479     @field
480     @type String
481     @default #rawRequest
482   */
483   encodedBody: function() {
484     var xhr = this.get('rawRequest');
485 
486     if (!xhr) { return null; }
487     if (this.get('isXML')) { return xhr.responseXML; }
488 
489     return xhr.responseText;
490   }.property('status').cacheable(),
491 
492   /**
493     Cancels the request.
494   */
495   cancelTransport: function() {
496     var rawRequest = this.get('rawRequest');
497     if (rawRequest) { rawRequest.abort(); }
498     this.set('rawRequest', null);
499   },
500 
501   /**
502     Starts the transport of the request
503 
504     @returns {XMLHttpRequest|ActiveXObject}
505   */
506   invokeTransport: function() {
507     var rawRequest, transport, handleReadyStateChange, async, headers;
508 
509     rawRequest = this.createRequest();
510     this.set('rawRequest', rawRequest);
511 
512     // configure async callback - differs per browser...
513     async = !!this.getPath('request.isAsynchronous');
514 
515     if (async) {
516       if (!SC.browser.isIE && !SC.browser.isOpera) {
517         SC.Event.add(rawRequest, 'readystatechange', this,
518                      this.finishRequest, rawRequest);
519       } else {
520         transport = this;
521         handleReadyStateChange = function() {
522           if (!transport) { return null; }
523           var ret = transport.finishRequest();
524           if (ret) { transport = null; }
525           return ret;
526         };
527         rawRequest.onreadystatechange = handleReadyStateChange;
528       }
529     }
530 
531     // initiate request.
532     rawRequest.open(this.get('type'), this.get('address'), async);
533 
534     // headers need to be set *after* the open call.
535     headers = this.getPath('request.headers');
536     for (var headerKey in headers) {
537       rawRequest.setRequestHeader(headerKey, headers[headerKey]) ;
538     }
539 
540     // now send the actual request body - for sync requests browser will
541     // block here
542     rawRequest.send(this.getPath('request.encodedBody')) ;
543     if (!async) { this.finishRequest(); }
544 
545     return rawRequest;
546   },
547 
548   /**
549     Creates the correct XMLHttpRequest object for this browser.
550 
551     You can override this if you need to, for example, create an XHR on a
552     different domain name from an iframe.
553 
554     @returns {XMLHttpRequest|ActiveXObject}
555   */
556   createRequest: function() {
557     function tryThese() {
558       for (var i=0; i < arguments.length; i++) {
559         try {
560           var item = arguments[i]();
561           return item;
562         } catch (e) {}
563       }
564       return NO;
565     }
566 
567     return tryThese(
568       function() { return new XMLHttpRequest(); },
569       function() { return new ActiveXObject('Msxml2.XMLHTTP'); },
570       function() { return new ActiveXObject('Microsoft.XMLHTTP'); }
571     );
572   },
573 
574   /**
575     @private
576 
577     Called by the XHR when it responds with some final results.
578 
579     @param {XMLHttpRequest} rawRequest the actual request
580     @returns {Boolean} YES if completed, NO otherwise
581   */
582   finishRequest: function(evt) {
583     var rawRequest = this.get('rawRequest'),
584         readyState = rawRequest.readyState,
585         error, status, msg;
586 
587     if (readyState === 4 && !this.get('timedOut')) {
588       this.receive(function(proceed) {
589         if (!proceed) { return; }
590 
591         // collect the status and decide if we're in an error state or not
592         status = -1;
593         try {
594           status = rawRequest.status || 0;
595         } catch (e) {}
596 
597         // if there was an error - setup error and save it
598         if ((status < 200) || (status >= 300)) {
599 
600           try {
601             msg = rawRequest.statusText || '';
602           } catch(e2) {
603             msg = '';
604           }
605 
606           error = SC.$error(msg || "HTTP Request failed", "Request", status);
607           error.set("errorValue", this) ;
608           this.set('isError', YES);
609           this.set('errorObject', error);
610         }
611 
612         // set the status - this will trigger changes on related properties
613         this.set('status', status);
614       }, this);
615 
616       // Avoid memory leaks
617       if (!SC.browser.isIE && !SC.browser.isOpera) {
618         SC.Event.remove(rawRequest, 'readystatechange', this, this.finishRequest);
619       } else {
620         rawRequest.onreadystatechange = null;
621       }
622 
623       return YES;
624     }
625     return NO;
626   }
