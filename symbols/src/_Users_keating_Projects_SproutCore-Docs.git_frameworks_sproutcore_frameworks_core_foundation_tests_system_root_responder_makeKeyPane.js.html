 10 test("returns receiver", function() {
 11   var p1 = SC.Pane.create(), p2 = SC.Pane.create({ acceptsKeyPane: YES });
 12   var r = SC.RootResponder.create();
 13   
 14   equals(r.makeKeyPane(p1), r, 'returns receiver even if pane does not accept key pane');
 15   equals(r.makeKeyPane(p2), r, 'returns receiver');

 18 test("changes keyPane to new pane if pane accepts key focus", function() {
 19   var p1 = SC.Pane.create({ acceptsKeyPane: NO }) ;
 20   var p2 = SC.Pane.create({ acceptsKeyPane: YES });
 21   var r = SC.RootResponder.create();
 22   
 23   r.makeKeyPane(p1);
 24   ok(r.get('keyPane') !== p1, 'keyPane should not change to view that does not accept key');
 25 
 26   r.makeKeyPane(p2);
 27   equals(r.get('keyPane'), p2, 'keyPane should change to view that does accept key');
 28   
 29 });

 31 test("setting nil sets key pane to mainPane if mainPane accepts key focus", function() {
 32   var main = SC.Pane.create({ acceptsKeyPane: YES });
 33   var key = SC.Pane.create({ acceptsKeyPane: YES });
 34   var r = SC.RootResponder.create({ mainPane: main, keyPane: key });
 35   
 36   // try to clear keyPane -- mainPane accepts key
 37   r.makeKeyPane(null);
 38   equals(r.get('keyPane'), main, 'keyPane should be main pane');
 39   
 40   r.makeKeyPane(key); // reset
 41   main.acceptsKeyPane = NO ;
 42   r.makeKeyPane(null); // try to clean - mainPane does not accept key
 43   equals(r.get('keyPane'), null, 'keyPane should be null, not main');
 44   
 45   // try another variety.  if keyPane is currently null and we try to set to
 46   // null do nothing, even if main DOES accept key.
 47   r.keyPane = null ;
 48   main.acceptsKeyPane = NO;
 49   r.makeKeyPane(null);
 50   equals(r.get('keyPane'), null, 'keyPane should remain null');
 51 });

 53 var p1, p2, r, callCount ;
 54 module("SC.RootResponder#makeKeyPane - testing notifications", {
 55   setup: function() {
 56     p1 = SC.Pane.create({ acceptsKeyPane: YES });    
 57     p2 = SC.Pane.create({ acceptsKeyPane: YES });    
 58     r = SC.RootResponder.create();
 59     callCount = 0 ;
 60   },
 61   
 62   teardown: function() { p1 = p2 = r ; }
 63 });
 64 
 65 test("should call willLoseKeyPaneTo on current keyPane", function() {
 66   r.keyPane = p1; //setup test
 67   p1.willLoseKeyPaneTo = function(pane) {
 68     equals(pane, p2, 'should pass new pane');
 69     callCount++;
 70   };
 71   
 72   r.makeKeyPane(p2);
 73   equals(callCount, 1, 'should invoke');
 74 });
 75 
 76 test("should call willBecomeKeyPaneFrom on new keyPane", function() {
 77   r.keyPane = p1; //setup test
 78   p2.willBecomeKeyPaneFrom = function(pane) {
 79     equals(pane, p1, 'should pass old pane');
 80     callCount++;
 81   };
 82   
 83   r.makeKeyPane(p2);
 84   equals(callCount, 1, 'should invoke');
 85 });

 88 test("should call didLoseKeyPaneTo on old keyPane", function() {

 91     equals(pane, p2, 'should pass new pane');
 92     callCount++;
 93   };
 94   
 95   r.makeKeyPane(p2);
 96   equals(callCount, 1, 'should invoke');
 97 });
 98 
 99 test("should call didBecomeKeyPaneFrom on new keyPane", function() {
100   r.keyPane = p1; //setup test
101   p2.didBecomeKeyPaneFrom = function(pane) {
102     equals(pane, p1, 'should pass old pane');
103     callCount++;

112   // instrument ...
