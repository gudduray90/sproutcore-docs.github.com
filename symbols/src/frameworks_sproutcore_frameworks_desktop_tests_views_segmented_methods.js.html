 10 var iconURL = sc_static("sproutcore-32.png");
 11 var pane, view;
 12 
 13 module("SC.SegmentedView", {
 14   setup: function() {
 15     SC.run(function () {
 16       pane = SC.MainPane.create({
 17         childViews: [
 18           SC.SegmentedView.extend({
 19             items: [
 20             { value: "Item1", icon: iconURL, width: 100 },
 21             { value: "Item2", icon: iconURL, width: 100 },
 22             { value: "Item3", icon: iconURL, width: 100 }],
 23             itemTitleKey: 'value',
 24             itemValueKey: 'value',
 25             itemIconKey: 'icon',
 26             itemWidthKey: 'width',
 27             itemActionKey: 'action',
 28             value: "Item1 Item3".w(),
 29             allowsEmptySelection: NO,
 30             layout: { height: 25, width: 400 }
 31           })]
 32       });
 33       pane.append(); // make sure there is a layer...
 34     });
 35 
 36     view = pane.childViews[0];
 37   },
 38 
 39   teardown: function() {
 40     SC.run(function () {
 41       pane.destroy();
 42       pane = view = null;
 43     });
 44   }
 45 });

 47 test("Check that properties are mapped correctly", function() {
 48   view.triggerItemAtIndex(1);
 49 
 50   SC.RunLoop.begin();
 51   view.set('isEnabled', YES);
 52   SC.RunLoop.end();
 53 
 54   equals(view.get('value'), "Item2", "the second item should be selected.");
 55 
 56   var childViews = view.get('childViews');
 57   equals(childViews[0].title, "Item1", 'Computed properties should match');
 58   equals(childViews[0].value, "Item1", 'Computed properties should match');
 59   equals(childViews[0].isEnabled, true, 'Computed properties should match');
 60   equals(childViews[0].icon, iconURL, 'Computed properties should match');
 61   equals(childViews[0].width, 100, 'Computed properties should match');
 62   equals(childViews[0].toolTip, null, 'Computed properties should match');
 63   equals(childViews[0].index, 0, 'Computed properties should match');
 64 });

 67 test("Check the values of value", function() {
 68   equals(SC.isArray(view.get('value')), true, "the value should initially be an array");
 69   equals(view.getPath('value.length'), 2, "the value array should have 2 items in it");
 70   view.triggerItemAtIndex(1);
 71   equals(SC.isArray(view.get('value')), false, "the value should not be an array if allowsMultipleSelection is false");
 72   equals(view.get('value'), "Item2", "the second item should be selected.");
 73 
 74   view.set('allowsMultipleSelection', true);
 75   view.triggerItemAtIndex(2);
 76   equals(SC.isArray(view.get('value')), true, "the value should be an array if allowsMultipleSelection is true");
 77   equals(view.getPath('value.length'), 2, "the value array should have 2 items in it");
 78   view.triggerItemAtIndex(1);
 79   equals(view.getPath('value.length'), 1, "the value array should have 1 item in it");
 80   view.triggerItemAtIndex(2);
 81   equals(view.getPath('value.length'), 1, "the value array should have 1 items in it, because allowsEmptySelection is false");
 82 
 83   view.set('allowsEmptySelection', true);
 84   view.triggerItemAtIndex(2);
 85   equals(view.getPath('value.length'), 0, "the value array should have 0 items in it,  because allowsEmptySelection is true");
 86 
 87   view.set('allowsMultipleSelection', false);
 88   view.triggerItemAtIndex(1);
 89   equals(SC.isArray(view.get('value')), false, "the value should not be an array if allowsMultipleSelection is false again");
 90   equals(view.get('value'), "Item2", "the second item should be selected.");
 91   view.triggerItemAtIndex(2);
 92   equals(view.get('value'), "Item3", "the third item should be selected.");
 93 
 94   view.set('allowsEmptySelection', false);
 95   view.triggerItemAtIndex(2);
 96   equals(view.get('value'), "Item3", "the third item should still be selected, because allowsEmptySelection is false");
 97 
 98   view.set('allowsEmptySelection', true);
 99   view.triggerItemAtIndex(2);
100   equals(view.get('value'), null, "the value should go to null if allowsMultipleSelection is false and allowsEmptySelection is true.");
101 });

103 // Test that if the clientX & clientY are over an element, that the proper index of that view is returned.
104 // Note: the test segmented view is centered by default and the items are only 50px wide.
105 test('displayItemIndexForEvent(evt)', function () {
106   var evt = {};
107 
108   // Check not over the view.
109   evt.target = view.get('layer');
110   evt.clientX = 500;
111   evt.clientY = 500;
112   equals(view.displayItemIndexForEvent(evt), -1, "Clicking at 500,500 which is not over the view element at all should return");
113 
114   // Check not over any child elements but in view at 0,0. This tests a special accessibility fix-up where events at 0,0
115   // are converted into a point over an element, because WebKit sends clientX & client& of 0,0 for accessibility events.
116   evt.target = view.get('layer');
117   evt.clientX = 0;
118   evt.clientY = 0;
119   equals(view.displayItemIndexForEvent(evt), -1, "Clicking at 0,0 which is not over any child elements, but is over the view element should return");
120 
121   // Check not over any child elements but in view at 1,1.
122   evt.clientX = 1;
123   evt.clientY = 1;
124   equals(view.displayItemIndexForEvent(evt), -1, "Clicking at 1,1 which is not over any child elements, but is over the view element should return");
125 
126   // Check over first child element.
127   evt.target = view.get('childViews')[0].get('layer');
128   evt.clientX = 51;
129   evt.clientY = 0;
130   equals(view.displayItemIndexForEvent(evt), 0, "Clicking at 0,51 which is over the first child element should return");
131 
132   // Check over second child element.
133   evt.target = view.get('childViews')[1].get('layer');
134   evt.clientX = 151;
135   evt.clientY = 0;
136   equals(view.displayItemIndexForEvent(evt), 1, "Clicking at 0,151 which is over the second child element should return");
137 
138   // Check over third child element.
139   evt.target = view.get('childViews')[2].get('layer');
140   evt.clientX = 251;
141   evt.clientY = 0;
142   equals(view.displayItemIndexForEvent(evt), 2, "Clicking at 0,251 which is over the third child element should return");
143 
144   // Check past third child element.
145   evt.target = view.get('layer');
146   evt.clientX = 351;
147   evt.clientY = 0;
148   equals(view.displayItemIndexForEvent(evt), -1, "Clicking at 0,351 which is past the third child element should return");
149 });
