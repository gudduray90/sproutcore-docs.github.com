 10 var iconURL = sc_static("sproutcore-32.png");
 11 var pane, view;
 12 
 13 module("SC.SegmentedView", {
 14   setup: function() {
 15     SC.RunLoop.begin();
 16     pane = SC.MainPane.create({
 17       childViews: [
 18         SC.SegmentedView.extend({
 19           items: [
 20           { value: "Item1", icon: iconURL },
 21           { value: "Item2", icon: iconURL },
 22           { value: "Item3", icon: iconURL }],
 23           itemTitleKey: 'value',
 24           itemValueKey: 'value',
 25           itemIconKey: 'icon',
 26           itemActionKey: 'action',
 27           value: "Item1 Item3".w(),
 28           allowsEmptySelection: NO,
 29           layout: { height: 25, width: 400 }
 30         })]
 31     });
 32     pane.append(); // make sure there is a layer...
 33     SC.RunLoop.end();
 34 
 35     view = pane.childViews[0];
 36   },
 37 
 38   teardown: function() {
 39     pane.remove();
 40     pane = view = null ;
 41   }
 42 });
 43 
 44 test("Check that properties are mapped correctly", function() {
 45   view.triggerItemAtIndex(1);
 46 
 47   SC.RunLoop.begin();
 48   view.set('isEnabled', YES);
 49   SC.RunLoop.end();
 50 
 51   equals(view.get('value'), "Item2", "the second item should be selected.");
 52 
 53   var childViews = view.get('childViews');
 54   equals(childViews[0].title, "Item1", 'Computed properties should match');
 55   equals(childViews[0].value, "Item1", 'Computed properties should match');
 56   equals(childViews[0].isEnabled, true, 'Computed properties should match');
 57   equals(childViews[0].icon, iconURL, 'Computed properties should match');
 58   equals(childViews[0].width, null, 'Computed properties should match');
 59   equals(childViews[0].toolTip, null, 'Computed properties should match');
 60   equals(childViews[0].index, 0, 'Computed properties should match');
 61 });
 62 
 63 
 64 test("Check the values of value", function() {
 65   equals(SC.isArray(view.get('value')), true, "the value should initially be an array");
 66   equals(view.getPath('value.length'), 2, "the value array should have 2 items in it");
 67   view.triggerItemAtIndex(1);
 68   equals(SC.isArray(view.get('value')), false, "the value should not be an array if allowsMultipleSelection is false");
 69   equals(view.get('value'), "Item2", "the second item should be selected.");
 70 
 71   view.set('allowsMultipleSelection', true);
 72   view.triggerItemAtIndex(2);
 73   equals(SC.isArray(view.get('value')), true, "the value should be an array if allowsMultipleSelection is true");
 74   equals(view.getPath('value.length'), 2, "the value array should have 2 items in it");
 75   view.triggerItemAtIndex(1);
 76   equals(view.getPath('value.length'), 1, "the value array should have 1 item in it");
 77   view.triggerItemAtIndex(2);
 78   equals(view.getPath('value.length'), 1, "the value array should have 1 items in it, because allowsEmptySelection is false");
 79 
 80   view.set('allowsEmptySelection', true);
 81   view.triggerItemAtIndex(2);
 82   equals(view.getPath('value.length'), 0, "the value array should have 0 items in it,  because allowsEmptySelection is true");
 83 
 84   view.set('allowsMultipleSelection', false);
 85   view.triggerItemAtIndex(1);
 86   equals(SC.isArray(view.get('value')), false, "the value should not be an array if allowsMultipleSelection is false again");
 87   equals(view.get('value'), "Item2", "the second item should be selected.");
 88   view.triggerItemAtIndex(2);
 89   equals(view.get('value'), "Item3", "the third item should be selected.");
 90 
 91   view.set('allowsEmptySelection', false);
 92   view.triggerItemAtIndex(2);
 93   equals(view.get('value'), "Item3", "the third item should still be selected, because allowsEmptySelection is false");
 94 
 95   view.set('allowsEmptySelection', true);
 96   view.triggerItemAtIndex(2);
 97   equals(view.get('value'), null, "the value should go to null if allowsMultipleSelection is false and allowsEmptySelection is true.");
 98 });
