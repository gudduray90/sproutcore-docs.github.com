  5 /*globals Greenhouse */
  6 /** @class
  7 
  8   provides access to the iframes memoryspace
  9   @extends SC.WebView
 10 */
 11 Greenhouse.WebView = SC.WebView.extend(
 12 /** @scope Greenhouse.WebView.prototype */ {
 13   
 14   iframeDidLoad: function() {
 15     //fit the iframe to size of the contents.
 16     if (this.get('shouldAutoResize') === YES){
 17       var contentWindow;
 18       var iframeElt = this.$('iframe')[0];
 19       if(iframeElt && iframeElt.contentWindow){
 20         contentWindow = iframeElt.contentWindow;
 21         this.contentWindow = contentWindow;
 22         if(contentWindow && contentWindow.document && contentWindow.document.documentElement){
 23           var docElement = contentWindow.document.documentElement;
 24           // setting the width before the height gives more accurate results.. 
 25           // at least for the test iframe content i'm using.
 26           //TODO: try out document flows other than top to bottom.
 27           if (!SC.browser.isIE){
 28             this.$().width(docElement.scrollWidth);
 29             this.$().height(docElement.scrollHeight);          
 30           } else {
 31             this.$().width(docElement.scrollWidth + 12);
 32             this.$().height(docElement.scrollHeight + 5);          
 33           }
 34         }
 35       }
 36     }
 37     else{
 38       var iframe = this.$('iframe')[0];
 39       if(iframe) this.contentWindow = iframe.contentWindow;
 40     }
 41     Greenhouse.set('iframe', this.contentWindow);
 42     Greenhouse.sendAction('iframeLoaded');
 43   }
 44 });
