 68 pane.verifyEmpty = function verifyEmpty(view, expectedHint) {
 69   var input = view.$('input');
 70   var layer = view.$();
 71   
 72   ok(!layer.hasClass('not-empty'), 'layer should not have not-empty class');
 73   if(SC.browser.webkit || parseInt(SC.browser.mozilla) >= 2.0) equals(input.val(), '', 'input should have empty value');
 74   else equals(input.val(), expectedHint, 'input should have empty value');
 75   
 76   if (expectedHint) {
 77     var hint = view.$('.sc-hint');
 78 
 79     if (hint.length===1) {
 80       hint = hint.text();
 81     } else {
 82       hint = view.$('input');
 83       hint = hint.attr('placeholder');
 84     }
 85     equals(hint, expectedHint, 'hint span should have expected hint');
 86   }
 87 };
 88 
 89 pane.verifyNotEmpty = function verifyNotEmpty(view, expectedValue, expectedHint) {
 90   var input = view.$('input');
 91   var layer = view.$();
 92   
 93   ok(layer.hasClass('not-empty'), 'layer should have not-empty class');
 94   equals(input.val(), expectedValue, 'input should have value');
 95   
 96   if (expectedHint) {
 97     var hint = view.$('.sc-hint');
 98 
 99     if (hint.length===1) {
100       hint = hint.text();
101     } else {
102       hint = view.$('input');
103       hint = hint.attr('placeholder');
104     }
105     equals(hint, expectedHint, 'hint span should have expected hint');
106   }
107 };
108 
109 pane.verifyDisabled = function verifyDisabled(view, isDisabled) {
110   var layer = view.$();
111   var input = view.$('input');
112   
113   if (isDisabled) {
114     ok(layer.hasClass('disabled'), 'layer should have disabled class');
115     ok(input.attr('disabled'), 'input should have disabled attr');
116   } else {
117     ok(!layer.hasClass('disabled'), 'layer should not have disabled class');
118     ok(!input.attr('disabled'), 'input should not have disabled attr');
119   }
120 };

282   var view = pane.view('with value');
283   
284   // test adding accessory view adds the view like it should
285   SC.RunLoop.begin();
286   var accessoryView = SC.View.create({

288   });

302   
303   // Test removing the accessory view.
304   SC.RunLoop.begin();
305   view.set('leftAccessoryView', null);

357 test("Adding both left and right accessory views", function() {  
358   var view = pane.view('with value');
359   
360   // test adding accessory view adds the view like it should
361   SC.RunLoop.begin();
362   var leftAccessoryView = SC.View.create({
363     layout:  { top:1, left:2, width:16, height:16 }
364   });
365   view.set('leftAccessoryView', leftAccessoryView);
366   var rightAccessoryView = SC.View.create({
367     layout:  { top:1, right:3, width:17, height:16 }
368   });
369   view.set('rightAccessoryView', rightAccessoryView);
370   SC.RunLoop.end();
371 
372   ok(view.get('childViews').length === 2, 'we should have two child views since we added both a left and a right accessory view');
373   
374   
375   // The hint and padding elements should automatically have their 'left' and
376   // 'right' values set to the accessory views' offset + width
377   //   *  left:   18 = 2 left offset + 16 width)
378   //   *  right:  20 = 3 left offset + 17 width)
379   var paddingElement = view.$('.padding')[0];
380   ok(paddingElement.style.left === '18px', 'padding element should get 18px left');
381   ok(paddingElement.style.right === '20px', 'padding element should get 20px right');
382   
383   
384   // Test removing the accessory views.
385   SC.RunLoop.begin();
386   view.set('rightAccessoryView', null);
387   SC.RunLoop.end();
388   ok(view.get('childViews').length === 1, 'after removing the right accessory view there should be one child view left (the left accessory view)');
389   ok(!paddingElement.style.right, 'after removing the right accessory view the padding element should have no right style');
390   SC.RunLoop.begin();
391   view.set('leftAccessoryView', null);
392   SC.RunLoop.end();
393   ok(view.get('childViews').length === 0, 'after removing both accessory views there should be no child views left');
394   ok(!paddingElement.style.left, 'after removing the left accessory view the padding element should have no left style');
395 });
