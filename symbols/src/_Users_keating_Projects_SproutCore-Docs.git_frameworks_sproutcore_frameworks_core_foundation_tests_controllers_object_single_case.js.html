  8 var content, controller;

 11 // SINGLE OBSERVABLE OBJECT

 13 
 14 module("SC.ObjectController - single_case - OBSERVABLE OBJECT", {
 15   setup: function() {
 16     content = SC.Object.create({ foo: "foo1", bar: "bar1" });
 17     controller = SC.ObjectController.create({ content: content });
 18   },
 19   
 20   teardown: function() {
 21     controller.destroy();
 22   }
 23 });
 24 
 25 test("getting any unknown value should pass through to object", function() {
 26   equals(controller.get("foo"), "foo1", 'controller.get(foo)');
 27   equals(controller.get("bar"), "bar1", 'controller.get(bar)');
 28 });
 29 
 30 test("setting any unknown value should pass through", function() {
 31   equals(controller.set("foo", "EDIT"), controller, 'controller.set(foo, EDIT) should return self');  
 32   equals(controller.set("bar", "EDIT"), controller, 'controller.set(bar, EDIT) should return self');
 33   equals(controller.set("baz", "EDIT"), controller, 'controller.set(baz, EDIT) should return self');
 34   
 35   equals(content.get("foo"), "EDIT", 'controller.get(foo)');
 36   equals(content.get("bar"), "EDIT", 'controller.get(bar)');
 37   equals(content.get("baz"), "EDIT", 'controller.get(bar)');
 38 });
 39 
 40 test("changing a property on the content", function() {
 41   var callCount = 0;
 42   controller.addObserver("foo", function() { callCount++; });
 43 
 44   equals(controller.get("foo"), "foo1", "controller.get(foo) before edit should have original value");
 45 
 46   content.set("foo", "EDIT");
 47   equals(controller.get("foo"), "EDIT", "controller.get(foo) after edit should have updated value");
 48   equals(callCount, 1, 'observer on controller should have fired');
 49 });
 50 
 51 test("changing the content from one to another", function() {
 52   var callCount = 0 ;
 53   var content2 = SC.Object.create({ foo: "foo2", bar: "bar2" });
 54   controller.addObserver("foo", function() { callCount++; });
 55 
 56   equals(controller.get("foo"), "foo1", "controller.get(foo) before edit should have original value");
 57 
 58   controller.set("content", content2);
 59 
 60   equals(controller.get("foo"), "foo2", "controller.get(foo) after content should contain new content");
 61   equals(callCount, 1, 'observer on controller should have fired');
 62 
 63   callCount = 0 ;
 64   content2.set("foo", "EDIT");
 65   equals(controller.get("foo"), "EDIT", "controller.get(foo) after edit should have updated value");
 66   equals(callCount, 1, 'observer on controller should have fired');
 67   
 68   callCount = 0;
 69   content.set("foo", "BAR");
 70   equals(controller.get("foo"), "EDIT", "controller.get(foo) after edit of non-content object should not change value");
 71   equals(callCount, 0, 'observer on controller should NOT have fired');
 72 });
 73 
 74 test("changing the content from one single to null and back", function() {
 75   var callCount = 0 ;
 76   controller.addObserver("foo", function() { callCount++; });
 77 
 78   equals(controller.get("foo"), "foo1", "controller.get(foo) before edit should have original value");
 79 
 80   controller.set("content", null);
 81 
 82   equals(controller.get("foo"), undefined, "controller.get(foo) after content change should be empty");
 83   equals(callCount, 1, 'observer on controller should have fired');
 84 
 85   callCount = 0;
 86   content.set("foo", "BAR");
 87   equals(controller.get("foo"), undefined, "controller.get(foo) after edit of non-content object should not change value");
 88   equals(callCount, 0, 'observer on controller should NOT have fired');
 89 
 90   callCount = 0 ;
 91   controller.set("content", content);
 92   equals(callCount, 1, 'observer on controller should have fired');
 93 
 94   callCount = 0 ;
 95   content.set("foo", "EDIT");
 96   equals(controller.get("foo"), "EDIT", "controller.get(foo) after edit should have updated value");
 97   equals(callCount, 1, 'observer on controller should have fired');
 98   
 99 });
100 
101 test("hasContent", function() {
102   equals(controller.get("hasContent"), YES, 'should have content');
103   
104   var callCount = 0;
105   controller.addObserver("hasContent", function() { callCount++; });
106   
107   controller.set("content", null);
108   equals(controller.get("hasContent"), NO, "hasContent should == NO after setting to null");
109   ok(callCount > 0, 'hasContent observer should fire when setting to null');
110   
111   callCount = 0;
112   controller.set("content", content);
113   equals(controller.get("hasContent"), YES, "hasContent should == YES after setting back to content");
114   ok(callCount > 0, "hasContent observer should fire");
115 });
