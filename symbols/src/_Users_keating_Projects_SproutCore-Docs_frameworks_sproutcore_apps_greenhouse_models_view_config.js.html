 14 Greenhouse.ViewConfig = SC.Record.extend(
 15 /** @scope Greenhouse.ViewConfig.prototype */ {
 16 
 17   primaryKey: 'path',
 18 
 19   views: SC.Record.toMany('Greenhouse.Design', {nested: YES}),
 20   panes: SC.Record.toMany('Greenhouse.Design', {nested: YES}),
 21   controllers: SC.Record.toMany('Greenhouse.Design', {nested: YES}),
 22   canEdit: SC.Record.attr(Boolean),
 23   name: SC.Record.attr(String),
 24   path: SC.Record.attr(String),
 25 
 26   body: function(){
 27     var ret = {name: this.get('name'), path: this.get('path'), views: [], controllers: [], panes: []},
 28         views = this.get('views'),
 29         controllers = this.get('controllers'),
 30         panes = this.get('panes');
 31 
 32 
 33     views.forEach(function(i){
 34       ret.views.push(i.get('attributes'));
 35     });
 36 
 37     controllers.forEach(function(i){
 38       ret.controllers.push(i.get('attributes'));
 39     });
 40 
 41     panes.forEach(function(i){
 42       ret.panes.push(i.get('attributes'));
 43     });
 44 
 45     return js_beautify(SC.json.encode(ret));
 46   }.property('views', 'panes', 'controllers')
 47 
 48 }) ;
 49 Greenhouse.CONFIG_QUERY = SC.Query.remote(Greenhouse.ViewConfig);
