262   /**

267     @returns {SC.Request} new request

270     var ret = {},

272         loc = keys.length,
273         key, listeners, headers;
274 
275     while(--loc >= 0) {
276       key = keys[loc];
277       if (this.hasOwnProperty(key)) {

279       }
280     }
281 
282     if (this.hasOwnProperty('listeners')) {

295   /**
296     To set headers on the request object. Pass either a single key/value
297     pair or a hash of key/value pairs. If you pass only a header name, this
298     will return the current value of the header.
299 
300     @param {String|Hash} key
301     @param {String} value
302     @returns {SC.Request|Object} receiver
303   */
304   header: function(key, value) {
305     var header, headers;
306 
307     if (SC.typeOf(key) === SC.T_STRING) {
308       headers = this._headers;
309       if (arguments.length === 1) {
310         return headers ? headers[key] : null;
311       } else {
312         this.propertyWillChange('headers');
313         if (!headers) { headers = this._headers = {}; }
314         headers[key] = value;
315         this.propertyDidChange('headers');
316         return this;
317       }
318 
319     // handle parsing hash of parameters
320     } else if (value === undefined) {
321       headers = key;
322       this.beginPropertyChanges();
323       for(header in headers) {
324         if (!headers.hasOwnProperty(header)) { continue; }
325         this.header(header, headers[header]);
326       }
327       this.endPropertyChanges();
328       return this;
329     }
330 
331     return this;
332   },

401   /**
402     Called just before a request is enqueued.  This will encode the body
403     into JSON if it is not already encoded, and set identifying headers
404   */
405   _prep: function() {
406     var hasContentType = !!this.header('Content-Type');
407 
408     if (this.get('attachIdentifyingHeaders')) {
409       this.header('X-Requested-With', 'XMLHttpRequest');
410       this.header('X-SproutCore-Version', SC.VERSION);
411     }
412 
413     // Set the Content-Type header only if not specified and the request
414     // includes a body.
415     if (!hasContentType && !!this.get('body')) {
416       if (this.get('isJSON')) {
417         this.header('Content-Type', 'application/json');
418       } else if (this.get('isXML')) {
419         this.header('Content-Type', 'text/xml');
420       }
421     }
422     return this;
423   },

425   /**
426     Will fire the actual request. If you have set the request to use JSON
427     mode then you can pass any object that can be converted to JSON as the
428     body. Otherwise you should pass a string body.
429 
430     @param {String|Object} [body]
431     @returns {SC.Response} New response object
432   */
433   send: function(body) {
434     // Sanity-check: Be sure a timeout value was not specified if the request
435     // is synchronous (because it wouldn't work).
436     var timeout = this.get('timeout');
437     if (timeout && !this.get('isAsynchronous')) {
438       throw new Error("Timeout values cannot be used with synchronous requests");
439     } else if (timeout === 0) {
440       throw new Error("The timeout value must either not be specified or must be greater than 0");
441     }
442 
443     if (body) { this.set('body', body); }
444     return SC.Request.manager.sendRequest(this.copy()._prep());
445   },

459   /**
460     Configures a callback to execute as a request progresses or completes. You
461     must pass at least a target and action/method to this and optionally an
462     event name or status code.
463 
464     You may also pass additional arguments which will then be passed along to
465     your callback.
466 
467     ## Scoping With Status Codes
468 
469     If you pass a status code as the first argument to this method, the
470     accompanying notification callback will only be called if the response
471     status matches the status code. For example, if you pass 201 (or
472     SC.Request.CREATED), the accompanying method will only be called if the
473     response status from the server is also 201.
474 
475     You can also pass "generic" status codes such as 200, 300, or 400, which
476     will be invoked anytime the status code is in the same range and if a more
477     specific notifier was not registered first and returned YES.
478 
479     Finally, passing a status code of 0 or no status at all will cause your
480     method to be executed no matter what the resulting status is unless a
481     more specific notifier was registered first and returned YES.
482 
483     For example,
484 
485         var req = SC.Request.create({type: 'POST'});
486         req.notify(201, this, this.reqWasCreated);  // Handle a specific status code
487         req.notify(401, this, this.reqWasUnauthorized);  // Handle a specific status code
488         req.notify(400, this, this.reqDidRedirect);  // Handle any 4xx status
489         req.notify(this, function(response, arg1, arg2) {
490           // do something
491         }, arg1, arg2);  // Handle any status.  Also, passing additional arguments to the callback handler
492 
493     ## Notifying on Progress Events
494 
495     If you pass a progress event name your callback will be called each time
496     the event fires on the response.  For example, the XMLHttpRequest Level 2
497     specification defines several progress events: loadstart, progress, abort,
498     error, load, timeout and loadend.  Therefore, when using the default
499     SC.Request responseClass, SC.XHRResponse, you can be notified of each of
500     these events by simply providing the matching event name.
501 
502     Note that many older browsers do not support XMLHttpRequest Level 2.  See
503     http://caniuse.com/xhr2 for a list of supported browsers.
504 
505     For example,
506 
507       var req = SC.Request.create({type: 'GET'});
508       req.notify('progress', this, this.reqDidProgress); // Handle 'progress' events
509       req.notify('abort', this, this.reqDidAbort); // Handle 'abort' events
510       req.notify('upload.progress', this, this.reqUploadDidProgress); // Handle 'progress' events on the XMLHttpRequestUpload
511       req.send();
512 
513     ## Callback Format
514 
515     Your notification callback should expect to receive the Response object as
516     the first parameter for status code notifications and the Event object for
517     progress notifications; plus any additional parameters that you pass. If
518     your callback handles the notification and to prevent further handling, it
519     should return YES.
520 
521     @param [statusOrEvent] {Number|String} A Number status code or String Event name.
522     @param target {Object} The target object for the callback action.
523     @param action {String|Function} The method name or function to call on the target.
524     @returns {SC.Request} The SC.Request object.
525   */
526   notify: function(statusOrEvent, target, action) {
527     var args;
528 
529     //@if (debug)
530     if (statusOrEvent === 'loadend' && SC.Request.WARN_ON_LOADEND) {
531       SC.warn("Developer Warning: You have called SC.Request#notify for the 'loadend' event. Note that on certain platforms, like older iPads, loadend is not supported and this notification will fail silently. You can protect against this by checking SC.platform.get('supportsXHR2LoadEndEvent'), and attaching listeners for load, error and abort instead. (This is not done automatically because your code may need to handle event type and fire order considerations.) To suppress this warning, set SC.Request.WARN_ON_LOADEND to NO.");
532     }
533     //@endif
534 
535     // Normalize arguments
536     if (SC.typeOf(statusOrEvent) !== SC.T_NUMBER && SC.typeOf(statusOrEvent) !== SC.T_STRING) {
537       // Accept multiple additional arguments (Do so before shifting the arguments!)
538       args = SC.A(arguments).slice(2);
539 
540       // Shift the arguments
541       action = target;
542       target = statusOrEvent;
543       statusOrEvent = 0;
544     } else {
545       // Accept multiple additional arguments.
546       args = SC.A(arguments).slice(3);
547     }
548 
549     // Prepare listeners for this object and notification target.
550     var listeners = this.get('listeners');
551     if (!listeners) { this.set('listeners', listeners = {}); }
552     if(!listeners[statusOrEvent]) { listeners[statusOrEvent] = []; }
553 
554     // Add another listener for the given status code or event name.
555     listeners[statusOrEvent].push({target: target, action: action, args: args});
556 
557     return this;
558   }

629 /**
630   @class
631 
632   The request manager coordinates all of the active XHR requests. It will
633   only allow a certain number of requests to be active at a time; queuing
634   any others. This allows you more precise control over which requests load
635   in which order.
636 
637   @since SproutCore 1.0
638 */
639 SC.Request.manager = SC.Object.create(
640 /** @scope SC.Request.manager */{
641 
642   /**
643     Maximum number of concurrent requests allowed. 6 for all browsers.
644 
645     @type Number
646     @default 6
647   */
648   maxRequests: 6,
649 
650   /**
651     Current requests that are inflight.
652 
653     @type Array
654     @default []
655   */
656   inflight: [],
657 
658   /**
659     Requests that are pending and have not been started yet.
660 
661     @type Array
662     @default []
663   */
664   pending: [],
665 
666 
667   // ..........................................................
668   // METHODS
669   //
670 
671   /**
672     Invoked by the send() method on a request. This will create a new low-
673     level transport object and queue it if needed.
674 
675     @param {SC.Request} request the request to send
676     @returns {SC.Object} response object
677   */
678   sendRequest: function(request) {
679     if (!request) { return null; }
680 
681     // create low-level transport.  copy all critical data for request over
682     // so that if the request has been reconfigured the transport will still
683     // work.
684     var response = request.get('responseClass').create({ request: request });
685 
686     // add to pending queue
687     this.get('pending').pushObject(response);
688     this.fireRequestIfNeeded();
689 
690     return response;
691   },
692 
693   /**
694     Cancels a specific request. If the request is pending it will simply
695     be removed. Otherwise it will actually be cancelled.
696 
697     @param {SC.Response} response a response object
698     @returns {Boolean} YES if cancelled
699   */
700   cancel: function(response) {
701     var pending = this.get('pending'),
702         inflight = this.get('inflight'),
703         idx;
704 
705     if (pending.indexOf(response) >= 0) {
706       this.propertyWillChange('pending');
707       pending.removeObject(response);
708       this.propertyDidChange('pending');
709       return YES;
710     } else if (inflight.indexOf(response) >= 0) {
711       response.cancel();
712 
713       inflight.removeObject(response);
714       this.fireRequestIfNeeded();
715       return YES;
716     }
717 
718     return NO;
719   },
720 
721   /**
722     Cancels all inflight and pending requests.
723 
724     @returns {Boolean} YES if any items were cancelled.
725   */
726   cancelAll: function() {
727     var pendingLen = this.getPath('pending.length'),
728       inflightLen = this.getPath('inflight.length'),
729       inflight = this.get('inflight'),
730       pending = this.get('pending');
731 
732     if(pendingLen || inflightLen) {
733       // Iterate backwards.
734       for( var i = inflightLen - 1; i >= 0; i--) {
735         // This will 'eventually' try to remove the request from
736         // inflight, but it's not fast enough for us.
737         var r = inflight.objectAt(i);
738         r.cancel();
739       }
740 
741       // Manually scrub the arrays without screwing up memory pointers.
742       pending.replace(0, pendingLen);
743       inflight.replace(0, inflightLen);
744 
745       return YES;
746 
747     }
748     return NO;
749   },
750 
751   /**
752     Checks the inflight queue. If there is an open slot, this will move a
753     request from pending to inflight.
754 
755     @returns {Object} receiver
756   */
757   fireRequestIfNeeded: function() {
758     var pending = this.get('pending'),
759         inflight = this.get('inflight'),
760         max = this.get('maxRequests'),
761         next;
762 
763     if ((pending.length>0) && (inflight.length<max)) {
764       next = pending.shiftObject();
765       inflight.pushObject(next);
766       next.fire();
767     }
768   },
