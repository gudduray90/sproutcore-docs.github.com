 18   /** @private */
 19   _testSupportFor: function (target, propertyName, testValue) {
 20     /*jshint eqnull:true*/
 21     var ret = target[propertyName] != null,
 22       originalValue;
 23 
 24     if (testValue != null) {
 25       originalValue = target[propertyName];
 26       target[propertyName] = testValue;
 27       ret = target[propertyName] === testValue;
 28       target[propertyName] = originalValue;
 29     }
 30 
 31     return ret;
 32   },

 34   /**
 35     Version Strings should not be compared against Numbers.  For example,
 36     the version "1.20" is greater than "1.2" and less than "1.200", but as
 37     Numbers, they are all 1.2.
 38 
 39     Pass in one of the browser versions: SC.browser.version,
 40     SC.browser.engineVersion or SC.browser.osVersion and a String to compare
 41     against.  The function will split each version on the decimals and compare
 42     the parts numerically.
 43 
 44     Examples:
 45 
 46       SC.browser.compare('1.20', '1.2') == 18
 47       SC.browser.compare('1.08', '1.8') == 0
 48       SC.browser.compare('1.1.1', '1.1.004') == -3
 49 
 50     @param {String} version One of SC.browser.version, SC.browser.engineVersion or SC.browser.osVersion
 51     @param {String} other The version to compare against.
 52     @returns {Number} The difference between the versions at the first difference.
 53   */
 54   compare: function (version, other) {
 55     var coerce,
 56         parts,
 57         tests;
 58 
 59     // Ensure that the versions are Strings.
 60     if (typeof version === 'number' || typeof other === 'number') {
 61       //@if(debug)
 62       SC.warn('Developer Warning: SC.browser.compare(): Versions compared against Numbers may not provide accurate results.  Use a String of decimal separated Numbers instead.');
 63       //@endif
 64       version = String(version);
 65       other = String(other);
 66     }
 67 
 68     // This function transforms the String to a Number or NaN
 69     coerce = function (part) {
 70       return Number(part.match(/^[0-9]+/));
 71     };
 72 
 73     parts = SC.A(version.split('.')).map(coerce);
 74     tests = SC.A(other.split('.')).map(coerce);
 75 
 76     // Test each part stopping when there is a difference.
 77     for (var i = 0; i < tests.length; i++) {
 78       var check = parts[i] - tests[i];
 79       if (isNaN(check)) return 0;
 80       if (check !== 0) return check;
 81     }
 82 
 83     return 0;
 84   },

 87     @deprecated Since 1.7. Use SC.browser.compare(version, otherVersion) instead.

 89     Pass any number of arguments, and this will check them against the browser
 90     version split on ".".  If any of them are not equal, return the inequality.
 91     If as many arguments as were passed in are equal, return 0.  If something
 92     is NaN, return 0.

 95   // Deprecation Note:

 98   // SC.browser.version, but the old value of SC.browser.version would
 99   // occasionally be the browser's version or the layout engine's version,
100   // which could cause unexpected results.  As well, there was no way to

102   compareVersion: function () {
103     //@if(debug)
104     SC.warn('Developer Warning: SC.browser.compareVersion() has been deprecated.  Please ' +
105         'use SC.browser.compare() instead.  Example: ' +
106         'SC.browser.compareVersion(16,0,912) < 0 becomes ' +
107         'SC.browser.compare(SC.browser.engineVersion, \'16.0.912\').');
108     //@endif
109 
110     if (this._versionSplit === undefined) {
111       var coerce = function (part) {
112         return Number(part.match(/^[0-9]+/));
113       };
114       this._versionSplit = SC.A(this.version.split('.')).map(coerce);
115     }
116 
117     var tests = SC.A(arguments).map(Number);
118     for (var i = 0; i < tests.length; i++) {
119       var check = this._versionSplit[i] - tests[i];
120       if (isNaN(check)) return 0;
121       if (check !== 0) return check;
122     }
123 
124     return 0;

128   /**
129     This simple method allows you to more safely use experimental properties and
130     methods in current and future browsers.
131 
132     Using browser specific methods and properties is a risky coding practice.
133     With sufficient testing, you may be able to match prefixes to today's
134     browsers, but this is prone to error and not future proof.  For instance,
135     if a property becomes standard and the browser drops the prefix, your code
136     could suddenly stop working.
137 
138     Instead, use SC.browser.experimentalNameFor(target, standardName), which
139     will check the existence of the standard name on the target and if not found
140     will try different camel-cased versions of the name with the current
141     browser's prefix appended.
142 
143     If it is still not found, SC.UNSUPPORTED will be returned, allowing
144     you a chance to recover from the lack of browser support.
145 
146     Note that `experimentalNameFor` is not really meant for determining browser
147     support, only to ensure that using browser prefixed properties and methods
148     is safe.  Instead, SC.platform provides several properties that can be used
149     to determine support for a certain platform feature, which should be
150     used before calling `experimentalNameFor` to safely use the feature.
151 
152     For example,
153 
154         // Checks for IndexedDB support first on the current platform.
155         if (SC.platform.supportsIndexedDB) {
156           var db = window.indexedDB,
157             // Example return values: 'getDatabaseNames', 'webkitGetDatabaseNames', 'MozGetDatabaseNames', SC.UNSUPPORTED.
158             getNamesMethod = SC.browser.experimentalNameFor(db, 'getDatabaseNames'),
159             names;
160 
161             if (getNamesMethod === SC.UNSUPPORTED) {
162               // Work without it.
163             } else {
164               names = db[getNamesMethod](...);
165             }
166         } else {
167           // Work without it.
168         }
169 
170     ## Improving deduction
171     Occasionally a target will appear to support a property, but will fail to
172     actually accept a value.  In order to ensure that the property doesn't just
173     exist but is also usable, you can provide an optional `testValue` that will
174     be temporarily assigned to the target to verify that the detected property
175     is usable.
176 
177     @param {Object} target The target for the method.
178     @param {String} standardName The standard name of the property or method we wish to check on the target.
179     @param {String} [testValue] A value to temporarily assign to the property.
180     @returns {string} The name of the property or method on the target or SC.UNSUPPORTED if no method found.
181   */
182   experimentalNameFor: function (target, standardName, testValue) {
183     var cachedNames = this._cachedNames,
184       targetGuid = SC.guidFor(target);
185 
186     // Fast path & cache initialization.
187     if (!cachedNames) {
188       cachedNames = this._cachedNames = {};
189       cachedNames[targetGuid] = {};
190     } else if (!cachedNames[targetGuid]) {
191       cachedNames[targetGuid] = {};
192     } else if (cachedNames[targetGuid][standardName]) {
193       return cachedNames[targetGuid][standardName];
194     }
195 
196     // Test the property name.
197     var ret = standardName;
198 
199     // ex. window.indexedDB.getDatabaseNames
200     if (!this._testSupportFor(target, ret, testValue)) {
201       // ex. window.WebKitCSSMatrix
202       ret = SC.browser.classPrefix + standardName.capitalize();
203       if (!this._testSupportFor(target, ret, testValue)) {
204         // No need to check if the prefix is the same for properties and classes
205         if (SC.browser.domPrefix === SC.browser.classPrefix) {
206           // Always show a warning so that production usage information has a
207           // better chance of filtering back to the developer(s).
208           SC.warn("SC.browser.experimentalNameFor(): target, %@, does not have property `%@` or `%@`.".fmt(target, standardName, ret));
209           ret = SC.UNSUPPORTED;
210         } else {
211           // ex. window.indexedDB.webkitGetDatabaseNames
212           ret = SC.browser.domPrefix + standardName.capitalize();
213           if (!this._testSupportFor(target, ret, testValue)) {
214             // Always show a warning so that production usage information has a
215             // better chance of filtering back to the developer(s).
216             SC.warn("SC.browser.experimentalNameFor(): target, %@, does not have property `%@`, '%@' or `%@`.".fmt(target, standardName, SC.browser.classPrefix + standardName.capitalize(), ret));
217             ret = SC.UNSUPPORTED;
218           }
219         }
220       }
221     }
222 
223     // Cache the experimental property name (even SC.UNSUPPORTED) for quick repeat access.
224     cachedNames[targetGuid][standardName] = ret;
225 
226     return ret;
227   },

229   /**
230     This method returns safe style names for current and future browsers.
231 
232     Using browser specific style prefixes is a risky coding practice.  With
233     sufficient testing, you may be able to match styles across today's most
234     popular browsers, but this is a lot of work and not future proof.  For
235     instance, if a browser drops the prefix and supports the standard style
236     name, your code will suddenly stop working.  This happens ALL the time!
237 
238     Instead, use SC.browser.experimentalStyleNameFor(standardStyleName), which
239     will test support for the standard style name and if not found will try the
240     prefixed version with the current browser's prefix appended.
241 
242     Note: the proper style name is only determined once per standard style
243     name tested and then cached.  Therefore, calling experimentalStyleNameFor
244     repeatedly has no performance detriment.
245 
246     For example,
247 
248         var boxShadowName = SC.browser.experimentalStyleNameFor('boxShadow'),
249           el = document.createElement('div');
250 
251         // `boxShadowName` may be "boxShadow", "WebkitBoxShadow", "msBoxShadow", etc. depending on the browser support.
252         el.style[boxShadowName] = "rgb(0,0,0) 0px 3px 5px";
253 
254     ## Improving deduction
255     Occasionally a browser will appear to support a style, but will fail to
256     actually accept a value.  In order to ensure that the style doesn't just
257     exist but is also usable, you can provide an optional `testValue` that will
258     be used to verify that the detected style is usable.
259 
260     @param {string} standardStyleName The standard name of the experimental style as it should be un-prefixed.  This is the DOM property name, which is camel-cased (ex. boxShadow)
261     @param {String} [testValue] A value to temporarily assign to the style to ensure support.
262     @returns {string} Future-proof style name for use in the current browser or SC.UNSUPPORTED if no style support found.
263   */
264   experimentalStyleNameFor: function (standardStyleName, testValue) {
265     // Test the style name.
266     var el = this._testEl;
267 
268     // Create a test element and cache it for repeated use.
269     if (!el) { el = this._testEl = document.createElement("div"); }
270 
271     return this.experimentalNameFor(el.style, standardStyleName, testValue);
272   },

274   /**
275     This method returns safe CSS attribute names for current and future browsers.
276 
277     Using browser specific CSS prefixes is a risky coding practice.  With
278     sufficient testing, you may be able to match attributes across today's most
279     popular browsers, but this is a lot of work and not future proof.  For
280     instance, if a browser drops the prefix and supports the standard CSS
281     name, your code will suddenly stop working.  This happens ALL the time!
282 
283     Instead, use SC.browser.experimentalCSSNameFor(standardCSSName), which
284     will test support for the standard CSS name and if not found will try the
285     prefixed version with the current browser's prefix appended.
286 
287     Note: the proper CSS name is only determined once per standard CSS
288     name tested and then cached.  Therefore, calling experimentalCSSNameFor
289     repeatedly has no performance detriment.
290 
291     For example,
292 
293         var boxShadowCSS = SC.browser.experimentalCSSNameFor('box-shadow'),
294           el = document.createElement('div');
295 
296         // `boxShadowCSS` may be "box-shadow", "-webkit-box-shadow", "-ms-box-shadow", etc. depending on the current browser.
297         el.style.cssText = boxShadowCSS + " rgb(0,0,0) 0px 3px 5px";
298 
299     ## Improving deduction
300     Occasionally a browser will appear to support a style, but will fail to
301     actually accept a value.  In order to ensure that the style doesn't just
302     exist but is also usable, you can provide an optional `testValue` that will
303     be used to verify that the detected style is usable.
304 
305     @param {string} standardCSSName The standard name of the experimental CSS attribute as it should be un-prefixed (ex. box-shadow).
306     @param {String} [testValue] A value to temporarily assign to the style to ensure support.
307     @returns {string} Future-proof CSS name for use in the current browser or SC.UNSUPPORTED if no style support found.
308   */
309   experimentalCSSNameFor: function (standardCSSName, testValue) {
310     var ret = standardCSSName,
311       standardStyleName = standardCSSName.camelize(),
312       styleName = this.experimentalStyleNameFor(standardStyleName, testValue);
313 
314     if (styleName === SC.UNSUPPORTED) {
315       ret = SC.UNSUPPORTED;
316     } else if (styleName !== standardStyleName) {
317       // If the DOM property is prefixed, then the CSS name should be prefixed.
318       ret = SC.browser.cssPrefix + standardCSSName;
319     }
320 
321     return ret;
322   }
