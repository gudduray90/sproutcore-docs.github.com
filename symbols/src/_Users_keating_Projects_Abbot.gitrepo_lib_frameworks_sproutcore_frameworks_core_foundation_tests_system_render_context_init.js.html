  8 /*global module test equals context */

 10 var context = null;
 11 
 12 module("SC.RenderContext#init");
 13 
 14 test("it should return a new context object with the passed tag name", function() {
 15   equals(SC.RenderContext('foo').tagName(), 'foo', 'tag name');
 16 });
 17 
 18 test("it should use a default tag name of div if not passed", function() {
 19   equals(SC.RenderContext().tagName(), 'div', 'tag name');
 20 });
 21 
 22 test("it should lowercase any tag name passed in", function() {
 23   equals(SC.RenderContext('DIV').tagName(), 'div', 'lowercase tag name');
 24 });
 25 
 26 test("it should have a length of 1 with a null value for the first time, saving space for the opening tag", function() {
 27   var context = SC.RenderContext();
 28   equals(context.length, 1, 'context length');
 29   equals(context.get(0), null, 'first item');
 30 });
 31 
 32 test("if script tag is passed, should mark context._selfClosing as NO" ,function() {
 33   var context = SC.RenderContext('script');
 34   ok(context._selfClosing === NO, "selfClosing MUST be no");
 35   
 36   context = SC.RenderContext('SCRIPT');
 37   ok(context._selfClosing === NO, "selfClosing MUST be no 2");
 38 });
 39 
 40 test("if element is passed, it should save element and not reserve space for string output", function() {
 41   var elem = document.createElement('div');
 42   var context = SC.RenderContext(elem);
 43   equals(context.length, 0, 'no length');
 44   equals(context._elem, elem, 'element');
 45   elem = context._elem = null; //cleanup
 46 });
 47 
 48 test("offset should should use offset + length of parent for self", function() {
 49   var context =SC.RenderContext('div');
 50   context.offset = 2;
 51   context.length = 3;
 52   var newContext = SC.RenderContext('div', context);
 53   equals(newContext.offset, 5, 'has proper offset');
 54 });
