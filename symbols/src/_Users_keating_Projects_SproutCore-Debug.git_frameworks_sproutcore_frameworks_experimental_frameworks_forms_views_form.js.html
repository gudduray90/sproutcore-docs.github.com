 13 /** 
 14   @class
 15   FormView lays out rows, manages their label widths, binds their
 16   content properties, and sets up their contentValueKeys as needed.
 17 
 18   Usually, you will place rows into the FormView:
 19   
 20       childViews: "fullName gender".w(),
 21       contentBinding: 'MyApp.personController',
 22 
 23       fullName: SC.FormView.row("Name:", SC.TextFieldView.extend({
 24         layout: {height: 20, width: 150}
 25       })),
 26 
 27       gender: SC.FormView.row("Gender:", SC.RadioView.design({
 28         layout: {width: 150, height: 40, centerY: 0},
 29         items: ["male", "female"]
 30       }))
 31 
 32   The name of the row (ie. 'fullName'), is passed down to the fields, and used as the key
 33   to bind the value property to the content. In this case it will bind content.fullName to the
 34   value property of the textFieldView.
 35 
 36 
 37   @extends SC.View
 38   @implements SC.FlowedLayout, SC.CalculatesEmptiness, SC.FormsEditMode
 39 */
 40 
 41 SC.FormView = SC.View.extend(SC.FlowedLayout, SC.CalculatesEmptiness, SC.FormsEditMode, /** @scope SC.FormView.prototype */ {
 42   // We lay out forms vertically. Each item gets its own "row". Wrapping makes
 43   // no sense, as the FormView should grow with each row.
 44   layoutDirection: SC.LAYOUT_VERTICAL,
 45   canWrap: NO,
 46 
 47   renderDelegateName: 'formRenderDelegate',
 48 
 49   /**
 50     The default padding around items in the form. By default, this comes from the theme.
 51     You can supply your own directly, or override the formRenderDelegate:
 52 
 53         // base it on the existing render delegate
 54         MyTheme.formRenderDelegate = SC.AceTheme.formRenderDelegate.create({
 55           flowSpacing: { left: 5, top: 5, right: 5, bottom: 5 }
 56         });
 57   */
 58   defaultFlowSpacing: SC.propertyFromRenderDelegate('flowSpacing', {}),
 59 
 60   classNames: ["sc-form-view"],
 61 
 62   /**
 63     Whether to automatically start editing.
 64   */
 65   editsByDefault: YES,
 66 
 67   /**
 68     The content to bind the form to. This content object is passed to all children.
 69   
 70     All child views, if added at design time via string-based childViews array, will get their
 71     contentValueKey set to their own key. Note that SC.RowView passes on its contentValueKey to its
 72     child field, and if its isNested property is YES, uses it to find its own content object.
 73   */
 74   content: null,
 75   
 76   /**
 77     Rows in the form do not have to be full SC.FormRowView at design time. They can also be hashes
 78     that get loaded into rows.
 79   */
 80   exampleRow: SC.FormRowView.extend({
 81     labelView: SC.FormRowView.LabelView.extend({ textAlign: SC.ALIGN_RIGHT })
 82   }),
 83 
 84   /**
 85      @private
 86   */
 87   init: function() {
 88     if (this.get("editsByDefault")) this.set("isEditing", YES);
 89     sc_super();
 90   },
 91 
 92   /**
 93   */
 94   createChildViews: function() {
 95     var childViews = this.get('childViews'),
 96         len        = childViews.length,
 97         idx, key, views, view,
 98         attrs, exampleRow = this.get('exampleRow');
 99 
100     this.beginPropertyChanges() ;
101 
102     // swap the array
103     for (idx=0; idx<len; ++idx) {
104       if (key = (view = childViews[idx])) {
105 
106         // is this is a key name, lookup view class
107         if (typeof key === SC.T_STRING) {
108           view = this[key];
109         } else {
110           key = null ;
111         }
112 
113         if (!view) {
114           SC.Logger.error ("No view with name "+key+" has been found in "+this.toString());
115           // skip this one.
116           continue;
117         }
118 
119         if(!view.isClass && SC.typeOf(view) === SC.T_HASH) {
120           attrs = view;
121           view = exampleRow;
122         }
123         else {
124           attrs = {};
125         }
126 
127         if((attrs.isFormField || view.prototype.isFormField) && (attrs.hasContentValueSupport || view.prototype.hasContentValueSupport)) {
128           attrs.contentValueKey = key;
129         }
130 
131         attrs.formKey = key;
132 
133         // createChildView creates the view if necessary, but also sets
134         // important properties, such as parentView
135         view = this.createChildView(view, attrs) ;
136 
137         if(!view.get('content')) {
138           view.bind('content', this, 'content');
139         }
140 
141         // for form rows, set up label measuring and the label itself.
142         if (view.isFormRow) {
143           // set label (if possible).
144           if (SC.none(view.get('label'))) {
145               view.set("label", key.humanize().titleize());
146           }
147 
148           // set the label size measuring stuff
149           if (this.get('labelWidth') !== null) {
150             view.set("shouldMeasureLabel", NO);
151           }
152         }
153 
154         if (key) { this[key] = view ; } // save on key name if passed
155       }
156       childViews[idx] = view;
157     }
158 
159     this.endPropertyChanges() ;
160 
161     this._hasCreatedRows = YES;
162     this.recalculateLabelWidth();
163 
164     return this ;
165   },
166 
167   
168   /**
169     Allows rows to use this to track label width.
170   */
171   isRowDelegate: YES,
172   
173   /**
174     Supply a label width to avoid automatically calculating the widths of the labels
175     in the form. Leave null to let SproutCore automatically determine the proper width
176     for the label.
177 
178     @type Number
179     @default null
180   */
181   labelWidth: null,
182   
183   /**
184     Tells the child rows whether they should measure their labels or not.
185   */
186   labelWidthDidChange: function() {
187     var childViews = this.get('childViews'), i, len = childViews.length,
188     shouldMeasure = SC.none(this.get('labelWidth'));
189     
190     for(i = 0; i < len; i++) {
191       childViews[i].set('shouldMeasureLabel', shouldMeasure);
192     }
193     
194     this.recalculateLabelWidth();
195   }.observes('labelWidth'),
196   
197   /**
198     Propagates the label width to the child rows, finding the measured size if necessary.
199   */
200   recalculateLabelWidth: function() {
201     if (!this._hasCreatedRows) {
202       return;
203     }
204     
205     var ret = this.get("labelWidth"), children = this.get("childViews"), idx, len = children.length, child;
206     
207     // calculate by looping through child views and getting size (if possible and if
208     // no label width is explicitly set)
209     if (ret === null) {
210       ret = 0;
211       for (idx = 0; idx < len; idx++) {
212         child = children[idx];
213       
214         // if it has a measurable row label
215         if (child.get("rowLabelMeasuredSize")) {
216           ret = Math.max(child.get("rowLabelMeasuredSize"), ret);
217         }
218       }
219     }
220     
221     // now set for all children
222     if (this._rowLabelSize !== ret) {
223       this._rowLabelSize = ret;
224       
225       // set by looping through child views
226       for (idx = 0; idx < len; idx++) {
227         child = children[idx];
228 
229         // if it has a measurable row label
230         if (child.get("hasRowLabel")) {
231           child.set("rowLabelSize", ret);
232         }
233       }
234       
235     }
236   },
237   
238   /**
239     Rows call this when their label width changes.
240   */
241   rowLabelMeasuredSizeDidChange: function(row, labelSize) {
242     this.invokeOnce("recalculateLabelWidth");
243   }
244

246 });

248 SC.mixin(SC.FormView, {
249   /**
250   Creates a form row.
251 
252   Can be called in two ways: `row(optionalClass, properties)`, which creates
253   a field with the properties, and puts it in a new row;
254   and `row(properties)`, which creates a new row—and it is up to you to add
255   any fields you want in the row.
256   
257   You can also supply some properties to extend the row itself with.
258   */
259   row: function(optionalClass, properties, rowExt)
260   {
261     return SC.FormRowView.row(optionalClass, properties, rowExt);
262   }
