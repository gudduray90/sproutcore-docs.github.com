 48 /**
 49   GridView used to position items according to its clippingFrame, which meant
 50   that a GridView could not specify a frame width in order to scroll
 51   horizontally.  This also manifested in a bug where itemsPerRow was based on
 52   frame but clippingFrame would change first so that GridView's would not
 53   update properly when clippingFrame changed and frame did not change.  For
 54   example during a rotation on an iPad.
 55 */
 56 test("GridView items should be positioned according to frame", function() {
 57   var columnWidth,
 58     itemsPerRow = view.get('itemsPerRow'),
 59     layout,
 60     expectedLayout;
 61 
 62   columnWidth = Math.floor(200/itemsPerRow);
 63 
 64   layout = view.layoutForContentIndex(0);
 65   expectedLayout = { left: 0, top: 0, height: 48, width: columnWidth };
 66   same(layout, expectedLayout, "Layout of first item should be");
 67 
 68   layout = view.layoutForContentIndex(2);
 69   expectedLayout = { left: 132, top: 0, height: 48, width: columnWidth };
 70   same(layout, expectedLayout, "Layout of third item should be");
 71 
 72   layout = view.layoutForContentIndex(3);
 73   expectedLayout = { left: 0, top: 48, height: 48, width: columnWidth };
 74   same(layout, expectedLayout, "Layout of fourth item should be");
 75 
 76   console.log(SC.stringFromRect(view.get('clippingFrame')));
 77   SC.RunLoop.begin();
 78   scrollView.adjust('width', 100);
 79   // view.set('clippingFrame', { x: 0, y: 0, width: 100, height: 400 });
 80   SC.RunLoop.end();
 81 
 82   console.log(SC.stringFromRect(view.get('clippingFrame')));
 83   layout = view.layoutForContentIndex(0);
 84   expectedLayout = { left: 0, top: 0, height: 48, width: columnWidth };
 85   same(layout, expectedLayout, "Layout of first item should be");
 86 
 87   layout = view.layoutForContentIndex(2);
 88   expectedLayout = { left: 132, top: 0, height: 48, width: columnWidth };
 89   same(layout, expectedLayout, "Layout of third item should be");
 90 
 91   layout = view.layoutForContentIndex(3);
 92   expectedLayout = { left: 0, top: 48, height: 48, width: columnWidth };
 93   same(layout, expectedLayout, "Layout of fourth item should be");
 94 });

 97 /**
 98   GridView would adjust all of its item views every time that its frame
 99   changed (which happens on scroll), which was very wasteful.  Instead, it was
100   improved to only reposition the nowShowing item views, which improved the
101   efficiency and also allowed the content to be sparse.  Finally, since the
102   position of the item views is only effected by changes to the frame's
103   width and not its x, y or height properties.  We can optimize it further to
104   only adjust item views when the width is different.
105 */
106 test("Optimized re-position of item views when the frame changes.", function() {
107   view.itemViewForContentIndex.expect(0);
108   view.layoutForContentIndex.expect(0);
109 
110   SC.RunLoop.begin();
111   view.notifyPropertyChange('frame');
112   SC.RunLoop.end();
113 
114   view.itemViewForContentIndex.expect(0);
115   view.layoutForContentIndex.expect(0);
116 
117   SC.RunLoop.begin();
118   view.adjust('height', 500);
119   SC.RunLoop.end();
120 
121   view.itemViewForContentIndex.expect(0);
122   view.layoutForContentIndex.expect(0);
123 
124   SC.RunLoop.begin();
125   view.adjust('width', 400);
126   SC.RunLoop.end();
127 
128   // Six item requests, twelve layout requests
129   view.itemViewForContentIndex.expect(6);
130   view.layoutForContentIndex.expect(12);
131 });
