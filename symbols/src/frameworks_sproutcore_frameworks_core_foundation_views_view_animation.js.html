 30 /**
 31   States that the view's layout can be set to if its animation is cancelled.
 32 
 33   ### START
 34 
 35   The previous layout of the view before calling animate.
 36 
 37   For example,
 38 
 39       myView.set('layout', { left: 0, top: 0, width: 100, bottom: 0 });
 40       myView.animate('left', 300, { duration: 1.5 });
 41 
 42       // later..
 43       myView.cancelAnimation(SC.LayoutState.START);
 44 
 45       myView.get('layout'); // => { left: 0, top: 0, width: 100, bottom: 0 }
 46 
 47   ### CURRENT
 48 
 49   The current layout of the view while it is animating.
 50 
 51   For example,
 52 
 53       myView.set('layout', { left: 0, top: 0, width: 100, bottom: 0 });
 54       myView.animate('left', 300, { duration: 1.5 });
 55 
 56       // later..
 57       myView.cancelAnimation(SC.LayoutState.CURRENT);
 58       myView.get('layout'); // => { left: 150, top: 0, width: 100, bottom: 0 }
 59 
 60   ### END
 61 
 62   The final layout of the view if the animation completed.
 63 
 64   For example,
 65 
 66       myView.set('layout', { left: 0, top: 0, width: 100, bottom: 0 });
 67       myView.animate('left', 300, { duration: 1.5 });
 68 
 69       // later..
 70       myView.cancelAnimation(SC.LayoutState.END);
 71       myView.get('layout'); // => { left: 300, top: 0, width: 100, bottom: 0 }
 72 
 73   @readonly
 74   @enum {Number}
 75 */
 76 SC.LayoutState = {
 77   START: 1,
 78   CURRENT: 2,
 79   END: 3
 80 };

 95     @param {boolean} animationResult.isCancelled Whether the animation was cancelled or not.
 96     @param {event} [animationResult.evt] The transitionend event if it exists.

100   /**
101     Animate a group of layout properties using CSS animations.
102 
103     On supported platforms, this will apply the proper CSS transition style
104     in order to animate the view to the new layout.  The properties object
105     should contain the names of the layout properties to animate with the new
106     layout values as values.
107 
108     # Options
109 
110     To control the transition, you must provide an options object that contains
111     at least the duration property and optionally the timing and delay
112     properties.  The options properties are as follows:
113 
114     - duration: The duration of the transition in seconds.  The default value is 0.25.
115 
116     - timing: The transition timing function.  This may be a predefined CSS timing
117       function (e.g. 'linear', 'ease', 'ease-in', 'ease-out', 'ease-in-out') or
118       it may be an array of values to make a cubic bezier (e.g. [0, 0, 0.58, 1.0]).
119       The default value is 'ease'.
120 
121       ** 'linear' - Specifies a transition effect with the same speed from start to end (equivalent to cubic-bezier(0,0,1,1))
122       ** 'ease' -  Specifies a transition effect with a slow start, then fast, then end slowly (equivalent to cubic-bezier(0.25,0.1,0.25,1))
123       ** 'ease-in' - Specifies a transition effect with a slow start (equivalent to cubic-bezier(0.42,0,1,1))
124       ** 'ease-out' -  Specifies a transition effect with a slow end (equivalent to cubic-bezier(0,0,0.58,1))
125       ** 'ease-in-out' - Specifies a transition effect with a slow start and end (equivalent to cubic-bezier(0.42,0,0.58,1))
126       ** 'cubic-bezier(n,n,n,n)' - Define your own values in the cubic-bezier function. Possible values are numeric values from 0 to 1
127 
128     - delay: The transition delay in seconds.  The default value is 0.
129 
130     For example,
131 
132         var myView = SC.View.create({
133           layout: { top: 10, left: 10, width: 200, height: 400 }
134         });
135 
136         MyApp.mainPane.appendChild(myView);
137 
138         // The view will animate to the new top & left values.
139         myView.animate(
140           { top: 200, left: 200 },  // properties
141           { duration: 0.75, timing: 'ease-out', delay: 0.5 } // options
142         );
143 
144     # Callbacks
145 
146     To execute code when the transition completes, you may provide an optional
147     target and/or method.  When the given group of transitions completes,
148     the callback function will be called once and passed an animationResult object with
149     properties containing the `event`, the `view` and a boolean `isCancelled` which
150     indicates if the animation had been cancelled or not.  The format of the
151     target and method follows the standard SproutCore format, where if the
152     target is not given then the view itself will be the target.  The
153     method can be a function or a property path to look up on the target.
154 
155     For example,
156 
157         // Passing a function for method.
158         myView.animate(
159           { top: 200, left: 200 },  // properties
160           { duration: 0.75 }, // options
161           function (animationResult) {  // method
162             // `this` will be myView
163           }
164         );
165 
166         // Passing a target and method.
167         myView.animate(
168           { scale: 0, opacity: 0 },  // properties
169           { duration: 1.5 }, // options
170           MyApp.statechart, // target
171           'myViewDidShrink' // method
172         );
173 
174     The animate functions are intelligent in how they apply animations and
175     calling animate in a manner that would effect an ongoing animation (i.e.
176     animating left again while it is still in transition) will result in
177     the ongoing animation callback firing immediately with isCancelled set to
178     true and adjusting the transition to accomodate the new settings.
179 
180     Note: This may not work if you are not using SproutCore for view layout,
181     which means you should not use `animate` if the view has `useStaticLayout`
182     set to true.
183 
184     ## A note about Hardware Acceleration.
185 
186     If a view has a fixed layout (i.e. view.get('isFixedLayout') == true) then
187     it will be eligible for hardware accelerated position transitions. Having a
188     fixed layout, simply means that the view has a fixed size (width and height)
189     and a fixed position (left and top).  If the view is eligible for hardware
190     acceleration, it must also set wantsAcceleratedLayer to true for animate to
191     use hardware accelerated transitions when animating its position.
192 
193     Occassionally, you may wish to animate a view with a non-fixed layout.  To
194     do so with hardware acceleration, you should convert the view to a fixed
195     layout temporarily and then set it back to a flexible layout after the
196     transition is complete.
197 
198     For example,
199 
200         // Flexible layout.
201         myView.set('layout', { left: 0, top: 10, right: 0, bottom: 10 });
202 
203         // Prepare to animate by converting to a fixed layout.
204         frame = myView.get('frame');
205         height = frame.height;
206         width = frame.width;
207         myView.adjust({ right: null, bottom: null, height: height, width: width });
208 
209         // Animate (will be hardware accelerated if myView.get('wantsAcceleratedLayout') is true).
210         myView.animate('left', width, { duration: 1 }, function () {
211           // Revert back to flexible layout.
212           myView.adjust({ right: -width, bottom: 10 });
213         });
214 
215     @param {Object|String} properties Hash of property names with new layout values or a single property name.
216     @param {Number} [value] The new layout value for a single property (only provide if the first parameter is a String).
217     @param {Object} options Hash of transition options.
218     @param {Object} [target=this] The target for the method.
219     @param {AnimateCallback|String} [method] The method to run when the transition completes.  May be a function or a property path.
220     @returns {SC.View} receiver
221   */
222   animate: function (key, value, options, target, method) {
223     var cur, curAnim,
224       valueDidChange = NO,
225       optionsDidChange = NO,
226       hash, layout,
227       optionsType,
228       pendingAnimations = this._pendingAnimations,
229       timing;
230 
231     //@if(debug)
232     // Provide a little developer support if they are doing something that should be considered wrong.
233     if (this.get('useStaticLayout')) {
234       SC.warn("Developer Warning: SC.View:animate() was called on a view with useStaticLayout and may not work.  If you are using CSS to layout the view (i.e. useStaticLayout: YES), then you should manage the animation manually.");
235     }
236     //@endif
237 
238     // Normalize arguments
239     if (typeof key === SC.T_STRING) {
240       hash = {};
241       hash[key] = value;
242     } else {
243       method = target;
244       target = options;
245       options = value;
246       hash = key;
247     }
248 
249     optionsType = SC.typeOf(options);
250     // This support should be deprecated.  Too much argument overloading.
251     if (optionsType === SC.T_NUMBER) {
252       //@if(debug)
253       // Provide a little developer support if they are doing something that should be considered wrong.
254       SC.warn("Developer Warning: The duration should be given as a property of the options object.");
255       //@endif
256       options = { duration: options };
257     } else if (optionsType !== SC.T_HASH) {
258       throw new Error("Must provide options hash!");
259     }
260 
261     // This support should be deprecated.  Too much argument overloading.
262     if (options.callback) {
263       //@if(debug)
264       // Provide a little developer support if they are doing something that should be considered wrong.
265       SC.warn("Developer Warning: The callback method should be given as an argument not as part of the options object.");
266       //@endif
267       method = options.callback;
268       delete options.callback;
269     }
270 
271     // Callback.  We need to keep the callback for each group of animations separate.
272     if (method === undefined) {
273       method = target;
274       target = this;
275     }
276 
277     // Support `null` being passed in for the target, rather than dropping the argument.
278     if (!target) target = this;
279 
280     if (method) {
281       if (typeof method === "string") method = target[method];
282       options.target = target;
283       options.method = method;
284     }
285 
286     // In the case of zero duration, just adjust and call the callback.
287     if (options.duration === 0) {
288       //@if(debug)
289       // Provide a little developer support if they are doing something that should be considered wrong.
290       SC.warn("Developer Warning: SC.View:animate() was called with a duration of 0 seconds.  The view will be adjusted and the callback will fire immediately in the next run loop.");
291       //@endif
292       this.adjust(hash);
293       this.runAnimationCallback(options, null, NO);
294       return this;
295     }
296 
297     // Timing function
298     timing = options.timing;
299     if (timing) {
300       if (typeof timing !== SC.T_STRING) {
301         options.timing = "cubic-bezier(" + timing[0] + ", " + timing[1] + ", " +
302                                          timing[2] + ", " + timing[3] + ")";
303       } // else leave as is (assume proper CSS timing String)
304     } else {
305       options.timing = 'ease';
306     }
307 
308     // Delay
309     if (SC.none(options.delay)) { options.delay = 0; }
310 
311     // Get the layout (may be a previous layout already animating).
312     if (!this._prevLayout) {
313       this._prevLayout = SC.clone(this.get('layout'));
314     }
315 
316     if (!pendingAnimations) { pendingAnimations = this._pendingAnimations = {}; }
317 
318     // Get the layout (may be a partially adjusted one already queued up).
319     layout = this._animateLayout || SC.clone(this.get('layout'));
320 
321     // Handle old style rotation.
322     if (!SC.none(hash.rotate)) {
323       //@if(debug)
324       SC.Logger.warn('Developer Warning: Please animate rotateX instead of rotate.');
325       //@endif
326       if (SC.none(hash.rotateX)) {
327         hash.rotateX = hash.rotate;
328       }
329       delete hash.rotate;
330     }
331 
332     // Go through the new animated properties and check for conflicts with
333     // previous calls to animate and changes to the current layout.
334     for (var property in hash) {
335       // Fast path.
336       if (!hash.hasOwnProperty(property) || !SC.ANIMATABLE_PROPERTIES[property]) {
337 
338         //@if(debug)
339         if (!SC.ANIMATABLE_PROPERTIES[property]) {
340           SC.warn("Developer Warning: The property `%@` is not animatable using SC.View:animate().".fmt(property));
341         }
342         //@endif
343         continue;
344       }
345 
346       value = hash[property];
347       cur = layout[property];
348       curAnim = pendingAnimations[property];
349 
350       if (SC.none(value)) { throw new Error("Can only animate to an actual value!"); }
351 
352       // If the new adjustment changes the previous adjustment's options before
353       // it has rendered, overwrite the previous adjustment.
354       if (curAnim && (curAnim.duration !== options.duration ||
355           curAnim.timing !== options.timing ||
356           curAnim.delay !== options.delay)) {
357         optionsDidChange = YES;
358         this.runAnimationCallback(curAnim, null, YES);
359       }
360 
361       if (cur !== value || optionsDidChange) {
362         valueDidChange = YES;
363         layout[property] = value;
364 
365         // Always update the animate hash to the newest options which may have been altered before this was applied.
366         pendingAnimations[property] = options;
367       }
368     }
369 
370     // Only animate to new values.
371     if (valueDidChange) {
372       // When animating height or width with centerX or centerY, we need to animate the margin property also to get a smooth change.
373       if (!SC.none(pendingAnimations.height) && !SC.none(layout.centerY) && SC.none(pendingAnimations.centerY)) {
374         // Don't animate less than 2px difference b/c the margin-top value won't differ.
375         if (Math.abs(hash.height - this.get('layout').height) >= 2) {
376           pendingAnimations.centerY = options;
377         }
378       }
379       if (!SC.none(pendingAnimations.width) && !SC.none(layout.centerX) && SC.none(pendingAnimations.centerX)) {
380         // Don't animate less than 2px difference b/c the margin-left value won't differ.
381         if (Math.abs(hash.width - this.get('layout').width) >= 2) {
382           pendingAnimations.centerX = options;
383         }
384       }
385 
386       this._animateLayout = layout;
387 
388       // Always run the animation asynchronously so that the original layout is guaranteed to be applied to the DOM.
389       this.invokeNext('_animate');
390     } else if (!optionsDidChange) {
391       this.runAnimationCallback(options, null, NO);
392     }
393 
394     return this;
395   },

414     Animates through the given frames.
415 
416     @param {Array} frames The array of frame objects.
417     @param {AnimateCallback} callback The callback function to call when the final frame is done animating.

421   // TODO: Do this using CSS animations instead.
422   _animateFrames: function (frames, callback, initialDelay, count) {
423     // Normalize the private argument `count`.
424     if (SC.none(count)) { count = 0; }
425 
426     var frame = frames[count];
427 
428     this.animate(frame.value, {
429       delay: initialDelay,
430       duration: frame.duration,
431       timing: frame.timing
432     }, function (data) {
433       count += 1;
434 
435       // Keep iterating while frames exist and the animations weren't cancelled.
436       if (!data.isCancelled && count < frames.length) {
437         // Only delay on the first animation.  Increase count to the next frame.
438         this._animateFrames(frames, callback, 0, count);
439       } else {
440         // Done.
441         callback(data);
442       }
443     });
444 
445     return this;

448   /**
449     Cancels the animation, adjusting the view's layout immediately to one of
450     three values depending on the `layoutState` parameter.
451 
452     If no `layoutState` is given or if SC.LayoutState.END is given, the view
453     will be adjusted to its final layout.  If SC.LayoutState.START is given,
454     the view will be adjusted back to its initial layout and if
455     SC.LayoutState.CURRENT is given, the view will stop at its current layout
456     value, which will be some transient value between the start and end values.
457 
458     Note: The animation callbacks will be called with the animationResult object's
459     isCancelled property set to YES.
460 
461     @param {SC.LayoutState} [layoutState=SC.LayoutState.END] The layout to immediately adjust the view to.
462     @returns {SC.View} this
463   */
464   cancelAnimation: function (layoutState) {
465     var activeAnimations = this._activeAnimations,
466       pendingAnimations = this._pendingAnimations,
467       animation,
468       key,
469       layout,
470       didCancel = NO;
471 
472     switch (layoutState) {
473     case SC.LayoutState.START:
474       // Revert back to the start layout.
475       layout = this._prevLayout;
476       break;
477     case SC.LayoutState.CURRENT:
478       // Stop at the current layout.
479       layout = this.get('liveAdjustments');
480       break;
481     default:
482     }
483 
484     // Immediately remove the pending animations while calling the callbacks.
485     for (key in pendingAnimations) {
486       animation = pendingAnimations[key];
487       didCancel = YES;
488 
489       // Update the animation hash.  Do this first, so callbacks can check for active animations.
490       delete pendingAnimations[key];
491 
492       // Run the callback.
493       this.runAnimationCallback(animation, null, YES);
494     }
495 
496     // Immediately remove the animation styles while calling the callbacks.
497     for (key in activeAnimations) {
498       animation = activeAnimations[key];
499       didCancel = YES;
500 
501       // Update the animation hash.  Do this first, so callbacks can check for active animations.
502       delete activeAnimations[key];
503 
504       // Remove the animation style without triggering a layout change.
505       this.removeAnimationFromLayout(key, YES);
506 
507       // Run the callback.
508       this.runAnimationCallback(animation, null, YES);
509     }
510 
511     // Adjust to final position.
512     if (didCancel && !!layout) {
513       this.adjust(layout);
514     }
515 
516     // Clean up.
517     delete this._prevLayout;
518     delete this._activeAnimations;
519     delete this._pendingAnimations;
520 
521     return this;
522   },

525     This method is called after the layout style is applied to the layer.  If
526     the platform didn't support CSS transitions, the callbacks will be fired

530

545   /** @private
546     Returns the live values of the properties being animated on a view while it
547     is animating.  Getting the layout of the view after a call to animate will
548     include the final values, some of which will not be the same as what they
549     are while the animation is in progress.
550 
551     Depending on the property being animated, determining the actual value can
552     be quite difficult.  For instance, accelerated views will animate certain
553     properties using a browser specific CSS transition on a CSS transform and
554     the current value may be a CSSMatrix that needs to be mapped back to a
555     regular layout format.
556 
557     This property is used by cancelAnimation() to stop the animation in its
558     current place.
559 
560     PRIVATE - because we may want to rename this function and change its output
561 
562     @returns {Object}
563   */
564   liveAdjustments: function () {
565     var activeAnimations = this._activeAnimations,
566       jqueryEl = this.$(),
567       ret = {},
568       transformKey = SC.browser.experimentalCSSNameFor('transform');
569 
570     if (activeAnimations) {
571       for (var key in activeAnimations) {
572         if (key === transformKey) {
573           var matrix = jqueryEl.css(key),
574             CSSMatrixClass = SC.browser.experimentalNameFor(window, 'CSSMatrix');
575 
576           if (CSSMatrixClass !== SC.UNSUPPORTED) {
577             matrix = new window[CSSMatrixClass](matrix);
578 
579             // Calculate rotateX
580             ret.rotateX = Math.atan2(matrix.m32, matrix.m33) * (180 / Math.PI);
581 
582             // Calculate rotateY
583             ret.rotateY = Math.atan2(-matrix.m31, Math.sqrt(Math.pow(matrix.m32, 2) + Math.pow(matrix.m33, 2))) * (180 / Math.PI);
584 
585             // Calculate rotateZ
586             ret.rotateZ = -Math.atan2(matrix.m21, matrix.m11) * (180 / Math.PI);
587 
588             // Calculate scaleX
589             ret.scale = Math.sqrt(Math.pow(matrix.m11, 2) + Math.pow(matrix.m12, 2) + Math.pow(matrix.m13, 2));
590             if (matrix.m11 < 0) ret.scale = ret.scale * -1;
591 
592             // Retrieve translateX & translateY
593             if (matrix.e > 0) { ret.left = matrix.e; }
594             if (matrix.f > 0) { ret.top = matrix.f; }
595           } else {
596             matrix = matrix.match(/^matrix\((.*)\)$/)[1].split(/,\s*/);
597             if (matrix) {
598               ret.left = parseInt(matrix[4], 10);
599               ret.top = parseInt(matrix[5], 10);
600             }
601           }
602         } else {
603           ret[key] = parseInt(jqueryEl.css(key), 10);
604         }
605       }
606     }
607 
608     return ret;

613     var activeAnimations = this._activeAnimations,
614       layer = this.get('layer');

617       var updatedCSS = [];

622           updatedCSS.push(activeAnimations[key].css);

624       }

642   /** @private */
643   runAnimationCallback: function (animation, evt, cancelled) {
644     var method = animation.method,
645       target = animation.target;
646 
647     if (method) {
648       // We're using invokeNext so we don't trigger any layout changes from
649       // the callback until the current layout is updated.
650       this.invokeNext(function () {
651         method.call(target, { event: evt, view: this, isCancelled: cancelled });
652       }, this);
653 
654       // Always clear the method from the hash to prevent it being called
655       // multiple times for animations in the group.
656       delete animation.method;
657       delete animation.target;
658     }

661   /** @private
662     Called when animation ends, should not usually be called manually
663   */
664   transitionDidEnd: function (evt) {
665     var propertyName = evt.originalEvent.propertyName,
666       activeAnimations = this._activeAnimations,
667       animation;
668 
669     // Fix up the centerX & centerY properties.
670     if (propertyName === 'margin-left') { propertyName = 'centerX'; }
671     if (propertyName === 'margin-top') { propertyName = 'centerY'; }
672     animation = activeAnimations ? activeAnimations[propertyName] : null;
673 
674     if (animation) {
675       // Update the animation hash.  Do this first, so callbacks can check for active animations.
676       delete activeAnimations[propertyName];
677 
678       // Remove the animation style without triggering a layout change.
679       this.removeAnimationFromLayout(propertyName, YES);
680 
681       // Run the callback.
682       this.runAnimationCallback(animation, evt, NO);
683 
684       // Clean up the internal hash.
685       this._activeAnimationsLength -= 1;
686       if (this._activeAnimationsLength === 0) {
687         delete this._activeAnimations;
688         delete this._prevLayout;
689       }

693   /** @private

695    animations have been defined for the view, they will be included in
696    this._pendingAnimations.  This method will clear out any conflicts between
697    pending and active animations.
698    */
699   willRenderAnimations: function () {
700     if (SC.platform.supportsCSSTransitions) {
701       var pendingAnimations = this._pendingAnimations;
702 
703       if (pendingAnimations) {
704         var activeAnimations = this._activeAnimations;
705 
706         if (!activeAnimations) {
707           this._activeAnimationsLength = 0;
708           activeAnimations = {};
709         }
710 
711         for (var key in pendingAnimations) {
712           if (!pendingAnimations.hasOwnProperty(key)) { continue; }
713 
714           var activeAnimation = activeAnimations[key],
715             pendingAnimation = pendingAnimations[key];
716 
717           if (activeAnimation) {
718             this.runAnimationCallback(activeAnimation, null, YES);
719           }
720 
721           activeAnimations[key] = pendingAnimation;
722           this._activeAnimationsLength += 1;
723         }
724 
725         this._activeAnimations = activeAnimations;
726         this._pendingAnimations = null;
727       }
728     }
