  7 /**
  8 
  9   The full set of targets available in the application.  This is populated
 10   automatically when you call loadTargets().
 11 
 12   @extends SC.ArrayController
 13 */
 14 TestRunner.targetsController = SC.ArrayController.create(
 15 /** @scope TestRunner.targetsController.prototype */ {
 16 
 17   /**
 18     Generates the root array of children objects whenever the target content
 19     changes.  Used in a tree node.
 20   */
 21   sourceRoot: function () {
 22 
 23     // break targets into their respective types.  Items that should not be
 24     // visible at the top level will not have a sort kind
 25     var kinds = {}, keys = [], kind, targets, ret;
 26 
 27     this.forEach(function (target) {
 28       kind = target.get('sortKind');
 29 
 30       if (kind) {
 31         targets = kinds[kind];
 32         if (!targets) { kinds[kind] = targets = []; }
 33 
 34         targets.push(target);
 35         if (keys.indexOf(kind) < 0) keys.push(kind);
 36       }
 37     }, this);
 38 
 39     // sort kinds alphabetically - with sproutcore at end and apps at top
 40     keys.sort();
 41     if (keys.indexOf('sproutcore') >= 0) {
 42       keys.removeObject('sproutcore').pushObject('sproutcore');
 43     }
 44     if (keys.indexOf('apps') >= 0) {
 45       keys.removeObject('apps').unshiftObject('apps');
 46     }
 47 
 48     // once divided into kinds, create group nodes for each kind
 49     ret = [];
 50     keys.forEach(function (kind) {
 51       targets = kinds[kind];
 52 
 53       var defKey = "SourceList.%@.isExpanded".fmt(kind),
 54           expanded = TestRunner.userDefaults.get(defKey);
 55 
 56       ret.push(SC.Object.create({
 57         displayName: "Kind.%@".fmt(kind).loc(),
 58         isExpanded: SC.none(expanded) ? (kind !== 'sproutcore') : expanded,
 59         children: targets.sortProperty('kind', 'displayName'),
 60 
 61         isExpandedDefaultKey: defKey,
 62         isExpandedDidChange: function () {
 63           TestRunner.userDefaults.set(this.get('isExpandedDefaultKey'), this.get('isExpanded'));
 64         }.observes('isExpanded')
 65       }));
 66     });
 67 
 68     return SC.Object.create({ children: ret, isExpanded: YES });
 69 
 70   }.property('[]').cacheable(),
 71 
 72   /**
 73     Send event when targets load.
 74   */
 75   statusDidChange: function () {
 76     if (this.get('status') === SC.Record.READY_CLEAN) {
 77       TestRunner.statechart.resumeGotoState();
 78     }
 79   }.observes('status')
 80 
 81 });
