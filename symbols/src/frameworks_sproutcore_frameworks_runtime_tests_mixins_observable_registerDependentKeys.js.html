 10 var object ;
 11 
 12 module("object.registerDependentKeys()", {  
 13   setup: function() {
 14     object = SC.Object.create({
 15 
 16         // normal properties
 17         firstName:  'John',
 18         lastName:   'Doe',
 19         observedValue: '',
 20 
 21         // computed property
 22         fullName: function() {
 23           return this.getEach('firstName','lastName').compact().join(' ');
 24         }.property(),
 25 
 26         // init to setup registerDependentKey...
 27         init: function() {
 28           sc_super();
 29           this.registerDependentKey('fullName', 'firstName', 'lastName');
 30         },
 31 
 32         //observer that should fire whenever the 'fullName' property changes
 33         fullNameDidChange:  function() {
 34           this.set('observedValue', this.get('fullName')) ;
 35         }.observes('fullName')
 36     });
 37   }
 38 });
 39 
 40 
 41 test("should indicate the registered property changes if the dependent key value changes", function() {
 42   // now, change the firstName...
 43   object.set('firstName', 'Jane');
 44 
 45   // since fullName is 'dependent' on firstName, then the observer for  
 46   // 'fullName' should fire here because you changed a dependent key.
 47   equals(object.get('observedValue'), 'Jane Doe');
 48 
 49   // now change the lastName
 50   object.set('lastName', 'Johnson');
 51 
 52   // again, fullName is 'dependent' on lastName, so observer for  
 53   // fullName should fire.
 54   equals(object.get('observedValue'), 'Jane Johnson');
 55 });
 56 
 57 
 58 test("should indicate the registered property changes if the dependent key value changes and change is within begin property loop ", function() {
 59   // Wrap the changes with begin property changes call
 60   object.beginPropertyChanges();
 61   
 62   // now, change the firstName & lastname...
 63   object.set('firstName', 'Jane');
 64   object.set('lastName', 'Johnson');
 65   
 66   // The observer for fullName should not have fired yet at this  
 67   // point because we are inside a propertyChange loop.
 68   equals(object.get('observedValue'), '');
 69   
 70   //End the property changes loop.
 71   object.endPropertyChanges();
 72   
 73   // now change the lastName
 74   object.set('lastName', 'Johnson');
 75 
 76   // again, fullName is 'dependent' on lastName, so observer for  
 77   // fullName should fire.
 78   equals(object.get('observedValue'), 'Jane Johnson');
 79 });

 81 module("object.registerDependentKeys() - @each");

 83 test("should invalidate computed property once per changed key", function() {
 84   var setCalls = 0;
 85   var getCalls = 0;
 86 
 87   window.people = SC.Object.create({
 88     content: [SC.Object.create({name:'Juan'}),
 89               SC.Object.create({name:'Camilo'}),
 90               SC.Object.create({name:'Pinzon'})],
 91 
 92     fullName: function(key, value) {
 93       if (value !== undefined) {
 94         setCalls++;
 95         this.content.setEach('name', value);
 96       } else {
 97         getCalls++;
 98       }
 99 
100       return this.content.getEach('name').join(' ');
101     }.property('content.@each.name')
102   });
103 
104   var peopleWatcher = SC.Object.create({
105     nameBinding: 'window.people.fullName'
106   });
107 
108   SC.run(function() { people.set('fullName', 'foo bar baz'); });
109   equals(setCalls, 1, "calls set once");
110   equals(getCalls, 3, "calls get three times");
111 });

113 test("should invalidate key when properties higher up in the chain change", function() {
114   var notified = 0;
115   
116   var obj = SC.Object.create({
117     contact: null,
118     
119     fullName: function(key, value) {
120       return [this.getPath('contact.firstName'), this.getPath('contact.lastName')].join(' ');
121     }.property('contact.firstName', 'contact.lastName').cacheable(),
122     
123     fullNameDidChange: function() {
124       notified++;
125     }.observes('fullName')
126   });
127 
128   var johnDoe = SC.Object.create({ firstName: 'John', lastName: 'Doe' });
129   var janeDoe = SC.Object.create({ firstName: 'Jane', lastName: 'Doe' });
130   
131   equals(notified, 0, 'should start empty');
132   SC.run(function() {  obj.set('contact', johnDoe);  });
133   equals(notified, 1, 'should notify once after set content=johnDoe');
134   equals(obj.get('fullName'), 'John Doe', 'should get proper name');
135   
136   notified = 0;
137   SC.run(function() { johnDoe.set('firstName', 'JOHNNY'); });
138   equals(notified, 1, 'should notify again after set firstName=JOHNNY');
139   equals(obj.get('fullName'), 'JOHNNY Doe', 'should get proper name');
140   
141   notified = 0;
142   SC.run(function() { obj.set('contact', janeDoe); });
143   equals(notified, 1, 'should notify again after set content=janeDoe');
144   equals(obj.get('fullName'), 'Jane Doe', 'should get proper name');
145 
146   notified = 0;
147   SC.run(function() { johnDoe.set('firstName', 'JON'); });
148   equals(notified, 0, 'should NOT notify again after set johnDoe.firstName=JON (johnDoe is not current contact)');
149   equals(obj.get('fullName'), 'Jane Doe', 'should get proper name while janeDoe is current');
150 
151   notified = 0;
152   SC.run(function() { janeDoe.set('firstName', 'Janna'); });
153   equals(notified, 1, 'should notify again after set janeDoe.firstName=Janna');
154   equals(obj.get('fullName'), 'Janna Doe', 'should get proper name after firstname=Janna');  
155 });
