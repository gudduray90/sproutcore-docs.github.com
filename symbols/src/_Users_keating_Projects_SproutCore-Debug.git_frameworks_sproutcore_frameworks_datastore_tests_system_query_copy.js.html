  8 // test parsing of query string
  9 module("SC.Query#copy");
 10 
 11 test("basic copy", function() {
 12   var q=  SC.Query.create({
 13     conditions: "foo = bar",
 14     parameters: { foo: "bar" },
 15     orderBy: "foo",
 16     recordType: SC.Record,
 17     recordTypes: [SC.Record],
 18     location: SC.Query.REMOTE,
 19     scope: SC.CoreSet.create()
 20   }).freeze();
 21   
 22   var keys = 'conditions orderBy recordType recordTypes parameters location scope'.w();
 23   var copy = q.copy();
 24  25   
 26   equals(copy.isFrozen, NO, 'copy should not be frozen');
 27   keys.forEach(function(key) {
 28     equals(copy.get(key), q.get(key), 'copy.%@ should = original.%@'.fmt(key, key));
 29   }, this);
 30   
 31 });
