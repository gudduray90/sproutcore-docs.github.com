  9 window.SC = window.SC || {};
 10 
 11 // Note:  We won't use SC.T_* here because those constants might not yet be
 12 //        defined.
 13 SC._mapDisplayNamesUseHashForSeenTypes = ['object', 'number', 'boolean', 'array', 'string', 'function', 'class', 'undefined', 'error'];  // 'hash' causes problems
 14 
 15 
 16 SC.mapDisplayNames = function(obj, level, path, seenHash, seenArray) {
 17   if (SC.browser.engine !== SC.ENGINE.webkit) return ;
 18 
 19   // Lazily instantiate the hash of types we'll use a hash for the "have we
 20   // seen this before?" structure.  (Some types are not safe to put in a hash
 21   // in this manner, so we'll use the hash for its algorithmic advantage when
 22   // possible, but fall back to an array using `indexOf()` when necessary.)
 23   if (!SC._mapDisplayNamesUseHashForSeenTypesHash) {
 24     var types = SC._mapDisplayNamesUseHashForSeenTypes ;
 25     var typesHash = {} ;
 26     var len = types.length ;
 27     for (var i = 0;  i < len;  ++i) {
 28       var type = types[i] ;
 29       typesHash[type] = true ;
 30     }
 31     SC._mapDisplayNamesUseHashForSeenTypesHash = typesHash ;
 32   }
 33 
 34 
 35   if (obj === undefined) obj = window ;
 36   if (level === undefined) level = 0 ;
 37   if (path === undefined) path = [] ;
 38  39   if (seenHash === undefined) seenHash = {} ;
 40   if (seenArray === undefined) seenArray = [] ;
 41 
 42   if (level > 5) return ;
 43 
 44   var useHash = !!SC._mapDisplayNamesUseHashForSeenTypesHash[SC.typeOf(obj)] ;
 45 
 46   var hash;
 47   var arrayToCheck;
 48   if (useHash) {
 49     hash = SC.hashFor(obj) ;
 50     arrayToCheck = seenHash[hash];
 51   }
 52   else {
 53     arrayToCheck = seenArray;
 54   }
 55 
 56   if (arrayToCheck  &&  arrayToCheck.indexOf(obj) !== -1) return ;
 57 
 58   if (arrayToCheck) {
 59     arrayToCheck.push(obj) ;
 60   }
 61   else if (useHash) {
 62     seenHash[hash] = [obj] ;
 63   }
 64 
 65   var loc = path.length, str, val, t;
 66   path[loc] = '';
 67 
 68   for(var key in obj) {
 69     if (obj.hasOwnProperty && !obj.hasOwnProperty(key)) continue ;
 70     if (!isNaN(Number(key))) continue ; // skip array indexes
 71     if (key === "constructor") continue ;
 72     if (key === "superclass") continue ;
 73     if (key === "document") continue ;
 74 
 75     // Avoid TypeError's in WebKit based browsers
 76     if (obj.type && obj.type === 'file') {
 77       if (key === 'selectionStart' || key === 'selectionEnd') continue;
 78     }
 79 
 80     try{
 81       val = obj[key];
 82     }catch(e){
 83       //This object might be special this get called when an app
 84      // using webView adds an static C object to JS.
 85       continue;
 86     }
 87     if (key === "SproutCore") key = "SC";
 88     t   = SC.typeOf(val);
 89     if (t === SC.T_FUNCTION) {
 90       if (!val.displayName) { // only name the first time it is encountered
 91         path[loc] = key ;
 92         str = path.join('.').replace('.prototype.', '#');
 93         val.displayName = str;
 94       }
 95 
 96       // handle constructor-style
 97       if (val.prototype) {
 98         path.push("prototype");
 99         SC.mapDisplayNames(val.prototype, level+1, path, seenHash, seenArray);
100         path.pop();
101       }
102 
103     } else if (t === SC.T_CLASS) {
104       path[loc] = key ;
105       SC.mapDisplayNames(val, level+1, path, seenHash, seenArray);
106 
107     } else if ((key.indexOf('_')!==0) && (t===SC.T_OBJECT || t===SC.T_HASH)) {
108       path[loc] = key ;
109       SC.mapDisplayNames(val, level+1, path, seenHash, seenArray);
110     }
111   }
112 
113   path.pop();
114 };
