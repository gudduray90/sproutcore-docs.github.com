  8 var parent, store, child, storeKey, json;
  9 module("SC.NestedStore#commitChangesFromNestedStore", {
 10   setup: function() {
 11     SC.RunLoop.begin();
 12 
 13     parent = SC.Store.create();
 14 
 15     json = {
 16       string: "string",
 17       number: 23,
 18       bool:   YES
 19     };
 20 
 21     storeKey = SC.Store.generateStoreKey();
 22 
 23     store = parent.chain();
 24     child = store.chain();  // test multiple levels deep
 25 
 26     // write basic status
 27     child.writeDataHash(storeKey, json, SC.Record.READY_DIRTY);
 28     child.dataHashDidChange(storeKey);
 29     child.changelog = SC.Set.create();
 30     child.changelog.add(storeKey);
 31 
 32     SC.RunLoop.end();
 33   }
 34 });
 35 
 36 test("copies changed data hashes, statuses, and revisions", function() {
 37 
 38   SC.RunLoop.begin();
 39 
 40   // verify preconditions
 41   equals(store.readDataHash(storeKey), null, 'precond - should not have data yet');
 42   ok(child.chainedChanges.contains(storeKey), 'precond - child changes should include storeKey');
 43 
 44   // perform action
 45   equals(store.commitChangesFromNestedStore(child, child.chainedChanges, NO), store, 'should return receiver');
 46   SC.RunLoop.end();
 47 
 48   // verify new status
 49   equals(store.readDataHash(storeKey), json, 'now should have json');
 50   equals(store.readStatus(storeKey), SC.Record.READY_DIRTY, 'now should have status');
 51   equals(store.revisions[storeKey], child.revisions[storeKey], 'now shoulave have revision from child');
 52 
 53 });
 54 
 55 test("adds lock on any items not already locked", function() {
 56 
 57   SC.RunLoop.begin();
 58 
 59   var storeKey2 = SC.Store.generateStoreKey();
 60   var json2 = { kind: "json2" };
 61 
 62   // verify preconditions
 63   store.readDataHash(storeKey);
 64   ok(store.locks[storeKey], 'precond - storeKey should have lock');
 65   ok(!store.locks[storeKey2], 'precond - storeKey2 should not have lock');
 66 
 67   // write another record into child store to commit changes.
 68   child.writeDataHash(storeKey2, json2, SC.Record.READY_DIRTY);
 69   child.dataHashDidChange(storeKey2);
 70 
 71   var changes = child.chainedChanges ;
 72   ok(changes.contains(storeKey), 'precond - child.chainedChanges should contain storeKey');
 73   ok(changes.contains(storeKey2), 'precond - child.chainedChanges should contain storeKey2');
 74 
 75   // now commit back to parent
 76   equals(store.commitChangesFromNestedStore(child, changes, NO), store, 'should return receiver');
 77   SC.RunLoop.end();
 78 
 79   // and verify that both have locks
 80   ok(store.locks[storeKey], 'storeKey should have lock after commit (actual: %@)'.fmt(store.locks[storeKey]));
 81   ok(store.locks[storeKey2], 'storeKey2 should have lock after commit (actual: %@)'.fmt(store.locks[storeKey2]));
 82 
 83 });
 84 
 85 test("adds items in chainedChanges to receiver chainedChanges", function() {
 86 
 87   SC.RunLoop.begin();
 88 
 89   var key1 = SC.Store.generateStoreKey();
 90 
 91   store.dataHashDidChange(key1);
 92 
 93   ok(child.chainedChanges.contains(storeKey), 'precond - child.chainedChanges should contain store key');
 94 
 95   equals(store.commitChangesFromNestedStore(child, child.chainedChanges, NO), store, 'should return receiver');
 96   SC.RunLoop.end();
 97 
 98   // changelog should merge nested store & existing
 99   ok(store.chainedChanges.contains(key1), 'chainedChanges should still contain key1');
100   ok(store.chainedChanges.contains(storeKey), 'chainedChanges should also contain storeKey');
101 });
102 
103 test("should set hasChanges to YES if has changes", function() {
104 
105   SC.RunLoop.begin();
106 
107   var changes = child.chainedChanges;
108   ok(changes.length>0, 'precond - should have some changes in child');
109   equals(store.get('hasChanges'), NO, 'precond - store should not have changes');
110 
111   store.commitChangesFromNestedStore(child, changes, NO);
112   equals(store.get('hasChanges'), YES, 'store should now have changes');
113 });
114 
115 test("should set hasChanges to NO if no changes", function() {
116 
117   SC.RunLoop.begin();
118 
119   child = store.chain() ; // get a new child store
120 
121   var changes = child.chainedChanges || SC.Set.create();
122   ok(!changes || !changes.length, 'precond - should have not have changes in child');
123   equals(store.get('hasChanges'), NO, 'precond - store should not have changes');
124 
125   store.commitChangesFromNestedStore(child, changes, NO);
126   SC.RunLoop.end();
127 
128   equals(store.get('hasChanges'), NO, 'store should NOT now have changes');
129 });

135 test("committing changes should chain back each step", function() {
136 
137   SC.RunLoop.begin();
138 
139   // preconditions
140   equals(child.readDataHash(storeKey), json, 'precond - child should have data');
141   equals(store.readDataHash(storeKey), null, 'precond - store should not have data');
142   equals(parent.readDataHash(storeKey), null, 'precond - parent should not have data');
143 
144   // do commits
145   child.commitChanges();
146 
147   equals(store.get('hasChanges'), YES, 'store should now have changes');
148   equals(store.readDataHash(storeKey), json, 'store should now have json');
149 
150   store.commitChanges();
151   equals(store.get('hasChanges'), NO, 'store should no longer have changes');
152   equals(parent.readDataHash(storeKey), json, 'parent should now have json');
153   SC.RunLoop.end();
154 
155 });
