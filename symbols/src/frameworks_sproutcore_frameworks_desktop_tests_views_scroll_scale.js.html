100 test("Content view is scaled based on scroll view's scale property", function() {
101   var scrollView = pane.view('scrollView with content view of fixed height'),
102     contentView = scrollView.get('contentView');
103 
104   SC.run(function() {
105     scrollView.set('scale', 0.1);
106   });
107 
108   equals(scrollView.get('scale'), 0.1, 'Scale has been updated to 0.1');
109 
110   equals(contentView.getPath('frame.width'), 0.1 * contentViewOriginalWidth, "Content view's width is scaled smaller based on scroll view's scale");
111   equals(contentView.getPath('frame.height'), 0.1 * contentViewOriginalHeight, "Content view's height is scaled smaller based on scroll view's scale");
112 
113   SC.run(function() {
114     scrollView.set('scale', 10);
115   });
116 
117   equals(scrollView.get('scale'), 10, 'Scale has been updated to 10');
118 
119   equals(contentView.getPath('frame.width'), 10 * contentViewOriginalWidth, "Content view's width is scaled larger based on scroll view's scale");
120   equals(contentView.getPath('frame.height'), 10 * contentViewOriginalHeight, "Content view's height is scaled larger based on scroll view's scale");
121 
122 });

128 test('When contentView is fully visible, it is positioned according to horizontalAlign', function() {
129   var scrollView = pane.view('scrollView with content view of fixed height'),
130     containerView = scrollView.get('containerView'),
131     contentView = scrollView.get('contentView');
132 
133   SC.run(function() {
134     scrollView.set('scale', 0.1);
135   });
136 
137   SC.run(function() {
138     scrollView.set('horizontalAlign', SC.ALIGN_LEFT);
139   });
140 
141   equals(contentView.getPath('frame.x'), 0, "Content view's x offset is 0 when horizontalAlign is set to SC.ALIGN_LEFT");
142 
143   SC.run(function() {
144     scrollView.set('horizontalAlign', SC.ALIGN_CENTER);
145   });
146 
147   var expectedContentViewOffsetX = (containerView.getPath('frame.width') - contentView.getPath('frame.width')) / 2;
148   expectedContentViewOffsetX = Math.round(expectedContentViewOffsetX);
149   equals(contentView.getPath('frame.x'), expectedContentViewOffsetX, "Content view is centered when horizontalAlign is set to SC.ALIGN_CENTER");
150 
151   SC.run(function() {
152     scrollView.set('horizontalAlign', SC.ALIGN_RIGHT);
153   });
154 
155   expectedContentViewOffsetX = containerView.getPath('frame.width') - contentView.getPath('frame.width');
156   expectedContentViewOffsetX = Math.round(expectedContentViewOffsetX);
157 
158   equals(contentView.getPath('frame.x'), expectedContentViewOffsetX, "Content view is aligned right when horizontalAlign is set to SC.ALIGN_RIGHT");
159 });

199 test('When zoomed into the contentView, the content view should remain centered around visible center', function() {
200   var scrollView = pane.view('scrollView with content view of fixed height'),
201     containerFrame = scrollView.getPath('containerView.frame');
202 
203   // The percentage offsets for horizontal: ALIGN_CENTER & vertical: ALIGN_TOP for the given content and container size.
204   var hPct = 0.5,
205       vPct = 0.1075;
206 
207   SC.run(function() {
208     // Note: Before any scrolling occurs, scale changes adhere to initial alignments.
209     scrollView.set('horizontalScrollOffset', scrollView.get('horizontalScrollOffset'));
210     scrollView.set('verticalScrollOffset', scrollView.get('verticalScrollOffset'));
211 
212     scrollView.set('scale', 10);
213   });
214 
215   var newOffsets = {
216     h: hPct * (scrollView.get('maximumHorizontalScrollOffset') + containerFrame.width) - (containerFrame.width / 2),
217     v: vPct * (scrollView.get('maximumVerticalScrollOffset') + containerFrame.height) - (containerFrame.height / 2)
218   };
219 
220   equals(scrollView.get('horizontalScrollOffset'), newOffsets.h, "Content view left offset remains centered to visible center when zoomed into contentView");
221   equals(scrollView.get('verticalScrollOffset'), newOffsets.v, "Content view top offset remains centered to visible center when zoomed into contentView");
222 });

373 test("Initial alignments different than alignments", function() {
374   // This view is aligned top-left but with initial position at bottom-right. It should start at maximum offsets; when
375   // scaled out it should instead pin to the minimum offsets.
376   var view = pane.view('scrollView with content view of fixed height and bottom-right initial alignment');
377   equals(view.get('horizontalScrollOffset'), view.get('maximumHorizontalScrollOffset'), "initialHorizontalAlign correctly informs initial position when not scaled out");
378   equals(view.get('verticalScrollOffset'), view.get('maximumVerticalScrollOffset'), "initialVerticalAlign correctly informs initial position when not scaled out");
379 
380   SC.run(function() {
381     view.set('scale', 0.1);
382   });
383 
384   equals(view.get('horizontalScrollOffset'), 0, "horizontalAlign correctly takes over from initialHorizontalAlign when scaled out");
385   equals(view.get('verticalScrollOffset'), 0, "verticalAlign correctly takes over from initialVerticalAlign when scaled out");
