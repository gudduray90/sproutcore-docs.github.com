 11   /**
 12     Handles changes in the layer id.
 13   */
 14   layerIdDidChange: function() {
 15     var layer  = this.get('layer'),
 16         lid    = this.get('layerId'),
 17         lastId = this._lastLayerId;
 18 
 19     if (lid !== lastId) {
 20       // if we had an earlier one, remove from view hash.
 21       if (lastId && SC.View.views[lastId] === this) {
 22         delete SC.View.views[lastId];
 23       }
 24 
 25       // set the current one as the new old one
 26       this._lastLayerId = lid;
 27 
 28       // and add the new one
 29       SC.View.views[lid] = this;
 30 
 31       // and finally, set the actual layer id.
 32       if (layer) { layer.id = lid; }
 33     }
 34   }.observes("layerId"),

 36   /**
 37     This method is called whenever the receiver's parentView has changed.
 38     The default implementation of this method marks the view's display
 39     location as dirty so that it will update at the end of the run loop.
 40 
 41     You will not usually need to override or call this method yourself, though
 42     if you manually patch the parentView hierarchy for some reason, you should
 43     call this method to notify the view that it's parentView has changed.
 44 
 45     @returns {SC.View} receiver
 46   */
 47   parentViewDidChange: function() {
 48     this.recomputeIsVisibleInWindow() ;
 49 
 50     this.resetBuildState();
 51     this.set('layerLocationNeedsUpdate', YES) ;
 52     this.invokeOnce(this.updateLayerLocationIfNeeded) ;
 53 
 54     // We also need to iterate down through the view hierarchy and invalidate
 55     // all our child view's caches for 'pane', since it could have changed.
 56     //
 57     // Note:  In theory we could try to avoid this invalidation if we
 58     //        do this only in cases where we "know" the 'pane' value might
 59     //        have changed, but those cases are few and far between.
 60 
 61     this._invalidatePaneCacheForSelfAndAllChildViews();
 62 
 63     return this ;
 64   },

 66   /** @private
 67     We want to cache the 'pane' property, but it's impossible for us to
 68     declare a dependence on all properties that can affect the value.  (For
 69     example, if our grandparent gets attached to a new pane, our pane will
 70     have changed.)  So when there's the potential for the pane changing, we
 71     need to invalidate the caches for all our child views, and their child
 72     views, and so on.
 73   */
 74   _invalidatePaneCacheForSelfAndAllChildViews: function () {
 75     var childView, childViews = this.get('childViews'),
 76         len = childViews.length, idx ;
 77 
 78     this.notifyPropertyChange('pane');
 79 
 80     for (idx=0; idx<len; ++idx) {
 81       childView = childViews[idx];
 82       if (childView._invalidatePaneCacheForSelfAndAllChildViews) {
 83         childView._invalidatePaneCacheForSelfAndAllChildViews();
 84       }
 85     }
 86   },

 92   /**
 93     Insert the view into the the receiver's childNodes array.
 94 
 95     The view will be added to the childNodes array before the beforeView.  If
 96     beforeView is null, then the view will be added to the end of the array.
 97     This will also add the view's rootElement DOM node to the receivers
 98     containerElement DOM node as a child.
 99 
100     If the specified view already belongs to another parent, it will be
101     removed from that view first.
102 
103     @param {SC.View} view
104     @param {SC.View} beforeView
105     @returns {SC.View} the receiver
106   */
107   insertBefore: function(view, beforeView) {
108     view.beginPropertyChanges(); // limit notifications
109 
110     // remove view from old parent if needed.  Also notify views.
111     if (view.get('parentView')) { view.removeFromParent() ; }
112     if (this.willAddChild) { this.willAddChild(view, beforeView) ; }
113     if (view.willAddToParent) { view.willAddToParent(this, beforeView) ; }
114 
115     // set parentView of child
116 117     view.set('parentView', this);
118 
119     // add to childView's array.
120     var idx, childViews = this.get('childViews') ;
121     if (childViews.needsClone) { this.set(childViews = []); }
122     idx = (beforeView) ? childViews.indexOf(beforeView) : childViews.length;
123     if (idx<0) { idx = childViews.length ; }
124     childViews.insertAt(idx, view) ;
125 
126     // The DOM will need some fixing up, note this on the view.
127     if(view.parentViewDidChange) view.parentViewDidChange();
128     if(view.layoutDidChange) view.layoutDidChange();
129 
130     view.endPropertyChanges();
131 
132     // Make sure all notifications are delayed since the appending
133     // doesn't complete until the end of the RunLoop
134     // There may be better ways to do this than with invokeLast,
135     // but it's the best I can do for now - PDW
136     this.invokeLast(function(){
137       var pane = view.get('pane');
138       if(pane && pane.get('isPaneAttached')) {
139         view._notifyDidAppendToDocument();
140       }
141     });
142 
143     // Even though its layer has not necessarily been created, the child views 
144     // are added immediately. Hence notify views immediately.
145     if (this.didAddChild) { this.didAddChild(view, beforeView) ; }
146     if (view.didAddToParent) { view.didAddToParent(this, beforeView) ; }
147 
148     return this ;
149   },

151   removeChild: function(original, view) {
152     if (!view) { return this; } // nothing to do
153     if (view.parentView !== this) {
154       throw "%@.removeChild(%@) must belong to parent".fmt(this,view);
155     }
156     // notify views
157     if (view.willRemoveFromParent) { view.willRemoveFromParent() ; }
158     if (this.willRemoveChild) { this.willRemoveChild(view) ; }
159 
160     original(view);
161 
162     // The DOM will need some fixing up, note this on the view.
163     if(view.parentViewDidChange) view.parentViewDidChange() ;
164 
165     // notify views
166     if (this.didRemoveChild) { this.didRemoveChild(view); }
167     if (view.didRemoveFromParent) { view.didRemoveFromParent(this) ; }
168 
169     return this;

172   /**
173     Replace the oldView with the specified view in the receivers childNodes
174     array. This will also replace the DOM node of the oldView with the DOM
175     node of the new view in the receivers DOM.
176 
177     If the specified view already belongs to another parent, it will be
178     removed from that view first.
179 
180     @param view {SC.View} the view to insert in the DOM
181     @param view {SC.View} the view to remove from the DOM.
182     @returns {SC.View} the receiver
183   */
184   replaceChild: function(view, oldView) {
185     // suspend notifications
186     view.beginPropertyChanges();
187     oldView.beginPropertyChanges();
188     this.beginPropertyChanges();
189 
190     this.insertBefore(view,oldView).removeChild(oldView) ;
191 
192     // resume notifications
193     this.endPropertyChanges();
194     oldView.endPropertyChanges();
195     view.endPropertyChanges();
196 
197     return this;
198   },

342     Usually called by parentViewDidChange, this resets the build state (calling resetBuild in the process).

344   resetBuildState: function() {
345     if (this.isBuildingIn) {
346       this.buildInDidCancel();
347       this.isBuildingIn = NO;
348     }
349     if (this.isBuildingOut) {
350       this.buildOutDidCancel();
351       this.isBuildingOut = NO;
352     }
353 
354     // finish cleaning up
355     this.buildingInTo = null;
356     this.buildingOutFrom = null;
357 
358     this.resetBuild();

397     // if we are already building out, do nothing.

401     if (this.isBuildingIn) {

403     }

405     // in any case, we need to build out
406     this.isBuildingOut = YES;
407     this.isBuildingIn = NO;
