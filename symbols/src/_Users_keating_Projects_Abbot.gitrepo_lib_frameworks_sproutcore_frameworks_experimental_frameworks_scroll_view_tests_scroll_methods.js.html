 10 var pane, view , view2;
 11 var appleURL='http://photos4.meetupstatic.com/photos/event/4/6/9/9/600_4518073.jpeg';
 12 module("SC.ScrollView",{
 13 	setup: function() {
 14 	  SC.RunLoop.begin();
 15 	    pane = SC.MainPane.create({
 16 		  childViews: [
 17 		   SC.ScrollView.extend({
 18 		     contentView: SC.ImageView.design({value: appleURL, layout: {height:4000, width:4000}})
 19 		   }),
 20 		   SC.ScrollView.extend({
 21 		     contentView: SC.ImageView.design({value: appleURL, layout: {height:2000, width:2000}})
 22 		   })
 23 		   
 24 		   ],
 25  26 		   
 27 		  expectedVertLine: function(line) {
 28 			var ret = view.get('verticalLineScroll')*line;
 29 			var alt = view.get('maximumVerticalScrollOffset');
 30 			ret = (ret > alt)? alt : ret;
 31 		    return ret;
 32 		  },
 33 		  expectedHorzLine: function(line) {
 34 			var ret = view.get('horizontalLineScroll')*line;
 35 			var alt = view.get('maximumHorizontalScrollOffset');
 36 			ret = (ret > alt)? alt : ret;
 37 		    return ret;
 38 		  },
 39 		  expectedVertPage: function(page) {
 40 			var ret = view.get('verticalPageScroll')*page;
 41 			var alt = view.get('maximumVerticalScrollOffset');
 42 			ret = (ret > alt)? alt : ret;
 43 		    return ret;
 44 		  },
 45 		  expectedHorzPage: function(page) {
 46 			var ret = view.get('horizontalPageScroll')*page;
 47 			var alt = view.get('maximumHorizontalScrollOffset');
 48 			ret = (ret > alt)? alt : ret;
 49 		    return ret;
 50 		  }
 51 		});
 52 		
 53 		
 54 		pane.append(); // make sure there is a layer...
 55 	  SC.RunLoop.end();	
 56 	  view = pane.childViews[0];
 57 	  view.get('containerView').get('frame').height = 100;
 58 	  view.get('containerView').get('frame').width = 100;
 59 
 60 	  view2 = pane.childViews[1];
 61 	  view2.get('containerView').get('frame').height = 100;
 62 	  view2.get('containerView').get('frame').width = 100;
 63 	  
 64 	},
 65 	
 66 	teardown: function() {
 67     	pane.remove();
 68     	pane = view = null ;
 69   	}
 70 });

 74 test("Scrolling to a certain co-ordinate of the container view", function() {
 75 	equals(view.get('horizontalScrollOffset'), 0, "Initial horizontal offset must be zero");
 76 	equals(view.get('verticalScrollOffset'), 0, "Initial vertical offset must be zero");
 77 	SC.RunLoop.begin();
 78 	view.scrollTo(100, 100);
 79 	SC.RunLoop.end();
 80 	
 81 	equals(view.get('horizontalScrollOffset'), 100, "After scrolling to 100, horizontal offset must be");
 82 	equals(view.get('verticalScrollOffset'), 100, "After scrolling to 100, vertical offset must be");
 83 	view.scrollTo(5000, 5000);
 84 	equals(view.get('horizontalScrollOffset'), view.get('maximumHorizontalScrollOffset'), "After scrolling to 400, horizontal offset must be maximum");
 85 	equals(view.get('verticalScrollOffset'), view.get('maximumVerticalScrollOffset'), "After scrolling to 400, vertical offset must be maximum");
 86 });

 88 test("Scrolling relative to the current possition of the container view", function() {
 89 	equals(view.get('horizontalScrollOffset'), 0, "Initial horizontal offset must be zero");
 90 	equals(view.get('verticalScrollOffset'), 0, "Initial vertical offset must be zero");
 91 	view.scrollBy(100, 100);
 92 	equals(view.get('horizontalScrollOffset'), 100, "After scrolling by 100, horizontal offset must be");
 93 	equals(view.get('verticalScrollOffset'), 100, "After scrolling by 100, vertical offset must be");
 94 	view.scrollBy(100, 100);
 95 	equals(view.get('horizontalScrollOffset'), 200, "After scrolling by 100, horizontal offset must be");
 96 	equals(view.get('verticalScrollOffset'), 200, "After scrolling by 100, vertical offset must be");
 97 	view.scrollBy(5000, 5000);
 98 	equals(view.get('horizontalScrollOffset'), view.get('maximumHorizontalScrollOffset'), "After scrolling by 400, horizontal offset must be maximum");
 99 	equals(view.get('verticalScrollOffset'), view.get('maximumVerticalScrollOffset'), "After scrolling by 400, vertical offset must be maximum");
100 });

102 test("Scrolling through line by line", function() {
103 	var line = 3;
104 	equals(view.get('horizontalScrollOffset'), 0, "Initial horizontal offset must be zero");
105 	equals(view.get('verticalScrollOffset'), 0, "Initial vertical offset must be zero");
106 	view.scrollDownLine(line);
107 	equals(view.get('horizontalScrollOffset'), 0, "After scrolling down by lines, horizontal offset is unchanged");
108 	equals(view.get('verticalScrollOffset'), pane.expectedVertLine(line), "After scrolling down by lines, vertical offset must be");
109 	view.scrollUpLine(line);
110 });

112 test("maximumHorizontalScrollOffset() returns the maximum horizontal scroll dimention", function() {
113   var old_horizontalScrollOffset=2;
114   var old_verticalScrollOffset=2;
115 
116   view2.set('horizontalScrollOffset',old_horizontalScrollOffset);
117   view2.set('verticalScrollOffset',old_verticalScrollOffset);
118   view2.scrollBy(5000, 0);
119   view2.get('horizontalScrollOffset');
120 
121   equals(view2.get('horizontalScrollOffset'),1900, 'maximum y coordinate should be 1900');	
122  
123   view2.set('horizontalScrollOffset',old_horizontalScrollOffset);
124   view2.set('verticalScrollOffset',old_verticalScrollOffset);
125   view2.scrollBy(-5000,0);
126   equals(view2.get('horizontalScrollOffset'),0, 'minimum y coordinate should be 0');	
127   	
128 });
