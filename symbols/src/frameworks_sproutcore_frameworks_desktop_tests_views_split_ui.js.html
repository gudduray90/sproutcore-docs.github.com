 10 (function() {
 11   var pane = SC.ControlTestPane.design()
 12     .add("horizontal", SC.SplitView, {
 13 	  layout: { right: 20, bottom: 20, width: 100, height: 23 },
 14       layoutDirection: SC.LAYOUT_HORIZONTAL,canCollapseViews: NO
 15     })
 16 
 17     .add("vertical", SC.SplitView, {
 18       layoutDirection: SC.LAYOUT_VERTICAL,canCollapseViews: NO
 19     })
 20   	.add("collapsable", SC.SplitView, {
 21       layoutDirection: SC.LAYOUT_HORIZONTAL,canCollapseViews: YES
 22     });
 23 
 24   // ..........................................................
 25   // TEST VIEWS
 26   //
 27   module('SC.SplitView ui', {
 28     setup: function(){
 29       htmlbody('<style> .sc-static-layout { border: 1px red dotted; } </style>');
 30       pane.standardSetup().setup();
 31     },
 32     teardown: function(){
 33       pane.standardSetup().teardown();
 34       clearHtmlbody();
 35     }
 36   });
 37 
 38   test("Horizontal and Vertical SplitView", function() {
 39   	var view = pane.view('horizontal');
 40     ok(!view.$().hasClass('vertical'), 'should not be vertically divided');
 41     ok(!view.$().hasClass('collapsable'), 'should not be collapsable');
 42     equals(view.get('layoutDirection'),SC.LAYOUT_HORIZONTAL,"the view is divided horizontally");
 43     view = pane.view('vertical');
 44     ok(!view.$().hasClass('horizontal'), 'should not be horizontally divided');
 45 	ok(!view.$().hasClass('collapsable'), 'should not be collapsable');
 46 	equals(view.get('layoutDirection'),SC.LAYOUT_VERTICAL,"the view is divided vertically");
 47   });
 48 
 49   test("can collapse split view", function() {
 50 	var view = pane.view('collapsable');
 51     ok(!view.$().hasClass('vertical'), 'should not be have vertical class');
 52 	equals(view.get('layoutDirection'),SC.LAYOUT_HORIZONTAL);
 53 	equals(view.get('canCollapseViews'),YES,"the view should be collapsable");
 54   });
 55  })();
