  4 /** @class
  5 
  6   @extends SC.DataSource
  7 */
  8 SC.ResourceDataSource = SC.DataSource.extend(
  9 /** @scope TP.DataSource.prototype */ {
 10 
 11   init: function() {
 12     this._cachedResourceRoutes = {};
 13   },
 14   
 15   resourceForRecordType: function(recordType) {
 16     var name, 
 17         result;
 18     
 19     result = this._cachedResourceRoutes[recordType];
 20     if (!result) {
 21       name = SC._object_className(recordType).split('.')[1].decamelize();
 22       result = this._cachedResourceRoutes[recordType] = SC.ModelResource.create({ recordType: recordType, name: name });
 23     }
 24     
 25     return result;
 26   },
 27   
 28   sendRequest: function(route, store, storeKey, id) {
 29     var body,
 30         request;
 31     
 32     if (storeKey) {
 33       id = id || store.idFor(storeKey);
 34       
 35       if (route.httpMethod !== SC.HTTP_METHOD.GET && route.httpMethod !== SC.HTTP_METHOD.DELETE) {
 36         body = route.bodyForRequest(store.readDataHash(storeKey));
 37       }
 38     }
 39     
 40     request = SC.Request.create({
 41       type: route.httpMethod,
 42       isJSON: route.format === SC.RESOURCE_ROUTE_FORMAT.JSON,
 43       isXML: route.format === SC.RESOURCE_ROUTE_FORMAT.XML,
 44       address: route.path(id),
 45       body: body
 46     });
 47     
 48     request.notify(route, route.didComplete, {
 49       store: store,
 50       storeKey: storeKey
 51     }).send();
 52   },
 53   
 54   fetch: function(store, query) {
 55     var recordType = query.get('recordType'),
 56         resource,
 57         route,
 58         path,
 59         request;
 60     
 61     if (query.parameters && query.parameters.route) {
 62       route =  query.parameters.route;
 63       path = query.parameters.path;
 64     } else if (!query.conditions) {
 65       resource = this.resourceForRecordType(recordType);
 66       route = resource.get('routes')['collection'];
 67     }
 68     
 69     if (route) this.sendRequest(route, store);
 70     
 71     return YES;
 72   },
 73   
 74   retrieveRecord: function(store, storeKey, id) {
 75     var recordType = SC.Store.recordTypeFor(storeKey),
 76         resource,
 77         route;
 78 
 79     console.log('retrieveRecord');
 80     resource = this.resourceForRecordType(recordType);
 81     route = resource.get('routes')['member'];
 82 
 83     this.sendRequest(route, store, storeKey, id);
 84 
 85     return YES;
 86   },
 87   
 88   createRecord: function(store, storeKey, params) {
 89     var recordType = store.recordTypeFor(storeKey),
 90         resource,
 91         route;
 92     
 93     console.log('createRecord');
 94     if (params && params.route) {
 95       route = params.route;
 96     } else {
 97       resource = this.resourceForRecordType(recordType);
 98       route = resource.get('routes')['create'];
 99     }
100 
101     this.sendRequest(route, store, storeKey);
102     
103     return YES;
104   },
105   
106   updateRecord: function(store, storeKey, params) {
107     var recordType = store.recordTypeFor(storeKey),
108         resource,
109         route;
110     
111     console.log('updateRecord');
112     if (params && params.route) {
113       route = params.route;
114     } else {
115       resource = this.resourceForRecordType(recordType);
116       route = resource.get('routes')['update'];
117     }
118     
119     this.sendRequest(route, store, storeKey);
120     
121     return YES;
122   },
123   
124   destroyRecord: function(store, storeKey, params) {
125     var recordType = store.recordTypeFor(storeKey),
126         resource,
127         route;
128     
129     console.log('destroyRecord');
130     if (params && params.route) {
131       route = params.route;
132     } else {
133       resource = this.resourceForRecordType(recordType);
134       route = resource.get('routes')['destroy'];
135     }
136 
137     this.sendRequest(route, store, storeKey);
138     
139     return YES;
140   }
141 });
