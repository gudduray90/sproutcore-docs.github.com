 13 var initModels = function(){
 14   NestedRecord.Address = SC.Record.extend({
 15     street: SC.Record.attr(String),
 16     city: SC.Record.attr(String),
 17     state: SC.Record.attr(String, {defaultValue: 'VA'})
 18   });
 19 
 20   NestedRecord.Person = SC.Record.extend({
 21     /** Child Record Namespace */
 22     nestedRecordNamespace: NestedRecord,
 23 
 24     name: SC.Record.attr(String),
 25     address: SC.Record.toOne('NestedRecord.Address', { nested: true }),
 26     children: SC.Record.toMany('NestedRecord.GrandchildRecordTest', { nested: true })
 27   });
 28 
 29   NestedRecord.ParentRecordTest = SC.Record.extend({
 30     /** Child Record Namespace */
 31     nestedRecordNamespace: NestedRecord,
 32 
 33     name: SC.Record.attr(String),
 34     person: SC.Record.toOne('NestedRecord.Person', { nested: true })
 35   });
 36 
 37   NestedRecord.GrandchildRecordTest = SC.Record.extend({
 38     name: SC.Record.attr(String)
 39   });
 40 };

 45 module("Basic SC.Record Functions w/ a Parent > Child > Child", {
 46 
 47   setup: function() {
 48     NestedRecord = SC.Object.create({
 49       store: SC.Store.create()
 50     });
 51     window.NestedRecord = NestedRecord;
 52     store = NestedRecord.store;
 53     initModels();
 54     SC.RunLoop.begin();
 55     testParent = store.createRecord(NestedRecord.ParentRecordTest, {
 56       name: 'Parent Name',
 57       person: {
 58         type: 'Person',
 59         name: 'Albert',
 60         address: {
 61           type: 'Address',
 62           street: '123 Sesame St',
 63           city: 'New York',
 64           state: 'NY'
 65         },
 66         children: [{
 67           name: 'Grandchild Name 1'
 68         },{
 69           name: 'Grandchild Name 2'
 70         }]
 71       }
 72     });
 73     SC.RunLoop.end();
 74   },
 75 
 76   teardown: function() {
 77     delete NestedRecord.ParentRecordTest;
 78     delete NestedRecord.Person;
 79     delete NestedRecord.Address;
 80     //delete window.NestedRecord;
 81     NestedRecord = null;
 82     testParent = null;
 83     store = null;
 84   }
 85 });

240 
241   var a = testParent.getPath('person.address');
242   // Check Model Class information

245 
246   // Check reference information
247   var aKey = a.get('id');
248   storeRef = store.find(NestedRecord.Address, aKey);
249   ok(storeRef, 'checking that the store has the instance of the (parent > child > child) record with proper primary key');
250   equals(a, storeRef, "checking the (parent > child > child) reference is the same as the direct store reference");
251   same(storeRef.get('attributes'), p.readAttribute('address'), "check that the ChildRecord's attributes are the same as the (parent > child.address)'s readAttribute for the reference");

254 test("Basic Write",
255 function() {
256   var oldP, p, key, oldKey, storeRef;
257   var a, parentAttrs;
258   // Test general gets
259   testParent.set('name', 'New Parent Name');
260   equals(testParent.get('name'), 'New Parent Name', "set() should change name attribute");
261   testParent.set('nothing', 'nothing');
262   equals(testParent.get('nothing'), 'nothing', "set should change non-existent property to a new property");
263 
264   // Test Child Record creation
265   oldP = testParent.get('person');
266   oldKey = oldP.get('id');
267   testParent.set('person', {
268     type: 'Person',
269     name: 'Al Gore',
270     address: {
271       type: 'Address',
272       street: '123 Crazy St',
273       city: 'Khacki Pants',
274       state: 'Insanity'
275     }
276   });
277   p = testParent.get('person');
278   // Check Model Class information
279   ok(SC.kindOf(p, SC.Record), "set() with an object creates an actual instance that is a kind of a SC.Record Object");
280   ok(SC.instanceOf(p, NestedRecord.Person), "set() with an object creates an actual instance of a ChildRecordTest Object");
281 
282   // Check reference information
283   key = p.get('id');
284   storeRef = store.find(NestedRecord.Person, key);
285   ok(storeRef, 'after a set() with an object, checking that the store has the instance of the child record with proper primary key');
286   equals(p, storeRef, "after a set with an object, checking the parent reference is the same as the direct store reference");
287   ok((oldKey === key), 'check to see that the old child record has the same key as the new child record');
288 
289   // Check for changes on the child bubble to the parent.
290   p.set('name', 'Child Name Change');
291   equals(p.get('name'), 'Child Name Change', "after a set('name', <new>) on child, checking that the value is updated");
292   ok(p.get('status') & SC.Record.DIRTY, 'check that the child record is dirty');
293   ok(testParent.get('status') & SC.Record.DIRTY, 'check that the parent record is dirty');
294   oldP = p;
295   p = testParent.get('person');
296   same(p, oldP, "after a set('name', <new>) on child, checking to see that the parent has received the changes from the child record");
297   same(testParent.readAttribute('person'), p.get('attributes'), "after a set('name', <new>) on child, readAttribute on the parent should be correct for person child attributes");
298 
299   // Check changes on the address
300   a = testParent.getPath('person.address');
301   a.set('street', '321 Nutty Professor Lane');
302   parentAttrs = testParent.readAttribute('person');
303   same(a.get('attributes'), parentAttrs.address, "after a set('street', <new>) on address child, checking to see that the parent has received the changes from the child record");
304 });

326 test("Modifying Grandchild Nested Record Dirties the Main Record", function () {
327   var childRecord,
328       grandChildren,
329       grandChild0;
330 
331   SC.run(function () {
332     childRecord = testParent.get('person');
333     grandChildren = childRecord.get('children');
334     grandChild0 = grandChildren.objectAt(0);
335   });
336 
337   // Check Model Class information
338   ok(SC.kindOf(grandChild0, SC.Record), "get() creates an actual instance that is a kind of a SC.Record Object");
339   ok(SC.instanceOf(grandChild0, NestedRecord.GrandchildRecordTest), "get() creates an actual instance of a GrandchildRecordTest Object");
340 
341   // Check for changes on the child bubble to the parent.
342   SC.run(function () {
343     grandChild0.set('name', 'Grandchild Name Change');
344   });
345 
346   equals(grandChild0.get('name'), 'Grandchild Name Change', "after a set('name', <new>) on grandchild, checking that the value is updated");
347   ok(grandChild0.get('status') & SC.Record.DIRTY, 'check that the grandchild record is dirty');
348   ok(childRecord.get('status') & SC.Record.DIRTY, 'check that the child record is dirty');
349   ok(testParent.get('status') & SC.Record.DIRTY, 'check that the parent record is dirty');
350 });
