  8 // NOTE: The test below are based on the Data Hashes state chart.  This models
  9 // the "read" event in the NestedStore portion of the diagram.
 10 
 11 var parent, store, child, storeKey, json;
 12 module("SC.NestedStore#readDataHash", {
 13   setup: function() {
 14     parent = SC.Store.create();
 15     
 16     json = {
 17       string: "string",
 18       number: 23,
 19       bool:   YES
 20     };
 21     
 22     storeKey = SC.Store.generateStoreKey();
 23 
 24     parent.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
 25     parent.editables = null; // manually patch to setup test state
 26     
 27     store = parent.chain(); // create nested store
 28     child = store.chain();  // for deep nested
 29   }
 30 });

 36 test("data state=INHERITED, lockOnRead=YES, parent editable=NO", function() {
 37   // preconditions
 38   equals(store.get('lockOnRead'), YES, 'precond - lockOnRead should be YES');
 39   equals(store.storeKeyEditState(storeKey), SC.Store.INHERITED, 'precond - storeKey should be inherited from parent');
 40   var oldrev = store.revisions[storeKey]; // save old rev for testing later
 41 
 42   // perform read
 43   equals(store.readDataHash(storeKey), json, 'should return json');
 44 
 45   // verify
 46   equals(store.storeKeyEditState(storeKey), SC.Store.LOCKED, 'storeKey should be read-locked now');
 47   ok(store.dataHashes.hasOwnProperty(storeKey), 'should copy reference to json');
 48 
 49   // test revisions...
 50   equals(store.revisions[storeKey], oldrev, 'should not change revision');
 51   if (!SC.none(oldrev)) {
 52     ok(store.revisions.hasOwnProperty(storeKey), 'should copy reference to revision');
 53   }
 54 });

 80 test("data state=INHERITED, lockOnRead=YES, parent editable=YES", function() {
 81 
 82   // preconditions
 83   
 84   // first, make parentStore record editable.  an editable record needs to be
 85   // cloned into nested stores on lock to avoid un-monitored edits
 86   parent.readEditableDataHash(storeKey);
 87   equals(parent.storeKeyEditState(storeKey), SC.Store.EDITABLE, 'precond - parent storeKey should be editable');
 88   equals(store.get('lockOnRead'), YES, 'precond - lockOnRead should be YES');
 89   equals(store.storeKeyEditState(storeKey), SC.Store.INHERITED, 'precond - storeKey should be inherited from parent');
 90   var oldrev = store.revisions[storeKey]; // save old rev for testing later
 91 
 92   // perform read
 93   var ret = store.readDataHash(storeKey);
 94   same(ret, json, 'should return equivalent json object');
 95   ok(!(ret === json), 'should return clone of json instance not exact same instance');
 96 
 97   // verify new state
 98   equals(store.storeKeyEditState(storeKey), SC.Store.EDITABLE, 'storeKey should be locked');
 99   ok(store.dataHashes.hasOwnProperty(storeKey), 'should have reference to json');
100 
101   // test revisions...
102   equals(store.revisions[storeKey], oldrev, 'should not change revision');
103   if (!SC.none(oldrev)) {
104     ok(store.revisions.hasOwnProperty(storeKey), 'should copy reference to revision');
105   }
106 });

108 test("data state=LOCKED", function() {
109   
110   // preconditions
111   store.set('lockOnRead', YES); // make sure reading will lock
112   var ret1 = store.readDataHash(storeKey);

161 test("locking deep nested store when top-level parent is editable and middle store is inherited", function() {
162 
163   // first, make the parent store data hash editable
164   json = parent.readEditableDataHash(storeKey);
165   equals(parent.storeKeyEditState(storeKey), SC.Store.EDITABLE, 'parent edit state should be EDITABLE');
166   equals(store.storeKeyEditState(storeKey), SC.Store.INHERITED, 'middle store edit state should be INHERITED');
167   equals(child.storeKeyEditState(storeKey), SC.Store.INHERITED, 'child store edit state should be INHERITED');
168   
169   // now read data hash from child, locking child
170   var json2 = child.readDataHash(storeKey);
171   equals(child.storeKeyEditState(storeKey), SC.Store.EDITABLE, 'child store edit state should be locked after reading data');
172   
173   // now edit the root json and make sure it does NOT propogate.
174   json.newItem = "bar";
175   ok(child.readDataHash(storeKey).newItem !== 'bar', 'child json should not pick up edit from parent store since it is now locked');
176 });
