  8 /** @class
  9 
 10   Implements a complete scroller view for menus.
 11   This class implements the arrows displayed in a menu to scroll.
 12 
 13   The main difference with `SC.ScrollerView` is that there is only vertical
 14   scrollers. Value Syncing between `SC.MenuScrollView` and `SC.MenuScrollerView`
 15   is done using valueBinding.
 16 
 17   @extends SC.CoreScrollerView
 18   @since SproutCore 1.6
 19  */
 20 SC.MenuScrollerView = SC.CoreScrollerView.extend(
 21   /** @scope SC.MenuScrollerView.prototype */{
 22 
 23   /**
 24     @type Array
 25     @default ['sc-menu-scroller-view']
 26     @see SC.View#classNames
 27   */
 28   classNames: ['sc-menu-scroller-view'],
 29 
 30   /**
 31     @type Array
 32     @default ['scrollDown']
 33     @see SC.View#displayProperties
 34    */
 35   displayProperties: ['scrollDown'],
 36 
 37   /**
 38     @type String
 39     @default 'menuScrollerRenderDelegate'
 40    */
 41   renderDelegateName: 'menuScrollerRenderDelegate',
 42 
 43   // ..........................................................
 44   // PROPERTIES
 45   //
 46 
 47   /**
 48     Used to set the scrolling direction of the scroller.
 49 
 50     @type Boolean
 51     @default NO
 52    */
 53   scrollDown: NO,
 54 
 55   /**
 56     The maximum offset value for the scroller.  This will be used to calculate
 57     the internal height/width of the scroller itself. It is not necessarily
 58     the same as the height of a scroll view's content view.
 59 
 60     When set less than the height of the scroller, the scroller is disabled.
 61 
 62  63     @type Number
 64     @default 0
 65    */
 66   maximum: 0,
 67 
 68   /**
 69     YES if enable scrollbar, NO to disable it.  Scrollbars will automatically
 70     disable if the maximum scroll width does not exceed their capacity.
 71 
 72     @type Boolean
 73     @default YES
 74    */
 75   isEnabled: YES,
 76 
 77   /**
 78     This function overrides the default function in SC.Scroller as
 79     menus only have vertical scrolling.
 80 
 81     @field
 82     @type String
 83     @default 'verticalScrollOffset'
 84    */
 85   ownerScrollValueKey: function() {
 86     return 'verticalScrollOffset';
 87   }.property('layoutDirection').cacheable(),
 88 
 89 
 90   // ..........................................................
 91   // INTERNAL SUPPORT
 92   //
 93 
 94   /** @private */
 95   init: function() {
 96     // Set the scrollerThickness based on controlSize
 97     switch (this.get('controlSize')) {
 98     case SC.TINY_CONTROL_SIZE:
 99       this.set('scrollerThickness', SC.MenuScrollerView.TINY_SCROLLER_THICKNESS);
100       break;
101     case SC.SMALL_CONTROL_SIZE:
102       this.set('scrollerThickness', SC.MenuScrollerView.SMALL_SCROLLER_THICKNESS);
103       break;
104 105     case SC.REGULAR_CONTROL_SIZE:
106       this.set('scrollerThickness', SC.MenuScrollerView.REGULAR_SCROLLER_THICKNESS);
107       break;
108     case SC.LARGE_CONTROL_SIZE:
109       this.set('scrollerThickness', SC.MenuScrollerView.LARGE_SCROLLER_THICKNESS);
110       break;
111     case SC.HUGE_CONTROL_SIZE:
112       this.set('scrollerThickness', SC.MenuScrollerView.HUGE_SCROLLER_THICKNESS);
113       break;
114     }
115 
116     return sc_super();
117   },
118 
119   /** @private */
120   mouseEntered: function(evt) {
121     this.set('isMouseOver', YES);
122     this._invokeScrollOnMouseOver();
123   },
124 
125   /** @private */
126   mouseExited: function(evt) {
127     this.set('isMouseOver', NO);
128   },
129 
130   mouseWheel: function (evt) {
131     var el = this.getPath('parentView.containerView.layer'),
132         rawEvent = evt.originalEvent;
133 
134     if (el && rawEvent) {
135       try {
136         if (SC.typeOf(el.fireEvent) === SC.T_FUNCTION) { // IE
137           el.fireEvent(rawEvent.type, rawEvent);
138         } else { // W3C
139           el.dispatchEvent(rawEvent);
140         }
141       } catch (x) {
142         // Can't dispatch the event; give up.
143       }
144     }
145   },
146 
147   /** @private */
148   _sc_scroller_armScrollTimer: function() {
149     if (!this._sc_scrollTimer) {
150       SC.RunLoop.begin();
151       var method = this._sc_scroller_scrollDidChange;
152       this._sc_scrollTimer = this.invokeLater(method, 50);
153       SC.RunLoop.end();
154     }
155   },
156 
157   /** @private */
158   _sc_scroller_scrollDidChange: function() {
159     var now = (new Date()).getTime(),
160         last = this._sc_lastScroll,
161         layer = this.get('layer'),
162         scroll = 0;
163 
164     if (last && (now - last) < 50) {
165       this._sc_scroller_armScrollTimer();
166 
167     } else {
168       this._sc_scrollTimer = null;
169       this._sc_lastScroll = now;
170 
171       SC.RunLoop.begin();
172       if (!this.get('isEnabled')) {
173         this._sc_scrollValue = scroll = layer.scrollTop;
174         this.set('value', scroll); // will now enforce minimum and maximum
175       }
176       SC.RunLoop.end();
177     }
178   },
179 
180   /** @private
181     Scroll the menu if it is is an up or down arrow. This is called by
182     the function that simulates mouseOver.
183    */
184   _scrollMenu: function(){
185     var val = this.get('value'),
186         lineScroll = this.getPath('parentView.verticalLineScroll'),
187         newval;
188 
189     if (this.get('scrollDown')) {
190       newval = val + lineScroll;
191       if (newval <= this.get('maximum')) {
192         this.set('value', newval);
193       }
194     } else {
195       newval = val - lineScroll;
196       if (newval >= 0) {
197         this.set('value', newval);
198       } else if (val <= this.verticalLineScroll && val > 0) {
199         this.set('value', 0);
200       }
201     }
202     return YES;
203   },
204 
205   /** @private
206 
207     We use this function to simulate mouseOver. It checks for the flag
208     isMouseOver which is turned on when mouseEntered is called and turned off
209     when mouseExited is called.
210   */
211   _invokeScrollOnMouseOver: function(){
212     this._scrollMenu();
213     if (this.get('isMouseOver')) {
214       this.invokeLater(this._invokeScrollOnMouseOver, 100);
215     }
216   }
217 });
218

220   @static
221   @type Number
222   @default 18
223 */

225

228   @type Number
229   @default 10
230 */
231 SC.MenuScrollerView.TINY_SCROLLER_THICKNESS = 10;
232

234   @static
235   @type Number
236   @default 14
237 */
238 SC.MenuScrollerView.SMALL_SCROLLER_THICKNESS = 14;
239 
240 /**
241   @static
242   @type Number
243   @default 23
244 */
245 SC.MenuScrollerView.LARGE_SCROLLER_THICKNESS = 23;
246 
247 /**
248   @static
249   @type Number
250   @default 26
251 */
252 SC.MenuScrollerView.HUGE_SCROLLER_THICKNESS = 26;
