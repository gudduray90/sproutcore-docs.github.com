  9 SC.BORDER_BEZEL  = 'sc-bezel-border';
 10 SC.BORDER_BLACK  = 'sc-black-border';
 11 SC.BORDER_GRAY   = 'sc-gray-border';
 12 SC.BORDER_TOP    = 'sc-top-border';
 13 SC.BORDER_BOTTOM = 'sc-bottom-border';
 14 SC.BORDER_NONE   = null ;
 15 
 16 /**
 17   @namespace
 18   @deprecated Border functionality is now included in SC.View.
 19 
 20   The `SC.Border` mixin has be deprecated in favour of applying border properties
 21   to an `SC.View`'s layout object. See {@link SC.View#layout}.
 22 
 23   The Border mixin can be applied to any view to give it a visual border.
 24   In addition to specifying the mixin itself, you should specify the border
 25   style with the borderStyle property on your view.
 26 
 27   Border style can be any predefined CSS class name or a border color.
 28 
 29   If you specify a CSS class name, it must end in "-border". Additionally,
 30   you should set the `borderTop`, `borderRight`, `borderBottom`, and
 31   `borderLeft` properties so SproutCore can accurately account for the size
 32   of your view.
 33 
 34   SproutCore pre-defines several useful border styles including:
 35 
 36    - `SC.BORDER_BEZEL`  -- displays an inlaid bezel
 37    - `SC.BORDER_BLACK`  -- displays a black border
 38    - `SC.BORDER_GRAY`   -- displays a gray border
 39    - `SC.BORDER_TOP`    -- displays a border on the top only
 40    - `SC.BORDER_BOTTOM` -- displays a border on the bottom only
 41    - `SC.BORDER_NONE`   -- disables the border
 42 
 43   @since SproutCore 1.0
 44 */
 45 SC.Border = {
 46 
 47   /**
 48     The thickness of the top border.
 49 
 50     @type Number
 51     @commonTask Border Dimensions
 52   */
 53   borderTop: 0,
 54 
 55   /**
 56     The thickness of the right border.
 57 
 58     @type Number
 59     @commonTask Border Dimensions
 60   */
 61   borderRight: 0,
 62 
 63   /**
 64     The thickness of the bottom border.
 65 
 66     @type Number
 67     @commonTask Border Dimensions
 68   */
 69   borderBottom: 0,
 70 
 71   /**
 72     The thickness of the left border.
 73 
 74     @type Number
 75     @commonTask Border Dimensions
 76   */
 77   borderLeft: 0,
 78 
 79   /**
 80     The style of the border. You may specify a color string (like 'red' or
 81     '#fff'), a CSS class name, or one of:
 82 
 83       - SC.BORDER_BEZEL
 84       - SC.BORDER_BLACK
 85       - SC.BORDER_GRAY
 86       - SC.BORDER_TOP
 87       - SC.BORDER_BOTTOM
 88       - SC.BORDER_NONE
 89 
 90     If you specify a CSS class name, it must end in "-border".
 91   */
 92   borderStyle: SC.BORDER_GRAY,
 93 
 94   /**
 95     Walk like a duck
 96 
 97     @private
 98   */
 99   hasBorder: YES,
100 
101   /**
102     Make sure we re-render if the `borderStyle` property changes.
103     @private
104   */
105   displayProperties: ['borderStyle'],
106 
107   /** @private */
108   _BORDER_REGEXP: (/-border$/),
109 
110   /** @private */
111   initMixin: function() {
112     //@if(debug)
113     SC.warn("Developer Warning: SC.Border is deprecated, please set border in your layout");
114     //@endif
115     this._sc_border_borderStyleDidChange();
116     this._sc_border_borderDimensionsDidChange();
117   },
118 
119   /** @private */
120   renderMixin: function(context, firstTime) {
121     var style = this.get('borderStyle');
122     if (style) {
123       if (this._BORDER_REGEXP.exec(style)) {
124         context.addClass(style);
125       } else context.addStyle('border', '1px '+style+' solid');
126     }
127   },
128 
129   /** @private */
130   _sc_border_borderStyleDidChange: function() {
131     var borderStyle = this.get('borderStyle'),
132         borderSize = SC.Border.dimensions[borderStyle];
133 
134     if (borderSize) {
135       this.beginPropertyChanges();
136       this.set('borderTop', borderSize);
137       this.set('borderRight', borderSize);
138       this.set('borderBottom', borderSize);
139       this.set('borderLeft', borderSize);
140       this.endPropertyChanges();
141     }
142   },
143 
144   /** @private */
145   _sc_border_borderDimensionsDidChange: function(){
146     var borderTop     = this.get('borderTop'),
147         borderRight   = this.get('borderRight'),
148         borderBottom  = this.get('borderBottom'),
149         borderLeft    = this.get('borderLeft');
150     this.adjust({ borderTop: borderTop, borderRight: borderRight, borderBottom: borderBottom, borderLeft: borderLeft });
151   }.observes('borderTop', 'borderRight', 'borderBottom', 'borderLeft')
152 
153 };
154 
155 SC.mixin(SC.Border, {
156   dimensions: {
157     'sc-bezel-border': 1,
158     'sc-black-border': 1,
159     'sc-gray-border': 1,
160     'sc-top-border': 1,
161     'sc-bottom-border': 1
162   }
163 });
