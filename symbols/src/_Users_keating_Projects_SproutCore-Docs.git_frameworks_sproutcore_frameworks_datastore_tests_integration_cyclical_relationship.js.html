  9 module("Cyclical relationships", { 
 10   setup: function() {
 11 
 12     // define the application space
 13     window.AB = SC.Object.create({
 14       store: SC.Store.create().from(SC.Record.fixtures)
 15     }); 
 16 
 17     // ..........................................................
 18     // MODEL
 19     // 
 20     
 21     // Describes a single contact
 22     AB.Contact = SC.Record.extend({
 23       name: SC.Record.attr(String),
 24       group: SC.Record.toOne('AB.Group'),
 25       isFavorite: SC.Record.attr(Boolean)
 26     });
 27     
 28     AB.Group = SC.Record.extend({
 29       name: SC.Record.attr(String),
 30       
 31       // dynamically discover contacts for a group using a foreign key
 32       contacts: function() {
 33         var q = SC.Query.local(AB.Contact, "group = {record}", {
 34           record: this
 35         });  
 36         return this.get('store').find(q);
 37       }.property().cacheable(),
 38       
 39       // discover favorite contacts only
 40       favoriteContacts: function() {
 41         return this.get('contacts').filterProperty('isFavorite', YES);
 42       }.property('contacts').cacheable(),
 43       
 44       // we need to reset favoriteContacts whenever the contacts themselves
 45       // change
 46       contactsContentDidChange: function() {
 47         this.notifyPropertyChange('favoriteContacts');
 48       }.observes('.contacts*[]')
 49       
 50     });
 51     
 52     AB.Group.FIXTURES = [
 53       { guid: 100, name: "G1" },
 54       { guid: 101, name: "G2" }
 55     ];
 56 
 57     AB.Contact.FIXTURES = [
 58       { guid: 1,
 59         name: "G1-Fav1",
 60         group: 100,
 61         isFavorite: YES },
 62 
 63       { guid: 2,
 64         name: "G1-Fav2",
 65         group: 100,
 66         isFavorite: YES },
 67 
 68       { guid: 3,
 69         name: "G1-Norm1",
 70         group: 100,
 71         isFavorite: NO },
 72 
 73       { guid: 4,
 74         name: "G2-Fav1",
 75         group: 101,
 76         isFavorite: YES },
 77 
 78       { guid: 5,
 79         name: "G1-Norm1",
 80         group: 101,
 81         isFavorite: NO }
 82     ];
 83     
 84     
 85     SC.RunLoop.begin();
 86     
 87   },
 88   
 89   teardown: function() {
 90     SC.RunLoop.end(); 
 91     AB = null;
 92   }
 93 });
 94 
 95 test("getting all contacts in a group", function() {
 96   var group  = AB.store.find(AB.Group, 100);
 97   var expected = AB.store.find(AB.Contact).filterProperty('group', group);
 98   same(group.get('contacts').toArray(), expected, 'contacts');
 99 });
100 
101 test("finding favoriteContacts", function() {
102   var group  = AB.store.find(AB.Group, 100);
103   var expected = AB.store.find(AB.Contact)
104     .filterProperty('group', group)
105     .filterProperty('isFavorite', YES);
106     
107   same(group.get('favoriteContacts'), expected, 'contacts');
108   
109   var c = AB.store.find(AB.Contact, 4) ;
110   c.set('group', group); // move to group...
111   SC.RunLoop.end();
112   SC.RunLoop.begin();
113   
114   expected.push(c);
115   same(group.get('favoriteContacts'), expected, 'favoriteContacts after adding extra');
116   
117 });
