  2 // SC.Statechart Unit Test

  4 /*globals SC statechart */
  5 
  6 var obj1, rootState1, stateA, stateB;
  7 var obj2;
  8 
  9 module("SC.Statechart: Create Statechart with Assigned Root State Tests", {
 10   setup: function() {
 11     obj1 = SC.Object.extend(SC.StatechartManager, {
 12       rootState: SC.State.design({
 13         
 14         initialSubstate: 'a',
 15         
 16         a: SC.State.design({
 17           foo: function() {
 18             this.gotoState('b');
 19  20           }
 21         }),
 22         
 23         b: SC.State.design({
 24           bar: function() {
 25             this.gotoState('a');
 26           }
 27         })
 28         
 29       })
 30     });
 31     
 32     obj1 = obj1.create();
 33     rootState1 = obj1.get('rootState');
 34     stateA = obj1.getState('a');
 35     stateB = obj1.getState('b');
 36     
 37     obj2 = SC.Object.extend(SC.StatechartManager, {
 38       autoInitStatechart: NO,
 39       rootState: SC.State.design()
 40     });
 41     
 42     obj2 = obj2.create();
 43   },
 44   
 45   teardown: function() {
 46     obj1 = rootState1 = stateA = stateB = null;
 47     obj2 = null;
 48   }
 49 });
 50 
 51 test("check obj1", function() {
 52   ok(obj1.get('isStatechart'), "obj should be statechart");
 53   ok(obj1.get('statechartIsInitialized'), "obj should be an initialized statechart");
 54   ok(SC.kindOf(rootState1, SC.State), "root state should be kind of SC.State");
 55   equals(obj1.get('initialState'), null, "obj initialState should be null");
 56   
 57   ok(stateA.get('isCurrentState'), "state A should be current state");
 58   ok(!stateB.get('isCurrentState'), "state B should not be current state");
 59   
 60   equals(rootState1.get('owner'), obj1, "root state's owner should be obj");
 61   equals(stateA.get('owner'), obj1, "state A's owner should be obj");
 62   equals(stateB.get('owner'), obj1, "state B's owner should be obj");
 63   
 64   obj1.sendEvent('foo');
 65   
 66   ok(!stateA.get('isCurrentState'), "state A should not be current state");
 67   ok(stateB.get('isCurrentState'), "state B should be current state");
 68 });
 69 
 70 test("check obj2", function() {
 71   ok(obj2.get('isStatechart'), "obj should be statechart");
 72   ok(!obj2.get('statechartIsInitialized'), "obj not should be an initialized statechart");
 73   
 74   obj2.initStatechart();
 75   
 76   ok(obj2.get('statechartIsInitialized'), "obj should be an initialized statechart");
 77 });
