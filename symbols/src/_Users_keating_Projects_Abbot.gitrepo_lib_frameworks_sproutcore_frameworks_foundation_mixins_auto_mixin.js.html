  8 /**
  9   @namespace 
 10 
 11   Use this mixin to automatically mix in a a collection of mixins into all
 12   child views created _by the view_ (that are created at view initialization).
 13   
 14   @since SproutCore 1.0
 15 */
 16 SC.AutoMixin = {
 17 
 18   /**
 19     The mixins to automatically mix in.
 20     @property
 21   */
 22   autoMixins: [],
 23   
 24   /**
 25     @private
 26     Override createChildViews to mix in the mixins defined in autoMixins.
 27   */
 28   createChildView: function(v, attrs) {
 29     if (!attrs) attrs = {};
 30     attrs.owner = attrs.parentView = this;
 31     attrs.isVisibleInWindow = this.get('isVisibleInWindow');
 32     if (!attrs.page) attrs.page = this.page;
 33     
 34     var applyMixins = this.get("autoMixins");
 35     applyMixins.push(attrs);
 36     
 37     v = v.create.apply(v, applyMixins);
 38     return v;
 39   }
 40 
 41 };
