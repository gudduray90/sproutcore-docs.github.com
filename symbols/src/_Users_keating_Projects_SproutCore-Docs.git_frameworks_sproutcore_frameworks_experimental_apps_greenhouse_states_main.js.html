 87

 90 
 91       enterState: function(){

104         this.gotoState('syncRunLoops');
105       }

107 
108     syncRunLoops: SC.State.design({
109 
110       enterState: function(){
111         this._setupRunLoops();
112         this._grabDropTargets();
113         this._setupGreenhouse();
114         this._setupEventBlocker();
115         this.gotoState('readyWaiting');
116         //this.invokeLater(function(){this.gotoState('readyWaiting');}); //totally cheating!!
117       },
118       exitState: function(){
119 
120       },
121 
122       // ..........................................................
123       // Monkey-Patch Run Loop
124       // 
125       _setupRunLoops: function(){
126         var iframe = Greenhouse.get('iframe'), innerBegin, outerBegin, innerEnd, outerEnd, outerSC = SC;
127 
128         // ..........................................................
129         // run loop patches...
130         // 
131         outerBegin = function() {    
132           var runLoop = outerSC.RunLoop.currentRunLoop;
133           if (!runLoop) runLoop = outerSC.RunLoop.currentRunLoop = outerSC.RunLoop.runLoopClass.create();
134           runLoop.beginRunLoop();
135           return outerSC.RunLoop ;
136         };
137         outerEnd = function() {
138           var runLoop = outerSC.RunLoop.currentRunLoop;
139           if (!runLoop) {
140             throw "SC.RunLoop.end() called outside of a runloop!";
141           }
142           runLoop.endRunLoop();
143           return outerSC.RunLoop ;
144         };
145 
146         innerBegin = function() {    
147           var runLoop = iframe.SC.RunLoop.currentRunLoop;
148           if (!runLoop) runLoop = iframe.SC.RunLoop.currentRunLoop = iframe.SC.RunLoop.runLoopClass.create();
149           runLoop.beginRunLoop();
150           return iframe.SC.RunLoop ;
151         };
152 
153         innerEnd = function() {
154           var runLoop = iframe.SC.RunLoop.currentRunLoop;
155           if (!runLoop) {
156             throw "SC.RunLoop.end() called outside of a runloop!";
157           }
158           runLoop.endRunLoop();
159           return iframe.SC.RunLoop ;
160         };
161 
162         //outer begin
163         outerSC.RunLoop.begin = function() { 
164           //console.log('outer begin');
165           var outer = outerBegin();
166           innerBegin();
167           return outer;
168         };
169 
170         //inner begin
171         iframe.SC.RunLoop.begin = function() {
172           //console.log('inner begin');
173           var inner = innerBegin();
174           outerBegin();
175           return inner;
176         };
177 
178         //inner end
179         iframe.SC.RunLoop.end = function() {
180           //console.log('inner end');
181           outerEnd();
182           return innerEnd();
183         };
184 
185         //Outer End
186         outerSC.RunLoop.end = function() {
187           //console.log('outer end');
188           innerEnd();
189           return outerEnd();
190         };
191        },
192 
193       _grabDropTargets: function(){
194         var iframe = Greenhouse.get('iframe'), 
195             innerTargets,
196             webViewFrame,
197             dropContainerLayout,
198             webView = Greenhouse.appPage.get('webView');
199 
200         var pv = webView.get('parentView');
201           webViewFrame = webView.get('frame');
202         webViewFrame = pv.convertFrameToView(webViewFrame, null);
203         //add the drop container to the adjusted layout
204         dropContainerLayout = iframe.SC.designPage.getPath('designMainPane.container').layout;
205         
206         if(dropContainerLayout){
207           webViewFrame.x += dropContainerLayout.left;
208           webViewFrame.y += dropContainerLayout.top;
209         }
210         //assign this to 
211         Greenhouse._webViewFrame = webViewFrame;
212         
213         //add existing targets
214         innerTargets = iframe.SC.Drag._dropTargets;
215 
216         for(var dt in innerTargets){
217           if(innerTargets.hasOwnProperty(dt)){
218             SC.Drag.addDropTarget(innerTargets[dt]);
219           }
220         }
221 
222         //make sure we get any new ones
223         iframe.SC.Drag.addDropTarget = function(target) {
224           iframe.SC.Drag._dropTargets[iframe.SC.guidFor(target)] = target ;
225           SC.Drag._dropTargets[iframe.SC.guidFor(target)] = target ;
226         };
227 
228 
229         iframe.SC.Drag.removeDropTarget = function(target) {
230           delete iframe.SC.Drag._dropTargets[iframe.SC.guidFor(target)] ;
231           delete SC.Drag._dropTargets[iframe.SC.guidFor(target)];
232         };
233         
234         //make sure all drags have an iframeTargetOffset...
235         SC.Drag.start = function(ops) {
236           var ret = this.create(ops);
237           ret.globalTargetOffset = Greenhouse._webViewFrame;
238           ret.startDrag();
239           return ret;
240         };
241 
242         //all inner drags are actually outer drags
243         iframe.SC.Drag.start = SC.Drag.start;
244 
245 
246         SC.Drag.prototype._findDropTarget = function(evt) {
247           var loc = { x: evt.pageX, y: evt.pageY }, globalOffset = this.globalTargetOffset ;
248           var target, frame ;
249           var ary = this._dropTargets() ;
250           for (var idx=0, len=ary.length; idx<len; idx++) {
251             target = ary[idx] ;
252             // If the target is not visible, it is not valid.
253             if (!target.get('isVisibleInWindow')) continue ;
254 
255             // get clippingFrame, converted to the pane.
256             frame = target.convertFrameToView(target.get('clippingFrame'), null) ;
257             //convert to iframe pane if it is in the winodw
258             if(globalOffset && target.inGlobalOffset){
259               frame.x += globalOffset.x;
260               frame.y += globalOffset.y;
261             }
262             // check to see if loc is inside.  If so, then make this the drop target
263             // unless there is a drop target and the current one is not deeper.
264             if(SC.pointInRect(loc, frame)) return target;
265 
266           } 
267           return null ;        };
268         //all inner drags are actually outer drags
269         iframe.SC.Drag.start = SC.Drag.start;
270       },
271 
272       _setupGreenhouse: function(){
273         var iframe = Greenhouse.get('iframe');
274         iframe.SC._Greenhouse = Greenhouse;
275       },
276

279         Greenhouse.set('eventBlocker', eventBlocker);
280       }
281     })
282   })
