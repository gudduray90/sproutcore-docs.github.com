  9 (function() {
 10 
 11 var fakeDelegate, fakeEditor, view;
 12 
 13 fakeDelegate = {
 14   acquireEditor: function() {
 15     return fakeEditor;
 16   },
 17 
 18   shouldDiscardCalled: NO,
 19   shouldDiscardAllowed: YES,
 20   inlineEditorShouldDiscardEditing: function() {
 21     this.shouldDiscardCalled = YES;
 22 
 23     return this.shouldDiscardAllowed;
 24   },
 25 
 26   willDiscardCalled: NO,
 27   inlineEditorWillDiscardEditing: function() {
 28     this.willDiscardCalled = YES;
 29   },
 30 
 31   didDiscardCalled: NO,
 32   inlineEditorDidDiscardEditing: function() {
 33     this.didDiscardCalled = YES;
 34     ok(this.willDiscardCalled, "willDiscard called before didDiscard");
 35 
 36     view._endEditing();
 37   }
 38 };
 39 
 40 fakeEditor = SC.View.create(SC.InlineEditor, {
 41   inlineEditorDelegate: fakeDelegate,
 42 
 43   beginEditing: function(original, editable) {
 44     return original(editable);
 45   }.enhance(),
 46 
 47   discardEditingCalled: NO,
 48   discardEditingAllowed: YES,
 49   discardEditing: function(original) {
 50     this.discardEditingCalled = YES;
 51 
 52     var ret = original();
 53 
 54     return this.discardEditingAllowed ? ret : NO;
 55   }.enhance()
 56 });
 57 
 58 view = SC.View.create(SC.InlineEditable, {
 59   inlineEditorDelegate: fakeDelegate
 60 });
 61 
 62 function reset() {
 63   if(fakeEditor.isEditing) fakeEditor.commitEditing();
 64 
 65   fakeDelegate.shouldDiscardCalled = NO;
 66   fakeEditor.discardEditingCalled = NO;
 67 
 68   fakeDelegate.willDiscardCalled = NO;
 69   fakeDelegate.didDiscardCalled = NO;
 70 }
 71 
 72 module('SC.InlineEditable.discardEditing');
 73 
 74 test("discardEditing should ask shouldDiscard and then call discardEditing", function() {
 75   reset();
 76 
 77   fakeDelegate.shouldDiscardAllowed = YES;
 78   fakeEditor.discardEditingAllowed = YES;
 79 
 80   ok(view.beginEditing(), "began editing");
 81 
 82   ok(view.discardEditing(), "discardEditing successful");
 83 
 84   ok(fakeEditor.discardEditingCalled, "discardEditing called");
 85 
 86   ok(fakeDelegate.shouldDiscardCalled, "shouldDiscard called");
 87 });
 88 
 89 test("discardEditing should fail if shouldDiscard returns false", function() {
 90   reset();
 91 
 92   fakeDelegate.shouldDiscardAllowed = NO;
 93   fakeEditor.discardEditingAllowed = YES;
 94 
 95   ok(view.beginEditing(), "began editing");
 96 
 97   ok(!view.discardEditing(), "discardEditing failed");
 98 
 99   ok(fakeEditor.discardEditingCalled, "discardEditing called");
100 
101   ok(fakeDelegate.shouldDiscardCalled, "shouldDiscard called");
102 });
103 
104 test("discardEditing should cleanup properly on success", function() {
105   reset();
106 
107   // test when successful
108   fakeDelegate.shouldDiscardAllowed = YES;
109   fakeEditor.discardEditingAllowed = YES;
110 
111   ok(view.beginEditing(), "began editing");
112 
113   ok(view.discardEditing(), "discardted editing");
114 
115   ok(view.beginEditing(), "cleaned up successfully on discard");
116 });
117 
118 test("delegate methods are called in order by discardEditing", function() {
119   reset();
120 
121   fakeDelegate.shouldDiscardAllowed = YES;
122   fakeEditor.discardEditingAllowed = YES;
123 
124   ok(view.beginEditing(), "began editing");
125 
126   SC.run(function() {
127     ok(view.discardEditing(), "discardted editing");
128   }, undefined, YES);
129 
130   ok(fakeDelegate.willDiscardCalled, "willDiscard was called");
131 
132   ok(fakeDelegate.didDiscardCalled, "didDiscard was called");
133 });
134 
135 test("delegate methods are not called when discardEditing fails", function() {
136   reset();
137 
138   fakeDelegate.shouldDiscardAllowed = NO;
139   fakeEditor.discardEditingAllowed = YES;
140 
141   ok(view.beginEditing(), "began editing");
142 
143   SC.run(function() {
144     ok(!view.discardEditing(), "discard failed");
145   }, undefined, YES);
146 
147   ok(!fakeDelegate.willDiscardCalled, "willDiscard was not called");
148 
149   ok(!fakeDelegate.didDiscardCalled, "didDiscard was not called");
150 });
151 
152 })();
