 11 test("notifies layoutStyle & frame change", function () {
 12 
 13   var view = SC.View.create();
 14   var layoutStyleCallCount = 0, frameCallCount = 0;
 15 
 16   view.addObserver('layoutStyle', function () { layoutStyleCallCount++; });
 17   view.addObserver('frame', function () { frameCallCount++; });
 18 
 19   SC.run(function () {
 20     // Manually indicate a layout change.
 21     view.layoutDidChange();
 22   });
 23 
 24   equals(frameCallCount, 1, 'should trigger observers for frame');
 25   equals(layoutStyleCallCount, 0, 'should not trigger observers for layoutStyle');
 26 
 27   // Attach to the document.
 28   view.createLayer()._doAttach(document.body);
 29 
 30   equals(frameCallCount, 2, 'should trigger observers for frame when attached to the document');
 31   equals(layoutStyleCallCount, 0, 'should still not trigger observers for layoutStyle');
 32 
 33   SC.run(function () {
 34     view.adjust('top', 20);
 35   });
 36 
 37   equals(frameCallCount, 3, 'should trigger observers for frame when attached to the document');
 38   equals(layoutStyleCallCount, 1, 'should trigger observers for layoutStyle');
 39 
 40   // Clean up.
 41   view.destroy();
 42 });

 44 test("invokes layoutDidChangeFor() on layoutView each time it is called", function () {
 45 
 46   var callCount = 0;
 47   var layoutView = SC.View.create({
 48     layoutDidChangeFor: function (changedView) {
 49       equals(this.get('childViewsNeedLayout'), YES, 'should set childViewsNeedLayout to YES before calling layoutDidChangeFor()');
 50 
 51       equals(view, changedView, 'should pass view');
 52       callCount++;
 53 
 54       // Original
 55       var set = this._needLayoutViews;
 56       if (!set) set = this._needLayoutViews = SC.CoreSet.create();
 57       set.add(changedView);
 58     }
 59   });
 60 
 61   var view = SC.View.create({ layoutView: layoutView });
 62 
 63   SC.run(function () {
 64     view.layoutDidChange();
 65     view.layoutDidChange();
 66     view.layoutDidChange();
 67   });
 68 
 69   equals(callCount, 3, 'should call layoutView.layoutDidChangeFor each time');
 70 
 71   // Clean up.
 72   layoutView.destroy();
 73   view.destroy();
 74 });

219 // The default implementation for viewDidResize calls internal layout-related
220 // methods on child views. This test confirms that child views that do not
221 // support layout do not cause this process to explode.
222 test("Calling viewDidResize on a view notifies its child views", function () {
223   var regularViewCounter = 0, coreViewCounter = 0;
224 
225   var view = SC.View.create({
226     childViews: ['regular', 'core'],
227 
228     regular: SC.View.create({
229       viewDidResize: function () {
230         regularViewCounter++;
231         // Make sure we call the default implementation to
232         // ensure potential blow-uppy behavior is invoked
233         sc_super();
234       }
235     }),
236 
237     core: SC.CoreView.create({
238       viewDidResize: function () {
239         coreViewCounter++;
240         sc_super();
241       }
242     })
243   });
244 
245   view.viewDidResize();
246 
247   equals(regularViewCounter, 1, "regular view's viewDidResize gets called");
248   equals(coreViewCounter, 1, "core view's viewDidResize gets called");
249 
250   // Clean up.
251   view.destroy();
252 });
