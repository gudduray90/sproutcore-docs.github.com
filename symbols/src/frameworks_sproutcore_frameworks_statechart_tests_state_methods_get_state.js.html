110 test("get state x from sibling states", function() {
111   var state,
112       foo = root.getSubstate('a.foo'),
113       bar = root.getSubstate('b.bar'),
114       z = root.getSubstate('c.z');
115       
116   state = foo.getState('x');
117   equals(state.get('fullPath'), 'a.x', "for state foo, should return state a.x for value 'x'");

120   equals(state.get('fullPath'), 'b.x', "for state bar, should return state b.x for value 'x'");

124 });
