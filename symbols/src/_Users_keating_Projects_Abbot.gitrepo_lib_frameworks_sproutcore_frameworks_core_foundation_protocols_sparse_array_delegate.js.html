  8 /** @namespace
  9 
 10   Delegate that provides data for a sparse array.  If you set the delegate for
 11   a sparse array to an object that implements one or more of these methods,
 12   they will be invoked by the sparse array to fetch data or to update the
 13   array content as needed.
 14 
 15   Your object does not need to implement all of these methods, but it should
 16   at least implment the sparseArrayDidRequestIndex() method.
 17 
 18   @since SproutCore 1.0
 19 */
 20 SC.SparseArrayDelegate = {
 21 
 22   /**
 23     Invoked when an object requests the length of the sparse array and the
 24     length has not yet been set.  You can implement this method to update
 25     the length property of the sparse array immediately or at a later time
 26     by calling the provideLength() method on the sparse array.
 27 
 28     This method will only be called once on your delegate unless you
 29     subsequently call provideLength(null) on the array, which will effectively
 30     "empty" the array and cause the array to invoke the delegate again the
 31     next time its length is request.
 32 
 33     If you do not set a length on the sparse array immediately, it will return
 34     a length of 0 until you provide the length.
 35 
 36     @param {SC.SparseArray} sparseArray the array that needs a length.
 37     @returns {void}
 38   */
 39   sparseArrayDidRequestLength: function(sparseArray) {
 40     // Default does nothing.
 41   },
 42 
 43   /**
 44     Invoked when an object requests an index on the sparse array that has not
 45     yet been set.  You should implement this method to set the object at the
 46     index using provideObjectsAtIndex() or provideObjectsInRange() on the
 47     sparse array.  You can call these methods immediately during this handler
 48     or you can wait and call them at a later time once you have loaded any
 49     data.
 50 
 51     This method will only be called when an index is requested on the sparse
 52     array that has not yet been filled.  If you have filled an index or range
 53     and you would like to reset it, call the objectsDidChangeInRange() method
 54     on the sparse array.
 55 
 56     Note that if you implement the sparseArrayDidRequestRange() method, that
 57     method will be invoked instead of this one whenever possible to allow you
 58     to fill in the array with the most efficiency possible.
 59 
 60     @param {SC.SparseArray} sparseArray the sparse array
 61     @param {Number} index the requested index
 62     @returns {void}
 63   */
 64   sparseArrayDidRequestIndex: function(sparseArray, index) {
 65 
 66   },
 67 
 68   /**
 69     Alternative method invoked when an object requests an index on the
 70     sparse array that has not yet been set.  If you set the
 71     rangeWindowSize property on the Sparse Array, then all object index
 72     requests will be expanded to to nearest range window and then this
 73     method will be called with that range.
 74 
 75     You should fill in the passed range by calling the provideObjectsInRange()
 76     method on the sparse array.
 77 
 78     If you do not implement this method but set the rangeWindowSize anyway,
 79     then the sparseArrayDidRequestIndex() method will be invoked instead.
 80 
 81     Note that the passed range is a temporary object.  Be sure to clone it if
 82     you want to keep the range for later use.
 83 
 84     @param {SC.SparseArray} sparseArray the sparse array
 85     @param {Range} range read only range.
 86     @returns {void}
 87   */
 88   sparseArrayDidRequestRange: function(sparseArray, range) {
 89 
 90   },
 91 
 92   /**
 93     Optional delegate method you can use to determine the index of a
 94     particular object.  If you do not implement this method, then the
 95     sparse array will just search the objects it has loaded already.
 96 
 97     @param {SC.SparseArray} sparseArray the sparse array
 98     @param {Object} object the object to find the index of
 99     @return {Number} the index or -1
100     @returns {void}
101   */
102   sparseArrayDidRequestIndexOf: function(sparseArray, object) {
103 
104   },
105 
106   /**
107     Optional delegate method invoked whenever the sparse array attempts to
108     changes its contents.  If you do not implement this method or if you
109     return NO from this method, then the edit will not be allowed.
110 
111     @param {SC.SparseArray} sparseArray the sparse array
112     @param {Number} idx the starting index to replace
113     @param {Number} amt the number if items to replace
114     @param {Array} objects the array of objects to insert
115     @returns {Boolean} YES to allow replace, NO to deny
116 117   */
118   sparseArrayShouldReplace: function(sparseArray, idx, amt, objects) {
119     return NO ;
120   },
121 
122   /**
123     Invoked whenever the sparse array is reset.  Resetting a sparse array
124     will cause it to flush its content and go back to the delegate for all
125     property requests again.
126 
127     @param {SC.SparseArray} sparseArray the sparse array
128     @returns {void}
129   */
130   sparseArrayDidReset: function(sparseArray) {
131   }
132 };
