  8 /*global module test htmlbody ok equals same stop start */
  9 var pane, menu, callCount = 0;
 10 
 11 module("SC.MenuItemView", {
 12   setup: function() {
 13    pane = SC.MainPane.create({
 14      layout: { width: 100, height: 20, centerX: 0, centerY: 0 },
 15      childViews: 'button'.w(),
 16 
 17      button: SC.ButtonView.design({
 18        menuItemAction: function() {
 19          callCount += 1;
 20        }
 21      })
 22    }).append();
 23 
 24    pane.makeFirstResponder(pane.button);
 25 
 26    menu = SC.MenuPane.create({
 27      items: [
 28       { title: 'Send Action', action: 'menuItemAction' }
 29      ]
 30    });
 31 
 32    menu.popup(pane.anchor);
 33   },
 34 
 35   teardown: function() {
 36     pane.remove();
 37     menu.remove();
 38     pane = menu = null;
 39   }
 40 });
 41 
 42 test('Sending an action with no target', function() {
 43   var itemView = menu.get('menuItemViews')[0];
 44   itemView.sendAction();
 45   equals(callCount, 1, 'firstResponder of main pane should be called');
 46 });
