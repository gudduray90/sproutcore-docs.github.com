 17 test("template view should call the function of the associated template", function() {
 18   var view;
 19 
 20   view = SC.TemplateView.create({
 21     templateName: 'test_template',
 22 
 23     templates: SC.Object.create({
 24       test_template: function(dataSource) {
 25         return "<h1 id='twas-called'>template was called</h1>";
 26       }
 27     })
 28   });
 29 
 30   view.createLayer();
 31 
 32   ok(view.$('#twas-called').length, "the named template was called");
 33 });

 35 test("template view should call the function of the associated template with itself as the context", function() {
 36   var view;

 48   });
 49 
 50   view.createLayer();

 63 test("template views return YES to mouseDown if there is a mouseUp method", function() {
 64   var view = SC.TemplateView.create();
 65 
 66   ok(!view.tryToPerform('mouseDown'), "view returns NO if there is no mouseUp method");
 67 
 68   view = SC.TemplateView.create({
 69     mouseUp: function() { }
 70   });
 71 
 72   ok(view.tryToPerform('mouseDown'), "view returns YES if we add a mouseUp method");
 73 });

 75 test("should add a 'sc-hidden' class to a template view when isVisible is true", function() {
 76   var view = SC.TemplateView.create();
 77 
 78   ok(view.get('isVisible'), "precond - views default to being visible");
 79 
 80   view.set('template', function() { return "foo"; });
 81 
 82   view.createLayer();
 83   ok(!view.$().hasClass('sc-hidden'), "does not have hidden class applied");
 84 
 85   SC.run(function() { view.set('isVisible', NO); });
 86   ok(view.$().hasClass('sc-hidden'), "adds hidden class when isVisible changes to NO");
 87 
 88   SC.run(function() { view.set('isVisible', YES); });
 89   ok(!view.$().hasClass('sc-hidden'), "removes hidden class when isVisible changes to YES");
 90 });
 91 
 92 test("should add a 'sc-hidden' class to template views if isVisible is false before their layer is created", function() {
 93   var view = SC.TemplateView.create({
 94     isVisible: false
 95   });
 96 
 97   ok(!view.get('isVisible'), "precond - view is not visible");
 98 
 99   view.set('template', function() { return "foo"; });
100 
101   view.createLayer();
102   ok(view.$().hasClass('sc-hidden'), "adds hidden class when rendering");
103   $(document.body).append(view.$());
104   ok(!view.$().is(':visible'), "should be hidden when hidden class is added");
105   view.$().remove();
106 
107   SC.run(function() { view.set('isVisible', YES); });
108   ok(!view.$().hasClass('sc-hidden'), "removes hidden class when isVisible changes");
109 });

122 test("should invalidate frame cache when layer is created", function() {
123   var pane = SC.MainPane.create().append();
124 
125   var view = SC.TemplateView.create({
126     template: function() { return "foo"; }
127   });
128 
129   var f = view.get('frame');
130   ok(f, "precond - returns a frame object");
131   equals(f.width, 0, "returns zero width because there is no layer");
132 
133   pane.appendChild(view);
134 
135   f = view.get('frame');
136   ok(f, "returns frame object");
137   equals(f.width, view.$().width(), "returns non-zero width");
138   pane.remove();
139 });
