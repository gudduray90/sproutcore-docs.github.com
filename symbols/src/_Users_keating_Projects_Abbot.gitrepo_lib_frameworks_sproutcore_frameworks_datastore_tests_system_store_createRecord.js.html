  8 var store, storeKey, json, hash, hash2;
  9 
 10 module("SC.Store#createRecord", {
 11   setup: function() {
 12     
 13     MyRecordType = SC.Record.extend({
 14       string: SC.Record.attr(String, { defaultValue: "Untitled" }),
 15       number: SC.Record.attr(Number, { defaultValue: 5 }),
 16       bool: SC.Record.attr(Boolean, { defaultValue: YES })
 17     });
 18 
 19     SC.RunLoop.begin();
 20 
 21     store = SC.Store.create();
 22     
 23     json = {
 24       string: "string",
 25       number: 23,
 26       bool:   YES
 27     };
 28     
 29     storeKey = SC.Store.generateStoreKey();
 30 
 31     store.writeDataHash(storeKey, json, SC.Record.READY_CLEAN);
 32 
 33     SC.RunLoop.end();
 34   }
 35 });
 36 
 37 test("create a record", function() {
 38   var sk;
 39   var rec = SC.Record.create();
 40   hash = {
 41     guid: "1234abcd",
 42     string: "abcd",
 43     number: 1,
 44     bool:   NO
 45     };
 46   hash2 = {
 47     string: "abcd",
 48     number: 1,
 49     bool:   NO
 50   };
 51 
 52   rec = store.createRecord(SC.Record, hash);
 53   ok(rec, "a record was created");
 54   sk=store.storeKeyFor(SC.Record, rec.id());
 55   equals(store.readDataHash(sk), hash, "data hashes are equivalent");
 56   equals(rec.id(), "1234abcd", "guids are the same");
 57 
 58   rec = store.createRecord(SC.Record, hash2, "priKey");
 59   ok(rec, "a record with a custom id was created");
 60   sk=store.storeKeyFor(SC.Record, "priKey");
 61   equals(store.readDataHash(sk), hash2, "data hashes are equivalent");
 62   equals(rec.id(), "priKey", "guids are the same");
 63   
 64   equals(store.changelog.length, 2, "The changelog has the following number of entries:");
 65   
 66   
 67 });
 68 
 69 test("Creating an empty (null) record should make the hash available", function() {
 70   
 71   store.createRecord(MyRecordType, null, 'guid8');
 72   var storeKey = store.storeKeyFor(MyRecordType, 'guid8');
 73   
 74   ok(store.readDataHash(storeKey), 'data hash should not be empty/undefined');
 75   
 76 });
