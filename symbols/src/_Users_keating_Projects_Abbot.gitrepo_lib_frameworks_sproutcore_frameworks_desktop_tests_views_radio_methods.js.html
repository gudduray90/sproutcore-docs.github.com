 10 var pane, view ;
 11 module("SC.RadioView Logic", {
 12   setup: function() {
 13     SC.RunLoop.begin();
 14     pane = SC.MainPane.create({
 15       childViews: [
 16         SC.RadioView.extend({
 17           layout: { right: 20, bottom: 20, width: 100, height: 23 },
 18           items: 'Red Green'.w(),
 19           value: 'Red',
 20           isEnabled: YES
 21         })]
 22     });
 23     pane.append(); // make sure there is a layer...
 24     SC.RunLoop.end();
 25     
 26     view = pane.childViews[0];
 27   }, 
 28   
 29   teardown: function() {
 30     pane.remove();
 31     pane = view = null ;
 32   }
 33 });
 34 
 35 test("renders an input tag with appropriate attributes", function() {
 36   equals(view.get('value'), 'Red', 'precon - value should be YES');
 37 
 38   var q = view.$();
 39   equals(q.attr('role'), 'radiogroup', 'should have role=radio');
 40   var radios=view.$('.sc-radio-button');
 41   equals(SC.$(radios[0]).attr('aria-checked'), 'true', 'should be checked');
 42 
 43 });
 44 
 45 test("changing the value should update the group", function() {
 46   var oldValue = view.get('value');
 47   SC.RunLoop.begin();
 48   view.set('value', 'Green');
 49   SC.RunLoop.end();
 50 
 51   ok(view.get('value') !== oldValue, 'precond - should have changed value');
 52   var q = Q$('.sel > span', view.get('layer'));
 53   equals(q.text(), view.get('value'), 'label should have new value');
 54 });
 55 
 56 test("isEnabled should alter group classname and sync with isEnabled property", function() {
 57 
 58   // check initial render state
 59   ok(view.get('isEnabled'), 'isEnabled should match value');
 60   ok(!view.$().hasClass('disabled'), 'should NOT have disabled class');
 61 
 62   // update value -- make sure isEnabled changes.  
 63   SC.RunLoop.begin();
 64   view.set('isEnabled', 0); // make falsy. (but not NO exactly)
 65   SC.RunLoop.end();
 66   
 67   ok(!view.get('isEnabled'), 'isEnabled should now be NO');
 68   ok(view.$().hasClass('disabled'), 'should have disabled class');
 69 
 70   // update isEnabled -- make sure it edits the value
 71   SC.RunLoop.begin();
 72   view.set('isEnabled', YES);
 73   SC.RunLoop.end();
 74   
 75   ok(view.get('isEnabled'), 'isEnabled should match value');
 76   ok(!view.$().hasClass('disabled'), 'should NOT have disabled class');
 77 });
 78 
 79 test("clicking on a radio button will change toggle the value", function() {
 80   equals(view.get('value'), 'Red', 'precond - value should be Red');
 81   var r = view.$('.sc-radio-button');
 82   SC.Event.trigger(r[1], 'mousedown');
 83   SC.Event.trigger(r[1], 'mouseup');
 84   equals(view.get('value'), 'Green', 'value should have changed');
 85 });
 86 
 87 test("pressing mouseDown and then mouseUp anywhere in a radio button should toggle the selection", function() {
 88   var elem = view.get('layer'), 
 89       r = view.$('.sc-radio-button');
 90 
 91   SC.Event.trigger(r[0], 'mousedown');
 92   equals(view.get('value'), 'Red', 'value should not change yet');
 93   ok(r.first().hasClass('active'), 'radio button should be active');
 94   
 95   // simulate mouseUp and browser-native change to control
 96   SC.Event.trigger(r[0],'mouseup');
 97   //input.attr('checked', NO);
 98   // loose focus of the element since it was changed
 99   SC.Event.trigger(r[1],'mousedown');
100   SC.Event.trigger(r[1],'mouseup');
101   
102   ok(!r.first().hasClass('active'), 'radio button should no longer be active');
103   equals(view.get('value'), 'Green', 'value should be undefined (none checked)');
104   
105   elem = null ;
106 });
107 
108 test("isEnabled=NO should add disabled attr to input", function() {
109   SC.RunLoop.begin();
110   view.set('isEnabled', NO);
111   SC.RunLoop.end();
112   
113   ok(view.$().hasClass('disabled'), 'should have disabled attr');
114   
115   ok(view.get('value'), 'precond - value should be true');
116   view.mouseDown();
117   view.mouseUp();
118   ok(view.get('value'), 'value should not change when clicked');
119 });
