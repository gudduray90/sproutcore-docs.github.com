 57 /**
 58   GridView used to position items according to its clippingFrame, which meant
 59   that a GridView could not specify a frame width in order to scroll
 60   horizontally.  This also manifested in a bug where itemsPerRow was based on
 61   frame but clippingFrame would change first so that GridView's would not
 62   update properly when clippingFrame changed and frame did not change.  For
 63   example during a rotation on an iPad.
 64 */
 65 test("GridView items should be positioned according to frame", function () {
 66   var layout,
 67     expectedLayout;
 68 
 69   layout = view.layoutForContentIndex(0);
 70   expectedLayout = { left: 0, top: 0, height: 48, width: 100 };
 71   same(layout, expectedLayout, "Layout of first item should be");
 72 
 73   layout = view.layoutForContentIndex(1);
 74   expectedLayout = { left: 100, top: 0, height: 48, width: 100 };
 75   same(layout, expectedLayout, "Layout of second item should be");
 76 
 77   layout = view.layoutForContentIndex(2);
 78   expectedLayout = { left: 0, top: 48, height: 48, width: 100 };
 79   same(layout, expectedLayout, "Layout of third item should be");
 80 
 81   layout = view.layoutForContentIndex(3);
 82   expectedLayout = { left: 100, top: 48, height: 48, width: 100 };
 83   same(layout, expectedLayout, "Layout of fourth item should be");
 84 
 85   SC.RunLoop.begin();
 86   scrollView.adjust('width', 100);
 87   SC.RunLoop.end();
 88 
 89   layout = view.layoutForContentIndex(0);
 90   expectedLayout = { left: 0, top: 0, height: 48, width: 100 };
 91   same(layout, expectedLayout, "Layout of first item should be");
 92 
 93   layout = view.layoutForContentIndex(1);
 94   expectedLayout = { left: 100, top: 0, height: 48, width: 100 };
 95   same(layout, expectedLayout, "Layout of second item should be");
 96 
 97   layout = view.layoutForContentIndex(2);
 98   expectedLayout = { left: 0, top: 48, height: 48, width: 100 };
 99   same(layout, expectedLayout, "Layout of third item should be");
100 
101   layout = view.layoutForContentIndex(3);
102   expectedLayout = { left: 100, top: 48, height: 48, width: 100 };
103   same(layout, expectedLayout, "Layout of fourth item should be");
104 });

107 /*
108   GridView would adjust all of its item views every time that its frame
109   changed (which happens on scroll), which was very wasteful.  Instead, it was
110   improved to only reposition the nowShowing item views, which improved the
111   efficiency and also allowed the content to be sparse.  Finally, since the
112   position of the item views is only effected by changes to the frame's
113   width and not its x, y or height properties.  We can optimize it further to
114   only adjust item views when the width is different.
115 */
116 test("Optimized re-position of item views when the frame changes.", function () {
117   view.itemViewForContentIndex.expect(0);
118   view.layoutForContentIndex.expect(0);
119 
120   SC.RunLoop.begin();
121   view.notifyPropertyChange('frame');
122   SC.RunLoop.end();
123 
124   view.itemViewForContentIndex.expect(0);
125   view.layoutForContentIndex.expect(0);
126 
127   SC.RunLoop.begin();
128   view.adjust('height', 500);
129   SC.RunLoop.end();
130 
131   view.itemViewForContentIndex.expect(0);
132   view.layoutForContentIndex.expect(0);
133 
134   SC.RunLoop.begin();
135   view.adjust('width', 400);
136   SC.RunLoop.end();
137 
138   // Six item requests, twelve layout requests
139   view.itemViewForContentIndex.expect(6);
140   view.layoutForContentIndex.expect(12);
141 });

144 /**
145   GridView used to not re-compute its layout when the frame changed, which meant
146   that if the itemsPerRow changed the calculated height of the grid would not
147   update to fit the new number of rows.
148 */
149 test("Adjusts layout when the itemsPerRow changes.", function () {
150   view.adjustLayout.expect(1);
151 
152   SC.RunLoop.begin();
153   view.notifyPropertyChange('frame');
154   SC.RunLoop.end();
155 
156   view.adjustLayout.expect(1);
157 
158   SC.RunLoop.begin();
159   view.adjust('height', 500);
160   SC.RunLoop.end();
161 
162   view.adjustLayout.expect(1);
163 
164   SC.RunLoop.begin();
165   view.adjust('width', 400);
166   SC.RunLoop.end();
167 
168   view.adjustLayout.expect(2);
169 });
