 10 // contentPropertyDidChange()

 12 var view, content ;
 13 module('SC.ButtonView#contentPropertyDidChange', {
 14   setup: function() {
 15     content = SC.Object.create();
 16     view = SC.ButtonView.create();
 17     view.set('title', 'hello world');
 18   },
 19   
 20   teardown: function() {
 21     content = null;
 22     view.destroy();
 23   }
 24 });
 25 
 26 test("invoked with key = * whenever content changes", function() {
 27   view.contentPropertyDidChange = function(target, key) {
 28     ok(target === content, 'should pass content object as target target=%@'.fmt(target));
 29     equals(key, '*', 'should pass * as key');    
 30   };
 31   view.set('content', content);
 32 });
 33 
 34 test("should not be invoked when arbitrary keys are changed", function() {
 35   var isTesting = NO, count = 0;
 36   view.contentPropertyDidChange = function(target, key) {
 37     if (!isTesting) return ; //wait until testing should begin...
 38     count++;
 39   };
 40 
 41   view.set('content', content);
 42   
 43   isTesting = YES ;
 44   
 45   content.set('foo', 'foo');
 46   content.set('bar', 'bar');
 47 
 48   equals(count, 0, "method was not invoked");
 49 });
 50 
 51 test("should no longer be invoked when a key is changed on a former content object", function() {
 52   var isTesting = NO;
 53   view.contentPropertyDidChange = function(target, key) {
 54     if (!isTesting) return ; //wait until testing should begin...
 55     ok(NO, 'should not invoke contentPropertyDidChange after content is removed');
 56   };
 57   
 58   view.set('content', content);
 59   content.set('foo', 'foo');
 60   view.set('content', null);
 61   
 62   isTesting= YES ;
 63   content.set('bar', 'bar');
 64 });
 65 
 66 test("should fire even on a content object set when the object is created", function() {
 67   var callCount = 0;
 68   var view = SC.ButtonView.create({
 69     contentPropertyDidChange: function() { callCount++; },
 70     content: content,
 71     contentTitleKey: 'title'
 72   });
 73   
 74   equals(callCount, 1, 'should call contentPropertyDidChange on init to do initial setup');
 75   
 76   content.set('title', 'title');
 77   equals(callCount, 2, 'should call contentPropertyDidChange when changing content.title');
 78 });

105 module("SC.ButtonView#updatePropertyFromContent()", {
106   setup: function() {
107     content = SC.Object.create({ foo: "foo", bar: "bar" });
108     view = SC.ButtonView.create({ content: content });
109   },
110   teardown: function() {
111     content = null ;
112     view.destroy();
113   }
114 }) ;
115 
116 test("copies value of content[key] to this[prop] if changed key.  Gets key from contentKey property", function() {
117   view.contentValueKey = 'foo'; // set key mapping.
118   view.updatePropertyFromContent('value', 'foo', 'contentValueKey');
119   equals(view.get('value'), 'foo', 'should copy foo from content.foo to value');
120 });
121 
122 test("does nothing of changed key does not match contentKey value" ,function() {
123   view.value = "foo";
124   view.contentValueKey = "foo" ;
125   view.updatePropertyFromContent('value', 'bar', 'contentValueKey');
126   equals(view.get('value'), 'foo', 'should not have changed "value"');
127 });
128 
129 test("if contentKey is not passed, assume contentPROPKey", function() {
130   view.contentValueKey = "foo";
131   view.updatePropertyFromContent("value", "foo");
132   equals(view.get('value'), 'foo', 'should have looked at foo since contentValueKey is set to foo');
133 });
134 
135 test("if contentFooKey is not set on receiver, ask displayDelegate", function() {
136   view.displayDelegate = SC.Object.create({ contentValueKey: "foo" });
137   view.updatePropertyFromContent("value", "foo", "contentValueKey");
138   equals(view.get('value'), 'foo', 'should have looked at foo since contentValueKey is set to foo');
139 });
140 
141 test("should be able to get value from a content object that is not SC.Object", function() {
142   view.content = { foo: "foo" };
143   view.contentValueKey = "foo";
144   view.updatePropertyFromContent("value", "foo", "contentValueKey");
145   equals(view.get('value'), 'foo', 'should have looked at foo since contentValueKey is set to foo');
146 });

152 // ..........................................................
153 // updateContentWithValueObserver()
154 // 
155 module("SC.ButtonView#updatePropertyFromContent()", {
156   setup: function() {
157     content = SC.Object.create({ foo: "foo", bar: "bar" });
158     view = SC.ButtonView.create({ 
159       value: "bar",
160       content: content,
161       contentValueKey: "bar",
162       displayDelegate: SC.Object.create({ contentValueKey: "foo" }) 
163     });
164   },
165   teardown: function() {
166     content = null ;
167     view.destroy();
168   }
169 }) ;
170 
171 test("if contentValueKey is set, changing value will be pushed to content", function() {
172   view.set('value', 'baz');
173   equals(content.get('bar'), 'baz', 'should copy from view.value to content');
174 });
175 
176 test("if contentValueKey is undefined, asks display delegate instead", function() {
177   delete view.contentValueKey ;
178   view.set('value', 'baz');
179   equals(content.get('foo'), 'baz', 'should copy from view.value to content');
180 });
181 
182 test("if contentValueKey is not set & displayDelegate not set, does nothing", function() {
183   delete view.contentValueKey;
184   delete view.displayDelegate;
185   view.set('value', 'baz');
186   equals(content.get('foo'), 'foo', 'should not change');
187 });
