 29 // quick and dirty function to combine passed browser props
 30 // with cleanBrowser
 31 function combine(hash) {
 32   var ret = {},
 33       key;
 34   
 35   for (key in cleanBrowser) {
 36     if (!cleanBrowser.hasOwnProperty(key)) continue;
 37     ret[key] = cleanBrowser[key];
 38   }
 39   
 40   for (key in hash) {
 41     if (!hash.hasOwnProperty(key)) continue;
 42     ret[key] = hash[key];
 43   }
 44   
 45   return ret;
 46 }
 47 
 48 function testUserAgent(userAgent, matchers) {
 49   var browser = SC._detectBrowser(userAgent), key;
 50   matchers = combine(matchers);
 51   for (key in matchers) equals(browser[key], matchers[key], "'" + key + "' should be '" + matchers[key] + "'");
 52 }
