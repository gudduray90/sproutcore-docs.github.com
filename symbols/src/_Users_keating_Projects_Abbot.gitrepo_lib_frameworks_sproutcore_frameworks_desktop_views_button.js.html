536   /**
537     Handle a key equivalent if set.  Trigger the default action for the 
538     button.  Depending on the implementation this may vary.
539     
540     @param {String} keystring
541     @param {SC.Event} evt
542     @returns {Boolean}  YES if handled, NO otherwise
543   */
544   performKeyEquivalent: function(keystring, evt) {
545     //If this is not visible
546     if (!this.get('isVisibleInWindow')) return NO;
547 
548     if (!this.get('isEnabled')) return NO;
549     var equiv = this.get('keyEquivalent');
550 
551     // button has defined a keyEquivalent and it matches!
552     // if triggering succeeded, true will be returned and the operation will 
553     // be handeled (i.e performKeyEquivalent will cease crawling the view 
554     // tree)
555     if (equiv) {
556       if (equiv === keystring) return this.triggerAction(evt);
557     
558     // should fire if isDefault OR isCancel.  This way if isDefault AND 
559     // isCancel, responds to both return and escape
560     } else if ((this.get('isDefault') && (keystring === 'return')) ||
561         (this.get('isCancel') && (keystring === 'escape'))) {
562           return this.triggerAction(evt);
563     }
564 
565     return NO; // did not handle it; keep searching
566   },

572   /**
573     This is the standard logic to compute a proposed isSelected state for a
574     new value.  This takes into account the `toggleOnValue`/`toggleOffValue`
575     properties, among other things.  It may return `YES`, `NO`, or
576     `SC.MIXED_STATE`.
577     
578     @param {Object} value
579     @returns {Boolean} return state
580   */
581   computeIsSelectedForValue: function(value) {
582     var targetValue = this.get('toggleOnValue'), state, next ;
583     
584     if (SC.typeOf(value) === SC.T_ARRAY) {
585 
586       // treat a single item array like a single value
587       if (value.length === 1) {
588         state = (value[0] == targetValue) ;
589         
590       // for a multiple item array, check the states of all items.
591       } else {
592         state = null;
593         value.find(function(x) {
594           next = (x == targetValue) ;
595           if (state === null) {
596             state = next ;
597           } else if (next !== state) state = SC.MIXED_STATE ;
598           return state === SC.MIXED_STATE ; // stop when we hit a mixed state.
599         });
600       }
601       
602     // for single values, just compare to the toggleOnValue...use truthiness
603     } else {
604       if(value === SC.MIXED_STATE) state = SC.MIXED_STATE;
605       else state = (value === targetValue) ;
606     }
607     return state ;
608   },

610   /** @private
611     Whenever the button value changes, update the selected state to match.
612   */
613   _button_valueDidChange: function() {

619   /** @private
620     Whenever the selected state is changed, make sure the button value is
621     also updated.  Note that this may be called because the value has just
622     changed.  In that case this should do nothing.
623   */
624   _button_isSelectedDidChange: function() {
625     var newState = this.get('isSelected'),
626         curState = this.computeIsSelectedForValue(this.get('value'));
627     
628     // fix up the value, but only if computed state does not match.
629     // never fix up value if isSelected is set to MIXED_STATE since this can
630     // only come from the value.
631     if ((newState !== SC.MIXED_STATE) && (curState !== newState)) {
632       var valueKey = (newState) ? 'toggleOnValue' : 'toggleOffValue' ;
633       this.set('value', this.get(valueKey));
634     }
635   }.observes('isSelected'),

743   /** @private */
744   touchesDragged: function(evt, touches) {
745     if (!this.touchIsInBoundary(evt)) {
746       if (!this._touch_exited) this.set('isActive', NO);
747       this._touch_exited = YES;
748     } else {
749       if (this._touch_exited) this.set('isActive', YES);

751     }

754     return YES;

757   /** @private */
758   touchEnd: function(touch){
759     this._touch_exited = NO;
760     this.set('isActive', NO); // track independently in case isEnabled has changed
761 
762     if (this.get('buttonBehavior') !== SC.HOLD_BEHAVIOR) {
763       if (this.touchIsInBoundary(touch) && this.get('isEnabled')) {
764         this._action();
765       }
766     }
767 
768     touch.preventDefault();
769     return YES ;

772   /** @private */
773   keyDown: function(evt) {
774     // handle tab key
775      if(!this.get('isEnabled')) return YES;
776     if (evt.which === 9 || evt.keyCode === 9) {
777       var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
778       if(view) view.becomeFirstResponder();
779       else evt.allowDefault();
780       return YES ; // handled
781     }
782     if (evt.which === 13 || evt.which === 32) {
783       this.triggerActionAfterDelay(evt);
784       return YES ; // handled
785     }
786 
787     // let other keys through to browser
788     evt.allowDefault();
789 
790     return NO;
791   },

793   /** @private
794     Perform an action based on the behavior of the button.
795 
796      - toggle behavior: switch to on/off state
797      - on behavior: turn on.
798      - off behavior: turn off.
799      - otherwise: invoke target/action
800   */
801   _action: function(evt, skipHoldRepeat) {
802     switch(this.get('buttonBehavior')) {
803 
804     // When toggling, try to invert like values. i.e. 1 => 0, etc.
805     case SC.TOGGLE_BEHAVIOR:
806       var sel = this.get('isSelected') ;
807       if (sel) {
808         this.set('value', this.get('toggleOffValue')) ;
809       } else {
810         this.set('value', this.get('toggleOnValue')) ;
811       }
812       break ;
813 
814     // set value to on.  change 0 => 1.
815     case SC.TOGGLE_ON_BEHAVIOR:
816       this.set('value', this.get('toggleOnValue')) ;
817       break ;
818 
819     // set the value to false. change 1 => 0
820     case SC.TOGGLE_OFF_BEHAVIOR:
821       this.set('value', this.get('toggleOffValue')) ;
822       break ;
823 
824     case SC.HOLD_BEHAVIOR:
825       this._runHoldAction(evt, skipHoldRepeat);
826       break ;
827 
828     // otherwise, just trigger an action if there is one.
829     default:
830       //if (this.action) this.action(evt);
831       this._runAction(evt);
832     }
833   },
834 
835   /** @private */
836   _runAction: function(evt) {
837     var action = this.get('action'),
838         target = this.get('target') || null,
839         rootResponder = this.getPath('pane.rootResponder');
840         
841     if (action) {
842       if (this._hasLegacyActionHandler()) {
843         // old school... V
844         this._triggerLegacyActionHandler(evt);
845       } else {
846         if (rootResponder) {
847           // newer action method + optional target syntax...
848           rootResponder.sendAction(action, target, this, this.get('pane'), null, this);
849         }
850       }
851     }    
852   },
853 
854   /** @private */
855   _runHoldAction: function(evt, skipRepeat) {
856     if (this.get('isActive')) {
857       this._runAction();
858 
859       if (!skipRepeat) {
860         // This run loop appears to only be necessary for testing
861         SC.RunLoop.begin();
862         this.invokeLater('_runHoldAction', this.get('holdInterval'), evt);
863         SC.RunLoop.end();
864       }
865     }
866   },
867 
868   /** @private */
869   _hasLegacyActionHandler: function() {
870     var action = this.get('action');
871     if (action && (SC.typeOf(action) === SC.T_FUNCTION)) return true;
872     if (action && (SC.typeOf(action) === SC.T_STRING) && (action.indexOf('.') != -1)) return true;
873     return false;
874   },
875   
876   /** @private */
877   _triggerLegacyActionHandler: function( evt )
878   {
879     if (!this._hasLegacyActionHandler()) return false;
880   
881     var action = this.get('action');
882     if (SC.typeOf(action) === SC.T_FUNCTION) this.action(evt);
883     if (SC.typeOf(action) === SC.T_STRING) {
884       eval("this.action = function(e) { return "+ action +"(this, e); };");
885       this.action(evt);
886     }
887   },

891     // focus the text field.

893       this._isFocused = YES ;
894       this.becomeFirstResponder();
895       if (this.get('isVisibleInWindow')) {

897       }
898     }
899   },

902   willLoseKeyResponderTo: function(responder) {
903     if (this._isFocused) this._isFocused = NO ;

907   didAppendToDocument: function() {
908     if(parseInt(SC.browser.msie, 0)===7 && this.get('useStaticLayout')){
909       var layout = this.get('layout'),
910           elem = this.$(), w=0;
911       if(elem && elem[0] && (w=elem[0].clientWidth) && w!==0 && this._labelMinWidthIE7===0){
912         var label = this.$('.sc-button-label'),
913             paddingRight = parseInt(label.css('paddingRight'),0),
914             paddingLeft = parseInt(label.css('paddingLeft'),0),
915             marginRight = parseInt(label.css('marginRight'),0),
916             marginLeft = parseInt(label.css('marginLeft'),0);
917         if(marginRight=='auto') SC.Logger.log(marginRight+","+marginLeft+","+paddingRight+","+paddingLeft);
918         if(!paddingRight && isNaN(paddingRight)) paddingRight = 0;
919         if(!paddingLeft && isNaN(paddingLeft)) paddingLeft = 0;
920         if(!marginRight && isNaN(marginRight)) marginRight = 0;
921         if(!marginLeft && isNaN(marginLeft)) marginLeft = 0;
922 
923         this._labelMinWidthIE7 = w-(paddingRight + paddingLeft)-(marginRight + marginLeft);
924         label.css('minWidth', this._labelMinWidthIE7+'px');
925       }else{
926         this.invokeLater(this.didAppendToDocument, 1);
927       }
928     }
