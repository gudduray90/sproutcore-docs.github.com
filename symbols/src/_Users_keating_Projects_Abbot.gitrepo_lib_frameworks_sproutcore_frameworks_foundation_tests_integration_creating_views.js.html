  9 module("Create a pane with some custom views and adding it to the window", {
 10   setup: function(){
 11     htmlbody('<style>.sc-view { border: 1px red solid; z-index: -1; position: absolute; }</style>');
 12   },
 13   teardown: function(){
 14     clearHtmlbody();
 15   }
 16 });
 17 
 18 test("layer creation and management", function() {
 19   
 20   // create custom view w/ render method
 21   var MyView = SC.View.extend({
 22     cv: 0,
 23     
 24     displayProperties: 'cv'.w(),
 25     
 26     render: function(context, firstTime) {
 27       this.renderChildViews(context, YES); // always re-render...
 28       context.begin()
 29         .push('View %@'.fmt(this.get('cv')))
 30       .end();
 31     }
 32   });
 33   
 34   // create design
 35   var pane = SC.Pane.design()
 36     .layout({ bottom: 10, right: 20, top: 20, left: 20 })
 37     .childView(MyView.design({ cv: 1 })
 38       .classNames('view-1')
 39       .layout({ top: 50 })
 40       .childView(MyView.design({ cv: 2 })
 41         .classNames('view-2')
 42         .layout({ bottom: 10, right: 20, width: 100, height: 23 })));
 43     
 44   
 45   // instantiate pane...
 46   pane = pane.create();
 47   var cv1 = pane.childViews[0], cv2 = cv1.childViews[0];
 48   
 49   // pane should have nested child views.
 50   equals(cv1.cv, 1, 'has childView 1');
 51   equals(cv2.cv, 2, 'has childView 2');
 52   
 53   // it should not have a layer yet (nor should children)
 54   ok(!pane.get('layer'), 'pane has no layer');
 55   ok(!cv1.get('layer'), 'cv1 has no layer');
 56   ok(!cv2.get('layer'), 'cv2 has no layer');
 57   
 58   // if we mark cv2 as needed an update, it should not do anything, even at
 59   // end of the runloop.
 60   // -- temporarily replace updateLayer()
 61   var runCount = 0; 
 62   cv2.updateLayer = function() { runCount++; };
 63   SC.RunLoop.begin();
 64   cv2.set('displayNeedsUpdate', YES) ;
 65   SC.RunLoop.end();
 66   
 67   // updateLayer should NOT have run...
 68   equals(runCount, 0, 'cv2.updateLayer did not run');
 69   
 70   // restore original updateLayer
 71   delete cv2.updateLayer ;
 72   
 73   // all the views should have their isVisibleInWindow set to NO now
 74   ok(!pane.get('isVisibleInWindow'), 'pane.isVisibleInWindow should be NO');
 75   ok(!cv1.get('isVisibleInWindow'), 'cv1.isVisibleInWindow should be NO');
 76   ok(!cv2.get('isVisibleInWindow'), 'cv2.isVisibleInWindow should be NO');
 77   // -----
 78   // OK, let's add this baby to the window
 79   SC.RunLoop.begin();
 80   pane.append();
 81   SC.RunLoop.end();
 82   
 83   // all the views should have their isVisibleInWindow set to YES now
 84   ok(pane.get('isVisibleInWindow'), 'pane.isVisibleInWindow should be YES');
 85   ok(cv1.get('isVisibleInWindow'), 'cv1.isVisibleInWindow should be YES');
 86   ok(cv2.get('isVisibleInWindow'), 'cv2.isVisibleInWindow should be YES');
 87 
 88   // pane should have layer...cv's should not have had to find their layers
 89   // yet.
 90   ok(pane.get('layer'), 'pane has layer');
 91   ok(!cv1._view_pane, 'cv1 has not found its layer yet') ;
 92   ok(!cv2._view_pane, 'cv2 has not found its layer yet') ;
 93   
 94   // now let's make sure the pane layer has some of the things we expect...
 95   var layer = pane.get('layer');
 96   equals(layer.firstChild.id, cv1.get('layerId'), 'has cv1 elem');
 97   equals(layer.firstChild.firstChild.id, cv2.get('layerId'), 'has cv2 elem');
 98   layer = null;
 99   
100   // but if we get the layer for cv1 & cv2 they should find them...
101   ok(cv1.get('layer'), 'cv1 can find its layer');
102   ok(cv2.get('layer'), 'cv2 can find its layer');
103   
104   // also, the pane should now belong to the main document body.  it should 
105   // have a parent.
106   equals(pane.get('layer').parentNode, document.body, 'panes layer is in document body');
107   
108   // ok, now we're going to change the view number for cv2 and see if that
109   // updates the UI.
110   SC.RunLoop.begin();
111   cv2.set('cv', 'foo');
112   SC.RunLoop.end();
113   
114   // the innerHTML for cv2 should now contain 'View foo'
115   layer = cv2.get('layer');
116   ok(layer.innerHTML.toString().match(/View foo/), 'does not have new label!');
117   
118 });
