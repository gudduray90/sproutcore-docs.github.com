  8 /*
  9   Tests SplitView logic responsible for managing child views; specifically,
 10   these properties that are added to the child views:
 11 
 12   - nextView
 13   - previousView
 14   - viewIndex
 15 
 16   We don't have to worry about orientation; the code being tested creates
 17   child views, but does nothing related to laying them out.
 18 
 19   We _do_ test these properties with dividers as well.
 20 */
 21 var splitView;
 22 module("SplitView - Child Management", {
 23   setup: function() {
 24     SC.RunLoop.begin();
 25     splitView = SC.SplitView.create({
 26       childViews: [ 'left', 'middle', 'right' ],
 27 
 28       left:  SC.View.extend(SC.SplitChild, { name: 'left', canCollapse: YES, collapseAtSize: 50 }),
 29       middle: SC.View.extend(SC.SplitChild, { name: 'middle' }),
 30       right: SC.View.extend(SC.SplitChild, { name: 'right', canCollapse: YES, collapseAtSize: 50 }),
 31 
 32       layout: { left: 0, top: 0, width: 500, height: 500 },
 33 
 34       splitDividerView: null // in most of these tests, we are not checking the behavior of split dividers
 35     });
 36     SC.RunLoop.end();
 37   }
 38 });
 39 
 40 function validateChildren(view, expected) {
 41   equals(view.childViews.length, expected, "Should have " + expected + " child views");
 42 
 43   var prev = null;
 44   for (var i = 0; i < view.childViews.length; i++) {
 45     equals(view.childViews[i].viewIndex, i, "View index should be " + i);
 46     equals(view.childViews[i].previousView, prev, "Should have proper previous view");
 47 
 48     var next = null;
 49     if (view.childViews.length > i + 1) {
 50       next = view.childViews[i+1];
 51     }
 52 
 53     equals(view.childViews[i].nextView, next, "Should have proper next view");
 54     prev = view.childViews[i];
 55   }
 56 }
 57 
 58 test("Initial settings correct", function() {
 59   validateChildren(splitView, 3);
 60 });
 61 
 62 test("Adding a child view doesn't disturb things", function() {
 63   var add = SC.View.create(SC.SplitChild, { name: 'add' });
 64 
 65   SC.RunLoop.begin();
 66   splitView.appendChild(add);
 67   SC.RunLoop.end();
 68 
 69   validateChildren(splitView, 4);
 70 });
 71 
 72 test("Inserting a child view doesn't disturb things", function() {
 73   var add = SC.View.create(SC.SplitChild, { name: 'add' });
 74 
 75   SC.RunLoop.begin();
 76   splitView.insertBefore(add, splitView.childViews[0]);
 77   SC.RunLoop.end();
 78 
 79   validateChildren(splitView, 4);
 80 });
 81 
 82 test("Removing a child view doesn't disturb things", function() {
 83   SC.RunLoop.begin();
 84   splitView.removeChild(splitView.childViews[0]);
 85   SC.RunLoop.end();
 86 
 87   validateChildren(splitView, 2);
 88 });
 89 
 90 test("Works with dividers", function() {
 91   var add = SC.View.create(SC.SplitChild, { name: 'add' });
 92 
 93   splitView.splitDividerView = SC.SplitDividerView;
 94 
 95   SC.RunLoop.begin();
 96   splitView.appendChild(add);
 97   SC.RunLoop.end();
 98 
 99   validateChildren(splitView, 7);
100 });
101 
102 test("Destroying the split view with dividers doesn't break things", function() {
103   var add = SC.View.create(SC.SplitChild, { name: 'add' });
104 
105   splitView.splitDividerView = SC.SplitDividerView;
106 
107   SC.RunLoop.begin();
108   splitView.appendChild(add); // trigger _scsv_setupChildViews
109   SC.RunLoop.end();
110 
111   try {
112     SC.RunLoop.begin();
113     splitView.destroy();
114     SC.RunLoop.end();
115     ok(true, "No error was thrown");
116   } catch(e) {
117     ok(false, "An error was thrown - " + e);
118   }
119 });
