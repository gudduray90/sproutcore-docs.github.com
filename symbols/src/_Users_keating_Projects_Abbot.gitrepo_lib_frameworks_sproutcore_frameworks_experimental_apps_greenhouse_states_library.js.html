  8 /** @mixin

 14 */
 15 Greenhouse.mixin( /** @scope Greenhouse */{
 16   
 17   libraryStates: SC.State.design({
 18     initialSubstate: 'libraryClosed',
 19     
 20     libraryClosed: SC.State.design({
 21       parallelStatechart: 'library',
 22 
 23       // ..........................................................
 24       // Events
 25       //
 26       openLibrary: function(){
 27         this.gotoState('openLibraryPicker');
 28       },
 29 
 30       toggleDockedLibrary: function(){
 31         this.gotoState('dockedLibrary');
 32       }
 33     }),
 34 
 35     openLibraryPicker: SC.State.design({
 36 
 37       parallelStatechart: 'library',
 38 
 39       enterState: function(){
 40         var picker = Greenhouse.appPage.get('libraryPicker'),
 41             button = Greenhouse.appPage.getPath('mainView.toolBar.library'),
 42             pickerContentView = Greenhouse.appPage.get('libraryPickerContentView');
 43 
 44         pickerContentView.setIfChanged('nowShowing', 'Greenhouse.appPage.libraryContentView');
 45         picker.popup(button, SC.PICKER_POINTER);
 46         picker.becomeFirstResponder();
 47       },
 48 
 49       exitState: function(){
 50         var picker = Greenhouse.appPage.get('libraryPicker'),
 51             pickerContentView = Greenhouse.appPage.get('libraryPickerContentView');
 52         pickerContentView.setIfChanged('nowShowing', null);
 53         picker.remove();
 54       },
 55 
 56       cancel: function(){
 57         this.gotoState('libraryClosed');
 58       },
 59 
 60       floatLibrary: function(){
 61         this.gotoState('libraryPalette');
 62       },
 63 
 64       toggleDockedLibrary: function(){
 65         this.gotoState('dockedLibrary');
 66       }
 67     }),
 68 
 69     libraryPalette: SC.State.design({
 70       parallelStatechart: 'library',
 71 
 72       enterState: function(){
 73         var ap = Greenhouse.appPage;
 74         var picker = ap.get('libraryPicker'),
 75             pickerContentView = ap.get('libraryPickerContentView');
 76 
 77         pickerContentView.setIfChanged('nowShowing', 'Greenhouse.appPage.libraryContentView');
 78         picker.append();
 79         picker.set('isModal', NO);
 80         picker.set('isAnchored', NO);
 81         picker.$().toggleClass('sc-picker', NO);
 82         var content = ap.getPath('libraryContentView.content'),
 83             toolbar = ap.getPath('libraryContentView.toolbar');
 84 
 85         content.adjust('top', 49);    
 86         toolbar.set('isVisible', YES); 
 87       },
 88       exitState: function(){
 89         var ap = Greenhouse.appPage;
 90         var picker = ap.get('libraryPicker'),
 91             pickerContentView = ap.get('libraryPickerContentView');
 92 
 93         pickerContentView.setIfChanged('nowShowing', null);
 94         picker.set('isModal', YES);
 95         picker.set('isAnchored', YES);
 96         picker.remove();
 97 
 98         var content = ap.getPath('libraryContentView.content'),
 99             toolbar = ap.getPath('libraryContentView.toolbar');
100 
101         content.adjust('top', 49);    
102         toolbar.set('isVisible', NO);
103       },
104 
105       closeLibrary: function(){
106         this.gotoState('libraryClosed');
107       },
108 
109       toggleDockedLibrary: function(){
110         this.gotoState('dockedLibrary');
111       }
112     }),
113 
114     dockedLibrary: SC.State.design({
115 
116       parallelStatechart: 'library',
117 
118       enterState: function(){
119         var libDock = Greenhouse.appPage.get('libraryDockView');
120         libDock.setIfChanged('nowShowing', 'Greenhouse.appPage.libraryContentView');
121       },
122       exitState: function(){
123         var libDock = Greenhouse.appPage.get('libraryDockView');
124         libDock.setIfChanged('nowShowing', null);
125       },
126 
127       // ..........................................................
128       // Events
129       //
130       toggleDockedLibrary: function(){
131         var states = Greenhouse.get('currentStates') || [];
132         if (states.indexOf(Greenhouse.getState('dockedInspector'))) Greenhouse.sendEvent('undock');
133 
134         this.gotoState('libraryClosed');
135       }
136     })
137     
138     
139   })
140   
141   
142 });
