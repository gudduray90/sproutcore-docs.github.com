1 /**
  2   Prepares the Handlebars templating library for use inside SproutCore's view
  3   system.
  4 
  5   The SC.Handlebars object is the standard Handlebars library, extended to use
  6   SproutCore's get() method instead of direct property access, which allows
  7   computed properties to be used inside templates.
  8 
  9   To use SC.Handlebars, call SC.Handlebars.compile().  This will return a
 10   function that you can call multiple times, with a context object as the first
 11   parameter:
 12 
 13       var template = SC.Handlebars.compile("my {{cool}} template");
 14       var result = template({
 15         cool: "awesome"
 16       });
 17 
 18       console.log(result); // prints "my awesome template"
 19 
 20   Note that you won't usually need to use SC.Handlebars yourself. Instead, use
 21   SC.TemplateView, which takes care of integration into the view layer for you.
 22 */
 23 
 24 SC.Handlebars = {};
 25 
 26 SC.Handlebars.Compiler = function() {};
 27 SC.Handlebars.Compiler.prototype = SC.beget(Handlebars.Compiler.prototype);
 28 SC.Handlebars.Compiler.prototype.compiler = SC.Handlebars.Compiler;
 29 
 30 SC.Handlebars.JavaScriptCompiler = function() {};
 31 SC.Handlebars.JavaScriptCompiler.prototype = SC.beget(Handlebars.JavaScriptCompiler.prototype);
 32 SC.Handlebars.JavaScriptCompiler.prototype.compiler = SC.Handlebars.JavaScriptCompiler;
 33 
 34 SC.Handlebars.JavaScriptCompiler.prototype.nameLookup = function(parent, name, type) {
 35   if (type === 'context') {
 36     return "SC.get(" + parent + ", " + this.quotedString(name) + ");";
 37   } else {
 38     return Handlebars.JavaScriptCompiler.prototype.nameLookup.call(this, parent, name, type);
 39   }
 40 };
 41 
 42 SC.Handlebars.Compiler.prototype.mustache = function(mustache) {
 43   if (mustache.params.length || mustache.hash) {
 44     return Handlebars.Compiler.prototype.mustache.call(this, mustache);
 45   } else {
 46     var id = new Handlebars.AST.IdNode(['bind']);
 47     mustache = new Handlebars.AST.MustacheNode([id].concat([mustache.id]), mustache.hash, !mustache.escaped);
 48     return Handlebars.Compiler.prototype.mustache.call(this, mustache);
 49   }
 50 };
 51 
 52 SC.Handlebars.compile = function(string) {
 53   var ast = Handlebars.parse(string);
 54   var environment = new SC.Handlebars.Compiler().compile(ast, {data: true, stringParams: true});
 55   var ret = new SC.Handlebars.JavaScriptCompiler().compile(environment, {data: true, stringParams: true});
 56   ret.rawTemplate = string;
 57   return ret;
 58 };

 60 /**
 61   Registers a helper in Handlebars that will be called if no property with the
 62   given name can be found on the current context object, and no helper with
 63   that name is registered.
 64 
 65   This throws an exception with a more helpful error message so the user can
 66   track down where the problem is happening.
 67 */
 68 Handlebars.registerHelper('helperMissing', function(path, options) {
 69   var error;
 70 
 71   error = "%@ Handlebars error: Could not find property '%@' on object %@.";
 72   throw error.fmt(options.data.view, path, this);
 73 });
