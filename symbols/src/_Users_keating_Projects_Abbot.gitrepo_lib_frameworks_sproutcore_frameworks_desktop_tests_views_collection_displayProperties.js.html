  8 var view ;
  9 module("SC.CollectionView#displayProperties", {
 10   setup: function() {
 11     view = SC.CollectionView.create({
 12         isVisibleInWindow: YES
 13     }).createLayer();
 14   },
 15 
 16   teardown: function() {
 17     view.destroy();
 18   }
 19 });
 20 
 21 test("should gain active class if isActive", function() {
 22   SC.RunLoop.begin();
 23   view.set('isActive', YES);
 24   SC.RunLoop.end();
 25   ok(view.$().hasClass('active'), 'should have active class');
 26 
 27   SC.RunLoop.begin();
 28   view.set('isActive', NO);
 29   SC.RunLoop.end();
 30   ok(!view.$().hasClass('active'), 'should remove active class');
 31 });
