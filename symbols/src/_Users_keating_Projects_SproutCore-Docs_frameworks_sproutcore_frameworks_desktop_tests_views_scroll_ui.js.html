141    test('ScrollView should readjust scrollTop/scrollLeft if layer changes', function() {
142      var view = pane.view('basic2'), cv = view.get('contentView'), container = view.get('containerView') ;
143      view.scrollTo(10, 10);
144      SC.RunLoop.begin().end();
145      equals(container.get('layer').scrollLeft, 10, 'precond - scrollLeft is set to 10');
146      equals(container.get('layer').scrollTop, 10, 'precond- scrollTop is set to 10');
147      cv.replaceLayer();
148      SC.RunLoop.begin().end();
149      equals(container.get('layer').scrollLeft, 10, 'scrollLeft should be readjusted to 10');
150      equals(container.get('layer').scrollTop, 10, 'scrollTop should be readjust to 10');
151    });
