 13   Implements support for Ajax requests using XHR, XHR 2 and other protocols.
 14 
 15   SC.Request is much like an inverted version of the request/response objects
 16   you receive when implementing HTTP servers.

 18   To send a request, you just need to create your request object, configure
 19   your options, and call send() to initiate the request.

227   /**
228     Returns a copy of the current request. This will only copy certain
229     properties so if you want to add additional properties to the copy you
230     will need to override copy() in a subclass.
231 
232     @returns {SC.Request} new request
233   */
234   copy: function() {
235     var ret = {},
236         keys = this.COPY_KEYS,
237         loc = keys.length,
238         key, listeners, headers;
239 
240     while(--loc >= 0) {
241       key = keys[loc];
242       if (this.hasOwnProperty(key)) {
243         ret[key] = this.get(key);
244       }
245     }
246 
247     if (this.hasOwnProperty('listeners')) {
248       ret.listeners = SC.copy(this.get('listeners'));
249     }
250 
251     if (this.hasOwnProperty('_headers')) {
252       ret._headers = SC.copy(this._headers);
253     }
254 
255     ret.source = this.get('source') || this;
256 
257     return this.constructor.create(ret);
258   },

260   /**
261     To set headers on the request object. Pass either a single key/value
262     pair or a hash of key/value pairs. If you pass only a header name, this
263     will return the current value of the header.
264 
265     @param {String|Hash} key
266     @param {String} value
267     @returns {SC.Request|Object} receiver
268   */
269   header: function(key, value) {
270     var header, headers;
271 
272     if (SC.typeOf(key) === SC.T_STRING) {
273       headers = this._headers;
274       if (arguments.length === 1) {
275         return headers ? headers[key] : null;
276       } else {
277         this.propertyWillChange('headers');
278         if (!headers) { headers = this._headers = {}; }
279         headers[key] = value;
280         this.propertyDidChange('headers');
281         return this;
282       }
283 
284     // handle parsing hash of parameters
285     } else if (value === undefined) {
286       headers = key;
287       this.beginPropertyChanges();
288       for(header in headers) {
289         if (!headers.hasOwnProperty(header)) { continue; }
290         this.header(header, headers[header]);
291       }
292       this.endPropertyChanges();
293       return this;
294     }
295 
296     return this;
297   },

355   /**
356     Called just before a request is enqueued.  This will encode the body
357     into JSON if it is not already encoded, and set identifying headers
358   */
359   _prep: function() {
360     var hasContentType = !!this.header('Content-Type');
361 
362     if (this.get('attachIdentifyingHeaders')) {
363       this.header('X-Requested-With', 'XMLHttpRequest');
364       this.header('X-SproutCore-Version', SC.VERSION);
365     }
366 
367     // Set the Content-Type header only if not specified and the request
368     // includes a body.
369     if (!hasContentType && !!this.get('body')) {
370       if (this.get('isJSON')) {
371         this.header('Content-Type', 'application/json');
372       } else if (this.get('isXML')) {
373         this.header('Content-Type', 'text/xml');
374       }
375     }
376     return this;
377   },

379   /**
380     Will fire the actual request. If you have set the request to use JSON
381     mode then you can pass any object that can be converted to JSON as the
382     body. Otherwise you should pass a string body.
383 
384     @param {String|Object} [body]
385     @returns {SC.Response} New response object
386   */
387   send: function(body) {
388     // Sanity-check: Be sure a timeout value was not specified if the request
389     // is synchronous (because it wouldn't work).
390     var timeout = this.get('timeout');
391     if (timeout && !this.get('isAsynchronous')) {
392       throw "Timeout values cannot be used with synchronous requests";
393     } else if (timeout === 0) {
394       throw "The timeout value must either not be specified or must be greater than 0";
395     }
396 
397     if (body) { this.set('body', body); }
398     return SC.Request.manager.sendRequest(this.copy()._prep());
399   },

401   /**
402     Resends the current request. This is more efficient than calling send()
403     for requests that have already been used in a send. Otherwise acts just
404     like send(). Does not take a body argument.
405 
406     @returns {SC.Response} new response object
407   */
408   resend: function() {
409     var req = this.get('source') ? this : this.copy()._prep();
410     return SC.Request.manager.sendRequest(req);

413   /**
414     Configures a callback to execute as a request progresses or completes. You
415     must pass at least a target and action/method to this and optionally an
416     event name or status code.
417 
418     You may also pass additional arguments which will then be passed along to
419     your callback.
420 
421     ## Scoping With Status Codes
422 
423     If you pass a status code as the first argument to this method, the
424     accompanying notification callback will only be called if the response
425     status matches the status code. For example, if you pass 201 (or
426     SC.Request.CREATED), the accompanying method will only be called if the
427     response status from the server is also 201.
428 
429     You can also pass "generic" status codes such as 200, 300, or 400, which
430     will be invoked anytime the status code is in the same range and if a more
431     specific notifier was not registered first and returned YES.
432 
433     Finally, passing a status code of 0 or no status at all will cause your
434     method to be executed no matter what the resulting status is unless a
435     more specific notifier was registered first and returned YES.
436 
437     For example,
438 
439         var req = SC.Request.create({type: 'POST'});
440         req.notify(201, this, this.reqWasCreated);  // Handle a specific status code
441         req.notify(401, this, this.reqWasUnauthorized);  // Handle a specific status code
442         req.notify(400, this, this.reqDidRedirect);  // Handle any 4xx status
443         req.notify(this, function(response, arg1, arg2) {
444           // do something
445         }, arg1, arg2);  // Handle any status.  Also, passing additional arguments to the callback handler
446 
447     ## Notifying on Progress Events
448 
449     If you pass a progress event name your callback will be called each time
450     the event fires on the response.  For example, the XMLHttpRequest Level 2
451     specification defines several progress events: loadstart, progress, abort,
452     error, load, timeout and loadend.  Therefore, when using the default
453     SC.Request responseClass, SC.XHRResponse, you can be notified of each of
454     these events by simply providing the matching event name.
455 
456     Note that many older browsers do not support XMLHttpRequest Level 2.  See
457     http://caniuse.com/xhr2 for a list of supported browsers.
458 
459     For example,
460 
461       var req = SC.Request.create({type: 'GET'});
462       req.notify('progress', this, this.reqDidProgress); // Handle 'progress' events
463       req.notify('abort', this, this.reqDidAbort); // Handle 'abort' events
464       req.notify('upload.progress', this, this.reqUploadDidProgress); // Handle 'progress' events on the XMLHttpRequestUpload
465       req.send();
466 
467     ## Callback Format
468 
469     Your notification callback should expect to receive the Response object as
470     the first parameter for status code notifications and the Event object for
471     progress notifications; plus any additional parameters that you pass. If
472     your callback handles the notification and to prevent further handling, it
473     should return YES.
474 
475     @param [statusOrEvent] {Number|String} A Number status code or String Event name.
476     @param target {Object} The target object for the callback action.
477     @param action {String|Function} The method name or function to call on the target.
478     @returns {SC.Request} The SC.Request object.
479   */
480   notify: function(statusOrEvent, target, action) {
481     var args;
482 
483     // Normalize arguments
484     if (SC.typeOf(statusOrEvent) !== SC.T_NUMBER && SC.typeOf(statusOrEvent) !== SC.T_STRING) {
485       // Accept multiple additional arguments (Do so before shifting the arguments!)
486       args = SC.A(arguments).slice(2);
487 
488       // Shift the arguments
489       action = target;
490       target = statusOrEvent;
491       statusOrEvent = 0;
492     } else {
493       // Accept multiple additional arguments.
494       args = SC.A(arguments).slice(3);
495     }
496 
497     // Prepare listeners for this object and notification target.
498     var listeners = this.get('listeners');
499     if (!listeners) { this.set('listeners', listeners = {}); }
500     if(!listeners[statusOrEvent]) { listeners[statusOrEvent] = []; }
501 
502     // Add another listener for the given status code or event name.
503     listeners[statusOrEvent].push({target: target, action: action, args: args});
504 
505     return this;
506   }

574 /**
575   @class
576 
577   The request manager coordinates all of the active XHR requests. It will
578   only allow a certain number of requests to be active at a time; queuing
579   any others. This allows you more precise control over which requests load
580   in which order.
581 
582   @since SproutCore 1.0
583 */
584 SC.Request.manager = SC.Object.create(
585 /** @scope SC.Request.manager */{
586 
587   /**
588     Maximum number of concurrent requests allowed. 6 for all browsers.
589 
590     @type Number
591     @default 6
592   */
593   maxRequests: 6,
594 
595   /**
596     Current requests that are inflight.
597 
598     @type Array
599     @default []
600   */
601   inflight: [],
602 
603   /**
604     Requests that are pending and have not been started yet.
605 
606     @type Array
607     @default []
608   */
609   pending: [],
610 
611 
612   // ..........................................................
613   // METHODS
614   //
615 
616   /**
617     Invoked by the send() method on a request. This will create a new low-
618     level transport object and queue it if needed.
619 
620     @param {SC.Request} request the request to send
621     @returns {SC.Object} response object
622   */
623   sendRequest: function(request) {
624     if (!request) { return null; }
625 
626     // create low-level transport.  copy all critical data for request over
627     // so that if the request has been reconfigured the transport will still
628     // work.
629     var response = request.get('responseClass').create({ request: request });
630 
631     // add to pending queue
632     this.get('pending').pushObject(response);
633     this.fireRequestIfNeeded();
634 
635     return response;
636   },
637 
638   /**
639     Cancels a specific request. If the request is pending it will simply
640     be removed. Otherwise it will actually be cancelled.
641 
642     @param {Object} response a response object
643     @returns {Boolean} YES if cancelled
644   */
645   cancel: function(response) {
646     var pending = this.get('pending'),
647         inflight = this.get('inflight'),
648         idx;
649 
650     if (pending.indexOf(response) >= 0) {
651       this.propertyWillChange('pending');
652       pending.removeObject(response);
653       this.propertyDidChange('pending');
654       return YES;
655     } else if (inflight.indexOf(response) >= 0) {
656       response.cancel();
657 
658       inflight.removeObject(response);
659       this.fireRequestIfNeeded();
660       return YES;
661     }
662 
663     return NO;
664   },
665 
666   /**
667     Cancels all inflight and pending requests.
668 
669     @returns {Boolean} YES if any items were cancelled.
670   */
671   cancelAll: function() {
672     if (this.get('pending').length || this.get('inflight').length) {
673       this.set('pending', []);
674       this.get('inflight').forEach(function(r) { r.cancel(); });
675       this.set('inflight', []);
676       return YES;
677     }
678 
679     return NO;
680   },
681 
682   /**
683     Checks the inflight queue. If there is an open slot, this will move a
684     request from pending to inflight.
685 
686     @returns {Object} receiver
687   */
688   fireRequestIfNeeded: function() {
689     var pending = this.get('pending'),
690         inflight = this.get('inflight'),
691         max = this.get('maxRequests'),
692         next;
693 
694     if ((pending.length>0) && (inflight.length<max)) {
695       next = pending.shiftObject();
696       inflight.pushObject(next);
697       next.fire();
698     }
699   },
700 
701   /**
702     Called by a response/transport object when finishes running. Removes
703     the transport from the queue and kicks off the next one.
704   */
705   transportDidClose: function(response) {
706     this.get('inflight').removeObject(response);
707     this.fireRequestIfNeeded();
708   }
709 
710 });
