 24 var FRAME = { x: 10, y: 10, width: 30, height: 30 };
 25 
 26 var pane, view; // test globals
 27 
 28 module("isVisible", {
 29 
 30   setup: function() {
 31     pane = SC.MainPane.create();
 32     view = SC.View.create();
 33   },
 34 
 35   teardown: function() {
 36     view.destroy();
 37     pane.remove().destroy();
 38     pane = view = null;
 39   }
 40 
 41 });
 42 
 43 test("a new view should not be visible initially", function() {
 44   ok(view.get('isVisible'), "view.get('isVisible') === NO");
 45 });
 46 
 47 test("initializing with isVisible: false, should still add the proper class on append", function() {
 48   var newView = SC.View.create({
 49     isVisible: false
 50   });
 51 
 52   SC.RunLoop.begin();
 53   pane.append();
 54   pane.appendChild(newView);
 55   SC.RunLoop.end();
 56   ok(newView.$().hasClass('sc-hidden'), "newView.$().hasClass('sc-hidden') should be true");
 57 });
 58 
 59 test("adding a new view to a visible pane should make it visible", function() {
 60   ok(view.get('isVisible'), "view.get('isVisible') === YES");
 61   ok(pane.get('isVisible'), "pane.get('isVisible') === YES");
 62   SC.RunLoop.begin();
 63   pane.appendChild(view);
 64   pane.append();
 65   view.set('isVisible', NO);
 66   SC.RunLoop.end();
 67   ok(!view.get('isVisible'), "after pane.appendChild(view), view.get('isVisible') === YES");
 68   ok(view.$().hasClass('sc-hidden'), "after view.set('isVisible', NO), view.$().hasClass('sc-hidden') should be true");
 69 });
 70 
 71 test("a view with visibility can have a child view without visibility", function() {
 72   var pane = SC.Pane.create({
 73     childViews: ['visibleChild'],
 74 
 75     visibleChild: SC.View.design({
 76       childViews: ['noVisibilityChild'],
 77       noVisibilityChild: SC.CoreView
 78     })
 79   });
 80 
 81   var errored = false;
 82 
 83   try {
 84     pane.append();
 85     pane.remove().destroy();
 86   } catch(e) {
 87     errored = true;
 88   } finally {
 89     try {
 90       pane.remove().destroy();
 91     } catch(e2) {
 92       errored = true;
 93     }
 94   }
 95 
 96   ok(!errored, "Inserting a pane containing a child with visibility that itself has a child without visibility does not cause an error");
 97 });
