1 var object, methodSpy;
  2 module('CoreTest.spyOn', {
  3   setup: function() {
  4     object = {method: function() {}};
  5     methodSpy = CoreTest.spyOn(object, 'method');
  6   }
  7 });
  8 
  9 test('when the spied upon method was called spy#wasCalled is true', function() {
 10   object.method();
 11 
 12   ok(methodSpy.wasCalled, 'spy should know that it was called');
 13 });
 14 
 15 test('when the spied upon method was not called spy#wasCalled is false', function() {
 16   ok(!methodSpy.wasCalled, 'spy should know that it was not called');
 17 });
 18 
 19 test('when the spied upon method is called with a single argument the spy should know when it was called with the proper arguments', function() {
 20   object.method('something');
 21 
 22   ok(methodSpy.wasCalledWith('something'), 'spy should know it was called with the proper arguments');
 23 });
 24 
 25 test('when the spied upon method is called with multiple arguments the spy should know when it was called with the proper arguments', function() {
 26   object.method('something', 'else');
 27 
 28   ok(methodSpy.wasCalledWith('something', 'else'), 'spy should know it was called with the proper arguments');
 29 });
 30 
 31 test('when the spied upon method is called with a single argument the spy should know when it was called with the wrong arguments', function() {
 32   object.method('something');
 33 
 34   ok(!methodSpy.wasCalledWith('else'), 'spy should know it was called with the wrong arguments');
 35 });
 36 
 37 test('when the spied upon method is called with multiple arguments the spy should know when it was called with the wrong arguments', function() {
 38   object.method('something', 'else');
 39 
 40   ok(!methodSpy.wasCalledWith('else', 'something'), 'spy should know it was called with the wrong arguments');
 41 });
