 10 /*globals module test */
 11 
 12 var object, object1,object3; //global variables
 13 
 14 module("Checking the tuple for property path",{
 15 	
 16 	setup: function(){
 17 		 object = SC.Object.create({
 18 			name:'SproutCore',
 19 			value:'',						//no value defined for the property
 20 			objectA:SC.Object.create({
 21 					propertyVal:"chainedProperty"
 22 			})		
 23 		 });
 24    }
 25 
 26 });
 27    
 28 
 29 test("should check for the tuple property", function() {
 30      var object2 = [];
 31      object2 = SC.tupleForPropertyPath(object.name,'');
 32      equals(object2[0], window, "the window object");
 33      equals(object2[1],'SproutCore',"the property name");	
 34      object2 = SC.tupleForPropertyPath(object.objectA.propertyVal,'object');
 35 	 equals(object2[0],'object',"the root");
 36      equals(object2[1],'chainedProperty',"a chained property");
 37 });
 38 
 39 test("should check for the tuple property when path is undefined",function(){     //test case where no property defined
 40      var object2;
 41      object2 = SC.tupleForPropertyPath(object.value,'');
 42      equals(YES,object2 === null,'returns null for undefined path');	
 43 });
