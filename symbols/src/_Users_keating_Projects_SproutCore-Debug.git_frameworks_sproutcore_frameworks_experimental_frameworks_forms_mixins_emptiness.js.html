  7 /**
  8   @namespace 
  9   A view is empty if all of its children are empty. A view is automatically 
 10   counted as empty if it is not visible, and not empty if it is being edited.
 11 
 12   Any field that does not mix in CalculatesEmptiness will be considered empty.
 13 */
 14 SC.CalculatesEmptiness = {
 15   
 16   hasCalculatesEmptiness: YES,
 17   
 18   /**
 19   YES if the value of the field is empty. Defaults to yes so if you don't override this, it will only consider child fields in emptiness calculation (this is the desired behavior for forms).
 20   */
 21   isValueEmpty: YES,
 22   
 23   /**
 24     Defaults to YES so that a field with no children will act properly.
 25   */
 26   _scce_childrenAreEmpty: YES,
 27   
 28   /**
 29     If YES, all visible fields are considered non-empty when editing.
 30     @type Boolean
 31  32     @default YES
 33   */
 34   isEditingAffectsIsEmpty: YES,
 35 
 36 
 37   _scce_isEditingDidChange: function() {
 38     if(this.get('isEditingAffectsIsEmpty')) {
 39       this.notifyPropertyChange('isEmpty');
 40     }
 41   }.observes('isEditing'),
 42   
 43   /**
 44     YES if the field itself is empty. Even if the value is non-empty, the field can be empty due to isVisible.
 45   */
 46   isEmpty: function() {
 47     // When not visible, it is empty. Period.
 48     if (!this.get('isVisible')) {
 49       return YES;
 50     }
 51 
 52     // if it is editing and edit mode affects emptiness, it is NOT empty.
 53     if (this.get('isEditingAffectsIsEmpty') && this.get('isEditing')) {
 54       return NO;
 55     }
 56 
 57     // otherwise, it is empty if its value AND all children are empty.
 58     return this.get('isValueEmpty') && this.get('_scce_childrenAreEmpty');
 59   }.property('isValueEmpty', 'isVisible', '_scce_childrenAreEmpty', 'isEditingAffectsIsEmpty').cacheable(),
 60 
 61   /**
 62     When emptiness changes tell the parent to re-check its own emptiness.
 63   */
 64   _scce_isEmptyDidChange: function() {
 65     var parentView = this.get('parentView');
 66 
 67     if (parentView && parentView._scce_emptinessDidChangeFor) {
 68       parentView._scce_emptinessDidChangeFor(this);
 69     }
 70   }.observes('isEmpty'),
 71 
 72   initMixin: function() {
 73     this._scce_emptinessDidChangeFor();
 74   },
 75  76 
  /**
 77   Called by fields when their emptiness changes.
 78 
 79   Always triggers (at end of run loop) a relayout of fields.
 80   */
 81   _scce_emptinessDidChangeFor: function(child) {
 82     this.invokeOnce('_scce_recalculateChildrensEmptiness');
 83   },
 84 
 85   /**
 86   By default, a view will check all of its fields to determine if it is empty. It is only empty if all of its value fields are.
 87   */
 88   _scce_recalculateChildrensEmptiness: function()
 89   {
 90     // in short, we get the value fields, if we come across one that is not empty
 91     // we cannot be empty.
 92     var views = this.get('childViews');
 93 
 94     var empty = YES,
 95     len = views.length,
 96     field;
 97 
 98     for (var i = 0; i < len; i++)
 99     {
100       field = views[i];
101 
102       if (!field.get('isEmpty') && field.hasCalculatesEmptiness) {
103         empty = NO;
104         break;
105       }
106     }
107     
108     this.setIfChanged('_scce_childrenAreEmpty', empty);
109   }
110 };
