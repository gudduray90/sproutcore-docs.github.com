/** @class
 10   @extends SC.View
 11 */
 12 SC.MediaSlider = SC.SliderView.extend(
 13 /** @scope SC.MediaSlider.prototype */{
 14   
 15   mediaView: null,
 16 
 17   leftHandleInsetBinding: "*owner.leftHandleInset",
 18   
 19   rightHandleInsetBinding: "*owner.rightHandleInset",
 20   
 21   handleWidthBinding: "*owner.handleWidth",
 22                                                              
 23   //for some reason, when the observer was hooked to this bound property,
 24   //it wouldn't fire even when the original setter was in an SC run loop!
 25   //what gives and/or what was I doing wrong? The observer is now direct.
 26   // loadedTimeRangesBinding: "*mediaView.loadedTimeRanges", 
 27   
 28   loadedRangesSelector: "span.sc-loaded-ranges",
 29   
 30   _defaults: {
 31     //these are necessary because I couldn't get SC.Binding.notEmpty to
 32     //work without values defined in the parent, which is a situation
 33     //I'd rather leave optional.
 34     leftHandleInset:-1,
 35     rightHandleInset:-8,
 36     handleWidth:16
 37   },           
 38 
 39   render: function(context, firstTime) {
 40     //sc_super(); we are overriding the ENTIRE previous render method,
 41     //so instead of calling sc_super, we want to skip the function
 42     //immediately above this one and go two levels up.
 43     arguments.callee.base.base.apply(this,arguments);
 44     
 45     //some variables
 46     var min = this.get('minimum'),
 47         max = this.get('maximum'),
 48         value = this.get('value'),
 49         step = this.get('step'),
 50         width = this.get('frame').width,
 51         left, loc, minLoc, maxLoc, i,
 52         blankImage = SC.BLANK_IMAGE_URL;
 53                                 
 54     //process the defaults if we haven't.
 55     if(!this._defaultsEnsured) {
 56       this._defaultsEnsured = YES;
 57       for(i in this._defaults) {
 58         if(SC.none(this.get(i))) {
 59           this.set(i,this._defaults[i]);
 60         }
 61       }
 62     }
 63 
 64     // determine the constrained value.  Must fit within min & max
 65     value = Math.min(Math.max(value, min), max);
 66 
 67     // limit to step value
 68     if (!SC.none(step) && step !== 0) {
 69       value = Math.round(value / step) * step;
 70     }
 71     
 72     // get a pointer to the actual handle DOM element, and, if we have no
 73     // width information for it, try to use its offsetWidth.
 74     if(!firstTime) {
 75       var handle = this.handleElement ||
 76  77                   (this.handleElement=this.$(this.get('handleSelector'))[0]);
 78       if(!this.handleWidth){
 79         this.handleWidth = this.handleElement.offsetWidth || this.handleWidth;
 80       }
 81     }
 82     
 83     //derive the extremes
 84     maxLoc = width-this.handleWidth/2-this.rightHandleInset;
 85     minLoc = this.handleWidth/2+this.leftHandleInset;
 86     
 87     if(min==max) loc=minLoc;
 88     else loc = (value - min) / (max - min) * (maxLoc - minLoc) + minLoc;
 89     
 90     left = loc + this.leftHandleInset - this.handleWidth/2;
 91     
 92     //console.log("RENDERING: max is "+max+" and min is "+min+" and value is "+value+" out of "+max+", left is "+left+" and loc is "+loc+" and handleWidth is "+this.handleWidth);
 93     
 94     if(!firstTime) {
 95       //if all our elements are already created, just edit them instead of
 96       //blowing them away and starting over.
 97       handle.style.left = left + "px";
 98     } else {
 99       context.push('<span class="sc-inner">',
100                     '<span class="sc-leftcap"></span>',
101                     '<span class="sc-rightcap"></span>',  
102                     '<span class="sc-track"></span>',
103                     '<span class="sc-loaded-ranges"></span>',
104                     '<img src="', blankImage, 
105                     '" class="sc-handle" style="left: ', left, 'px" />',
106                    '</span>');
107     }
108     
109     //next, we take care of the loaded ranges of time that will darken
110     //the slider. This should happen whether it's the first render or not!
111     this.renderLoadedTimeRanges();
112   },
113 
114   renderLoadedTimeRanges: function() {
115       //first, get our quick variables we know we'll need...
116       var max = this.get('maximum'),
117           min = this.get('minimum'),
118           width = this.get('frame').width,
119           i;
120       //get our ranges, which will be an array of Numbers with twice as many
121       //elements as there are ranges. odd indices are starts, and evens are the
122       //ends that correspond to the start times immediately preceding them.
123       //only act if we have the ranges array. if it doesn't exist, don't bother
124       //to render the loaded ranges.
125       var mediaView = this.get('mediaView');
126       if(!mediaView) return;
127       var ranges = mediaView.get('loadedTimeRanges');
128       if(!ranges) return;
129       //make sure we have the pointer to our element which will house all the
130       //clones of the loaded-style background:
131       if(!this.loadedRangesElement){
132         this.loadedRangesElement = this.$(this.get('loadedRangesSelector'))[0];
133       }
134       //now, make sure we have a loaded-style slider background template, which
135       //will be cloned as many times as is necessary to render all loaded ranges:
136       if(!this.loadedBGTemplate) {
137         this.loadedBGTemplate = document.createElement('span');
138         this.loadedBGTemplate.className = "sc-loaded-range";
139         this.loadedBGTemplate.innerHTML = [
140           '<span class="sc-loaded-range-inner">',
141             '<span class="sc-leftcap loaded"></span>',
142             '<span class="sc-rightcap loaded"></span>',  
143             '<span class="sc-track loaded"></span>',
144           '</span>'
145         ].join('');
146       }
147       //now, make sure we have our cache of clones of the slider background
148       //template. we don't want to be spawning new HTML elements every time
149       //we render, so we will use this pool of them.
150       if(!this.loadedRangeElements) this.loadedRangeElements = [];
151       //now, for each range, obtain or create a clone of the background
152       //template, and style it such that the inner element lands
153       //exactly on top of the original slider background, but clipped
154       //by the outer element.
155       for(i=0; i<ranges.length/2; i++) {
156         var clone = this.loadedRangeElements[i] || 
157                    (this.loadedRangeElements[i] = this.loadedBGTemplate.cloneNode(true));
158         var leftClip  = Math.round(((ranges[ 2*i ]+min)/max) * width);
159         var rightClip = Math.round(((ranges[2*i+1]+min)/max) * width);
160         clone.style.left = leftClip+"px";
161         clone.style.width = rightClip-leftClip+"px";
162         clone.firstChild.style.left = 0-leftClip+"px";
163         clone.firstChild.style.width = width+"px";
164         if(!clone.parentNode) this.loadedRangesElement.appendChild(clone);
165       }
166       //now, make sure there aren't any EXTRA clones of the darkened background
167       //still hanging around. loop through the end of the loadedRangeElements
168       //list at indices greater than ranges.length/2, and remove the elements.
169       for(i=ranges.length/2; i<this.loadedRangeElements.length; i++) {
170         if(this.loadedRangeElements[i].parentNode) {
171           this.loadedRangeElements[i].parentNode.removeChild(this.loadedRangeElements[i]);
172         }
173       }
174       
175       //console.log('rendered loaded time ranges for '+ranges);
176   }.observes('*mediaView.loadedTimeRanges'),
177 
178   _triggerHandle: function(evt, firstEvent) {
179     var width = this.get('frame').width,
180         min = this.get('minimum'), max=this.get('maximum'),  
181         step = this.get('step'), v=this.get('value'), loc, maxLoc, minLoc,
182         handle = this.handleElement ||
183                 (this.handleElement=this.$(this.get('handleSelector'))[0]);
184     
185     //try to get a handleWidth from offsetWidth (if we don't already have a handleWidth);
186     //but make sure to favor zero over something like NaN we might get...
187     if(!this.handleWidth){
188       this.handleWidth = this.handleElement.offsetWidth || this.handleWidth;
189     }
190     //get the location of the cursor relative to the view
191     if(firstEvent){    
192       loc = this.convertFrameFromView({ x: evt.pageX }).x;
193       this._evtDiff = evt.pageX - loc;
194     }else{
195       loc = evt.pageX-this._evtDiff;
196     }
197 
198     //derive the extremes
199     maxLoc = width-this.handleWidth/2-this.rightHandleInset;
200     minLoc = this.handleWidth/2+this.leftHandleInset;
201 
202     //constrain loc to at least half a knob-width from either extreme.
203     loc = Math.max(Math.min(loc,maxLoc),minLoc);
204     
205     // convert to fraction of available width
206     loc = (loc-minLoc) / (maxLoc-minLoc);
207     
208     // convert to value using minimum/maximum then constrain to steps
209     loc = min + ((max-min)*loc);
210     if (step !== 0) loc = Math.round(loc / step) * step ;
211 
212     // if changes by more than a rounding amount, set v.
213     if (Math.abs(v-loc)>=0.01) this.set('value', loc); // adjust 
214     return YES ;
215   },
216 
217   mouseDown: function(evt) {
218     var media=this.get('mediaView');
219     if(media) media.startSeek();
220     return sc_super();
221   },
222 223 
  mouseUp: function(evt) {
224     var media=this.get('mediaView');
225     if(media) media.endSeek();
226     return sc_super();
227   }, 
228   
229   mouseWheel: function(){
230     var media, ret;
231     SC.RunLoop.begin();
232     media=this.get('mediaView');
233     if(media) media.startSeek();
234     ret = sc_super();
235     SC.RunLoop.end();
236     SC.RunLoop.begin();
237     if(media) media.endSeek();
238     SC.RunLoop.end();
239     return ret;
240   }  
241 });
