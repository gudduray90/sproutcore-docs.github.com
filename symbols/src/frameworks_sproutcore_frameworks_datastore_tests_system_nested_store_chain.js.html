 32 test("initial setup for chained store", function() {
 33   var parent = SC.Store.create();
 34   var store  = parent.chain();
 35   
 36   ok(store !== parent, 'chain should return new child store');
 37   
 38   equals(store.get('parentStore'), parent, 'should have parentStore');
 39   
 40   equals(SC.typeOf(store.dataHashes), SC.T_HASH, 'should have dataHashes');
 41   parent.dataHashes.foo = 'bar';
 42   equals(store.dataHashes.foo, 'bar', 'dataHashes should inherit from parent');
 43     
 44   equals(SC.typeOf(store.revisions), SC.T_HASH, 'should have revisions');
 45   parent.revisions.foo = 'bar';
 46   equals(store.revisions.foo, 'bar', 'revisions should inherit from parent');
 47 
 48   equals(SC.typeOf(store.statuses), SC.T_HASH, 'should have statuses');
 49   parent.statuses.foo = 'bar';
 50   equals(store.statuses.foo, 'bar', 'statuses should inherit from parent');
 51   
 52   ok(!store.locks, 'should not have locks');
 53   ok(!store.chainedChanges, 'should not have chainedChanges');
 54   ok(!store.editables, 'should not have editables');
 55 });

 57 test("allow for custom subclasses of SC.NestedStore", function() {
 58   var parent = SC.Store.create();
 59   
 60   // We should get an exception if we specify a "subclass" that's not a class
 61   var ex = null;
 62   try {
 63     var bogus = parent.chain({}, "I am not a class");
 64   }
 65   catch(e) {
 66     ex = e;
 67   }
 68   ok(ex  &&  ex.message  &&  ex.message.indexOf('not a valid class') !== -1, 'chain should report that our bogus "class" it is not a valid class');
 69   
 70   // We should get an exception if we specify a class that's not a subclass of
 71   // SC.NestedStore
 72   ex = null;
 73   try {
 74     var bogus = parent.chain({}, SC.Store);
 75   }
 76   catch(e) {
 77     ex = e;
 78   }
 79   ok(ex  &&  ex.message  &&  ex.message.indexOf('is not a type of SC.NestedStore') !== -1, 'chain should report that our class needs to be a subclass of SC.NestedStore');
 80   
 81   
 82   // Our specified (proper!) subclass should be respected.
 83   var MyNestedStoreSubclass = SC.NestedStore.extend();
 84   var nested = parent.chain({}, MyNestedStoreSubclass);
 85   ok(nested.kindOf(MyNestedStoreSubclass), 'our nested store should be the SC.NestedStore subclass we specified');
 86 });

 93 test("chained store changes should propagate reliably", function() {
 94   var parent = SC.Store.create(), rec, store, rec2;
 95 
 96   SC.run(function() {
 97     parent.loadRecords(Rec, [{ title: "foo", guid: 1 }]);
 98   });
 99   
100   rec = parent.find(Rec, 1);
101   ok(rec && rec.get('title')==='foo', 'precond - base store should have record');
102 
103   // run several times to make sure this works reliably when used several 
104   // times in the same app
105   
106   // trial 1
107   SC.RunLoop.begin();
108   store = parent.chain();
109   rec2  = store.find(Rec, 1);
110   ok(rec2 && rec2.get('title')==='foo', 'chain store should have record');
111   
112   rec.reset();
113   rec2.set('title', 'bar');
114   SC.RunLoop.end();
115   
116   equals(rec2.get('title'), 'bar', 'chained rec.title should changed');
117   equals(rec.get('title'), 'foo', 'original rec.title should NOT change');
118   equals(store.get('hasChanges'), YES, 'chained store.hasChanges');
119   equals(rec.fired, NO, 'original rec.title should not have notified');
120   
121   SC.RunLoop.begin();
122   rec.reset();
123   store.commitChanges();
124   store.destroy();
125   SC.RunLoop.end();
126 
127   equals(rec.get('title'), 'bar', 'original rec.title should change');
128   equals(rec.fired, YES, 'original rec.title should have notified');  
129 
130 
131   // trial 2
132   SC.RunLoop.begin();
133   store = parent.chain();
134   rec2  = store.find(Rec, 1);
135   ok(rec2 && rec2.get('title')==='bar', 'chain store should have record');
136   
137   rec.reset();
138   rec2.set('title', 'baz');
139   SC.RunLoop.end();
140   
141   equals(rec2.get('title'), 'baz', 'chained rec.title should changed');
142   equals(rec.get('title'), 'bar', 'original rec.title should NOT change');
143   equals(store.get('hasChanges'), YES, 'chained store.hasChanges');
144   equals(rec.fired, NO, 'original rec.title should not have notified');
145   
146   SC.RunLoop.begin();
147   rec.reset();
148   store.commitChanges();
149   store.destroy();
150   SC.RunLoop.end();
151 
152   equals(rec.get('title'), 'baz', 'original rec.title should change');
153   equals(rec.fired, YES, 'original rec.title should have notified');  
154   
155 
156   // trial 3
157   SC.RunLoop.begin();
158   store = parent.chain();
159   rec2  = store.find(Rec, 1);
160   ok(rec2 && rec2.get('title')==='baz', 'chain store should have record');
161   
162   rec.reset();
163   rec2.set('title', 'FOO2');
164   SC.RunLoop.end();
165   
166   equals(rec2.get('title'), 'FOO2', 'chained rec.title should changed');
167   equals(rec.get('title'), 'baz', 'original rec.title should NOT change');
168   equals(store.get('hasChanges'), YES, 'chained store.hasChanges');
169   equals(rec.fired, NO, 'original rec.title should not have notified');
170   
171   SC.RunLoop.begin();
172   rec.reset();
173   store.commitChanges();
174   store.destroy();
175   SC.RunLoop.end();
176 
177   equals(rec.get('title'), 'FOO2', 'original rec.title should change');
178   equals(rec.fired, YES, 'original rec.title should have notified');  
179 });

181 test("record retrievals triggered from a chained store and returned to the parent store should be reflected in the chained store", function() {
182   var parent = SC.Store.create().from(SC.DataSource.create({
183     retrieveRecords: function(store, storeKeys, ids) {
184       this.invokeLast(function() {
185         storeKeys.forEach(function(key, i) {
186           store.dataSourceDidComplete(key, { title: 'Stupendous! AB-solutely corking.' });
187         });
188       })
189       return YES;
190     }
191   }));
192 
193   var chained = parent.chain(),
194       chainedRec, storeKey;
195 
196   SC.run(function() {
197     chainedRec = chained.find(Rec, 1);
198     storeKey = chainedRec.get('storeKey');
199     equals(chained.peekStatus(storeKey), SC.Record.BUSY_LOADING, "While retrieving, the record's status should be BUSY_LOADING");
200   });
201 
202   // This immediate status update is sensitive to the data source's use of invokeNext to load the record.
203   equals(chained.peekStatus(storeKey), SC.Record.READY_CLEAN, "After retrieving, the record's status should be READY_CLEAN");
204 
205 
206 });
