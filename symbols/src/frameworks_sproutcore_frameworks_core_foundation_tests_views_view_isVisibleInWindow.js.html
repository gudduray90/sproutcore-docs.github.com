 24 var pane, view; // test globals
 25 
 26 module("isVisibleInWindow", {
 27 
 28   setup: function () {
 29     pane = SC.MainPane.create();
 30     pane.append();
 31     view = SC.View.create();
 32   },
 33 
 34   teardown: function () {
 35     view.destroy();
 36     pane.remove().destroy();
 37     pane = null;
 38   }
 39 
 40 });
 41 
 42 test("a new view should not be visible initially", function () {
 43   ok(!view.get('isVisibleInWindow'), "view.get('isVisibleInWindow') === NO");
 44 });
 45 
 46 test("adding a new view to a visible pane should make it visible", function () {
 47   ok(!view.get('isVisibleInWindow'), "view.get('isVisibleInWindow') === NO");
 48   ok(pane.get('isVisibleInWindow'), "pane.get('isVisibleInWindow') === YES");
 49 
 50   pane.appendChild(view);
 51   ok(view.get('isVisibleInWindow'), "after pane.appendChild(view), view.get('isVisibleInWindow') === YES");
 52 });
 53 
 54 test("removing a view from a visible pane should make it invisible again", function () {
 55   ok(!view.get('isVisibleInWindow'), "view.get('isVisibleInWindow') === NO");
 56   ok(pane.get('isVisibleInWindow'), "pane.get('isVisibleInWindow') === YES");
 57   pane.appendChild(view);
 58   ok(view.get('isVisibleInWindow'), "after pane.appendChild(view), view.get('isVisibleInWindow') === YES");
 59 
 60   view.removeFromParent();
 61   ok(!view.get('isVisibleInWindow'), "after view.removeFromParent(), view.get('isVisibleInWindow') === NO");
 62 });

 67 test("_executeDoUpdateContent should not be invoked even if layer becomes dirty until isVisibleInWindow changes, then it should invoke", function () {
 68 
 69   var callCount = 0;
 70   view._executeDoUpdateContent = function () {
 71     SC.View.prototype._executeDoUpdateContent.apply(this, arguments);
 72     callCount++;
 73   };
 74   ok(!view.get('isVisibleInWindow'), 'precond - view should not be visible to start');
 75 
 76   SC.run(function () {
 77     view.displayDidChange();
 78   });
 79   equals(callCount, 0, '_executeDoUpdateContent should not run b/c it\'s not visible');
 80 
 81   view.set('isVisible', false);
 82 
 83   SC.run(function () {
 84     pane.appendChild(view); // Attach the view.
 85     view.displayDidChange();
 86   });
 87   equals(callCount, 0, '_executeDoUpdateContent should not run b/c it\'s not visible');
 88 
 89   SC.run(function () {
 90     view.set('isVisible', true);
 91     ok(view.get('isVisibleInWindow'), 'view should now be visible in window');
 92   });
 93   equals(callCount, 1, '_executeDoUpdateContent should exec now b/c the view is visible');
 94 });
 95 
 96 test("_doUpdateLayoutStyle should not be invoked even if layer needs layout until isVisibleInWindow changes, then it should invoke", function () {
 97 
 98   var child = SC.View.create();
 99   view.appendChild(child);
100 
101   var callCount = 0;
102   child._doUpdateLayoutStyle = function () { callCount++; };
103   ok(!view.get('isVisibleInWindow'), 'precond - view should not be visible to start');
104 
105   SC.run(function () {
106     child.layoutDidChange();
107   });
108 
109   equals(callCount, 0, '_doUpdateLayoutStyle should not run b/c its not shown');
110 
111   view.set('isVisible', false);
112 
113   SC.run(function () {
114     pane.appendChild(view); // Attach the view.
115     child.layoutDidChange();
116   });
117   equals(callCount, 0, '_doUpdateLayoutStyle should not run b/c its not shown');
118 
119   SC.run(function () {
120     view.set('isVisible', true);
121     ok(view.get('isVisibleInWindow'), 'view should now be visible in window');
122   });
123   equals(callCount, 1, '_doUpdateLayoutStyle should exec now b/c the child was appended to a shown parent');
124 });
125 
126 test("setting isVisible to NO should trigger a layer update to hide the view", function () {
127 
128   SC.RunLoop.begin();
129   pane.appendChild(view);
130   SC.RunLoop.end();
131 
132   SC.RunLoop.begin();
133   view.set('isVisible', NO);
134   SC.RunLoop.end();
135 
136   ok(view.renderContext(view.get('layer')).classes().indexOf('sc-hidden') >= 0, "layer should have the 'sc-hidden' class");
137 });
