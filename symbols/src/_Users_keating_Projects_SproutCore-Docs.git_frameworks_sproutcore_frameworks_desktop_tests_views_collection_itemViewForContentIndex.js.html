 10 module("SC.CollectionView.itemViewForContentIndex", {
 11   setup: function() {
 12     content = "a b c".w().map(function(x) { 
 13       return SC.Object.create({ title: x });
 14     });
 15     
 16     del = {
 17       fixture: {
 18         isEnabled: YES,
 19         isSelected: YES,
 20         outlineLevel: 3,
 21         disclosureState: SC.LEAF_NODE
 22       },
 23       
 24       contentIndexIsEnabled: function() { 
 25         return this.fixture.isEnabled; 
 26       },
 27       
 28       contentIndexIsSelected: function() { 
 29         return this.fixture.isSelected; 
 30       },
 31 
 32       contentIndexOutlineLevel: function() { 
 33         return this.fixture.outlineLevel; 
 34       },
 35       
 36       contentIndexDisclosureState: function() {
 37         return this.fixture.disclosureState ;
 38       }
 39     };
 40 
 41     // NOTE: delegate methods above are added here.
 42     view = SC.CollectionView.create(del, {
 43       content: content,
 44       
 45       layoutForContentIndex: function(contentIndex) {
 46         return this.fixtureLayout ;
 47       },
 48       
 49       fixtureLayout: { left: 0, right: 0, top:0, bottom: 0 },
 50       
 51       groupExampleView: SC.View.extend(), // custom for testing
 52       
 53       exampleView: SC.View.extend(), // custom for testing
 54       
 55       testAsGroup: NO,
 56       
 57       contentIndexIsGroup: function() { return this.testAsGroup; },
 58       
 59       contentGroupIndexes: function() {
 60         if (this.testAsGroup) {
 61           return SC.IndexSet.create(0, this.get('length'));
 62         } else return null ;
 63       },
 64       
 65       fixtureNowShowing: SC.IndexSet.create(0,3),
 66       computeNowShowing: function() {
 67         return this.fixtureNowShowing;
 68       }
 69       
 70     });
 71     
 72     // add in delegate mixin
 73     del = SC.mixin({}, SC.CollectionContent, del);
 74     
 75   }
 76 });

 78 function shouldMatchFixture(itemView, fixture) {
 79   var key;
 80   for(key in fixture) {
 81     if (!fixture.hasOwnProperty(key)) continue;
 82     equals(itemView.get(key), fixture[key], 'itemView.%@ should match delegate value'.fmt(key));
 83   }
 84 }

 86 test("creating basic item view", function() {
 87   var itemView = view.itemViewForContentIndex(1);
 88   
 89   // should use exampleView
 90   ok(itemView, 'should return itemView');
 91   ok(itemView.kindOf(view.exampleView), 'itemView %@ should be kindOf %@'.fmt(itemView, view.exampleView));
 92   
 93   // set added properties
 94   equals(itemView.get('content'), content.objectAt(1), 'itemView.content should be set to content item');
 95   equals(itemView.get('contentIndex'), 1, 'itemView.contentIndex should be set');
 96   equals(itemView.get('owner'), view, 'itemView.owner should be collection view');
 97   equals(itemView.get('displayDelegate'), view, 'itemView.displayDelegate should be collection view');
 98   equals(itemView.get('parentView'), view, 'itemView.parentView should be collection view');
 99   
100   // test data from delegate
101   shouldMatchFixture(itemView, view.fixture);
102 });

104 test("returning item from cache", function() {
105   
106   var itemView1 = view.itemViewForContentIndex(1);
107   ok(itemView1, 'precond - first call returns an item view');
108   
109   var itemView2 = view.itemViewForContentIndex(1);
110   equals(itemView2, itemView1, 'retrieving multiple times should same instance');
111 
112   // Test internal case
113   var itemView3 = view.itemViewForContentIndex(1, YES);
114   ok(itemView1 !== itemView3, 'itemViewForContentIndex(1, YES) should return new item even if it is already cached actual :%@'.fmt(itemView3));
115 
116   var itemView4 = view.itemViewForContentIndex(1, NO);
117   equals(itemView4, itemView3, 'itemViewForContentIndex(1) [no reload] should return newly cached item after recache');
118   
119 });

263 test("after making an item visible then invisible again", function() {
264 
265   view.isVisibleInWindow = YES ;
266   
267   // STEP 1- setup with some nowShowing
268   SC.run(function() {
269     view.set('fixtureNowShowing', SC.IndexSet.create(1));
270     view.notifyPropertyChange('nowShowing');
271   });
272   equals(view.get('childViews').length, 1, 'precond - should have a child view');
273 
274   var itemView = view.itemViewForContentIndex(1);
275   equals(itemView.get('parentView'), view, 'itemView has parent view');
276   
277   // STEP 2- setup with NONE visible
278   SC.run(function() {
279     view.set('fixtureNowShowing', SC.IndexSet.create());
280     view.notifyPropertyChange('nowShowing');
281   });
282   equals(view.get('childViews').length, 0, 'precond - should have no childview');
283 
284   itemView = view.itemViewForContentIndex(1);
285   equals(itemView.get('parentView'), view, 'itemView has parent view');
286 
287 
288   // STEP 3- go back to nowShowing
289   SC.run(function() {
290     view.set('fixtureNowShowing', SC.IndexSet.create(1));
291     view.notifyPropertyChange('nowShowing');
292   });
293   equals(view.get('childViews').length, 1, 'precond - should have a child view');
294 
295   itemView = view.itemViewForContentIndex(1);
296   equals(itemView.get('parentView'), view, 'itemView has parent view');
297   
298 });
