  8 sc_require('mixins/split_child');
  9 sc_require('mixins/split_thumb');
 10 
 11 /**
 12   @class
 13 
 14   A SplitDividerView sits between any two other views in a SplitView.
 15   The SplitDivider mixes in SC.SplitThumb to allow the user to drag
 16   it around. The dragging process will cause SC.SplitView to adjust
 17   other children as needed.
 18 
 19   @extends SC.View
 20   @author Alex Iskander
 21 */
 22 SC.SplitDividerView = SC.View.extend(SC.SplitChild, SC.SplitThumb,
 23 {
 24   /** @scope SC.SplitDividerView.prototype */
 25   classNames: ['sc-split-divider-view'],
 26   
 27   // set to prevent SC.SplitView from automatically creating dividers
 28   // to sit between this divider and another view.
 29   isSplitDivider: YES,
 30 
 31   // NOTE: 'sc-fixed-size' is only hard-coded because SC.SplitView requires
 32   // this file, and SC.FIXED_SIZE is defined inside SC.SplitView.
 33   autoResizeStyle: 'sc-fixed-size',
 34   
 35   movesSibling: SC.MOVES_CHILD,
 36   
 37   size: SC.propertyFromRenderDelegate('dividerSize', 10),
 38 
 39   renderDelegateName: 'splitDividerRenderDelegate'
 40  41 });
