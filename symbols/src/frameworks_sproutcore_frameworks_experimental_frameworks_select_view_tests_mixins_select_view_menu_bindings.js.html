  3 function setValues(obj, props, toValue) {
  4   for (var idx = 0; idx < props.length; idx++) {
  5     var prop = props[idx];
  6     if (SC.typeOf(prop) === SC.T_HASH) prop = prop.from;
  7     obj.set(prop, toValue);
  8   }
  9 }
 10 
 11 function validateValues(obj, props, value) {
 12   for (var idx = 0; idx < props.length; idx++) {
 13     equals(obj.get(props[idx]), value, props[idx]);
 14   }
 15 }
